70      @BeforeEach
71      void before() {
72          when(database.getName()).thenReturn("schema");

74          when(database.getRuleMetaData()).thenReturn(new ShardingSphereRuleMetaData(Collections.emptyList()));

77      @Test
78      void assertUpdate() {
79          AlterShardingTableRuleStatement statement = new AlterShardingTableRuleStatement(Arrays.asList(createCompleteAutoTableRule("t_order_item"), createCompleteTableRule("t_order")));
80          updater.checkSQLStatement(database, statement, currentRuleConfig);
81          ShardingRuleConfiguration toBeAlteredRuleConfig = updater.buildToBeAlteredRuleConfiguration(statement);
82          updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeAlteredRuleConfig);
83          assertThat(currentRuleConfig.getTables().size(), is(1));
84          ShardingTableRuleConfiguration tableRule = currentRuleConfig.getTables().iterator().next();
85          assertThat(tableRule.getLogicTable(), is("t_order"));
86          assertThat(tableRule.getActualDataNodes(), is("ds_${0..1}.t_order${0..1}"));
87          assertThat(tableRule.getTableShardingStrategy(), instanceOf(StandardShardingStrategyConfiguration.class));
88          assertThat(((StandardShardingStrategyConfiguration) tableRule.getTableShardingStrategy()).getShardingColumn(), is("product_id"));
89          assertThat(tableRule.getTableShardingStrategy().getShardingAlgorithmName(), is("t_order_table_core.standard.fixture"));
90          assertThat(tableRule.getDatabaseShardingStrategy(), instanceOf(StandardShardingStrategyConfiguration.class));
91          assertThat(tableRule.getDatabaseShardingStrategy().getShardingAlgorithmName(), is("t_order_database_inline"));
92          assertThat(currentRuleConfig.getTables().size(), is(1));
93          ShardingAutoTableRuleConfiguration autoTableRule = currentRuleConfig.getAutoTables().iterator().next();
94          assertThat(autoTableRule.getLogicTable(), is("t_order_item"));
95          assertThat(autoTableRule.getActualDataSources(), is("ds_0,ds_1"));
96          assertThat(autoTableRule.getShardingStrategy().getShardingAlgorithmName(), is("t_order_item_foo.distsql.fixture"));
97          assertThat(((StandardShardingStrategyConfiguration) autoTableRule.getShardingStrategy()).getShardingColumn(), is("order_id"));
98          assertThat(autoTableRule.getKeyGenerateStrategy().getColumn(), is("product_id"));
99          assertThat(autoTableRule.getKeyGenerateStrategy().getKeyGeneratorName(), is("t_order_item_distsql.fixture"));
100     }

152     private AutoTableRuleSegment createCompleteAutoTableRule(final String logicTableName) {
153         AutoTableRuleSegment result = new AutoTableRuleSegment(logicTableName, Arrays.asList("ds_0", "ds_1"));
154         result.setKeyGenerateStrategySegment(new KeyGenerateStrategySegment("product_id", new AlgorithmSegment("DISTSQL.FIXTURE", new Properties())));
155         result.setShardingColumn("order_id");
156         result.setShardingAlgorithmSegment(new AlgorithmSegment("FOO.DISTSQL.FIXTURE", PropertiesBuilder.build(new Property("", ""))));
157         return result;
158     }
159     
160     private TableRuleSegment createCompleteTableRule(final String logicTableName) {
161         KeyGenerateStrategySegment keyGenerator = new KeyGenerateStrategySegment("product_id", new AlgorithmSegment("DISTSQL.FIXTURE", new Properties()));
162         TableRuleSegment result = new TableRuleSegment(logicTableName, Collections.singletonList("ds_${0..1}.t_order${0..1}"), keyGenerator, null);
163         result.setTableStrategySegment(new ShardingStrategySegment("standard", "product_id", new AlgorithmSegment("CORE.STANDARD.FIXTURE", new Properties())));
164         AlgorithmSegment databaseAlgorithmSegment = new AlgorithmSegment("inline", PropertiesBuilder.build(new Property("algorithm-expression", "ds_${user_id % 2}")));
165         result.setDatabaseStrategySegment(new ShardingStrategySegment("standard", "product_id", databaseAlgorithmSegment));
166         return result;
167     }
168     
169     private ShardingRuleConfiguration createCurrentShardingRuleConfiguration() {
170         ShardingRuleConfiguration result = new ShardingRuleConfiguration();
171         result.getTables().add(createTableRuleConfiguration());
172         result.getAutoTables().add(createAutoTableRuleConfiguration());
173         result.getShardingAlgorithms().put("t_order_algorithm", new AlgorithmConfiguration("hash_mod", PropertiesBuilder.build(new Property("sharding-count", "4"))));
174         result.getKeyGenerators().put("t_order_item_snowflake", new AlgorithmConfiguration("snowflake", new Properties()));
175         return result;
176     }
177     
178     private ShardingTableRuleConfiguration createTableRuleConfiguration() {
179         ShardingTableRuleConfiguration result = new ShardingTableRuleConfiguration("t_order", "ds_${0..1}.t_order${0..1}");
180         result.setTableShardingStrategy(new StandardShardingStrategyConfiguration("order_id", "t_order_algorithm"));
181         return result;
182     }
183     
184     private ShardingAutoTableRuleConfiguration createAutoTableRuleConfiguration() {
185         ShardingAutoTableRuleConfiguration result = new ShardingAutoTableRuleConfiguration("t_order_item", "ds_0");
186         result.setShardingStrategy(new StandardShardingStrategyConfiguration("order_id", "t_order_MOD_TEST"));
187         result.setKeyGenerateStrategy(new KeyGenerateStrategyConfiguration("product_id", "product_id_snowflake_test"));
188         return result;
189     }

195         return result;
