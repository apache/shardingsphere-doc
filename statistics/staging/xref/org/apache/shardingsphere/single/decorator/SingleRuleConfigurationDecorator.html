45  /**
46   * Single rule configuration decorator.
47   */
48  public final class SingleRuleConfigurationDecorator implements RuleConfigurationDecorator<SingleRuleConfiguration> {
49      
50      @Override
51      public SingleRuleConfiguration decorate(final String databaseName, final Map<String, DataSource> dataSources,
52                                              final Collection<ShardingSphereRule> builtRules, final SingleRuleConfiguration ruleConfig) {
53          return new SingleRuleConfiguration(decorateTables(databaseName, dataSources, new LinkedList<>(builtRules), ruleConfig.getTables()), ruleConfig.getDefaultDataSource().orElse(null));
54      }
55      
56      private Collection<String> decorateTables(final String databaseName, final Map<String, DataSource> dataSources, final Collection<ShardingSphereRule> builtRules, final Collection<String> tables) {
57          builtRules.removeIf(SingleRule.class::isInstance);
58          if (tables.isEmpty() && builtRules.isEmpty()) {
59              return Collections.emptyList();
60          }
61          Collection<String> splitTables = SingleTableLoadUtils.splitTableLines(tables);
62          if (!isExpandRequired(splitTables)) {
63              return splitTables;
64          }
65          Map<String, DataSource> aggregatedDataSources = SingleTableLoadUtils.getAggregatedDataSourceMap(dataSources, builtRules);
66          DatabaseType databaseType = dataSources.isEmpty() ? DatabaseTypeEngine.getDefaultStorageType() : DatabaseTypeEngine.getStorageType(dataSources.values().iterator().next());
67          Collection<String> excludedTables = SingleTableLoadUtils.getExcludedTables(builtRules);
68          Map<String, Collection<DataNode>> actualDataNodes = SingleTableDataNodeLoader.load(databaseName, aggregatedDataSources, excludedTables);
69          boolean isSchemaSupportedDatabaseType = new DatabaseTypeRegistry(databaseType).getDialectDatabaseMetaData().getDefaultSchema().isPresent();
70          if (splitTables.contains(SingleTableConstants.ALL_TABLES) || splitTables.contains(SingleTableConstants.ALL_SCHEMA_TABLES)) {
71              return loadAllTables(isSchemaSupportedDatabaseType, actualDataNodes);
72          }
73          Collection<DataNode> configuredDataNodes = SingleTableLoadUtils.convertToDataNodes(databaseName, databaseType, splitTables);
74          return loadSpecifiedTables(isSchemaSupportedDatabaseType, actualDataNodes, builtRules, configuredDataNodes);
75      }
76      
77      private boolean isExpandRequired(final Collection<String> splitTables) {
78          return splitTables.stream().anyMatch(each -> each.contains(SingleTableConstants.ASTERISK));
79      }
80      
81      private Collection<String> loadAllTables(final boolean isSchemaSupportedDatabaseType, final Map<String, Collection<DataNode>> actualDataNodes) {
82          return actualDataNodes.values().stream().map(each -> getTableNodeString(isSchemaSupportedDatabaseType, each.iterator().next())).collect(Collectors.toList());
83      }
84      
85      private String getTableNodeString(final boolean isSchemaSupportedDatabaseType, final DataNode dataNode) {
86          return isSchemaSupportedDatabaseType
87                  ? formatTableName(dataNode.getDataSourceName(), dataNode.getSchemaName(), dataNode.getTableName())
88                  : formatTableName(dataNode.getDataSourceName(), dataNode.getTableName());
89      }
90      
91      private String formatTableName(final String dataSourceName, final String schemaName, final String tableName) {
92          return String.format("%s.%s.%s", dataSourceName, schemaName, tableName);
93      }
94      
95      private String formatTableName(final String dataSourceName, final String tableName) {
96          return String.format("%s.%s", dataSourceName, tableName);
97      }
98      
99      private Collection<String> loadSpecifiedTables(final boolean isSchemaSupportedDatabaseType, final Map<String, Collection<DataNode>> actualDataNodes,
100                                                    final Collection<ShardingSphereRule> builtRules, final Collection<DataNode> configuredDataNodes) {
101         Collection<String> expandRequiredDataSources = new LinkedHashSet<>(configuredDataNodes.size(), 1F);
102         Map<String, DataNode> expectedDataNodes = new LinkedHashMap<>(configuredDataNodes.size(), 1F);
103         for (DataNode each : configuredDataNodes) {
104             if (SingleTableConstants.ASTERISK.equals(each.getTableName())) {
105                 expandRequiredDataSources.add(each.getDataSourceName());
106             } else {
107                 expectedDataNodes.put(each.getTableName(), each);
108             }
109         }
110         return expandRequiredDataSources.isEmpty()
111                 ? loadSpecifiedTablesWithoutExpand(isSchemaSupportedDatabaseType, actualDataNodes, configuredDataNodes)
112                 : loadSpecifiedTablesWithExpand(
113                         isSchemaSupportedDatabaseType, actualDataNodes, SingleTableLoadUtils.getFeatureRequiredSingleTables(builtRules), expandRequiredDataSources, expectedDataNodes);
114     }
115     
116     private Collection<String> loadSpecifiedTablesWithExpand(final boolean isSchemaSupportedDatabaseType, final Map<String, Collection<DataNode>> actualDataNodes,
117                                                              final Collection<String> featureRequiredSingleTables, final Collection<String> expandRequiredDataSources,
118                                                              final Map<String, DataNode> expectedDataNodes) {
119         Collection<String> result = new LinkedHashSet<>(actualDataNodes.size(), 1F);
120         for (Entry<String, Collection<DataNode>> entry : actualDataNodes.entrySet()) {
121             if (featureRequiredSingleTables.contains(entry.getKey())) {
122                 continue;
123             }
124             DataNode physicalDataNode = entry.getValue().iterator().next();
125             if (expandRequiredDataSources.contains(physicalDataNode.getDataSourceName())) {
126                 result.add(getTableNodeString(isSchemaSupportedDatabaseType, physicalDataNode));
127                 continue;
128             }
129             if (expectedDataNodes.containsKey(entry.getKey())) {
130                 DataNode dataNode = expectedDataNodes.get(entry.getKey());
131                 String tableNodeStr = getTableNodeString(isSchemaSupportedDatabaseType, physicalDataNode);
132                 ShardingSpherePreconditions.checkState(physicalDataNode.equals(dataNode),
133                         () -> new InvalidSingleRuleConfigurationException(String.format("Single table `%s` is found that does not match %s", tableNodeStr,
134                                 getTableNodeString(isSchemaSupportedDatabaseType, dataNode))));
135                 result.add(tableNodeStr);
136             }
137         }
138         return result;
139     }
140     
141     private Collection<String> loadSpecifiedTablesWithoutExpand(final boolean isSchemaSupportedDatabaseType,
142                                                                 final Map<String, Collection<DataNode>> actualDataNodes, final Collection<DataNode> configuredDataNodes) {
143         Collection<String> result = new LinkedHashSet<>(configuredDataNodes.size(), 1F);
144         for (DataNode each : configuredDataNodes) {
145             ShardingSpherePreconditions.checkContainsKey(actualDataNodes, each.getTableName(), () -> new SingleTableNotFoundException(getTableNodeString(isSchemaSupportedDatabaseType, each)));
146             DataNode actualDataNode = actualDataNodes.get(each.getTableName()).iterator().next();
147             String tableNodeStr = getTableNodeString(isSchemaSupportedDatabaseType, actualDataNode);
148             ShardingSpherePreconditions.checkState(actualDataNode.equals(each), () -> new InvalidSingleRuleConfigurationException(
149                     String.format("Single table '%s' is found that does not match %s", tableNodeStr, getTableNodeString(isSchemaSupportedDatabaseType, each))));
150             result.add(tableNodeStr);
151         }
152         return result;
153     }
154     
155     @Override
156     public Class<SingleRuleConfiguration> getType() {
157         return SingleRuleConfiguration.class;
158     }
159 }
