18  package org.apache.shardingsphere.data.pipeline.core.check.consistency.algorithm;
19  
20  import lombok.Getter;
21  import lombok.RequiredArgsConstructor;
22  import lombok.extern.slf4j.Slf4j;
23  import org.apache.shardingsphere.data.pipeline.api.check.consistency.DataConsistencyCalculateParameter;
24  import org.apache.shardingsphere.data.pipeline.api.check.consistency.DataConsistencyCalculatedResult;
25  
26  import java.util.Iterator;
27  import java.util.Optional;
28  
29  /**
30   * Streaming data consistency calculate algorithm.
31   */
32  @Getter
33  @Slf4j
34  public abstract class AbstractStreamingDataConsistencyCalculateAlgorithm extends AbstractDataConsistencyCalculateAlgorithm {
35      
36      @Override
37      public final Iterable<DataConsistencyCalculatedResult> calculate(final DataConsistencyCalculateParameter param) {
38          return new ResultIterable(param);
39      }
40      
41      /**
42       * Calculate chunked records at one time.
43       *
44       * @param param data consistency calculate parameter
45       * @return optional calculated result, empty means there's no more result
46       */
47      protected abstract Optional<DataConsistencyCalculatedResult> calculateChunk(DataConsistencyCalculateParameter param);
48      
49      /**
50       * It's not thread-safe, it should be executed in only one thread at the same time.
51       */
52      @RequiredArgsConstructor
53      private final class ResultIterable implements Iterable<DataConsistencyCalculatedResult> {
54          
55          private final DataConsistencyCalculateParameter param;
56          
57          @Override
58          public Iterator<DataConsistencyCalculatedResult> iterator() {
59              return new ResultIterator(param);
60          }
61      }
62      
63      @RequiredArgsConstructor
64      private final class ResultIterator implements Iterator<DataConsistencyCalculatedResult> {
65          
66          private final DataConsistencyCalculateParameter param;
67          
68          private volatile Optional<DataConsistencyCalculatedResult> nextResult;
69          
70          @Override
71          public boolean hasNext() {
72              calculateIfNecessary();
73              return nextResult.isPresent();
74          }
75          
76          @Override
77          public DataConsistencyCalculatedResult next() {
78              calculateIfNecessary();
79              Optional<DataConsistencyCalculatedResult> nextResult = this.nextResult;
80              this.nextResult = null;
81              return nextResult.orElse(null);
82          }
83          
84          private void calculateIfNecessary() {
85              if (null != nextResult) {
86                  return;
87              }
88              nextResult = calculateChunk(param);
89          }
90      }
91  }
