54  /**
55   * Expression converter.
56   */
57  public final class ExpressionConverter implements SQLSegmentConverter<ExpressionSegment, SqlNode> {
58      
59      @Override
60      public Optional<SqlNode> convert(final ExpressionSegment segment) {
61          if (null == segment) {
62              return Optional.empty();
63          }
64          if (segment instanceof LiteralExpressionSegment) {
65              return new LiteralExpressionConverter().convert((LiteralExpressionSegment) segment);
66          }
67          if (segment instanceof CommonExpressionSegment) {
68              // TODO
69              throw new UnsupportedSQLOperationException("unsupported CommonExpressionSegment");
70          }
71          if (segment instanceof ListExpression) {
72              return new ListExpressionConverter().convert((ListExpression) segment);
73          }
74          if (segment instanceof BinaryOperationExpression) {
75              return new BinaryOperationExpressionConverter().convert((BinaryOperationExpression) segment);
76          }
77          if (segment instanceof ColumnSegment) {
78              return new ColumnConverter().convert((ColumnSegment) segment);
79          }
80          if (segment instanceof ExistsSubqueryExpression) {
81              return new ExistsSubqueryExpressionConverter().convert((ExistsSubqueryExpression) segment);
82          }
83          if (segment instanceof SubqueryExpressionSegment) {
84              return new SubqueryExpressionConverter().convert((SubqueryExpressionSegment) segment);
85          }
86          if (segment instanceof InExpression) {
87              return new InExpressionConverter().convert((InExpression) segment);
88          }
89          if (segment instanceof BetweenExpression) {
90              return new BetweenExpressionConverter().convert((BetweenExpression) segment);
91          }
92          if (segment instanceof ParameterMarkerExpressionSegment) {
93              return new ParameterMarkerExpressionConverter().convert((ParameterMarkerExpressionSegment) segment);
94          }
95          if (segment instanceof FunctionSegment) {
96              return new FunctionConverter().convert((FunctionSegment) segment);
97          }
98          if (segment instanceof AggregationProjectionSegment) {
99              return new AggregationProjectionConverter().convert((AggregationProjectionSegment) segment);
100         }
101         if (segment instanceof DataTypeSegment) {
102             return new DataTypeConverter().convert((DataTypeSegment) segment);
103         }
104         if (segment instanceof CaseWhenExpression) {
105             return new CaseWhenExpressionConverter().convert((CaseWhenExpression) segment);
106         }
107         throw new UnsupportedSQLOperationException("unsupported TableSegment type: " + segment.getClass());
108     }
109 }
