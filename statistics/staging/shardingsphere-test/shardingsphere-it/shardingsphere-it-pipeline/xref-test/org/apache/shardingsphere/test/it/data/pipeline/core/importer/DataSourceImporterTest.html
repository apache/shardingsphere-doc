61  @ExtendWith(MockitoExtension.class)
62  class DataSourceImporterTest {
63      
64      private static final String TABLE_NAME = "test_table";
65      
66      @Mock
67      private PipelineDataSourceManager dataSourceManager;
68      
69      private final PipelineDataSourceConfiguration dataSourceConfig = new StandardPipelineDataSourceConfiguration(
70              "jdbc:h2:mem:test_db;DB_CLOSE_DELAY=-1;DATABASE_TO_UPPER=false;MODE=MySQL;USER=root;PASSWORD=root", "root", "root");
71      
72      @Mock
73      private PipelineChannel channel;
74      
75      @Mock
76      private PipelineDataSourceWrapper dataSource;
77      
78      @Mock
79      private Connection connection;
80      
81      @Mock
82      private PreparedStatement preparedStatement;
83      
84      private DataSourceImporter jdbcImporter;
85      
86      @BeforeEach
87      void setUp() throws SQLException {
88          ImporterConnector importerConnector = new DataSourceImporterConnector(dataSourceManager);
89          jdbcImporter = new DataSourceImporter(mockImporterConfiguration(), importerConnector, channel, new FixtureInventoryIncrementalJobItemContext());
90          when(dataSourceManager.getDataSource(dataSourceConfig)).thenReturn(dataSource);
91          when(dataSource.getConnection()).thenReturn(connection);
92      }
93      
94      @Test
95      void assertWriteInsertDataRecord() throws SQLException {
96          DataRecord insertRecord = getDataRecord("INSERT");
97          when(connection.prepareStatement(any())).thenReturn(preparedStatement);
98          when(channel.fetchRecords(anyInt(), anyInt())).thenReturn(mockRecords(insertRecord));
99          jdbcImporter.run();
100         verify(preparedStatement).setObject(1, 1);
101         verify(preparedStatement).setObject(2, 10);
102         verify(preparedStatement).setObject(3, "INSERT");
103         verify(preparedStatement).addBatch();
104     }
105     
106     @Test
107     void assertDeleteDataRecord() throws SQLException {
108         DataRecord deleteRecord = getDataRecord("DELETE");
109         when(connection.prepareStatement(any())).thenReturn(preparedStatement);
110         when(channel.fetchRecords(anyInt(), anyInt())).thenReturn(mockRecords(deleteRecord));
111         jdbcImporter.run();
112         verify(preparedStatement).setObject(1, 1);
113         verify(preparedStatement).setObject(2, 10);
114         verify(preparedStatement).addBatch();
115     }
116     
117     @Test
118     void assertUpdateDataRecord() throws SQLException {
119         DataRecord updateRecord = getDataRecord("UPDATE");
120         when(connection.prepareStatement(any())).thenReturn(preparedStatement);
121         when(channel.fetchRecords(anyInt(), anyInt())).thenReturn(mockRecords(updateRecord));
122         jdbcImporter.run();
123         verify(preparedStatement).setObject(1, 10);
124         verify(preparedStatement).setObject(2, "UPDATE");
125         verify(preparedStatement).setObject(3, 1);
126         verify(preparedStatement).setObject(4, 10);
127         verify(preparedStatement).executeUpdate();
128     }
129     
130     @Test
131     void assertUpdatePrimaryKeyDataRecord() throws SQLException {
132         DataRecord updateRecord = getUpdatePrimaryKeyDataRecord();
133         when(connection.prepareStatement(any())).thenReturn(preparedStatement);
134         when(channel.fetchRecords(anyInt(), anyInt())).thenReturn(mockRecords(updateRecord));
135         jdbcImporter.run();
136         InOrder inOrder = inOrder(preparedStatement);
137         inOrder.verify(preparedStatement).setObject(1, 2);
138         inOrder.verify(preparedStatement).setObject(2, 10);
139         inOrder.verify(preparedStatement).setObject(3, "UPDATE");
140         inOrder.verify(preparedStatement).setObject(4, 1);
141         inOrder.verify(preparedStatement).setObject(5, 10);
142         inOrder.verify(preparedStatement).executeUpdate();
143     }
144     
145     private DataRecord getUpdatePrimaryKeyDataRecord() {
146         DataRecord result = new DataRecord(new PlaceholderPosition(), 3);
147         result.setTableName(TABLE_NAME);
148         result.setType("UPDATE");
149         result.addColumn(new Column("id", 1, 2, true, true));
150         result.addColumn(new Column("user", 10, true, false));
151         result.addColumn(new Column("status", "UPDATE", true, false));
152         return result;
153     }
154     
155     private Collection<Column> mockConditionColumns(final DataRecord dataRecord) {
156         return RecordUtils.extractConditionColumns(dataRecord, Collections.singleton("user"));
157     }
158     
159     private List<Record> mockRecords(final DataRecord dataRecord) {
160         List<Record> result = new LinkedList<>();
161         result.add(dataRecord);
162         result.add(new FinishedRecord(new PlaceholderPosition()));
163         return result;
164     }
165     
166     private DataRecord getDataRecord(final String recordType) {
167         DataRecord result = new DataRecord(new PlaceholderPosition(), 3);
168         result.setTableName(TABLE_NAME);
169         result.setType(recordType);
170         result.addColumn(new Column("id", 1, false, true));
171         result.addColumn(new Column("user", 10, true, false));
172         result.addColumn(new Column("status", recordType, true, false));
173         return result;
174     }
175     
176     private ImporterConfiguration mockImporterConfiguration() {
177         Map<LogicTableName, Set<String>> shardingColumnsMap = Collections.singletonMap(new LogicTableName("test_table"), Collections.singleton("user"));
178         return new ImporterConfiguration(dataSourceConfig, shardingColumnsMap, new TableNameSchemaNameMapping(Collections.emptyMap()), 1000, null, 3, 3);
179     }
