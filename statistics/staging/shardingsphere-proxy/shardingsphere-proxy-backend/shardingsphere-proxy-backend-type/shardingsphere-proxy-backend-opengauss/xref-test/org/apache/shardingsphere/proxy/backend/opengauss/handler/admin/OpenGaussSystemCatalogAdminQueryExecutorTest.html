42  @ExtendWith(AutoMockExtension.class)
43  @StaticMockSettings(ProxyContext.class)
44  class OpenGaussSystemCatalogAdminQueryExecutorTest {
45      
46      @Test
47      void assertExecuteSelectFromPgDatabase() throws SQLException {
48          when(ProxyContext.getInstance()).thenReturn(mock(ProxyContext.class, RETURNS_DEEP_STUBS));
49          when(ProxyContext.getInstance().getAllDatabaseNames()).thenReturn(Arrays.asList("foo", "bar", "sharding_db", "other_db"));
50          OpenGaussSystemCatalogAdminQueryExecutor executor = new OpenGaussSystemCatalogAdminQueryExecutor("select datname, datcompatibility from pg_database where datname = 'sharding_db'");
51          ConnectionSession connectionSession = mock(ConnectionSession.class);
52          when(connectionSession.getProtocolType()).thenReturn(new OpenGaussDatabaseType());
53          executor.execute(connectionSession);
54          QueryResultMetaData actualMetaData = executor.getQueryResultMetaData();
55          assertThat(actualMetaData.getColumnCount(), is(2));
56          assertThat(actualMetaData.getColumnName(1), is("datname"));
57          assertThat(actualMetaData.getColumnName(2), is("datcompatibility"));
58          MergedResult actualResult = executor.getMergedResult();
59          assertTrue(actualResult.next());
60          assertThat(actualResult.getValue(1, String.class), is("sharding_db"));
61          assertThat(actualResult.getValue(2, String.class), is("PG"));
62      }
63      
64      @Test
65      void assertExecuteSelectVersion() throws SQLException {
66          when(ProxyContext.getInstance()).thenReturn(mock(ProxyContext.class, RETURNS_DEEP_STUBS));
67          OpenGaussSystemCatalogAdminQueryExecutor executor = new OpenGaussSystemCatalogAdminQueryExecutor("select VERSION()");
68          ConnectionSession connectionSession = mock(ConnectionSession.class);
69          when(connectionSession.getProtocolType()).thenReturn(new OpenGaussDatabaseType());
70          executor.execute(connectionSession);
71          QueryResultMetaData actualMetaData = executor.getQueryResultMetaData();
72          assertThat(actualMetaData.getColumnCount(), is(1));
73          assertThat(actualMetaData.getColumnType(1), is(Types.VARCHAR));
74          MergedResult actualResult = executor.getMergedResult();
75          assertTrue(actualResult.next());
76          assertThat((String) actualResult.getValue(1, String.class), containsString("ShardingSphere-Proxy"));
77      }
78      
79      @Test
80      void assertExecuteSelectGsPasswordDeadlineAndIntervalToNum() throws SQLException {
81          when(ProxyContext.getInstance()).thenReturn(mock(ProxyContext.class, RETURNS_DEEP_STUBS));
82          OpenGaussSystemCatalogAdminQueryExecutor executor = new OpenGaussSystemCatalogAdminQueryExecutor("select intervaltonum(gs_password_deadline())");
83          ConnectionSession connectionSession = mock(ConnectionSession.class);
84          when(connectionSession.getProtocolType()).thenReturn(new OpenGaussDatabaseType());
85          executor.execute(connectionSession);
86          QueryResultMetaData actualMetaData = executor.getQueryResultMetaData();
87          assertThat(actualMetaData.getColumnCount(), is(1));
88          assertThat(actualMetaData.getColumnType(1), is(Types.INTEGER));
89          MergedResult actualResult = executor.getMergedResult();
90          assertTrue(actualResult.next());
91          assertThat(actualResult.getValue(1, Integer.class), is(90));
92      }
93

95      void assertExecuteSelectGsPasswordNotifyTime() throws SQLException {
96          when(ProxyContext.getInstance()).thenReturn(mock(ProxyContext.class, RETURNS_DEEP_STUBS));
97          OpenGaussSystemCatalogAdminQueryExecutor executor = new OpenGaussSystemCatalogAdminQueryExecutor("select gs_password_notifytime()");

99          when(connectionSession.getProtocolType()).thenReturn(new OpenGaussDatabaseType());
100         executor.execute(connectionSession);
