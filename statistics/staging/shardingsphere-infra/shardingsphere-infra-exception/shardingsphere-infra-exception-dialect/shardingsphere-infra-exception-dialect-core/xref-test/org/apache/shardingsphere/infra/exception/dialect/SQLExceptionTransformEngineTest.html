18  package org.apache.shardingsphere.infra.exception.dialect;
19  
20  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
21  import org.apache.shardingsphere.infra.exception.core.external.server.ShardingSphereServerException;
22  import org.apache.shardingsphere.infra.exception.core.external.sql.ShardingSphereSQLException;
23  import org.apache.shardingsphere.infra.exception.dialect.exception.SQLDialectException;
24  import org.apache.shardingsphere.infra.exception.dialect.exception.protocol.DatabaseProtocolException;
25  import org.apache.shardingsphere.infra.spi.type.typed.TypedSPILoader;
26  import org.junit.jupiter.api.Test;
27  
28  import java.sql.SQLException;
29  
30  import static org.hamcrest.CoreMatchers.is;
31  import static org.hamcrest.MatcherAssert.assertThat;
32  import static org.mockito.Mockito.mock;
33  import static org.mockito.Mockito.when;
34  
35  class SQLExceptionTransformEngineTest {
36      
37      private final DatabaseType databaseType = TypedSPILoader.getService(DatabaseType.class, "FIXTURE");
38      
39      @Test
40      void assertToSQLExceptionWithSQLException() {
41          SQLException cause = new SQLException("");
42          assertThat(SQLExceptionTransformEngine.toSQLException(cause, databaseType), is(cause));
43      }
44      
45      @Test
46      void assertToSQLExceptionWithShardingSphereSQLException() {
47          ShardingSphereSQLException cause = mock(ShardingSphereSQLException.class);
48          SQLException expected = new SQLException("");
49          when(cause.toSQLException()).thenReturn(expected);
50          assertThat(SQLExceptionTransformEngine.toSQLException(cause, databaseType), is(expected));
51      }
52      
53      @Test
54      void assertToSQLExceptionWithDatabaseProtocolException() {
55          DatabaseProtocolException cause = mock(DatabaseProtocolException.class);
56          when(cause.getMessage()).thenReturn("No reason");
57          SQLException actual = SQLExceptionTransformEngine.toSQLException(cause, databaseType);
58          assertThat(actual.getSQLState(), is("HY000"));
59          assertThat(actual.getErrorCode(), is(30002));
60          assertThat(actual.getMessage(), is("Database protocol exception: No reason"));
61      }
62      
63      @Test
64      void assertToSQLExceptionWithSQLDialectException() {
65          assertThat(SQLExceptionTransformEngine.toSQLException(mock(SQLDialectException.class), databaseType).getMessage(), is("Dialect exception"));
66      }
67      
68      @Test
69      void assertToSQLExceptionWithShardingSphereServerException() {
70          ShardingSphereServerException cause = mock(ShardingSphereServerException.class);
71          when(cause.getMessage()).thenReturn("No reason");
72          SQLException actual = SQLExceptionTransformEngine.toSQLException(cause, databaseType);
73          assertThat(actual.getSQLState(), is("HY000"));
74          assertThat(actual.getErrorCode(), is(30004));
75          assertThat(actual.getMessage(), is("Server exception." + System.lineSeparator() + "More details: " + cause));
76      }
77      
78      @Test
79      void assertToSQLExceptionWithOtherException() {
80          SQLException actual = SQLExceptionTransformEngine.toSQLException(new Exception("No reason"), databaseType);
81          assertThat(actual.getSQLState(), is("HY000"));
82          assertThat(actual.getErrorCode(), is(30000));
83          assertThat(actual.getMessage(), is("Unknown exception." + System.lineSeparator() + "More details: java.lang.Exception: No reason"));
84      }
85  }
