42  /**
43   * Merged result for show create table.
44   */
45  public final class ShowCreateTableMergedResult extends LogicTablesMergedResult {
46      
47      public ShowCreateTableMergedResult(final ShardingRule shardingRule,
48                                         final SQLStatementContext sqlStatementContext, final ShardingSphereSchema schema, final List<QueryResult> queryResults) throws SQLException {
49          super(shardingRule, sqlStatementContext, schema, queryResults);
50      }
51      
52      @Override
53      protected void setCellValue(final MemoryQueryResultRow memoryResultSetRow, final String logicTableName, final String actualTableName,
54                                  final ShardingSphereTable table, final ShardingRule shardingRule) {
55          memoryResultSetRow.setCell(2, memoryResultSetRow.getCell(2).toString().replaceFirst(actualTableName, logicTableName));
56          setViewCellValue(memoryResultSetRow, logicTableName, actualTableName, shardingRule);
57          for (ShardingSphereIndex each : table.getIndexValues()) {
58              String actualIndexName = IndexMetaDataUtils.getActualIndexName(each.getName(), actualTableName);
59              memoryResultSetRow.setCell(2, memoryResultSetRow.getCell(2).toString().replace(actualIndexName, each.getName()));
60          }
61          for (ShardingSphereConstraint each : table.getConstraintValues()) {
62              String actualIndexName = IndexMetaDataUtils.getActualIndexName(each.getName(), actualTableName);
63              memoryResultSetRow.setCell(2, memoryResultSetRow.getCell(2).toString().replace(actualIndexName, each.getName()));
64              Optional<ShardingTable> shardingTable = shardingRule.findShardingTable(each.getReferencedTableName());
65              if (!shardingTable.isPresent()) {
66                  continue;
67              }
68              for (DataNode dataNode : shardingTable.get().getActualDataNodes()) {
69                  memoryResultSetRow.setCell(2, memoryResultSetRow.getCell(2).toString().replace(dataNode.getTableName(), each.getReferencedTableName()));
70              }
71          }
72      }
73      
74      private void setViewCellValue(final MemoryQueryResultRow memoryResultSetRow, final String logicTableName, final String actualTableName, final ShardingRule shardingRule) {
75          Optional<ShardingTable> shardingTable = shardingRule.findShardingTable(logicTableName);
76          Optional<BindingTableRule> bindingTableRule = shardingRule.findBindingTableRule(logicTableName);
77          if (shardingTable.isPresent() && bindingTableRule.isPresent()) {
78              Collection<DataNode> actualDataNodes = shardingTable.get().getActualDataNodes().stream().filter(each -> each.getTableName().equalsIgnoreCase(actualTableName)).collect(Collectors.toList());
79              Map<String, String> logicAndActualTablesFromBindingTables = new CaseInsensitiveMap<>();
80              for (DataNode each : actualDataNodes) {
81                  logicAndActualTablesFromBindingTables
82                          .putAll(shardingRule.getLogicAndActualTablesFromBindingTable(each.getDataSourceName(), logicTableName, actualTableName, bindingTableRule.get().getAllLogicTables()));
83              }
84              for (Entry<String, String> entry : logicAndActualTablesFromBindingTables.entrySet()) {
85                  memoryResultSetRow.setCell(2, memoryResultSetRow.getCell(2).toString().replaceFirst(entry.getValue(), entry.getKey()));
86              }
87          }
88      }
89  }
