28  import java.util.HashMap;
29  import java.util.Map;
30  import java.util.Optional;
31  import java.util.Set;
32  import java.util.HashSet;
33  
34  /**
35   * Database meta data of Oracle.
36   */
37  public final class OracleDatabaseMetaData implements DialectDatabaseMetaData {
38      
39      private static final Set<String> RESERVED_KEYWORDS = new HashSet<>(Arrays.asList("ACCESS", "ADD", "ALL", "ALTER", "AND", "ANY", "ARRAYLEN", "AS", "ASC", "AUDIT", "BETWEEN", "BY", "CHAR", "CHECK",
40              "CLUSTER", "COLUMN", "COMMENT", "COMPRESS", "CONNECT", "CREATE", "CURRENT", "DATE", "DECIMAL", "DEFAULT", "DELETE", "DESC", "DISTINCT", "DROP", "ELSE", "EXCLUSIVE", "EXISTS", "FILE",
41              "FLOAT", "FOR", "FROM", "GRANT", "GROUP", "HAVING", "IDENTIFIED", "IMMEDIATE", "IN", "INCREMENT", "INDEX", "INITIAL", "INSERT", "INTEGER", "INTERSECT", "INTO", "IS", "LEVEL", "LIKE",
42              "LOCK", "LONG", "MAXEXTENTS", "MINUS", "MODE", "MODIFY", "NOAUDIT", "NOCOMPRESS", "NOT", "NOTFOUND", "NOWAIT", "NULL", "NUMBER", "OF", "OFFLINE", "ON", "ONLINE", "OPTION", "OR", "ORDER",
43              "PCTFREE", "PRIOR", "PRIVILEGES", "PUBLIC", "RAW", "RENAME", "RESOURCE", "REVOKE", "ROW", "ROWID", "ROWLABEL", "ROWNUM", "ROWS", "START", "SELECT", "SESSION", "SET", "SHARE", "SIZE",
44              "SMALLINT", "SQLBUF", "SUCCESSFUL", "SYNONYM", "SYSDATE", "TABLE", "THEN", "TO", "TRIGGER", "UID", "UNION", "UNIQUE", "UPDATE", "USER", "VALIDATE", "VALUES", "VARCHAR", "VARCHAR2",
45              "VIEW", "WHENEVER", "WHERE", "WITH"));
46      
47      @Override
48      public QuoteCharacter getQuoteCharacter() {
49          return QuoteCharacter.QUOTE;
50      }
51      
52      @Override
53      public NullsOrderType getDefaultNullsOrderType() {
54          return NullsOrderType.LAST;
55      }
56      
57      @Override
58      public boolean isReservedWord(final String identifier) {
59          return RESERVED_KEYWORDS.contains(identifier.toUpperCase());
60      }
61      
62      @Override
63      public boolean isSchemaAvailable() {
64          return true;
65      }
66      
67      @Override
68      public String getSchema(final Connection connection) {
69          try {
70              return Optional.ofNullable(connection.getMetaData().getUserName()).map(String::toUpperCase).orElse(null);
71          } catch (final SQLException ignored) {
72              return null;
73          }
74      }
75      
76      @Override
77      public String formatTableNamePattern(final String tableNamePattern) {
78          return tableNamePattern.toUpperCase();
79      }
80      
81      @Override
82      public boolean isInstanceConnectionAvailable() {
83          return true;
84      }
85      
86      @Override
87      public String getDatabaseType() {
88          return "Oracle";
89      }
90      
91      @Override
92      public Map<String, Integer> getExtraDataTypes() {
93          Map<String, Integer> result = new HashMap<>(8);
94          result.put("TEXT", Types.LONGVARCHAR);
95          result.put("CHARACTER", Types.CHAR);
96          result.put("VARCHAR2", Types.VARCHAR);
97          result.put("DATETIME", Types.TIMESTAMP);
98          result.put("ROWID", Types.ROWID);
99          result.put("BINARY_DOUBLE", Types.DOUBLE);
100         result.put("BINARY_FLOAT", Types.FLOAT);
101         result.put("NUMBER", Types.NUMERIC);
102         return result;
103     }
104 }
