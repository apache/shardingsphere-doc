65  /**
66   * Sharding statement validator factory.
67   */
68  @NoArgsConstructor(access = AccessLevel.PRIVATE)
69  public final class ShardingStatementValidatorFactory {
70      
71      /**
72       * New instance of sharding statement validator.
73       * 
74       * @param sqlStatement SQL statement
75       * @param shardingConditions sharding conditions
76       * @return created instance
77       */
78      public static Optional<ShardingStatementValidator> newInstance(final SQLStatement sqlStatement, final ShardingConditions shardingConditions) {
79          if (sqlStatement instanceof DDLStatement) {
80              return getDDLStatementValidator(sqlStatement);
81          }
82          if (sqlStatement instanceof DMLStatement) {
83              return getDMLStatementValidator(sqlStatement, shardingConditions);
84          }
85          return Optional.empty();
86      }
87      
88      private static Optional<ShardingStatementValidator> getDDLStatementValidator(final SQLStatement sqlStatement) {
89          if (sqlStatement instanceof CreateTableStatement) {
90              return Optional.of(new ShardingCreateTableStatementValidator());
91          }
92          if (sqlStatement instanceof CreateFunctionStatement) {
93              return Optional.of(new ShardingCreateFunctionStatementValidator());
94          }
95          if (sqlStatement instanceof CreateProcedureStatement) {
96              return Optional.of(new ShardingCreateProcedureStatementValidator());
97          }
98          if (sqlStatement instanceof CreateViewStatement) {
99              return Optional.of(new ShardingCreateViewStatementValidator());
100         }
101         if (sqlStatement instanceof CreateIndexStatement) {
102             return Optional.of(new ShardingCreateIndexStatementValidator());
103         }
104         if (sqlStatement instanceof AlterTableStatement) {
105             return Optional.of(new ShardingAlterTableStatementValidator());
106         }
107         if (sqlStatement instanceof RenameTableStatement) {
108             return Optional.of(new ShardingRenameTableStatementValidator());
109         }
110         if (sqlStatement instanceof AlterViewStatement) {
111             return Optional.of(new ShardingAlterViewStatementValidator());
112         }
113         if (sqlStatement instanceof AlterIndexStatement) {
114             return Optional.of(new ShardingAlterIndexStatementValidator());
115         }
116         if (sqlStatement instanceof DropTableStatement) {
117             return Optional.of(new ShardingDropTableStatementValidator());
118         }
119         if (sqlStatement instanceof DropIndexStatement) {
120             return Optional.of(new ShardingDropIndexStatementValidator());
121         }
122         if (sqlStatement instanceof PrepareStatement) {
123             return Optional.of(new ShardingPrepareStatementValidator());
124         }
125         return Optional.empty();
126     }
127     
128     private static Optional<ShardingStatementValidator> getDMLStatementValidator(final SQLStatement sqlStatement, final ShardingConditions shardingConditions) {
129         if (sqlStatement instanceof InsertStatement) {
130             return Optional.of(new ShardingInsertStatementValidator(shardingConditions));
131         }
132         if (sqlStatement instanceof UpdateStatement) {
133             return Optional.of(new ShardingUpdateStatementValidator());
134         }
135         if (sqlStatement instanceof DeleteStatement) {
136             return Optional.of(new ShardingDeleteStatementValidator());
137         }
138         if (sqlStatement instanceof CopyStatement) {
139             return Optional.of(new ShardingCopyStatementValidator());
140         }
141         if (sqlStatement instanceof MySQLLoadDataStatement) {
142             return Optional.of(new ShardingLoadDataStatementValidator());
143         }
144         if (sqlStatement instanceof MySQLLoadXMLStatement) {
145             return Optional.of(new ShardingLoadXMLStatementValidator());
146         }
147         return Optional.empty();
148     }
