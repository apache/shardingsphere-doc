104     public DriverExecutor(final ShardingSphereConnection connection) {
105         this.connection = connection;
106         MetaDataContexts metaDataContexts = connection.getContextManager().getMetaDataContexts();
107         ExecutorEngine executorEngine = connection.getContextManager().getExecutorEngine();
108         JDBCExecutor jdbcExecutor = new JDBCExecutor(executorEngine, connection.getDatabaseConnectionManager().getConnectionContext());
109         regularExecutor = new DriverJDBCExecutor(connection.getDatabaseName(), connection.getContextManager(), jdbcExecutor);
110         rawExecutor = new RawExecutor(executorEngine, connection.getDatabaseConnectionManager().getConnectionContext());
111         String schemaName = new DatabaseTypeRegistry(metaDataContexts.getMetaData().getDatabase(connection.getDatabaseName()).getProtocolType()).getDefaultSchemaName(connection.getDatabaseName());
112         trafficExecutor = new TrafficExecutor();
113         sqlFederationEngine = new SQLFederationEngine(connection.getDatabaseName(), schemaName, metaDataContexts.getMetaData(), metaDataContexts.getStatistics(), jdbcExecutor);
114         kernelProcessor = new KernelProcessor();
115     }

117     /**
118      * Execute query.
119      *
120      * @param metaData meta data
121      * @param database database
122      * @param queryContext query context
123      * @param prepareEngine prepare engine
124      * @param statement statement
125      * @param columnLabelAndIndexMap column label and index map
126      * @param statementReplayCallback statement replay callback
127      * @return result set
128      * @throws SQLException SQL exception
129      */
130     @SuppressWarnings("rawtypes")
131     public ResultSet executeQuery(final ShardingSphereMetaData metaData, final ShardingSphereDatabase database, final QueryContext queryContext,
132                                   final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine, final Statement statement,
133                                   final Map<String, Integer> columnLabelAndIndexMap, final StatementReplayCallback statementReplayCallback) throws SQLException {
134         Optional<String> trafficInstanceId = connection.getTrafficInstanceId(metaData.getGlobalRuleMetaData().getSingleRule(TrafficRule.class), queryContext);
135         if (trafficInstanceId.isPresent()) {
136             return trafficExecutor.execute(
137                     connection.getProcessId(), database.getName(), trafficInstanceId.get(), queryContext, prepareEngine, getTrafficExecuteQueryCallback(prepareEngine.getType()));
138         }
139         if (sqlFederationEngine.decide(queryContext.getSqlStatementContext(), queryContext.getParameters(), database, metaData.getGlobalRuleMetaData())) {
140             return sqlFederationEngine.executeQuery(
141                     prepareEngine, getExecuteQueryCallback(database, queryContext, prepareEngine.getType()), new SQLFederationContext(false, queryContext, metaData, connection.getProcessId()));
142         }
143         List<QueryResult> queryResults = executePushDownQuery(metaData, database, queryContext, prepareEngine, statementReplayCallback);
144         MergedResult mergedResult = mergeQuery(metaData, database, queryResults, queryContext.getSqlStatementContext());
145         boolean selectContainsEnhancedTable = queryContext.getSqlStatementContext() instanceof SelectStatementContext
146                 && ((SelectStatementContext) queryContext.getSqlStatementContext()).isContainsEnhancedTable();
147         List<ResultSet> resultSets = getResultSets();
148         return new ShardingSphereResultSet(resultSets, mergedResult, statement, selectContainsEnhancedTable, queryContext.getSqlStatementContext(),
149                 null == columnLabelAndIndexMap
150                         ? ShardingSphereResultSetUtils.createColumnLabelAndIndexMap(queryContext.getSqlStatementContext(), selectContainsEnhancedTable, resultSets.get(0).getMetaData())
151                         : columnLabelAndIndexMap);
152     }
153     
154     private TrafficExecutorCallback<ResultSet> getTrafficExecuteQueryCallback(final String jdbcDriverType) {
155         return JDBCDriverType.STATEMENT.equals(jdbcDriverType) ? Statement::executeQuery : ((statement, sql) -> ((PreparedStatement) statement).executeQuery());
156     }
157     
158     private ExecuteQueryCallback getExecuteQueryCallback(final ShardingSphereDatabase database, final QueryContext queryContext, final String jdbcDriverType) {
159         return JDBCDriverType.STATEMENT.equals(jdbcDriverType)
160                 ? new StatementExecuteQueryCallback(database.getProtocolType(), database.getResourceMetaData(),
161                         queryContext.getSqlStatementContext().getSqlStatement(), SQLExecutorExceptionHandler.isExceptionThrown())
162                 : new PreparedStatementExecuteQueryCallback(database.getProtocolType(), database.getResourceMetaData(),
163                         queryContext.getSqlStatementContext().getSqlStatement(), SQLExecutorExceptionHandler.isExceptionThrown());
164     }
165     
166     @SuppressWarnings({"rawtypes", "unchecked"})
167     private List<QueryResult> executePushDownQuery(final ShardingSphereMetaData metaData, final ShardingSphereDatabase database, final QueryContext queryContext,
168                                                    final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
169                                                    final StatementReplayCallback statementReplayCallback) throws SQLException {
170         ExecutionContext executionContext = createExecutionContext(metaData, database, queryContext);
171         if (hasRawExecutionRule(database)) {
172             return rawExecutor.execute(createRawExecutionGroupContext(metaData, database, executionContext),
173                     queryContext, new RawSQLExecutorCallback()).stream().map(QueryResult.class::cast).collect(Collectors.toList());
174         }
175         ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext = prepareEngine.prepare(executionContext.getRouteContext(), executionContext.getExecutionUnits(),
176                 new ExecutionGroupReportContext(connection.getProcessId(), database.getName(), new Grantee("", "")));
177         for (ExecutionGroup<JDBCExecutionUnit> each : executionGroupContext.getInputGroups()) {
178             statements.addAll(getStatements(each));
179             if (JDBCDriverType.PREPARED_STATEMENT.equals(prepareEngine.getType())) {
180                 parameterSets.addAll(getParameterSets(each));
181             }
182         }
183         statementReplayCallback.replay(statements, parameterSets);
184         return regularExecutor.executeQuery(executionGroupContext, queryContext, getExecuteQueryCallback(database, queryContext, prepareEngine.getType()));
185     }
186     
187     private boolean hasRawExecutionRule(final ShardingSphereDatabase database) {
188         return !database.getRuleMetaData().getAttributes(RawExecutionRuleAttribute.class).isEmpty();
189     }
190     
191     private Collection<Statement> getStatements(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
192         Collection<Statement> result = new LinkedList<>();
193         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
194             result.add(each.getStorageResource());
195         }
196         return result;
197     }
198     
199     private Collection<List<Object>> getParameterSets(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
200         Collection<List<Object>> result = new LinkedList<>();
201         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
202             result.add(each.getExecutionUnit().getSqlUnit().getParameters());
203         }
204         return result;
205     }
206     
207     private ExecutionContext createExecutionContext(final ShardingSphereMetaData metaData, final ShardingSphereDatabase database, final QueryContext queryContext) throws SQLException {
208         clearStatements();
209         RuleMetaData globalRuleMetaData = metaData.getGlobalRuleMetaData();
210         SQLAuditEngine.audit(queryContext.getSqlStatementContext(), queryContext.getParameters(), globalRuleMetaData, database, null, queryContext.getHintValueContext());
211         return kernelProcessor.generateExecutionContext(queryContext, database, globalRuleMetaData, metaData.getProps(), connection.getDatabaseConnectionManager().getConnectionContext());
212     }
213     
214     private void clearStatements() throws SQLException {
215         for (Statement each : statements) {
216             each.close();
217         }
218         statements.clear();
219     }
220     
221     private ExecutionGroupContext<RawSQLExecutionUnit> createRawExecutionGroupContext(final ShardingSphereMetaData metaData,
222                                                                                       final ShardingSphereDatabase database, final ExecutionContext executionContext) throws SQLException {
223         int maxConnectionsSizePerQuery = metaData.getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
224         return new RawExecutionPrepareEngine(maxConnectionsSizePerQuery, database.getRuleMetaData().getRules()).prepare(
225                 executionContext.getRouteContext(), executionContext.getExecutionUnits(), new ExecutionGroupReportContext(connection.getProcessId(), database.getName(), new Grantee("", "")));
226     }
227     
228     private MergedResult mergeQuery(final ShardingSphereMetaData metaData, final ShardingSphereDatabase database,
229                                     final List<QueryResult> queryResults, final SQLStatementContext sqlStatementContext) throws SQLException {
230         MergeEngine mergeEngine = new MergeEngine(metaData.getGlobalRuleMetaData(), database, metaData.getProps(), connection.getDatabaseConnectionManager().getConnectionContext());
231         return mergeEngine.merge(queryResults, sqlStatementContext);
232     }
233     
234     private List<ResultSet> getResultSets() throws SQLException {
235         List<ResultSet> result = new ArrayList<>(statements.size());
236         for (Statement each : statements) {
237             if (null != each.getResultSet()) {
238                 result.add(each.getResultSet());
239             }
240         }
241         return result;
242     }

244     /**
245      * Execute advance update.
246      *
247      * @param metaData meta data
248      * @param database database
249      * @param queryContext query context
250      * @param prepareEngine prepare engine
251      * @param trafficCallback traffic callback
252      * @return updated row count
253      * @throws SQLException SQL exception
254      */
255     public Optional<Integer> executeAdvanceUpdate(final ShardingSphereMetaData metaData, final ShardingSphereDatabase database, final QueryContext queryContext,
256                                                   final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
257                                                   final TrafficExecutorCallback<Integer> trafficCallback) throws SQLException {
258         Optional<String> trafficInstanceId = connection.getTrafficInstanceId(metaData.getGlobalRuleMetaData().getSingleRule(TrafficRule.class), queryContext);
259         if (trafficInstanceId.isPresent()) {
260             return Optional.of(trafficExecutor.execute(connection.getProcessId(), database.getName(), trafficInstanceId.get(), queryContext, prepareEngine, trafficCallback));
261         }
262         return Optional.empty();
263     }
264     
265     /**
266      * Execute advance.
267      *
268      * @param metaData meta data
269      * @param database database
270      * @param queryContext query context
271      * @param prepareEngine prepare engine
272      * @param trafficCallback traffic callback
273      * @return execute result
274      * @throws SQLException SQL exception
275      */
276     public Optional<Boolean> executeAdvance(final ShardingSphereMetaData metaData, final ShardingSphereDatabase database,
277                                             final QueryContext queryContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
278                                             final TrafficExecutorCallback<Boolean> trafficCallback) throws SQLException {
279         Optional<String> trafficInstanceId = connection.getTrafficInstanceId(metaData.getGlobalRuleMetaData().getSingleRule(TrafficRule.class), queryContext);
280         if (trafficInstanceId.isPresent()) {
281             executeType = ExecuteType.TRAFFIC;
282             return Optional.of(trafficExecutor.execute(connection.getProcessId(), database.getName(), trafficInstanceId.get(), queryContext, prepareEngine, trafficCallback));
283         }
284         if (sqlFederationEngine.decide(queryContext.getSqlStatementContext(), queryContext.getParameters(), database, metaData.getGlobalRuleMetaData())) {
285             executeType = ExecuteType.FEDERATION;
286             ResultSet resultSet = sqlFederationEngine.executeQuery(
287                     prepareEngine, getExecuteQueryCallback(database, queryContext, prepareEngine.getType()), new SQLFederationContext(false, queryContext, metaData, connection.getProcessId()));
288             return Optional.of(null != resultSet);
289         }
290         return Optional.empty();
291     }
292     
293     /**
294      * Get advanced result set.
295      *
296      * @return advanced result set
297      */
298     public Optional<ResultSet> getAdvancedResultSet() {
299         switch (executeType) {
300             case TRAFFIC:
301                 return Optional.of(trafficExecutor.getResultSet());
302             case FEDERATION:
303                 return Optional.of(sqlFederationEngine.getResultSet());
304             default:
305                 return Optional.empty();
306         }
307     }
308     
309     /**
310      * Clear.
311      */
312     public void clear() {
313         statements.clear();
314         parameterSets.clear();
315     }
316     
317     @Override
318     public void close() throws SQLException {
319         sqlFederationEngine.close();
320         trafficExecutor.close();
321     }
322     
323     public enum ExecuteType {
324         
325         TRAFFIC, FEDERATION, REGULAR
326     }
