18  package org.apache.shardingsphere.infra.datanode;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  
23  import java.util.Collection;
24  import java.util.Collections;
25  import java.util.Iterator;
26  import java.util.LinkedHashMap;
27  import java.util.LinkedList;
28  import java.util.List;
29  import java.util.Map;
30  
31  /**
32   * Data node utility class.
33   */
34  @NoArgsConstructor(access = AccessLevel.PRIVATE)
35  public final class DataNodeUtils {
36      
37      /**
38       * Get data node groups.
39       *
40       * @param dataNodes data nodes
41       * @return data node groups, key is data source name, values are data nodes belong to this data source
42       */
43      public static Map<String, List<DataNode>> getDataNodeGroups(final Collection<DataNode> dataNodes) {
44          Map<String, List<DataNode>> result = new LinkedHashMap<>(dataNodes.size(), 1F);
45          for (DataNode each : dataNodes) {
46              result.computeIfAbsent(each.getDataSourceName(), unused -> new LinkedList<>()).add(each);
47          }
48          return result;
49      }
50      
51      /**
52       * Build data node.
53       *
54       * @param dataNode data node
55       * @param dataSources dataSource map
56       * @return data node collection
57       */
58      public static Collection<DataNode> buildDataNode(final DataNode dataNode, final Map<String, Collection<String>> dataSources) {
59          if (!dataSources.containsKey(dataNode.getDataSourceName())) {
60              return Collections.singletonList(dataNode);
61          }
62          Collection<DataNode> result = new LinkedList<>();
63          for (String each : dataSources.get(dataNode.getDataSourceName())) {
64              result.add(new DataNode(each, dataNode.getTableName()));
65          }
66          return result;
67      }
68      
69      /**
70       * Get format data nodes.
71       *
72       * @param amount amount
73       * @param logicTable logic table
74       * @param dataSources data source names
75       * @return data node list
76       */
77      public static List<String> getFormatDataNodes(final int amount, final String logicTable, final Collection<String> dataSources) {
78          List<String> result = new LinkedList<>();
79          Iterator<String> iterator = dataSources.iterator();
80          for (int i = 0; i < amount; i++) {
81              if (!iterator.hasNext()) {
82                  iterator = dataSources.iterator();
83              }
84              result.add(String.format("%s.%s_%s", iterator.next(), logicTable, i));
85          }
86          return result;
87      }
88  }
