43      @ParameterizedTest(name = "{0}")
44      @ArgumentsSource(TestCaseArgumentsProvider.class)
45      void assertSQLFormat(final String caseId, final String inputSQL, final String expectFormattedSQL) {
46          CodePointBuffer buffer = CodePointBuffer.withChars(CharBuffer.wrap(inputSQL.toCharArray()));
47          MySQLLexer lexer = new MySQLLexer(CodePointCharStream.fromBuffer(buffer));
48          MySQLParser parser = new MySQLParser(new CommonTokenStream(lexer));
49          ParseTree tree = ((ParseASTNode) parser.parse()).getRootNode();
50          MySQLFormatSQLVisitor visitor = new MySQLDMLFormatSQLVisitor();
51          visitor.setParameterized(false);
52          assertThat(visitor.visit(tree), is(expectFormattedSQL));
53      }
