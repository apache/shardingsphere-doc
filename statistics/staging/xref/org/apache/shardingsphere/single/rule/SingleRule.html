45  /**
46   * Single rule.
47   */
48  public final class SingleRule implements DatabaseRule {
49      
50      @Getter
51      private final SingleRuleConfiguration configuration;
52      
53      private final String defaultDataSource;
54      
55      @Getter
56      private final Collection<String> dataSourceNames;
57      
58      private final Map<String, Collection<DataNode>> singleTableDataNodes;
59      
60      private final DatabaseType protocolType;
61      
62      private final SingleMutableDataNodeRuleAttribute mutableDataNodeRuleAttribute;
63      
64      @Getter
65      private final RuleAttributes attributes;
66      
67      public SingleRule(final SingleRuleConfiguration ruleConfig, final String databaseName,
68                        final DatabaseType protocolType, final Map<String, DataSource> dataSourceMap, final Collection<ShardingSphereRule> builtRules) {
69          configuration = ruleConfig;
70          defaultDataSource = ruleConfig.getDefaultDataSource().orElse(null);
71          Map<String, DataSource> aggregateDataSourceMap = SingleTableLoadUtils.getAggregatedDataSourceMap(dataSourceMap, builtRules);
72          dataSourceNames = aggregateDataSourceMap.keySet();
73          this.protocolType = protocolType;
74          singleTableDataNodes = SingleTableDataNodeLoader.load(databaseName, protocolType, aggregateDataSourceMap, builtRules, configuration.getTables());
75          SingleTableMapperRuleAttribute tableMapperRuleAttribute = new SingleTableMapperRuleAttribute(singleTableDataNodes.values());
76          mutableDataNodeRuleAttribute = new SingleMutableDataNodeRuleAttribute(configuration, dataSourceNames, singleTableDataNodes, protocolType, tableMapperRuleAttribute);
77          attributes = new RuleAttributes(
78                  new SingleDataNodeRuleAttribute(singleTableDataNodes), tableMapperRuleAttribute, new SingleExportableRuleAttribute(tableMapperRuleAttribute), mutableDataNodeRuleAttribute);
79      }
80      
81      /**
82       * Assign new data source name.
83       *
84       * @return assigned data source name
85       */
86      public String assignNewDataSourceName() {
87          return null == defaultDataSource ? new ArrayList<>(dataSourceNames).get(ThreadLocalRandom.current().nextInt(dataSourceNames.size())) : defaultDataSource;
88      }
89      
90      /**
91       * Judge whether all tables are in same compute node or not.
92       * 
93       * @param dataNodes data nodes
94       * @param singleTables single tables
95       * @return whether all tables are in same compute node or not
96       */
97      public boolean isAllTablesInSameComputeNode(final Collection<DataNode> dataNodes, final Collection<QualifiedTable> singleTables) {
98          if (!isSingleTablesInSameComputeNode(singleTables)) {
99              return false;
100         }
101         QualifiedTable sampleTable = singleTables.iterator().next();
102         Optional<DataNode> sampleDataNode = mutableDataNodeRuleAttribute.findTableDataNode(sampleTable.getSchemaName(), sampleTable.getTableName());
103         if (sampleDataNode.isPresent()) {
104             for (DataNode each : dataNodes) {
105                 if (!isSameComputeNode(sampleDataNode.get().getDataSourceName(), each.getDataSourceName())) {
106                     return false;
107                 }
108             }
109         }
110         return true;
111     }
112     
113     private boolean isSameComputeNode(final String sampleDataSourceName, final String dataSourceName) {
114         return sampleDataSourceName.equalsIgnoreCase(dataSourceName);
115     }
116     
117     private boolean isSingleTablesInSameComputeNode(final Collection<QualifiedTable> singleTables) {
118         String sampleDataSourceName = null;
119         for (QualifiedTable each : singleTables) {
120             Optional<DataNode> dataNode = mutableDataNodeRuleAttribute.findTableDataNode(each.getSchemaName(), each.getTableName());
121             if (!dataNode.isPresent()) {
122                 continue;
123             }
124             if (null == sampleDataSourceName) {
125                 sampleDataSourceName = dataNode.get().getDataSourceName();
126                 continue;
127             }
128             if (!isSameComputeNode(sampleDataSourceName, dataNode.get().getDataSourceName())) {
129                 return false;
130             }
131         }
132         return true;
133     }
134     
135     /**
136      * Get single table names.
137      *
138      * @param qualifiedTables qualified tables
139      * @return single table names
140      */
141     public Collection<QualifiedTable> getSingleTables(final Collection<QualifiedTable> qualifiedTables) {
142         Collection<QualifiedTable> result = new LinkedList<>();
143         for (QualifiedTable each : qualifiedTables) {
144             Collection<DataNode> dataNodes = singleTableDataNodes.getOrDefault(each.getTableName().toLowerCase(), new LinkedList<>());
145             if (!dataNodes.isEmpty() && containsDataNode(each, dataNodes)) {
146                 result.add(each);
147             }
148         }
149         return result;
150     }
151     
152     private boolean containsDataNode(final QualifiedTable qualifiedTable, final Collection<DataNode> dataNodes) {
153         for (DataNode each : dataNodes) {
154             if (qualifiedTable.getSchemaName().equalsIgnoreCase(each.getSchemaName())) {
155                 return true;
156             }
157         }
158         return false;
159     }
160     
161     /**
162      * Get qualified tables.
163      *
164      * @param sqlStatementContext sql statement context
165      * @param database database
166      * @return qualified tables
167      */
168     public Collection<QualifiedTable> getQualifiedTables(final SQLStatementContext sqlStatementContext, final ShardingSphereDatabase database) {
169         Collection<QualifiedTable> result = getQualifiedTables(database, protocolType, sqlStatementContext.getTablesContext().getSimpleTables());
170         if (result.isEmpty() && sqlStatementContext instanceof IndexAvailable) {
171             result = IndexMetaDataUtils.getTableNames(database, protocolType, ((IndexAvailable) sqlStatementContext).getIndexes());
172         }
173         return result;
174     }
175     
176     private Collection<QualifiedTable> getQualifiedTables(final ShardingSphereDatabase database, final DatabaseType databaseType, final Collection<SimpleTableSegment> tableSegments) {
177         Collection<QualifiedTable> result = new ArrayList<>(tableSegments.size());
178         String schemaName = new DatabaseTypeRegistry(databaseType).getDefaultSchemaName(database.getName());
179         for (SimpleTableSegment each : tableSegments) {
180             String actualSchemaName = each.getOwner().map(optional -> optional.getIdentifier().getValue()).orElse(schemaName);
181             result.add(new QualifiedTable(actualSchemaName, each.getTableName().getIdentifier().getValue()));
182         }
183         return result;
184     }
