44  /**
45   * Insert on duplicate key update parameter rewriter for encrypt.
46   */
47  @Setter
48  public final class EncryptInsertOnDuplicateKeyUpdateValueParameterRewriter implements ParameterRewriter<InsertStatementContext>, EncryptRuleAware, DatabaseNameAware {
49      
50      private EncryptRule encryptRule;
51      
52      private String databaseName;
53      
54      @Override
55      public boolean isNeedRewrite(final SQLStatementContext<?> sqlStatementContext) {
56          return sqlStatementContext instanceof InsertStatementContext
57                  && InsertStatementHandler.getOnDuplicateKeyColumnsSegment(((InsertStatementContext) sqlStatementContext).getSqlStatement()).isPresent();
58      }
59      
60      @SuppressWarnings({"rawtypes", "unchecked"})
61      @Override
62      public void rewrite(final ParameterBuilder paramBuilder, final InsertStatementContext insertStatementContext, final List<Object> params) {
63          String tableName = insertStatementContext.getSqlStatement().getTable().getTableName().getIdentifier().getValue();
64          GroupedParameterBuilder groupedParamBuilder = (GroupedParameterBuilder) paramBuilder;
65          OnDuplicateUpdateContext onDuplicateKeyUpdateValueContext = insertStatementContext.getOnDuplicateKeyUpdateValueContext();
66          String schemaName = insertStatementContext.getTablesContext().getSchemaName().orElseGet(() -> DatabaseTypeEngine.getDefaultSchemaName(insertStatementContext.getDatabaseType(), databaseName));
67          for (int index = 0; index < onDuplicateKeyUpdateValueContext.getValueExpressions().size(); index++) {
68              String encryptLogicColumnName = onDuplicateKeyUpdateValueContext.getColumn(index).getIdentifier().getValue();
69              EncryptContext encryptContext = EncryptContextBuilder.build(databaseName, schemaName, tableName, encryptLogicColumnName);
70              Optional<StandardEncryptAlgorithm> encryptor = encryptRule.findEncryptor(tableName, encryptLogicColumnName);
71              if (!encryptor.isPresent()) {
72                  continue;
73              }
74              Object plainColumnValue = onDuplicateKeyUpdateValueContext.getValue(index);
75              if (plainColumnValue instanceof FunctionSegment && "VALUES".equalsIgnoreCase(((FunctionSegment) plainColumnValue).getFunctionName())) {
76                  return;
77              }
78              Object cipherColumnValue = encryptor.get().encrypt(plainColumnValue, encryptContext);
79              groupedParamBuilder.getGenericParameterBuilder().addReplacedParameters(index, cipherColumnValue);
80              Collection<Object> addedParams = new LinkedList<>();
81              Optional<StandardEncryptAlgorithm> assistedQueryEncryptor = encryptRule.findAssistedQueryEncryptor(tableName, encryptLogicColumnName);
82              if (assistedQueryEncryptor.isPresent()) {
83                  Optional<String> assistedColumnName = encryptRule.findAssistedQueryColumn(tableName, encryptLogicColumnName);
84                  Preconditions.checkArgument(assistedColumnName.isPresent(), "Can not find assisted query Column Name");
85                  addedParams.add(assistedQueryEncryptor.get().encrypt(plainColumnValue, encryptContext));
86              }
87              Optional<LikeEncryptAlgorithm> likeQueryEncryptor = encryptRule.findLikeQueryEncryptor(tableName, encryptLogicColumnName);
88              if (likeQueryEncryptor.isPresent()) {
89                  Optional<String> likeColumnName = encryptRule.findLikeQueryColumn(tableName, encryptLogicColumnName);
90                  Preconditions.checkArgument(likeColumnName.isPresent(), "Can not find assisted query Column Name");
91                  addedParams.add(likeQueryEncryptor.get().encrypt(plainColumnValue, encryptContext));
92              }
93              if (encryptRule.findPlainColumn(tableName, encryptLogicColumnName).isPresent()) {
94                  addedParams.add(plainColumnValue);
95              }
96              if (!addedParams.isEmpty()) {
97                  if (!groupedParamBuilder.getGenericParameterBuilder().getAddedIndexAndParameters().containsKey(index)) {
98                      groupedParamBuilder.getGenericParameterBuilder().getAddedIndexAndParameters().put(index, new LinkedList<>());
99                  }
100                 groupedParamBuilder.getGenericParameterBuilder().getAddedIndexAndParameters().get(index).addAll(addedParams);
101             }
102         }
103     }
104 }
