47  /**
48   * Driver raw push down execute query executor.
49   */
50  public final class DriverRawPushDownExecuteQueryExecutor {
51      
52      private final ConnectionContext connectionContext;
53      
54      private final String processId;
55      
56      private final RuleMetaData globalRuleMetaData;
57      
58      private final ConfigurationProperties props;
59      
60      private final RawExecutor rawExecutor;
61      
62      public DriverRawPushDownExecuteQueryExecutor(final ShardingSphereConnection connection, final ShardingSphereMetaData metaData, final RawExecutor rawExecutor) {
63          connectionContext = connection.getDatabaseConnectionManager().getConnectionContext();
64          processId = connection.getProcessId();
65          globalRuleMetaData = metaData.getGlobalRuleMetaData();
66          props = metaData.getProps();
67          this.rawExecutor = rawExecutor;
68      }
69      
70      /**
71       * Execute query.
72       * 
73       * @param database database
74       * @param queryContext query context
75       * @param statement statement
76       * @param columnLabelAndIndexMap column label and index map
77       * @return result set
78       * @throws SQLException SQL exception
79       */
80      public ResultSet executeQuery(final ShardingSphereDatabase database, final QueryContext queryContext, final Statement statement,
81                                    final Map<String, Integer> columnLabelAndIndexMap) throws SQLException {
82          List<QueryResult> queryResults = getQueryResults(database, queryContext);
83          return new ShardingSphereResultSetFactory(connectionContext, globalRuleMetaData, props, Collections.emptyList())
84                  .newInstance(database, queryContext, queryResults, statement, columnLabelAndIndexMap);
85      }
86      
87      private List<QueryResult> getQueryResults(final ShardingSphereDatabase database, final QueryContext queryContext) throws SQLException {
88          ExecutionContext executionContext = new KernelProcessor().generateExecutionContext(queryContext, globalRuleMetaData, props, connectionContext);
89          return rawExecutor.execute(
90                  createRawExecutionGroupContext(database, executionContext), queryContext, new RawSQLExecutorCallback()).stream().map(QueryResult.class::cast).collect(Collectors.toList());
91      }
92      
93      private ExecutionGroupContext<RawSQLExecutionUnit> createRawExecutionGroupContext(final ShardingSphereDatabase database, final ExecutionContext executionContext) throws SQLException {
94          int maxConnectionsSizePerQuery = props.<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
95          return new RawExecutionPrepareEngine(maxConnectionsSizePerQuery, database.getRuleMetaData().getRules()).prepare(database.getName(),
96                  executionContext.getRouteContext(), executionContext.getExecutionUnits(), new ExecutionGroupReportContext(processId, database.getName(), connectionContext.getGrantee()));
97      }
98  }
