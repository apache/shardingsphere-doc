59  /**
60   * Meta data contexts factory.
61   */
62  @NoArgsConstructor(access = AccessLevel.PRIVATE)
63  public final class MetaDataContextsFactory {
64      
65      /**
66       * Create meta data contexts.
67       * 
68       * @param persistService persist service
69       * @param param context manager builder parameter
70       * @param computeNodeInstanceContext compute node instance context
71       * @return meta data contexts
72       * @throws SQLException SQL exception
73       */
74      public static MetaDataContexts create(final MetaDataPersistService persistService, final ContextManagerBuilderParameter param,
75                                            final ComputeNodeInstanceContext computeNodeInstanceContext) throws SQLException {
76          boolean isDatabaseMetaDataExisted = !persistService.getDatabaseMetaDataService().loadAllDatabaseNames().isEmpty();
77          Map<String, DatabaseConfiguration> effectiveDatabaseConfigs = isDatabaseMetaDataExisted
78                  ? createEffectiveDatabaseConfigurations(getDatabaseNames(computeNodeInstanceContext, param.getDatabaseConfigs(), persistService), param.getDatabaseConfigs(), persistService)
79                  : param.getDatabaseConfigs();
80          // TODO load global data sources from persist service
81          Map<String, DataSource> globalDataSources = param.getGlobalDataSources();
82          Collection<RuleConfiguration> globalRuleConfigs = isDatabaseMetaDataExisted ? persistService.getGlobalRuleService().load() : param.getGlobalRuleConfigs();
83          ConfigurationProperties props = isDatabaseMetaDataExisted ? new ConfigurationProperties(persistService.getPropsService().load()) : new ConfigurationProperties(param.getProps());
84          Map<String, ShardingSphereDatabase> databases = isDatabaseMetaDataExisted
85                  ? InternalMetaDataFactory.create(persistService, effectiveDatabaseConfigs, props, computeNodeInstanceContext)
86                  : ExternalMetaDataFactory.create(effectiveDatabaseConfigs, props, computeNodeInstanceContext);
87          ResourceMetaData globalResourceMetaData = new ResourceMetaData(globalDataSources);
88          RuleMetaData globalRuleMetaData = new RuleMetaData(GlobalRulesBuilder.buildRules(globalRuleConfigs, databases, props));
89          ShardingSphereMetaData shardingSphereMetaData = new ShardingSphereMetaData(databases, globalResourceMetaData, globalRuleMetaData, props);
90          ShardingSphereStatistics shardingSphereStatistics = initStatistics(persistService, shardingSphereMetaData);
91          MetaDataContexts result = new MetaDataContexts(shardingSphereMetaData, shardingSphereStatistics);
92          if (!isDatabaseMetaDataExisted) {
93              persistDatabaseConfigurations(result, param, persistService);
94              persistMetaData(result, persistService);
95          }
96          return result;
97      }
98      
99      /**
100      * Create meta data contexts.
101      *
102      * @param persistService meta data persist service
103      * @param metaData shardingsphere meta data
104      * @return meta data contexts
105      */
106     public static MetaDataContexts create(final MetaDataPersistService persistService, final ShardingSphereMetaData metaData) {
107         return new MetaDataContexts(metaData, initStatistics(persistService, metaData));
108     }
109     
110     private static Collection<String> getDatabaseNames(final ComputeNodeInstanceContext computeNodeInstanceContext,
111                                                        final Map<String, DatabaseConfiguration> databaseConfigs, final MetaDataPersistService persistService) {
112         return computeNodeInstanceContext.getInstance().getMetaData() instanceof JDBCInstanceMetaData ? databaseConfigs.keySet() : persistService.getDatabaseMetaDataService().loadAllDatabaseNames();
113     }
114     
115     private static Map<String, DatabaseConfiguration> createEffectiveDatabaseConfigurations(final Collection<String> databaseNames,
116                                                                                             final Map<String, DatabaseConfiguration> databaseConfigs, final MetaDataPersistService persistService) {
117         return databaseNames.stream().collect(Collectors.toMap(each -> each, each -> createEffectiveDatabaseConfiguration(each, databaseConfigs, persistService)));
118     }
119     
120     private static DatabaseConfiguration createEffectiveDatabaseConfiguration(final String databaseName,
121                                                                               final Map<String, DatabaseConfiguration> databaseConfigs, final MetaDataPersistService persistService) {
122         closeGeneratedDataSources(databaseName, databaseConfigs);
123         Map<String, DataSourceConfiguration> dataSources = persistService.loadDataSourceConfigurations(databaseName);
124         Collection<RuleConfiguration> databaseRuleConfigs = persistService.getDatabaseRulePersistService().load(databaseName);
125         return new DataSourceGeneratedDatabaseConfiguration(dataSources, databaseRuleConfigs);
126     }
127     
128     private static void closeGeneratedDataSources(final String databaseName, final Map<String, ? extends DatabaseConfiguration> databaseConfigs) {
129         if (databaseConfigs.containsKey(databaseName) && !databaseConfigs.get(databaseName).getStorageUnits().isEmpty()) {
130             databaseConfigs.get(databaseName).getDataSources().values().forEach(each -> new DataSourcePoolDestroyer(each).asyncDestroy());
131         }
132     }
133     
134     private static ShardingSphereStatistics initStatistics(final MetaDataPersistService persistService, final ShardingSphereMetaData metaData) {
135         if (metaData.getDatabases().isEmpty()) {
136             return new ShardingSphereStatistics();
137         }
138         DatabaseType protocolType = metaData.getDatabases().values().iterator().next().getProtocolType();
139         DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(protocolType).getDialectDatabaseMetaData();
140         // TODO can `protocolType instanceof SchemaSupportedDatabaseType ? "PostgreSQL" : protocolType.getType()` replace to trunk database type?
141         DatabaseType databaseType = dialectDatabaseMetaData.getDefaultSchema().isPresent() ? TypedSPILoader.getService(DatabaseType.class, "PostgreSQL") : protocolType;
142         Optional<ShardingSphereStatisticsBuilder> statisticsBuilder = DatabaseTypedSPILoader.findService(ShardingSphereStatisticsBuilder.class, databaseType);
143         if (!statisticsBuilder.isPresent()) {
144             return new ShardingSphereStatistics();
145         }
146         ShardingSphereStatistics result = statisticsBuilder.get().build(metaData);
147         Optional<ShardingSphereStatistics> loadedStatistics = persistService.getShardingSphereDataPersistService().load(metaData);
148         loadedStatistics.ifPresent(optional -> useLoadedToReplaceInit(result, optional));
149         return result;
150     }
151     
152     private static void useLoadedToReplaceInit(final ShardingSphereStatistics initStatistics, final ShardingSphereStatistics loadedStatistics) {
153         for (Entry<String, ShardingSphereDatabaseData> entry : initStatistics.getDatabaseData().entrySet()) {
154             if (loadedStatistics.getDatabaseData().containsKey(entry.getKey())) {
155                 useLoadedToReplaceInitByDatabaseData(entry.getValue(), loadedStatistics.getDatabaseData().get(entry.getKey()));
156             }
157         }
158     }
159     
160     private static void useLoadedToReplaceInitByDatabaseData(final ShardingSphereDatabaseData initDatabaseData, final ShardingSphereDatabaseData loadedDatabaseData) {
161         for (Entry<String, ShardingSphereSchemaData> entry : initDatabaseData.getSchemaData().entrySet()) {
162             if (loadedDatabaseData.getSchemaData().containsKey(entry.getKey())) {
163                 useLoadedToReplaceInitBySchemaData(entry.getValue(), loadedDatabaseData.getSchemaData().get(entry.getKey()));
164             }
165         }
166     }
167     
168     private static void useLoadedToReplaceInitBySchemaData(final ShardingSphereSchemaData initSchemaData, final ShardingSphereSchemaData loadedSchemaData) {
169         for (Entry<String, ShardingSphereTableData> entry : initSchemaData.getTableData().entrySet()) {
170             if (loadedSchemaData.getTableData().containsKey(entry.getKey())) {
171                 entry.setValue(loadedSchemaData.getTableData().get(entry.getKey()));
172             }
173         }
174     }
175     
176     private static void persistDatabaseConfigurations(final MetaDataContexts metadataContexts, final ContextManagerBuilderParameter param, final MetaDataPersistService persistService) {
177         Collection<RuleConfiguration> globalRuleConfigs = param.getGlobalRuleConfigs();
178         persistService.persistGlobalRuleConfiguration(globalRuleConfigs, param.getProps());
179         for (Entry<String, ? extends DatabaseConfiguration> entry : param.getDatabaseConfigs().entrySet()) {
180             String databaseName = entry.getKey();
181             persistService.persistConfigurations(entry.getKey(), entry.getValue(),
182                     metadataContexts.getMetaData().getDatabase(databaseName).getResourceMetaData().getStorageUnits().entrySet().stream()
183                             .collect(Collectors.toMap(Entry::getKey, each -> each.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
184                     metadataContexts.getMetaData().getDatabase(databaseName).getRuleMetaData().getRules());
185         }
186     }
187     
188     private static void persistMetaData(final MetaDataContexts metaDataContexts, final MetaDataPersistService persistService) {
189         metaDataContexts.getMetaData().getDatabases().values().forEach(each -> each.getSchemas().forEach((schemaName, schema) -> persistService.getDatabaseMetaDataService()
190                 .persistByAlterConfiguration(each.getName(), schemaName, schema)));
191         metaDataContexts.getStatistics().getDatabaseData().forEach((databaseName, databaseData) -> databaseData.getSchemaData().forEach((schemaName, schemaData) -> persistService
192                 .getShardingSphereDataPersistService().persist(databaseName, schemaName, schemaData, metaDataContexts.getMetaData().getDatabases())));
193     }
