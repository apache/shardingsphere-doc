18  package org.apache.shardingsphere.data.pipeline.core.util;
19  
20  import org.apache.shardingsphere.infra.metadata.caseinsensitive.CaseInsensitiveIdentifier;
21  import org.apache.shardingsphere.infra.yaml.config.pojo.rule.YamlRuleConfiguration;
22  import org.apache.shardingsphere.sharding.api.config.ShardingRuleConfiguration;
23  import org.apache.shardingsphere.sharding.api.config.rule.ShardingAutoTableRuleConfiguration;
24  import org.apache.shardingsphere.sharding.api.config.rule.ShardingTableRuleConfiguration;
25  import org.apache.shardingsphere.sharding.api.config.strategy.sharding.ComplexShardingStrategyConfiguration;
26  import org.apache.shardingsphere.sharding.api.config.strategy.sharding.ShardingStrategyConfiguration;
27  import org.apache.shardingsphere.sharding.api.config.strategy.sharding.StandardShardingStrategyConfiguration;
28  import org.apache.shardingsphere.sharding.yaml.swapper.ShardingRuleConfigurationConverter;
29  
30  import java.util.Arrays;
31  import java.util.Collection;
32  import java.util.Collections;
33  import java.util.HashSet;
34  import java.util.Map;
35  import java.util.Optional;
36  import java.util.Set;
37  import java.util.concurrent.ConcurrentHashMap;
38  
39  /**
40   * Sharding columns extractor.
41   */
42  public final class ShardingColumnsExtractor {
43      
44      /**
45       * Get sharding columns map.
46       *
47       * @param yamlRuleConfigs YAML rule configurations
48       * @param logicTableNames logic table names
49       * @return sharding columns map
50       */
51      public Map<CaseInsensitiveIdentifier, Set<String>> getShardingColumnsMap(final Collection<YamlRuleConfiguration> yamlRuleConfigs, final Set<CaseInsensitiveIdentifier> logicTableNames) {
52          Optional<ShardingRuleConfiguration> shardingRuleConfig = ShardingRuleConfigurationConverter.findAndConvertShardingRuleConfiguration(yamlRuleConfigs);
53          if (!shardingRuleConfig.isPresent()) {
54              return Collections.emptyMap();
55          }
56          Set<String> defaultDatabaseShardingColumns = extractShardingColumns(shardingRuleConfig.get().getDefaultDatabaseShardingStrategy());
57          Set<String> defaultTableShardingColumns = extractShardingColumns(shardingRuleConfig.get().getDefaultTableShardingStrategy());
58          Map<CaseInsensitiveIdentifier, Set<String>> result = new ConcurrentHashMap<>(shardingRuleConfig.get().getTables().size(), 1F);
59          for (ShardingTableRuleConfiguration each : shardingRuleConfig.get().getTables()) {
60              CaseInsensitiveIdentifier logicTableName = new CaseInsensitiveIdentifier(each.getLogicTable());
61              if (!logicTableNames.contains(logicTableName)) {
62                  continue;
63              }
64              Set<String> shardingColumns = new HashSet<>();
65              shardingColumns.addAll(null == each.getDatabaseShardingStrategy() ? defaultDatabaseShardingColumns : extractShardingColumns(each.getDatabaseShardingStrategy()));
66              shardingColumns.addAll(null == each.getTableShardingStrategy() ? defaultTableShardingColumns : extractShardingColumns(each.getTableShardingStrategy()));
67              result.put(logicTableName, shardingColumns);
68          }
69          for (ShardingAutoTableRuleConfiguration each : shardingRuleConfig.get().getAutoTables()) {
70              CaseInsensitiveIdentifier logicTableName = new CaseInsensitiveIdentifier(each.getLogicTable());
71              if (!logicTableNames.contains(logicTableName)) {
72                  continue;
73              }
74              ShardingStrategyConfiguration shardingStrategy = each.getShardingStrategy();
75              Set<String> shardingColumns = new HashSet<>(extractShardingColumns(shardingStrategy));
76              result.put(logicTableName, shardingColumns);
77          }
78          return result;
79      }
80      
81      private Set<String> extractShardingColumns(final ShardingStrategyConfiguration shardingStrategy) {
82          if (shardingStrategy instanceof StandardShardingStrategyConfiguration) {
83              return new HashSet<>(Collections.singleton(((StandardShardingStrategyConfiguration) shardingStrategy).getShardingColumn()));
84          }
85          if (shardingStrategy instanceof ComplexShardingStrategyConfiguration) {
86              return new HashSet<>(Arrays.asList(((ComplexShardingStrategyConfiguration) shardingStrategy).getShardingColumns().split(",")));
87          }
88          return Collections.emptySet();
89      }
90  }
