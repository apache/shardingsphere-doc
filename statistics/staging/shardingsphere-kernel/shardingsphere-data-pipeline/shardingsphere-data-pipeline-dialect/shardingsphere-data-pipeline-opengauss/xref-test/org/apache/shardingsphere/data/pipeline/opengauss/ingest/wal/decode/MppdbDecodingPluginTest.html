62      @Test
63      void assertDecodeWriteRowEvent() {
64          MppTableData tableData = new MppTableData();
65          tableData.setTableName("public.test");
66          tableData.setOpType("INSERT");
67          String[] insertTypes = new String[]{"character varying", "text", "char", "character", "nchar", "varchar2", "nvarchar2", "clob"};
68          tableData.setColumnsType(insertTypes);
69          tableData.setColumnsName(IntStream.range(0, insertTypes.length).mapToObj(idx -> "data" + idx).toArray(String[]::new));
70          tableData.setColumnsVal(IntStream.range(0, insertTypes.length).mapToObj(idx -> "'1 2 3'").toArray(String[]::new));
71          ByteBuffer data = ByteBuffer.wrap(toJSON(tableData).getBytes());
72          WriteRowEvent actual = (WriteRowEvent) new MppdbDecodingPlugin(null).decode(data, logSequenceNumber);
73          assertThat(actual.getLogSequenceNumber(), is(logSequenceNumber));
74          assertThat(actual.getTableName(), is("test"));
75          IntStream.range(0, insertTypes.length).forEach(each -> assertThat(actual.getAfterRow().get(each), is("1 2 3")));
76      }

147     @Test
148     void assertDecodeWriteRowEventWithDateAndTime() throws SQLException {
149         MppTableData tableData = new MppTableData();
150         tableData.setTableName("public.test");
151         tableData.setOpType("INSERT");
152         String[] insertTypes = new String[]{
153                 "time without time zone", "time with time zone", "timestamp without time zone", "timestamp with time zone", "smalldatetime", "date", "interval", "reltime"};
154         String[] insertValues = new String[]{"'21:21:21'", "'21:21:21 pst'", "'2010-12-12'", "'2013-12-11 pst'", "'2003-04-12 04:05:06'", "'2021-10-10'", "'3 days'", "'2 mons'"};
155         final String[] compareValues = {
156                 "21:21:21", "13:21:21", "2010-12-12 00:00:00.0", "2013-12-11 16:00:00.0", "2003-04-12 04:05:00.0", "2021-10-10", "0 years 0 mons 3 days 0 hours 0 mins 0.00 secs", "2 mons"};
157         tableData.setColumnsName(IntStream.range(0, insertTypes.length).mapToObj(idx -> "data" + idx).toArray(String[]::new));
158         tableData.setColumnsType(insertTypes);
159         tableData.setColumnsVal(insertValues);
160         ByteBuffer data = ByteBuffer.wrap(toJSON(tableData).getBytes());
161         TimestampUtils timestampUtils = mock(TimestampUtils.class);
162         when(timestampUtils.toTime(null, "21:21:21")).thenReturn(Time.valueOf("21:21:21"));
163         when(timestampUtils.toTime(null, "21:21:21 pst")).thenReturn(Time.valueOf("13:21:21"));
164         when(timestampUtils.toTimestamp(null, "2010-12-12")).thenReturn(Timestamp.valueOf("2010-12-12 00:00:00.0"));
165         when(timestampUtils.toTimestamp(null, "2013-12-11 pst")).thenReturn(Timestamp.valueOf("2013-12-11 16:00:00.0"));
166         when(timestampUtils.toTimestamp(null, "2003-04-12 04:05:06")).thenReturn(Timestamp.valueOf("2003-04-12 04:05:00.0"));
167         WriteRowEvent actual = (WriteRowEvent) new MppdbDecodingPlugin(new OpenGaussTimestampUtils(timestampUtils)).decode(data, logSequenceNumber);
168         assertThat(actual.getLogSequenceNumber(), is(logSequenceNumber));
169         assertThat(actual.getTableName(), is("test"));
170         IntStream.range(0, insertTypes.length).forEach(each -> assertThat(actual.getAfterRow().get(each).toString(), is(compareValues[each])));
171     }

239     @Test
240     void assertDecodeWithXid() throws JsonProcessingException {
241         MppTableData tableData = new MppTableData();
242         tableData.setTableName("public.test");
243         tableData.setOpType("INSERT");
244         tableData.setColumnsName(new String[]{"data"});
245         tableData.setColumnsType(new String[]{"raw"});
246         tableData.setColumnsVal(new String[]{"'7D'"});
247         List<String> dataList = Arrays.asList("BEGIN 1", OBJECT_MAPPER.writeValueAsString(tableData), OBJECT_MAPPER.writeValueAsString(tableData),
248                 "COMMIT 1 (at 2022-10-27 04:19:39.476261+00) CSN 3468");
249         MppdbDecodingPlugin mppdbDecodingPlugin = new MppdbDecodingPlugin(null, true);
250         List<AbstractWALEvent> expectedEvent = new LinkedList<>();
251         for (String each : dataList) {
252             expectedEvent.add(mppdbDecodingPlugin.decode(ByteBuffer.wrap(each.getBytes()), logSequenceNumber));
253         }
254         assertThat(expectedEvent.size(), is(4));
255         AbstractWALEvent actualFirstEvent = expectedEvent.get(0);
256         assertTrue(actualFirstEvent instanceof BeginTXEvent);
257         assertThat(((BeginTXEvent) actualFirstEvent).getXid(), is(1L));
258         AbstractWALEvent actualLastEvent = expectedEvent.get(expectedEvent.size() - 1);
259         assertTrue(actualLastEvent instanceof CommitTXEvent);
260         assertThat(((CommitTXEvent) actualLastEvent).getCsn(), is(3468L));
261         assertThat(((CommitTXEvent) actualLastEvent).getXid(), is(1L));
262     }
