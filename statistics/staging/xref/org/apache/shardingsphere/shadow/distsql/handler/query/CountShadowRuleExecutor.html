18  package org.apache.shardingsphere.shadow.distsql.handler.query;

20  import org.apache.shardingsphere.distsql.handler.query.RQLExecutor;
21  import org.apache.shardingsphere.infra.merge.result.impl.local.LocalDataQueryResultRow;
22  import org.apache.shardingsphere.infra.metadata.database.ShardingSphereDatabase;
23  import org.apache.shardingsphere.shadow.distsql.parser.statement.CountShadowRuleStatement;
24  import org.apache.shardingsphere.shadow.rule.ShadowRule;
25  
26  import java.util.Arrays;
27  import java.util.Collection;
28  import java.util.Iterator;
29  import java.util.LinkedHashMap;
30  import java.util.LinkedList;
31  import java.util.Map;
32  import java.util.Map.Entry;
33  import java.util.Optional;
34  import java.util.function.Supplier;
35  
36  /**
37   * Count shadow rule executor.
38   */
39  public final class CountShadowRuleExecutor implements RQLExecutor<CountShadowRuleStatement> {
40      
41      private static final String SHADOW = "shadow";
42      
43      @Override
44      public Collection<String> getColumnNames() {
45          return Arrays.asList("rule_name", "database", "count");
46      }
47      
48      @Override
49      public Collection<LocalDataQueryResultRow> getRows(final ShardingSphereDatabase database, final CountShadowRuleStatement sqlStatement) {
50          Optional<ShadowRule> rule = database.getRuleMetaData().findSingleRule(ShadowRule.class);
51          Map<String, LinkedList<Object>> rowMap = new LinkedHashMap<>();
52          rule.ifPresent(optional -> addShadowData(rowMap, database.getName(), rule.get()));
53          Iterator<Entry<String, LinkedList<Object>>> data = rowMap.entrySet().iterator();
54          Collection<LocalDataQueryResultRow> result = new LinkedList<>();
55          while (data.hasNext()) {
56              Entry<String, LinkedList<Object>> entry = data.next();
57              entry.getValue().addFirst(entry.getKey());
58              result.add(new LocalDataQueryResultRow(entry.getValue()));
59          }
60          return result;
61      }
62      
63      private void addShadowData(final Map<String, LinkedList<Object>> rowMap, final String databaseName, final ShadowRule rule) {
64          addData(rowMap, SHADOW, databaseName, () -> rule.getDataSourceMapper().size());
65      }
66      
67      private void addData(final Map<String, LinkedList<Object>> rowMap, final String dataKey, final String databaseName, final Supplier<Integer> apply) {
68          rowMap.compute(dataKey, (key, value) -> buildRow(value, databaseName, apply.get()));
69      }
70      
71      private LinkedList<Object> buildRow(final LinkedList<Object> value, final String databaseName, final int count) {
72          if (null == value) {
73              return new LinkedList<>(Arrays.asList(databaseName, count));
74          }
75          value.set(1, (Integer) value.get(1) + count);
76          return value;
77      }
78      
79      @Override
80      public String getType() {
81          return CountShadowRuleStatement.class.getName();
82      }
83  }
