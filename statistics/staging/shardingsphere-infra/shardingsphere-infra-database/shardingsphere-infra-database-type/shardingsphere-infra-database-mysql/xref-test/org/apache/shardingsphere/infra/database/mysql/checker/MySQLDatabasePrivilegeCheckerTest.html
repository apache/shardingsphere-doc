18  package org.apache.shardingsphere.infra.database.mysql.checker;
19  
20  import org.apache.shardingsphere.infra.database.core.checker.PrivilegeCheckType;
21  import org.apache.shardingsphere.infra.database.core.exception.CheckDatabaseEnvironmentFailedException;
22  import org.apache.shardingsphere.infra.database.core.exception.MissingRequiredPrivilegeException;
23  import org.junit.jupiter.api.BeforeEach;
24  import org.junit.jupiter.api.Test;
25  import org.junit.jupiter.api.extension.ExtendWith;
26  import org.mockito.Answers;
27  import org.mockito.Mock;
28  import org.mockito.junit.jupiter.MockitoExtension;
29  
30  import javax.sql.DataSource;
31  import java.sql.PreparedStatement;
32  import java.sql.ResultSet;
33  import java.sql.SQLException;
34  
35  import static org.junit.jupiter.api.Assertions.assertThrows;
36  import static org.mockito.ArgumentMatchers.anyString;
37  import static org.mockito.Mockito.times;
38  import static org.mockito.Mockito.verify;
39  import static org.mockito.Mockito.when;
40  
41  @ExtendWith(MockitoExtension.class)
42  class MySQLDatabasePrivilegeCheckerTest {
43      
44      @Mock
45      private PreparedStatement preparedStatement;
46      
47      @Mock
48      private ResultSet resultSet;
49      
50      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
51      private DataSource dataSource;
52      
53      @BeforeEach
54      void setUp() throws SQLException {
55          when(dataSource.getConnection().prepareStatement(anyString())).thenReturn(preparedStatement);
56      }
57      
58      @Test
59      void assertCheckPrivilegeWithParticularSuccess() throws SQLException {
60          when(preparedStatement.executeQuery()).thenReturn(resultSet);
61          when(resultSet.next()).thenReturn(true);
62          when(resultSet.getString(1)).thenReturn("GRANT REPLICATION SLAVE, REPLICATION CLIENT ON *.* TO '%'@'%'");
63          new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.PIPELINE);
64          verify(preparedStatement).executeQuery();
65      }
66      
67      @Test
68      void assertCheckPrivilegeWithAllSuccess() throws SQLException {
69          when(preparedStatement.executeQuery()).thenReturn(resultSet);
70          when(resultSet.next()).thenReturn(true);
71          when(resultSet.getString(1)).thenReturn("GRANT ALL PRIVILEGES CLIENT ON *.* TO '%'@'%'");
72          new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.PIPELINE);
73          verify(preparedStatement).executeQuery();
74      }
75      
76      @Test
77      void assertCheckPrivilegeLackPrivileges() throws SQLException {
78          when(preparedStatement.executeQuery()).thenReturn(resultSet);
79          assertThrows(MissingRequiredPrivilegeException.class, () -> new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.PIPELINE));
80      }
81      
82      @Test
83      void assertCheckPrivilegeFailure() throws SQLException {
84          when(preparedStatement.executeQuery()).thenReturn(resultSet);
85          when(resultSet.next()).thenThrow(new SQLException(""));
86          assertThrows(CheckDatabaseEnvironmentFailedException.class, () -> new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.PIPELINE));
87      }
88      
89      @Test
90      void assertCheckXAPrivilegeWithParticularSuccessInMySQL8() throws SQLException {
91          when(preparedStatement.executeQuery()).thenReturn(resultSet);
92          when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
93          when(resultSet.next()).thenReturn(true);
94          when(resultSet.getString(1)).thenReturn("GRANT XA_RECOVER_ADMIN ON *.* TO '%'@'%'");
95          new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.XA);
96          verify(preparedStatement).executeQuery();
97      }
98      
99      @Test
100     void assertUnCheckXAPrivilegeInMySQL5() throws SQLException {
101         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(5);
102         new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.XA);
103         verify(preparedStatement, times(0)).executeQuery();
104     }
105     
106     @Test
107     void assertCheckXAPrivilegeWithAllSuccessInMySQL8() throws SQLException {
108         when(preparedStatement.executeQuery()).thenReturn(resultSet);
109         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
110         when(resultSet.next()).thenReturn(true);
111         when(resultSet.getString(1)).thenReturn("GRANT ALL PRIVILEGES ON *.* TO '%'@'%'");
112         new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.XA);
113         verify(preparedStatement).executeQuery();
114     }
115     
116     @Test
117     void assertCheckXAPrivilegeLackPrivilegesInMySQL8() throws SQLException {
118         when(preparedStatement.executeQuery()).thenReturn(resultSet);
119         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
120         assertThrows(MissingRequiredPrivilegeException.class, () -> new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.XA));
121     }
122     
123     @Test
124     void assertCheckXAPrivilegeFailureInMySQL8() throws SQLException {
125         when(preparedStatement.executeQuery()).thenReturn(resultSet);
126         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
127         when(resultSet.next()).thenThrow(new SQLException(""));
128         assertThrows(CheckDatabaseEnvironmentFailedException.class, () -> new MySQLDatabasePrivilegeChecker().check(dataSource, PrivilegeCheckType.XA));
129     }
130 }
