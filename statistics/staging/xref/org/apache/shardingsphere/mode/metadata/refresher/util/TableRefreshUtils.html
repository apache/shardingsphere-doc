42  /**
43   * Table refresh utils.
44   */
45  @NoArgsConstructor(access = AccessLevel.PRIVATE)
46  public final class TableRefreshUtils {
47      
48      /**
49       * Get table name.
50       *
51       * @param databaseType database type
52       * @param identifierValue identifier value
53       * @return table name
54       */
55      public static String getTableName(final DatabaseType databaseType, final IdentifierValue identifierValue) {
56          return databaseType instanceof OracleDatabaseType && QuoteCharacter.NONE == identifierValue.getQuoteCharacter() ? identifierValue.getValue().toUpperCase() : identifierValue.getValue();
57      }
58      
59      /**
60       * Judge whether single table.
61       *
62       * @param tableName table name
63       * @param database database
64       * @return whether single table
65       */
66      public static boolean isSingleTable(final String tableName, final ShardingSphereDatabase database) {
67          return database.getRuleMetaData().getAttributes(TableMapperRuleAttribute.class).stream().noneMatch(each -> each.getDistributedTableNames().contains(tableName));
68      }
69      
70      /**
71       * Judge whether the rule need to be refreshed.
72       *
73       * @param ruleMetaData rule meta data
74       * @param schemaName schema name
75       * @param tableSegments table segments
76       * @return whether the rule need to be refreshed
77       */
78      public static boolean isRuleRefreshRequired(final RuleMetaData ruleMetaData, final String schemaName, final Collection<SimpleTableSegment> tableSegments) {
79          for (SimpleTableSegment each : tableSegments) {
80              if (isRuleRefreshRequired(ruleMetaData, schemaName, each.getTableName().getIdentifier().getValue())) {
81                  return true;
82              }
83          }
84          return false;
85      }
86      
87      /**
88       * Judge whether the rule need to be refreshed.
89       * 
90       * @param ruleMetaData rule meta data
91       * @param schemaName schema name
92       * @param tableName table name
93       * @return whether the rule need to be refreshed
94       */
95      public static boolean isRuleRefreshRequired(final RuleMetaData ruleMetaData, final String schemaName, final String tableName) {
96          Collection<ShardingSphereRule> rules = new LinkedList<>();
97          for (ShardingSphereRule each : ruleMetaData.getRules()) {
98              each.getAttributes().findAttribute(MutableDataNodeRuleAttribute.class).ifPresent(optional -> rules.add(each));
99          }
100         if (rules.isEmpty()) {
101             return false;
102         }
103         ShardingSphereRule rule = rules.iterator().next();
104         RuleConfiguration ruleConfig = rule.getConfiguration();
105         if (!(ruleConfig instanceof SingleRuleConfiguration)) {
106             return false;
107         }
108         Collection<String> tablesConfig = ((SingleRuleConfiguration) ruleConfig).getTables();
109         if (tablesConfig.contains(SingleTableConstants.ALL_TABLES) || tablesConfig.contains(SingleTableConstants.ALL_SCHEMA_TABLES)) {
110             return false;
111         }
112         Optional<DataNode> dataNode = rule.getAttributes().getAttribute(MutableDataNodeRuleAttribute.class).findTableDataNode(schemaName, tableName);
113         if (!dataNode.isPresent()) {
114             return false;
115         }
116         DataNode actualNode = dataNode.get();
117         return !tablesConfig.contains(joinDataNodeSegments(actualNode.getDataSourceName(), SingleTableConstants.ASTERISK))
118                 && !tablesConfig.contains(joinDataNodeSegments(actualNode.getDataSourceName(), SingleTableConstants.ASTERISK, SingleTableConstants.ASTERISK))
119                 && !tablesConfig.contains(joinDataNodeSegments(actualNode.getDataSourceName(), actualNode.getSchemaName(), SingleTableConstants.ASTERISK));
120     }
121     
122     private static String joinDataNodeSegments(final String... segments) {
123         return Joiner.on(".").join(segments);
124     }
125 }
