41  /**
42   * Show unused sharding key generator executor.
43   */
44  public final class ShowUnusedShardingKeyGeneratorExecutor implements RQLExecutor<ShowUnusedShardingKeyGeneratorsStatement> {
45      
46      @Override
47      public Collection<LocalDataQueryResultRow> getRows(final ShardingSphereDatabase database, final ShowUnusedShardingKeyGeneratorsStatement sqlStatement) {
48          Optional<ShardingRule> rule = database.getRuleMetaData().findSingleRule(ShardingRule.class);
49          if (!rule.isPresent()) {
50              return Collections.emptyList();
51          }
52          ShardingRuleConfiguration shardingRuleConfig = (ShardingRuleConfiguration) rule.get().getConfiguration();
53          Collection<LocalDataQueryResultRow> result = new LinkedList<>();
54          Collection<String> inUsedKeyGenerators = getUsedKeyGenerators(shardingRuleConfig);
55          for (Entry<String, AlgorithmConfiguration> entry : shardingRuleConfig.getKeyGenerators().entrySet()) {
56              if (!inUsedKeyGenerators.contains(entry.getKey())) {
57                  result.add(new LocalDataQueryResultRow(entry.getKey(), entry.getValue().getType(), buildProps(entry.getValue().getProps())));
58              }
59          }
60          return result;
61      }
62      
63      private Collection<String> getUsedKeyGenerators(final ShardingRuleConfiguration shardingRuleConfig) {
64          Collection<String> result = new LinkedHashSet<>();
65          shardingRuleConfig.getTables().stream().filter(each -> Objects.nonNull(each.getKeyGenerateStrategy())).forEach(each -> result.add(each.getKeyGenerateStrategy().getKeyGeneratorName()));
66          shardingRuleConfig.getAutoTables().stream().filter(each -> Objects.nonNull(each.getKeyGenerateStrategy())).forEach(each -> result.add(each.getKeyGenerateStrategy().getKeyGeneratorName()));
67          KeyGenerateStrategyConfiguration keyGenerateStrategy = shardingRuleConfig.getDefaultKeyGenerateStrategy();
68          if (Objects.nonNull(keyGenerateStrategy) && !Strings.isNullOrEmpty(keyGenerateStrategy.getKeyGeneratorName())) {
69              result.add(keyGenerateStrategy.getKeyGeneratorName());
70          }
71          return result;
72      }
73      
74      @Override
75      public Collection<String> getColumnNames() {
76          return Arrays.asList("name", "type", "props");
77      }
78      
79      private Object buildProps(final Properties props) {
80          return Objects.nonNull(props) ? PropertiesConverter.convert(props) : "";
81      }
82      
83      @Override
84      public String getType() {
85          return ShowUnusedShardingKeyGeneratorsStatement.class.getName();
86      }
87  }
