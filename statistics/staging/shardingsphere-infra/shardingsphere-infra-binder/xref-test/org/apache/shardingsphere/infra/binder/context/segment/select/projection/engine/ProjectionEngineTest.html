182         when(schema.getVisibleColumnNames("t_order")).thenReturn(Arrays.asList("order_id", "customer_id"));
183         when(databaseType.getQuoteCharacter()).thenReturn(QuoteCharacter.NONE);
184         SimpleTableSegment customersTableSegment = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_customer")));
185         when(schema.getVisibleColumnNames("t_customer")).thenReturn(Collections.singletonList("customer_id"));
186         JoinTableSegment table = new JoinTableSegment();
187         SimpleTableSegment ordersTableSegment = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")));
188         table.setLeft(ordersTableSegment);
189         table.setRight(customersTableSegment);
190         table.setCondition(new CommonExpressionSegment(0, 0, "t_order.customer_id=t_customer.customer_id"));
191         ShorthandProjectionSegment shorthandProjectionSegment = new ShorthandProjectionSegment(0, 10);
192         Optional<Projection> actual = new ProjectionEngine(
193                 DefaultDatabase.LOGIC_NAME, Collections.singletonMap(DefaultDatabase.LOGIC_NAME, schema), databaseType).createProjection(table, shorthandProjectionSegment);
194         assertTrue(actual.isPresent());
195         assertThat(actual.get(), instanceOf(ShorthandProjection.class));
196         Collection<ColumnProjection> columnProjections = ((ShorthandProjection) actual.get()).getColumnProjections();
197         assertThat(columnProjections.size(), is(3));
198         Iterator<ColumnProjection> iterator = columnProjections.iterator();
199         assertThat(iterator.next(), is(new ColumnProjection("t_order", "order_id", null, databaseType)));
200         assertThat(iterator.next(), is(new ColumnProjection("t_order", "customer_id", null, databaseType)));
201         assertThat(iterator.next(), is(new ColumnProjection("t_customer", "customer_id", null, databaseType)));
202     }

268         SimpleTableSegment left = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")));
269         left.setAlias(new AliasSegment(0, 0, new IdentifierValue("o")));
270         SimpleTableSegment right = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order_item")));
271         right.setAlias(new AliasSegment(0, 0, new IdentifierValue("i")));
272         JoinTableSegment result = new JoinTableSegment();
273         result.setLeft(left);
274         result.setRight(right);
275         result.setJoinType(JoinType.RIGHT.name());
276         result.setUsing(Arrays.asList(new ColumnSegment(0, 0, new IdentifierValue("user_id")), new ColumnSegment(0, 0, new IdentifierValue("order_id")),
277                 new ColumnSegment(0, 0, new IdentifierValue("creation_date"))));
278         return result;
279     }
