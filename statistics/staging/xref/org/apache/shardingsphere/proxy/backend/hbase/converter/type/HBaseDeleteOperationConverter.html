18  package org.apache.shardingsphere.proxy.backend.hbase.converter.type;
19  
20  import com.google.common.base.Preconditions;
21  import lombok.RequiredArgsConstructor;
22  import org.apache.hadoop.hbase.client.Delete;
23  import org.apache.hadoop.hbase.util.Bytes;
24  import org.apache.shardingsphere.infra.binder.context.statement.SQLStatementContext;
25  import org.apache.shardingsphere.infra.binder.context.statement.dml.DeleteStatementContext;
26  import org.apache.shardingsphere.proxy.backend.hbase.bean.HBaseOperation;
27  import org.apache.shardingsphere.proxy.backend.hbase.converter.HBaseOperationConverter;
28  import org.apache.shardingsphere.proxy.backend.hbase.converter.HBaseRowKeyExtractor;
29  import org.apache.shardingsphere.proxy.backend.hbase.converter.operation.HBaseDeleteOperation;
30  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.BinaryOperationExpression;
31  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.InExpression;
32  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.predicate.WhereSegment;
33  
34  import java.util.List;
35  import java.util.Optional;
36  import java.util.stream.Collectors;
37  
38  /**
39   * HBase delete operation converter.
40   */
41  @RequiredArgsConstructor
42  public final class HBaseDeleteOperationConverter implements HBaseOperationConverter {
43      
44      private final SQLStatementContext sqlStatementContext;
45      
46      @Override
47      public HBaseOperation convert() {
48          DeleteStatementContext deleteStatementContext = (DeleteStatementContext) sqlStatementContext;
49          Optional<WhereSegment> whereSegment = deleteStatementContext.getWhereSegments().stream().findFirst();
50          Preconditions.checkArgument(whereSegment.isPresent(), "Where segment is absent.");
51          return whereSegment.get().getExpr() instanceof InExpression
52                  ? createDeleteMultipleRowKeysOperation(deleteStatementContext, whereSegment.get())
53                  : createDeleteSingleRowKeyOperation(deleteStatementContext, whereSegment.get());
54      }
55      
56      private HBaseOperation createDeleteMultipleRowKeysOperation(final DeleteStatementContext deleteStatementContext, final WhereSegment whereSegment) {
57          String tableName = deleteStatementContext.getTablesContext().getTableNames().iterator().next();
58          List<String> rowKeys = HBaseRowKeyExtractor.getRowKeys((InExpression) whereSegment.getExpr());
59          List<Delete> deletes = rowKeys.stream().map(each -> new Delete(Bytes.toBytes(each))).collect(Collectors.toList());
60          return new HBaseOperation(tableName, new HBaseDeleteOperation(deletes));
61      }
62      
63      private HBaseOperation createDeleteSingleRowKeyOperation(final DeleteStatementContext deleteStatementContext, final WhereSegment whereSegment) {
64          String tableName = deleteStatementContext.getTablesContext().getTableNames().iterator().next();
65          String rowKey = HBaseRowKeyExtractor.getRowKey((BinaryOperationExpression) whereSegment.getExpr());
66          Delete delete = new Delete(Bytes.toBytes(rowKey));
67          return new HBaseOperation(tableName, delete);
68      }
69  }
