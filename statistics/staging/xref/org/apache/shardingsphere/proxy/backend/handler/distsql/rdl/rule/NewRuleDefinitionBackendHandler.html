51  /**
52   * TODO Rename to RuleDefinitionBackendHandler when metadata structure adjustment completed. #25485
53   * Rule definition backend handler.
54   *
55   * @param <T> type of rule definition statement
56   */
57  public final class NewRuleDefinitionBackendHandler<T extends RuleDefinitionStatement> extends RDLBackendHandler<T> {
58      
59      public NewRuleDefinitionBackendHandler(final T sqlStatement, final ConnectionSession connectionSession) {
60          super(sqlStatement, connectionSession);
61      }
62      
63      @SuppressWarnings({"rawtypes", "unchecked"})
64      @Override
65      protected ResponseHeader execute(final String databaseName, final T sqlStatement) {
66          RuleDefinitionUpdater ruleDefinitionUpdater = TypedSPILoader.getService(RuleDefinitionUpdater.class, sqlStatement.getClass().getName());
67          Class<? extends RuleConfiguration> ruleConfigClass = ruleDefinitionUpdater.getRuleConfigurationClass();
68          ShardingSphereDatabase database = ProxyContext.getInstance().getDatabase(databaseName);
69          RuleConfiguration currentRuleConfig = findCurrentRuleConfiguration(database, ruleConfigClass).orElse(null);
70          ruleDefinitionUpdater.checkSQLStatement(database, sqlStatement, currentRuleConfig);
71          if (getRefreshStatus(sqlStatement, currentRuleConfig, ruleDefinitionUpdater)) {
72              ProxyContext.getInstance().getContextManager().getMetaDataContexts().getPersistService().getMetaDataVersionPersistService().switchActiveVersion(
73                      processSQLStatement(database, sqlStatement, ruleDefinitionUpdater, currentRuleConfig));
74          }
75          return new UpdateResponseHeader(sqlStatement);
76      }
77      
78      private Optional<RuleConfiguration> findCurrentRuleConfiguration(final ShardingSphereDatabase database, final Class<? extends RuleConfiguration> ruleConfigClass) {
79          for (RuleConfiguration each : database.getRuleMetaData().getConfigurations()) {
80              if (ruleConfigClass.isAssignableFrom(each.getClass())) {
81                  return Optional.of(each);
82              }
83          }
84          return Optional.empty();
85      }
86      
87      @SuppressWarnings("rawtypes")
88      private Collection<MetaDataVersion> processSQLStatement(final ShardingSphereDatabase database,
89                                                              final T sqlStatement, final RuleDefinitionUpdater updater, final RuleConfiguration currentRuleConfig) {
90          if (updater instanceof RuleDefinitionCreateUpdater) {
91              return processCreate(database, sqlStatement, (RuleDefinitionCreateUpdater) updater, currentRuleConfig);
92          }
93          if (updater instanceof RuleDefinitionAlterUpdater) {
94              return processAlter(database, sqlStatement, (RuleDefinitionAlterUpdater) updater, currentRuleConfig);
95          }
96          if (updater instanceof RuleDefinitionDropUpdater) {
97              return processDrop(database, sqlStatement, (RuleDefinitionDropUpdater) updater, currentRuleConfig);
98          }
99          throw new UnsupportedSQLOperationException(String.format("Cannot support RDL updater type `%s`", updater.getClass().getName()));
100     }
101     
102     @SuppressWarnings({"rawtypes", "unchecked"})
103     private Collection<MetaDataVersion> processCreate(final ShardingSphereDatabase database, final T sqlStatement, final RuleDefinitionCreateUpdater updater,
104                                                       final RuleConfiguration currentRuleConfig) {
105         RuleConfiguration toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, sqlStatement);
106         if (null != currentRuleConfig) {
107             updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
108         }
109         if (sqlStatement instanceof LoadSingleTableStatement || sqlStatement instanceof SetDefaultSingleTableStorageUnitStatement) {
110             return ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().alterRuleConfiguration(database.getName(),
111                     null == currentRuleConfig ? decorateRuleConfiguration(database, toBeCreatedRuleConfig) : decorateRuleConfiguration(database, currentRuleConfig));
112         }
113         return ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().alterRuleConfiguration(database.getName(), toBeCreatedRuleConfig);
114     }
115     
116     @SuppressWarnings({"rawtypes", "unchecked"})
117     private Collection<MetaDataVersion> processAlter(final ShardingSphereDatabase database, final T sqlStatement, final RuleDefinitionAlterUpdater updater, final RuleConfiguration currentRuleConfig) {
118         RuleConfiguration toBeAlteredRuleConfig = updater.buildToBeAlteredRuleConfiguration(sqlStatement);
119         updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeAlteredRuleConfig);
120         if (sqlStatement instanceof UnloadSingleTableStatement) {
121             return ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager()
122                     .alterRuleConfiguration(database.getName(), decorateRuleConfiguration(database, currentRuleConfig));
123         }
124         RuleConfiguration toBeDroppedRuleConfig = updater.buildToBeDroppedRuleConfiguration(currentRuleConfig, toBeAlteredRuleConfig);
125         ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().removeRuleConfigurationItem(database.getName(), toBeDroppedRuleConfig);
126         return ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().alterRuleConfiguration(database.getName(), toBeAlteredRuleConfig);
127     }
128     
129     @SuppressWarnings("unchecked")
130     private RuleConfiguration decorateRuleConfiguration(final ShardingSphereDatabase database, final RuleConfiguration ruleConfig) {
131         if (TypedSPILoader.contains(RuleConfigurationDecorator.class, ruleConfig.getClass().getName())) {
132             return TypedSPILoader.getService(RuleConfigurationDecorator.class, ruleConfig.getClass().getName()).decorate(database.getName(),
133                     database.getResourceMetaData().getDataSources(), database.getRuleMetaData().getRules(), ruleConfig);
134         }
135         return ruleConfig;
136     }
137     
138     @SuppressWarnings({"rawtypes", "unchecked"})
139     private Collection<MetaDataVersion> processDrop(final ShardingSphereDatabase database, final T sqlStatement, final RuleDefinitionDropUpdater updater, final RuleConfiguration currentRuleConfig) {
140         if (!updater.hasAnyOneToBeDropped(sqlStatement, currentRuleConfig)) {
141             return Collections.emptyList();
142         }
143         ModeContextManager modeContextManager = ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager();
144         final RuleConfiguration toBeDroppedRuleConfig = updater.buildToBeDroppedRuleConfiguration(currentRuleConfig, sqlStatement);
145         // TODO remove updateCurrentRuleConfiguration after update refactor completed.
146         if (updater.updateCurrentRuleConfiguration(sqlStatement, currentRuleConfig) && ((DatabaseRuleConfiguration) currentRuleConfig).isEmpty()) {
147             modeContextManager.removeRuleConfigurationItem(database.getName(), toBeDroppedRuleConfig);
148             new NewYamlRuleConfigurationSwapperEngine().swapToYamlRuleConfigurations(Collections.singleton(currentRuleConfig)).values().stream().findFirst()
149                     .ifPresent(swapper -> modeContextManager.removeRuleConfiguration(database.getName(), swapper.getRuleTagName().toLowerCase()));
150             return Collections.emptyList();
151         }
152         if (updater instanceof DropReadwriteSplittingRuleStatementUpdater) {
153             database.getRuleMetaData().findSingleRule(StaticDataSourceContainedRule.class)
154                     .ifPresent(optional -> ((DropReadwriteSplittingRuleStatement) sqlStatement).getNames().forEach(optional::cleanStorageNodeDataSource));
155             // TODO refactor to new metadata refresh way
156         }
157         modeContextManager.removeRuleConfigurationItem(database.getName(), toBeDroppedRuleConfig);
158         final RuleConfiguration toBeAlteredRuleConfig = updater.buildToBeAlteredRuleConfiguration(currentRuleConfig, sqlStatement);
159         return modeContextManager.alterRuleConfiguration(database.getName(), toBeAlteredRuleConfig);
160     }
161     
162     @SuppressWarnings({"rawtypes", "unchecked"})
163     private boolean getRefreshStatus(final SQLStatement sqlStatement, final RuleConfiguration currentRuleConfig, final RuleDefinitionUpdater<?, ?> updater) {
164         return !(updater instanceof RuleDefinitionDropUpdater) || ((RuleDefinitionDropUpdater) updater).hasAnyOneToBeDropped(sqlStatement, currentRuleConfig);
165     }
