45  /**
46   * Standard pipeline table meta data loader.
47   */
48  @RequiredArgsConstructor
49  @Slf4j
50  public final class StandardPipelineTableMetaDataLoader implements PipelineTableMetaDataLoader {
51      
52      // It doesn't support ShardingSphereDataSource
53      private final PipelineDataSourceWrapper dataSource;
54      
55      private final Map<CaseInsensitiveIdentifier, PipelineTableMetaData> tableMetaDataMap = new ConcurrentHashMap<>();
56      
57      @Override
58      public PipelineTableMetaData getTableMetaData(final String schemaName, final String tableName) {
59          PipelineTableMetaData result = tableMetaDataMap.get(new CaseInsensitiveIdentifier(tableName));
60          if (null != result) {
61              return result;
62          }
63          try {
64              loadTableMetaData(schemaName, tableName);
65          } catch (final SQLException ex) {
66              throw new PipelineInternalException(String.format("Load meta data for schema '%s' and table '%s' failed", schemaName, tableName), ex);
67          }
68          result = tableMetaDataMap.get(new CaseInsensitiveIdentifier(tableName));
69          if (null == result) {
70              log.warn("Can not load meta data for table '{}'", tableName);
71          }
72          return result;
73      }
74      
75      private void loadTableMetaData(final String schemaName, final String tableNamePattern) throws SQLException {
76          try (Connection connection = dataSource.getConnection()) {
77              DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(dataSource.getDatabaseType()).getDialectDatabaseMetaData();
78              Map<CaseInsensitiveIdentifier, PipelineTableMetaData> tableMetaDataMap = loadTableMetaData0(connection, dialectDatabaseMetaData.isSchemaAvailable() ? schemaName : null, tableNamePattern);
79              this.tableMetaDataMap.putAll(tableMetaDataMap);
80          }
81      }
82      
83      private Map<CaseInsensitiveIdentifier, PipelineTableMetaData> loadTableMetaData0(final Connection connection, final String schemaName, final String tableNamePattern) throws SQLException {
84          Collection<String> tableNames = new LinkedList<>();
85          try (ResultSet resultSet = connection.getMetaData().getTables(connection.getCatalog(), schemaName, tableNamePattern, null)) {
86              while (resultSet.next()) {
87                  String tableName = resultSet.getString("TABLE_NAME");
88                  tableNames.add(tableName);
89              }
90          }
91          Map<CaseInsensitiveIdentifier, PipelineTableMetaData> result = new LinkedHashMap<>(tableNames.size(), 1F);
92          for (String each : tableNames) {
93              Collection<CaseInsensitiveIdentifier> primaryKeys = loadPrimaryKeys(connection, schemaName, each);
94              Map<CaseInsensitiveIdentifier, Collection<CaseInsensitiveIdentifier>> uniqueKeys = loadUniqueIndexesOfTable(connection, schemaName, each);
95              Map<CaseInsensitiveIdentifier, PipelineColumnMetaData> columnMetaDataMap = new LinkedHashMap<>();
96              try (ResultSet resultSet = connection.getMetaData().getColumns(connection.getCatalog(), schemaName, each, "%")) {
97                  while (resultSet.next()) {
98                      int ordinalPosition = resultSet.getInt("ORDINAL_POSITION");
99                      CaseInsensitiveIdentifier columnName = new CaseInsensitiveIdentifier(resultSet.getString("COLUMN_NAME"));
100                     if (columnMetaDataMap.containsKey(columnName)) {
101                         continue;
102                     }
103                     int dataType = resultSet.getInt("DATA_TYPE");
104                     String dataTypeName = resultSet.getString("TYPE_NAME");
105                     boolean primaryKey = primaryKeys.contains(columnName);
106                     boolean isNullable = "YES".equals(resultSet.getString("IS_NULLABLE"));
107                     boolean isUniqueKey = uniqueKeys.values().stream().anyMatch(names -> names.contains(columnName));
108                     PipelineColumnMetaData columnMetaData = new PipelineColumnMetaData(ordinalPosition, columnName.toString(), dataType, dataTypeName, isNullable, primaryKey, isUniqueKey);
109                     columnMetaDataMap.put(columnName, columnMetaData);
110                 }
111             }
112             Collection<PipelineIndexMetaData> uniqueIndexMetaData = uniqueKeys.entrySet().stream()
113                     .map(entry -> new PipelineIndexMetaData(entry.getKey(), entry.getValue().stream().map(columnMetaDataMap::get).collect(Collectors.toList()),
114                             DataConsistencyCheckUtils.compareLists(primaryKeys, entry.getValue())))
115                     .collect(Collectors.toList());
116             result.put(new CaseInsensitiveIdentifier(each), new PipelineTableMetaData(each, columnMetaDataMap, uniqueIndexMetaData));
117         }
118         return result;
119     }
120     
121     private Map<CaseInsensitiveIdentifier, Collection<CaseInsensitiveIdentifier>> loadUniqueIndexesOfTable(final Connection connection,
122                                                                                                            final String schemaName, final String tableName) throws SQLException {
123         Map<String, SortedMap<Short, CaseInsensitiveIdentifier>> orderedColumnsOfIndexes = new LinkedHashMap<>();
124         try (ResultSet resultSet = connection.getMetaData().getIndexInfo(connection.getCatalog(), schemaName, tableName, true, false)) {
125             while (resultSet.next()) {
126                 String indexName = resultSet.getString("INDEX_NAME");
127                 if (null == indexName) {
128                     continue;
129                 }
130                 orderedColumnsOfIndexes.computeIfAbsent(indexName, unused -> new TreeMap<>()).put(
131                         resultSet.getShort("ORDINAL_POSITION"), new CaseInsensitiveIdentifier(resultSet.getString("COLUMN_NAME")));
132             }
133         }
134         Map<CaseInsensitiveIdentifier, Collection<CaseInsensitiveIdentifier>> result = new LinkedHashMap<>();
135         for (Entry<String, SortedMap<Short, CaseInsensitiveIdentifier>> entry : orderedColumnsOfIndexes.entrySet()) {
136             Collection<CaseInsensitiveIdentifier> columnNames = result.computeIfAbsent(new CaseInsensitiveIdentifier(entry.getKey()), unused -> new LinkedList<>());
137             columnNames.addAll(entry.getValue().values());
138         }
139         return result;
140     }
141     
142     private Collection<CaseInsensitiveIdentifier> loadPrimaryKeys(final Connection connection, final String schemaName, final String tableName) throws SQLException {
143         SortedMap<Short, CaseInsensitiveIdentifier> result = new TreeMap<>();
144         try (ResultSet resultSet = connection.getMetaData().getPrimaryKeys(connection.getCatalog(), schemaName, tableName)) {
145             while (resultSet.next()) {
146                 result.put(resultSet.getShort("KEY_SEQ"), new CaseInsensitiveIdentifier(resultSet.getString("COLUMN_NAME")));
147             }
148         }
149         return result.values();
150     }
