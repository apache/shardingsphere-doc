92      @Override
93      public Optional<GovernanceEvent> build(final String databaseName, final DataChangedEvent event) {
94          if (!ShardingNodeConverter.isShardingPath(event.getKey()) || Strings.isNullOrEmpty(event.getValue())) {
95              return Optional.empty();
96          }
97          Optional<String> tableName = ShardingNodeConverter.getTableName(event.getKey());
98          if (tableName.isPresent() && !Strings.isNullOrEmpty(event.getValue())) {
99              Optional<String> tableNameVersion = ShardingNodeConverter.getTableNameVersion(event.getKey());
100             if (tableNameVersion.isPresent()) {
101                 return createShardingTableConfigEvent(databaseName, tableName.get(), Integer.parseInt(tableNameVersion.get()), event);
102             }
103         }
104         Optional<String> autoTableName = ShardingNodeConverter.getAutoTableName(event.getKey());
105         if (autoTableName.isPresent() && !Strings.isNullOrEmpty(event.getValue())) {
106             Optional<String> autoTableNameVersion = ShardingNodeConverter.getAutoTableNameVersion(event.getKey());
107             if (autoTableNameVersion.isPresent()) {
108                 return createShardingAutoTableConfigEvent(databaseName, autoTableName.get(), Integer.parseInt(autoTableNameVersion.get()), event);
109             }
110         }
111         Optional<String> bindingTableName = ShardingNodeConverter.getBindingTableName(event.getKey());
112         if (bindingTableName.isPresent() && !Strings.isNullOrEmpty(event.getValue())) {
113             Optional<String> bingingTableNameVersion = ShardingNodeConverter.getBindingTableNameVersion(event.getKey());
114             if (bingingTableNameVersion.isPresent()) {
115                 return createShardingTableReferenceConfigEvent(databaseName, bindingTableName.get(), Integer.parseInt(bingingTableNameVersion.get()), event);
116             }
117         }
118         if (ShardingNodeConverter.isDefaultDatabaseStrategyPath(event.getKey()) && !Strings.isNullOrEmpty(event.getValue())) {
119             Optional<String> defaultDatabaseStrategyVersion = ShardingNodeConverter.getDefaultDatabaseStrategyVersion(event.getKey());
120             if (defaultDatabaseStrategyVersion.isPresent()) {
121                 return createDefaultDatabaseStrategyConfigEvent(databaseName, Integer.parseInt(defaultDatabaseStrategyVersion.get()), event);
122             }
123         }
124         if (ShardingNodeConverter.isDefaultTableStrategyPath(event.getKey()) && !Strings.isNullOrEmpty(event.getValue())) {
125             Optional<String> defaultTableStrategyVersion = ShardingNodeConverter.getDefaultTableStrategyVersion(event.getKey());
126             if (defaultTableStrategyVersion.isPresent()) {
127                 return createDefaultTableStrategyConfigEvent(databaseName, Integer.parseInt(defaultTableStrategyVersion.get()), event);
128             }
129         }
130         if (ShardingNodeConverter.isDefaultKeyGenerateStrategyPath(event.getKey()) && !Strings.isNullOrEmpty(event.getValue())) {
131             Optional<String> defaultKeyGenerateStrategyVersion = ShardingNodeConverter.getDefaultKeyGenerateStrategyVersion(event.getKey());
132             if (defaultKeyGenerateStrategyVersion.isPresent()) {
133                 return createDefaultKeyGenerateStrategyConfigEvent(databaseName, Integer.parseInt(defaultKeyGenerateStrategyVersion.get()), event);
134             }
135         }
136         if (ShardingNodeConverter.isDefaultAuditStrategyPath(event.getKey()) && !Strings.isNullOrEmpty(event.getValue())) {
137             Optional<String> defaultAuditStrategyVersion = ShardingNodeConverter.getDefaultAuditStrategyVersion(event.getKey());
138             if (defaultAuditStrategyVersion.isPresent()) {
139                 return createDefaultShardingAuditorStrategyConfigEvent(databaseName, Integer.parseInt(defaultAuditStrategyVersion.get()), event);
140             }
141         }
142         if (ShardingNodeConverter.isDefaultShardingColumnPath(event.getKey()) && !Strings.isNullOrEmpty(event.getValue())) {
143             Optional<String> defaultShardingColumnVersion = ShardingNodeConverter.getDefaultShardingColumnVersion(event.getKey());
144             if (defaultShardingColumnVersion.isPresent()) {
145                 return createDefaultShardingColumnEvent(databaseName, Integer.parseInt(defaultShardingColumnVersion.get()), event);
146             }
147         }
148         Optional<String> algorithmName = ShardingNodeConverter.getShardingAlgorithmName(event.getKey());
149         if (algorithmName.isPresent() && !Strings.isNullOrEmpty(event.getValue())) {
150             Optional<String> algorithmVersion = ShardingNodeConverter.getShardingAlgorithmVersion(event.getKey());
151             if (algorithmVersion.isPresent()) {
152                 return createShardingAlgorithmEvent(databaseName, algorithmName.get(), Integer.parseInt(algorithmVersion.get()), event);
153             }
154         }
155         Optional<String> keyGeneratorName = ShardingNodeConverter.getKeyGeneratorName(event.getKey());
156         if (keyGeneratorName.isPresent() && !Strings.isNullOrEmpty(event.getValue())) {
157             Optional<String> keyGeneratorVersion = ShardingNodeConverter.getKeyGeneratorVersion(event.getKey());
158             if (keyGeneratorVersion.isPresent()) {
159                 return createKeyGeneratorEvent(databaseName, keyGeneratorName.get(), Integer.parseInt(keyGeneratorVersion.get()), event);
160             }
161         }
162         Optional<String> auditorName = ShardingNodeConverter.getAuditorName(event.getKey());
163         if (auditorName.isPresent() && !Strings.isNullOrEmpty(event.getValue())) {
164             Optional<String> auditorVersion = ShardingNodeConverter.getAuditorVersion(event.getKey());
165             if (auditorVersion.isPresent()) {
166                 return createAuditorEvent(databaseName, auditorName.get(), Integer.parseInt(auditorVersion.get()), event);
167             }
168         }
169         if (ShardingNodeConverter.isShardingCachePath(event.getKey()) && !Strings.isNullOrEmpty(event.getValue())) {
170             Optional<String> shardingCacheVersion = ShardingNodeConverter.getShardingCacheVersion(event.getKey());
171             if (shardingCacheVersion.isPresent()) {
172                 return createShardingCacheEvent(databaseName, Integer.parseInt(shardingCacheVersion.get()), event);
173             }
174         }
175         return Optional.empty();
176     }

178     private Optional<GovernanceEvent> createShardingTableConfigEvent(final String databaseName, final String tableName, final int version, final DataChangedEvent event) {
179         if (Type.ADDED == event.getType()) {
180             return Optional.of(new AddShardingTableConfigurationEvent<>(databaseName, swapShardingTableRuleConfig(event.getValue()), event.getKey(), version));
181         }
182         if (Type.UPDATED == event.getType()) {
183             return Optional.of(new AlterShardingTableConfigurationEvent<>(databaseName, tableName, swapShardingTableRuleConfig(event.getValue()), event.getKey(), version));
184         }
185         return Optional.of(new DeleteShardingTableConfigurationEvent(databaseName, tableName, event.getKey(), version));
186     }

189         return new YamlShardingTableRuleConfigurationSwapper().swapToObject(YamlEngine.unmarshal(yamlContext, YamlTableRuleConfiguration.class));
