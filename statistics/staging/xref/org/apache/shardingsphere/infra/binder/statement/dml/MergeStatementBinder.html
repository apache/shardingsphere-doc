60  /**
61   * Merge statement binder.
62   */
63  public final class MergeStatementBinder implements SQLStatementBinder<MergeStatement> {
64      
65      @Override
66      public MergeStatement bind(final MergeStatement sqlStatement, final ShardingSphereMetaData metaData, final String currentDatabaseName) {
67          return bind(sqlStatement, metaData, currentDatabaseName, Collections.emptyMap());
68      }
69      
70      @SneakyThrows(ReflectiveOperationException.class)
71      private MergeStatement bind(final MergeStatement sqlStatement, final ShardingSphereMetaData metaData, final String currentDatabaseName,
72                                  final Map<String, TableSegmentBinderContext> externalTableBinderContexts) {
73          MergeStatement result = sqlStatement.getClass().getDeclaredConstructor().newInstance();
74          SQLStatementBinderContext statementBinderContext = new SQLStatementBinderContext(metaData, currentDatabaseName, sqlStatement.getDatabaseType(), sqlStatement.getVariableNames());
75          statementBinderContext.getExternalTableBinderContexts().putAll(externalTableBinderContexts);
76          Map<String, TableSegmentBinderContext> targetTableBinderContexts = new CaseInsensitiveMap<>();
77          TableSegment boundedTargetTableSegment = TableSegmentBinder.bind(sqlStatement.getTarget(), statementBinderContext, targetTableBinderContexts, Collections.emptyMap());
78          Map<String, TableSegmentBinderContext> sourceTableBinderContexts = new CaseInsensitiveMap<>();
79          TableSegment boundedSourceTableSegment = TableSegmentBinder.bind(sqlStatement.getSource(), statementBinderContext, sourceTableBinderContexts, Collections.emptyMap());
80          result.setTarget(boundedTargetTableSegment);
81          result.setSource(boundedSourceTableSegment);
82          Map<String, TableSegmentBinderContext> tableBinderContexts = new LinkedHashMap<>();
83          tableBinderContexts.putAll(sourceTableBinderContexts);
84          tableBinderContexts.putAll(targetTableBinderContexts);
85          if (null != sqlStatement.getExpression()) {
86              ExpressionWithParamsSegment expression = new ExpressionWithParamsSegment(sqlStatement.getExpression().getStartIndex(), sqlStatement.getExpression().getStopIndex(),
87                      ExpressionSegmentBinder.bind(sqlStatement.getExpression().getExpr(), SegmentType.JOIN_ON, statementBinderContext, tableBinderContexts, Collections.emptyMap()));
88              expression.getParameterMarkerSegments().addAll(sqlStatement.getExpression().getParameterMarkerSegments());
89              result.setExpression(expression);
90          }
91          sqlStatement.getInsert().ifPresent(
92                  optional -> result.setInsert(bindMergeInsert(optional, (SimpleTableSegment) boundedTargetTableSegment, statementBinderContext, targetTableBinderContexts, sourceTableBinderContexts)));
93          sqlStatement.getUpdate().ifPresent(
94                  optional -> result.setUpdate(bindMergeUpdate(optional, (SimpleTableSegment) boundedTargetTableSegment, statementBinderContext, targetTableBinderContexts, sourceTableBinderContexts)));
95          addParameterMarkerSegments(result);
96          result.getCommentSegments().addAll(sqlStatement.getCommentSegments());
97          return result;
98      }
99      
100     private void addParameterMarkerSegments(final MergeStatement mergeStatement) {
101         // TODO bind parameter marker segments for merge statement
102         mergeStatement.addParameterMarkerSegments(getSourceSubqueryTableProjectionParameterMarkers(mergeStatement.getSource()));
103         mergeStatement.getInsert().ifPresent(optional -> mergeStatement.addParameterMarkerSegments(optional.getParameterMarkerSegments()));
104         mergeStatement.getUpdate().ifPresent(optional -> mergeStatement.addParameterMarkerSegments(optional.getParameterMarkerSegments()));
105     }
106     
107     private Collection<ParameterMarkerSegment> getSourceSubqueryTableProjectionParameterMarkers(final TableSegment tableSegment) {
108         if (!(tableSegment instanceof SubqueryTableSegment)) {
109             return Collections.emptyList();
110         }
111         SubqueryTableSegment subqueryTable = (SubqueryTableSegment) tableSegment;
112         Collection<ParameterMarkerSegment> result = new LinkedList<>();
113         for (ProjectionSegment each : subqueryTable.getSubquery().getSelect().getProjections().getProjections()) {
114             if (each instanceof ParameterMarkerExpressionSegment) {
115                 result.add((ParameterMarkerSegment) each);
116             }
117         }
118         return result;
119     }

121     @SneakyThrows
122     private InsertStatement bindMergeInsert(final InsertStatement sqlStatement, final SimpleTableSegment tableSegment, final SQLStatementBinderContext statementBinderContext,
123                                             final Map<String, TableSegmentBinderContext> targetTableBinderContexts, final Map<String, TableSegmentBinderContext> sourceTableBinderContexts) {
124         SQLStatementBinderContext insertStatementBinderContext = new SQLStatementBinderContext(statementBinderContext.getMetaData(), statementBinderContext.getCurrentDatabaseName(),
125                 statementBinderContext.getDatabaseType(), statementBinderContext.getVariableNames());
126         insertStatementBinderContext.getExternalTableBinderContexts().putAll(statementBinderContext.getExternalTableBinderContexts());
127         insertStatementBinderContext.getExternalTableBinderContexts().putAll(sourceTableBinderContexts);
128         InsertStatement result = sqlStatement.getClass().getDeclaredConstructor().newInstance();
129         result.setTable(tableSegment);
130         sqlStatement.getInsertColumns()
131                 .ifPresent(optional -> result.setInsertColumns(InsertColumnsSegmentBinder.bind(sqlStatement.getInsertColumns().get(), statementBinderContext, targetTableBinderContexts)));
132         sqlStatement.getInsertSelect().ifPresent(result::setInsertSelect);
133         Collection<InsertValuesSegment> insertValues = new LinkedList<>();
134         Map<ParameterMarkerSegment, ColumnSegmentBoundedInfo> parameterMarkerSegmentBoundedInfos = new LinkedHashMap<>();
135         List<ColumnSegment> columnSegments = new ArrayList<>(result.getInsertColumns().map(InsertColumnsSegment::getColumns)
136                 .orElseGet(() -> getVisibleColumns(targetTableBinderContexts.values().iterator().next().getProjectionSegments())));
137         for (InsertValuesSegment each : sqlStatement.getValues()) {
138             List<ExpressionSegment> values = new LinkedList<>();
139             int index = 0;
140             for (ExpressionSegment expression : each.getValues()) {
141                 values.add(ExpressionSegmentBinder.bind(expression, SegmentType.VALUES, insertStatementBinderContext, targetTableBinderContexts, sourceTableBinderContexts));
142                 if (expression instanceof ParameterMarkerSegment) {
143                     parameterMarkerSegmentBoundedInfos.put((ParameterMarkerSegment) expression, columnSegments.get(index).getColumnBoundedInfo());
144                 }
145                 index++;
146             }
147             insertValues.add(new InsertValuesSegment(each.getStartIndex(), each.getStopIndex(), values));
148         }
149         result.getValues().addAll(insertValues);
150         sqlStatement.getOnDuplicateKeyColumns().ifPresent(result::setOnDuplicateKeyColumns);
151         sqlStatement.getSetAssignment().ifPresent(result::setSetAssignment);
152         sqlStatement.getWithSegment().ifPresent(result::setWithSegment);
153         sqlStatement.getOutputSegment().ifPresent(result::setOutputSegment);
154         sqlStatement.getMultiTableInsertType().ifPresent(result::setMultiTableInsertType);
155         sqlStatement.getMultiTableInsertIntoSegment().ifPresent(result::setMultiTableInsertIntoSegment);
156         sqlStatement.getMultiTableConditionalIntoSegment().ifPresent(result::setMultiTableConditionalIntoSegment);
157         sqlStatement.getReturningSegment().ifPresent(result::setReturningSegment);
158         sqlStatement.getWhere().ifPresent(optional -> result.setWhere(WhereSegmentBinder.bind(optional, insertStatementBinderContext, targetTableBinderContexts, sourceTableBinderContexts)));
159         result.addParameterMarkerSegments(ParameterMarkerSegmentBinder.bind(sqlStatement.getParameterMarkerSegments(), parameterMarkerSegmentBoundedInfos));
160         result.getCommentSegments().addAll(sqlStatement.getCommentSegments());
161         return result;
162     }
163     
164     private Collection<ColumnSegment> getVisibleColumns(final Collection<ProjectionSegment> projectionSegments) {
165         Collection<ColumnSegment> result = new LinkedList<>();
166         for (ProjectionSegment each : projectionSegments) {
167             if (each instanceof ColumnProjectionSegment && each.isVisible()) {
168                 result.add(((ColumnProjectionSegment) each).getColumn());
169             }
170         }
171         return result;
172     }
173     
174     @SneakyThrows
175     private UpdateStatement bindMergeUpdate(final UpdateStatement sqlStatement, final SimpleTableSegment tableSegment, final SQLStatementBinderContext statementBinderContext,
176                                             final Map<String, TableSegmentBinderContext> targetTableBinderContexts, final Map<String, TableSegmentBinderContext> sourceTableBinderContexts) {
177         UpdateStatement result = sqlStatement.getClass().getDeclaredConstructor().newInstance();
178         result.setTable(tableSegment);
179         Collection<ColumnAssignmentSegment> assignments = new LinkedList<>();
180         SQLStatementBinderContext updateStatementBinderContext = new SQLStatementBinderContext(statementBinderContext.getMetaData(), statementBinderContext.getCurrentDatabaseName(),
181                 statementBinderContext.getDatabaseType(), statementBinderContext.getVariableNames());
182         updateStatementBinderContext.getExternalTableBinderContexts().putAll(statementBinderContext.getExternalTableBinderContexts());
183         updateStatementBinderContext.getExternalTableBinderContexts().putAll(sourceTableBinderContexts);
184         Map<ParameterMarkerSegment, ColumnSegmentBoundedInfo> parameterMarkerSegmentBoundedInfos = new LinkedHashMap<>(sqlStatement.getSetAssignment().getAssignments().size(), 1F);
185         for (ColumnAssignmentSegment each : sqlStatement.getSetAssignment().getAssignments()) {
186             List<ColumnSegment> columnSegments = new ArrayList<>(each.getColumns().size());
187             each.getColumns().forEach(column -> columnSegments.add(
188                     ColumnSegmentBinder.bind(column, SegmentType.SET_ASSIGNMENT, updateStatementBinderContext, targetTableBinderContexts, Collections.emptyMap())));
189             ExpressionSegment expression = ExpressionSegmentBinder.bind(each.getValue(), SegmentType.SET_ASSIGNMENT, updateStatementBinderContext, targetTableBinderContexts, Collections.emptyMap());
190             ColumnAssignmentSegment columnAssignmentSegment = new ColumnAssignmentSegment(each.getStartIndex(), each.getStopIndex(), columnSegments, expression);
191             assignments.add(columnAssignmentSegment);
192             if (expression instanceof ParameterMarkerSegment) {
193                 parameterMarkerSegmentBoundedInfos.put((ParameterMarkerSegment) expression, columnAssignmentSegment.getColumns().get(0).getColumnBoundedInfo());
194             }
195         }
196         SetAssignmentSegment setAssignmentSegment = new SetAssignmentSegment(sqlStatement.getSetAssignment().getStartIndex(), sqlStatement.getSetAssignment().getStopIndex(), assignments);
197         result.setSetAssignment(setAssignmentSegment);
198         sqlStatement.getWhere().ifPresent(optional -> result.setWhere(WhereSegmentBinder.bind(optional, updateStatementBinderContext, targetTableBinderContexts, Collections.emptyMap())));
199         sqlStatement.getDeleteWhere().ifPresent(optional -> result.setDeleteWhere(WhereSegmentBinder.bind(optional, updateStatementBinderContext, targetTableBinderContexts, Collections.emptyMap())));
200         sqlStatement.getOrderBy().ifPresent(result::setOrderBy);
201         sqlStatement.getLimit().ifPresent(result::setLimit);
202         sqlStatement.getWithSegment().ifPresent(result::setWithSegment);
203         result.addParameterMarkerSegments(ParameterMarkerSegmentBinder.bind(sqlStatement.getParameterMarkerSegments(), parameterMarkerSegmentBoundedInfos));
204         result.getCommentSegments().addAll(sqlStatement.getCommentSegments());
205         return result;
206     }
