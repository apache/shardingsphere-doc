20  import org.apache.shardingsphere.infra.binder.statement.SQLStatementContext;
21  import org.apache.shardingsphere.proxy.backend.handler.admin.executor.DatabaseAdminExecutor;
22  import org.apache.shardingsphere.proxy.backend.handler.admin.executor.DatabaseAdminExecutorCreator;
23  import org.apache.shardingsphere.proxy.backend.postgresql.handler.admin.PostgreSQLAdminExecutorCreator;
24  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.item.ExpressionProjectionSegment;
25  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.item.ProjectionSegment;
26  import org.apache.shardingsphere.sql.parser.sql.common.statement.dml.SelectStatement;
27  
28  import java.util.Collection;
29  import java.util.List;
30  import java.util.Optional;
31  import java.util.Set;
32  import java.util.TreeSet;
33  
34  /**
35   * Database admin executor creator for openGauss.
36   */
37  public final class OpenGaussAdminExecutorCreator implements DatabaseAdminExecutorCreator {
38      
39      private static final Set<String> SYSTEM_CATALOG_QUERY_EXPRESSIONS = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
40      
41      private static final Set<String> SYSTEM_CATALOG_TABLES = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
42      
43      static {
44          SYSTEM_CATALOG_QUERY_EXPRESSIONS.add("VERSION()");
45          SYSTEM_CATALOG_QUERY_EXPRESSIONS.add("intervaltonum(gs_password_deadline())");
46          SYSTEM_CATALOG_QUERY_EXPRESSIONS.add("gs_password_notifytime()");
47          SYSTEM_CATALOG_TABLES.add("pg_database");
48          SYSTEM_CATALOG_TABLES.add("pg_tables");
49          SYSTEM_CATALOG_TABLES.add("pg_roles");
50      }
51      
52      private final PostgreSQLAdminExecutorCreator delegated = new PostgreSQLAdminExecutorCreator();
53      
54      @Override
55      public Optional<DatabaseAdminExecutor> create(final SQLStatementContext sqlStatementContext) {
56          return delegated.create(sqlStatementContext);
57      }
58      
59      @Override
60      public Optional<DatabaseAdminExecutor> create(final SQLStatementContext sqlStatementContext, final String sql, final String databaseName, final List<Object> parameters) {
61          if (isSystemCatalogQuery(sqlStatementContext)) {
62              return Optional.of(new OpenGaussSystemCatalogAdminQueryExecutor(sqlStatementContext, sql, databaseName, parameters));
63          }
64          return delegated.create(sqlStatementContext, sql, databaseName, parameters);
65      }
66      
67      private boolean isSystemCatalogQuery(final SQLStatementContext sqlStatementContext) {
68          if (sqlStatementContext.getTablesContext().getTableNames().stream().anyMatch(SYSTEM_CATALOG_TABLES::contains)) {
69              return true;
70          }
71          if (!(sqlStatementContext.getSqlStatement() instanceof SelectStatement)) {
72              return false;
73          }
74          SelectStatement selectStatement = (SelectStatement) sqlStatementContext.getSqlStatement();
75          Collection<ProjectionSegment> projections = selectStatement.getProjections().getProjections();
76          return 1 == projections.size() && projections.iterator().next() instanceof ExpressionProjectionSegment
77                  && SYSTEM_CATALOG_QUERY_EXPRESSIONS.contains(((ExpressionProjectionSegment) projections.iterator().next()).getText());
78      }
79      
80      @Override
81      public String getDatabaseType() {
82          return "openGauss";
83      }
84  }
