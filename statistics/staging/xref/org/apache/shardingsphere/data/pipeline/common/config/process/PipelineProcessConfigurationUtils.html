18  package org.apache.shardingsphere.data.pipeline.common.config.process;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.data.pipeline.common.config.process.yaml.YamlPipelineProcessConfiguration;
23  import org.apache.shardingsphere.data.pipeline.common.config.process.yaml.YamlPipelineReadConfiguration;
24  import org.apache.shardingsphere.data.pipeline.common.config.process.yaml.YamlPipelineWriteConfiguration;
25  import org.apache.shardingsphere.data.pipeline.common.config.process.yaml.swapper.YamlPipelineProcessConfigurationSwapper;
26  import org.apache.shardingsphere.infra.yaml.config.pojo.algorithm.YamlAlgorithmConfiguration;
27  
28  import java.util.Properties;
29  
30  /**
31   * Pipeline process configuration utility class.
32   */
33  @NoArgsConstructor(access = AccessLevel.PRIVATE)
34  public final class PipelineProcessConfigurationUtils {
35      
36      private static final YamlPipelineProcessConfigurationSwapper SWAPPER = new YamlPipelineProcessConfigurationSwapper();
37      
38      /**
39       * Convert with default value.
40       *
41       * @param originalConfig original process configuration, nullable
42       * @return process configuration
43       */
44      public static PipelineProcessConfiguration convertWithDefaultValue(final PipelineProcessConfiguration originalConfig) {
45          YamlPipelineProcessConfiguration yamlConfig = null != originalConfig ? SWAPPER.swapToYamlConfiguration(originalConfig) : new YamlPipelineProcessConfiguration();
46          fillInDefaultValue(yamlConfig);
47          return SWAPPER.swapToObject(yamlConfig);
48      }
49      
50      /**
51       * Fill in default value.
52       *
53       * @param yamlConfig YAML configuration, non-null
54       */
55      public static void fillInDefaultValue(final YamlPipelineProcessConfiguration yamlConfig) {
56          if (null == yamlConfig.getRead()) {
57              yamlConfig.setRead(YamlPipelineReadConfiguration.buildWithDefaultValue());
58          } else {
59              yamlConfig.getRead().fillInNullFieldsWithDefaultValue();
60          }
61          if (null == yamlConfig.getWrite()) {
62              yamlConfig.setWrite(YamlPipelineWriteConfiguration.buildWithDefaultValue());
63          } else {
64              yamlConfig.getWrite().fillInNullFieldsWithDefaultValue();
65          }
66          if (null == yamlConfig.getStreamChannel()) {
67              YamlAlgorithmConfiguration yamlAlgorithmConfig = new YamlAlgorithmConfiguration();
68              yamlAlgorithmConfig.setType("MEMORY");
69              Properties props = new Properties();
70              props.setProperty("block-queue-size", "2000");
71              yamlAlgorithmConfig.setProps(props);
72              yamlConfig.setStreamChannel(yamlAlgorithmConfig);
73          }
74      }
75  }
