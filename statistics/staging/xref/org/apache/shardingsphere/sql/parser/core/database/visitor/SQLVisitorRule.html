725     /**
726      * Value of visitor rule.
727      * 
728      * @param parseTreeClass parse tree class
729      * @return visitor rule
730      * @throws SQLASTVisitorException SQL AST visitor exception
731      */
732     public static SQLVisitorRule valueOf(final Class<? extends ParseTree> parseTreeClass) {
733         String parseTreeClassName = parseTreeClass.getSimpleName();
734         for (SQLVisitorRule each : values()) {
735             if (each.getContextName().equals(parseTreeClassName)) {
736                 return each;
737             }
738         }
739         throw new SQLASTVisitorException(parseTreeClass);
