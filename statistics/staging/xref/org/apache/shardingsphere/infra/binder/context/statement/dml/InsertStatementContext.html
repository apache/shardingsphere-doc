58  /**
59   * Insert SQL statement context.
60   */
61  @Getter
62  public final class InsertStatementContext extends CommonSQLStatementContext implements TableAvailable, ParameterAware {
63      
64      private final TablesContext tablesContext;
65      
66      private final List<String> columnNames;
67      
68      private final ShardingSphereMetaData metaData;
69      
70      private final String currentDatabaseName;
71      
72      private final List<String> insertColumnNames;
73      
74      private final List<List<ExpressionSegment>> valueExpressions;
75      
76      private List<InsertValueContext> insertValueContexts;
77      
78      private InsertSelectContext insertSelectContext;
79      
80      private OnDuplicateUpdateContext onDuplicateKeyUpdateValueContext;
81      
82      private GeneratedKeyContext generatedKeyContext;
83      
84      public InsertStatementContext(final ShardingSphereMetaData metaData, final List<Object> params, final InsertStatement sqlStatement, final String currentDatabaseName) {
85          super(sqlStatement);
86          this.metaData = metaData;
87          this.currentDatabaseName = currentDatabaseName;
88          insertColumnNames = getInsertColumnNames();
89          valueExpressions = getAllValueExpressions(sqlStatement);
90          AtomicInteger parametersOffset = new AtomicInteger(0);
91          insertValueContexts = getInsertValueContexts(params, parametersOffset, valueExpressions);
92          insertSelectContext = getInsertSelectContext(metaData, params, parametersOffset, currentDatabaseName).orElse(null);
93          onDuplicateKeyUpdateValueContext = getOnDuplicateKeyUpdateValueContext(params, parametersOffset).orElse(null);
94          tablesContext = new TablesContext(getAllSimpleTableSegments(), getDatabaseType(), currentDatabaseName);
95          ShardingSphereSchema schema = getSchema(metaData, currentDatabaseName);
96          columnNames = containsInsertColumns() ? insertColumnNames
97                  : Optional.ofNullable(sqlStatement.getTable()).map(optional -> schema.getVisibleColumnNames(optional.getTableName().getIdentifier().getValue())).orElseGet(Collections::emptyList);
98          generatedKeyContext = new GeneratedKeyContextEngine(sqlStatement, schema).createGenerateKeyContext(insertColumnNames, insertValueContexts, params).orElse(null);
99      }
100     
101     private ShardingSphereSchema getSchema(final ShardingSphereMetaData metaData, final String currentDatabaseName) {
102         String databaseName = tablesContext.getDatabaseName().orElse(currentDatabaseName);
103         ShardingSpherePreconditions.checkNotNull(databaseName, NoDatabaseSelectedException::new);
104         ShardingSphereDatabase database = metaData.getDatabase(databaseName);
105         ShardingSpherePreconditions.checkNotNull(database, () -> new UnknownDatabaseException(databaseName));
106         String defaultSchema = new DatabaseTypeRegistry(getDatabaseType()).getDefaultSchemaName(databaseName);
107         return tablesContext.getSchemaName().map(database::getSchema).orElseGet(() -> database.getSchema(defaultSchema));
108     }
109     
110     private Collection<SimpleTableSegment> getAllSimpleTableSegments() {
111         TableExtractor tableExtractor = new TableExtractor();
112         tableExtractor.extractTablesFromInsert(getSqlStatement());
113         return tableExtractor.getRewriteTables();
114     }
115     
116     private List<InsertValueContext> getInsertValueContexts(final List<Object> params, final AtomicInteger paramsOffset, final List<List<ExpressionSegment>> valueExpressions) {
117         List<InsertValueContext> result = new LinkedList<>();
118         for (Collection<ExpressionSegment> each : valueExpressions) {
119             InsertValueContext insertValueContext = new InsertValueContext(each, params, paramsOffset.get());
120             result.add(insertValueContext);
121             paramsOffset.addAndGet(insertValueContext.getParameterCount());
122         }
123         return result;
124     }
125     
126     private Optional<InsertSelectContext> getInsertSelectContext(final ShardingSphereMetaData metaData, final List<Object> params,
127                                                                  final AtomicInteger paramsOffset, final String currentDatabaseName) {
128         if (!getSqlStatement().getInsertSelect().isPresent()) {
129             return Optional.empty();
130         }
131         SubquerySegment insertSelectSegment = getSqlStatement().getInsertSelect().get();
132         SelectStatementContext selectStatementContext = new SelectStatementContext(metaData, params, insertSelectSegment.getSelect(), currentDatabaseName, Collections.emptyList());
133         selectStatementContext.setSubqueryType(SubqueryType.INSERT_SELECT_SUBQUERY);
134         InsertSelectContext insertSelectContext = new InsertSelectContext(selectStatementContext, params, paramsOffset.get());
135         paramsOffset.addAndGet(insertSelectContext.getParameterCount());
136         return Optional.of(insertSelectContext);
137     }
138     
139     private Optional<OnDuplicateUpdateContext> getOnDuplicateKeyUpdateValueContext(final List<Object> params, final AtomicInteger parametersOffset) {
140         Optional<OnDuplicateKeyColumnsSegment> onDuplicateKeyColumnsSegment = getSqlStatement().getOnDuplicateKeyColumns();
141         if (!onDuplicateKeyColumnsSegment.isPresent()) {
142             return Optional.empty();
143         }
144         Collection<ColumnAssignmentSegment> onDuplicateKeyColumns = onDuplicateKeyColumnsSegment.get().getColumns();
145         OnDuplicateUpdateContext onDuplicateUpdateContext = new OnDuplicateUpdateContext(onDuplicateKeyColumns, params, parametersOffset.get());
146         parametersOffset.addAndGet(onDuplicateUpdateContext.getParameterCount());
147         return Optional.of(onDuplicateUpdateContext);
148     }
149     
150     /**
151      * Get column names for descending order.
152      *
153      * @return column names for descending order
154      */
155     public Iterator<String> getDescendingColumnNames() {
156         return new LinkedList<>(columnNames).descendingIterator();
157     }
158     
159     /**
160      * Get grouped parameters.
161      *
162      * @return grouped parameters
163      */
164     public List<List<Object>> getGroupedParameters() {
165         List<List<Object>> result = new LinkedList<>();
166         for (InsertValueContext each : insertValueContexts) {
167             result.add(each.getParameters());
168         }
169         if (null != insertSelectContext && !insertSelectContext.getParameters().isEmpty()) {
170             result.add(insertSelectContext.getParameters());
171         }
172         return result;
173     }
174     
175     /**
176      * Get on duplicate key update parameters.
177      *
178      * @return on duplicate key update parameters
179      */
180     public List<Object> getOnDuplicateKeyUpdateParameters() {
181         return null == onDuplicateKeyUpdateValueContext ? new ArrayList<>() : onDuplicateKeyUpdateValueContext.getParameters();
182     }
183     
184     /**
185      * Get generated key context.
186      *
187      * @return generated key context
188      */
189     public Optional<GeneratedKeyContext> getGeneratedKeyContext() {
190         return Optional.ofNullable(generatedKeyContext);
191     }
192     
193     /**
194      * Judge whether contains insert columns.
195      *
196      * @return contains insert columns or not
197      */
198     public boolean containsInsertColumns() {
199         InsertStatement insertStatement = getSqlStatement();
200         return !insertStatement.getColumns().isEmpty() || insertStatement.getSetAssignment().isPresent();
201     }
202     
203     /**
204      * Get value list count.
205      *
206      * @return value list count
207      */
208     public int getValueListCount() {
209         InsertStatement insertStatement = getSqlStatement();
210         return insertStatement.getSetAssignment().isPresent() ? 1 : insertStatement.getValues().size();
211     }
212     
213     /**
214      * Get insert column names.
215      *
216      * @return column names collection
217      */
218     public List<String> getInsertColumnNames() {
219         InsertStatement insertStatement = getSqlStatement();
220         return insertStatement.getSetAssignment().map(this::getColumnNamesForSetAssignment).orElseGet(() -> getColumnNamesForInsertColumns(insertStatement.getColumns()));
221     }
222     
223     private List<String> getColumnNamesForSetAssignment(final SetAssignmentSegment setAssignment) {
224         List<String> result = new LinkedList<>();
225         for (ColumnAssignmentSegment each : setAssignment.getAssignments()) {
226             result.add(each.getColumns().get(0).getIdentifier().getValue().toLowerCase());
227         }
228         return result;
229     }
230     
231     private List<String> getColumnNamesForInsertColumns(final Collection<ColumnSegment> columns) {
232         List<String> result = new LinkedList<>();
233         for (ColumnSegment each : columns) {
234             result.add(each.getIdentifier().getValue().toLowerCase());
235         }
236         return result;
237     }
238     
239     private List<List<ExpressionSegment>> getAllValueExpressions(final InsertStatement insertStatement) {
240         Optional<SetAssignmentSegment> setAssignment = insertStatement.getSetAssignment();
241         return setAssignment
242                 .map(optional -> Collections.singletonList(getAllValueExpressionsFromSetAssignment(optional))).orElseGet(() -> getAllValueExpressionsFromValues(insertStatement.getValues()));
243     }
244     
245     private List<ExpressionSegment> getAllValueExpressionsFromSetAssignment(final SetAssignmentSegment setAssignment) {
246         List<ExpressionSegment> result = new ArrayList<>(setAssignment.getAssignments().size());
247         for (ColumnAssignmentSegment each : setAssignment.getAssignments()) {
248             result.add(each.getValue());
249         }
250         return result;
251     }
252     
253     private List<List<ExpressionSegment>> getAllValueExpressionsFromValues(final Collection<InsertValuesSegment> values) {
254         List<List<ExpressionSegment>> result = new ArrayList<>(values.size());
255         for (InsertValuesSegment each : values) {
256             result.add(each.getValues());
257         }
258         return result;
259     }
260     
261     @Override
262     public InsertStatement getSqlStatement() {
263         return (InsertStatement) super.getSqlStatement();
264     }
265     
266     @Override
267     public void setUpParameters(final List<Object> params) {
268         AtomicInteger parametersOffset = new AtomicInteger(0);
269         insertValueContexts = getInsertValueContexts(params, parametersOffset, valueExpressions);
270         insertSelectContext = getInsertSelectContext(metaData, params, parametersOffset, currentDatabaseName).orElse(null);
271         onDuplicateKeyUpdateValueContext = getOnDuplicateKeyUpdateValueContext(params, parametersOffset).orElse(null);
272         ShardingSphereSchema schema = getSchema(metaData, currentDatabaseName);
273         generatedKeyContext = new GeneratedKeyContextEngine(getSqlStatement(), schema).createGenerateKeyContext(insertColumnNames, insertValueContexts, params).orElse(null);
274     }
