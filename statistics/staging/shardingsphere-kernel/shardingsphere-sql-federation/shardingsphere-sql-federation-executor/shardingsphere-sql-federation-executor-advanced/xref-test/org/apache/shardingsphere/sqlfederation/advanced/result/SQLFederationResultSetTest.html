68      @BeforeEach
69      void setUp() {
70          enumerator = createEnumerator();
71          federationResultSet = new SQLFederationResultSet(enumerator, mock(ShardingSphereSchema.class), mock(FilterableSchema.class), createSelectStatementContext(), mock(RelDataType.class));
72      }
73      
74      private static SelectStatementContext createSelectStatementContext() {
75          SelectStatementContext result = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
76          List<Projection> projections = Arrays.asList(new ColumnProjection("o", "order_id", null), new ColumnProjection("o", "user_id", null), new ColumnProjection("o", "status", null),
77                  new ColumnProjection("i", "item_id", null));
78          when(result.getProjectionsContext().getExpandProjections()).thenReturn(projections);
79          TablesContext tablesContext = mock(TablesContext.class);
80          when(tablesContext.getTableNames()).thenReturn(Collections.emptyList());
81          when(result.getTablesContext()).thenReturn(tablesContext);
82          return result;
83      }
84      
85      @SuppressWarnings("unchecked")
86      private Enumerator<Object> createEnumerator() {
87          Enumerator<Object> result = mock(Enumerator.class);
88          when(result.moveNext()).thenReturn(true, false);
89          when(result.current()).thenReturn(new Object[]{1, 1, "OK", 1});
90          return result;
91      }
92

94      void clean() {
95          enumerator.close();
96          federationResultSet.close();
97      }
