199 /**
200  * DDL statement visitor for Doris.
201  */
202 public final class DorisDDLStatementVisitor extends DorisStatementVisitor implements DDLStatementVisitor {
203     
204     @Override
205     public ASTNode visitCreateView(final CreateViewContext ctx) {
206         DorisCreateViewStatement result = new DorisCreateViewStatement();
207         result.setView((SimpleTableSegment) visit(ctx.viewName()));
208         result.setViewDefinition(getOriginalText(ctx.select()));
209         result.setSelect((DorisSelectStatement) visit(ctx.select()));
210         return result;
211     }
212     
213     @Override
214     public ASTNode visitAlterView(final AlterViewContext ctx) {
215         DorisAlterViewStatement result = new DorisAlterViewStatement();
216         result.setView((SimpleTableSegment) visit(ctx.viewName()));
217         result.setViewDefinition(getOriginalText(ctx.select()));
218         result.setSelect((DorisSelectStatement) visit(ctx.select()));
219         return result;
220     }
221     
222     @SuppressWarnings("unchecked")
223     @Override
224     public ASTNode visitDropView(final DropViewContext ctx) {
225         DorisDropViewStatement result = new DorisDropViewStatement();
226         result.getViews().addAll(((CollectionValue<SimpleTableSegment>) visit(ctx.viewNames())).getValue());
227         return result;
228     }
229     
230     @Override
231     public ASTNode visitCreateDatabase(final CreateDatabaseContext ctx) {
232         DorisCreateDatabaseStatement result = new DorisCreateDatabaseStatement();
233         result.setDatabaseName(new IdentifierValue(ctx.schemaName().getText()).getValue());
234         result.setIfNotExists(null != ctx.ifNotExists());
235         return result;
236     }
237     
238     @Override
239     public ASTNode visitAlterDatabase(final AlterDatabaseContext ctx) {
240         return new DorisAlterDatabaseStatement();
241     }
242     
243     @Override
244     public ASTNode visitDropDatabase(final DropDatabaseContext ctx) {
245         DorisDropDatabaseStatement result = new DorisDropDatabaseStatement();
246         result.setDatabaseName(new IdentifierValue(ctx.schemaName().getText()).getValue());
247         result.setIfExists(null != ctx.ifExists());
248         return result;
249     }
250     
251     @SuppressWarnings("unchecked")
252     @Override
253     public ASTNode visitCreateTable(final CreateTableContext ctx) {
254         DorisCreateTableStatement result = new DorisCreateTableStatement(null != ctx.ifNotExists());
255         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
256         if (null != ctx.createDefinitionClause()) {
257             CollectionValue<CreateDefinitionSegment> createDefinitions = (CollectionValue<CreateDefinitionSegment>) visit(ctx.createDefinitionClause());
258             for (CreateDefinitionSegment each : createDefinitions.getValue()) {
259                 if (each instanceof ColumnDefinitionSegment) {
260                     result.getColumnDefinitions().add((ColumnDefinitionSegment) each);
261                 } else if (each instanceof ConstraintDefinitionSegment) {
262                     result.getConstraintDefinitions().add((ConstraintDefinitionSegment) each);
263                 }
264             }
265         }
266         if (null != ctx.createLikeClause()) {
267             result.setLikeTable((SimpleTableSegment) visit(ctx.createLikeClause()));
268         }
269         if (null != ctx.createTableOptions()) {
270             result.setCreateTableOption((CreateTableOptionSegment) visit(ctx.createTableOptions()));
271         }
272         // DORIS ADDED BEGIN
273         if (null != ctx.duplicateAsQueryExpression()) {
274             result.setSelectStatement((SelectStatement) visit(ctx.duplicateAsQueryExpression().select()));
275         }
276         // DORIS ADDED END
277         return result;
278     }
279     
280     @Override
281     public ASTNode visitCreateTableOptions(final CreateTableOptionsContext ctx) {
282         CreateTableOptionSegment result = new CreateTableOptionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
283         for (CreateTableOptionContext each : ctx.createTableOption()) {
284             if (null != each.engineRef()) {
285                 result.setEngine((EngineSegment) visit(each.engineRef()));
286             } else if (null != each.COMMENT()) {
287                 result.setCommentSegment(new CommentSegment(each.string_().getText(), each.string_().getStart().getStartIndex(), each.string_().getStop().getStopIndex()));
288             }
289         }
290         return result;
291     }
292     
293     @Override
294     public ASTNode visitCreateDefinitionClause(final CreateDefinitionClauseContext ctx) {
295         CollectionValue<CreateDefinitionSegment> result = new CollectionValue<>();
296         for (TableElementContext each : ctx.tableElementList().tableElement()) {
297             if (null != each.columnDefinition()) {
298                 result.getValue().add((ColumnDefinitionSegment) visit(each.columnDefinition()));
299             }
300             if (null != each.tableConstraintDef()) {
301                 result.getValue().add((ConstraintDefinitionSegment) visit(each.tableConstraintDef()));
302             }
303         }
304         return result;
305     }
306     
307     @Override
308     public ASTNode visitCreateLikeClause(final CreateLikeClauseContext ctx) {
309         return visit(ctx.tableName());
310     }
311     
312     @SuppressWarnings("unchecked")
313     @Override
314     public ASTNode visitAlterTable(final AlterTableContext ctx) {
315         DorisAlterTableStatement result = new DorisAlterTableStatement();
316         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
317         if (null == ctx.alterTableActions() || null == ctx.alterTableActions().alterCommandList() || null == ctx.alterTableActions().alterCommandList().alterList()) {
318             return result;
319         }
320         for (AlterDefinitionSegment each : ((CollectionValue<AlterDefinitionSegment>) visit(ctx.alterTableActions().alterCommandList().alterList())).getValue()) {
321             setAlterDefinition(result, each);
322         }
323         return result;
324     }
325     
326     private void setAlterDefinition(final DorisAlterTableStatement alterTableStatement, final AlterDefinitionSegment alterDefinitionSegment) {
327         if (alterDefinitionSegment instanceof AddColumnDefinitionSegment) {
328             alterTableStatement.getAddColumnDefinitions().add((AddColumnDefinitionSegment) alterDefinitionSegment);
329         } else if (alterDefinitionSegment instanceof ModifyColumnDefinitionSegment) {
330             alterTableStatement.getModifyColumnDefinitions().add((ModifyColumnDefinitionSegment) alterDefinitionSegment);
331         } else if (alterDefinitionSegment instanceof ChangeColumnDefinitionSegment) {
332             alterTableStatement.getChangeColumnDefinitions().add((ChangeColumnDefinitionSegment) alterDefinitionSegment);
333         } else if (alterDefinitionSegment instanceof DropColumnDefinitionSegment) {
334             alterTableStatement.getDropColumnDefinitions().add((DropColumnDefinitionSegment) alterDefinitionSegment);
335         } else if (alterDefinitionSegment instanceof AddConstraintDefinitionSegment) {
336             alterTableStatement.getAddConstraintDefinitions().add((AddConstraintDefinitionSegment) alterDefinitionSegment);
337         } else if (alterDefinitionSegment instanceof DropConstraintDefinitionSegment) {
338             alterTableStatement.getDropConstraintDefinitions().add((DropConstraintDefinitionSegment) alterDefinitionSegment);
339         } else if (alterDefinitionSegment instanceof RenameTableDefinitionSegment) {
340             alterTableStatement.setRenameTable(((RenameTableDefinitionSegment) alterDefinitionSegment).getRenameTable());
341         } else if (alterDefinitionSegment instanceof ConvertTableDefinitionSegment) {
342             alterTableStatement.setConvertTableDefinition((ConvertTableDefinitionSegment) alterDefinitionSegment);
343         } else if (alterDefinitionSegment instanceof DropIndexDefinitionSegment) {
344             alterTableStatement.getDropIndexDefinitions().add((DropIndexDefinitionSegment) alterDefinitionSegment);
345         } else if (alterDefinitionSegment instanceof RenameIndexDefinitionSegment) {
346             alterTableStatement.getRenameIndexDefinitions().add((RenameIndexDefinitionSegment) alterDefinitionSegment);
347         } else if (alterDefinitionSegment instanceof RenameColumnSegment) {
348             alterTableStatement.getRenameColumnDefinitions().add((RenameColumnSegment) alterDefinitionSegment);
349         } else if (alterDefinitionSegment instanceof AlgorithmTypeSegment) {
350             alterTableStatement.setAlgorithmSegment((AlgorithmTypeSegment) alterDefinitionSegment);
351         } else if (alterDefinitionSegment instanceof LockTableSegment) {
352             alterTableStatement.setLockTableSegment((LockTableSegment) alterDefinitionSegment);
353         }
354     }
355     
356     private ColumnDefinitionSegment generateColumnDefinitionSegment(final ColumnSegment column, final FieldDefinitionContext ctx) {
357         DataTypeSegment dataTypeSegment = (DataTypeSegment) visit(ctx.dataType());
358         boolean isPrimaryKey = ctx.columnAttribute().stream().anyMatch(each -> null != each.KEY() && null == each.UNIQUE());
359         boolean isAutoIncrement = ctx.columnAttribute().stream().anyMatch(each -> null != each.AUTO_INCREMENT());

396     private Optional<AlterDefinitionSegment> getAlterDefinitionSegment(final AlterListContext alterListContext, final AlterListItemContext alterListItemContext) {
397         if (alterListItemContext instanceof AddColumnContext) {
398             return Optional.of((AddColumnDefinitionSegment) visit(alterListItemContext));
399         }
400         if (alterListItemContext instanceof AlterConstraintContext || alterListItemContext instanceof AlterCheckContext) {
401             return Optional.of((AlterDefinitionSegment) visit(alterListItemContext));
402         }
403         if (alterListItemContext instanceof ChangeColumnContext) {
404             return Optional.of(generateModifyColumnDefinitionSegment((ChangeColumnContext) alterListItemContext));
405         }
406         if (alterListItemContext instanceof ModifyColumnContext) {
407             return Optional.of(generateModifyColumnDefinitionSegment((ModifyColumnContext) alterListItemContext));
408         }
409         if (alterListItemContext instanceof AlterTableDropContext) {
410             return getDropItemDefinitionSegment(alterListContext, (AlterTableDropContext) alterListItemContext);
411         }
412         if (alterListItemContext instanceof AddTableConstraintContext) {
413             return Optional.of((AddConstraintDefinitionSegment) visit(alterListItemContext));
414         }
415         if (alterListItemContext instanceof AlterRenameTableContext) {
416             return Optional.of((RenameTableDefinitionSegment) visit(alterListItemContext));
417         }
418         if (alterListItemContext instanceof AlterConvertContext) {
419             return Optional.of((ConvertTableDefinitionSegment) visit(alterListItemContext));
420         }
421         if (alterListItemContext instanceof RenameColumnContext) {
422             return Optional.of((RenameColumnSegment) visit(alterListItemContext));
423         }
424         if (alterListItemContext instanceof RenameIndexContext) {
425             return Optional.of((RenameIndexDefinitionSegment) visit(alterListItemContext));
426         }
427         return Optional.empty();
428     }

430     private Optional<AlterDefinitionSegment> getDropItemDefinitionSegment(final AlterListContext alterListContext, final AlterTableDropContext alterTableDrop) {
431         if (null != alterTableDrop.CHECK() || null != alterTableDrop.CONSTRAINT()) {
432             ConstraintSegment constraint = new ConstraintSegment(alterTableDrop.identifier().getStart().getStartIndex(), alterTableDrop.identifier().getStop().getStopIndex(),
433                     (IdentifierValue) visit(alterTableDrop.identifier()));
434             return Optional.of(new DropConstraintDefinitionSegment(alterListContext.getStart().getStartIndex(), alterListContext.getStop().getStopIndex(), constraint));
435         }
436         if (null == alterTableDrop.KEY() && null == alterTableDrop.keyOrIndex()) {
437             ColumnSegment column = new ColumnSegment(alterTableDrop.columnInternalRef.start.getStartIndex(), alterTableDrop.columnInternalRef.stop.getStopIndex(),
438                     (IdentifierValue) visit(alterTableDrop.columnInternalRef));
439             return Optional.of(new DropColumnDefinitionSegment(alterTableDrop.getStart().getStartIndex(), alterTableDrop.getStop().getStopIndex(), Collections.singleton(column)));
440         }
441         if (null != alterTableDrop.keyOrIndex()) {
442             return Optional.of(
443                     new DropIndexDefinitionSegment(alterListContext.getStart().getStartIndex(), alterListContext.getStop().getStopIndex(), (IndexSegment) visit(alterTableDrop.indexName())));
444         }
445         return Optional.empty();
446     }

597     @Override
598     public ASTNode visitTableConstraintDef(final TableConstraintDefContext ctx) {
599         ConstraintDefinitionSegment result = new ConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
600         if (null != ctx.constraintClause() && null != ctx.constraintClause().constraintName()) {
601             result.setConstraintName((ConstraintSegment) visit(ctx.constraintClause().constraintName()));
602         }
603         if (null != ctx.KEY() && null != ctx.PRIMARY()) {
604             result.setPrimaryKey(true);
605             result.getPrimaryKeyColumns().addAll(((CollectionValue) visit(ctx.keyListWithExpression())).getValue());
606             return result;
607         }
608         if (null != ctx.FOREIGN()) {
609             result.setReferencedTable((SimpleTableSegment) visit(ctx.referenceDefinition()));
610             return result;
611         }
612         if (null != ctx.UNIQUE()) {
613             result.setUniqueKey(true);
614             result.getIndexColumns().addAll(((CollectionValue) visit(ctx.keyListWithExpression())).getValue());
615             if (null != ctx.indexName()) {
616                 result.setIndexName((IndexSegment) visit(ctx.indexName()));
617             }
618             return result;
619         }
620         if (null != ctx.checkConstraint()) {
621             return result;
622         }
623         result.getIndexColumns().addAll(((CollectionValue) visit(ctx.keyListWithExpression())).getValue());
624         if (null != ctx.indexName()) {
625             result.setIndexName((IndexSegment) visit(ctx.indexName()));
626         }
627         return result;
628     }
