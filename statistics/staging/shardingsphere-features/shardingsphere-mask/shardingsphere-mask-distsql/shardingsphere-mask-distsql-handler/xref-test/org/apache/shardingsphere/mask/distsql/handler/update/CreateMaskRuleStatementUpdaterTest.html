41  import static org.hamcrest.CoreMatchers.is;
42  import static org.hamcrest.MatcherAssert.assertThat;
43  import static org.junit.jupiter.api.Assertions.assertThrows;
44  import static org.junit.jupiter.api.Assertions.assertTrue;
45  
46  @ExtendWith(MockitoExtension.class)
47  class CreateMaskRuleStatementUpdaterTest {
48      
49      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
50      private ShardingSphereDatabase database;
51      
52      private final CreateMaskRuleStatementUpdater updater = new CreateMaskRuleStatementUpdater();
53      
54      @Test
55      void assertCheckSQLStatementWithDuplicateMaskRule() {
56          assertThrows(DuplicateRuleException.class, () -> updater.checkSQLStatement(database, createDuplicatedSQLStatement(false, "MD5"), getCurrentRuleConfig()));
57      }
58      
59      @Test
60      void assertCheckSQLStatementWithInvalidAlgorithm() {
61          assertThrows(ServiceProviderNotFoundServerException.class, () -> updater.checkSQLStatement(database, createSQLStatement(false, "INVALID_TYPE"), null));
62      }
63      
64      @Test
65      void assertCreateMaskRule() {
66          MaskRuleConfiguration currentRuleConfig = getCurrentRuleConfig();
67          CreateMaskRuleStatement sqlStatement = createSQLStatement(false, "MD5");
68          updater.checkSQLStatement(database, sqlStatement, currentRuleConfig);
69          MaskRuleConfiguration toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, sqlStatement);
70          updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
71          assertThat(currentRuleConfig.getTables().size(), is(4));
72          assertTrue(currentRuleConfig.getMaskAlgorithms().containsKey("t_mask_1_user_id_md5"));
73          assertTrue(currentRuleConfig.getMaskAlgorithms().containsKey("t_order_1_order_id_md5"));
74      }
75      
76      @Test
77      void assertCreateMaskRuleWithIfNotExists() {
78          MaskRuleConfiguration currentRuleConfig = getCurrentRuleConfig();
79          CreateMaskRuleStatement sqlStatement = createSQLStatement(false, "MD5");
80          updater.checkSQLStatement(database, sqlStatement, currentRuleConfig);
81          MaskRuleConfiguration toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, sqlStatement);
82          updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
83          sqlStatement = createSQLStatement(true, "MD5");
84          updater.checkSQLStatement(database, sqlStatement, currentRuleConfig);
85          toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, sqlStatement);
86          updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
87          assertThat(currentRuleConfig.getTables().size(), is(4));
88          assertTrue(currentRuleConfig.getMaskAlgorithms().containsKey("t_mask_1_user_id_md5"));
89          assertTrue(currentRuleConfig.getMaskAlgorithms().containsKey("t_order_1_order_id_md5"));
90      }
91      
92      private CreateMaskRuleStatement createSQLStatement(final boolean ifNotExists, final String algorithmType) {
93          MaskColumnSegment tMaskColumnSegment = new MaskColumnSegment("user_id", new AlgorithmSegment(algorithmType, new Properties()));
94          MaskColumnSegment tOrderColumnSegment = new MaskColumnSegment("order_id", new AlgorithmSegment(algorithmType, new Properties()));
95          MaskRuleSegment tMaskRuleSegment = new MaskRuleSegment("t_mask_1", Collections.singleton(tMaskColumnSegment));
96          MaskRuleSegment tOrderRuleSegment = new MaskRuleSegment("t_order_1", Collections.singleton(tOrderColumnSegment));
97          Collection<MaskRuleSegment> rules = new LinkedList<>();
98          rules.add(tMaskRuleSegment);
99          rules.add(tOrderRuleSegment);
100         return new CreateMaskRuleStatement(ifNotExists, rules);
101     }
102     
103     private CreateMaskRuleStatement createDuplicatedSQLStatement(final boolean ifNotExists, final String algorithmType) {
104         MaskColumnSegment tMaskColumnSegment = new MaskColumnSegment("user_id", new AlgorithmSegment(algorithmType, new Properties()));
105         MaskColumnSegment tOrderColumnSegment = new MaskColumnSegment("order_id", new AlgorithmSegment(algorithmType, new Properties()));
106         MaskRuleSegment tMaskRuleSegment = new MaskRuleSegment("t_mask", Collections.singleton(tMaskColumnSegment));
107         MaskRuleSegment tOrderRuleSegment = new MaskRuleSegment("t_order", Collections.singleton(tOrderColumnSegment));
108         Collection<MaskRuleSegment> rules = new LinkedList<>();
109         rules.add(tMaskRuleSegment);
110         rules.add(tOrderRuleSegment);
111         return new CreateMaskRuleStatement(ifNotExists, rules);
112     }
113     
114     private MaskRuleConfiguration getCurrentRuleConfig() {
115         Collection<MaskTableRuleConfiguration> rules = new LinkedList<>();
116         rules.add(new MaskTableRuleConfiguration("t_mask", Collections.emptyList()));
117         rules.add(new MaskTableRuleConfiguration("t_order", Collections.emptyList()));
118         return new MaskRuleConfiguration(rules, new HashMap<>());
119     }
