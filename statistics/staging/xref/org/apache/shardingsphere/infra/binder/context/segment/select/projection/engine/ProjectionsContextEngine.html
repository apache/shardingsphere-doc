40  import java.util.Collection;
41  import java.util.LinkedList;

43  /**
44   * Projections context engine.
45   */
46  public final class ProjectionsContextEngine {
47      
48      private final ProjectionEngine projectionEngine;
49      
50      public ProjectionsContextEngine(final DatabaseType databaseType) {
51          projectionEngine = new ProjectionEngine(databaseType);
52      }
53      
54      /**
55       * Create projections context.
56       *
57       * @param projectionsSegment projection segments
58       * @param groupByContext group by context
59       * @param orderByContext order by context
60       * @return projections context
61       */
62      public ProjectionsContext createProjectionsContext(final ProjectionsSegment projectionsSegment, final GroupByContext groupByContext, final OrderByContext orderByContext) {
63          Collection<Projection> projections = getProjections(projectionsSegment);
64          ProjectionsContext result = new ProjectionsContext(projectionsSegment.getStartIndex(), projectionsSegment.getStopIndex(), projectionsSegment.isDistinctRow(), projections);
65          result.getProjections().addAll(getDerivedGroupByColumns(groupByContext, projections));
66          result.getProjections().addAll(getDerivedOrderByColumns(orderByContext, projections));
67          return result;
68      }
69      
70      private Collection<Projection> getProjections(final ProjectionsSegment projectionsSegment) {
71          Collection<Projection> result = new LinkedList<>();
72          for (ProjectionSegment each : projectionsSegment.getProjections()) {
73              projectionEngine.createProjection(each).ifPresent(result::add);
74          }
75          return result;
76      }
77      
78      private Collection<Projection> getDerivedGroupByColumns(final GroupByContext groupByContext, final Collection<Projection> projections) {
79          return getDerivedOrderColumns(groupByContext.getItems(), DerivedColumn.GROUP_BY_ALIAS, projections);
80      }
81      
82      private Collection<Projection> getDerivedOrderByColumns(final OrderByContext orderByContext, final Collection<Projection> projections) {
83          return getDerivedOrderColumns(orderByContext.getItems(), DerivedColumn.ORDER_BY_ALIAS, projections);
84      }
85      
86      private Collection<Projection> getDerivedOrderColumns(final Collection<OrderByItem> orderItems, final DerivedColumn derivedColumn, final Collection<Projection> projections) {
87          Collection<Projection> result = new LinkedList<>();
88          int derivedColumnOffset = 0;
89          for (OrderByItem each : orderItems) {
90              if (!containsProjection(each.getSegment(), projections)) {
91                  result.add(new DerivedProjection(((TextOrderByItemSegment) each.getSegment()).getText(), new IdentifierValue(derivedColumn.getDerivedColumnAlias(derivedColumnOffset++)),
92                          each.getSegment()));
93              }
94          }
95          return result;
96      }
97      
98      private boolean containsProjection(final OrderByItemSegment orderByItem, final Collection<Projection> projections) {
99          if (orderByItem instanceof IndexOrderByItemSegment) {
100             return true;
101         }
102         for (Projection each : projections) {
103             if (orderByItem instanceof ColumnOrderByItemSegment && isSameColumn(each, ((ColumnOrderByItemSegment) orderByItem).getColumn())) {
104                 return true;
105             }
106             String text = ((TextOrderByItemSegment) orderByItem).getText();
107             if (isSameAlias(each, text) || isSameQualifiedName(each, text)) {
108                 return true;
109             }
110         }
111         return false;
112     }
113     
114     private boolean isSameColumn(final Projection projection, final ColumnSegment columnSegment) {
115         Collection<ColumnProjection> columns = getColumnProjections(projection);
116         if (columns.isEmpty()) {
117             return false;
118         }
119         boolean columnSegmentPresent = columnSegment.getOwner().isPresent();
120         for (ColumnProjection each : columns) {
121             if (columnSegmentPresent ? isSameQualifiedName(each, columnSegment.getQualifiedName()) : isSameName(each, columnSegment.getQualifiedName())) {
122                 return true;
123             }
124         }
125         return false;
126     }
127     
128     private Collection<ColumnProjection> getColumnProjections(final Projection projection) {
129         Collection<ColumnProjection> result = new LinkedList<>();
130         if (projection instanceof ColumnProjection) {
131             result.add((ColumnProjection) projection);
132         }
133         if (projection instanceof ShorthandProjection) {
134             result.addAll(((ShorthandProjection) projection).getColumnProjections());
135         }
136         return result;
137     }
138     
139     private boolean isSameName(final ColumnProjection projection, final String text) {
140         return SQLUtils.getExactlyValue(text).equalsIgnoreCase(projection.getName().getValue());
141     }
142     
143     private boolean isSameAlias(final Projection projection, final String text) {
144         return projection.getAlias().isPresent() && SQLUtils.getExactlyValue(text).equalsIgnoreCase(SQLUtils.getExactlyValue(projection.getAlias().get().getValue()));
145     }
146     
147     private boolean isSameQualifiedName(final Projection projection, final String text) {
148         return SQLUtils.getExactlyValue(text).equalsIgnoreCase(SQLUtils.getExactlyValue(projection.getExpression()));
149     }
150 }
