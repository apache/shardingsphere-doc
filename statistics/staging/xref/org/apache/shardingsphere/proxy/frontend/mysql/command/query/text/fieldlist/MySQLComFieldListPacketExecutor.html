50  /**
51   * COM_FIELD_LIST packet executor for MySQL.
52   */
53  @RequiredArgsConstructor
54  public final class MySQLComFieldListPacketExecutor implements CommandExecutor {
55      
56      private static final String SQL = "SHOW COLUMNS FROM %s FROM %s";
57      
58      private final MySQLComFieldListPacket packet;
59      
60      private final ConnectionSession connectionSession;
61      
62      private DatabaseConnector databaseConnector;
63      
64      @Override
65      public Collection<DatabasePacket> execute() throws SQLException {
66          String databaseName = connectionSession.getCurrentDatabaseName();
67          String sql = String.format(SQL, packet.getTable(), databaseName);
68          MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
69          SQLParserRule sqlParserRule = metaDataContexts.getMetaData().getGlobalRuleMetaData().getSingleRule(SQLParserRule.class);
70          SQLStatement sqlStatement = sqlParserRule.getSQLParserEngine(TypedSPILoader.getService(DatabaseType.class, "MySQL")).parse(sql, false);
71          HintValueContext hintValueContext = SQLHintUtils.extractHint(sql);
72          SQLStatementContext sqlStatementContext = new SQLBindEngine(metaDataContexts.getMetaData(), databaseName, hintValueContext).bind(sqlStatement, Collections.emptyList());
73          ProxyDatabaseConnectionManager databaseConnectionManager = connectionSession.getDatabaseConnectionManager();
74          QueryContext queryContext = new QueryContext(sqlStatementContext, sql, Collections.emptyList(), hintValueContext, connectionSession.getConnectionContext(), metaDataContexts.getMetaData());
75          databaseConnector = DatabaseConnectorFactory.getInstance().newInstance(queryContext, databaseConnectionManager, false);
76          databaseConnector.execute();
77          return createColumnDefinition41Packets(databaseName);
78      }
79      
80      private Collection<DatabasePacket> createColumnDefinition41Packets(final String databaseName) throws SQLException {
81          Collection<DatabasePacket> result = new LinkedList<>();
82          int characterSet = connectionSession.getAttributeMap().attr(MySQLConstants.CHARACTER_SET_ATTRIBUTE_KEY).get().getId();
83          while (databaseConnector.next()) {
84              String columnName = databaseConnector.getRowData().getCells().iterator().next().getData().toString();
85              result.add(new MySQLColumnDefinition41Packet(
86                      characterSet, databaseName, packet.getTable(), packet.getTable(), columnName, columnName, 100, MySQLBinaryColumnType.VARCHAR, 0, true));
87          }
88          result.add(new MySQLEofPacket(ServerStatusFlagCalculator.calculateFor(connectionSession)));
89          return result;
90      }
91      
92      @Override
93      public void close() throws SQLException {
94          databaseConnector.close();
95      }
96  }
