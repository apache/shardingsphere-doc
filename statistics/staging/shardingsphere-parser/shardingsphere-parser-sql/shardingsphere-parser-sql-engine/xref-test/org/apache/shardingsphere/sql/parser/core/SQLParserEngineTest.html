18  package org.apache.shardingsphere.sql.parser.core;
19  
20  import com.github.benmanes.caffeine.cache.CacheLoader;
21  import com.github.benmanes.caffeine.cache.Caffeine;
22  import com.github.benmanes.caffeine.cache.LoadingCache;
23  import org.apache.shardingsphere.sql.parser.core.database.parser.SQLParserExecutor;
24  import org.junit.jupiter.api.Test;
25  
26  import javax.annotation.ParametersAreNonnullByDefault;
27  
28  import static org.mockito.Mockito.mock;
29  import static org.mockito.Mockito.when;
30  import static org.mockito.Mockito.verify;
31  import static org.mockito.Mockito.times;
32  
33  class SQLParserEngineTest {
34      
35      @Test
36      void assertParse() {
37          SQLParserExecutor sqlParserExecutor = mock(SQLParserExecutor.class);
38          when(sqlParserExecutor.parse("")).thenReturn(mock(ParseASTNode.class));
39          LoadingCache<String, ParseASTNode> parseTreeCache = Caffeine.newBuilder().softValues()
40                  .initialCapacity(128).maximumSize(1024L).build(new CacheLoader<String, ParseASTNode>() {
41                      
42                      @ParametersAreNonnullByDefault
43                      @Override
44                      public ParseASTNode load(final String sql) {
45                          return sqlParserExecutor.parse(sql);
46                      }
47                  });
48          parseTreeCache.get("");
49          verify(sqlParserExecutor).parse("");
50          parseTreeCache.get("");
51          verify(sqlParserExecutor).parse("");
52          sqlParserExecutor.parse("");
53          verify(sqlParserExecutor, times(2)).parse("");
54      }
55  }
