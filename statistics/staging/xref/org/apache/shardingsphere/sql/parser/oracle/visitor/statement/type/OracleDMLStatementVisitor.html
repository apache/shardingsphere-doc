204     @Override
205     public ASTNode visitUpdate(final UpdateContext ctx) {
206         OracleUpdateStatement result = new OracleUpdateStatement();
207         result.setTable((TableSegment) visit(ctx.updateSpecification()));
208         if (null != ctx.alias()) {
209             result.getTable().setAlias((AliasSegment) visit(ctx.alias()));
210         }
211         result.setSetAssignment((SetAssignmentSegment) visit(ctx.updateSetClause()));
212         if (null != ctx.whereClause()) {
213             result.setWhere((WhereSegment) visit(ctx.whereClause()));
214         }
215         result.addParameterMarkerSegments(ctx.getParent() instanceof ExecuteContext ? getGlobalParameterMarkerSegments() : popAllStatementParameterMarkerSegments());
216         result.getVariableNames().addAll(getVariableNames());
217         return result;
218     }
219     
220     @Override
221     public ASTNode visitUpdateSpecification(final UpdateSpecificationContext ctx) {
222         if (null != ctx.dmlTableExprClause().dmlTableClause()) {
223             return visit(ctx.dmlTableExprClause().dmlTableClause());
224         }
225         if (null != ctx.dmlTableExprClause().dmlSubqueryClause()) {
226             SubquerySegment subquerySegment = (SubquerySegment) visit(ctx.dmlTableExprClause().dmlSubqueryClause());
227             return new SubqueryTableSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), subquerySegment);
228         }
229         SubquerySegment subquerySegment = (SubquerySegment) visit(ctx.dmlTableExprClause().tableCollectionExpr());
230         return new SubqueryTableSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), subquerySegment);
231     }
232     
233     @Override
234     public ASTNode visitUpdateSetClause(final UpdateSetClauseContext ctx) {
235         Collection<ColumnAssignmentSegment> assignments = new LinkedList<>();
236         if (null != ctx.updateSetColumnList()) {
237             for (UpdateSetColumnClauseContext each : ctx.updateSetColumnList().updateSetColumnClause()) {
238                 assignments.add((ColumnAssignmentSegment) visit(each));
239             }
240         } else {
241             assignments.add((ColumnAssignmentSegment) visit(ctx.updateSetValueClause()));
242         }
243         return new SetAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), assignments);
244     }
245     
246     @Override
247     public ASTNode visitUpdateSetColumnClause(final UpdateSetColumnClauseContext ctx) {
248         return 1 == ctx.columnName().size() ? createAssignmentSegmentFromSingleColumnAssignment(ctx) : createAssignmentSegmentFromMultiColumnAssignment(ctx);
249     }
250     
251     private ColumnAssignmentSegment createAssignmentSegmentFromSingleColumnAssignment(final UpdateSetColumnClauseContext ctx) {
252         ColumnSegment column = (ColumnSegment) visitColumnName(ctx.columnName(0));
253         List<ColumnSegment> columnSegments = new LinkedList<>();
254         columnSegments.add(column);
255         if (null != ctx.expr()) {
256             ExpressionSegment value = (ExpressionSegment) visit(ctx.expr());
257             return new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
258         }
259         if (null != ctx.selectSubquery()) {
260             SubquerySegment subquerySegment = new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(),
261                     (OracleSelectStatement) visit(ctx.selectSubquery()), getOriginalText(ctx.selectSubquery()));
262             SubqueryExpressionSegment value = new SubqueryExpressionSegment(subquerySegment);
263             ColumnAssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
264             result.getColumns().add(column);
265             return result;
266         }
267         CommonExpressionSegment value = new CommonExpressionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), ctx.DEFAULT().getText());
268         ColumnAssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
269         result.getColumns().add(column);
270         return result;
271     }
272     
273     private ColumnAssignmentSegment createAssignmentSegmentFromMultiColumnAssignment(final UpdateSetColumnClauseContext ctx) {
274         List<ColumnSegment> columnSegments = new LinkedList<>();
275         for (ColumnNameContext each : ctx.columnName()) {
276             columnSegments.add((ColumnSegment) visit(each));
277         }
278         SubquerySegment subquerySegment = new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(),
279                 (OracleSelectStatement) visit(ctx.selectSubquery()), getOriginalText(ctx.selectSubquery()));
280         SubqueryExpressionSegment value = new SubqueryExpressionSegment(subquerySegment);
281         return new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
282     }
283     
284     @Override
285     public ASTNode visitUpdateSetValueClause(final UpdateSetValueClauseContext ctx) {
286         ColumnSegment column = (ColumnSegment) visit(ctx.columnName());
287         List<ColumnSegment> columnSegments = new LinkedList<>();
288         columnSegments.add(column);
289         if (null != ctx.expr()) {
290             ExpressionSegment value = (ExpressionSegment) visit(ctx.expr());
291             ColumnAssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
292             result.getColumns().add(column);
293             return result;
294         } else {
295             SubquerySegment subquerySegment = new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(),
296                     (OracleSelectStatement) visit(ctx.selectSubquery()), getOriginalText(ctx.selectSubquery()));
297             SubqueryExpressionSegment value = new SubqueryExpressionSegment(subquerySegment);
298             ColumnAssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
299             result.getColumns().add(column);
300             return result;
301         }
302     }

304     @Override
305     public ASTNode visitInsert(final InsertContext ctx) {
306         // TODO :FIXME, since there is no segment for insertValuesClause, InsertStatement is created by sub rule.
307         OracleInsertStatement result = (OracleInsertStatement) (null == ctx.insertSingleTable() ? visit(ctx.insertMultiTable()) : visit(ctx.insertSingleTable()));
308         result.addParameterMarkerSegments(ctx.getParent() instanceof ExecuteContext ? getGlobalParameterMarkerSegments() : popAllStatementParameterMarkerSegments());
309         return result;
310     }

312     @Override
313     public ASTNode visitInsertSingleTable(final InsertSingleTableContext ctx) {
314         OracleInsertStatement result = (OracleInsertStatement) visit(ctx.insertIntoClause());
315         if (null != ctx.insertValuesClause()) {
316             result.getValues().addAll(createInsertValuesSegments(ctx.insertValuesClause().assignmentValues()));
317         }
318         if (null != ctx.selectSubquery()) {
319             OracleSelectStatement subquery = (OracleSelectStatement) visit(ctx.selectSubquery());
320             SubquerySegment subquerySegment = new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(), subquery,
321                     getOriginalText(ctx.selectSubquery()));
322             result.setInsertSelect(subquerySegment);
323         }
324         result.getVariableNames().addAll(getVariableNames());
325         return result;
326     }
327     
328     private Collection<InsertValuesSegment> createInsertValuesSegments(final AssignmentValuesContext ctx) {
329         Collection<InsertValuesSegment> result = new LinkedList<>();
330         result.add((InsertValuesSegment) visit(ctx));
331         return result;
332     }

334     @Override

819     private ASTNode createProjection(final SelectProjectionExprClauseContext ctx) {
820         AliasSegment alias = null == ctx.alias() ? null : (AliasSegment) visit(ctx.alias());
821         ASTNode projection = visit(ctx.expr());
822         if (projection instanceof AliasAvailable) {
823             ((AliasAvailable) projection).setAlias(alias);
824             return projection;
825         }
826         if (projection instanceof ComplexExpressionSegment) {
827             return createProjectionForComplexExpressionSegment(projection, alias);
828         }
829         if (projection instanceof SimpleExpressionSegment) {
830             return createProjectionForSimpleExpressionSegment(projection, alias, ctx);
831         }
832         if (projection instanceof ExpressionSegment) {
833             return createProjectionForExpressionSegment(projection, alias);
834         }
835         throw new UnsupportedOperationException("Unhandled case");
836     }

838     private ASTNode createProjectionForComplexExpressionSegment(final ASTNode projection, final AliasSegment alias) {
839         if (projection instanceof FunctionSegment) {
840             FunctionSegment segment = (FunctionSegment) projection;
841             ExpressionProjectionSegment result = new ExpressionProjectionSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getText(), segment);
842             result.setAlias(alias);

844         }
845         if (projection instanceof CommonExpressionSegment) {
846             CommonExpressionSegment segment = (CommonExpressionSegment) projection;
847             ExpressionProjectionSegment result = new ExpressionProjectionSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getText(), segment);
848             result.setAlias(alias);
849             return result;
850         }
851         if (projection instanceof XmlQueryAndExistsFunctionSegment || projection instanceof XmlPiFunctionSegment || projection instanceof XmlSerializeFunctionSegment
852                 || projection instanceof XmlElementFunctionSegment) {

854         }
855         throw new UnsupportedOperationException("Unsupported Complex Expression");

898         throw new UnsupportedOperationException("Unsupported Expression");
