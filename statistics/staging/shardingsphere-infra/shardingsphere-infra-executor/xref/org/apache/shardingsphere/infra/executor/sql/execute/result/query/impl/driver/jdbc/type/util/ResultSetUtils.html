20  import com.google.common.primitives.Ints;
21  import com.google.common.primitives.Longs;
22  import com.google.common.primitives.Shorts;
23  import lombok.AccessLevel;
24  import lombok.NoArgsConstructor;
25  import org.apache.shardingsphere.infra.executor.exception.UnsupportedDataTypeConversionException;
26  import org.apache.shardingsphere.infra.util.exception.ShardingSpherePreconditions;
27  
28  import java.math.BigDecimal;
29  import java.math.RoundingMode;
30  import java.net.MalformedURLException;
31  import java.net.URL;
32  import java.sql.SQLFeatureNotSupportedException;
33  import java.sql.Time;
34  import java.sql.Timestamp;
35  import java.time.LocalDate;
36  import java.time.LocalDateTime;
37  import java.time.LocalTime;
38  import java.time.OffsetDateTime;
39  import java.time.ZoneId;
40  import java.util.Date;
41  
42  /**
43   * Result set utility class.
44   */
45  @NoArgsConstructor(access = AccessLevel.PRIVATE)
46  public final class ResultSetUtils {
47      
48      /**
49       * Convert value via expected class type.
50       *
51       * @param value original value
52       * @param convertType expected class type
53       * @return converted value
54       * @throws SQLFeatureNotSupportedException SQL feature not supported exception
55       */
56      public static Object convertValue(final Object value, final Class<?> convertType) throws SQLFeatureNotSupportedException {
57          ShardingSpherePreconditions.checkState(null != convertType, () -> new SQLFeatureNotSupportedException("Type can not be null"));
58          if (null == value) {
59              return convertNullValue(convertType);
60          }
61          if (value.getClass() == convertType) {
62              return value;
63          }
64          if (value instanceof LocalDateTime) {
65              return convertLocalDateTimeValue((LocalDateTime) value, convertType);
66          }
67          if (value instanceof Timestamp) {
68              return convertTimestampValue((Timestamp) value, convertType);
69          }
70          if (URL.class.equals(convertType)) {
71              return convertURL(value);
72          }
73          if (value instanceof Number) {
74              return convertNumberValue(value, convertType);
75          }
76          if (value instanceof Date) {
77              return convertDateValue((Date) value, convertType);
78          }
79          if (value instanceof byte[]) {
80              return convertByteArrayValue((byte[]) value, convertType);
81          }
82          if (boolean.class.equals(convertType)) {
83              return convertBooleanValue(value);
84          }
85          if (String.class.equals(convertType)) {
86              return value.toString();
87          }
88          try {
89              return convertType.cast(value);
90          } catch (final ClassCastException ignored) {
91              throw new SQLFeatureNotSupportedException("getObject with type");
92          }
93      }
94      
95      private static Object convertURL(final Object value) {
96          try {
97              return new URL(value.toString());
98          } catch (final MalformedURLException ignored) {
99              throw new UnsupportedDataTypeConversionException(URL.class, value);
100         }
101     }
102     
103     /**
104      * Convert object to BigDecimal.
105      *
106      * @param value current db object
107      * @param needScale need scale
108      * @param scale scale size
109      * @return big decimal
110      * @throws UnsupportedDataTypeConversionException unsupported data type conversion exception
111      */
112     public static Object convertBigDecimalValue(final Object value, final boolean needScale, final int scale) {
113         if (null == value) {
114             return convertNullValue(BigDecimal.class);
115         }
116         if (BigDecimal.class == value.getClass()) {
117             return adjustBigDecimalResult((BigDecimal) value, needScale, scale);
118         }
119         if (value instanceof Number || value instanceof String) {
120             BigDecimal bigDecimal = new BigDecimal(value.toString());
121             return adjustBigDecimalResult(bigDecimal, needScale, scale);
122         }
123         throw new UnsupportedDataTypeConversionException(BigDecimal.class, value);
124     }
125     
126     private static BigDecimal adjustBigDecimalResult(final BigDecimal value, final boolean needScale, final int scale) {
127         if (needScale) {
128             try {
129                 return value.setScale(scale);
130             } catch (final ArithmeticException ex) {
131                 return value.setScale(scale, RoundingMode.HALF_UP);
132             }
133         }
134         return value;
135     }
136     
137     private static Object convertLocalDateTimeValue(final LocalDateTime value, final Class<?> convertType) {
138         if (Timestamp.class.equals(convertType)) {
139             return Timestamp.valueOf(value);
140         }
141         if (String.class.equals(convertType)) {
142             return value.toString();
143         }
144         return value;
145     }
146     
147     private static Object convertTimestampValue(final Timestamp value, final Class<?> convertType) {
148         if (LocalDateTime.class.equals(convertType)) {
149             return value.toInstant().atZone(ZoneId.systemDefault()).toLocalDateTime();
150         }
151         if (LocalDate.class.equals(convertType)) {
152             return value.toInstant().atZone(ZoneId.systemDefault()).toLocalDate();
153         }
154         if (LocalTime.class.equals(convertType)) {
155             return value.toInstant().atZone(ZoneId.systemDefault()).toLocalTime();
156         }
157         if (OffsetDateTime.class.equals(convertType)) {
158             return value.toInstant().atZone(ZoneId.systemDefault()).toOffsetDateTime();
159         }
160         if (String.class.equals(convertType)) {
161             return value.toString();
162         }
163         return value;
164     }
165     
166     private static Object convertNullValue(final Class<?> convertType) {
167         switch (convertType.getName()) {
168             case "boolean":
169                 return false;
170             case "byte":
171                 return (byte) 0;
172             case "short":
173                 return (short) 0;
174             case "int":
175                 return 0;
176             case "long":
177                 return 0L;
178             case "float":
179                 return 0.0F;
180             case "double":
181                 return 0.0D;
182             default:
183                 return null;
184         }
185     }
186     
187     private static Object convertNumberValue(final Object value, final Class<?> convertType) {
188         Number number = (Number) value;
189         switch (convertType.getName()) {
190             case "boolean":
191                 return longToBoolean(number.longValue());
192             case "byte":
193             case "java.lang.Byte":
194                 return number.byteValue();
195             case "short":
196             case "java.lang.Short":
197                 return number.shortValue();
198             case "int":
199             case "java.lang.Integer":
200                 return number.intValue();
201             case "long":
202             case "java.lang.Long":
203                 return number.longValue();
204             case "double":
205             case "java.lang.Double":
206                 return number.doubleValue();
207             case "float":
208             case "java.lang.Float":
209                 return number.floatValue();
210             case "java.math.BigDecimal":
211                 return new BigDecimal(number.toString());
212             case "java.lang.Object":
213                 return value;
214             case "java.lang.String":
215                 return value.toString();
216             default:
217                 throw new UnsupportedDataTypeConversionException(convertType, value);
218         }
219     }
220     
221     private static Object convertDateValue(final Date value, final Class<?> convertType) {
222         switch (convertType.getName()) {
223             case "java.sql.Date":
224                 return new java.sql.Date(value.getTime());
225             case "java.sql.Time":
226                 return new Time(value.getTime());
227             case "java.sql.Timestamp":
228                 return new Timestamp(value.getTime());
229             case "java.lang.String":
230                 return value.toString();
231             default:
232                 throw new UnsupportedDataTypeConversionException(convertType, value);
233         }
234     }
235     
236     private static Object convertByteArrayValue(final byte[] value, final Class<?> convertType) {
237         switch (value.length) {
238             case 1:
239                 return convertNumberValue(value[0], convertType);
240             case Shorts.BYTES:
241                 return convertNumberValue(Shorts.fromByteArray(value), convertType);
242             case Ints.BYTES:
243                 return convertNumberValue(Ints.fromByteArray(value), convertType);
244             case Longs.BYTES:
245                 return convertNumberValue(Longs.fromByteArray(value), convertType);
246             default:
247                 return value;
248         }
249     }
250     
251     private static Object convertBooleanValue(final Object value) {
252         if (value instanceof Boolean) {
253             return value;
254         }
255         String stringVal = value.toString();
256         if (stringVal.length() > 0) {
257             int firstChar = Character.toLowerCase(stringVal.charAt(0));
258             return 't' == firstChar || 'y' == firstChar || '1' == firstChar || "-1".equals(stringVal);
259         }
260         return false;
261     }
262     
263     private static Boolean longToBoolean(final long longVal) {
264         return -1 == longVal || longVal > 0;
265     }
