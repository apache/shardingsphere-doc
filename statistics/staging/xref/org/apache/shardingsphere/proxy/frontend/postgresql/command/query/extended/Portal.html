64  /**
65   * PostgreSQL portal.
66   */
67  public final class Portal {
68      
69      @Getter
70      private final String name;
71      
72      @Getter
73      private final SQLStatement sqlStatement;
74      
75      private final List<PostgreSQLValueFormat> resultFormats;
76      
77      private final ProxyBackendHandler proxyBackendHandler;
78      
79      private final ProxyDatabaseConnectionManager databaseConnectionManager;
80      
81      private ResponseHeader responseHeader;
82      
83      public Portal(final String name, final PostgreSQLServerPreparedStatement preparedStatement, final List<Object> params, final List<PostgreSQLValueFormat> resultFormats,
84                    final ProxyDatabaseConnectionManager databaseConnectionManager) throws SQLException {
85          this.name = name;
86          this.sqlStatement = preparedStatement.getSqlStatementContext().getSqlStatement();
87          this.resultFormats = resultFormats;
88          this.databaseConnectionManager = databaseConnectionManager;
89          String databaseName = databaseConnectionManager.getConnectionSession().getDefaultDatabaseName();
90          SQLStatementContext sqlStatementContext = preparedStatement.getSqlStatementContext();
91          if (sqlStatementContext instanceof ParameterAware) {
92              ((ParameterAware) sqlStatementContext).setUpParameters(params);
93          }
94          DatabaseType protocolType = ProxyContext.getInstance().getContextManager().getDatabase(databaseName).getProtocolType();
95          QueryContext queryContext = new QueryContext(sqlStatementContext, preparedStatement.getSql(), params, preparedStatement.getHintValueContext(), true);
96          databaseConnectionManager.getConnectionSession().setQueryContext(queryContext);
97          proxyBackendHandler = ProxyBackendHandlerFactory.newInstance(protocolType, queryContext, databaseConnectionManager.getConnectionSession(), true);
98      }
99      
100     /**
101      * Do bind.
102      * 
103      * @throws SQLException SQL exception
104      */
105     public void bind() throws SQLException {
106         responseHeader = proxyBackendHandler.execute();
107     }
108     
109     /**
110      * Describe portal.
111      *
112      * @return portal description packet
113      * @throws IllegalStateException illegal state exception
114      */
115     public PostgreSQLPacket describe() {
116         if (responseHeader instanceof QueryResponseHeader) {
117             return createRowDescriptionPacket((QueryResponseHeader) responseHeader);
118         }
119         if (responseHeader instanceof UpdateResponseHeader) {
120             return PostgreSQLNoDataPacket.getInstance();
121         }
122         throw new IllegalStateException(String.format("Can not describe portal `%s` before bind", name));
123     }
124     
125     private PostgreSQLRowDescriptionPacket createRowDescriptionPacket(final QueryResponseHeader queryResponseHeader) {
126         return new PostgreSQLRowDescriptionPacket(createColumnDescriptions(queryResponseHeader));
127     }
128     
129     private Collection<PostgreSQLColumnDescription> createColumnDescriptions(final QueryResponseHeader queryResponseHeader) {
130         Collection<PostgreSQLColumnDescription> result = new LinkedList<>();
131         int columnIndex = 0;
132         for (QueryHeader each : queryResponseHeader.getQueryHeaders()) {
133             PostgreSQLValueFormat valueFormat = determineValueFormat(columnIndex);
134             result.add(new PostgreSQLColumnDescription(each.getColumnLabel(), ++columnIndex, each.getColumnType(), each.getColumnLength(), each.getColumnTypeName(), valueFormat.getCode()));
135         }
136         return result;
137     }
138     
139     /**
140      * Execute portal.
141      *
142      * @param maxRows max rows of query result
143      * @return execute result
144      * @throws SQLException SQL exception
145      */
146     public List<DatabasePacket> execute(final int maxRows) throws SQLException {
147         int fetchSize = maxRows > 0 ? maxRows : Integer.MAX_VALUE;
148         List<DatabasePacket> result = new LinkedList<>();
149         for (int i = 0; i < fetchSize && hasNext(); i++) {
150             result.add(nextPacket());
151         }
152         if (responseHeader instanceof UpdateResponseHeader && sqlStatement instanceof SetStatement) {
153             result.addAll(createParameterStatusResponse((SetStatement) sqlStatement));
154             return result;
155         }
156         result.add(createExecutionCompletedPacket(maxRows > 0 && maxRows == result.size(), result.size()));
157         return result;
158     }
159     
160     private List<PostgreSQLPacket> createParameterStatusResponse(final SetStatement sqlStatement) {
161         List<PostgreSQLPacket> result = new ArrayList<>(2);
162         result.add(new PostgreSQLCommandCompletePacket("SET", 0L));
163         for (VariableAssignSegment each : sqlStatement.getVariableAssigns()) {
164             result.add(new PostgreSQLParameterStatusPacket(each.getVariable().getVariable(), IdentifierValue.getQuotedContent(each.getAssignValue())));
165         }
166         return result;
167     }
168     
169     private boolean hasNext() throws SQLException {
170         return proxyBackendHandler.next();
171     }
172     
173     private PostgreSQLPacket nextPacket() throws SQLException {
174         return new PostgreSQLDataRowPacket(getData(proxyBackendHandler.getRowData()));
175     }
176     
177     private List<Object> getData(final QueryResponseRow queryResponseRow) {
178         Collection<QueryResponseCell> cells = queryResponseRow.getCells();
179         List<Object> result = new ArrayList<>(cells.size());
180         List<QueryResponseCell> columns = new ArrayList<>(cells);
181         for (int i = 0; i < columns.size(); i++) {
182             PostgreSQLValueFormat format = determineValueFormat(i);
183             result.add(PostgreSQLValueFormat.BINARY == format ? createBinaryCell(columns.get(i)) : getCellData(columns.get(i)));
184         }
185         return result;
186     }
187     
188     private PostgreSQLValueFormat determineValueFormat(final int columnIndex) {
189         return resultFormats.isEmpty() ? PostgreSQLValueFormat.TEXT : resultFormats.get(columnIndex % resultFormats.size());
190     }
191     
192     private BinaryCell createBinaryCell(final QueryResponseCell cell) {
193         return new BinaryCell(PostgreSQLColumnType.valueOfJDBCType(cell.getJdbcType(), cell.getColumnTypeName().orElse(null)), getCellData(cell));
194     }
195     
196     private Object getCellData(final QueryResponseCell cell) {
197         if (PostgreSQLColumnType.isBit(cell.getJdbcType(), cell.getColumnTypeName().orElse(null))) {
198             return PostgreSQLTextBitUtils.getTextValue(cell.getData());
199         }
200         if (PostgreSQLColumnType.isBool(cell.getJdbcType(), cell.getColumnTypeName().orElse(null))) {
201             return PostgreSQLTextBoolUtils.getTextValue(cell.getData());
202         }
203         return cell.getData();
204     }
205     
206     private PostgreSQLIdentifierPacket createExecutionCompletedPacket(final boolean isSuspended, final int fetchedRows) {
207         if (isSuspended) {
208             suspendPortal();
209             return new PostgreSQLPortalSuspendedPacket();
210         }
211         if (getSqlStatement() instanceof EmptyStatement) {
212             return new PostgreSQLEmptyQueryResponsePacket();
213         }
214         String sqlCommand = PostgreSQLCommand.valueOf(getSqlStatement().getClass()).map(PostgreSQLCommand::getTag).orElse("");
215         return new PostgreSQLCommandCompletePacket(sqlCommand, Math.max(fetchedRows, getUpdateCount()));
216     }
217     
218     private void suspendPortal() {
219         databaseConnectionManager.markResourceInUse(proxyBackendHandler);
220     }
221     
222     private long getUpdateCount() {
223         return responseHeader instanceof UpdateResponseHeader ? ((UpdateResponseHeader) responseHeader).getUpdateCount() : 0L;
224     }
225     
226     /**
227      * Close portal.
228      *
229      * @throws SQLException SQL exception
230      */
231     public void close() throws SQLException {
232         databaseConnectionManager.unmarkResourceInUse(proxyBackendHandler);
233         proxyBackendHandler.close();
234     }
