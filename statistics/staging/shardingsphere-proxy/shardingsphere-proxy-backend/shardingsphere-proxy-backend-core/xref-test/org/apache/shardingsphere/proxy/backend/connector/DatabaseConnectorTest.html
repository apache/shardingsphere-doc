136     @Test
137     void assertExecuteFederationAndClose() throws SQLException {
138         SQLStatementContext<?> sqlStatementContext = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
139         when(sqlStatementContext.getDatabaseType()).thenReturn(new MySQLDatabaseType());
140         DatabaseConnector engine =
141                 DatabaseConnectorFactory.getInstance().newInstance(new QueryContext(sqlStatementContext, "schemaName", Collections.emptyList()), backendConnection, true);
142         when(backendConnection.getConnectionSession().getStatementManager()).thenReturn(new JDBCBackendStatement());
143         SQLFederationExecutor federationExecutor = mock(SQLFederationExecutor.class);
144         when(SystemSchemaUtils.containsSystemSchema(any(DatabaseType.class), any(), any(ShardingSphereDatabase.class))).thenReturn(true);
145         try (MockedStatic<TypedSPILoader> typedSPILoader = mockStatic(TypedSPILoader.class)) {
146             when(federationExecutor.executeQuery(any(DriverExecutionPrepareEngine.class), any(ProxyJDBCExecutorCallback.class), any(SQLFederationExecutorContext.class))).thenReturn(resultSet);
147             when(resultSet.getMetaData().getColumnCount()).thenReturn(1);
148             when(resultSet.next()).thenReturn(true, false);
149             when(resultSet.getObject(1)).thenReturn(Integer.MAX_VALUE);
150             typedSPILoader.when(() -> TypedSPILoader.getService(SQLFederationExecutor.class, "NONE")).thenReturn(federationExecutor);
151             typedSPILoader.when(() -> TypedSPILoader.getService(QueryHeaderBuilder.class, "H2")).thenReturn(new QueryHeaderBuilderFixture());
152             engine.execute();
153         }
154         assertTrue(engine.next());
155         QueryResponseRow actualRow = engine.getRowData();
156         assertThat(actualRow.getCells().get(0).getJdbcType(), is(Types.INTEGER));
157         assertThat(actualRow.getCells().get(0).getData(), is(Integer.MAX_VALUE));
158         assertFalse(engine.next());
159         engine.close();
160         verify(federationExecutor).close();
161     }

163     @Test
164     void assertBinaryProtocolQueryHeader() throws SQLException, NoSuchFieldException, IllegalAccessException {
165         SQLStatementContext<?> sqlStatementContext = mock(SQLStatementContext.class, RETURNS_DEEP_STUBS);
166         when(sqlStatementContext.getTablesContext().getSchemaNames()).thenReturn(Collections.emptyList());
167         DatabaseConnector engine = DatabaseConnectorFactory.getInstance().newInstance(new QueryContext(sqlStatementContext, "schemaName", Collections.emptyList()), backendConnection, true);
168         assertNotNull(engine);
169         assertThat(engine, instanceOf(DatabaseConnector.class));
170         Field queryHeadersField = DatabaseConnector.class.getDeclaredField("queryHeaders");
171         ShardingSphereDatabase database = createDatabaseMetaData();
172         MemberAccessor accessor = Plugins.getMemberAccessor();
173         try (MockedStatic<TypedSPILoader> typedSPILoader = mockStatic(TypedSPILoader.class)) {
174             typedSPILoader.when(() -> TypedSPILoader.getService(QueryHeaderBuilder.class, "MySQL")).thenReturn(new QueryHeaderBuilderFixture());
175             accessor.set(queryHeadersField, engine, Collections.singletonList(new QueryHeaderBuilderEngine(new MySQLDatabaseType()).build(createQueryResultMetaData(), database, 1)));
176             Field mergedResultField = DatabaseConnector.class.getDeclaredField("mergedResult");
177             accessor.set(mergedResultField, engine, new MemoryMergedResult<ShardingSphereRule>(null, null, null, Collections.emptyList()) {
178                 
179                 @Override
180                 protected List<MemoryQueryResultRow> init(final ShardingSphereRule rule, final ShardingSphereSchema schema,
181                                                           final SQLStatementContext<?> sqlStatementContext, final List<QueryResult> queryResults) {
182                     return Collections.singletonList(mock(MemoryQueryResultRow.class));
183                 }
184             });
185             Exception ex = null;
186             try {
187                 engine.getRowData();
188             } catch (final SQLException | IndexOutOfBoundsException e) {
189                 ex = e;
190             } finally {
191                 assertFalse(ex instanceof IndexOutOfBoundsException);
192             }
193         }
194     }

235     void assertCloseCorrectly() throws SQLException {
236         SQLStatementContext<?> sqlStatementContext = mock(SQLStatementContext.class, RETURNS_DEEP_STUBS);
237         when(sqlStatementContext.getTablesContext().getSchemaNames()).thenReturn(Collections.emptyList());
238         DatabaseConnector engine = DatabaseConnectorFactory.getInstance().newInstance(new QueryContext(sqlStatementContext, "schemaName", Collections.emptyList()), backendConnection, false);
239         Collection<ResultSet> cachedResultSets = getField(engine, "cachedResultSets");
240         cachedResultSets.add(resultSet);
241         Collection<Statement> cachedStatements = getField(engine, "cachedStatements");
242         cachedStatements.add(statement);
243         engine.close();

246         verify(statement).close();
247         assertTrue(cachedResultSets.isEmpty());
248         assertTrue(cachedStatements.isEmpty());

251     @Test
252     void assertCloseResultSetsWithExceptionThrown() throws SQLException {
253         SQLStatementContext<?> sqlStatementContext = mock(SQLStatementContext.class, RETURNS_DEEP_STUBS);
254         when(sqlStatementContext.getTablesContext().getSchemaNames()).thenReturn(Collections.emptyList());
255         DatabaseConnector engine = DatabaseConnectorFactory.getInstance().newInstance(new QueryContext(sqlStatementContext, "schemaName", Collections.emptyList()), backendConnection, false);
256         Collection<ResultSet> cachedResultSets = getField(engine, "cachedResultSets");
257         SQLException sqlExceptionByResultSet = new SQLException("ResultSet");
258         doThrow(sqlExceptionByResultSet).when(resultSet).close();
259         cachedResultSets.add(resultSet);
260         Collection<Statement> cachedStatements = getField(engine, "cachedStatements");
261         SQLException sqlExceptionByStatement = new SQLException("Statement");
262         doThrow(sqlExceptionByStatement).when(statement).close();
263         cachedStatements.add(statement);
264         SQLException actual = null;
265         try {
266             engine.close();
267         } catch (final SQLException ex) {
268             actual = ex;
269         }
270         verify(resultSet).close();
271         verify(statement).close();
272         assertTrue(cachedResultSets.isEmpty());
273         assertTrue(cachedStatements.isEmpty());
274         assertNotNull(actual);
275         assertThat(actual.getNextException(), is(sqlExceptionByResultSet));
276         assertThat(actual.getNextException().getNextException(), is(sqlExceptionByStatement));
277     }
