56  /**
57   * Default importer.
58   */
59  @Slf4j
60  public final class DataSourceImporter extends AbstractLifecycleExecutor implements Importer {
61      
62      private static final DataRecordMerger MERGER = new DataRecordMerger();
63      
64      @Getter(AccessLevel.PROTECTED)
65      private final ImporterConfiguration importerConfig;
66      
67      private final PipelineDataSourceManager dataSourceManager;
68      
69      private final PipelineSQLBuilder pipelineSqlBuilder;
70      
71      private final PipelineChannel channel;
72      
73      private final PipelineJobProgressListener jobProgressListener;
74      
75      private final JobRateLimitAlgorithm rateLimitAlgorithm;
76      
77      private volatile Statement batchInsertStatement;
78      
79      private volatile Statement updateStatement;
80      
81      private volatile Statement batchDeleteStatement;
82      
83      public DataSourceImporter(final ImporterConfiguration importerConfig, final ImporterConnector importerConnector, final PipelineChannel channel,
84                                final PipelineJobProgressListener jobProgressListener) {
85          this.importerConfig = importerConfig;
86          rateLimitAlgorithm = importerConfig.getRateLimitAlgorithm();
87          this.dataSourceManager = (PipelineDataSourceManager) importerConnector.getConnector();
88          this.channel = channel;
89          pipelineSqlBuilder = PipelineTypedSPILoader.getDatabaseTypedService(PipelineSQLBuilder.class, importerConfig.getDataSourceConfig().getDatabaseType().getType());
90          this.jobProgressListener = jobProgressListener;
91      }
92      
93      @Override
94      protected void runBlocking() {
95          int batchSize = importerConfig.getBatchSize() * 2;
96          while (isRunning()) {
97              List<Record> records = channel.fetchRecords(batchSize, 3);
98              if (null != records && !records.isEmpty()) {
99                  PipelineJobProgressUpdatedParameter updatedParam = flush(dataSourceManager.getDataSource(importerConfig.getDataSourceConfig()), records);
100                 channel.ack(records);
101                 jobProgressListener.onProgressUpdated(updatedParam);
102                 if (FinishedRecord.class.equals(records.get(records.size() - 1).getClass())) {
103                     break;
104                 }
105             }
106         }
107     }
108     
109     private PipelineJobProgressUpdatedParameter flush(final DataSource dataSource, final List<Record> buffer) {
110         List<DataRecord> dataRecords = buffer.stream().filter(each -> each instanceof DataRecord).map(each -> (DataRecord) each).collect(Collectors.toList());
111         int insertRecordNumber = 0;
112         if (dataRecords.isEmpty()) {
113             return new PipelineJobProgressUpdatedParameter(0);
114         }
115         for (DataRecord each : dataRecords) {
116             if (IngestDataChangeType.INSERT.equals(each.getType())) {
117                 insertRecordNumber++;
118             }
119         }
120         List<GroupedDataRecord> result = MERGER.group(dataRecords);
121         for (GroupedDataRecord each : result) {
122             flushInternal(dataSource, each.getDeleteDataRecords());
123             flushInternal(dataSource, each.getInsertDataRecords());
124             flushInternal(dataSource, each.getUpdateDataRecords());
125         }
126         return new PipelineJobProgressUpdatedParameter(insertRecordNumber);
127     }
128     
129     private void flushInternal(final DataSource dataSource, final List<DataRecord> buffer) {
130         if (null == buffer || buffer.isEmpty()) {
131             return;
132         }
133         boolean success = tryFlush(dataSource, buffer);
134         ShardingSpherePreconditions.checkState(!isRunning() || success, PipelineImporterJobWriteException::new);
135     }
136     
137     @SneakyThrows(InterruptedException.class)
138     private boolean tryFlush(final DataSource dataSource, final List<DataRecord> buffer) {
139         for (int i = 0; isRunning() && i <= importerConfig.getRetryTimes(); i++) {
140             try {
141                 doFlush(dataSource, buffer);
142                 return true;
143             } catch (final SQLException ex) {
144                 log.error("flush failed {}/{} times.", i, importerConfig.getRetryTimes(), ex);
145                 Thread.sleep(Math.min(5 * 60 * 1000L, 1000L << i));
146             }
147         }
148         return false;
149     }
150     
151     private void doFlush(final DataSource dataSource, final List<DataRecord> buffer) throws SQLException {
152         try (Connection connection = dataSource.getConnection()) {
153             connection.setAutoCommit(false);
154             switch (buffer.get(0).getType()) {
155                 case IngestDataChangeType.INSERT:
156                     if (null != rateLimitAlgorithm) {
157                         rateLimitAlgorithm.intercept(JobOperationType.INSERT, 1);
158                     }
159                     executeBatchInsert(connection, buffer);
160                     break;
161                 case IngestDataChangeType.UPDATE:
162                     if (null != rateLimitAlgorithm) {
163                         rateLimitAlgorithm.intercept(JobOperationType.UPDATE, 1);
164                     }
165                     executeUpdate(connection, buffer);
166                     break;
167                 case IngestDataChangeType.DELETE:
168                     if (null != rateLimitAlgorithm) {
169                         rateLimitAlgorithm.intercept(JobOperationType.DELETE, 1);
170                     }
171                     executeBatchDelete(connection, buffer);
172                     break;
173                 default:
174                     break;
175             }
176             connection.commit();
177         }
178     }
179     
180     private void executeBatchInsert(final Connection connection, final List<DataRecord> dataRecords) throws SQLException {
181         DataRecord dataRecord = dataRecords.get(0);
182         String insertSql = pipelineSqlBuilder.buildInsertSQL(getSchemaName(dataRecord.getTableName()), dataRecord);
183         try (PreparedStatement preparedStatement = connection.prepareStatement(insertSql)) {
184             batchInsertStatement = preparedStatement;
185             preparedStatement.setQueryTimeout(30);
186             for (DataRecord each : dataRecords) {
187                 for (int i = 0; i < each.getColumnCount(); i++) {
188                     preparedStatement.setObject(i + 1, each.getColumn(i).getValue());
189                 }
190                 preparedStatement.addBatch();
191             }
192             preparedStatement.executeBatch();
193         } finally {
194             batchInsertStatement = null;
195         }
196     }
197     
198     private String getSchemaName(final String logicTableName) {
199         return getImporterConfig().getSchemaName(new LogicTableName(logicTableName));
200     }
201     
202     private void executeUpdate(final Connection connection, final List<DataRecord> dataRecords) throws SQLException {
203         for (DataRecord each : dataRecords) {
204             executeUpdate(connection, each);
205         }
206     }
207     
208     private void executeUpdate(final Connection connection, final DataRecord record) throws SQLException {
209         Set<String> shardingColumns = importerConfig.getShardingColumns(record.getTableName());
210         List<Column> conditionColumns = RecordUtils.extractConditionColumns(record, shardingColumns);
211         List<Column> updatedColumns = pipelineSqlBuilder.extractUpdatedColumns(record);
212         String updateSql = pipelineSqlBuilder.buildUpdateSQL(getSchemaName(record.getTableName()), record, conditionColumns);
213         try (PreparedStatement preparedStatement = connection.prepareStatement(updateSql)) {
214             updateStatement = preparedStatement;
215             for (int i = 0; i < updatedColumns.size(); i++) {
216                 preparedStatement.setObject(i + 1, updatedColumns.get(i).getValue());
217             }
218             for (int i = 0; i < conditionColumns.size(); i++) {
219                 Column keyColumn = conditionColumns.get(i);
220                 preparedStatement.setObject(updatedColumns.size() + i + 1, (keyColumn.isUniqueKey() && keyColumn.isUpdated()) ? keyColumn.getOldValue() : keyColumn.getValue());
221             }
222             int updateCount = preparedStatement.executeUpdate();
223             if (1 != updateCount) {
224                 log.warn("executeUpdate failed, updateCount={}, updateSql={}, updatedColumns={}, conditionColumns={}", updateCount, updateSql, updatedColumns, conditionColumns);
225             }
226         } finally {
227             updateStatement = null;
228         }
229     }
230     
231     private void executeBatchDelete(final Connection connection, final List<DataRecord> dataRecords) throws SQLException {
232         DataRecord dataRecord = dataRecords.get(0);
233         List<Column> conditionColumns = RecordUtils.extractConditionColumns(dataRecord, importerConfig.getShardingColumns(dataRecord.getTableName()));
234         String deleteSQL = pipelineSqlBuilder.buildDeleteSQL(getSchemaName(dataRecord.getTableName()), dataRecord, conditionColumns);
235         try (PreparedStatement preparedStatement = connection.prepareStatement(deleteSQL)) {
236             batchDeleteStatement = preparedStatement;
237             preparedStatement.setQueryTimeout(30);
238             for (DataRecord each : dataRecords) {
239                 conditionColumns = RecordUtils.extractConditionColumns(each, importerConfig.getShardingColumns(each.getTableName()));
240                 for (int i = 0; i < conditionColumns.size(); i++) {
241                     preparedStatement.setObject(i + 1, conditionColumns.get(i).getValue());
242                 }
243                 preparedStatement.addBatch();
244             }
245             preparedStatement.executeBatch();
246         } finally {
247             batchDeleteStatement = null;
248         }
249     }
250     
251     @Override
252     protected void doStop() throws SQLException {
253         cancelStatement(batchInsertStatement);
254         cancelStatement(updateStatement);
255         cancelStatement(batchDeleteStatement);
256     }
