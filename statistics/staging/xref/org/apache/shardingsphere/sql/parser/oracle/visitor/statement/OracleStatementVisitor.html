176     @Override
177     public final ASTNode visitLiterals(final LiteralsContext ctx) {
178         if (null != ctx.stringLiterals()) {
179             return visit(ctx.stringLiterals());
180         }
181         if (null != ctx.numberLiterals()) {
182             return visit(ctx.numberLiterals());
183         }
184         if (null != ctx.hexadecimalLiterals()) {
185             return visit(ctx.hexadecimalLiterals());
186         }
187         if (null != ctx.bitValueLiterals()) {
188             return visit(ctx.bitValueLiterals());
189         }
190         if (null != ctx.booleanLiterals()) {
191             return visit(ctx.booleanLiterals());
192         }
193         if (null != ctx.nullValueLiterals()) {
194             return visit(ctx.nullValueLiterals());
195         }
196         if (null != ctx.dateTimeLiterals()) {
197             return visit(ctx.dateTimeLiterals());
198         }
199         if (null != ctx.intervalLiterals()) {
200             return visit(ctx.intervalLiterals());
201         }
202         throw new IllegalStateException("Literals must have string, number, dateTime, hex, bit, interval, boolean or null.");
203     }

339     @Override
340     public final ASTNode visitExpr(final ExprContext ctx) {
341         if (null != ctx.booleanPrimary()) {
342             return visit(ctx.booleanPrimary());
343         }
344         if (null != ctx.LP_()) {
345             return visit(ctx.expr(0));
346         }
347         if (null != ctx.andOperator()) {
348             return createBinaryOperationExpression(ctx, ctx.andOperator().getText());
349         }
350         if (null != ctx.orOperator()) {
351             return createBinaryOperationExpression(ctx, ctx.orOperator().getText());
352         }
353         if (null != ctx.datetimeExpr()) {
354             return createDatetimeExpression(ctx, ctx.datetimeExpr());
355         }
356         return new NotExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), (ExpressionSegment) visit(ctx.expr(0)), false);
357     }
358     
359     private ASTNode createDatetimeExpression(final ExprContext ctx, final DatetimeExprContext datetimeExpr) {
360         ExpressionSegment left = (ExpressionSegment) visit(ctx.expr(0));
361         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
362         if (null != datetimeExpr.expr()) {
363             ExpressionSegment right = new ExpressionProjectionSegment(datetimeExpr.getStart().getStartIndex(),
364                     datetimeExpr.getStop().getStopIndex(), datetimeExpr.getText(), (ExpressionSegment) visit(datetimeExpr.expr()));
365             return new DatetimeExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, text);
366         }
367         return new DatetimeExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, text);
368     }
369     
370     private ASTNode createBinaryOperationExpression(final ExprContext ctx, final String operator) {
371         ExpressionSegment left = (ExpressionSegment) visit(ctx.expr(0));
372         ExpressionSegment right = (ExpressionSegment) visit(ctx.expr(1));
373         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
374         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
375     }
376     
377     @Override
378     public final ASTNode visitBooleanPrimary(final BooleanPrimaryContext ctx) {
379         if (null == ctx.IS()) {
380             return null == ctx.comparisonOperator() && null == ctx.SAFE_EQ_() ? visit(ctx.predicate()) : createCompareSegment(ctx);
381         }
382         String rightText = "";
383         if (null != ctx.NOT()) {
384             rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(ctx.NOT().getSymbol().getStartIndex(), ctx.NOT().getSymbol().getStopIndex()))).concat(" ");
385         }
386         Token operatorToken = null;
387         if (null != ctx.NULL()) {
388             operatorToken = ctx.NULL().getSymbol();
389         }
390         if (null != ctx.TRUE()) {
391             operatorToken = ctx.TRUE().getSymbol();
392         }
393         if (null != ctx.FALSE()) {
394             operatorToken = ctx.FALSE().getSymbol();
395         }
396         int startIndex = null == operatorToken ? ctx.IS().getSymbol().getStopIndex() + 2 : operatorToken.getStartIndex();
397         rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(startIndex, ctx.stop.getStopIndex())));
398         ExpressionSegment right = new LiteralExpressionSegment(ctx.IS().getSymbol().getStopIndex() + 2, ctx.stop.getStopIndex(), rightText);
399         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
400         ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
401         String operator = "IS";
402         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
403     }
404     
405     private ASTNode createCompareSegment(final BooleanPrimaryContext ctx) {
406         ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
407         String operator = null == ctx.SAFE_EQ_() ? ctx.comparisonOperator().getText() : ctx.SAFE_EQ_().getText();
408         ExpressionSegment right = null != ctx.predicate() ? (ExpressionSegment) visit(ctx.predicate()) : (ExpressionSegment) visit(ctx.subquery());
409         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
410         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
411     }

413     @Override
414     public final ASTNode visitPredicate(final PredicateContext ctx) {
415         if (null != ctx.IN()) {
416             return createInSegment(ctx);
417         }
418         if (null != ctx.BETWEEN()) {
419             return createBetweenSegment(ctx);
420         }
421         if (null != ctx.LIKE()) {
422             return createBinaryOperationExpressionFromLike(ctx);
423         }
424         return visit(ctx.bitExpr(0));
425     }

427     private InExpression createInSegment(final PredicateContext ctx) {
428         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
429         ExpressionSegment right;
430         if (null == ctx.subquery()) {
431             ListExpression listExpression = new ListExpression(ctx.LP_().getSymbol().getStartIndex(), ctx.RP_().getSymbol().getStopIndex());
432             for (ExprContext each : ctx.expr()) {
433                 listExpression.getItems().add((ExpressionSegment) visit(each));
434             }
435             right = listExpression;
436         } else {
437             right = new SubqueryExpressionSegment(new SubquerySegment(ctx.subquery().start.getStartIndex(), ctx.subquery().stop.getStopIndex(), (OracleSelectStatement) visit(ctx.subquery())));
438         }
439         boolean not = null != ctx.NOT();
440         return new InExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, not);
441     }

443     private BinaryOperationExpression createBinaryOperationExpressionFromLike(final PredicateContext ctx) {
444         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
445         ListExpression right = new ListExpression(ctx.simpleExpr(0).start.getStartIndex(), ctx.simpleExpr().get(ctx.simpleExpr().size() - 1).stop.getStopIndex());
446         for (SimpleExprContext each : ctx.simpleExpr()) {
447             right.getItems().add((ExpressionSegment) visit(each));
448         }
449         String operator = null != ctx.NOT() ? "NOT LIKE" : "LIKE";
450         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
451         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
452     }

454     private BetweenExpression createBetweenSegment(final PredicateContext ctx) {
455         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
456         ExpressionSegment between = (ExpressionSegment) visit(ctx.bitExpr(1));
457         ExpressionSegment and = (ExpressionSegment) visit(ctx.predicate());
458         boolean not = null != ctx.NOT();
459         return new BetweenExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, between, and, not);
460     }

463     public final ASTNode visitBitExpr(final BitExprContext ctx) {
464         if (null != ctx.simpleExpr()) {
465             return createExpressionSegment(visit(ctx.simpleExpr()), ctx);
466         }
467         ExpressionSegment left = (ExpressionSegment) visit(ctx.getChild(0));

471         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
472     }

474     private ASTNode createExpressionSegment(final ASTNode astNode, final ParserRuleContext context) {
475         if (astNode instanceof StringLiteralValue) {
476             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((StringLiteralValue) astNode).getValue());
477         }
478         if (astNode instanceof NumberLiteralValue) {
479             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((NumberLiteralValue) astNode).getValue());
480         }
481         if (astNode instanceof BooleanLiteralValue) {
482             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((BooleanLiteralValue) astNode).getValue());
483         }
484         if (astNode instanceof ParameterMarkerValue) {
485             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) astNode;
486             ParameterMarkerExpressionSegment segment = new ParameterMarkerExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(),
487                     parameterMarker.getValue(), parameterMarker.getType());
488             parameterMarkerSegments.add(segment);
489             return segment;
490         }
491         if (astNode instanceof SubquerySegment) {
492             return new SubqueryExpressionSegment((SubquerySegment) astNode);
493         }
494         if (astNode instanceof OtherLiteralValue) {
495             return new CommonExpressionSegment(context.getStart().getStartIndex(), context.getStop().getStopIndex(), context.getText());
496         }
497         return astNode;
498     }

500     @Override
501     public final ASTNode visitSimpleExpr(final SimpleExprContext ctx) {
502         int startIndex = ctx.getStart().getStartIndex();
503         int stopIndex = ctx.getStop().getStopIndex();
504         if (null != ctx.subquery()) {
505             return new SubquerySegment(startIndex, stopIndex, (OracleSelectStatement) visit(ctx.subquery()));
506         }
507         if (null != ctx.parameterMarker()) {
508             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) visit(ctx.parameterMarker());
509             ParameterMarkerExpressionSegment segment = new ParameterMarkerExpressionSegment(startIndex, stopIndex, parameterMarker.getValue(), parameterMarker.getType());
510             parameterMarkerSegments.add(segment);
511             return segment;
512         }
513         if (null != ctx.literals()) {
514             return SQLUtils.createLiteralExpression(visit(ctx.literals()), startIndex, stopIndex, ctx.literals().start.getInputStream().getText(new Interval(startIndex, stopIndex)));
515         }
516         if (null != ctx.functionCall()) {
517             return visit(ctx.functionCall());
518         }
519         if (null != ctx.columnName()) {
520             return null == ctx.joinOperator() ? visit(ctx.columnName())
521                     : new ColumnWithJoinOperatorSegment(startIndex, stopIndex, (ColumnSegment) visitColumnName(ctx.columnName()), ctx.joinOperator().getText());
522         }
523         if (null != ctx.privateExprOfDb()) {
524             return visit(ctx.privateExprOfDb());
525         }
526         return new CommonExpressionSegment(startIndex, stopIndex, ctx.getText());
527     }

944         OrderDirection orderDirection = null != ctx.DESC() ? OrderDirection.DESC : OrderDirection.ASC;
945         NullsOrderType nullsOrderType = generateNullsOrderType(ctx);
946         if (null != ctx.columnName()) {
947             ColumnSegment column = (ColumnSegment) visit(ctx.columnName());
948             return new ColumnOrderByItemSegment(column, orderDirection, nullsOrderType);
949         }
950         if (null != ctx.numberLiterals()) {
951             return new IndexOrderByItemSegment(ctx.numberLiterals().getStart().getStartIndex(), ctx.numberLiterals().getStop().getStopIndex(),
952                     SQLUtils.getExactlyNumber(ctx.numberLiterals().getText(), 10).intValue(), orderDirection, nullsOrderType);
953         }
954         return new ExpressionOrderByItemSegment(ctx.expr().getStart().getStartIndex(), ctx.expr().getStop().getStopIndex(),
955                 getOriginalText(ctx.expr()), orderDirection, nullsOrderType, (ExpressionSegment) visit(ctx.expr()));
956     }
957     
958     private NullsOrderType generateNullsOrderType(final OrderByItemContext ctx) {
959         if (null == ctx.FIRST() && null == ctx.LAST()) {
960             return null;
961         }
962         return null == ctx.FIRST() ? NullsOrderType.LAST : NullsOrderType.FIRST;
963     }

965     @Override
966     public final ASTNode visitDataType(final DataTypeContext ctx) {
967         DataTypeSegment result = new DataTypeSegment();
968         if (null != ctx.dataTypeName()) {
969             result.setDataTypeName(((KeywordValue) visit(ctx.dataTypeName())).getValue());
970         }
971         if (null != ctx.specialDatatype()) {
972             result.setDataTypeName(((KeywordValue) visit(ctx.specialDatatype().dataTypeName())).getValue());
973         }
974         result.setStartIndex(ctx.start.getStartIndex());
975         result.setStopIndex(ctx.stop.getStopIndex());
976         if (null != ctx.dataTypeLength()) {
977             DataTypeLengthSegment dataTypeLengthSegment = (DataTypeLengthSegment) visit(ctx.dataTypeLength());
978             result.setDataLength(dataTypeLengthSegment);
979         }
980         return result;
981     }

983     @Override
984     public final ASTNode visitDataTypeLength(final DataTypeLengthContext ctx) {
985         DataTypeLengthSegment result = new DataTypeLengthSegment();
986         result.setStartIndex(ctx.start.getStartIndex());
987         result.setStopIndex(ctx.stop.getStartIndex());
988         List<TerminalNode> numbers = ctx.INTEGER_();
989         if (numbers.size() == 1) {
990             result.setPrecision(Integer.parseInt(numbers.get(0).getText()));
991         }
992         if (numbers.size() == 2) {
993             result.setPrecision(Integer.parseInt(numbers.get(0).getText()));
994             result.setScale(Integer.parseInt(numbers.get(1).getText()));
995         }
996         return result;
997     }
