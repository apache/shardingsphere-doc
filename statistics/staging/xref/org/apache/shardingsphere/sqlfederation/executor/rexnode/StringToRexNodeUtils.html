18  package org.apache.shardingsphere.sqlfederation.executor.rexnode;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.antlr.v4.runtime.CharStream;
23  import org.antlr.v4.runtime.CharStreams;
24  import org.antlr.v4.runtime.CommonTokenStream;
25  import org.antlr.v4.runtime.tree.ParseTree;
26  import org.apache.calcite.jdbc.JavaTypeFactoryImpl;
27  import org.apache.calcite.rel.type.RelDataTypeSystem;
28  import org.apache.calcite.rex.RexBuilder;
29  import org.apache.calcite.rex.RexNode;
30  import org.apache.shardingsphere.rexnode.autogen.SQLOptimizerRexNodeLexer;
31  import org.apache.shardingsphere.rexnode.autogen.SQLOptimizerRexNodeParser;
32  
33  import java.util.Map;
34  
35  /**
36   * Parsing string and generate rex node utility class.
37   */
38  @NoArgsConstructor(access = AccessLevel.PRIVATE)
39  public final class StringToRexNodeUtils {
40      
41      /**
42       * Parse string and generate rex node.
43       * 
44       * @param filterValue filter condition
45       * @param rexBuilder used to build rex node
46       * @param parameters parameters for SQL placeholder
47       * @param columnMap mapping of column id and column type
48       * @return rex node
49       */
50      public static RexNode buildRexNode(final String filterValue, final RexBuilder rexBuilder, final Map<String, Object> parameters, final Map<Integer, Integer> columnMap) {
51          CharStream input = CharStreams.fromString(filterValue);
52          SQLOptimizerRexNodeLexer lexer = new SQLOptimizerRexNodeLexer(input);
53          CommonTokenStream tokens = new CommonTokenStream(lexer);
54          SQLOptimizerRexNodeParser parser = new SQLOptimizerRexNodeParser(tokens);
55          ParseTree tree = parser.expression();
56          SQLOptimizerRexNodeVisitor visitor = new SQLOptimizerRexNodeVisitor(rexBuilder, new JavaTypeFactoryImpl(RelDataTypeSystem.DEFAULT), parameters, columnMap);
57          return visitor.visit(tree);
58      }
59  }
