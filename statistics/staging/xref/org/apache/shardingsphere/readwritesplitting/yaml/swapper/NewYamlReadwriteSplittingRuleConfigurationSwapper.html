45  /**
46   * TODO Rename YamlReadwriteSplittingRuleConfigurationSwapper when metadata structure adjustment completed. #25485
47   * YAML readwrite-splitting rule configuration swapper.
48   */
49  public final class NewYamlReadwriteSplittingRuleConfigurationSwapper implements NewYamlRuleConfigurationSwapper<ReadwriteSplittingRuleConfiguration> {
50      
51      private final YamlAlgorithmConfigurationSwapper algorithmSwapper = new YamlAlgorithmConfigurationSwapper();
52      
53      private final RuleNodePath readwriteSplittingRuleNodePath = new ReadwriteSplittingRuleNodePathProvider().getRuleNodePath();
54      
55      @Override
56      public Collection<YamlDataNode> swapToDataNodes(final ReadwriteSplittingRuleConfiguration data) {
57          Collection<YamlDataNode> result = new LinkedHashSet<>();
58          for (Entry<String, AlgorithmConfiguration> entry : data.getLoadBalancers().entrySet()) {
59              result.add(new YamlDataNode(readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.LOAD_BALANCERS).getPath(entry.getKey()),
60                      YamlEngine.marshal(algorithmSwapper.swapToYamlConfiguration(entry.getValue()))));
61          }
62          for (ReadwriteSplittingDataSourceRuleConfiguration each : data.getDataSources()) {
63              result.add(new YamlDataNode(readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.DATA_SOURCES).getPath(each.getName()),
64                      YamlEngine.marshal(swapToYamlConfiguration(each))));
65          }
66          return result;
67      }
68      
69      private YamlReadwriteSplittingDataSourceRuleConfiguration swapToYamlConfiguration(final ReadwriteSplittingDataSourceRuleConfiguration dataSourceRuleConfig) {
70          YamlReadwriteSplittingDataSourceRuleConfiguration result = new YamlReadwriteSplittingDataSourceRuleConfiguration();
71          result.setWriteDataSourceName(dataSourceRuleConfig.getWriteDataSourceName());
72          result.setReadDataSourceNames(dataSourceRuleConfig.getReadDataSourceNames());
73          result.setTransactionalReadQueryStrategy(dataSourceRuleConfig.getTransactionalReadQueryStrategy().name());
74          result.setLoadBalancerName(dataSourceRuleConfig.getLoadBalancerName());
75          return result;
76      }
77      
78      @Override
79      public Optional<ReadwriteSplittingRuleConfiguration> swapToObject(final Collection<YamlDataNode> dataNodes) {
80          List<YamlDataNode> validDataNodes = dataNodes.stream().filter(each -> readwriteSplittingRuleNodePath.getRoot().isValidatedPath(each.getKey())).collect(Collectors.toList());
81          if (validDataNodes.isEmpty()) {
82              return Optional.empty();
83          }
84          Collection<ReadwriteSplittingDataSourceRuleConfiguration> dataSources = new LinkedList<>();
85          Map<String, AlgorithmConfiguration> loadBalancerMap = new LinkedHashMap<>();
86          for (YamlDataNode each : validDataNodes) {
87              readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.DATA_SOURCES).getName(each.getKey())
88                      .ifPresent(optional -> dataSources.add(swapDataSource(optional, YamlEngine.unmarshal(each.getValue(), YamlReadwriteSplittingDataSourceRuleConfiguration.class))));
89              readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.LOAD_BALANCERS).getName(each.getKey())
90                      .ifPresent(optional -> loadBalancerMap.put(optional, algorithmSwapper.swapToObject(YamlEngine.unmarshal(each.getValue(), YamlAlgorithmConfiguration.class))));
91          }
92          return Optional.of(new ReadwriteSplittingRuleConfiguration(dataSources, loadBalancerMap));
93      }
94      
95      private ReadwriteSplittingDataSourceRuleConfiguration swapDataSource(final String name, final YamlReadwriteSplittingDataSourceRuleConfiguration yamlDataSourceRuleConfig) {
96          return new ReadwriteSplittingDataSourceRuleConfiguration(name, yamlDataSourceRuleConfig.getWriteDataSourceName(), yamlDataSourceRuleConfig.getReadDataSourceNames(),
97                  getTransactionalReadQueryStrategy(yamlDataSourceRuleConfig), yamlDataSourceRuleConfig.getLoadBalancerName());
98      }
99      
100     private TransactionalReadQueryStrategy getTransactionalReadQueryStrategy(final YamlReadwriteSplittingDataSourceRuleConfiguration yamlDataSourceRuleConfig) {
101         return Strings.isNullOrEmpty(yamlDataSourceRuleConfig.getTransactionalReadQueryStrategy())
102                 ? TransactionalReadQueryStrategy.DYNAMIC
103                 : TransactionalReadQueryStrategy.valueOf(yamlDataSourceRuleConfig.getTransactionalReadQueryStrategy());
104     }
105     
106     @Override
107     public Class<ReadwriteSplittingRuleConfiguration> getTypeClass() {
108         return ReadwriteSplittingRuleConfiguration.class;
109     }
110     
111     @Override
112     public String getRuleTagName() {
113         return "READWRITE_SPLITTING";
114     }
115     
116     @Override
117     public int getOrder() {
118         return ReadwriteSplittingOrder.ORDER;
119     }
120 }
