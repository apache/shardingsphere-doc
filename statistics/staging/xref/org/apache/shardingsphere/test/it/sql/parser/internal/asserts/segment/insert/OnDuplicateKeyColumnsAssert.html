29  import static org.hamcrest.CoreMatchers.is;
30  import static org.hamcrest.MatcherAssert.assertThat;
31  import static org.junit.jupiter.api.Assertions.assertNotNull;
32  
33  /**
34   * On duplicate key columns assert.
35   */
36  @NoArgsConstructor(access = AccessLevel.PRIVATE)
37  public final class OnDuplicateKeyColumnsAssert {
38      
39      /**
40       * Assert actual on duplicate key columns segment is correct with expected on duplicate key columns.
41       *
42       * @param assertContext assert context
43       * @param actual actual on duplicate key columns segment
44       * @param expected expected on duplicate key columns
45       */
46      public static void assertIs(final SQLCaseAssertContext assertContext, final OnDuplicateKeyColumnsSegment actual, final ExpectedOnDuplicateKeyColumns expected) {
47          assertNotNull(expected, assertContext.getText("On duplicate key columns should exist."));
48          assertThat(assertContext.getText("On duplicate key columns size assertion error: "), actual.getColumns().size(), is(expected.getAssignments().size()));
49          int count = 0;
50          for (ColumnAssignmentSegment each : actual.getColumns()) {
51              AssignmentAssert.assertIs(assertContext, each, expected.getAssignments().get(count));
52              count++;
53          }
54          SQLSegmentAssert.assertIs(assertContext, actual, expected);
55      }
56  }
