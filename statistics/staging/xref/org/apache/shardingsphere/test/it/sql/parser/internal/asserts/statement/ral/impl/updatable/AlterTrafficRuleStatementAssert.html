35  import static org.hamcrest.CoreMatchers.is;

38  /**
39   * Alter traffic rule statement assert.
40   */
41  @NoArgsConstructor(access = AccessLevel.PRIVATE)
42  public final class AlterTrafficRuleStatementAssert {
43      
44      /**
45       * Assert alter traffic rule statement is correct with expected parser result.
46       *
47       * @param assertContext assert context
48       * @param actual actual alter traffic rule statement
49       * @param expected expected alter traffic rule statement test case
50       */
51      public static void assertIs(final SQLCaseAssertContext assertContext, final AlterTrafficRuleStatement actual, final AlterTrafficRuleStatementTestCase expected) {
52          if (ExistingAssert.assertIs(assertContext, actual, expected)) {
53              assertTrafficRuleSegments(assertContext, actual.getSegments(), expected.getRules());
54          }
55      }
56      
57      private static void assertTrafficRuleSegments(final SQLCaseAssertContext assertContext, final Collection<TrafficRuleSegment> actual, final List<ExpectedTrafficRule> expected) {
58          Map<String, TrafficRuleSegment> actualMap = actual.stream().collect(Collectors.toMap(TrafficRuleSegment::getName, each -> each));
59          for (ExpectedTrafficRule each : expected) {
60              TrafficRuleSegment actualRule = actualMap.get(each.getName());
61              assertThat(actualRule.getName(), is(each.getName()));
62              assertThat(actualRule.getLabels(), is(each.getLabels()));
63              AlgorithmAssert.assertIs(assertContext, actualRule.getAlgorithm(), each.getTrafficAlgorithm());
64              AlgorithmAssert.assertIs(assertContext, actualRule.getLoadBalancer(), each.getLoadBalancer());
65          }
66      }
67  }
