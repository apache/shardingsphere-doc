18  package org.apache.shardingsphere.mask.merge;
19  
20  import org.apache.shardingsphere.infra.binder.context.statement.SQLStatementContext;
21  import org.apache.shardingsphere.infra.binder.context.statement.dml.SelectStatementContext;
22  import org.apache.shardingsphere.infra.config.props.ConfigurationProperties;
23  import org.apache.shardingsphere.infra.merge.engine.decorator.ResultDecorator;
24  import org.apache.shardingsphere.infra.merge.engine.decorator.ResultDecoratorEngine;
25  import org.apache.shardingsphere.infra.metadata.database.ShardingSphereDatabase;
26  import org.apache.shardingsphere.infra.metadata.database.rule.RuleMetaData;
27  import org.apache.shardingsphere.mask.constant.MaskOrder;
28  import org.apache.shardingsphere.mask.merge.dql.MaskDQLResultDecorator;
29  import org.apache.shardingsphere.mask.rule.MaskRule;
30  
31  import java.util.Optional;
32  
33  /**
34   * Result decorator engine for mask.
35   */
36  public final class MaskResultDecoratorEngine implements ResultDecoratorEngine<MaskRule> {
37      
38      @Override
39      public Optional<ResultDecorator<MaskRule>> newInstance(final RuleMetaData globalRuleMetaData, final ShardingSphereDatabase database,
40                                                             final MaskRule maskRule, final ConfigurationProperties props, final SQLStatementContext sqlStatementContext) {
41          return sqlStatementContext instanceof SelectStatementContext ? Optional.of(new MaskDQLResultDecorator(maskRule, (SelectStatementContext) sqlStatementContext)) : Optional.empty();
42      }
43      
44      @Override
45      public int getOrder() {
46          return MaskOrder.ORDER;
47      }
48      
49      @Override
50      public Class<MaskRule> getTypeClass() {
51          return MaskRule.class;
52      }
53  }
