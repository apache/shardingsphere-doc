54  /**
55   * Select statement assert.
56   */
57  @NoArgsConstructor(access = AccessLevel.PRIVATE)
58  public final class SelectStatementAssert {
59      
60      /**
61       * Assert select statement is correct with expected parser result.
62       *
63       * @param assertContext assert context
64       * @param actual actual select statement
65       * @param expected expected select statement test case
66       */
67      public static void assertIs(final SQLCaseAssertContext assertContext, final SelectStatement actual, final SelectStatementTestCase expected) {
68          assertProjection(assertContext, actual, expected);
69          assertWhereClause(assertContext, actual, expected);
70          assertGroupByClause(assertContext, actual, expected);
71          assertHavingClause(assertContext, actual, expected);
72          assertWindowClause(assertContext, actual, expected);
73          assertOrderByClause(assertContext, actual, expected);
74          assertLimitClause(assertContext, actual, expected);
75          assertTable(assertContext, actual, expected);
76          assertLockClause(assertContext, actual, expected);
77          assertWithClause(assertContext, actual, expected);
78          assertCombineClause(assertContext, actual, expected);
79          assertModelClause(assertContext, actual, expected);
80          assertIntoClause(assertContext, actual, expected);
81      }
82      
83      private static void assertWindowClause(final SQLCaseAssertContext assertContext, final SelectStatement actual, final SelectStatementTestCase expected) {
84          Optional<WindowSegment> windowSegment = actual.getWindow();
85          if (null == expected.getWindowClause()) {
86              assertFalse(windowSegment.isPresent(), assertContext.getText("Actual window segment should not exist."));
87          } else {
88              assertTrue(windowSegment.isPresent(), assertContext.getText("Actual window segment should exist."));
89              WindowClauseAssert.assertIs(assertContext, windowSegment.get(), expected.getWindowClause());
90          }
91      }
92      
93      private static void assertHavingClause(final SQLCaseAssertContext assertContext, final SelectStatement actual, final SelectStatementTestCase expected) {
94          if (null == expected.getHavingClause()) {
95              assertFalse(actual.getHaving().isPresent(), assertContext.getText("Actual having segment should not exist."));
96          } else {
97              assertTrue(actual.getHaving().isPresent(), assertContext.getText("Actual having segment should exist."));
98              HavingClauseAssert.assertIs(assertContext, actual.getHaving().get(), expected.getHavingClause());
99          }
100     }
101     
102     private static void assertProjection(final SQLCaseAssertContext assertContext, final SelectStatement actual, final SelectStatementTestCase expected) {
103         if (null == actual.getProjections() && 0 == expected.getProjections().getSize()) {
104             return;
105         }
106         ProjectionAssert.assertIs(assertContext, actual.getProjections(), expected.getProjections());
107     }
108     
109     private static void assertTable(final SQLCaseAssertContext assertContext, final SelectStatement actual, final SelectStatementTestCase expected) {
110         if (null == expected.getFrom()) {
111             assertFalse(actual.getFrom().isPresent(), assertContext.getText("Actual simple-table should not exist."));
112         } else {
113             assertTrue(actual.getFrom().isPresent(), assertContext.getText("Actual from segment should exist."));
114             TableAssert.assertIs(assertContext, actual.getFrom().get(), expected.getFrom());
115         }
116         if (actual instanceof MySQLSelectStatement) {
117             if (null == expected.getSimpleTable()) {
118                 assertFalse(((MySQLSelectStatement) actual).getTable().isPresent(), assertContext.getText("Actual simple-table should not exist."));
119             } else {
120                 Optional<SimpleTableSegment> table = ((MySQLSelectStatement) actual).getTable();
121                 assertTrue(table.isPresent(), assertContext.getText("Actual table segment should exist."));
122                 TableAssert.assertIs(assertContext, table.orElse(null), expected.getSimpleTable());
123             }
124         }
125     }
126

154     private static void assertLimitClause(final SQLCaseAssertContext assertContext, final SelectStatement actual, final SelectStatementTestCase expected) {
155         Optional<LimitSegment> limitSegment = actual.getLimit();
156         if (null == expected.getLimitClause()) {
157             assertFalse(limitSegment.isPresent(), assertContext.getText("Actual limit segment should not exist."));
158         } else {
159             assertTrue(limitSegment.isPresent(), assertContext.getText("Actual limit segment should exist."));
160             LimitClauseAssert.assertOffset(assertContext, limitSegment.get().getOffset().orElse(null), expected.getLimitClause().getOffset());
161             LimitClauseAssert.assertRowCount(assertContext, limitSegment.get().getRowCount().orElse(null), expected.getLimitClause().getRowCount());
162             SQLSegmentAssert.assertIs(assertContext, limitSegment.get(), expected.getLimitClause());
163         }
164     }

185     
186     private static void assertCombineClause(final SQLCaseAssertContext assertContext, final SelectStatement actual, final SelectStatementTestCase expected) {
187         Optional<CombineSegment> combineSegment = actual.getCombine();
188         if (null == expected.getCombineClause()) {
189             assertFalse(combineSegment.isPresent(), assertContext.getText("Actual combine segment should not exist."));
190         } else {
191             assertTrue(combineSegment.isPresent(), assertContext.getText("Actual combine segment should exist."));
192             assertThat(assertContext.getText("Combine type assertion error: "), combineSegment.get().getCombineType().name(), is(expected.getCombineClause().getCombineType()));
193             SQLSegmentAssert.assertIs(assertContext, combineSegment.get(), expected.getCombineClause());
194             assertIs(assertContext, combineSegment.get().getLeft().getSelect(), expected.getCombineClause().getLeft());
195             assertIs(assertContext, combineSegment.get().getRight().getSelect(), expected.getCombineClause().getRight());
196         }
197     }
198
