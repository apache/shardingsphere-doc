28  import org.apache.shardingsphere.sharding.yaml.swapper.ShardingRuleConfigurationConverter;
29  
30  import javax.sql.DataSource;
31  import java.sql.SQLException;
32  import java.util.Optional;
33  
34  /**
35   * ShardingSphere pipeline data source creator.
36   */
37  public final class ShardingSpherePipelineDataSourceCreator implements PipelineDataSourceCreator {
38      
39      @Override
40      public DataSource createPipelineDataSource(final Object dataSourceConfig) throws SQLException {
41          YamlRootConfiguration rootConfig = YamlEngine.unmarshal(YamlEngine.marshal(dataSourceConfig), YamlRootConfiguration.class);
42          enableStreamingQuery(rootConfig);
43          Optional<YamlShardingRuleConfiguration> yamlShardingRuleConfig = ShardingRuleConfigurationConverter.findYamlShardingRuleConfiguration(rootConfig.getRules());
44          if (yamlShardingRuleConfig.isPresent()) {
45              enableRangeQueryForInline(yamlShardingRuleConfig.get());
46              removeAuditStrategy(yamlShardingRuleConfig.get());
47          }
48          rootConfig.setDatabaseName(rootConfig.getDatabaseName());
49          rootConfig.setSchemaName(rootConfig.getSchemaName());
50          return YamlShardingSphereDataSourceFactory.createDataSourceWithoutCache(rootConfig);
51      }
52      
53      // TODO Another way is improving ExecuteQueryCallback.executeSQL to enable streaming query, then remove it
54      private void enableStreamingQuery(final YamlRootConfiguration rootConfig) {
55          // Set a large enough value to enable ConnectionMode.MEMORY_STRICTLY, make sure streaming query work.
56          rootConfig.getProps().put(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY.getKey(), 100000);
57      }
58      
59      private void enableRangeQueryForInline(final YamlShardingRuleConfiguration yamlShardingRuleConfig) {
60          for (YamlAlgorithmConfiguration each : yamlShardingRuleConfig.getShardingAlgorithms().values()) {
61              if ("INLINE".equalsIgnoreCase(each.getType())) {
62                  each.getProps().put("allow-range-query-with-inline-sharding", Boolean.TRUE.toString());
63              }
64          }
65      }
66      
67      private void removeAuditStrategy(final YamlShardingRuleConfiguration yamlShardingRuleConfig) {
68          yamlShardingRuleConfig.setDefaultAuditStrategy(null);
69          yamlShardingRuleConfig.setAuditors(null);
70          if (null != yamlShardingRuleConfig.getTables()) {
71              yamlShardingRuleConfig.getTables().forEach((key, value) -> value.setAuditStrategy(null));
72          }
73          if (null != yamlShardingRuleConfig.getAutoTables()) {
74              yamlShardingRuleConfig.getAutoTables().forEach((key, value) -> value.setAuditStrategy(null));
75          }
76      }
77      
78      @Override
79      public String getType() {
80          return ShardingSpherePipelineDataSourceConfiguration.TYPE;
81      }
82  }
