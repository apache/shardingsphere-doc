116 /**
117  * SQL statement context factory.
118  */
119 @NoArgsConstructor(access = AccessLevel.PRIVATE)
120 public final class SQLStatementContextFactory {
121     
122     /**
123      * Create SQL statement context.
124      *
125      * @param metaData metadata
126      * @param params SQL parameters
127      * @param sqlStatement SQL statement
128      * @param defaultDatabaseName default database name
129      * @return SQL statement context
130      */
131     public static SQLStatementContext newInstance(final ShardingSphereMetaData metaData, final List<Object> params, final SQLStatement sqlStatement, final String defaultDatabaseName) {
132         if (sqlStatement instanceof DMLStatement) {
133             return getDMLStatementContext(metaData, params, (DMLStatement) sqlStatement, defaultDatabaseName);
134         }
135         if (sqlStatement instanceof DDLStatement) {
136             return getDDLStatementContext(metaData, params, (DDLStatement) sqlStatement, defaultDatabaseName);
137         }
138         if (sqlStatement instanceof DCLStatement) {
139             return getDCLStatementContext((DCLStatement) sqlStatement);
140         }
141         if (sqlStatement instanceof DALStatement) {
142             return getDALStatementContext((DALStatement) sqlStatement);
143         }
144         return new UnknownSQLStatementContext(sqlStatement);
145     }
146     
147     private static SQLStatementContext getDMLStatementContext(final ShardingSphereMetaData metaData,
148                                                               final List<Object> params, final DMLStatement sqlStatement, final String defaultDatabaseName) {
149         if (sqlStatement instanceof SelectStatement) {
150             return new SelectStatementContext(metaData, params, (SelectStatement) sqlStatement, defaultDatabaseName);
151         }
152         if (sqlStatement instanceof UpdateStatement) {
153             return new UpdateStatementContext((UpdateStatement) sqlStatement);
154         }
155         if (sqlStatement instanceof DeleteStatement) {
156             return new DeleteStatementContext((DeleteStatement) sqlStatement);
157         }
158         if (sqlStatement instanceof InsertStatement) {
159             return new InsertStatementContext(metaData, params, (InsertStatement) sqlStatement, defaultDatabaseName);
160         }
161         if (sqlStatement instanceof CallStatement) {
162             return new CallStatementContext((CallStatement) sqlStatement);
163         }
164         if (sqlStatement instanceof CopyStatement) {
165             return new CopyStatementContext((CopyStatement) sqlStatement);
166         }
167         if (sqlStatement instanceof DoStatement) {
168             return new DoStatementContext((DoStatement) sqlStatement);
169         }
170         if (sqlStatement instanceof MySQLLoadDataStatement) {
171             return new LoadDataStatementContext((MySQLLoadDataStatement) sqlStatement);
172         }
173         if (sqlStatement instanceof MySQLLoadXMLStatement) {
174             return new LoadXMLStatementContext((MySQLLoadXMLStatement) sqlStatement);
175         }
176         throw new UnsupportedSQLOperationException(String.format("Unsupported SQL statement `%s`", sqlStatement.getClass().getSimpleName()));
177     }
178     
179     private static SQLStatementContext getDDLStatementContext(final ShardingSphereMetaData metaData, final List<Object> params,
180                                                               final DDLStatement sqlStatement, final String defaultDatabaseName) {
181         if (sqlStatement instanceof CreateSchemaStatement) {
182             return new CreateSchemaStatementContext((CreateSchemaStatement) sqlStatement);
183         }
184         if (sqlStatement instanceof CreateTableStatement) {
185             return new CreateTableStatementContext((CreateTableStatement) sqlStatement);
186         }
187         if (sqlStatement instanceof AlterTableStatement) {
188             return new AlterTableStatementContext((AlterTableStatement) sqlStatement);
189         }
190         if (sqlStatement instanceof RenameTableStatement) {
191             return new RenameTableStatementContext((RenameTableStatement) sqlStatement);
192         }
193         if (sqlStatement instanceof DropTableStatement) {
194             return new DropTableStatementContext((DropTableStatement) sqlStatement);
195         }
196         if (sqlStatement instanceof CreateIndexStatement) {
197             return new CreateIndexStatementContext((CreateIndexStatement) sqlStatement);
198         }
199         if (sqlStatement instanceof AlterIndexStatement) {
200             return new AlterIndexStatementContext((AlterIndexStatement) sqlStatement);
201         }
202         if (sqlStatement instanceof DropIndexStatement) {
203             return new DropIndexStatementContext((DropIndexStatement) sqlStatement);
204         }
205         if (sqlStatement instanceof TruncateStatement) {
206             return new TruncateStatementContext((TruncateStatement) sqlStatement);
207         }
208         if (sqlStatement instanceof CreateFunctionStatement) {
209             return new CreateFunctionStatementContext((CreateFunctionStatement) sqlStatement);
210         }
211         if (sqlStatement instanceof CreateProcedureStatement) {
212             return new CreateProcedureStatementContext((CreateProcedureStatement) sqlStatement);
213         }
214         if (sqlStatement instanceof CreateViewStatement) {
215             return new CreateViewStatementContext((CreateViewStatement) sqlStatement);
216         }
217         if (sqlStatement instanceof AlterViewStatement) {
218             return new AlterViewStatementContext((AlterViewStatement) sqlStatement);
219         }
220         if (sqlStatement instanceof DropViewStatement) {
221             return new DropViewStatementContext((DropViewStatement) sqlStatement);
222         }
223         if (sqlStatement instanceof PrepareStatement) {
224             return new PrepareStatementContext((PrepareStatement) sqlStatement);
225         }
226         if (sqlStatement instanceof CommentStatement) {
227             return new CommentStatementContext((CommentStatement) sqlStatement);
228         }
229         if (sqlStatement instanceof OpenGaussCursorStatement) {
230             return new CursorStatementContext(metaData, params, (OpenGaussCursorStatement) sqlStatement, defaultDatabaseName);
231         }
232         if (sqlStatement instanceof CloseStatement) {
233             return new CloseStatementContext((CloseStatement) sqlStatement);
234         }
235         if (sqlStatement instanceof MoveStatement) {
236             return new MoveStatementContext((MoveStatement) sqlStatement);
237         }
238         if (sqlStatement instanceof FetchStatement) {
239             return new FetchStatementContext((FetchStatement) sqlStatement);
240         }
241         return new UnknownSQLStatementContext(sqlStatement);
242     }
243     
244     private static SQLStatementContext getDCLStatementContext(final DCLStatement sqlStatement) {
245         if (sqlStatement instanceof GrantStatement) {
246             return new GrantStatementContext((GrantStatement) sqlStatement);
247         }
248         if (sqlStatement instanceof RevokeStatement) {
249             return new RevokeStatementContext((RevokeStatement) sqlStatement);
250         }
251         if (sqlStatement instanceof SQLServerDenyUserStatement) {
252             return new DenyUserStatementContext((SQLServerDenyUserStatement) sqlStatement);
253         }
254         return new UnknownSQLStatementContext(sqlStatement);
255     }
256     
257     private static SQLStatementContext getDALStatementContext(final DALStatement sqlStatement) {
258         if (sqlStatement instanceof ExplainStatement) {
259             return new ExplainStatementContext((ExplainStatement) sqlStatement);
260         }
261         if (sqlStatement instanceof MySQLShowCreateTableStatement) {
262             return new ShowCreateTableStatementContext((MySQLShowCreateTableStatement) sqlStatement);
263         }
264         if (sqlStatement instanceof MySQLShowColumnsStatement) {
265             return new ShowColumnsStatementContext((MySQLShowColumnsStatement) sqlStatement);
266         }
267         if (sqlStatement instanceof MySQLShowTablesStatement) {
268             return new ShowTablesStatementContext((MySQLShowTablesStatement) sqlStatement);
269         }
270         if (sqlStatement instanceof MySQLShowTableStatusStatement) {
271             return new ShowTableStatusStatementContext((MySQLShowTableStatusStatement) sqlStatement);
272         }
273         if (sqlStatement instanceof MySQLShowIndexStatement) {
274             return new ShowIndexStatementContext((MySQLShowIndexStatement) sqlStatement);
275         }
276         if (sqlStatement instanceof AnalyzeTableStatement) {
277             return new AnalyzeTableStatementContext((AnalyzeTableStatement) sqlStatement);
278         }
279         if (sqlStatement instanceof FlushStatement) {
280             return new FlushStatementContext((FlushStatement) sqlStatement);
281         }
282         if (sqlStatement instanceof MySQLOptimizeTableStatement) {
283             return new OptimizeTableStatementContext((MySQLOptimizeTableStatement) sqlStatement);
284         }
285         if (sqlStatement instanceof MySQLKillStatement) {
286             return new KillStatementContext((MySQLKillStatement) sqlStatement);
287         }
288         return new UnknownSQLStatementContext(sqlStatement);
289     }
