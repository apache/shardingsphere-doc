18  package org.apache.shardingsphere.shadow.distsql.handler.query;
19  
20  import lombok.Setter;
21  import org.apache.shardingsphere.distsql.handler.aware.DistSQLExecutorRuleAware;
22  import org.apache.shardingsphere.distsql.handler.engine.query.DistSQLQueryExecutor;
23  import org.apache.shardingsphere.infra.merge.result.impl.local.LocalDataQueryResultRow;
24  import org.apache.shardingsphere.mode.manager.ContextManager;
25  import org.apache.shardingsphere.shadow.config.ShadowRuleConfiguration;
26  import org.apache.shardingsphere.shadow.distsql.statement.ShowShadowTableRulesStatement;
27  import org.apache.shardingsphere.shadow.rule.ShadowRule;
28  
29  import java.util.ArrayList;
30  import java.util.Arrays;
31  import java.util.Collection;
32  import java.util.HashMap;
33  import java.util.Map;
34  import java.util.stream.Collectors;
35  
36  /**
37   * Show shadow table rules executor.
38   */
39  @Setter
40  public final class ShowShadowTableRulesExecutor implements DistSQLQueryExecutor<ShowShadowTableRulesStatement>, DistSQLExecutorRuleAware<ShadowRule> {
41      
42      private static final String SHADOW_TABLE = "shadow_table";
43      
44      private static final String SHADOW_ALGORITHM_NAME = "shadow_algorithm_name";
45      
46      private ShadowRule rule;
47      
48      @Override
49      public Collection<String> getColumnNames(final ShowShadowTableRulesStatement sqlStatement) {
50          return Arrays.asList(SHADOW_TABLE, SHADOW_ALGORITHM_NAME);
51      }
52      
53      @Override
54      public Collection<LocalDataQueryResultRow> getRows(final ShowShadowTableRulesStatement sqlStatement, final ContextManager contextManager) {
55          return buildData(rule.getConfiguration(), sqlStatement).stream()
56                  .map(each -> new LocalDataQueryResultRow(each.get(SHADOW_TABLE), each.get(SHADOW_ALGORITHM_NAME))).collect(Collectors.toList());
57      }
58      
59      private Collection<Map<String, String>> buildData(final ShadowRuleConfiguration ruleConfig, final ShowShadowTableRulesStatement sqlStatement) {
60          Collection<Map<String, String>> result = new ArrayList<>();
61          if (isSpecified(sqlStatement)) {
62              ruleConfig.getTables().forEach((key, value) -> {
63                  Map<String, String> map = new HashMap<>();
64                  if (key.equalsIgnoreCase(sqlStatement.getTableName())) {
65                      map.put(SHADOW_TABLE, key);
66                      map.put(SHADOW_ALGORITHM_NAME, convertToString(value.getShadowAlgorithmNames()));
67                  }
68                  result.add(map);
69              });
70          } else {
71              ruleConfig.getTables().forEach((key, value) -> {
72                  Map<String, String> map = new HashMap<>();
73                  map.put(SHADOW_TABLE, key);
74                  map.put(SHADOW_ALGORITHM_NAME, convertToString(value.getShadowAlgorithmNames()));
75                  result.add(map);
76              });
77          }
78          return result;
79      }
80      
81      private boolean isSpecified(final ShowShadowTableRulesStatement sqlStatement) {
82          return null != sqlStatement.getTableName() && !sqlStatement.getTableName().isEmpty();
83      }
84      
85      private String convertToString(final Collection<String> shadowTables) {
86          return null == shadowTables ? "" : String.join(",", shadowTables);
87      }
88      
89      @Override
90      public Class<ShadowRule> getRuleClass() {
91          return ShadowRule.class;
92      }
93      
94      @Override
95      public Class<ShowShadowTableRulesStatement> getType() {
96          return ShowShadowTableRulesStatement.class;
97      }
98  }
