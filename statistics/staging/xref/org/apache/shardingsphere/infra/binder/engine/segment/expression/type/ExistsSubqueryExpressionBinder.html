18  package org.apache.shardingsphere.infra.binder.engine.segment.expression.type;
19  
20  import com.cedarsoftware.util.CaseInsensitiveMap.CaseInsensitiveString;
21  import com.google.common.collect.Multimap;
22  import lombok.AccessLevel;
23  import lombok.NoArgsConstructor;
24  import org.apache.shardingsphere.infra.binder.engine.segment.from.context.TableSegmentBinderContext;
25  import org.apache.shardingsphere.infra.binder.engine.statement.SQLStatementBinderContext;
26  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.ExistsSubqueryExpression;
27  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.subquery.SubquerySegment;
28  
29  /**
30   * Exists subquery expression binder.
31   */
32  @NoArgsConstructor(access = AccessLevel.PRIVATE)
33  public final class ExistsSubqueryExpressionBinder {
34      
35      /**
36       * Bind exists subquery expression.
37       *
38       * @param segment exists subquery expression segment
39       * @param binderContext SQL statement binder context
40       * @param tableBinderContexts table binder contexts
41       * @return bound exists subquery expression segment
42       */
43      public static ExistsSubqueryExpression bind(final ExistsSubqueryExpression segment, final SQLStatementBinderContext binderContext,
44                                                  final Multimap<CaseInsensitiveString, TableSegmentBinderContext> tableBinderContexts) {
45          SubquerySegment boundSubquery = SubquerySegmentBinder.bind(segment.getSubquery(), binderContext, tableBinderContexts);
46          ExistsSubqueryExpression result = new ExistsSubqueryExpression(segment.getStartIndex(), segment.getStopIndex(), boundSubquery);
47          result.setNot(segment.isNot());
48          return result;
49      }
50  }
