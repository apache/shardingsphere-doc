18  package org.apache.shardingsphere.infra.binder.type;

20  import lombok.RequiredArgsConstructor;
21  import org.apache.shardingsphere.infra.binder.statement.dml.DeleteStatementBinder;
22  import org.apache.shardingsphere.infra.binder.statement.dml.InsertStatementBinder;
23  import org.apache.shardingsphere.infra.binder.statement.dml.MergeStatementBinder;
24  import org.apache.shardingsphere.infra.binder.statement.dml.SelectStatementBinder;
25  import org.apache.shardingsphere.infra.binder.statement.dml.UpdateStatementBinder;
26  import org.apache.shardingsphere.infra.metadata.ShardingSphereMetaData;
27  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.DMLStatement;
28  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.DeleteStatement;
29  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.InsertStatement;
30  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.MergeStatement;
31  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.SelectStatement;
32  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.UpdateStatement;
33  
34  /**
35   * DML statement bind engine.
36   */
37  @RequiredArgsConstructor
38  public final class DMLStatementBindEngine {
39      
40      private final ShardingSphereMetaData metaData;
41      
42      private final String currentDatabaseName;
43      
44      /**
45       * Bind DML statement.
46       *
47       * @param statement to be bound DML statement
48       * @return bound DML statement
49       */
50      public DMLStatement bind(final DMLStatement statement) {
51          if (statement instanceof SelectStatement) {
52              return new SelectStatementBinder().bind((SelectStatement) statement, metaData, currentDatabaseName);
53          }
54          if (statement instanceof InsertStatement) {
55              return new InsertStatementBinder().bind((InsertStatement) statement, metaData, currentDatabaseName);
56          }
57          if (statement instanceof UpdateStatement) {
58              return new UpdateStatementBinder().bind((UpdateStatement) statement, metaData, currentDatabaseName);
59          }
60          if (statement instanceof DeleteStatement) {
61              return new DeleteStatementBinder().bind((DeleteStatement) statement, metaData, currentDatabaseName);
62          }
63          if (statement instanceof MergeStatement) {
64              return new MergeStatementBinder().bind((MergeStatement) statement, metaData, currentDatabaseName);
65          }
66          return statement;
67      }
68  }
