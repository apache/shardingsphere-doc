52  /**
53   * Inventory dumper context splitter.
54   */
55  @RequiredArgsConstructor
56  public final class InventoryDumperContextSplitter {
57      
58      private final PipelineDataSourceWrapper sourceDataSource;
59      
60      private final InventoryDumperContext dumperContext;
61      
62      /**
63       * Split inventory dumper context.
64       *
65       * @param jobItemContext job item context
66       * @return inventory dumper contexts
67       */
68      public Collection<InventoryDumperContext> split(final TransmissionJobItemContext jobItemContext) {
69          return splitByTable().stream().flatMap(each -> splitByPrimaryKey(each, jobItemContext).stream()).collect(Collectors.toList());
70      }
71      
72      private Collection<InventoryDumperContext> splitByTable() {
73          return dumperContext.getCommonContext().getTableNameMapper().getTableNameMap().entrySet()
74                  .stream().map(entry -> createTableSpLitDumperContext(entry.getKey(), entry.getValue())).collect(Collectors.toList());
75      }
76      
77      private InventoryDumperContext createTableSpLitDumperContext(final CaseInsensitiveIdentifier actualTableName, final CaseInsensitiveIdentifier logicTableName) {
78          InventoryDumperContext result = new InventoryDumperContext(dumperContext.getCommonContext());
79          result.setActualTableName(actualTableName.toString());
80          result.setLogicTableName(logicTableName.toString());
81          result.getCommonContext().setPosition(new IngestPlaceholderPosition());
82          result.setInsertColumnNames(dumperContext.getInsertColumnNames());
83          result.setUniqueKeyColumns(dumperContext.getUniqueKeyColumns());
84          return result;
85      }
86      
87      private Collection<InventoryDumperContext> splitByPrimaryKey(final InventoryDumperContext dumperContext, final TransmissionJobItemContext jobItemContext) {
88          if (null == dumperContext.getUniqueKeyColumns()) {
89              dumperContext.setUniqueKeyColumns(getTableUniqueKeys(dumperContext, jobItemContext));
90          }
91          Collection<InventoryDumperContext> result = new LinkedList<>();
92          TransmissionProcessContext jobProcessContext = jobItemContext.getJobProcessContext();
93          int batchSize = jobProcessContext.getProcessConfiguration().getRead().getBatchSize();
94          JobRateLimitAlgorithm rateLimitAlgorithm = jobProcessContext.getReadRateLimitAlgorithm();
95          int i = 0;
96          for (IngestPosition each : getInventoryPositions(dumperContext, jobItemContext)) {
97              result.add(createPrimaryKeySplitDumperContext(dumperContext, each, i++, batchSize, rateLimitAlgorithm));
98          }
99          return result;
100     }
101     
102     private List<PipelineColumnMetaData> getTableUniqueKeys(final InventoryDumperContext dumperContext, final TransmissionJobItemContext jobItemContext) {
103         String schemaName = dumperContext.getCommonContext().getTableAndSchemaNameMapper().getSchemaName(dumperContext.getLogicTableName());
104         return PipelineTableMetaDataUtils.getUniqueKeyColumns(schemaName, dumperContext.getActualTableName(), jobItemContext.getSourceMetaDataLoader());
105     }
106     
107     private Collection<IngestPosition> getInventoryPositions(final InventoryDumperContext dumperContext, final TransmissionJobItemContext jobItemContext) {
108         TransmissionJobItemProgress initProgress = jobItemContext.getInitProgress();
109         if (null != initProgress) {
110             // Do NOT filter FinishedPosition here, since whole inventory tasks are required in job progress when persisting to register center.
111             Collection<IngestPosition> result = initProgress.getInventory().getInventoryPosition(dumperContext.getActualTableName()).values();
112             if (!result.isEmpty()) {
113                 return result;
114             }
115         }
116         long tableRecordsCount = InventoryRecordsCountCalculator.getTableRecordsCount(dumperContext, sourceDataSource);
117         jobItemContext.updateInventoryRecordsCount(tableRecordsCount);
118         if (!dumperContext.hasUniqueKey()) {
119             return Collections.singleton(new UnsupportedKeyIngestPosition());
120         }
121         List<PipelineColumnMetaData> uniqueKeyColumns = dumperContext.getUniqueKeyColumns();
122         if (1 == uniqueKeyColumns.size()) {
123             int firstColumnDataType = uniqueKeyColumns.get(0).getDataType();
124             if (PipelineJdbcUtils.isIntegerColumn(firstColumnDataType)) {
125                 return getPositionByIntegerUniqueKeyRange(dumperContext, tableRecordsCount, jobItemContext);
126             }
127             if (PipelineJdbcUtils.isStringColumn(firstColumnDataType)) {
128                 // TODO Support string unique key table splitting. Ascii characters ordering are different in different versions of databases.
129                 return Collections.singleton(new StringPrimaryKeyIngestPosition(null, null));
130             }
131         }
132         return Collections.singleton(new UnsupportedKeyIngestPosition());
133     }
134     
135     private Collection<IngestPosition> getPositionByIntegerUniqueKeyRange(final InventoryDumperContext dumperContext, final long tableRecordsCount, final TransmissionJobItemContext jobItemContext) {
136         if (0L == tableRecordsCount) {
137             return Collections.singletonList(new IntegerPrimaryKeyIngestPosition(0L, 0L));
138         }
139         Collection<IngestPosition> result = new LinkedList<>();
140         Range<Long> uniqueKeyValuesRange = getUniqueKeyValuesRange(jobItemContext, dumperContext);
141         int shardingSize = jobItemContext.getJobProcessContext().getProcessConfiguration().getRead().getShardingSize();
142         long splitCount = tableRecordsCount / shardingSize + (tableRecordsCount % shardingSize > 0L ? 1 : 0);
143         long interval = (uniqueKeyValuesRange.getMaximum() - uniqueKeyValuesRange.getMinimum()) / splitCount;
144         IntervalToRangeIterator rangeIterator = new IntervalToRangeIterator(uniqueKeyValuesRange.getMinimum(), uniqueKeyValuesRange.getMaximum(), interval);
145         while (rangeIterator.hasNext()) {
146             Range<Long> range = rangeIterator.next();
147             result.add(new IntegerPrimaryKeyIngestPosition(range.getMinimum(), range.getMaximum()));
148         }
149         return result;
150     }
151     
152     private Range<Long> getUniqueKeyValuesRange(final TransmissionJobItemContext jobItemContext, final InventoryDumperContext dumperContext) {
153         String uniqueKey = dumperContext.getUniqueKeyColumns().get(0).getName();
154         PipelinePrepareSQLBuilder pipelineSQLBuilder = new PipelinePrepareSQLBuilder(jobItemContext.getJobConfig().getSourceDatabaseType());
155         String sql = pipelineSQLBuilder.buildUniqueKeyMinMaxValuesSQL(
156                 dumperContext.getCommonContext().getTableAndSchemaNameMapper().getSchemaName(dumperContext.getLogicTableName()), dumperContext.getActualTableName(), uniqueKey);
157         try (
158                 Connection connection = sourceDataSource.getConnection();
159                 Statement statement = connection.createStatement();
160                 ResultSet resultSet = statement.executeQuery(sql)) {
161             resultSet.next();
162             return Range.of(resultSet.getLong(1), resultSet.getLong(2));
163         } catch (final SQLException ex) {
164             throw new SplitPipelineJobByUniqueKeyException(dumperContext.getActualTableName(), uniqueKey, ex);
165         }
166     }
167     
168     private InventoryDumperContext createPrimaryKeySplitDumperContext(final InventoryDumperContext dumperContext, final IngestPosition position,
169                                                                       final int shardingItem, final int batchSize, final JobRateLimitAlgorithm rateLimitAlgorithm) {
170         InventoryDumperContext result = new InventoryDumperContext(dumperContext.getCommonContext());
171         result.getCommonContext().setPosition(position);
172         result.setShardingItem(shardingItem);
173         result.setActualTableName(dumperContext.getActualTableName());
174         result.setLogicTableName(dumperContext.getLogicTableName());
175         result.setUniqueKeyColumns(dumperContext.getUniqueKeyColumns());
176         result.setInsertColumnNames(dumperContext.getInsertColumnNames());
177         result.setBatchSize(batchSize);
178         result.setRateLimitAlgorithm(rateLimitAlgorithm);
179         return result;
180     }
