55  /**
56   * Create rule statement assert.
57   */
58  @NoArgsConstructor(access = AccessLevel.PRIVATE)
59  public final class CreateRuleStatementAssert {
60      
61      /**
62       * Assert create rule statement is correct with expected parser result.
63       *
64       * @param assertContext assert context
65       * @param actual actual create rule statement
66       * @param expected expected create rule statement test case
67       */
68      public static void assertIs(final SQLCaseAssertContext assertContext, final CreateRuleStatement actual, final SQLParserTestCase expected) {
69          if (actual instanceof CreateEncryptRuleStatement) {
70              CreateEncryptRuleStatementAssert.assertIs(assertContext, (CreateEncryptRuleStatement) actual, (CreateEncryptRuleStatementTestCase) expected);
71          } else if (actual instanceof CreateReadwriteSplittingRuleStatement) {
72              CreateReadwriteSplittingRuleStatementAssert.assertIs(assertContext, (CreateReadwriteSplittingRuleStatement) actual, (CreateReadwriteSplittingRuleStatementTestCase) expected);
73          } else if (actual instanceof CreateShardingTableReferenceRuleStatement) {
74              CreateShardingTableReferenceRuleStatementAssert.assertIs(assertContext, (CreateShardingTableReferenceRuleStatement) actual, (CreateShardingTableReferenceRuleStatementTestCase) expected);
75          } else if (actual instanceof CreateBroadcastTableRuleStatement) {
76              CreateBroadcastTableRuleStatementAssert.assertIs(assertContext, (CreateBroadcastTableRuleStatement) actual, (CreateBroadcastTableRuleStatementTestCase) expected);
77          } else if (actual instanceof CreateShardingTableRuleStatement) {
78              CreateShardingTableRuleStatementAssert.assertIs(assertContext, (CreateShardingTableRuleStatement) actual, expected);
79          } else if (actual instanceof CreateShadowRuleStatement) {
80              CreateShadowRuleStatementAssert.assertIs(assertContext, (CreateShadowRuleStatement) actual, (CreateShadowRuleStatementTestCase) expected);
81          } else if (actual instanceof CreateDefaultShardingStrategyStatement) {
82              CreateDefaultShardingStrategyStatementAssert.assertIs(assertContext, (CreateDefaultShardingStrategyStatement) actual, (CreateDefaultShardingStrategyStatementTestCase) expected);
83          } else if (actual instanceof CreateDefaultShadowAlgorithmStatement) {
84              CreateDefaultShadowAlgorithmStatementAssert.assertIs(assertContext, (CreateDefaultShadowAlgorithmStatement) actual, (CreateDefaultShadowAlgorithmStatementTestCase) expected);
85          } else if (actual instanceof CreateMaskRuleStatement) {
86              CreateMaskRuleStatementAssert.assertIs(assertContext, (CreateMaskRuleStatement) actual, (CreateMaskRuleStatementTestCase) expected);
87          } else if (actual instanceof SetDefaultSingleTableStorageUnitStatement) {
88              SetDefaultSingleTableStorageUnitStatementAssert.assertIs(assertContext, (SetDefaultSingleTableStorageUnitStatement) actual, (SetDefaultSingleTableStorageUnitStatementTestCase) expected);
89          }
90      }
