18  package org.apache.shardingsphere.encrypt.rewrite.token.comparator;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.encrypt.rule.EncryptRule;
23  import org.apache.shardingsphere.encrypt.spi.EncryptAlgorithm;
24  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.column.ColumnSegment;
25  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.BinaryOperationExpression;
26  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.bound.ColumnSegmentBoundInfo;
27  
28  import java.util.Collection;
29  
30  /**
31   * Join conditions encryptor comparator.
32   */
33  @NoArgsConstructor(access = AccessLevel.PRIVATE)
34  public final class JoinConditionsEncryptorComparator {
35      
36      /**
37       * Compare whether same encryptor.
38       *
39       * @param joinConditions join conditions
40       * @param encryptRule encrypt rule
41       * @return same encryptors or not
42       */
43      public static boolean isSame(final Collection<BinaryOperationExpression> joinConditions, final EncryptRule encryptRule) {
44          for (BinaryOperationExpression each : joinConditions) {
45              if (!(each.getLeft() instanceof ColumnSegment) || !(each.getRight() instanceof ColumnSegment)) {
46                  continue;
47              }
48              ColumnSegmentBoundInfo leftColumnInfo = ((ColumnSegment) each.getLeft()).getColumnBoundInfo();
49              EncryptAlgorithm leftColumnEncryptor = encryptRule.findQueryEncryptor(leftColumnInfo.getOriginalTable().getValue(), leftColumnInfo.getOriginalColumn().getValue()).orElse(null);
50              ColumnSegmentBoundInfo rightColumnInfo = ((ColumnSegment) each.getRight()).getColumnBoundInfo();
51              EncryptAlgorithm rightColumnEncryptor = encryptRule.findQueryEncryptor(rightColumnInfo.getOriginalTable().getValue(), rightColumnInfo.getOriginalColumn().getValue()).orElse(null);
52              if (!EncryptorComparator.isSame(leftColumnEncryptor, rightColumnEncryptor)) {
53                  return false;
54              }
55          }
56          return true;
57      }
58  }
