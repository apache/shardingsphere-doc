18  package org.apache.shardingsphere.data.pipeline.mysql.datasource;
19  
20  import org.apache.shardingsphere.data.pipeline.core.checker.DialectPipelineDatabaseVariableChecker;
21  import org.apache.shardingsphere.infra.database.core.exception.CheckDatabaseEnvironmentFailedException;
22  import org.apache.shardingsphere.infra.database.core.exception.UnexpectedVariableValueException;
23  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
24  
25  import javax.sql.DataSource;
26  import java.sql.Connection;
27  import java.sql.PreparedStatement;
28  import java.sql.ResultSet;
29  import java.sql.SQLException;
30  import java.util.HashMap;
31  import java.util.Map;
32  import java.util.Map.Entry;
33  import java.util.stream.Collectors;
34  
35  /**
36   * Pipeline database variable checker for MySQL.
37   */
38  public final class MySQLPipelineDatabaseVariableChecker implements DialectPipelineDatabaseVariableChecker {
39      
40      private static final Map<String, String> REQUIRED_VARIABLES = new HashMap<>(3, 1F);
41      
42      private static final String SHOW_VARIABLES_SQL;
43      
44      static {
45          REQUIRED_VARIABLES.put("LOG_BIN", "ON");
46          REQUIRED_VARIABLES.put("BINLOG_FORMAT", "ROW");
47          // It does not exist in all versions of MySQL
48          REQUIRED_VARIABLES.put("BINLOG_ROW_IMAGE", "FULL");
49          SHOW_VARIABLES_SQL = String.format("SHOW VARIABLES WHERE Variable_name IN (%s)", REQUIRED_VARIABLES.keySet().stream().map(each -> "?").collect(Collectors.joining(",")));
50      }
51      
52      @Override
53      public void check(final DataSource dataSource) {
54          try (
55                  Connection connection = dataSource.getConnection();
56                  PreparedStatement preparedStatement = connection.prepareStatement(SHOW_VARIABLES_SQL)) {
57              int parameterIndex = 1;
58              for (Entry<String, String> entry : REQUIRED_VARIABLES.entrySet()) {
59                  preparedStatement.setString(parameterIndex++, entry.getKey());
60              }
61              try (ResultSet resultSet = preparedStatement.executeQuery()) {
62                  while (resultSet.next()) {
63                      String variableName = resultSet.getString(1).toUpperCase();
64                      String expectedValue = REQUIRED_VARIABLES.get(variableName);
65                      String actualValue = resultSet.getString(2);
66                      ShardingSpherePreconditions.checkState(expectedValue.equalsIgnoreCase(actualValue), () -> new UnexpectedVariableValueException(variableName, expectedValue, actualValue));
67                  }
68              }
69          } catch (final SQLException ex) {
70              throw new CheckDatabaseEnvironmentFailedException(ex);
71          }
72      }
73      
74      @Override
75      public String getDatabaseType() {
76          return "MySQL";
77      }
78  }
