48  /**
49   * Assignment generator for encrypt.
50   */
51  @Setter
52  public final class EncryptAssignmentTokenGenerator implements CollectionSQLTokenGenerator<SQLStatementContext<?>>, EncryptRuleAware, DatabaseNameAware {
53      
54      private EncryptRule encryptRule;
55      
56      private String databaseName;
57      
58      @Override
59      public boolean isGenerateSQLToken(final SQLStatementContext<?> sqlStatementContext) {
60          return sqlStatementContext instanceof UpdateStatementContext || sqlStatementContext instanceof InsertStatementContext
61                  && InsertStatementHandler.getSetAssignmentSegment(((InsertStatementContext) sqlStatementContext).getSqlStatement()).isPresent();
62      }
63      
64      @Override
65      public Collection<EncryptAssignmentToken> generateSQLTokens(final SQLStatementContext<?> sqlStatementContext) {
66          Collection<EncryptAssignmentToken> result = new LinkedList<>();
67          String tableName = ((TableAvailable) sqlStatementContext).getAllTables().iterator().next().getTableName().getIdentifier().getValue();
68          String schemaName = sqlStatementContext.getTablesContext().getSchemaName().orElseGet(() -> DatabaseTypeEngine.getDefaultSchemaName(sqlStatementContext.getDatabaseType(), databaseName));
69          for (AssignmentSegment each : getSetAssignmentSegment(sqlStatementContext.getSqlStatement()).getAssignments()) {
70              if (encryptRule.findEncryptor(tableName, each.getColumns().get(0).getIdentifier().getValue()).isPresent()) {
71                  generateSQLToken(schemaName, tableName, each).ifPresent(result::add);
72              }
73          }
74          return result;
75      }
76      
77      private SetAssignmentSegment getSetAssignmentSegment(final SQLStatement sqlStatement) {
78          if (sqlStatement instanceof InsertStatement) {
79              Optional<SetAssignmentSegment> result = InsertStatementHandler.getSetAssignmentSegment((InsertStatement) sqlStatement);
80              Preconditions.checkState(result.isPresent());
81              return result.get();
82          }
83          return ((UpdateStatement) sqlStatement).getSetAssignment();
84      }
85      
86      private Optional<EncryptAssignmentToken> generateSQLToken(final String schemaName, final String tableName, final AssignmentSegment assignmentSegment) {
87          if (assignmentSegment.getValue() instanceof ParameterMarkerExpressionSegment) {
88              return Optional.of(generateParameterSQLToken(tableName, assignmentSegment));
89          }
90          if (assignmentSegment.getValue() instanceof LiteralExpressionSegment) {
91              return Optional.of(generateLiteralSQLToken(schemaName, tableName, assignmentSegment));
92          }
93          return Optional.empty();
94      }
95      
96      private EncryptAssignmentToken generateParameterSQLToken(final String tableName, final AssignmentSegment assignmentSegment) {
97          EncryptParameterAssignmentToken result = new EncryptParameterAssignmentToken(assignmentSegment.getColumns().get(0).getStartIndex(), assignmentSegment.getStopIndex());
98          String columnName = assignmentSegment.getColumns().get(0).getIdentifier().getValue();
99          addCipherColumn(tableName, columnName, result);
100         addAssistedQueryColumn(tableName, columnName, result);
101         addLikeQueryColumn(tableName, columnName, result);
102         addPlainColumn(tableName, columnName, result);
103         return result;
104     }
105     
106     private void addCipherColumn(final String tableName, final String columnName, final EncryptParameterAssignmentToken token) {
107         token.addColumnName(encryptRule.getCipherColumn(tableName, columnName));
108     }
109     
110     private void addAssistedQueryColumn(final String tableName, final String columnName, final EncryptParameterAssignmentToken token) {
111         encryptRule.findAssistedQueryColumn(tableName, columnName).ifPresent(token::addColumnName);
112     }
113     
114     private void addLikeQueryColumn(final String tableName, final String columnName, final EncryptParameterAssignmentToken token) {
115         encryptRule.findLikeQueryColumn(tableName, columnName).ifPresent(token::addColumnName);
116     }
117     
118     private void addPlainColumn(final String tableName, final String columnName, final EncryptParameterAssignmentToken token) {
119         encryptRule.findPlainColumn(tableName, columnName).ifPresent(token::addColumnName);
120     }
121     
122     private EncryptAssignmentToken generateLiteralSQLToken(final String schemaName, final String tableName, final AssignmentSegment assignmentSegment) {
123         EncryptLiteralAssignmentToken result = new EncryptLiteralAssignmentToken(assignmentSegment.getColumns().get(0).getStartIndex(), assignmentSegment.getStopIndex());
124         addCipherAssignment(schemaName, tableName, assignmentSegment, result);
125         addAssistedQueryAssignment(schemaName, tableName, assignmentSegment, result);
126         addLikeAssignment(schemaName, tableName, assignmentSegment, result);
127         addPlainAssignment(tableName, assignmentSegment, result);
128         return result;
129     }
130     
131     private void addCipherAssignment(final String schemaName, final String tableName, final AssignmentSegment assignmentSegment, final EncryptLiteralAssignmentToken token) {
132         Object originalValue = ((LiteralExpressionSegment) assignmentSegment.getValue()).getLiterals();
133         Object cipherValue = encryptRule.getEncryptValues(databaseName, schemaName, tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue(),
134                 Collections.singletonList(originalValue)).iterator().next();
135         token.addAssignment(encryptRule.getCipherColumn(tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue()), cipherValue);
136     }
137     
138     private void addAssistedQueryAssignment(final String schemaName, final String tableName, final AssignmentSegment assignmentSegment, final EncryptLiteralAssignmentToken token) {
139         Object originalValue = ((LiteralExpressionSegment) assignmentSegment.getValue()).getLiterals();
140         Optional<String> assistedQueryColumn = encryptRule.findAssistedQueryColumn(tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue());
141         if (assistedQueryColumn.isPresent()) {
142             Object assistedQueryValue = encryptRule.getEncryptAssistedQueryValues(databaseName, schemaName,
143                     tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue(), Collections.singletonList(originalValue)).iterator().next();
144             token.addAssignment(assistedQueryColumn.get(), assistedQueryValue);
145         }
146     }
147     
148     private void addLikeAssignment(final String schemaName, final String tableName, final AssignmentSegment assignmentSegment, final EncryptLiteralAssignmentToken token) {
149         Object originalValue = ((LiteralExpressionSegment) assignmentSegment.getValue()).getLiterals();
150         Optional<String> assistedQueryColumn = encryptRule.findLikeQueryColumn(tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue());
151         if (assistedQueryColumn.isPresent()) {
152             Object assistedQueryValue = encryptRule.getEncryptLikeQueryValues(databaseName, schemaName,
153                     tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue(), Collections.singletonList(originalValue)).iterator().next();
154             token.addAssignment(assistedQueryColumn.get(), assistedQueryValue);
155         }
156     }
157     
158     private void addPlainAssignment(final String tableName, final AssignmentSegment assignmentSegment, final EncryptLiteralAssignmentToken token) {
159         Object originalValue = ((LiteralExpressionSegment) assignmentSegment.getValue()).getLiterals();
160         encryptRule.findPlainColumn(tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue()).ifPresent(optional -> token.addAssignment(optional, originalValue));
161     }
