57  @ExtendWith(AutoMockExtension.class)
58  @StaticMockSettings({ProxyContext.class, DatabaseConnectorFactory.class})
59  class TransactionBackendHandlerFactoryTest {
60      
61      @SuppressWarnings("unchecked")
62      @Test
63      void assertTransactionBackendHandlerReturnedWhenTCLStatementInstanceOfCommitStatement() {
64          ConnectionSession connectionSession = mock(ConnectionSession.class, Answers.RETURNS_DEEP_STUBS);
65          BackendConnection backendConnection = mock(BackendConnection.class);
66          when(backendConnection.getConnectionSession()).thenReturn(connectionSession);
67          when(connectionSession.getBackendConnection()).thenReturn(backendConnection);
68          SQLStatementContext<CommitStatement> context = mock(SQLStatementContext.class);
69          when(context.getSqlStatement()).thenReturn(mock(CommitStatement.class));
70          ContextManager contextManager = mockContextManager();
71          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
72          ProxyBackendHandler proxyBackendHandler = TransactionBackendHandlerFactory.newInstance(context, null, connectionSession);
73          assertThat(proxyBackendHandler, instanceOf(TransactionBackendHandler.class));
74          TransactionBackendHandler transactionBackendHandler = (TransactionBackendHandler) proxyBackendHandler;
75          assertFieldOfInstance(transactionBackendHandler, "operationType", is(TransactionOperationType.COMMIT));
76          assertFieldOfInstance(getBackendTransactionManager(transactionBackendHandler), "connection", is(backendConnection));
77      }
78      
79      @Test
80      void assertTransactionBackendHandlerReturnedWhenTCLStatementInstanceOfRollbackStatement() {
81          ConnectionSession connectionSession = mock(ConnectionSession.class, Answers.RETURNS_DEEP_STUBS);
82          BackendConnection backendConnection = mock(BackendConnection.class);
83          when(backendConnection.getConnectionSession()).thenReturn(connectionSession);
84          when(connectionSession.getBackendConnection()).thenReturn(backendConnection);
85          SQLStatementContext<RollbackStatement> context = mock(SQLStatementContext.class);
86          when(context.getSqlStatement()).thenReturn(mock(RollbackStatement.class));
87          ContextManager contextManager = mockContextManager();
88          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
89          ProxyBackendHandler proxyBackendHandler = TransactionBackendHandlerFactory.newInstance(context, null, connectionSession);
90          assertThat(proxyBackendHandler, instanceOf(TransactionBackendHandler.class));
91          TransactionBackendHandler transactionBackendHandler = (TransactionBackendHandler) proxyBackendHandler;
92          assertFieldOfInstance(transactionBackendHandler, "operationType", is(TransactionOperationType.ROLLBACK));
93          assertFieldOfInstance(getBackendTransactionManager(transactionBackendHandler), "connection", is(backendConnection));
94      }
95      
96      private ContextManager mockContextManager() {
97          ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
98          when(result.getMetaDataContexts().getMetaData().getGlobalRuleMetaData()).thenReturn(new ShardingSphereRuleMetaData(Collections.singleton(mock(TransactionRule.class))));
99          return result;
100     }
101     
102     @SuppressWarnings("unchecked")
103     @Test
104     void assertBroadcastBackendHandlerReturnedWhenTCLStatementNotHit() {
105         SQLStatementContext<TCLStatement> context = mock(SQLStatementContext.class);
106         when(context.getSqlStatement()).thenReturn(mock(TCLStatement.class));
107         DatabaseConnectorFactory mockFactory = mock(DatabaseConnectorFactory.class);
108         when(DatabaseConnectorFactory.getInstance()).thenReturn(mockFactory);
109         when(mockFactory.newInstance(any(QueryContext.class), nullable(BackendConnection.class), anyBoolean())).thenReturn(mock(DatabaseConnector.class));
110         assertThat(TransactionBackendHandlerFactory.newInstance(context, null, mock(ConnectionSession.class)), instanceOf(DatabaseConnector.class));
111     }
112     
113     @SuppressWarnings("unchecked")
114     @SneakyThrows(ReflectiveOperationException.class)
115     private <S, T> void assertFieldOfInstance(final S classInstance, final String fieldName, final Matcher<T> matcher) {
116         T value = (T) Plugins.getMemberAccessor().get(classInstance.getClass().getDeclaredField(fieldName), classInstance);
117         assertThat(value, matcher);
118     }
119     
120     @SneakyThrows(ReflectiveOperationException.class)
121     private BackendTransactionManager getBackendTransactionManager(final TransactionBackendHandler transactionBackendHandler) {
122         return (BackendTransactionManager) Plugins.getMemberAccessor().get(transactionBackendHandler.getClass().getDeclaredField("backendTransactionManager"), transactionBackendHandler);
123     }
