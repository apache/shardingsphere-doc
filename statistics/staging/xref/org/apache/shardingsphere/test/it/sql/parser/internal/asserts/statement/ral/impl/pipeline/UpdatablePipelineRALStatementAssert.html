59  /**
60   * Updatable pipeline RAL statement assert.
61   */
62  @NoArgsConstructor(access = AccessLevel.PRIVATE)
63  public final class UpdatablePipelineRALStatementAssert {
64      
65      /**
66       * Assert updatable pipeline RAL statement is correct with expected parser result.
67       *
68       * @param assertContext assert context
69       * @param actual actual updatable pipeline RAL statement
70       * @param expected expected updatable pipeline RAL statement test case
71       */
72      public static void assertIs(final SQLCaseAssertContext assertContext, final UpdatablePipelineRALStatement actual, final SQLParserTestCase expected) {
73          // TODO add more test case
74          if (actual instanceof MigrateTableStatement) {
75              MigrateTableStatementAssert.assertIs(assertContext, (MigrateTableStatement) actual, (MigrateTableStatementTestCase) expected);
76          } else if (actual instanceof StopMigrationStatement) {
77              StopMigrationStatementAssert.assertIs(assertContext, (StopMigrationStatement) actual, (StopMigrationStatementTestCase) expected);
78          } else if (actual instanceof CommitMigrationStatement) {
79              CommitMigrationStatementAssert.assertIs(assertContext, (CommitMigrationStatement) actual, (CommitMigrationStatementTestCase) expected);
80          } else if (actual instanceof RollbackMigrationStatement) {
81              RollbackMigrationStatementAssert.assertIs(assertContext, (RollbackMigrationStatement) actual, (RollbackMigrationStatementTestCase) expected);
82          } else if (actual instanceof StartMigrationStatement) {
83              StartMigrationStatementAssert.assertIs(assertContext, (StartMigrationStatement) actual, (StartMigrationStatementTestCase) expected);
84          } else if (actual instanceof RegisterMigrationSourceStorageUnitStatement) {
85              RegisterMigrationSourceStorageUnitStatementAssert.assertIs(
86                      assertContext, (RegisterMigrationSourceStorageUnitStatement) actual, (RegisterMigrationSourceStorageUnitStatementTestCase) expected);
87          } else if (actual instanceof UnregisterMigrationSourceStorageUnitStatement) {
88              UnregisterMigrationSourceStorageUnitStatementAssert.assertIs(
89                      assertContext, (UnregisterMigrationSourceStorageUnitStatement) actual, (UnregisterMigrationSourceStorageUnitStatementTestCase) expected);
90          } else if (actual instanceof CheckMigrationStatement) {
91              CheckMigrationStatementAssert.assertIs(assertContext, (CheckMigrationStatement) actual, (CheckMigrationStatementTestCase) expected);
92          } else if (actual instanceof StartMigrationCheckStatement) {
93              StartMigrationCheckStatementAssert.assertIs(assertContext, (StartMigrationCheckStatement) actual, (StartMigrationCheckStatementTestCase) expected);
94          } else if (actual instanceof StopMigrationCheckStatement) {
95              StopMigrationCheckStatementAssert.assertIs(assertContext, (StopMigrationCheckStatement) actual, (StopMigrationCheckStatementTestCase) expected);
96          } else if (actual instanceof DropStreamingStatement) {
97              DropStreamingStatementAssert.assertIs(assertContext, (DropStreamingStatement) actual, (DropStreamingStatementTestCase) expected);
98          }
99      }
