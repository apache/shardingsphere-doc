18  package org.apache.shardingsphere.readwritesplitting.distsql.handler.provider;
19  
20  import com.google.common.base.Strings;
21  import org.apache.shardingsphere.distsql.handler.ral.constant.DistSQLScriptConstants;
22  import org.apache.shardingsphere.distsql.handler.ral.query.ConvertRuleConfigurationProvider;
23  import org.apache.shardingsphere.infra.config.algorithm.AlgorithmConfiguration;
24  import org.apache.shardingsphere.infra.config.rule.RuleConfiguration;
25  import org.apache.shardingsphere.readwritesplitting.api.ReadwriteSplittingRuleConfiguration;
26  import org.apache.shardingsphere.readwritesplitting.api.rule.ReadwriteSplittingDataSourceRuleConfiguration;
27  
28  import java.util.Collection;
29  import java.util.Iterator;
30  import java.util.Map;
31  import java.util.Properties;
32  import java.util.TreeMap;
33  
34  /**
35   * Readwrite-splitting convert rule configuration provider.
36   */
37  public class ReadwriteSplittingConvertRuleConfigurationProvider implements ConvertRuleConfigurationProvider {
38      
39      @Override
40      public String convert(final RuleConfiguration ruleConfig) {
41          return getReadWriteSplittingDistSQL((ReadwriteSplittingRuleConfiguration) ruleConfig);
42      }
43      
44      private String getReadWriteSplittingDistSQL(final ReadwriteSplittingRuleConfiguration ruleConfig) {
45          if (ruleConfig.getDataSources().isEmpty()) {
46              return "";
47          }
48          StringBuilder result = new StringBuilder();
49          result.append(DistSQLScriptConstants.CREATE_READWRITE_SPLITTING_RULE);
50          Iterator<ReadwriteSplittingDataSourceRuleConfiguration> iterator = ruleConfig.getDataSources().iterator();
51          while (iterator.hasNext()) {
52              appendStaticReadWriteSplittingRule(ruleConfig.getLoadBalancers(), iterator.next(), result);
53              if (iterator.hasNext()) {
54                  result.append(DistSQLScriptConstants.COMMA);
55              }
56          }
57          result.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
58          return result.toString();
59      }
60      
61      private void appendStaticReadWriteSplittingRule(final Map<String, AlgorithmConfiguration> loadBalancers,
62                                                      final ReadwriteSplittingDataSourceRuleConfiguration dataSourceRuleConfig, final StringBuilder stringBuilder) {
63          String readDataSourceNames = getReadDataSourceNames(dataSourceRuleConfig.getReadDataSourceNames());
64          String transactionalReadQueryStrategy = dataSourceRuleConfig.getTransactionalReadQueryStrategy().name();
65          String loadBalancerType = getLoadBalancerType(loadBalancers.get(dataSourceRuleConfig.getLoadBalancerName()));
66          stringBuilder.append(String.format(DistSQLScriptConstants.READWRITE_SPLITTING_FOR_STATIC,
67                  dataSourceRuleConfig.getName(), dataSourceRuleConfig.getWriteDataSourceName(), readDataSourceNames, transactionalReadQueryStrategy, loadBalancerType));
68      }
69      
70      private String getReadDataSourceNames(final Collection<String> readDataSourceNames) {
71          StringBuilder result = new StringBuilder();
72          Iterator<String> iterator = readDataSourceNames.iterator();
73          while (iterator.hasNext()) {
74              result.append(String.format(DistSQLScriptConstants.READ_RESOURCE, iterator.next()));
75              if (iterator.hasNext()) {
76                  result.append(DistSQLScriptConstants.COMMA);
77              }
78          }
79          return result.toString();
80      }
81      
82      private String getLoadBalancerType(final AlgorithmConfiguration algorithmConfig) {
83          StringBuilder result = new StringBuilder();
84          String loadBalancerType = getAlgorithmType(algorithmConfig);
85          if (!Strings.isNullOrEmpty(loadBalancerType)) {
86              result.append(DistSQLScriptConstants.COMMA).append(System.lineSeparator()).append(loadBalancerType);
87          }
88          return result.toString();
89      }
90      
91      private String getAlgorithmType(final AlgorithmConfiguration algorithmConfig) {
92          StringBuilder result = new StringBuilder();
93          if (null == algorithmConfig) {
94              return result.toString();
95          }
96          String type = algorithmConfig.getType().toLowerCase();
97          if (algorithmConfig.getProps().isEmpty()) {
98              result.append(String.format(DistSQLScriptConstants.ALGORITHM_TYPE_WITHOUT_PROPS, type));
99          } else {
100             result.append(String.format(DistSQLScriptConstants.ALGORITHM_TYPE, type, getAlgorithmProperties(algorithmConfig.getProps())));
101         }
102         return result.toString();
103     }
104     
105     @SuppressWarnings({"unchecked", "rawtypes"})
106     private String getAlgorithmProperties(final Properties props) {
107         StringBuilder result = new StringBuilder();
108         Iterator<String> iterator = new TreeMap(props).keySet().iterator();
109         while (iterator.hasNext()) {
110             String key = iterator.next();
111             Object value = props.get(key);
112             if (null == value) {
113                 continue;
114             }
115             result.append(String.format(DistSQLScriptConstants.PROPERTY, key, value));
116             if (iterator.hasNext()) {
117                 result.append(DistSQLScriptConstants.COMMA).append(' ');
118             }
119         }
120         return result.toString();
121     }
122     
123     @Override
124     public String getType() {
125         return ReadwriteSplittingRuleConfiguration.class.getName();
126     }
127 }
