51  /**
52   * Compute node state changed watcher.
53   */
54  public final class ComputeNodeStateChangedWatcher implements GovernanceWatcher<GovernanceEvent> {
55      
56      @Override
57      public Collection<String> getWatchingKeys(final Collection<String> databaseNames) {
58          return Collections.singleton(ComputeNode.getComputeNodePath());
59      }
60      
61      @Override
62      public Collection<Type> getWatchingTypes() {
63          return Arrays.asList(Type.ADDED, Type.UPDATED, Type.DELETED);
64      }
65      
66      @Override
67      public Optional<GovernanceEvent> createGovernanceEvent(final DataChangedEvent event) {
68          String instanceId = ComputeNode.getInstanceIdByComputeNode(event.getKey());
69          if (!Strings.isNullOrEmpty(instanceId)) {
70              Optional<GovernanceEvent> result = createInstanceGovernanceEvent(event, instanceId);
71              if (result.isPresent()) {
72                  return result;
73              }
74          }
75          if (event.getKey().startsWith(ComputeNode.getOnlineInstanceNodePath())) {
76              return createInstanceEvent(event);
77          }
78          if (event.getKey().startsWith(ComputeNode.getShowProcessListTriggerNodePath())) {
79              return createReportLocalProcessesEvent(event);
80          }
81          if (event.getKey().startsWith(ComputeNode.getKillProcessTriggerNodePath())) {
82              return createKillLocalProcessEvent(event);
83          }
84          return Optional.empty();
85      }
86      
87      @SuppressWarnings("unchecked")
88      private Optional<GovernanceEvent> createInstanceGovernanceEvent(final DataChangedEvent event, final String instanceId) {
89          if (event.getKey().equals(ComputeNode.getInstanceStatusNodePath(instanceId)) && Type.DELETED != event.getType()) {
90              return Optional.of(new ComputeNodeInstanceStateChangedEvent(instanceId, event.getValue()));
91          }
92          if (event.getKey().equals(ComputeNode.getInstanceLabelsNodePath(instanceId)) && Type.DELETED != event.getType()) {
93              return Optional.of(new LabelsEvent(instanceId, Strings.isNullOrEmpty(event.getValue()) ? new ArrayList<>() : YamlEngine.unmarshal(event.getValue(), Collection.class)));
94          }
95          if (event.getKey().equals(ComputeNode.getInstanceWorkerIdNodePath(instanceId))) {
96              return Optional.of(new WorkerIdEvent(instanceId, Strings.isNullOrEmpty(event.getValue()) ? null : Integer.valueOf(event.getValue())));
97          }
98          return Optional.empty();
99      }
100     
101     private Optional<GovernanceEvent> createInstanceEvent(final DataChangedEvent event) {
102         Matcher matcher = getInstanceOnlinePathMatcher(event.getKey());
103         if (matcher.find()) {
104             ComputeNodeData computeNodeData = new YamlComputeNodeDataSwapper().swapToObject(YamlEngine.unmarshal(event.getValue(), YamlComputeNodeData.class));
105             InstanceMetaData instanceMetaData = InstanceMetaDataFactory.create(matcher.group(2),
106                     InstanceType.valueOf(matcher.group(1).toUpperCase()), computeNodeData.getAttribute(), computeNodeData.getVersion());
107             if (Type.ADDED == event.getType()) {
108                 return Optional.of(new InstanceOnlineEvent(instanceMetaData));
109             }
110             if (Type.DELETED == event.getType()) {
111                 return Optional.of(new InstanceOfflineEvent(instanceMetaData));
112             }
113         }
114         return Optional.empty();
115     }
116     
117     private Matcher getInstanceOnlinePathMatcher(final String onlineInstancePath) {
118         return Pattern.compile(ComputeNode.getOnlineInstanceNodePath() + "/([\\S]+)/([\\S]+)$", Pattern.CASE_INSENSITIVE).matcher(onlineInstancePath);
119     }
120     
121     private Optional<GovernanceEvent> createReportLocalProcessesEvent(final DataChangedEvent event) {
122         Matcher matcher = getShowProcessListTriggerMatcher(event);
123         if (!matcher.find()) {
124             return Optional.empty();
125         }
126         if (Type.ADDED == event.getType()) {
127             return Optional.of(new ReportLocalProcessesEvent(matcher.group(1), matcher.group(2)));
128         }
129         if (Type.DELETED == event.getType()) {
130             return Optional.of(new ReportLocalProcessesCompletedEvent(matcher.group(2)));
131         }
132         return Optional.empty();
133     }
134     
135     private Matcher getShowProcessListTriggerMatcher(final DataChangedEvent event) {
136         return Pattern.compile(ComputeNode.getShowProcessListTriggerNodePath() + "/([\\S]+):([\\S]+)$", Pattern.CASE_INSENSITIVE).matcher(event.getKey());
137     }
138     
139     private Optional<GovernanceEvent> createKillLocalProcessEvent(final DataChangedEvent event) {
140         Matcher matcher = getKillProcessTriggerMatcher(event);
141         if (!matcher.find()) {
142             return Optional.empty();
143         }
144         if (Type.ADDED == event.getType()) {
145             return Optional.of(new KillLocalProcessEvent(matcher.group(1), matcher.group(2)));
146         }
147         if (Type.DELETED == event.getType()) {
148             return Optional.of(new KillLocalProcessCompletedEvent(matcher.group(2)));
149         }
150         return Optional.empty();
151     }
152     
153     private Matcher getKillProcessTriggerMatcher(final DataChangedEvent event) {
154         Pattern pattern = Pattern.compile(ComputeNode.getKillProcessTriggerNodePath() + "/([\\S]+):([\\S]+)$", Pattern.CASE_INSENSITIVE);
155         return pattern.matcher(event.getKey());
156     }
