18  package org.apache.shardingsphere.infra.database.core.connector.url;
19  
20  import org.junit.jupiter.api.Test;
21  
22  import static org.hamcrest.CoreMatchers.is;
23  import static org.hamcrest.MatcherAssert.assertThat;
24  import static org.junit.jupiter.api.Assertions.assertThrows;
25  import static org.junit.jupiter.api.Assertions.assertTrue;
26  
27  class StandardJdbcUrlParserTest {
28      
29      @Test
30      void assertParseSimpleJdbcUrl() {
31          JdbcUrl actual = new StandardJdbcUrlParser().parse("jdbc:mock://127.0.0.1/");
32          assertThat(actual.getHostname(), is("127.0.0.1"));
33          assertThat(actual.getPort(), is(-1));
34          assertThat(actual.getDatabase(), is(""));
35          assertTrue(actual.getQueryProperties().isEmpty());
36      }
37      
38      @Test
39      void assertParseMySQLJdbcUrl() {
40          JdbcUrl actual = new StandardJdbcUrlParser().parse("jdbc:mysql://127.0.0.1:3306/demo_ds?serverTimezone=UTC&useSSL=false&sessionVariables=group_concat_max_len=204800,SQL_SAFE_UPDATES=0");
41          assertThat(actual.getHostname(), is("127.0.0.1"));
42          assertThat(actual.getPort(), is(3306));
43          assertThat(actual.getDatabase(), is("demo_ds"));
44          assertThat(actual.getQueryProperties().size(), is(3));
45          assertThat(actual.getQueryProperties().get("serverTimezone"), is("UTC"));
46          assertThat(actual.getQueryProperties().get("useSSL"), is(Boolean.FALSE.toString()));
47          assertThat(actual.getQueryProperties().get("sessionVariables"), is("group_concat_max_len=204800,SQL_SAFE_UPDATES=0"));
48      }
49      
50      @Test
51      void assertParseMySQLJdbcUrlWithReplication() {
52          JdbcUrl actual = new StandardJdbcUrlParser().parse("jdbc:mysql:replication://master-ip:3306,slave-1-ip:3306,slave-2-ip:3306/demo_ds?useUnicode=true");
53          assertThat(actual.getHostname(), is("master-ip"));
54          assertThat(actual.getPort(), is(3306));
55          assertThat(actual.getDatabase(), is("demo_ds"));
56          assertThat(actual.getQueryProperties().size(), is(1));
57          assertThat(actual.getQueryProperties().get("useUnicode"), is(Boolean.TRUE.toString()));
58      }
59      
60      @Test
61      void assertParsePostgreSQLJdbcUrl() {
62          JdbcUrl actual = new StandardJdbcUrlParser().parse("jdbc:postgresql://127.0.0.1:5432/demo_ds?prepareThreshold=1&preferQueryMode=extendedForPrepared");
63          assertThat(actual.getHostname(), is("127.0.0.1"));
64          assertThat(actual.getPort(), is(5432));
65          assertThat(actual.getDatabase(), is("demo_ds"));
66          assertThat(actual.getQueryProperties().size(), is(2));
67          assertThat(actual.getQueryProperties().get("prepareThreshold"), is("1"));
68          assertThat(actual.getQueryProperties().get("preferQueryMode"), is("extendedForPrepared"));
69      }
70      
71      @Test
72      void assertParseMicrosoftSQLServerJdbcUrl() {
73          JdbcUrl actual = new StandardJdbcUrlParser().parse("jdbc:microsoft:sqlserver://127.0.0.1:3306/demo_ds");
74          assertThat(actual.getHostname(), is("127.0.0.1"));
75          assertThat(actual.getPort(), is(3306));
76          assertThat(actual.getDatabase(), is("demo_ds"));
77          assertTrue(actual.getQueryProperties().isEmpty());
78      }
79      
80      @Test
81      void assertParseIncorrectURL() {
82          assertThrows(UnrecognizedDatabaseURLException.class, () -> new StandardJdbcUrlParser().parse("jdbc:h2:mem:test;DB_CLOSE_DELAY=-1;DATABASE_TO_UPPER=false;MODE=MySQL"));
83      }
84      
85      @Test
86      void assertParseTestContainersJDBCUrl() {
87          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:mysql:5.7.34:///demo_ds").getDatabase(), is("demo_ds"));
88          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:postgresql:9.6.8:///demo_ds").getDatabase(), is("demo_ds"));
89          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:postgis:9.6-2.5:///demo_ds").getDatabase(), is("demo_ds"));
90          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:timescaledb:2.1.0-pg13:///demo_ds").getDatabase(), is("demo_ds"));
91          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:trino:352://localhost/memory/default").getDatabase(), is("memory/default"));
92          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:cockroach:v21.2.3:///demo_ds").getDatabase(), is("demo_ds"));
93          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:tidb:v6.1.0:///demo_ds").getDatabase(), is("demo_ds"));
94          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:mysql:5.7.34:///demo_ds?TC_INITSCRIPT=somepath/init_mysql.sql")
95                  .getQueryProperties().get("TC_INITSCRIPT"), is("somepath/init_mysql.sql"));
96          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:mysql:5.7.34:///demo_ds?TC_INITSCRIPT=file:src/main/resources/init_mysql.sql")
97                  .getQueryProperties().get("TC_INITSCRIPT"), is("file:src/main/resources/init_mysql.sql"));
98          assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:mysql:5.7.34:///demo_ds?TC_INITFUNCTION=org.testcontainers.jdbc.JDBCDriverTest::sampleInitFunction")
99                  .getQueryProperties().get("TC_INITFUNCTION"), is("org.testcontainers.jdbc.JDBCDriverTest::sampleInitFunction"));
100         assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:mysql:5.7.34:///demo_ds?TC_DAEMON=true").getQueryProperties().get("TC_DAEMON"), is("true"));
101         assertThat(new StandardJdbcUrlParser().parse("jdbc:tc:postgresql:9.6.8:///demo_ds?TC_TMPFS=/testtmpfs:rw").getQueryProperties().get("TC_TMPFS"), is("/testtmpfs:rw"));
102     }
103 }
