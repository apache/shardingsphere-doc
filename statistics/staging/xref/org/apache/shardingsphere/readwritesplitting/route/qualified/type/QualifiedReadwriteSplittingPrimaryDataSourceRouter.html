18  package org.apache.shardingsphere.readwritesplitting.route.qualified.type;
19  
20  import org.apache.shardingsphere.infra.binder.context.statement.SQLStatementContext;
21  import org.apache.shardingsphere.infra.binder.context.statement.dml.SelectStatementContext;
22  import org.apache.shardingsphere.infra.hint.HintManager;
23  import org.apache.shardingsphere.infra.hint.HintValueContext;
24  import org.apache.shardingsphere.readwritesplitting.route.qualified.QualifiedReadwriteSplittingDataSourceRouter;
25  import org.apache.shardingsphere.readwritesplitting.rule.ReadwriteSplittingDataSourceGroupRule;
26  import org.apache.shardingsphere.sql.parser.sql.common.statement.SQLStatement;
27  import org.apache.shardingsphere.sql.parser.sql.common.statement.dml.SelectStatement;
28  
29  /**
30   * Qualified data source primary router for readwrite-splitting.
31   */
32  public final class QualifiedReadwriteSplittingPrimaryDataSourceRouter implements QualifiedReadwriteSplittingDataSourceRouter {
33      
34      @Override
35      public boolean isQualified(final SQLStatementContext sqlStatementContext, final ReadwriteSplittingDataSourceGroupRule rule, final HintValueContext hintValueContext) {
36          return isPrimaryRoute(sqlStatementContext, hintValueContext);
37      }
38      
39      private boolean isPrimaryRoute(final SQLStatementContext sqlStatementContext, final HintValueContext hintValueContext) {
40          return isWriteRouteStatement(sqlStatementContext) || isHintWriteRouteOnly(hintValueContext);
41      }
42      
43      private boolean isWriteRouteStatement(final SQLStatementContext sqlStatementContext) {
44          SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
45          return containsLockSegment(sqlStatement) || containsLastInsertIdProjection(sqlStatementContext) || !(sqlStatement instanceof SelectStatement);
46      }
47      
48      private boolean containsLockSegment(final SQLStatement sqlStatement) {
49          return sqlStatement instanceof SelectStatement && ((SelectStatement) sqlStatement).getLock().isPresent();
50      }
51      
52      private boolean containsLastInsertIdProjection(final SQLStatementContext sqlStatementContext) {
53          return sqlStatementContext instanceof SelectStatementContext && ((SelectStatementContext) sqlStatementContext).getProjectionsContext().isContainsLastInsertIdProjection();
54      }
55      
56      private boolean isHintWriteRouteOnly(final HintValueContext hintValueContext) {
57          return HintManager.isWriteRouteOnly() || hintValueContext.isWriteRouteOnly();
58      }
59      
60      @Override
61      public String route(final ReadwriteSplittingDataSourceGroupRule rule) {
62          return rule.getWriteDataSource();
63      }
64  }
