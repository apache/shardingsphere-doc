45  /**
46   * MySQL binlog event packet decoder.
47   */
48  @Slf4j
49  public final class MySQLBinlogEventPacketDecoder extends ByteToMessageDecoder {
50      
51      private final BinlogContext binlogContext;
52      
53      public MySQLBinlogEventPacketDecoder(final int checksumLength, final Map<Long, MySQLBinlogTableMapEventPacket> tableMap) {
54          binlogContext = new BinlogContext(checksumLength, tableMap);
55      }
56      
57      @Override
58      protected void decode(final ChannelHandlerContext ctx, final ByteBuf in, final List<Object> out) {
59          // readable bytes must greater + statusCode(1b) + header-length(19b) +
60          while (in.readableBytes() >= 1 + MySQLBinlogEventHeader.MYSQL_BINLOG_EVENT_HEADER_LENGTH) {
61              in.markReaderIndex();
62              MySQLPacketPayload payload = new MySQLPacketPayload(in, ctx.channel().attr(CommonConstants.CHARSET_ATTRIBUTE_KEY).get());
63              checkError(payload);
64              MySQLBinlogEventHeader binlogEventHeader = new MySQLBinlogEventHeader(payload, binlogContext.getChecksumLength());
65              // make sure event has complete body
66              if (in.readableBytes() < binlogEventHeader.getEventSize() - MySQLBinlogEventHeader.MYSQL_BINLOG_EVENT_HEADER_LENGTH) {
67                  log.debug("the event body is not complete, event size={}, readable bytes={}", binlogEventHeader.getEventSize(), in.readableBytes());
68                  in.resetReaderIndex();
69                  break;
70              }
71              Optional.ofNullable(decodeEvent(payload, binlogEventHeader)).ifPresent(out::add);
72              skipChecksum(binlogEventHeader.getEventType(), in);
73          }
74      }
75      
76      private AbstractBinlogEvent decodeEvent(final MySQLPacketPayload payload, final MySQLBinlogEventHeader binlogEventHeader) {
77          MySQLBinlogEventType eventType = MySQLBinlogEventType.valueOf(binlogEventHeader.getEventType()).orElse(MySQLBinlogEventType.UNKNOWN_EVENT);
78          switch (eventType) {
79              case ROTATE_EVENT:
80                  decodeRotateEvent(binlogEventHeader, payload);
81                  return null;
82              case FORMAT_DESCRIPTION_EVENT:
83                  MySQLBinlogFormatDescriptionEventPacket formatDescriptionEventPacket = new MySQLBinlogFormatDescriptionEventPacket(binlogEventHeader, payload);
84                  // MySQL mgr checksum length is 0, but the event ends up with 4 extra bytes, need to skip them.
85                  int readableBytes = payload.getByteBuf().readableBytes();
86                  if (binlogEventHeader.getChecksumLength() <= 0 && readableBytes > 0) {
87                      if (readableBytes != 4) {
88                          log.warn("the format description event has extra bytes, readable bytes length={}, binlogEventHeader={}, formatDescriptionEvent={}", readableBytes, binlogEventHeader,
89                                  formatDescriptionEventPacket);
90                      }
91                      payload.getByteBuf().skipBytes(readableBytes);
92                  }
93                  return null;
94              case TABLE_MAP_EVENT:
95                  decodeTableMapEvent(binlogEventHeader, payload);
96                  return null;
97              case WRITE_ROWS_EVENT_V1:
98              case WRITE_ROWS_EVENT_V2:
99                  return decodeWriteRowsEventV2(binlogEventHeader, payload);
100             case UPDATE_ROWS_EVENT_V1:
101             case UPDATE_ROWS_EVENT_V2:
102                 return decodeUpdateRowsEventV2(binlogEventHeader, payload);
103             case DELETE_ROWS_EVENT_V1:
104             case DELETE_ROWS_EVENT_V2:
105                 return decodeDeleteRowsEventV2(binlogEventHeader, payload);
106             default:
107                 PlaceholderEvent result = createPlaceholderEvent(binlogEventHeader);
108                 int remainDataLength = binlogEventHeader.getEventSize() + 1 - binlogEventHeader.getChecksumLength() - payload.getByteBuf().readerIndex();
109                 if (remainDataLength > 0) {
110                     payload.skipReserved(remainDataLength);
111                 }
112                 return result;
113         }
114     }
115     
116     private void checkError(final MySQLPacketPayload payload) {
117         int statusCode = payload.readInt1();
118         if (255 == statusCode) {
119             int errorNo = payload.readInt2();
120             payload.skipReserved(1);
121             String sqlState = payload.readStringFix(5);
122             throw new RuntimeException(String.format("Decode binlog event failed, errorCode: %d, sqlState: %s, errorMessage: %s", errorNo, sqlState, payload.readStringEOF()));
123         }
124         if (0 != statusCode) {
125             if (log.isDebugEnabled()) {
126                 log.debug("Illegal binlog status code {}, remaining packet \n{}", statusCode, readRemainPacket(payload));
127             }
128         }
129     }
130     
131     private String readRemainPacket(final MySQLPacketPayload payload) {
132         return ByteBufUtil.hexDump(payload.readStringFixByBytes(payload.getByteBuf().readableBytes()));
133     }
134     
135     private void skipChecksum(final int eventType, final ByteBuf in) {
136         if (0 < binlogContext.getChecksumLength() && MySQLBinlogEventType.FORMAT_DESCRIPTION_EVENT.getValue() != eventType) {
137             in.skipBytes(binlogContext.getChecksumLength());
138         }
139     }
140     
141     private void decodeRotateEvent(final MySQLBinlogEventHeader binlogEventHeader, final MySQLPacketPayload payload) {
142         MySQLBinlogRotateEventPacket rotateEventPacket = new MySQLBinlogRotateEventPacket(binlogEventHeader, payload);
143         binlogContext.setFileName(rotateEventPacket.getNextBinlogName());
144     }
145     
146     private void decodeTableMapEvent(final MySQLBinlogEventHeader binlogEventHeader, final MySQLPacketPayload payload) {
147         MySQLBinlogTableMapEventPacket tableMapEventPacket = new MySQLBinlogTableMapEventPacket(binlogEventHeader, payload);
148         binlogContext.putTableMapEvent(tableMapEventPacket.getTableId(), tableMapEventPacket);
149     }
150     
151     private DeleteRowsEvent decodeDeleteRowsEventV2(final MySQLBinlogEventHeader binlogEventHeader, final MySQLPacketPayload payload) {
152         MySQLBinlogRowsEventPacket rowsEventPacket = new MySQLBinlogRowsEventPacket(binlogEventHeader, payload);
153         rowsEventPacket.readRows(binlogContext.getTableMapEvent(rowsEventPacket.getTableId()), payload);
154         DeleteRowsEvent result = new DeleteRowsEvent();
155         initRowsEvent(result, binlogEventHeader, rowsEventPacket.getTableId());
156         result.setBeforeRows(rowsEventPacket.getRows());
157         return result;
158     }
159     
160     private UpdateRowsEvent decodeUpdateRowsEventV2(final MySQLBinlogEventHeader binlogEventHeader, final MySQLPacketPayload payload) {
161         MySQLBinlogRowsEventPacket rowsEventPacket = new MySQLBinlogRowsEventPacket(binlogEventHeader, payload);
162         rowsEventPacket.readRows(binlogContext.getTableMapEvent(rowsEventPacket.getTableId()), payload);
163         UpdateRowsEvent result = new UpdateRowsEvent();
164         initRowsEvent(result, binlogEventHeader, rowsEventPacket.getTableId());
165         result.setBeforeRows(rowsEventPacket.getRows());
166         result.setAfterRows(rowsEventPacket.getRows2());
167         return result;
168     }
169     
170     private WriteRowsEvent decodeWriteRowsEventV2(final MySQLBinlogEventHeader binlogEventHeader, final MySQLPacketPayload payload) {
171         MySQLBinlogRowsEventPacket rowsEventPacket = new MySQLBinlogRowsEventPacket(binlogEventHeader, payload);
172         rowsEventPacket.readRows(binlogContext.getTableMapEvent(rowsEventPacket.getTableId()), payload);
173         WriteRowsEvent result = new WriteRowsEvent();
174         initRowsEvent(result, binlogEventHeader, rowsEventPacket.getTableId());
175         result.setAfterRows(rowsEventPacket.getRows());
176         return result;
177     }
178     
179     private void initRowsEvent(final AbstractRowsEvent rowsEvent, final MySQLBinlogEventHeader binlogEventHeader, final long tableId) {
180         rowsEvent.setDatabaseName(binlogContext.getDatabaseName(tableId));
181         rowsEvent.setTableName(binlogContext.getTableName(tableId));
182         rowsEvent.setFileName(binlogContext.getFileName());
183         rowsEvent.setPosition(binlogEventHeader.getLogPos());
184         rowsEvent.setTimestamp(binlogEventHeader.getTimestamp());
185         rowsEvent.setServerId(binlogEventHeader.getServerId());
186     }
187     
188     private PlaceholderEvent createPlaceholderEvent(final MySQLBinlogEventHeader binlogEventHeader) {
189         PlaceholderEvent result = new PlaceholderEvent();
190         result.setFileName(binlogContext.getFileName());
191         result.setPosition(binlogEventHeader.getLogPos());
192         result.setTimestamp(binlogEventHeader.getTimestamp());
193         return result;
194     }
