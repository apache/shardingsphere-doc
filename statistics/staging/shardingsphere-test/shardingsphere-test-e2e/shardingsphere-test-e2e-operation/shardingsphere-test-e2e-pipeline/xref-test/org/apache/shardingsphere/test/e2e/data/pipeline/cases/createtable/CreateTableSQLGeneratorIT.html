63  @PipelineE2ESettings(database = {
64          @PipelineE2EDatabaseSettings(type = "MySQL", scenarioFiles = "env/scenario/create-table-generator/mysql/create-table-sql-generator.xml"),
65          @PipelineE2EDatabaseSettings(type = "PostgreSQL", scenarioFiles = "env/scenario/create-table-generator/postgresql/create-table-sql-generator.xml"),
66          @PipelineE2EDatabaseSettings(type = "openGauss", scenarioFiles = "env/scenario/create-table-generator/opengauss/create-table-sql-generator.xml")})
67  class CreateTableSQLGeneratorIT {
68      
69      private static final String DEFAULT_SCHEMA = "public";
70      
71      private static final String DEFAULT_DATABASE = "pipeline_it_0";
72      
73      private static final Pattern REPLACE_LINE_SPACE = Pattern.compile("\\s*|\t|\r|\n");
74      
75      private DockerStorageContainer storageContainer;
76      
77      @AfterEach
78      void stopContainer() {
79          storageContainer.stop();
80      }
81      
82      @ParameterizedTest(name = "{0}")
83      @EnabledIf("isEnabled")
84      @ArgumentsSource(PipelineE2ETestCaseArgumentsProvider.class)
85      void assertGenerateCreateTableSQL(final PipelineTestParameter testParam) throws SQLException {
86          startStorageContainer(testParam.getDatabaseType(), testParam.getStorageContainerImage());
87          CreateTableSQLGeneratorAssertionsRootEntity rootEntity = JAXB.unmarshal(
88                  Objects.requireNonNull(CreateTableSQLGeneratorIT.class.getClassLoader().getResource(testParam.getScenario())), CreateTableSQLGeneratorAssertionsRootEntity.class);
89          DataSource dataSource = storageContainer.createAccessDataSource(DEFAULT_DATABASE);
90          try (
91                  Connection connection = dataSource.getConnection();
92                  Statement statement = connection.createStatement()) {
93              int majorVersion = connection.getMetaData().getDatabaseMajorVersion();
94              for (CreateTableSQLGeneratorAssertionEntity each : rootEntity.getAssertions()) {
95                  statement.execute(each.getInput().getSql());
96                  Collection<String> actualDDLs = TypedSPILoader.getService(
97                          CreateTableSQLGenerator.class, testParam.getDatabaseType().getType()).generate(dataSource, DEFAULT_SCHEMA, each.getInput().getTable());
98                  assertSQL(actualDDLs, getVersionOutput(each.getOutputs(), majorVersion));
99              }
100         }
101     }
102     
103     private void startStorageContainer(final DatabaseType databaseType, final String storageContainerImage) {
104         StorageContainerConfiguration storageContainerConfig = createStorageContainerConfiguration(databaseType, storageContainerImage);
105         storageContainer = (DockerStorageContainer) StorageContainerFactory.newInstance(databaseType, storageContainerImage, "", storageContainerConfig);
106         storageContainer.start();
107     }
108     
109     private StorageContainerConfiguration createStorageContainerConfiguration(final DatabaseType databaseType, final String storageContainerImage) {
110         if (DatabaseTypeUtils.isMySQL(databaseType)) {
111             int majorVersion = new DockerImageVersion(storageContainerImage).getMajorVersion();
112             Map<String, String> mountedResources = Collections.singletonMap(String.format("/env/mysql/mysql%s/my.cnf", majorVersion), MySQLContainer.MYSQL_CONF_IN_CONTAINER);
113             return MySQLContainerConfigurationFactory.newInstance(null, null, mountedResources);
114         }
115         return StorageContainerConfigurationFactory.newInstance(databaseType);
116     }
117     
118     private void assertSQL(final Collection<String> actualSQL, final Collection<String> expectedSQL) {
119         Iterator<String> expected = expectedSQL.iterator();
120         for (String each : actualSQL) {
121             assertThat(REPLACE_LINE_SPACE.matcher(each).replaceAll(""), is(REPLACE_LINE_SPACE.matcher(expected.next()).replaceAll("")));
122         }
123     }
124     
125     private Collection<String> getVersionOutput(final Collection<CreateTableSQLGeneratorOutputEntity> outputs, final int majorVersion) {
126         Collection<String> result = new LinkedList<>();
127         for (CreateTableSQLGeneratorOutputEntity each : outputs) {
128             if ("default".equals(each.getVersion())) {
129                 result = each.getMultiSQL();
130             }
131             if (String.valueOf(majorVersion).equals(each.getVersion())) {
132                 return each.getMultiSQL();
133             }
134         }
135         return result;
136     }
137     
138     private static boolean isEnabled() {
139         return PipelineE2ECondition.isEnabled(new MySQLDatabaseType(), new PostgreSQLDatabaseType(), new OpenGaussDatabaseType());
140     }
