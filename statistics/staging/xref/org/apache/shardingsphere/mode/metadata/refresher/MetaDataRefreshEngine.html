53  /**
54   * Meta data refresh engine.
55   */
56  @RequiredArgsConstructor
57  public final class MetaDataRefreshEngine {
58      
59      private static final Collection<Class<? extends DDLStatement>> DDL_STATEMENT_CLASSES = Arrays.asList(CreateTableStatement.class, AlterTableStatement.class, DropTableStatement.class,
60              CreateViewStatement.class, AlterViewStatement.class, DropViewStatement.class, CreateIndexStatement.class, AlterIndexStatement.class, DropIndexStatement.class, CreateSchemaStatement.class,
61              AlterSchemaStatement.class, DropSchemaStatement.class, RenameTableStatement.class);
62      
63      private final MetaDataManagerPersistService metaDataManagerPersistService;
64      
65      private final ShardingSphereDatabase database;
66      
67      private final ConfigurationProperties props;
68      
69      /**
70       * Refresh meta data.
71       *
72       * @param sqlStatementContext SQL statement context
73       * @param routeUnits route units
74       * @throws SQLException SQL exception
75       */
76      @SuppressWarnings({"unchecked", "rawtypes"})
77      public void refresh(final SQLStatementContext sqlStatementContext, final Collection<RouteUnit> routeUnits) throws SQLException {
78          Class sqlStatementClass = sqlStatementContext.getSqlStatement().getClass().getSuperclass();
79          if (!DDL_STATEMENT_CLASSES.contains(sqlStatementClass)) {
80              return;
81          }
82          Optional<MetaDataRefresher> schemaRefresher = TypedSPILoader.findService(MetaDataRefresher.class, sqlStatementClass);
83          if (schemaRefresher.isPresent()) {
84              Collection<String> logicDataSourceNames = routeUnits.stream().map(each -> each.getDataSourceMapper().getLogicName()).collect(Collectors.toList());
85              String schemaName = sqlStatementContext instanceof TableAvailable ? getSchemaName(sqlStatementContext) : null;
86              DatabaseType databaseType = routeUnits.stream().map(each -> database.getResourceMetaData().getStorageUnits().get(each.getDataSourceMapper().getActualName()))
87                      .filter(Objects::nonNull).findFirst().map(StorageUnit::getStorageType).orElseGet(sqlStatementContext::getDatabaseType);
88              schemaRefresher.get().refresh(metaDataManagerPersistService, database, logicDataSourceNames, schemaName, databaseType, sqlStatementContext.getSqlStatement(), props);
89          }
90      }
91      
92      /**
93       * Refresh meta data for federation.
94       *
95       * @param sqlStatementContext SQL statement context
96       */
97      @SuppressWarnings("unchecked")
98      public void refresh(final SQLStatementContext sqlStatementContext) {
99          getFederationMetaDataRefresher(sqlStatementContext).ifPresent(optional -> optional.refresh(
100                 metaDataManagerPersistService, database, getSchemaName(sqlStatementContext), sqlStatementContext.getDatabaseType(), sqlStatementContext.getSqlStatement()));
101     }
102     
103     private String getSchemaName(final SQLStatementContext sqlStatementContext) {
104         return ((TableAvailable) sqlStatementContext).getTablesContext().getSchemaName()
105                 .orElseGet(() -> new DatabaseTypeRegistry(sqlStatementContext.getDatabaseType()).getDefaultSchemaName(database.getName())).toLowerCase();
106     }
107     
108     /**
109      * SQL statement is federation or not.
110      *
111      * @param sqlStatementContext SQL statement context
112      * @return is federation or not
113      */
114     public boolean isFederation(final SQLStatementContext sqlStatementContext) {
115         return getFederationMetaDataRefresher(sqlStatementContext).isPresent();
116     }
117     
118     @SuppressWarnings("rawtypes")
119     private Optional<FederationMetaDataRefresher> getFederationMetaDataRefresher(final SQLStatementContext sqlStatementContext) {
120         return TypedSPILoader.findService(FederationMetaDataRefresher.class, sqlStatementContext.getSqlStatement().getClass().getSuperclass());
121     }
122     
123     /**
124      * Is refresh meta data required.
125      *
126      * @param sqlStatementContext SQL statement context
127      * @return is refresh meta data required or not
128      */
129     public static boolean isRefreshMetaDataRequired(final SQLStatementContext sqlStatementContext) {
130         return DDL_STATEMENT_CLASSES.contains(sqlStatementContext.getSqlStatement().getClass().getSuperclass());
131     }
132 }
