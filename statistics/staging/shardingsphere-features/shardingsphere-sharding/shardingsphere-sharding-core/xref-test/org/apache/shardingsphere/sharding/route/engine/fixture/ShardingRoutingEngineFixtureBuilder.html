207     private static ShardingTableRuleConfiguration createInlineTableRuleConfig(final String tableName,
208                                                                               final String actualDataNodes, final String algorithmExpression, final String dsAlgorithmExpression) {
209         return createTableRuleConfig(tableName, actualDataNodes,
210                 createStandardShardingStrategyConfiguration("ds_inline", dsAlgorithmExpression), createStandardShardingStrategyConfiguration(tableName + "_inline", algorithmExpression));
211     }
212     
213     private static StandardShardingStrategyConfiguration createStandardShardingStrategyConfiguration(final String algorithmName, final String algorithmExpression) {
214         int startIndex = algorithmExpression.indexOf('{');
215         int endIndex = algorithmExpression.indexOf('%');
216         String shardingColumn = algorithmExpression.substring(startIndex + 1, endIndex).trim();
217         return new StandardShardingStrategyConfiguration(shardingColumn, algorithmName);
218     }
