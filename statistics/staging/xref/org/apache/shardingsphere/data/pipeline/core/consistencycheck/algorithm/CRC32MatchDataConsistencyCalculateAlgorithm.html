43  /**
44   * CRC32 match data consistency calculate algorithm.
45   */
46  @SPIDescription("Match CRC32 of records.")
47  @Slf4j
48  public final class CRC32MatchDataConsistencyCalculateAlgorithm extends AbstractDataConsistencyCalculateAlgorithm {
49      
50      private static final Collection<DatabaseType> SUPPORTED_DATABASE_TYPES = DatabaseTypeEngine.getTrunkAndBranchDatabaseTypes(Collections.singleton("MySQL"));
51      
52      @Override
53      public Iterable<DataConsistencyCalculatedResult> calculate(final DataConsistencyCalculateParameter param) {
54          PipelineDataConsistencyCalculateSQLBuilder pipelineSQLBuilder = new PipelineDataConsistencyCalculateSQLBuilder(param.getDatabaseType());
55          List<CalculatedItem> calculatedItems = param.getColumnNames().stream().map(each -> calculateCRC32(pipelineSQLBuilder, param, each)).collect(Collectors.toList());
56          return Collections.singletonList(new CalculatedResult(calculatedItems.get(0).getRecordsCount(), calculatedItems.stream().map(CalculatedItem::getCrc32).collect(Collectors.toList())));
57      }
58      
59      private CalculatedItem calculateCRC32(final PipelineDataConsistencyCalculateSQLBuilder pipelineSQLBuilder, final DataConsistencyCalculateParameter param, final String columnName) {
60          Optional<String> sql = pipelineSQLBuilder.buildCRC32SQL(param.getSchemaName(), param.getLogicTableName(), columnName);
61          ShardingSpherePreconditions.checkState(sql.isPresent(), () -> new UnsupportedCRC32DataConsistencyCalculateAlgorithmException(param.getDatabaseType()));
62          try (
63                  Connection connection = param.getDataSource().getConnection();
64                  PreparedStatement preparedStatement = connection.prepareStatement(sql.get());
65                  ResultSet resultSet = preparedStatement.executeQuery()) {
66              setCurrentStatement(preparedStatement);
67              resultSet.next();
68              long crc32 = resultSet.getLong(1);
69              int recordsCount = resultSet.getInt(2);
70              return new CalculatedItem(crc32, recordsCount);
71          } catch (final SQLException ex) {
72              throw new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), ex);
73          }
74      }
75      
76      @Override
77      public String getType() {
78          return "CRC32_MATCH";
79      }
80      
81      @Override
82      public Collection<DatabaseType> getSupportedDatabaseTypes() {
83          return SUPPORTED_DATABASE_TYPES;
84      }
85      
86      @RequiredArgsConstructor
87      @Getter
88      private static final class CalculatedItem {
89          
90          private final long crc32;
91          
92          private final int recordsCount;
93      }
94      
95      @RequiredArgsConstructor
96      @Getter
97      private static final class CalculatedResult implements DataConsistencyCalculatedResult {
98          
99          private final int recordsCount;
100         
101         private final Collection<Long> columnsCrc32;
102         
103         @Override
104         public boolean equals(final Object o) {
105             if (null == o) {
106                 return false;
107             }
108             if (this == o) {
109                 return true;
110             }
111             if (getClass() != o.getClass()) {
112                 log.warn("DataMatchCalculatedResult type not match, o.className={}", o.getClass().getName());
113                 return false;
114             }
115             final CalculatedResult that = (CalculatedResult) o;
116             if (recordsCount != that.recordsCount) {
117                 log.info("recordsCount not match, recordsCount={}, that.recordsCount={}", recordsCount, that.recordsCount);
118                 return false;
119             }
120             if (!columnsCrc32.equals(that.columnsCrc32)) {
121                 log.info("columnsCrc32 not match, columnsCrc32={}, that.columnsCrc32={}", columnsCrc32, that.columnsCrc32);
122                 return false;
123             }
124             return true;
125         }
126         
127         @Override
128         public int hashCode() {
129             int result = recordsCount;
130             result = 31 * result + columnsCrc32.hashCode();
131             return result;
132         }
133         
134         // TODO not support now
135         @Override
136         public Optional<Object> getMaxUniqueKeyValue() {
137             return Optional.empty();
138         }
139     }
