18  package org.apache.shardingsphere.infra.rewrite.sql.token.generator;
19  
20  import org.apache.shardingsphere.infra.binder.context.statement.SQLStatementContext;
21  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereSchema;
22  import org.apache.shardingsphere.infra.rewrite.sql.token.generator.aware.ConnectionContextAware;
23  import org.apache.shardingsphere.infra.rewrite.sql.token.generator.aware.ParametersAware;
24  import org.apache.shardingsphere.infra.rewrite.sql.token.generator.aware.PreviousSQLTokensAware;
25  import org.apache.shardingsphere.infra.rewrite.sql.token.generator.aware.SchemaMetaDataAware;
26  import org.apache.shardingsphere.infra.rewrite.sql.token.pojo.SQLToken;
27  import org.apache.shardingsphere.infra.session.connection.ConnectionContext;
28  
29  import java.util.Collection;
30  import java.util.LinkedList;
31  import java.util.List;
32  import java.util.Map;
33  
34  /**
35   * SQL token generators.
36   */
37  public final class SQLTokenGenerators {
38      
39      private final Collection<SQLTokenGenerator> generators = new LinkedList<>();
40      
41      /**
42       * Add all SQL token generators.
43       *
44       * @param sqlTokenGenerators SQL token generators
45       */
46      public void addAll(final Collection<SQLTokenGenerator> sqlTokenGenerators) {
47          generators.addAll(sqlTokenGenerators);
48      }
49      
50      /**
51       * Generate SQL tokens.
52       *
53       * @param databaseName database name
54       * @param sqlStatementContext SQL statement context
55       * @param params SQL parameters
56       * @param schemas schema map
57       * @param connectionContext connection context
58       * @return SQL tokens
59       */
60      @SuppressWarnings({"rawtypes", "unchecked"})
61      public List<SQLToken> generateSQLTokens(final String databaseName, final Map<String, ShardingSphereSchema> schemas,
62                                              final SQLStatementContext sqlStatementContext, final List<Object> params, final ConnectionContext connectionContext) {
63          List<SQLToken> result = new LinkedList<>();
64          for (SQLTokenGenerator each : generators) {
65              setUpSQLTokenGenerator(each, params, databaseName, schemas, result, connectionContext);
66              if (each instanceof OptionalSQLTokenGenerator) {
67                  SQLToken sqlToken = ((OptionalSQLTokenGenerator) each).generateSQLToken(sqlStatementContext);
68                  if (!result.contains(sqlToken)) {
69                      result.add(sqlToken);
70                  }
71              } else if (each instanceof CollectionSQLTokenGenerator) {
72                  result.addAll(((CollectionSQLTokenGenerator) each).generateSQLTokens(sqlStatementContext));
73              }
74          }
75          return result;
76      }
77      
78      private void setUpSQLTokenGenerator(final SQLTokenGenerator sqlTokenGenerator, final List<Object> params,
79                                          final String databaseName, final Map<String, ShardingSphereSchema> schemas, final List<SQLToken> previousSQLTokens, final ConnectionContext connectionContext) {
80          if (sqlTokenGenerator instanceof ParametersAware) {
81              ((ParametersAware) sqlTokenGenerator).setParameters(params);
82          }
83          if (sqlTokenGenerator instanceof SchemaMetaDataAware) {
84              ((SchemaMetaDataAware) sqlTokenGenerator).setSchemas(schemas);
85              ((SchemaMetaDataAware) sqlTokenGenerator).setDatabaseName(databaseName);
86          }
87          if (sqlTokenGenerator instanceof PreviousSQLTokensAware) {
88              ((PreviousSQLTokensAware) sqlTokenGenerator).setPreviousSQLTokens(previousSQLTokens);
89          }
90          if (sqlTokenGenerator instanceof ConnectionContextAware) {
91              ((ConnectionContextAware) sqlTokenGenerator).setConnectionContext(connectionContext);
92          }
93      }
94  }
