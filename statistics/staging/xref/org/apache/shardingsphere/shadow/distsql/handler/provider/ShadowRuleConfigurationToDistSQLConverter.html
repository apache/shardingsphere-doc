18  package org.apache.shardingsphere.shadow.distsql.handler.provider;

20  import org.apache.shardingsphere.distsql.handler.constant.DistSQLConstants;
21  import org.apache.shardingsphere.distsql.handler.engine.query.ral.convert.AlgorithmDistSQLConverter;
22  import org.apache.shardingsphere.distsql.handler.engine.query.ral.convert.RuleConfigurationToDistSQLConverter;
23  import org.apache.shardingsphere.infra.algorithm.core.config.AlgorithmConfiguration;
24  import org.apache.shardingsphere.shadow.config.ShadowRuleConfiguration;
25  import org.apache.shardingsphere.shadow.config.datasource.ShadowDataSourceConfiguration;
26  import org.apache.shardingsphere.shadow.config.table.ShadowTableConfiguration;
27  import org.apache.shardingsphere.shadow.distsql.handler.constant.ShadowDistSQLConstants;
28  
29  import java.util.Collection;
30  import java.util.Iterator;
31  import java.util.Map;
32  import java.util.Map.Entry;
33  
34  /**
35   * Shadow rule configuration to DistSQL converter.
36   */
37  public final class ShadowRuleConfigurationToDistSQLConverter implements RuleConfigurationToDistSQLConverter<ShadowRuleConfiguration> {
38      
39      @Override
40      public String convert(final ShadowRuleConfiguration ruleConfig) {
41          if (ruleConfig.getDataSources().isEmpty()) {
42              return "";
43          }
44          StringBuilder result = new StringBuilder(ShadowDistSQLConstants.CREATE_SHADOW);
45          Iterator<ShadowDataSourceConfiguration> iterator = ruleConfig.getDataSources().iterator();
46          while (iterator.hasNext()) {
47              ShadowDataSourceConfiguration dataSourceConfig = iterator.next();
48              String shadowRuleName = dataSourceConfig.getName();
49              String shadowTables = getShadowTables(shadowRuleName, ruleConfig.getTables(), ruleConfig.getShadowAlgorithms());
50              result.append(String.format(ShadowDistSQLConstants.SHADOW, shadowRuleName, dataSourceConfig.getProductionDataSourceName(), dataSourceConfig.getShadowDataSourceName(), shadowTables));
51              if (iterator.hasNext()) {
52                  result.append(DistSQLConstants.COMMA);
53              }
54          }
55          result.append(DistSQLConstants.SEMI);
56          return result.toString();
57      }
58      
59      private String getShadowTables(final String shadowRuleName, final Map<String, ShadowTableConfiguration> ruleConfig, final Map<String, AlgorithmConfiguration> algorithmConfigs) {
60          StringBuilder result = new StringBuilder();
61          Iterator<Entry<String, ShadowTableConfiguration>> iterator = ruleConfig.entrySet().iterator();
62          while (iterator.hasNext()) {
63              Entry<String, ShadowTableConfiguration> shadowTableConfig = iterator.next();
64              if (shadowTableConfig.getValue().getDataSourceNames().contains(shadowRuleName)) {
65                  String shadowTableTypes = getShadowTableTypes(shadowTableConfig.getValue().getShadowAlgorithmNames(), algorithmConfigs);
66                  result.append(String.format(ShadowDistSQLConstants.SHADOW_TABLE, shadowTableConfig.getKey(), shadowTableTypes));
67              }
68              if (iterator.hasNext()) {
69                  result.append(DistSQLConstants.COMMA).append(System.lineSeparator());
70              }
71          }
72          return result.toString();
73      }
74      
75      private String getShadowTableTypes(final Collection<String> shadowAlgorithmNames, final Map<String, AlgorithmConfiguration> algorithmConfigs) {
76          StringBuilder result = new StringBuilder();
77          Iterator<String> iterator = shadowAlgorithmNames.iterator();
78          while (iterator.hasNext()) {
79              result.append(AlgorithmDistSQLConverter.getAlgorithmType(algorithmConfigs.get(iterator.next())));
80              if (iterator.hasNext()) {
81                  result.append(DistSQLConstants.COMMA).append(' ');
82              }
83          }
84          return result.toString();
85      }
86      
87      @Override
88      public Class<ShadowRuleConfiguration> getType() {
89          return ShadowRuleConfiguration.class;
90      }
91  }
