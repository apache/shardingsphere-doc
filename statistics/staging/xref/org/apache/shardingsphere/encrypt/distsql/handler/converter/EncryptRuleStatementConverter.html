18  package org.apache.shardingsphere.encrypt.distsql.handler.converter;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.encrypt.api.config.EncryptRuleConfiguration;
23  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptColumnItemRuleConfiguration;
24  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptColumnRuleConfiguration;
25  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptTableRuleConfiguration;
26  import org.apache.shardingsphere.encrypt.distsql.parser.segment.EncryptColumnItemSegment;
27  import org.apache.shardingsphere.encrypt.distsql.parser.segment.EncryptColumnSegment;
28  import org.apache.shardingsphere.encrypt.distsql.parser.segment.EncryptRuleSegment;
29  import org.apache.shardingsphere.infra.config.algorithm.AlgorithmConfiguration;
30  
31  import java.util.Collection;
32  import java.util.HashMap;
33  import java.util.LinkedList;
34  import java.util.Map;
35  
36  /**
37   * Encrypt rule statement converter.
38   */
39  @NoArgsConstructor(access = AccessLevel.PRIVATE)
40  public final class EncryptRuleStatementConverter {
41      
42      /**
43       * Convert encrypt rule segments to encrypt rule configuration.
44       *
45       * @param ruleSegments encrypt rule segments
46       * @return encrypt rule configuration
47       */
48      public static EncryptRuleConfiguration convert(final Collection<EncryptRuleSegment> ruleSegments) {
49          Collection<EncryptTableRuleConfiguration> tables = new LinkedList<>();
50          Map<String, AlgorithmConfiguration> encryptors = new HashMap<>();
51          for (EncryptRuleSegment each : ruleSegments) {
52              tables.add(createEncryptTableRuleConfiguration(each));
53              encryptors.putAll(createEncryptorConfigurations(each));
54          }
55          return new EncryptRuleConfiguration(tables, encryptors);
56      }
57      
58      private static EncryptTableRuleConfiguration createEncryptTableRuleConfiguration(final EncryptRuleSegment ruleSegment) {
59          Collection<EncryptColumnRuleConfiguration> columns = new LinkedList<>();
60          for (EncryptColumnSegment each : ruleSegment.getColumns()) {
61              columns.add(createEncryptColumnRuleConfiguration(ruleSegment.getTableName(), each));
62          }
63          return new EncryptTableRuleConfiguration(ruleSegment.getTableName(), columns);
64      }
65      
66      private static EncryptColumnRuleConfiguration createEncryptColumnRuleConfiguration(final String tableName, final EncryptColumnSegment columnSegment) {
67          EncryptColumnItemRuleConfiguration cipherColumnConfig = new EncryptColumnItemRuleConfiguration(
68                  columnSegment.getCipher().getName(), getEncryptorName(tableName, columnSegment.getName()));
69          EncryptColumnRuleConfiguration result = new EncryptColumnRuleConfiguration(columnSegment.getName(), cipherColumnConfig);
70          if (null != columnSegment.getAssistedQuery()) {
71              setAssistedQuery(tableName, columnSegment, result);
72          }
73          if (null != columnSegment.getLikeQuery()) {
74              setLikeQuery(tableName, columnSegment, result);
75          }
76          return result;
77      }
78      
79      private static void setAssistedQuery(final String tableName, final EncryptColumnSegment columnSegment, final EncryptColumnRuleConfiguration result) {
80          String assistedQueryEncryptorName = null == columnSegment.getAssistedQuery().getEncryptor() ? null
81                  : getAssistedQueryEncryptorName(tableName, columnSegment.getName());
82          EncryptColumnItemRuleConfiguration assistedQueryColumnConfig = new EncryptColumnItemRuleConfiguration(columnSegment.getAssistedQuery().getName(), assistedQueryEncryptorName);
83          result.setAssistedQuery(assistedQueryColumnConfig);
84      }
85      
86      private static void setLikeQuery(final String tableName, final EncryptColumnSegment columnSegment, final EncryptColumnRuleConfiguration result) {
87          String likeQueryEncryptorName = null == columnSegment.getLikeQuery().getEncryptor() ? null : getLikeQueryEncryptorName(tableName, columnSegment.getName());
88          EncryptColumnItemRuleConfiguration likeQueryColumnConfig = new EncryptColumnItemRuleConfiguration(columnSegment.getLikeQuery().getName(), likeQueryEncryptorName);
89          result.setLikeQuery(likeQueryColumnConfig);
90      }
91      
92      private static Map<String, AlgorithmConfiguration> createEncryptorConfigurations(final EncryptRuleSegment ruleSegment) {
93          Map<String, AlgorithmConfiguration> result = new HashMap<>(ruleSegment.getColumns().size(), 1F);
94          for (EncryptColumnSegment each : ruleSegment.getColumns()) {
95              result.put(getEncryptorName(ruleSegment.getTableName(), each.getName()), createEncryptorConfiguration(each.getCipher()));
96              if (null != each.getAssistedQuery() && null != each.getAssistedQuery().getEncryptor()) {
97                  result.put(getAssistedQueryEncryptorName(ruleSegment.getTableName(), each.getName()), createEncryptorConfiguration(each.getAssistedQuery()));
98              }
99              if (null != each.getLikeQuery() && null != each.getLikeQuery().getEncryptor()) {
100                 result.put(getLikeQueryEncryptorName(ruleSegment.getTableName(), each.getName()), createEncryptorConfiguration(each.getLikeQuery()));
101             }
102         }
103         return result;
104     }
105     
106     private static AlgorithmConfiguration createEncryptorConfiguration(final EncryptColumnItemSegment columnSegment) {
107         return new AlgorithmConfiguration(columnSegment.getEncryptor().getName(), columnSegment.getEncryptor().getProps());
108     }
109     
110     private static String getEncryptorName(final String tableName, final String columnName) {
111         return String.format("%s_%s", tableName, columnName);
112     }
113     
114     private static String getAssistedQueryEncryptorName(final String tableName, final String columnName) {
115         return String.format("assist_%s_%s", tableName, columnName);
116     }
117     
118     private static String getLikeQueryEncryptorName(final String tableName, final String columnName) {
119         return String.format("like_%s_%s", tableName, columnName);
120     }
121 }
