1761      * Get all SQL parser test cases.
1762      *
1763      * @return got test cases
1764      */
1765     @SuppressWarnings("unchecked")
1766     @SneakyThrows(IllegalAccessException.class)
1767     public Map<String, SQLParserTestCase> getAllCases() {
1768         Map<String, SQLParserTestCase> result = new HashMap<>();
1769         for (Field each : RootSQLParserTestCases.class.getDeclaredFields()) {
1770             if (!isXmlElementField(each)) {
1771                 continue;
1772             }
1773             List<? extends SQLParserTestCase> cases = (List<? extends SQLParserTestCase>) Plugins.getMemberAccessor().get(each, this);
1774             if (cases.isEmpty()) {
1775                 continue;
1776             }
1777             Map<String, SQLParserTestCase> caseMap = getTestCaseMap(cases);
1778             checkDuplicatedTestCases(caseMap, result);
1779             result.putAll(caseMap);
1780         }
1781         return result;
1782     }

1784     private boolean isXmlElementField(final Field field) {
1785         return field.isAnnotationPresent(XmlElement.class) && List.class == field.getType() && field.getGenericType() instanceof ParameterizedType
1786                 && SQLParserTestCase.class.isAssignableFrom((Class<?>) ((ParameterizedType) field.getGenericType()).getActualTypeArguments()[0]);
1787     }

1789     private Map<String, SQLParserTestCase> getTestCaseMap(final List<? extends SQLParserTestCase> cases) {
1790         Map<String, SQLParserTestCase> result = new HashMap<>(cases.size(), 1F);
1791         for (SQLParserTestCase each : cases) {
1792             checkDuplicatedTestCase(each, result);
1793             result.put(each.getSqlCaseId(), each);
1794         }
1795         return result;
1796     }

1798     private void checkDuplicatedTestCase(final SQLParserTestCase newTestCase, final Map<String, SQLParserTestCase> existedTestCases) {
1799         Preconditions.checkState(!existedTestCases.containsKey(newTestCase.getSqlCaseId()), "Find duplicated SQL Case ID: %s.", newTestCase.getSqlCaseId());
1800     }

1802     private void checkDuplicatedTestCases(final Map<String, SQLParserTestCase> newTestCases, final Map<String, SQLParserTestCase> existedTestCases) {
1803         Collection<String> caseIds = new HashSet<>(newTestCases.keySet());
1804         caseIds.retainAll(existedTestCases.keySet());
1805         Preconditions.checkState(caseIds.isEmpty(), "Find duplicated SQL Case IDs: %s.", caseIds);
