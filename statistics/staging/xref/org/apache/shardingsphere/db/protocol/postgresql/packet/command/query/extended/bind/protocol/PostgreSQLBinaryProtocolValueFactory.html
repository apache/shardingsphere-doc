20  import com.google.common.base.Preconditions;
21  import lombok.AccessLevel;
22  import lombok.NoArgsConstructor;
23  import org.apache.shardingsphere.db.protocol.binary.BinaryColumnType;

26  import java.util.HashMap;

29  /**
30   * Binary protocol value factory for PostgreSQL.
31   */
32  @NoArgsConstructor(access = AccessLevel.PRIVATE)
33  public final class PostgreSQLBinaryProtocolValueFactory {
34      
35      private static final Map<BinaryColumnType, PostgreSQLBinaryProtocolValue> BINARY_PROTOCOL_VALUES = new HashMap<>();
36      
37      static {
38          setUnspecifiedBinaryProtocolValue();
39          setStringLenencBinaryProtocolValue();
40          setInt8BinaryProtocolValue();
41          setInt4BinaryProtocolValue();
42          setInt2BinaryProtocolValue();
43          setDoubleBinaryProtocolValue();
44          setFloatBinaryProtocolValue();
45          setNumericBinaryProtocolValue();
46          setDateBinaryProtocolValue();
47          setTimeBinaryProtocolValue();
48          setInt2ArrayBinaryProtocolValue();
49          setInt4ArrayBinaryProtocolValue();
50          setInt8ArrayBinaryProtocolValue();
51          setFloat4ArrayBinaryProtocolValue();
52          setFloat8ArrayBinaryProtocolValue();
53          setBoolArrayBinaryProtocolValue();
54          setStringArrayBinaryProtocolValue();
55          setByteaBinaryProtocolValue();
56          setUUIDBinaryProtocolValue();
57          setBoolBinaryProtocolValue();
58      }
59      
60      private static void setUnspecifiedBinaryProtocolValue() {
61          PostgreSQLUnspecifiedBinaryProtocolValue binaryProtocolValue = new PostgreSQLUnspecifiedBinaryProtocolValue();
62          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.UNSPECIFIED, binaryProtocolValue);
63      }
64      
65      private static void setStringLenencBinaryProtocolValue() {
66          PostgreSQLStringBinaryProtocolValue binaryProtocolValue = new PostgreSQLStringBinaryProtocolValue();
67          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.VARCHAR, binaryProtocolValue);
68          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.CHAR, binaryProtocolValue);
69      }
70      
71      private static void setInt8BinaryProtocolValue() {
72          PostgreSQLInt8BinaryProtocolValue binaryProtocolValue = new PostgreSQLInt8BinaryProtocolValue();
73          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.INT8, binaryProtocolValue);
74      }
75      
76      private static void setInt4BinaryProtocolValue() {
77          PostgreSQLInt4BinaryProtocolValue binaryProtocolValue = new PostgreSQLInt4BinaryProtocolValue();
78          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.INT4, binaryProtocolValue);
79      }
80      
81      private static void setInt2BinaryProtocolValue() {
82          PostgreSQLInt2BinaryProtocolValue binaryProtocolValue = new PostgreSQLInt2BinaryProtocolValue();
83          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.INT2, binaryProtocolValue);
84      }
85      
86      private static void setDoubleBinaryProtocolValue() {
87          PostgreSQLDoubleBinaryProtocolValue binaryProtocolValue = new PostgreSQLDoubleBinaryProtocolValue();
88          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.FLOAT8, binaryProtocolValue);
89      }
90      
91      private static void setFloatBinaryProtocolValue() {
92          PostgreSQLFloatBinaryProtocolValue binaryProtocolValue = new PostgreSQLFloatBinaryProtocolValue();
93          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.FLOAT4, binaryProtocolValue);
94      }
95      
96      private static void setNumericBinaryProtocolValue() {
97          PostgreSQLNumericBinaryProtocolValue binaryProtocolValue = new PostgreSQLNumericBinaryProtocolValue();
98          BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.NUMERIC, binaryProtocolValue);
99      }
100     
101     private static void setDateBinaryProtocolValue() {
102         PostgreSQLDateBinaryProtocolValue binaryProtocolValue = new PostgreSQLDateBinaryProtocolValue();
103         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.DATE, binaryProtocolValue);
104     }
105     
106     private static void setTimeBinaryProtocolValue() {
107         PostgreSQLTimeBinaryProtocolValue binaryProtocolValue = new PostgreSQLTimeBinaryProtocolValue();
108         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.TIMESTAMP, binaryProtocolValue);
109     }
110     
111     private static void setInt2ArrayBinaryProtocolValue() {
112         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.INT2_ARRAY, new PostgreSQLInt2ArrayBinaryProtocolValue());
113     }
114     
115     private static void setInt4ArrayBinaryProtocolValue() {
116         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.INT4_ARRAY, new PostgreSQLInt4ArrayBinaryProtocolValue());
117     }
118     
119     private static void setInt8ArrayBinaryProtocolValue() {
120         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.INT8_ARRAY, new PostgreSQLInt8ArrayBinaryProtocolValue());
121     }
122     
123     private static void setFloat4ArrayBinaryProtocolValue() {
124         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.FLOAT4_ARRAY, new PostgreSQLFloat4ArrayBinaryProtocolValue());
125     }
126     
127     private static void setFloat8ArrayBinaryProtocolValue() {
128         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.FLOAT8_ARRAY, new PostgreSQLFloat8ArrayBinaryProtocolValue());
129     }
130     
131     private static void setBoolArrayBinaryProtocolValue() {
132         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.BOOL_ARRAY, new PostgreSQLBoolArrayBinaryProtocolValue());
133     }
134     
135     private static void setStringArrayBinaryProtocolValue() {
136         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.VARCHAR_ARRAY, new PostgreSQLStringArrayBinaryProtocolValue());
137         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.CHAR_ARRAY, new PostgreSQLStringArrayBinaryProtocolValue());
138         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.TEXT_ARRAY, new PostgreSQLTextArrayBinaryProtocolValue());
139     }
140     
141     private static void setByteaBinaryProtocolValue() {
142         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.BYTEA, new PostgreSQLByteaBinaryProtocolValue());
143     }
144     
145     private static void setUUIDBinaryProtocolValue() {
146         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.UUID, new PostgreSQLUUIDBinaryProtocolValue());
147     }
148     
149     private static void setBoolBinaryProtocolValue() {
150         BINARY_PROTOCOL_VALUES.put(PostgreSQLColumnType.BOOL, new PostgreSQLBoolBinaryProtocolValue());
151     }
152     
153     /**
154      * Get binary protocol value.
155      *
156      * @param binaryColumnType binary column type
157      * @return binary protocol value
158      */
159     public static PostgreSQLBinaryProtocolValue getBinaryProtocolValue(final BinaryColumnType binaryColumnType) {
160         Preconditions.checkArgument(BINARY_PROTOCOL_VALUES.containsKey(binaryColumnType), "Cannot find PostgreSQL type '%s' in column type when process binary protocol value", binaryColumnType);
161         return BINARY_PROTOCOL_VALUES.get(binaryColumnType);
162     }
163 }
