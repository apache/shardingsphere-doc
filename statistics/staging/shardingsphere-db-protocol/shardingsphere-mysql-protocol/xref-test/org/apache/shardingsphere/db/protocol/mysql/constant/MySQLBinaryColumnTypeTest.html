18  package org.apache.shardingsphere.db.protocol.mysql.constant;
19  
20  import org.junit.jupiter.api.Test;
21  
22  import java.sql.Types;
23  
24  import static org.hamcrest.CoreMatchers.is;
25  import static org.hamcrest.MatcherAssert.assertThat;
26  import static org.junit.jupiter.api.Assertions.assertThrows;
27  
28  class MySQLBinaryColumnTypeTest {
29      
30      @Test
31      void assertValueOfJDBC() {
32          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.BIT), is(MySQLBinaryColumnType.MYSQL_TYPE_BIT));
33          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.TINYINT), is(MySQLBinaryColumnType.MYSQL_TYPE_TINY));
34          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.SMALLINT), is(MySQLBinaryColumnType.MYSQL_TYPE_SHORT));
35          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.INTEGER), is(MySQLBinaryColumnType.MYSQL_TYPE_LONG));
36          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.BIGINT), is(MySQLBinaryColumnType.MYSQL_TYPE_LONGLONG));
37          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.FLOAT), is(MySQLBinaryColumnType.MYSQL_TYPE_FLOAT));
38          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.REAL), is(MySQLBinaryColumnType.MYSQL_TYPE_FLOAT));
39          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.DOUBLE), is(MySQLBinaryColumnType.MYSQL_TYPE_DOUBLE));
40          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.NUMERIC), is(MySQLBinaryColumnType.MYSQL_TYPE_NEWDECIMAL));
41          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.DECIMAL), is(MySQLBinaryColumnType.MYSQL_TYPE_NEWDECIMAL));
42          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.CHAR), is(MySQLBinaryColumnType.MYSQL_TYPE_STRING));
43          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.VARCHAR), is(MySQLBinaryColumnType.MYSQL_TYPE_VAR_STRING));
44          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.LONGVARCHAR), is(MySQLBinaryColumnType.MYSQL_TYPE_VAR_STRING));
45          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.DATE), is(MySQLBinaryColumnType.MYSQL_TYPE_DATE));
46          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.TIME), is(MySQLBinaryColumnType.MYSQL_TYPE_TIME));
47          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.TIMESTAMP), is(MySQLBinaryColumnType.MYSQL_TYPE_TIMESTAMP));
48          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.BINARY), is(MySQLBinaryColumnType.MYSQL_TYPE_STRING));
49          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.VARBINARY), is(MySQLBinaryColumnType.MYSQL_TYPE_VAR_STRING));
50          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.LONGVARBINARY), is(MySQLBinaryColumnType.MYSQL_TYPE_VAR_STRING));
51          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.NULL), is(MySQLBinaryColumnType.MYSQL_TYPE_NULL));
52          assertThat(MySQLBinaryColumnType.valueOfJDBCType(Types.BLOB), is(MySQLBinaryColumnType.MYSQL_TYPE_BLOB));
53      }
54      
55      @Test
56      void assertValueOfJDBCIllegalArgument() {
57          assertThrows(IllegalArgumentException.class, () -> MySQLBinaryColumnType.valueOfJDBCType(9999));
58      }
59      
60      @Test
61      void assertValueOf() {
62          assertThat(MySQLBinaryColumnType.valueOf(MySQLBinaryColumnType.MYSQL_TYPE_DECIMAL.getValue()), is(MySQLBinaryColumnType.MYSQL_TYPE_DECIMAL));
63      }
64      
65      @Test
66      void assertValueOfWithIllegalArgument() {
67          assertThrows(IllegalArgumentException.class, () -> MySQLBinaryColumnType.valueOf(-1));
68      }
69  }
