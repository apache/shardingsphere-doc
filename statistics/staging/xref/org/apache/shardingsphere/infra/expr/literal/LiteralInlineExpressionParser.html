18  package org.apache.shardingsphere.infra.expr.literal;
19  
20  import com.google.common.base.Strings;
21  import org.apache.shardingsphere.infra.expr.spi.InlineExpressionParser;
22  
23  import java.util.ArrayList;
24  import java.util.Collections;
25  import java.util.List;
26  import java.util.Properties;
27  
28  /**
29   * literal inline expression parser.
30   */
31  public final class LiteralInlineExpressionParser implements InlineExpressionParser {
32      
33      private static final char SPLITTER = ',';
34      
35      private String inlineExpression;
36      
37      @Override
38      public void init(final Properties props) {
39          inlineExpression = props.getProperty(INLINE_EXPRESSION_KEY);
40      }
41      
42      @Override
43      public List<String> splitAndEvaluate() {
44          return Strings.isNullOrEmpty(inlineExpression) ? Collections.emptyList() : split(inlineExpression);
45      }
46      
47      private List<String> split(final String inlineExpression) {
48          List<String> result = new ArrayList<>(inlineExpression.length());
49          StringBuilder segment = new StringBuilder();
50          for (int i = 0; i < inlineExpression.length(); i++) {
51              char each = inlineExpression.charAt(i);
52              if (SPLITTER == each) {
53                  result.add(segment.toString().trim());
54                  segment.setLength(0);
55              } else {
56                  segment.append(each);
57              }
58          }
59          if (segment.length() > 0) {
60              result.add(segment.toString().trim());
61          }
62          return result;
63      }
64      
65      @Override
66      public String getType() {
67          return "LITERAL";
68      }
69  }
