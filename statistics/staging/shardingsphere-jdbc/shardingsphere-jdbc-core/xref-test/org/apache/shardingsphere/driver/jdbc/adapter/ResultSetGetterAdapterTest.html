436     private ShardingSphereResultSet mockShardingSphereResultSet(final MergedResult mergedResult) throws SQLException {
437         ResultSetMetaData resultSetMetaData = mock(ResultSetMetaData.class);
438         when(resultSetMetaData.getColumnLabel(1)).thenReturn("col");
439         ResultSet resultSet = mock(ResultSet.class);
440         when(resultSetMetaData.getColumnCount()).thenReturn(1);
441         when(resultSet.getMetaData()).thenReturn(resultSetMetaData);
442         return new ShardingSphereResultSet(Collections.singletonList(resultSet), mergedResult, mock(Statement.class), true, mock(ExecutionContext.class));
443     }
