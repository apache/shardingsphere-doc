46  @ExtendWith(MockitoExtension.class)
47  class CreateDefaultShardingStrategyStatementUpdaterTest {
48      
49      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
50      private ShardingSphereDatabase database;
51      
52      private final CreateDefaultShardingStrategyStatementUpdater updater = new CreateDefaultShardingStrategyStatementUpdater();
53      
54      @BeforeEach
55      void before() {
56          when(database.getName()).thenReturn("foo_db");
57      }
58      
59      @Test
60      void assertExecuteWithInvalidStrategyType() {
61          CreateDefaultShardingStrategyStatement statement = new CreateDefaultShardingStrategyStatement(false, "TABLE", "invalidType", null, null);
62          assertThrows(InvalidAlgorithmConfigurationException.class, () -> updater.checkSQLStatement(database, statement, new ShardingRuleConfiguration()));
63      }
64      
65      @Test
66      void assertExecuteWithoutAlgorithm() {
67          CreateDefaultShardingStrategyStatement statement = new CreateDefaultShardingStrategyStatement(false, "TABLE", "standard", "order_id", null);
68          assertThrows(MissingRequiredAlgorithmException.class, () -> updater.checkSQLStatement(database, statement, null));
69      }
70      
71      @Test
72      void assertExecuteWithExist() {
73          AlgorithmSegment algorithm = new AlgorithmSegment("order_id_algorithm", new Properties());
74          CreateDefaultShardingStrategyStatement statement = new CreateDefaultShardingStrategyStatement(false, "TABLE", "standard", "order_id", algorithm);
75          ShardingRuleConfiguration currentRuleConfig = new ShardingRuleConfiguration();
76          currentRuleConfig.setDefaultTableShardingStrategy(new StandardShardingStrategyConfiguration("order_id", "orderAlgorithm"));
77          currentRuleConfig.getShardingAlgorithms().put("order_id_algorithm", null);
78          assertThrows(DuplicateRuleException.class, () -> updater.checkSQLStatement(database, statement, currentRuleConfig));
79      }
80      
81      @Test
82      void assertExecuteWithUnmatchedStrategy() {
83          CreateDefaultShardingStrategyStatement statement = new CreateDefaultShardingStrategyStatement(false, "TABLE", "standard", "order_id,user_id", null);
84          ShardingRuleConfiguration currentRuleConfig = new ShardingRuleConfiguration();
85          currentRuleConfig.setDefaultTableShardingStrategy(new StandardShardingStrategyConfiguration("order_id", "orderAlgorithm"));
86          currentRuleConfig.getShardingAlgorithms().put("order_id_algorithm", null);
87          assertThrows(InvalidAlgorithmConfigurationException.class, () -> updater.checkSQLStatement(database, statement, currentRuleConfig));
88      }
89      
90      @Test
91      void assertCreateDefaultTableShardingStrategy() {
92          AlgorithmSegment algorithm = new AlgorithmSegment("order_id_algorithm", new Properties());
93          CreateDefaultShardingStrategyStatement statement = new CreateDefaultShardingStrategyStatement(false, "TABLE", "standard", "order_id", algorithm);
94          ShardingRuleConfiguration currentRuleConfig = new ShardingRuleConfiguration();
95          currentRuleConfig.setDefaultDatabaseShardingStrategy(new StandardShardingStrategyConfiguration("order_id", "orderAlgorithm"));
96          currentRuleConfig.getShardingAlgorithms().put("order_id_algorithm", null);
97          updater.checkSQLStatement(database, statement, currentRuleConfig);
98          ShardingRuleConfiguration toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, statement);
99          updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
100         StandardShardingStrategyConfiguration defaultTableShardingStrategy = (StandardShardingStrategyConfiguration) currentRuleConfig.getDefaultTableShardingStrategy();
101         assertThat(defaultTableShardingStrategy.getShardingAlgorithmName(), is("default_table_order_id_algorithm"));
102         assertThat(defaultTableShardingStrategy.getShardingColumn(), is("order_id"));
103     }

106     void assertCreateDefaultDatabaseShardingStrategy() {
107         AlgorithmSegment databaseAlgorithmSegment = new AlgorithmSegment("inline", PropertiesBuilder.build(new Property("algorithm-expression", "ds_${user_id % 2}")));
108         CreateDefaultShardingStrategyStatement statement = new CreateDefaultShardingStrategyStatement(false, "DATABASE", "standard", "user_id", databaseAlgorithmSegment);
109         ShardingRuleConfiguration currentRuleConfig = new ShardingRuleConfiguration();

111         ShardingRuleConfiguration toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, statement);
112         updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
113         StandardShardingStrategyConfiguration defaultDatabaseShardingStrategy = (StandardShardingStrategyConfiguration) currentRuleConfig.getDefaultDatabaseShardingStrategy();
114         assertThat(defaultDatabaseShardingStrategy.getShardingAlgorithmName(), is("default_database_inline"));
115         assertThat(defaultDatabaseShardingStrategy.getShardingColumn(), is("user_id"));
116     }
