19  
20  import org.apache.shardingsphere.agent.plugin.core.context.ShardingSphereDataSourceContext;
21  import org.apache.shardingsphere.agent.plugin.core.holder.ShardingSphereDataSourceContextHolder;
22  import org.apache.shardingsphere.agent.plugin.metrics.core.collector.MetricsCollectorRegistry;
23  import org.apache.shardingsphere.agent.plugin.metrics.core.collector.type.GaugeMetricFamilyMetricsCollector;
24  import org.apache.shardingsphere.agent.plugin.metrics.core.config.MetricCollectorType;
25  import org.apache.shardingsphere.agent.plugin.metrics.core.config.MetricConfiguration;
26  import org.apache.shardingsphere.agent.plugin.metrics.core.exporter.MetricsExporter;
27  
28  import java.util.Arrays;
29  import java.util.Map.Entry;
30  import java.util.Optional;
31  
32  /**
33   * JDBC meta data information exporter.
34   */
35  public final class JDBCMetaDataInfoExporter implements MetricsExporter {
36      
37      private final MetricConfiguration config = new MetricConfiguration("jdbc_meta_data_info", MetricCollectorType.GAUGE_METRIC_FAMILY,
38              "Meta data information of ShardingSphere-JDBC",
39              Arrays.asList("driver_instance", "database", "type"));
40      
41      @Override
42      public Optional<GaugeMetricFamilyMetricsCollector> export(final String pluginType) {
43          GaugeMetricFamilyMetricsCollector result = MetricsCollectorRegistry.get(config, pluginType);
44          result.cleanMetrics();
45          for (Entry<String, ShardingSphereDataSourceContext> entry : ShardingSphereDataSourceContextHolder.getShardingSphereDataSourceContexts().entrySet()) {
46              Optional.ofNullable(entry.getValue().getContextManager().getDatabase(entry.getValue().getDatabaseName()))
47                      .ifPresent(optional -> result.addMetric(Arrays.asList(entry.getKey(), optional.getName(), "storage_unit_count"), optional.getResourceMetaData().getStorageUnits().size()));
48          }
49          return Optional.of(result);
50      }
51  }
