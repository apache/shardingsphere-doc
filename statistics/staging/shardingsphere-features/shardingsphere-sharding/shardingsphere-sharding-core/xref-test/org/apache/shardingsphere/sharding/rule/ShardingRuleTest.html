743     private void assertGetDataNodes(final Collection<DataNode> dataNodes, final String tableNamePrefix) {
744         int dataSourceNameSuffix = 0;
745         int tableNameSuffix = 0;
746         for (final DataNode each : dataNodes) {
747             assertThat(each.getDataSourceName(), is("ds_" + dataSourceNameSuffix));
748             assertThat(each.getTableName(), is(tableNamePrefix + tableNameSuffix));
749             if (++tableNameSuffix == (dataNodes.size() / 2)) {
750                 tableNameSuffix = 0;
751                 dataSourceNameSuffix++;
752             }
753         }
754     }
