18  package org.apache.shardingsphere.driver.jdbc.core.driver.spi;
19  
20  import com.google.common.base.Preconditions;
21  import lombok.SneakyThrows;
22  import org.apache.commons.lang3.StringUtils;
23  import org.apache.shardingsphere.driver.jdbc.core.driver.ShardingSphereDriverURLProvider;
24  
25  import java.io.BufferedReader;
26  import java.io.IOException;
27  import java.io.InputStream;
28  import java.io.InputStreamReader;
29  import java.nio.charset.StandardCharsets;
30  
31  /**
32   * Classpath driver URL provider.
33   */
34  public final class ClasspathDriverURLProvider implements ShardingSphereDriverURLProvider {
35      
36      private static final String CLASSPATH_TYPE = "classpath:";
37      
38      @Override
39      public boolean accept(final String url) {
40          return StringUtils.isNotBlank(url) && url.contains(CLASSPATH_TYPE);
41      }
42      
43      @Override
44      @SneakyThrows(IOException.class)
45      public byte[] getContent(final String url, final String urlPrefix) {
46          String configuredFile = url.substring(urlPrefix.length(), url.contains("?") ? url.indexOf('?') : url.length());
47          String file = configuredFile.substring(CLASSPATH_TYPE.length());
48          Preconditions.checkArgument(!file.isEmpty(), "Configuration file is required in ShardingSphere driver URL.");
49          try (
50                  InputStream stream = getResourceAsStream(file);
51                  BufferedReader reader = new BufferedReader(new InputStreamReader(stream, StandardCharsets.UTF_8))) {
52              StringBuilder builder = new StringBuilder();
53              String line;
54              while (null != (line = reader.readLine())) {
55                  if (!line.startsWith("#")) {
56                      builder.append(line).append('\n');
57                  }
58              }
59              return builder.toString().getBytes(StandardCharsets.UTF_8);
60          }
61      }
62      
63      private InputStream getResourceAsStream(final String resource) {
64          InputStream result = Thread.currentThread().getContextClassLoader().getResourceAsStream(resource);
65          result = null == result ? Thread.currentThread().getContextClassLoader().getResourceAsStream("/" + resource) : result;
66          if (null != result) {
67              return result;
68          }
69          throw new IllegalArgumentException(String.format("Can not find configuration file `%s`.", resource));
70      }
71  }
