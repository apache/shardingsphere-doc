64      private static final String TABLE_NAME = "table";
65      
66      @Mock
67      private DataSource dataSource;
68      
69      @Mock
70      private Connection connection;
71      
72      @Mock
73      private DatabaseMetaData databaseMetaData;
74      
75      @Mock
76      private ResultSet resultSet;
77      
78      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
79      private ShardingSphereConnection shardingSphereConnection;
80      
81      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
82      private MetaDataContexts metaDataContexts;
83      
84      private final Map<String, DataSource> dataSourceMap = new HashMap<>();
85      
86      private ShardingSphereDatabaseMetaData shardingSphereDatabaseMetaData;
87      
88      @BeforeEach
89      void setUp() throws SQLException {
90          dataSourceMap.put(DATA_SOURCE_NAME, dataSource);
91          when(dataSource.getConnection()).thenReturn(connection);
92          when(connection.getMetaData()).thenReturn(databaseMetaData);
93          when(resultSet.getMetaData()).thenReturn(mock(ResultSetMetaData.class));
94          when(shardingSphereConnection.getConnectionManager().getRandomPhysicalDataSourceName()).thenReturn(DATA_SOURCE_NAME);
95          when(shardingSphereConnection.getConnectionManager().getRandomConnection()).thenReturn(connection);
96          when(shardingSphereConnection.getContextManager().getMetaDataContexts()).thenReturn(metaDataContexts);
97          when(shardingSphereConnection.getContextManager().getDataSourceMap(DefaultDatabase.LOGIC_NAME)).thenReturn(dataSourceMap);
98          when(shardingSphereConnection.getDatabaseName()).thenReturn(DefaultDatabase.LOGIC_NAME);
99          ShardingSphereDatabase database = mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS);
100         when(metaDataContexts.getMetaData().getDatabase(shardingSphereConnection.getDatabaseName())).thenReturn(database);
101         CachedDatabaseMetaData cachedDatabaseMetaData = new CachedDatabaseMetaData(databaseMetaData);
102         when(shardingSphereConnection.getJdbcContext().getCachedDatabaseMetaData()).thenReturn(cachedDatabaseMetaData);
103         ShardingRule shardingRule = mockShardingRule();
104         when(database.getRuleMetaData().getRules()).thenReturn(Collections.singleton(shardingRule));
105         shardingSphereDatabaseMetaData = new ShardingSphereDatabaseMetaData(shardingSphereConnection);
106     }
107     
108     private ShardingRule mockShardingRule() {
109         ShardingRuleConfiguration ruleConfig = new ShardingRuleConfiguration();
110         ShardingTableRuleConfiguration shardingTableRuleConfig = new ShardingTableRuleConfiguration(TABLE_NAME, DATA_SOURCE_NAME + "." + TABLE_NAME);
111         ruleConfig.setTables(Collections.singletonList(shardingTableRuleConfig));
112         return new ShardingRule(ruleConfig, Collections.singletonList(DATA_SOURCE_NAME), mock(InstanceContext.class));
113     }
