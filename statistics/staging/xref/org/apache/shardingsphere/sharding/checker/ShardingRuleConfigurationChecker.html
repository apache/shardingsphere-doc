51  /**
52   * Sharding rule configuration checker.
53   */
54  public final class ShardingRuleConfigurationChecker implements RuleConfigurationChecker<ShardingRuleConfiguration> {
55      
56      @Override
57      public void check(final String databaseName, final ShardingRuleConfiguration ruleConfig, final Map<String, DataSource> dataSourceMap, final Collection<ShardingSphereRule> builtRules) {
58          checkShardingAlgorithms(ruleConfig.getShardingAlgorithms().values());
59          checkKeyGeneratorAlgorithms(ruleConfig.getKeyGenerators().values());
60          Collection<String> keyGenerators = ruleConfig.getKeyGenerators().keySet();
61          Collection<String> auditors = ruleConfig.getAuditors().keySet();
62          Collection<String> shardingAlgorithms = ruleConfig.getShardingAlgorithms().keySet();
63          checkTables(databaseName, ruleConfig.getTables(), ruleConfig.getAutoTables(), keyGenerators, auditors, shardingAlgorithms);
64          checkKeyGenerateStrategy(databaseName, ruleConfig.getDefaultKeyGenerateStrategy(), keyGenerators);
65          checkAuditStrategy(databaseName, ruleConfig.getDefaultAuditStrategy(), auditors);
66          checkShardingStrategy(databaseName, ruleConfig.getDefaultDatabaseShardingStrategy(), shardingAlgorithms);
67          checkShardingStrategy(databaseName, ruleConfig.getDefaultTableShardingStrategy(), shardingAlgorithms);
68      }
69      
70      private void checkShardingAlgorithms(final Collection<AlgorithmConfiguration> algorithmConfigs) {
71          algorithmConfigs.forEach(each -> TypedSPILoader.checkService(ShardingAlgorithm.class, each.getType(), each.getProps()));
72      }
73      
74      private void checkKeyGeneratorAlgorithms(final Collection<AlgorithmConfiguration> algorithmConfigs) {
75          algorithmConfigs.stream().filter(Objects::nonNull).forEach(each -> TypedSPILoader.checkService(KeyGenerateAlgorithm.class, each.getType(), each.getProps()));
76      }
77      
78      private void checkTables(final String databaseName, final Collection<ShardingTableRuleConfiguration> tables, final Collection<ShardingAutoTableRuleConfiguration> autoTables,
79                               final Collection<String> keyGenerators, final Collection<String> auditors, final Collection<String> shardingAlgorithms) {
80          for (ShardingTableRuleConfiguration each : tables) {
81              checkLogicTable(databaseName, each.getLogicTable());
82              checkKeyGenerateStrategy(databaseName, each.getKeyGenerateStrategy(), keyGenerators);
83              checkAuditStrategy(databaseName, each.getAuditStrategy(), auditors);
84              checkShardingStrategy(databaseName, each.getDatabaseShardingStrategy(), shardingAlgorithms);
85              checkShardingStrategy(databaseName, each.getTableShardingStrategy(), shardingAlgorithms);
86          }
87          for (ShardingAutoTableRuleConfiguration each : autoTables) {
88              checkLogicTable(databaseName, each.getLogicTable());
89              checkKeyGenerateStrategy(databaseName, each.getKeyGenerateStrategy(), keyGenerators);
90              checkAuditStrategy(databaseName, each.getAuditStrategy(), auditors);
91              checkShardingStrategy(databaseName, each.getShardingStrategy(), shardingAlgorithms);
92          }
93      }
94      
95      private void checkLogicTable(final String databaseName, final String logicTable) {
96          ShardingSpherePreconditions.checkNotEmpty(logicTable, () -> new MissingRequiredShardingConfigurationException("Sharding logic table", databaseName));
97      }
98      
99      private void checkKeyGenerateStrategy(final String databaseName, final KeyGenerateStrategyConfiguration keyGenerateStrategy, final Collection<String> keyGenerators) {
100         if (null == keyGenerateStrategy) {
101             return;
102         }
103         ShardingSpherePreconditions.checkNotEmpty(keyGenerateStrategy.getColumn(), () -> new MissingRequiredShardingConfigurationException("Key generate column", databaseName));
104         ShardingSpherePreconditions.checkContains(keyGenerators, keyGenerateStrategy.getKeyGeneratorName(),
105                 () -> new UnregisteredAlgorithmException("Key generate", keyGenerateStrategy.getKeyGeneratorName(), new SQLExceptionIdentifier(databaseName)));
106     }
107     
108     private void checkAuditStrategy(final String databaseName, final ShardingAuditStrategyConfiguration auditStrategy, final Collection<String> auditors) {
109         if (null == auditStrategy) {
110             return;
111         }
112         ShardingSpherePreconditions.checkState(auditors.containsAll(auditStrategy.getAuditorNames()),
113                 () -> new UnregisteredAlgorithmException("Sharding audit", Joiner.on(",").join(auditStrategy.getAuditorNames()), new SQLExceptionIdentifier(databaseName)));
114     }
115     
116     private void checkShardingStrategy(final String databaseName, final ShardingStrategyConfiguration shardingStrategy, final Collection<String> shardingAlgorithms) {
117         if (null == shardingStrategy || shardingStrategy instanceof NoneShardingStrategyConfiguration) {
118             return;
119         }
120         if (shardingStrategy instanceof ComplexShardingStrategyConfiguration) {
121             ShardingSpherePreconditions.checkNotEmpty(((ComplexShardingStrategyConfiguration) shardingStrategy).getShardingColumns(),
122                     () -> new MissingRequiredShardingConfigurationException("Complex sharding columns", databaseName));
123         }
124         ShardingSpherePreconditions.checkNotNull(shardingStrategy.getShardingAlgorithmName(), () -> new MissingRequiredShardingConfigurationException("Sharding algorithm name", databaseName));
125         ShardingSpherePreconditions.checkContains(shardingAlgorithms, shardingStrategy.getShardingAlgorithmName(),
126                 () -> new UnregisteredAlgorithmException("Key generate", shardingStrategy.getShardingAlgorithmName(), new SQLExceptionIdentifier(databaseName)));
127     }
128     
129     @Override
130     public Collection<String> getRequiredDataSourceNames(final ShardingRuleConfiguration ruleConfig) {
131         Collection<String> result = new LinkedHashSet<>();
132         ruleConfig.getTables().forEach(each -> result.addAll(getDataSourceNames(each)));
133         ruleConfig.getAutoTables().forEach(each -> result.addAll(getDataSourceNames(each)));
134         return result;
135     }
136     
137     private Collection<String> getDataSourceNames(final ShardingTableRuleConfiguration shardingTableRuleConfig) {
138         Collection<String> actualDataNodes = InlineExpressionParserFactory.newInstance(shardingTableRuleConfig.getActualDataNodes()).splitAndEvaluate();
139         return actualDataNodes.stream().map(each -> new DataNode(each).getDataSourceName()).collect(Collectors.toList());
140     }
141     
142     private Collection<String> getDataSourceNames(final ShardingAutoTableRuleConfiguration shardingAutoTableRuleConfig) {
143         return new HashSet<>(InlineExpressionParserFactory.newInstance(shardingAutoTableRuleConfig.getActualDataSources()).splitAndEvaluate());
144     }
145     
146     @Override
147     public Collection<String> getTableNames(final ShardingRuleConfiguration ruleConfig) {
148         Collection<String> result = ruleConfig.getTables().stream().map(ShardingTableRuleConfiguration::getLogicTable).collect(Collectors.toList());
149         result.addAll(ruleConfig.getAutoTables().stream().map(ShardingAutoTableRuleConfiguration::getLogicTable).collect(Collectors.toList()));
150         return result;
151     }
152     
153     @Override
154     public int getOrder() {
155         return ShardingOrder.ORDER;
156     }
157     
158     @Override
159     public Class<ShardingRuleConfiguration> getTypeClass() {
160         return ShardingRuleConfiguration.class;
161     }
162 }
