18  package org.apache.shardingsphere.encrypt.algorithm.like;
19  
20  import com.google.common.base.Strings;
21  import lombok.EqualsAndHashCode;
22  import lombok.SneakyThrows;
23  import org.apache.shardingsphere.encrypt.api.context.EncryptContext;
24  import org.apache.shardingsphere.encrypt.api.encrypt.like.LikeEncryptAlgorithm;
25  import org.apache.shardingsphere.encrypt.exception.algorithm.EncryptAlgorithmInitializationException;
26  
27  import java.io.IOException;
28  import java.io.InputStream;
29  import java.util.Map;
30  import java.util.Objects;
31  import java.util.Properties;
32  import java.util.Scanner;
33  import java.util.stream.Collectors;
34  import java.util.stream.IntStream;
35  
36  /**
37   * Char digest like encrypt algorithm.
38   */
39  @EqualsAndHashCode
40  public final class CharDigestLikeEncryptAlgorithm implements LikeEncryptAlgorithm {
41      
42      private static final String DELTA_KEY = "delta";
43      
44      private static final String MASK_KEY = "mask";
45      
46      private static final String START_KEY = "start";
47      
48      private static final String DICT_KEY = "dict";
49      
50      private static final int DEFAULT_DELTA = 1;
51      
52      private static final int DEFAULT_MASK = 0b1111_0111_1101;
53      
54      private static final int DEFAULT_START = 0x4e00;
55      
56      private static final int MAX_NUMERIC_LETTER_CHAR = 255;
57      
58      private int delta;
59      
60      private int mask;
61      
62      private int start;
63      
64      private Map<Character, Integer> charIndexes;
65      
66      @Override
67      public void init(final Properties props) {
68          delta = createDelta(props);
69          mask = createMask(props);
70          start = createStart(props);
71          charIndexes = createCharIndexes(props);
72      }
73      
74      private int createDelta(final Properties props) {
75          if (props.containsKey(DELTA_KEY)) {
76              try {
77                  return Integer.parseInt(props.getProperty(DELTA_KEY));
78              } catch (final NumberFormatException ignored) {
79                  throw new EncryptAlgorithmInitializationException(getType(), "delta can only be a decimal number");
80              }
81          }
82          return DEFAULT_DELTA;
83      }
84      
85      private int createMask(final Properties props) {
86          if (props.containsKey(MASK_KEY)) {
87              try {
88                  return Integer.parseInt(props.getProperty(MASK_KEY));
89              } catch (final NumberFormatException ignored) {
90                  throw new EncryptAlgorithmInitializationException(getType(), "mask can only be a decimal number");
91              }
92          }
93          return DEFAULT_MASK;
94      }
95      
96      private int createStart(final Properties props) {
97          if (props.containsKey(START_KEY)) {
98              try {
99                  return Integer.parseInt(props.getProperty(START_KEY));
100             } catch (final NumberFormatException ignored) {
101                 throw new EncryptAlgorithmInitializationException(getType(), "start can only be a decimal number");
102             }
103         }
104         return DEFAULT_START;
105     }
106     
107     private Map<Character, Integer> createCharIndexes(final Properties props) {
108         String dictContent = props.containsKey(DICT_KEY) && !Strings.isNullOrEmpty(props.getProperty(DICT_KEY)) ? props.getProperty(DICT_KEY) : initDefaultDict();
109         return IntStream.range(0, dictContent.length()).boxed().collect(Collectors.toMap(dictContent::charAt, index -> index, (oldValue, currentValue) -> oldValue));
110     }
111     
112     @SneakyThrows(IOException.class)
113     private String initDefaultDict() {
114         StringBuilder result = new StringBuilder();
115         try (
116                 InputStream inputStream = Objects.requireNonNull(Thread.currentThread().getContextClassLoader().getResourceAsStream("algorithm/like/common_chinese_character.dict"));
117                 Scanner scanner = new Scanner(inputStream)) {
118             while (scanner.hasNextLine()) {
119                 String line = scanner.nextLine();
120                 if (!line.isEmpty() && !line.startsWith("#")) {
121                     result.append(line);
122                 }
123             }
124         }
125         return result.toString();
126     }
127     
128     @Override
129     public String encrypt(final Object plainValue, final EncryptContext encryptContext) {
130         return null == plainValue ? null : digest(String.valueOf(plainValue));
131     }
132     
133     private String digest(final String plainValue) {
134         StringBuilder result = new StringBuilder(plainValue.length());
135         for (char each : plainValue.toCharArray()) {
136             char maskedChar = getMaskedChar(each);
137             if ('%' == maskedChar || '_' == maskedChar) {
138                 result.append(each);
139             } else {
140                 result.append(maskedChar);
141             }
142         }
143         return result.toString();
144     }
145     
146     private char getMaskedChar(final char originalChar) {
147         if ('%' == originalChar || '_' == originalChar) {
148             return originalChar;
149         }
150         if (originalChar <= MAX_NUMERIC_LETTER_CHAR) {
151             return (char) ((originalChar + delta) & mask);
152         }
153         if (charIndexes.containsKey(originalChar)) {
154             return (char) (((charIndexes.get(originalChar) + delta) & mask) + start);
155         }
156         return (char) (((originalChar + delta) & mask) + start);
157     }
158     
159     @Override
160     public String getType() {
161         return "CHAR_DIGEST_LIKE";
162     }
163 }
