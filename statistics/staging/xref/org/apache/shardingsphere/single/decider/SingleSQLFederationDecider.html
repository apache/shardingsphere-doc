18  package org.apache.shardingsphere.single.decider;
19  
20  import org.apache.shardingsphere.infra.binder.context.statement.dml.SelectStatementContext;
21  import org.apache.shardingsphere.infra.datanode.DataNode;
22  import org.apache.shardingsphere.infra.metadata.database.ShardingSphereDatabase;
23  import org.apache.shardingsphere.infra.metadata.database.rule.RuleMetaData;
24  import org.apache.shardingsphere.infra.metadata.database.schema.QualifiedTable;
25  import org.apache.shardingsphere.infra.rule.attribute.datanode.MutableDataNodeRuleAttribute;
26  import org.apache.shardingsphere.single.constant.SingleOrder;
27  import org.apache.shardingsphere.single.rule.SingleRule;
28  import org.apache.shardingsphere.sqlfederation.spi.SQLFederationDecider;
29  
30  import java.util.Collection;
31  import java.util.HashSet;
32  import java.util.List;
33  
34  /**
35   * Single SQL federation decider.
36   */
37  public final class SingleSQLFederationDecider implements SQLFederationDecider<SingleRule> {
38      
39      @Override
40      public boolean decide(final SelectStatementContext selectStatementContext, final List<Object> parameters,
41                            final RuleMetaData globalRuleMetaData, final ShardingSphereDatabase database, final SingleRule rule, final Collection<DataNode> includedDataNodes) {
42          Collection<QualifiedTable> singleTables = getSingleTables(selectStatementContext, database, rule);
43          if (singleTables.isEmpty()) {
44              return false;
45          }
46          if (containsView(database, singleTables)) {
47              return true;
48          }
49          boolean isAllTablesInSameComputeNode = rule.isAllTablesInSameComputeNode(includedDataNodes, singleTables);
50          includedDataNodes.addAll(getTableDataNodes(rule, singleTables));
51          return !isAllTablesInSameComputeNode;
52      }
53      
54      private Collection<QualifiedTable> getSingleTables(final SelectStatementContext selectStatementContext, final ShardingSphereDatabase database, final SingleRule rule) {
55          Collection<QualifiedTable> qualifiedTables = rule.getQualifiedTables(selectStatementContext, database);
56          return rule.getSingleTables(qualifiedTables);
57      }
58      
59      private boolean containsView(final ShardingSphereDatabase database, final Collection<QualifiedTable> singleTables) {
60          for (QualifiedTable each : singleTables) {
61              if (database.getSchema(each.getSchemaName()).containsView(each.getTableName())) {
62                  return true;
63              }
64          }
65          return false;
66      }
67      
68      private Collection<DataNode> getTableDataNodes(final SingleRule rule, final Collection<QualifiedTable> singleTables) {
69          Collection<DataNode> result = new HashSet<>(singleTables.size(), 1F);
70          for (QualifiedTable each : singleTables) {
71              rule.getAttributes().getAttribute(MutableDataNodeRuleAttribute.class).findTableDataNode(each.getSchemaName(), each.getTableName()).ifPresent(result::add);
72          }
73          return result;
74      }
75      
76      @Override
77      public int getOrder() {
78          return SingleOrder.ORDER;
79      }
80      
81      @Override
82      public Class<SingleRule> getTypeClass() {
83          return SingleRule.class;
84      }
85  }
