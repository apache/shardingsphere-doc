77  /**
78   * Preview executor.
79   */
80  public final class PreviewExecutor implements ConnectionSessionRequiredRULExecutor<PreviewStatement> {
81      
82      private final KernelProcessor kernelProcessor = new KernelProcessor();
83      
84      @Override
85      public Collection<String> getColumnNames() {
86          return Arrays.asList("data_source_name", "actual_sql");
87      }
88      
89      @Override
90      public Collection<LocalDataQueryResultRow> getRows(final ShardingSphereMetaData metaData, final ConnectionSession connectionSession, final PreviewStatement sqlStatement) throws SQLException {
91          MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
92          String databaseName = getDatabaseName(connectionSession);
93          RuleMetaData globalRuleMetaData = metaDataContexts.getMetaData().getGlobalRuleMetaData();
94          SQLParserRule sqlParserRule = globalRuleMetaData.getSingleRule(SQLParserRule.class);
95          String sql = SQLHintUtils.removeHint(sqlStatement.getSql());
96          HintValueContext hintValueContext = SQLHintUtils.extractHint(sqlStatement.getSql()).orElseGet(HintValueContext::new);
97          DatabaseType databaseType = metaDataContexts.getMetaData().getDatabase(databaseName).getProtocolType();
98          SQLStatement previewedStatement = sqlParserRule.getSQLParserEngine(databaseType).parse(sql, false);
99          SQLStatementContext sqlStatementContext = new SQLBindEngine(metaDataContexts.getMetaData(), databaseName, hintValueContext).bind(previewedStatement, Collections.emptyList());
100         QueryContext queryContext = new QueryContext(sqlStatementContext, sql, Collections.emptyList(), hintValueContext);
101         connectionSession.setQueryContext(queryContext);
102         if (sqlStatementContext instanceof CursorAvailable && sqlStatementContext instanceof CursorDefinitionAware) {
103             setUpCursorDefinition(sqlStatementContext, connectionSession);
104         }
105         ShardingSphereDatabase database = ProxyContext.getInstance().getDatabase(connectionSession.getDatabaseName());
106         ShardingSpherePreconditions.checkState(database.isComplete(), () -> new RuleNotExistedException(connectionSession.getDatabaseName()));
107         ConfigurationProperties props = metaDataContexts.getMetaData().getProps();
108         String schemaName = queryContext.getSqlStatementContext().getTablesContext().getSchemaName()
109                 .orElseGet(() -> new DatabaseTypeRegistry(database.getProtocolType()).getDefaultSchemaName(databaseName));
110         SQLFederationEngine sqlFederationEngine = new SQLFederationEngine(databaseName, schemaName, metaDataContexts.getMetaData(), metaDataContexts.getStatistics(),
111                 new JDBCExecutor(BackendExecutorContext.getInstance().getExecutorEngine(), connectionSession.getConnectionContext()));
112         Collection<ExecutionUnit> executionUnits = isUseFederation(queryContext, metaDataContexts, connectionSession, sqlFederationEngine)
113                 ? getFederationExecutionUnits(queryContext, metaDataContexts, connectionSession, sqlFederationEngine)
114                 : kernelProcessor.generateExecutionContext(queryContext, database, globalRuleMetaData, props, connectionSession.getConnectionContext()).getExecutionUnits();
115         return executionUnits.stream().map(this::buildRow).collect(Collectors.toList());
116     }
117     
118     private void setUpCursorDefinition(final SQLStatementContext sqlStatementContext, final ConnectionSession connectionSession) {
119         if (!((CursorAvailable) sqlStatementContext).getCursorName().isPresent()) {
120             return;
121         }
122         String cursorName = ((CursorAvailable) sqlStatementContext).getCursorName().get().getIdentifier().getValue().toLowerCase();
123         CursorStatementContext cursorStatementContext = (CursorStatementContext) connectionSession.getConnectionContext().getCursorContext().getCursorDefinitions().get(cursorName);
124         Preconditions.checkArgument(null != cursorStatementContext, "Cursor %s does not exist.", cursorName);
125         ((CursorDefinitionAware) sqlStatementContext).setUpCursorDefinition(cursorStatementContext);
126     }
127     
128     private boolean isUseFederation(final QueryContext queryContext, final MetaDataContexts metaDataContexts, final ConnectionSession connectionSession,
129                                     final SQLFederationEngine sqlFederationEngine) {
130         return sqlFederationEngine.decide(queryContext.getSqlStatementContext(), queryContext.getParameters(),
131                 metaDataContexts.getMetaData().getDatabase(connectionSession.getDatabaseName()), metaDataContexts.getMetaData().getGlobalRuleMetaData());
132     }
133     
134     private LocalDataQueryResultRow buildRow(final ExecutionUnit unit) {
135         return new LocalDataQueryResultRow(unit.getDataSourceName(), unit.getSqlUnit().getSql());
136     }
137     
138     private Collection<ExecutionUnit> getFederationExecutionUnits(final QueryContext queryContext, final MetaDataContexts metaDataContexts,
139                                                                   final ConnectionSession connectionSession, final SQLFederationEngine sqlFederationEngine) {
140         SQLStatement sqlStatement = queryContext.getSqlStatementContext().getSqlStatement();
141         boolean isReturnGeneratedKeys = sqlStatement instanceof MySQLInsertStatement;
142         DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine = createDriverExecutionPrepareEngine(isReturnGeneratedKeys, metaDataContexts, connectionSession);
143         SQLFederationExecutorContext context = new SQLFederationExecutorContext(true, queryContext, metaDataContexts.getMetaData());
144         ShardingSphereDatabase database = metaDataContexts.getMetaData().getDatabase(getDatabaseName(connectionSession));
145         sqlFederationEngine.executeQuery(prepareEngine, createPreviewFederationCallback(database.getProtocolType(), database.getResourceMetaData(), sqlStatement), context);
146         return context.getExecutionUnits();
147     }
148     
149     private JDBCExecutorCallback<ExecuteResult> createPreviewFederationCallback(final DatabaseType protocolType, final ResourceMetaData resourceMetaData,
150                                                                                 final SQLStatement sqlStatement) {
151         return new JDBCExecutorCallback<ExecuteResult>(protocolType, resourceMetaData, sqlStatement, SQLExecutorExceptionHandler.isExceptionThrown()) {
152             
153             @Override
154             protected ExecuteResult executeSQL(final String sql, final Statement statement, final ConnectionMode connectionMode, final DatabaseType storageType) throws SQLException {
155                 return new JDBCStreamQueryResult(statement.executeQuery(sql));
156             }
157             
158             @Override
159             protected Optional<ExecuteResult> getSaneResult(final SQLStatement sqlStatement, final SQLException ex) {
160                 return Optional.empty();
161             }
162         };
163     }
164     
165     private DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> createDriverExecutionPrepareEngine(final boolean isReturnGeneratedKeys, final MetaDataContexts metaDataContexts,
166                                                                                                            final ConnectionSession connectionSession) {
167         int maxConnectionsSizePerQuery = metaDataContexts.getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
168         return new DriverExecutionPrepareEngine<>(JDBCDriverType.STATEMENT, maxConnectionsSizePerQuery, connectionSession.getDatabaseConnectionManager(),
169                 (JDBCBackendStatement) connectionSession.getStatementManager(), new StatementOption(isReturnGeneratedKeys),
170                 metaDataContexts.getMetaData().getDatabase(getDatabaseName(connectionSession)).getRuleMetaData().getRules(),
171                 metaDataContexts.getMetaData().getDatabase(getDatabaseName(connectionSession)).getResourceMetaData().getStorageUnits());
172     }
173     
174     private String getDatabaseName(final ConnectionSession connectionSession) {
175         String result = Strings.isNullOrEmpty(connectionSession.getDatabaseName()) ? connectionSession.getDefaultDatabaseName() : connectionSession.getDatabaseName();
176         ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(result), NoDatabaseSelectedException::new);
177         ShardingSpherePreconditions.checkState(ProxyContext.getInstance().databaseExists(result), () -> new UnknownDatabaseException(result));
178         return result;
179     }
180     
181     @Override
182     public Class<PreviewStatement> getType() {
183         return PreviewStatement.class;
184     }
