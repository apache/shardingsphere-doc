18  package org.apache.shardingsphere.infra.database.core.connector.url;
19  
20  import org.apache.shardingsphere.infra.database.core.connector.ConnectionPropertiesParser;
21  import org.apache.shardingsphere.infra.database.core.spi.DatabaseTypedSPILoader;
22  import org.apache.shardingsphere.infra.database.core.type.DatabaseTypeFactory;
23  
24  import java.util.Properties;
25  import java.util.stream.Collectors;
26  
27  /**
28   * JDBC URL appender.
29   */
30  public final class JdbcUrlAppender {
31      
32      /**
33       * Append query properties.
34       *
35       * @param jdbcUrl JDBC URL to be appended
36       * @param queryProps query properties to be appended
37       * @return appended JDBC URL
38       */
39      public String appendQueryProperties(final String jdbcUrl, final Properties queryProps) {
40          Properties currentQueryProps = DatabaseTypedSPILoader.getService(ConnectionPropertiesParser.class, DatabaseTypeFactory.get(jdbcUrl)).parse(jdbcUrl, null, null).getQueryProperties();
41          if (queryProps.isEmpty()) {
42              return jdbcUrl;
43          }
44          return hasConflictedQueryProperties(currentQueryProps, queryProps)
45                  ? jdbcUrl.substring(0, jdbcUrl.indexOf('?') + 1) + concat(getMergedProperties(currentQueryProps, queryProps))
46                  : jdbcUrl + getURLDelimiter(currentQueryProps) + concat(queryProps);
47      }
48      
49      private boolean hasConflictedQueryProperties(final Properties currentQueryProps, final Properties toBeAppendedQueryProps) {
50          return toBeAppendedQueryProps.keySet().stream().anyMatch(currentQueryProps::containsKey);
51      }
52      
53      private Properties getMergedProperties(final Properties currentQueryProps, final Properties toBeAppendedQueryProps) {
54          Properties result = new Properties();
55          result.putAll(currentQueryProps);
56          result.putAll(toBeAppendedQueryProps);
57          return result;
58      }
59      
60      private String getURLDelimiter(final Properties currentQueryProps) {
61          return currentQueryProps.isEmpty() ? "?" : "&";
62      }
63      
64      private String concat(final Properties queryProps) {
65          return queryProps.entrySet().stream().map(entry -> String.join("=", entry.getKey().toString(), entry.getValue().toString())).collect(Collectors.joining("&"));
66      }
67  }
