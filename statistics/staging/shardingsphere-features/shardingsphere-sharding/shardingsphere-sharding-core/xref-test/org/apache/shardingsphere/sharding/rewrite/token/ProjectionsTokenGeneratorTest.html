49  import static org.hamcrest.CoreMatchers.is;
50  import static org.hamcrest.MatcherAssert.assertThat;
51  import static org.junit.jupiter.api.Assertions.assertFalse;
52  import static org.junit.jupiter.api.Assertions.assertTrue;
53  import static org.mockito.ArgumentMatchers.anyString;
54  import static org.mockito.Mockito.RETURNS_DEEP_STUBS;
55  import static org.mockito.Mockito.mock;
56  import static org.mockito.Mockito.when;
57  
58  @ExtendWith(MockitoExtension.class)
59  @MockitoSettings(strictness = Strictness.LENIENT)
60  class ProjectionsTokenGeneratorTest {
61      
62      private static final String TEST_AGGREGATION_DISTINCT_PROJECTION_DISTINCT_INNER_EXPRESSION = "TEST_AGGREGATION_DISTINCT_PROJECTION_DISTINCT_INNER_EXPRESSION";
63      
64      private static final String TEST_AGGREGATION_DISTINCT_PROJECTION_ALIAS = "TEST_AGGREGATION_DISTINCT_PROJECTION_ALIAS";
65      
66      private static final String TEST_DERIVED_PROJECTION_ALIAS = "TEST_DERIVED_PROJECTION_ALIAS";
67      
68      private static final String TEST_LOGIC_TABLE_NAME = "TEST_LOGIC_TABLE_NAME";
69      
70      private static final String TEST_ACTUAL_TABLE_NAME_WRAPPED = "TEST_ACTUAL_TABLE_NAME_WRAPPED";
71      
72      private static final String TEST_OTHER_DERIVED_PROJECTION_ALIAS = "TEST_OTHER_DERIVED_PROJECTION_ALIAS";
73      
74      private static final String TEST_OTHER_DERIVED_PROJECTION_EXPRESSION = "TEST_OTHER_DERIVED_PROJECTION_EXPRESSION";
75      
76      @Mock
77      private RouteUnit routeUnit;
78      
79      @BeforeEach
80      void setup() {
81          RouteMapper routeMapper = mock(RouteMapper.class);
82          when(routeMapper.getLogicName()).thenReturn(TEST_LOGIC_TABLE_NAME);
83          when(routeMapper.getActualName()).thenReturn("TEST_ACTUAL_TABLE_NAME");
84          when(routeUnit.getTableMappers()).thenReturn(Collections.singleton(routeMapper));
85      }
86      
87      @Test
88      void assertIsGenerateInsertToken() {
89          assertFalse(getProjectionsTokenGenerator().isGenerateSQLToken(mock(InsertStatementContext.class)));
90      }
91      
92      @Test
93      void assertIsNotGenerateSelectToken() {
94          assertFalse(getProjectionsTokenGenerator().isGenerateSQLToken(mock(SelectStatementContext.class, RETURNS_DEEP_STUBS)));
95      }
96      
97      @Test
98      void assertIsGenerateSelectToken() {
99          SelectStatementContext selectStatementContext = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
100         AggregationProjection aggregationProjection = getAggregationProjection();
101         when(selectStatementContext.getProjectionsContext().getProjections()).thenReturn(Collections.singleton(aggregationProjection));
102         assertTrue(getProjectionsTokenGenerator().isGenerateSQLToken(selectStatementContext));
103     }
104     
105     @Test
106     void assertGenerateSQLToken() {
107         SelectStatementContext selectStatementContext = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
108         Collection<Projection> projections = Arrays.asList(getAggregationProjection(), getDerivedProjection(), getOtherDerivedProjection());
109         when(selectStatementContext.getProjectionsContext().getProjections()).thenReturn(projections);
110         when(selectStatementContext.getProjectionsContext().getStopIndex()).thenReturn(2);
111         when(selectStatementContext.getSqlStatement()).thenReturn(new MySQLSelectStatement());
112         ProjectionsTokenGenerator generator = getProjectionsTokenGenerator();
113         ProjectionsToken actual = generator.generateSQLToken(selectStatementContext);
114         assertThat(actual.toString(routeUnit), is(", " + TEST_AGGREGATION_DISTINCT_PROJECTION_DISTINCT_INNER_EXPRESSION + " AS " + TEST_AGGREGATION_DISTINCT_PROJECTION_ALIAS + " "
115                 + ", " + TEST_ACTUAL_TABLE_NAME_WRAPPED + ".null" + " AS " + TEST_DERIVED_PROJECTION_ALIAS + " "
116                 + ", " + TEST_OTHER_DERIVED_PROJECTION_EXPRESSION + " AS " + TEST_OTHER_DERIVED_PROJECTION_ALIAS + " "));
117     }
118     
119     private ProjectionsTokenGenerator getProjectionsTokenGenerator() {
120         RouteContext routeContext = mock(RouteContext.class);
121         when(routeContext.getRouteUnits()).thenReturn(Collections.singleton(routeUnit));
122         ProjectionsTokenGenerator result = new ProjectionsTokenGenerator();
123         result.setRouteContext(routeContext);
124         return result;
125     }
126     
127     private AggregationProjection getAggregationProjection() {
128         AggregationDistinctProjection derivedAggregationDistinctProjection = mock(AggregationDistinctProjection.class);
129         when(derivedAggregationDistinctProjection.getDistinctInnerExpression()).thenReturn(TEST_AGGREGATION_DISTINCT_PROJECTION_DISTINCT_INNER_EXPRESSION);
130         when(derivedAggregationDistinctProjection.getAlias()).thenReturn(Optional.of(TEST_AGGREGATION_DISTINCT_PROJECTION_ALIAS));
131         AggregationProjection result = mock(AggregationProjection.class);
132         when(result.getDerivedAggregationProjections()).thenReturn(Collections.singletonList(derivedAggregationDistinctProjection));
133         return result;
134     }
135     
136     private DerivedProjection getDerivedProjection() {
137         OwnerSegment ownerSegment = mock(OwnerSegment.class, RETURNS_DEEP_STUBS);
138         when(ownerSegment.getIdentifier().getValue()).thenReturn(TEST_LOGIC_TABLE_NAME);
139         when(ownerSegment.getIdentifier().getQuoteCharacter().wrap(anyString())).thenReturn(TEST_ACTUAL_TABLE_NAME_WRAPPED);
140         ColumnOrderByItemSegment oldColumnOrderByItemSegment = mock(ColumnOrderByItemSegment.class, RETURNS_DEEP_STUBS);
141         when(oldColumnOrderByItemSegment.getColumn().getOwner()).thenReturn(Optional.of(ownerSegment));
142         when(oldColumnOrderByItemSegment.getOrderDirection()).thenReturn(mock(OrderDirection.class));
143         when(oldColumnOrderByItemSegment.getColumn().getIdentifier()).thenReturn(mock(IdentifierValue.class));
144         DerivedProjection result = mock(DerivedProjection.class);
145         when(result.getAlias()).thenReturn(Optional.of(TEST_DERIVED_PROJECTION_ALIAS));
146         when(result.getDerivedProjection()).thenReturn(oldColumnOrderByItemSegment);
147         return result;
148     }
149     
150     private DerivedProjection getOtherDerivedProjection() {
151         DerivedProjection result = mock(DerivedProjection.class);
152         when(result.getDerivedProjection()).thenReturn(null);
153         when(result.getAlias()).thenReturn(Optional.of(TEST_OTHER_DERIVED_PROJECTION_ALIAS));
154         when(result.getExpression()).thenReturn(TEST_OTHER_DERIVED_PROJECTION_EXPRESSION);
155         return result;
156     }
