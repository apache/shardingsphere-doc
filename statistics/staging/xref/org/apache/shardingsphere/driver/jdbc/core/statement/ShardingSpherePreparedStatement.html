147     private ShardingSpherePreparedStatement(final ShardingSphereConnection connection, final String sql, final int resultSetType,
148                                             final int resultSetConcurrency, final int resultSetHoldability, final boolean returnGeneratedKeys, final String[] columns) throws SQLException {
149         ShardingSpherePreconditions.checkNotEmpty(sql, () -> new EmptySQLException().toSQLException());
150         this.connection = connection;
151         metaData = connection.getContextManager().getMetaDataContexts().getMetaData();
152         hintValueContext = SQLHintUtils.extractHint(sql);
153         this.sql = SQLHintUtils.removeHint(sql);
154         SQLParserRule sqlParserRule = metaData.getGlobalRuleMetaData().getSingleRule(SQLParserRule.class);
155         SQLParserEngine sqlParserEngine = sqlParserRule.getSQLParserEngine(metaData.getDatabase(connection.getDatabaseName()).getProtocolType());
156         SQLStatement sqlStatement = sqlParserEngine.parse(this.sql, true);
157         sqlStatementContext = new SQLBindEngine(metaData, connection.getDatabaseName(), hintValueContext).bind(sqlStatement, Collections.emptyList());
158         databaseName = sqlStatementContext.getTablesContext().getDatabaseName().orElse(connection.getDatabaseName());
159         connection.getDatabaseConnectionManager().getConnectionContext().setCurrentDatabase(databaseName);
160         parameterMetaData = new ShardingSphereParameterMetaData(sqlStatement);
161         statementOption = returnGeneratedKeys ? new StatementOption(true, columns) : new StatementOption(resultSetType, resultSetConcurrency, resultSetHoldability);
162         ShardingSphereDatabase database = metaData.getDatabase(databaseName);
163         driverExecutorFacade = new DriverExecutorFacade(connection, database);
164         statementsCacheable = isStatementsCacheable(database.getRuleMetaData());
165         selectContainsEnhancedTable = sqlStatementContext instanceof SelectStatementContext && ((SelectStatementContext) sqlStatementContext).isContainsEnhancedTable();
166         statementManager = new StatementManager();
167     }
168     
169     private boolean isStatementsCacheable(final RuleMetaData databaseRuleMetaData) {
170         return databaseRuleMetaData.getAttributes(StorageConnectorReusableRuleAttribute.class).size() == databaseRuleMetaData.getRules().size() && !HintManager.isInstantiated();
171     }
172     
173     @Override
174     public ResultSet executeQuery() throws SQLException {
175         try {
176             if (statementsCacheable && !statements.isEmpty()) {
177                 resetParameters();
178                 return statements.iterator().next().executeQuery();
179             }
180             clearPrevious();
181             QueryContext queryContext = createQueryContext();
182             handleAutoCommit(queryContext.getSqlStatementContext().getSqlStatement());
183             ShardingSphereDatabase database = metaData.getDatabase(databaseName);
184             findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
185             currentResultSet = driverExecutorFacade.executeQuery(database, queryContext, createDriverExecutionPrepareEngine(database), this, columnLabelAndIndexMap,
186                     (StatementAddCallback<PreparedStatement>) this::addStatements, this::replay);
187             if (currentResultSet instanceof ShardingSphereResultSet) {
188                 columnLabelAndIndexMap = ((ShardingSphereResultSet) currentResultSet).getColumnLabelAndIndexMap();
189             }
190             return currentResultSet;
191             // CHECKSTYLE:OFF
192         } catch (final RuntimeException ex) {
193             // CHECKSTYLE:ON
194             handleExceptionInTransaction(connection, metaData);
195             throw SQLExceptionTransformEngine.toSQLException(ex, metaData.getDatabase(databaseName).getProtocolType());
196         } finally {
197             driverExecutorFacade.clear();
198             clearParameters();
199         }
200     }
201     
202     private void handleAutoCommit(final SQLStatement sqlStatement) throws SQLException {
203         if (AutoCommitUtils.needOpenTransaction(sqlStatement)) {
204             connection.handleAutoCommit();
205         }
206     }

208     private void addStatements(final Collection<PreparedStatement> statements, final Collection<List<Object>> parameterSets) {
209         this.statements.addAll(statements);
210         this.parameterSets.addAll(parameterSets);

213     private void resetParameters() throws SQLException {
214         replaySetParameter(statements, Collections.singletonList(getParameters()));
215     }

217     private DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> createDriverExecutionPrepareEngine(final ShardingSphereDatabase database) {
218         int maxConnectionsSizePerQuery = metaData.getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
219         return new DriverExecutionPrepareEngine<>(JDBCDriverType.PREPARED_STATEMENT, maxConnectionsSizePerQuery, connection.getDatabaseConnectionManager(), statementManager, statementOption,
220                 database.getRuleMetaData().getRules(), database.getResourceMetaData().getStorageUnits());
221     }

223     @Override
224     public int executeUpdate() throws SQLException {
225         try {
226             if (statementsCacheable && !statements.isEmpty()) {
227                 resetParameters();
228                 return statements.iterator().next().executeUpdate();
229             }
230             clearPrevious();
231             QueryContext queryContext = createQueryContext();
232             handleAutoCommit(queryContext.getSqlStatementContext().getSqlStatement());
233             ShardingSphereDatabase database = metaData.getDatabase(databaseName);
234             int result = driverExecutorFacade.executeUpdate(database, queryContext, createDriverExecutionPrepareEngine(database),
235                     (sql, statement) -> ((PreparedStatement) statement).executeUpdate(), (StatementAddCallback<PreparedStatement>) this::addStatements, this::replay);
236             findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
237             return result;
238             // CHECKSTYLE:OFF
239         } catch (final RuntimeException ex) {
240             // CHECKSTYLE:ON
241             handleExceptionInTransaction(connection, metaData);
242             throw SQLExceptionTransformEngine.toSQLException(ex, metaData.getDatabase(databaseName).getProtocolType());
243         } finally {
244             clearBatch();
245         }
246     }

248     @Override
249     public boolean execute() throws SQLException {
250         try {
251             if (statementsCacheable && !statements.isEmpty()) {
252                 resetParameters();
253                 return statements.iterator().next().execute();
254             }
255             clearPrevious();
256             QueryContext queryContext = createQueryContext();
257             handleAutoCommit(queryContext.getSqlStatementContext().getSqlStatement());
258             ShardingSphereDatabase database = metaData.getDatabase(databaseName);
259             boolean result = driverExecutorFacade.execute(database, queryContext, createDriverExecutionPrepareEngine(database), (sql, statement) -> ((PreparedStatement) statement).execute(),
260                     (StatementAddCallback<PreparedStatement>) this::addStatements, this::replay);
261             findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
262             return result;
263             // CHECKSTYLE:OFF
264         } catch (final RuntimeException ex) {
265             // CHECKSTYLE:ON
266             handleExceptionInTransaction(connection, metaData);
267             throw SQLExceptionTransformEngine.toSQLException(ex, metaData.getDatabase(databaseName).getProtocolType());
268         } finally {
269             clearBatch();
270         }
271     }

273     @Override
274     public ResultSet getResultSet() throws SQLException {
275         if (null != currentResultSet) {
276             return currentResultSet;
277         }
278         Optional<ResultSet> resultSet = driverExecutorFacade.getResultSet();
279         if (resultSet.isPresent()) {
280             return resultSet.get();
281         }
282         if (sqlStatementContext instanceof SelectStatementContext || sqlStatementContext.getSqlStatement() instanceof DALStatement) {
283             List<ResultSet> resultSets = getResultSets();
284             if (resultSets.isEmpty()) {
285                 return currentResultSet;
286             }
287             MergedResult mergedResult = mergeQuery(getQueryResults(resultSets), sqlStatementContext);
288             if (null == columnLabelAndIndexMap) {
289                 columnLabelAndIndexMap = ShardingSphereResultSetUtils.createColumnLabelAndIndexMap(sqlStatementContext, selectContainsEnhancedTable, resultSets.get(0).getMetaData());
290             }
291             currentResultSet = new ShardingSphereResultSet(resultSets, mergedResult, this, selectContainsEnhancedTable, sqlStatementContext, columnLabelAndIndexMap);
292         }
293         return currentResultSet;
294     }
295     
296     private List<ResultSet> getResultSets() throws SQLException {
297         List<ResultSet> result = new ArrayList<>(statements.size());
298         for (Statement each : statements) {
299             if (null != each.getResultSet()) {
300                 result.add(each.getResultSet());
301             }
302         }
303         return result;
304     }
305     
306     private List<QueryResult> getQueryResults(final List<ResultSet> resultSets) throws SQLException {
307         List<QueryResult> result = new ArrayList<>(resultSets.size());
308         for (ResultSet each : resultSets) {
309             if (null != each) {
310                 result.add(new JDBCStreamQueryResult(each));
311             }
312         }
313         return result;
314     }
315     
316     private QueryContext createQueryContext() {
317         List<Object> params = new ArrayList<>(getParameters());
318         if (sqlStatementContext instanceof ParameterAware) {
319             ((ParameterAware) sqlStatementContext).setUpParameters(params);
320         }
321         return new QueryContext(sqlStatementContext, sql, params, hintValueContext, true);
322     }
323     
324     private MergedResult mergeQuery(final List<QueryResult> queryResults, final SQLStatementContext sqlStatementContext) throws SQLException {
325         MergeEngine mergeEngine = new MergeEngine(metaData.getGlobalRuleMetaData(), metaData.getDatabase(databaseName),
326                 metaData.getProps(), connection.getDatabaseConnectionManager().getConnectionContext());
327         return mergeEngine.merge(queryResults, sqlStatementContext);
328     }
329     
330     private void replay() throws SQLException {
331         replaySetParameter(statements, parameterSets);
332         for (Statement each : statements) {
333             getMethodInvocationRecorder().replay(each);
334         }
335     }
336     
337     private void replaySetParameter(final List<PreparedStatement> statements, final List<List<Object>> parameterSets) throws SQLException {
338         for (int i = 0; i < statements.size(); i++) {
339             replaySetParameter(statements.get(i), parameterSets.get(i));
340         }
341     }

343     private void clearPrevious() {
344         currentResultSet = null;
345         statements.clear();
346         parameterSets.clear();
347         generatedValues.clear();
348     }
349     
350     private Optional<GeneratedKeyContext> findGeneratedKey() {
351         return sqlStatementContext instanceof InsertStatementContext ? ((InsertStatementContext) sqlStatementContext).getGeneratedKeyContext() : Optional.empty();
352     }
353     
354     @Override
355     public ResultSet getGeneratedKeys() throws SQLException {
356         if (null != currentBatchGeneratedKeysResultSet) {
357             return currentBatchGeneratedKeysResultSet;
358         }
359         Optional<GeneratedKeyContext> generatedKey = findGeneratedKey();
360         if (generatedKey.isPresent() && statementOption.isReturnGeneratedKeys() && !generatedValues.isEmpty()) {
361             return new GeneratedKeysResultSet(getGeneratedKeysColumnName(generatedKey.get().getColumnName()), generatedValues.iterator(), this);
362         }
363         for (PreparedStatement each : statements) {
364             ResultSet resultSet = each.getGeneratedKeys();
365             while (resultSet.next()) {
366                 generatedValues.add((Comparable<?>) resultSet.getObject(1));
367             }
368         }
369         String columnName = generatedKey.map(GeneratedKeyContext::getColumnName).orElse(null);
370         return new GeneratedKeysResultSet(getGeneratedKeysColumnName(columnName), generatedValues.iterator(), this);
371     }
372     
373     private String getGeneratedKeysColumnName(final String columnName) {
374         return DatabaseTypedSPILoader.findService(GeneratedKeyColumnProvider.class, metaData.getDatabase(databaseName).getProtocolType())
375                 .map(GeneratedKeyColumnProvider::getColumnName).orElse(columnName);
376     }
377     
378     @Override
379     public void addBatch() {
380         currentResultSet = null;
381         QueryContext queryContext = createQueryContext();
382         driverExecutorFacade.addBatch(queryContext, metaData.getDatabase(databaseName));
383         findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
384         clearParameters();
385     }
386     
387     @Override
388     public int[] executeBatch() throws SQLException {
389         ShardingSphereDatabase database = metaData.getDatabase(databaseName);
390         try {
391             return driverExecutorFacade.executeBatch(database, sqlStatementContext, generatedValues, statementOption,
392                     createDriverExecutionPrepareEngine(database), (StatementAddCallback<PreparedStatement>) (statements, parameterSets) -> this.statements.addAll(statements),
393                     this::replaySetParameter,
394                     () -> {
395                         currentBatchGeneratedKeysResultSet = getGeneratedKeys();
396                         statements.clear();
397                     });
398             // CHECKSTYLE:OFF
399         } catch (final RuntimeException ex) {
400             // CHECKSTYLE:ON
401             handleExceptionInTransaction(connection, metaData);
402             throw SQLExceptionTransformEngine.toSQLException(ex, database.getProtocolType());
403         } finally {
404             clearBatch();
405         }
406     }

432     @Override
433     public boolean isAccumulate() {
434         for (DataNodeRuleAttribute each : metaData.getDatabase(databaseName).getRuleMetaData().getAttributes(DataNodeRuleAttribute.class)) {
435             if (each.isNeedAccumulate(sqlStatementContext.getTablesContext().getTableNames())) {
436                 return true;
437             }
438         }
439         return false;
440     }
