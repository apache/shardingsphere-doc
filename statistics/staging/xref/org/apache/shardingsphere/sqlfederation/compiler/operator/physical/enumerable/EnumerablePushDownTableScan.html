55  /**
56   * Enumerable push down table scan.
57   */
58  @Getter
59  public final class EnumerablePushDownTableScan extends TableScan implements EnumerableRel {
60      
61      private final SQLFederationTable sqlFederationTable;
62      
63      private final int[] fields;
64      
65      private final List<RexNode> filters;
66      
67      private final int number;
68      
69      private final List<RexNode> expressions;
70      
71      public EnumerablePushDownTableScan(final RelOptCluster cluster, final RelOptTable table, final SQLFederationTable sqlFederationTable, final int[] fields) {
72          super(cluster, cluster.traitSetOf(EnumerableConvention.INSTANCE), ImmutableList.of(), table);
73          this.sqlFederationTable = sqlFederationTable;
74          this.fields = fields;
75          this.number = fields.length;
76          this.filters = null;
77          this.expressions = new ArrayList<>();
78      }
79      
80      public EnumerablePushDownTableScan(final RelOptCluster cluster, final RelOptTable table, final SQLFederationTable sqlFederationTable, final int[] fields, final int number) {
81          super(cluster, cluster.traitSetOf(EnumerableConvention.INSTANCE), ImmutableList.of(), table);
82          this.sqlFederationTable = sqlFederationTable;
83          this.fields = fields;
84          this.number = number;
85          this.filters = null;
86          this.expressions = new ArrayList<>();
87      }
88      
89      public EnumerablePushDownTableScan(final RelOptCluster cluster, final RelOptTable table, final SQLFederationTable sqlFederationTable,
90                                         final List<RexNode> filters, final int[] fields) {
91          super(cluster, cluster.traitSetOf(EnumerableConvention.INSTANCE), ImmutableList.of(), table);
92          this.sqlFederationTable = sqlFederationTable;
93          this.fields = fields;
94          this.number = fields.length;
95          this.filters = filters;
96          this.expressions = new ArrayList<>();
97      }
98      
99      public EnumerablePushDownTableScan(final RelOptCluster cluster, final RelOptTable table, final SQLFederationTable sqlFederationTable,
100                                        final List<RexNode> filters, final int[] fields, final int number, final List<RexNode> expressions) {
101         super(cluster, cluster.traitSetOf(EnumerableConvention.INSTANCE), ImmutableList.of(), table);
102         this.sqlFederationTable = sqlFederationTable;
103         this.fields = fields;
104         this.number = number;
105         this.filters = filters;
106         this.expressions = expressions;
107     }
108     
109     @Override
110     public RelNode copy(final RelTraitSet traitSet, final List<RelNode> inputs) {
111         return new EnumerablePushDownTableScan(getCluster(), table, sqlFederationTable, fields, number);
112     }
113     
114     @Override
115     public String toString() {
116         if (null == filters) {
117             return "EnumerablePushDownTableScan{sqlFederationTable=" + sqlFederationTable + ", fields=" + Arrays.toString(fields) + '}';
118         }
119         String[] filterValues = new String[number];
120         addFilter(filters, filterValues);
121         return "EnumerablePushDownTableScan{sqlFederationTable=" + sqlFederationTable + ", fields=" + Arrays.toString(fields) + ", filters=" + Arrays.toString(filterValues) + '}';
122     }
123     
124     @Override
125     public RelWriter explainTerms(final RelWriter relWriter) {
126         if (null == filters) {
127             return super.explainTerms(relWriter).item("fields", Primitive.asList(fields));
128         }
129         String[] filterValues = new String[number];
130         addFilter(filters, filterValues);
131         return super.explainTerms(relWriter).item("fields", Primitive.asList(fields)).item("filters", Primitive.asList(filterValues));
132     }
133     
134     @Override
135     public RelDataType deriveRowType() {
136         List<RelDataTypeField> fieldList = table.getRowType().getFieldList();
137         RelDataTypeFactory.Builder builder = getCluster().getTypeFactory().builder();
138         for (int field : fields) {
139             builder.add(fieldList.get(field));
140         }
141         return builder.build();
142     }
143     
144     @Override
145     public void register(final RelOptPlanner planner) {
146         planner.addRule(PushFilterIntoScanRule.INSTANCE);
147         planner.addRule(PushProjectIntoScanRule.INSTANCE);
148     }
149     
150     @Override
151     public RelOptCost computeSelfCost(final RelOptPlanner planner, final RelMetadataQuery mq) {
152         return super.computeSelfCost(planner, mq).multiplyBy((number + 2D) / (table.getRowType().getFieldCount() + 2D));
153     }
154     
155     @Override
156     public Result implement(final EnumerableRelImplementor implementor, final Prefer pref) {
157         PhysType physType = PhysTypeImpl.of(implementor.getTypeFactory(), getRowType(), pref.preferArray());
158         if (null == filters) {
159             return generateCodeForNullFilters(implementor, physType);
160         } else {
161             String[] filterValues = new String[number];
162             addFilter(filters, filterValues);
163             return generateCodeForFilters(implementor, physType, filterValues);
164         }
165     }
166     
167     private Result generateCodeForNullFilters(final EnumerableRelImplementor implementor, final PhysType physType) {
168         if (fields.length == 1) {
169             return implementor.result(physType, Blocks.toBlock(Expressions.call(table.getExpression(SQLFederationTable.class),
170                     "projectScalar", implementor.getRootExpression(), Expressions.constant(fields))));
171         }
172         return implementor.result(physType, Blocks.toBlock(Expressions.call(table.getExpression(SQLFederationTable.class),
173                 "project", implementor.getRootExpression(), Expressions.constant(fields))));
174     }
175     
176     private Result generateCodeForFilters(final EnumerableRelImplementor implementor, final PhysType physType, final String[] filterValues) {
177         if (fields.length == 1) {
178             return implementor.result(physType, Blocks.toBlock(Expressions.call(table.getExpression(SQLFederationTable.class),
179                     "projectAndFilterScalar", implementor.getRootExpression(), Expressions.constant(filterValues), Expressions.constant(fields))));
180         }
181         return implementor.result(physType, Blocks.toBlock(Expressions.call(table.getExpression(SQLFederationTable.class),
182                 "projectAndFilter", implementor.getRootExpression(), Expressions.constant(filterValues), Expressions.constant(fields))));
183     }
184     
185     private void addFilter(final List<RexNode> filters, final String[] filterValues) {
186         int index = 0;
187         for (RexNode each : filters) {
188             RexCall call = (RexCall) each;
189             String columnMap = generateColumnMap(call);
190             filterValues[index] = each + columnMap;
191             index++;
192         }
193     }
194     
195     private String generateColumnMap(final RexCall call) {
196         Map<Integer, Integer> result = new HashMap<>();
197         traverseRexCall(call, result);
198         return result.toString();
199     }
200     
201     private void traverseRexCall(final RexCall call, final Map<Integer, Integer> columnMap) {
202         for (RexNode each : call.getOperands()) {
203             if (each instanceof RexInputRef) {
204                 RexInputRef reference = (RexInputRef) each;
205                 String referenceName = reference.getName();
206                 int columnId = Integer.parseInt(referenceName.replace("$", ""));
207                 int columnType = sqlFederationTable.getColumnType(columnId);
208                 columnMap.put(columnId, columnType);
209             }
210             if (each instanceof RexCall) {
211                 traverseRexCall((RexCall) each, columnMap);
212             }
213         }
214     }
