64      void assertGetValueWhenConfigPlainColumn() throws SQLException {
65          when(queryResult.next()).thenReturn(true).thenReturn(false);
66          when(queryResult.getValue(2, String.class)).thenReturn(
67                  "CREATE TABLE `t_encrypt` (`id` INT NOT NULL, `user_id_cipher` VARCHAR(100) NOT NULL, "
68                          + "`user_id` VARCHAR(100) NOT NULL, `order_id` VARCHAR(30) NOT NULL, PRIMARY KEY (`id`)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;");
69          MergedEncryptShowCreateTableMergedResult actual = createMergedEncryptShowCreateTableMergedResult(queryResult,
70                  mockEncryptRule(Collections.singletonList(new EncryptColumnRuleConfiguration("user_id", "user_id_cipher", null, "", "user_id", null, false))));
71          assertTrue(actual.next());
72          assertThat(actual.getValue(2, String.class),
73                  is("CREATE TABLE `t_encrypt` (`id` INT NOT NULL, `user_id` VARCHAR(100) NOT NULL, `order_id` VARCHAR(30) NOT NULL, PRIMARY KEY (`id`)) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;"));

116     private EncryptRule mockEncryptRule(final Collection<EncryptColumnRuleConfiguration> columnRuleConfigurations) {
117         EncryptRule result = mock(EncryptRule.class);
118         EncryptTable encryptTable = new EncryptTable(new EncryptTableRuleConfiguration("t_encrypt", columnRuleConfigurations, false));
119         when(result.findEncryptTable("t_encrypt")).thenReturn(Optional.of(encryptTable));
120         return result;
121     }
122     
123     @Test
124     void assertWasNull() throws SQLException {
125         assertFalse(createMergedEncryptShowCreateTableMergedResult(queryResult, mock(EncryptRule.class)).wasNull());
126     }
127     
128     private MergedEncryptShowCreateTableMergedResult createMergedEncryptShowCreateTableMergedResult(final QueryResult queryResult, final EncryptRule encryptRule) {
129         ShowCreateTableStatementContext sqlStatementContext = mock(ShowCreateTableStatementContext.class);
130         IdentifierValue identifierValue = new IdentifierValue("t_encrypt");
131         TableNameSegment tableNameSegment = new TableNameSegment(1, 4, identifierValue);
132         SimpleTableSegment simpleTableSegment = new SimpleTableSegment(tableNameSegment);
133         when(sqlStatementContext.getAllTables()).thenReturn(Collections.singletonList(simpleTableSegment));
134         return new MergedEncryptShowCreateTableMergedResult(queryResult, sqlStatementContext, encryptRule);
135     }
