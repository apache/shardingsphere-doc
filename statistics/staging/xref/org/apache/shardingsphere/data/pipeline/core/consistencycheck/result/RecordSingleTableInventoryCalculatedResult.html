18  package org.apache.shardingsphere.data.pipeline.core.consistencycheck.result;
19  
20  import lombok.Getter;
21  import lombok.extern.slf4j.Slf4j;
22  import org.apache.commons.lang3.builder.EqualsBuilder;
23  import org.apache.commons.lang3.builder.HashCodeBuilder;
24  import org.apache.shardingsphere.data.pipeline.core.consistencycheck.DataConsistencyCheckUtils;
25  
26  import java.util.Iterator;
27  import java.util.List;
28  import java.util.Map;
29  import java.util.Optional;
30  
31  /**
32   * Record single table inventory calculated result.
33   */
34  @Getter
35  @Slf4j
36  public final class RecordSingleTableInventoryCalculatedResult implements SingleTableInventoryCalculatedResult {
37      
38      private final Object maxUniqueKeyValue;
39      
40      private final int recordsCount;
41      
42      private final List<Map<String, Object>> records;
43      
44      public RecordSingleTableInventoryCalculatedResult(final Object maxUniqueKeyValue, final List<Map<String, Object>> records) {
45          this.maxUniqueKeyValue = maxUniqueKeyValue;
46          recordsCount = records.size();
47          this.records = records;
48      }
49      
50      @Override
51      public Optional<Object> getMaxUniqueKeyValue() {
52          return Optional.of(maxUniqueKeyValue);
53      }
54      
55      @Override
56      public boolean equals(final Object o) {
57          if (null == o) {
58              return false;
59          }
60          if (this == o) {
61              return true;
62          }
63          if (!(o instanceof RecordSingleTableInventoryCalculatedResult)) {
64              log.warn("RecordSingleTableInventoryCalculatedResult type not match, o.className={}.", o.getClass().getName());
65              return false;
66          }
67          final RecordSingleTableInventoryCalculatedResult that = (RecordSingleTableInventoryCalculatedResult) o;
68          EqualsBuilder equalsBuilder = new EqualsBuilder();
69          if (recordsCount != that.recordsCount || !DataConsistencyCheckUtils.isMatched(equalsBuilder, maxUniqueKeyValue, that.maxUniqueKeyValue)) {
70              log.warn("Record count or max unique key value not match, recordCount1={}, recordCount2={}, maxUniqueKeyValue1={}, maxUniqueKeyValue2={}, value1.class={}, value2.class={}.",
71                      recordsCount, that.recordsCount, maxUniqueKeyValue, that.maxUniqueKeyValue,
72                      null == maxUniqueKeyValue ? "" : maxUniqueKeyValue.getClass().getName(), null == that.maxUniqueKeyValue ? "" : that.maxUniqueKeyValue.getClass().getName());
73              return false;
74          }
75          Iterator<Map<String, Object>> thisRecordsIterator = records.iterator();
76          Iterator<Map<String, Object>> thatRecordsIterator = that.records.iterator();
77          while (thisRecordsIterator.hasNext() && thatRecordsIterator.hasNext()) {
78              Map<String, Object> thisRecord = thisRecordsIterator.next();
79              Map<String, Object> thatRecord = thatRecordsIterator.next();
80              if (thisRecord.size() != thatRecord.size()) {
81                  log.warn("Record column size not match, size1={}, size2={}, record1={}, record2={}.", thisRecord.size(), thatRecord.size(), thisRecord, thatRecord);
82                  return false;
83              }
84              if (!DataConsistencyCheckUtils.recordsEquals(thisRecord, thatRecord, equalsBuilder)) {
85                  log.warn("Records not equals, record1={}, record2={}.", thisRecord, thatRecord);
86                  return false;
87              }
88          }
89          return true;
90      }
91      
92      @Override
93      public int hashCode() {
94          return new HashCodeBuilder(17, 37).append(getMaxUniqueKeyValue().orElse(null)).append(getRecordsCount()).append(records).toHashCode();
95      }
96  }
