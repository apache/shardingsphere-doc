18  package org.apache.shardingsphere.proxy.backend.postgresql.handler.admin.executor;
19  
20  import org.apache.shardingsphere.infra.metadata.database.resource.ShardingSphereResourceMetaData;
21  import org.apache.shardingsphere.proxy.backend.context.ProxyContext;
22  import org.apache.shardingsphere.proxy.backend.session.ConnectionSession;
23  import org.apache.shardingsphere.proxy.backend.handler.admin.executor.AbstractDatabaseMetaDataExecutor;
24  import org.apache.shardingsphere.proxy.backend.handler.admin.executor.AbstractDatabaseMetaDataExecutor.DefaultDatabaseMetaDataExecutor;
25  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.item.ColumnProjectionSegment;
26  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.item.ProjectionSegment;
27  import org.apache.shardingsphere.sql.parser.sql.common.statement.dml.SelectStatement;
28  
29  import java.util.Collection;
30  import java.util.Collections;
31  import java.util.LinkedHashSet;
32  import java.util.LinkedList;
33  import java.util.List;
34  import java.util.Map;
35  import java.util.Set;
36  import java.util.stream.Collectors;
37  
38  /**
39   * Database meta data query executor, used to query the schemata database.
40   */
41  public final class SelectDatabaseExecutor extends DefaultDatabaseMetaDataExecutor {
42      
43      private static final String DATABASE_NAME = "databasename";
44      
45      private static final String DATNAME = "datname";
46      
47      private static final String NAME = "name";
48      
49      private final Set<String> columnNames = new LinkedHashSet<>();
50      
51      private final SelectStatement sqlStatement;
52      
53      private String databaseNameAlias = DATABASE_NAME;
54      
55      private boolean isQueryDatabase;
56      
57      public SelectDatabaseExecutor(final SelectStatement sqlStatement, final String sql) {
58          super(sql);
59          this.sqlStatement = sqlStatement;
60      }
61      
62      @Override
63      protected void createPreProcessing() {
64          removeDuplicatedRow();
65          addDefaultRow();
66      }
67      
68      private void addDefaultRow() {
69          Collection<String> schemaWithoutDataSource = ProxyContext.getInstance().getAllDatabaseNames().stream().filter(each -> !hasDataSource(each)).collect(Collectors.toList());
70          schemaWithoutDataSource.forEach(each -> getRows().addLast(getDefaultRowData(each)));
71      }
72      
73      private void removeDuplicatedRow() {
74          if (isQueryDatabase) {
75              List<Map<String, Object>> toBeRemovedRow = getRows().stream().collect(Collectors.groupingBy(each -> each.get(databaseNameAlias), Collectors.toCollection(LinkedList::new)))
76                      .values().stream().filter(each -> each.size() > 1).map(LinkedList::getLast).collect(Collectors.toList());
77              toBeRemovedRow.forEach(each -> getRows().remove(each));
78          }
79      }
80      
81      @Override
82      protected List<String> getDatabaseNames(final ConnectionSession connectionSession) {
83          Collection<String> databaseNames = ProxyContext.getInstance().getAllDatabaseNames().stream().filter(each -> isAuthorized(each, connectionSession.getGrantee())).collect(Collectors.toList());
84          return databaseNames.stream().filter(AbstractDatabaseMetaDataExecutor::hasDataSource).collect(Collectors.toList());
85      }
86      
87      @Override
88      protected void rowPostProcessing(final String databaseName, final Map<String, Object> rowMap, final Map<String, String> aliasMap) {
89          buildColumnNames(aliasMap);
90          ShardingSphereResourceMetaData resourceMetaData = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getDatabase(databaseName).getResourceMetaData();
91          Set<String> catalogs = resourceMetaData.getDataSources().keySet().stream().map(each -> resourceMetaData.getDataSourceMetaData(each).getCatalog()).collect(Collectors.toSet());
92          databaseNameAlias = aliasMap.getOrDefault(DATABASE_NAME, aliasMap.getOrDefault(DATNAME, aliasMap.getOrDefault(NAME, "")));
93          String rowValue = rowMap.getOrDefault(databaseNameAlias, "").toString();
94          isQueryDatabase = !rowValue.isEmpty();
95          if (catalogs.contains(rowValue)) {
96              rowMap.replace(databaseNameAlias, databaseName);
97          } else {
98              rowMap.clear();
99          }
100     }
101     
102     private void buildColumnNames(final Map<String, String> aliasMap) {
103         aliasMap.forEach((key, value) -> columnNames.add(value.isEmpty() ? key : value));
104         
105     }
106     
107     private Map<String, Object> getDefaultRowData(final String schemaName) {
108         Map<String, Object> result;
109         if (columnNames.isEmpty()) {
110             columnNames.addAll(getDefaultColumnNames());
111         }
112         result = columnNames.stream().collect(Collectors.toMap(each -> each, each -> ""));
113         result.replace(databaseNameAlias, schemaName);
114         return result;
115     }
116     
117     private Set<String> getDefaultColumnNames() {
118         Collection<ProjectionSegment> projections = sqlStatement.getProjections().getProjections();
119         if (projections.stream().anyMatch(each -> !(each instanceof ColumnProjectionSegment))) {
120             return Collections.singleton(databaseNameAlias);
121         }
122         return projections.stream().map(each -> {
123             ColumnProjectionSegment segment = (ColumnProjectionSegment) each;
124             return segment.getAlias().isPresent() ? segment.getAlias().get() : segment.getColumn().getIdentifier().getValue();
125         }).collect(Collectors.toCollection(LinkedHashSet::new));
126     }
127 }
