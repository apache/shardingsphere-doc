335     private void initProxyContext(final Map<String, ShardingSphereDatabase> databases) {
336         ContextManager contextManager = mock(ContextManager.class, RETURNS_DEEP_STUBS);
337         MetaDataContexts metaDataContexts = new MetaDataContexts(mock(MetaDataPersistService.class),
338                 new ShardingSphereMetaData(databases, mock(ShardingSphereRuleMetaData.class), new ConfigurationProperties(new Properties())));
339         when(contextManager.getMetaDataContexts()).thenReturn(metaDataContexts);
340         when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
341     }
