33  import static org.hamcrest.CoreMatchers.instanceOf;
34  import static org.hamcrest.CoreMatchers.is;
35  import static org.hamcrest.MatcherAssert.assertThat;
36  import static org.junit.jupiter.api.Assertions.assertNull;
37  import static org.junit.jupiter.api.Assertions.assertThrows;
38  import static org.mockito.Mockito.mock;
39  import static org.mockito.Mockito.when;

41  class TestDecodingPluginTest {
42      
43      private final LogSequenceNumber pgSequenceNumber = LogSequenceNumber.valueOf("0/14EFDB8");
44      
45      private final PostgreSQLLogSequenceNumber logSequenceNumber = new PostgreSQLLogSequenceNumber(pgSequenceNumber);
46      
47      @Test
48      void assertDecodeWriteRowEvent() {
49          ByteBuffer data = ByteBuffer.wrap(("table public.test: INSERT: data[character varying]:' 1 2 3'' ðŸ˜Šä¸­' t_json_empty[json]:'{}' t_json[json]:'{\"test\":\"ä¸­ä¸­{ä¸­ä¸­}' ä¸­\"}'"
50                  + " t_jsonb[jsonb]:'{\"test\":\"ðŸ˜ŠEmojiä¸­\"}'").getBytes());
51          WriteRowEvent actual = (WriteRowEvent) new TestDecodingPlugin(null).decode(data, logSequenceNumber);
52          assertThat(actual.getLogSequenceNumber(), is(logSequenceNumber));
53          assertThat(actual.getTableName(), is("test"));
54          assertThat(actual.getAfterRow().get(0), is(" 1 2 3' ðŸ˜Šä¸­"));
55          assertThat(actual.getAfterRow().get(1), is("{}"));
56          assertThat(actual.getAfterRow().get(2), is("{\"test\":\"ä¸­ä¸­{ä¸­ä¸­}' ä¸­\"}"));
57          assertThat(actual.getAfterRow().get(3), is("{\"test\":\"ðŸ˜ŠEmojiä¸­\"}"));
58      }
59      
60      @Test
61      void assertDecodeUpdateRowEvent() {
62          ByteBuffer data = ByteBuffer.wrap("table public.test: UPDATE: unicode[character varying]:' 1 2 3'' ðŸ˜Šä¸­ ' t_json_empty[json]:'{}' t_json[json]:'{\"test\":\"ä¸­ä¸­{ä¸­ä¸­}' ä¸­\"}'".getBytes());
63          UpdateRowEvent actual = (UpdateRowEvent) new TestDecodingPlugin(null).decode(data, logSequenceNumber);
64          assertThat(actual.getLogSequenceNumber(), is(logSequenceNumber));
65          assertThat(actual.getTableName(), is("test"));
66          assertThat(actual.getAfterRow().get(0), is(" 1 2 3' ðŸ˜Šä¸­ "));
67          assertThat(actual.getAfterRow().get(1), is("{}"));
68          assertThat(actual.getAfterRow().get(2), is("{\"test\":\"ä¸­ä¸­{ä¸­ä¸­}' ä¸­\"}"));
69      }
70      
71      @Test
72      void assertDecodeDeleteRowEvent() {
73          ByteBuffer data = ByteBuffer.wrap("table public.test: DELETE: data[integer]:1".getBytes());
74          DeleteRowEvent actual = (DeleteRowEvent) new TestDecodingPlugin(null).decode(data, logSequenceNumber);
75          assertThat(actual.getLogSequenceNumber(), is(logSequenceNumber));
76          assertThat(actual.getTableName(), is("test"));
77          assertThat(actual.getPrimaryKeys().get(0), is(1));
78      }
79      
80      @Test
81      void assertDecodeWriteRowEventWithByteA() {
82          ByteBuffer data = ByteBuffer.wrap("table public.test: INSERT: data[bytea]:'\\xff00ab'".getBytes());
83          WriteRowEvent actual = (WriteRowEvent) new TestDecodingPlugin(null).decode(data, logSequenceNumber);
84          assertThat(actual.getLogSequenceNumber(), is(logSequenceNumber));
85          assertThat(actual.getTableName(), is("test"));
86          assertThat(actual.getAfterRow().get(0), is(new byte[]{(byte) 0xff, (byte) 0, (byte) 0xab}));
87      }
88      
89      @Test
90      void assertDecodeUnknownTableType() {
91          ByteBuffer data = ByteBuffer.wrap("unknown".getBytes());
92          assertThat(new TestDecodingPlugin(null).decode(data, logSequenceNumber), instanceOf(PlaceholderEvent.class));
93      }
94      
95      @Test
96      void assertDecodeUnknownRowEventType() {
97          ByteBuffer data = ByteBuffer.wrap("table public.test: UNKNOWN: data[character varying]:'1 2 3'''".getBytes());
98          assertThrows(IngestException.class, () -> new TestDecodingPlugin(null).decode(data, logSequenceNumber));
99      }
100     
101     @Test
102     void assertDecodeTime() throws SQLException {
103         TimestampUtils timestampUtils = mock(TimestampUtils.class);
104         when(timestampUtils.toTime(null, "1 2 3'")).thenThrow(new SQLException(""));
105         ByteBuffer data = ByteBuffer.wrap("table public.test: INSERT: data[time without time zone]:'1 2 3'''".getBytes());
106         assertThrows(DecodingException.class, () -> new TestDecodingPlugin(new PostgreSQLTimestampUtils(timestampUtils)).decode(data, logSequenceNumber));
107     }
108     
109     @Test
110     void assertDecodeInsertWithNullValue() {
111         ByteBuffer data = ByteBuffer.wrap("table public.test: INSERT: id[integer]:123 col0[integer]:null col1[character varying]:null col2[character varying]:'nonnull'".getBytes());
112         AbstractWALEvent actual = new TestDecodingPlugin(null).decode(data, logSequenceNumber);
113         assertThat(actual, instanceOf(WriteRowEvent.class));
114         WriteRowEvent actualWriteRowEvent = (WriteRowEvent) actual;
115         assertThat(actualWriteRowEvent.getAfterRow().get(0), is(123));
116         assertNull(actualWriteRowEvent.getAfterRow().get(1));
117         assertNull(actualWriteRowEvent.getAfterRow().get(2));
118         assertThat(actualWriteRowEvent.getAfterRow().get(3), is("nonnull"));
119     }
120     
121     @Test
122     void assertDecodeJsonValue() {
123         ByteBuffer data = ByteBuffer.wrap("table public.test: INSERT: id[integer]:123 ".getBytes());
124         AbstractWALEvent actual = new TestDecodingPlugin(null).decode(data, logSequenceNumber);
125         assertThat(actual, instanceOf(WriteRowEvent.class));
126     }
127 }
