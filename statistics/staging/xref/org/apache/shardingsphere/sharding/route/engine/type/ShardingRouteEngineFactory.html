71  /**
72   * Sharding routing engine factory.
73   */
74  @NoArgsConstructor(access = AccessLevel.PRIVATE)
75  public final class ShardingRouteEngineFactory {
76      
77      /**
78       * Create new instance of routing engine.
79       *
80       * @param shardingRule sharding rule
81       * @param database database
82       * @param queryContext query context
83       * @param shardingConditions shardingConditions
84       * @param props ShardingSphere properties
85       * @param connectionContext connection context
86       * @return created instance
87       */
88      public static ShardingRouteEngine newInstance(final ShardingRule shardingRule, final ShardingSphereDatabase database, final QueryContext queryContext,
89                                                    final ShardingConditions shardingConditions, final ConfigurationProperties props, final ConnectionContext connectionContext) {
90          SQLStatementContext sqlStatementContext = queryContext.getSqlStatementContext();
91          SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
92          if (sqlStatement instanceof TCLStatement) {
93              return new ShardingDatabaseBroadcastRoutingEngine();
94          }
95          if (sqlStatement instanceof DDLStatement) {
96              if (sqlStatementContext instanceof CursorAvailable) {
97                  return getCursorRouteEngine(shardingRule, database, sqlStatementContext, queryContext.getHintValueContext(), shardingConditions, props);
98              }
99              return getDDLRoutingEngine(shardingRule, database, sqlStatementContext);
100         }
101         if (sqlStatement instanceof DALStatement) {
102             return getDALRoutingEngine(shardingRule, database, sqlStatementContext, connectionContext);
103         }
104         if (sqlStatement instanceof DCLStatement) {
105             return getDCLRoutingEngine(shardingRule, database, sqlStatementContext);
106         }
107         return getDQLRoutingEngine(shardingRule, database, sqlStatementContext, queryContext.getHintValueContext(), shardingConditions, props, connectionContext);
108     }
109     
110     private static ShardingRouteEngine getDDLRoutingEngine(final ShardingRule shardingRule, final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext) {
111         SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
112         boolean functionStatement = sqlStatement instanceof CreateFunctionStatement || sqlStatement instanceof AlterFunctionStatement || sqlStatement instanceof DropFunctionStatement;
113         boolean procedureStatement = sqlStatement instanceof CreateProcedureStatement || sqlStatement instanceof AlterProcedureStatement || sqlStatement instanceof DropProcedureStatement;
114         if (functionStatement || procedureStatement) {
115             return new ShardingDatabaseBroadcastRoutingEngine();
116         }
117         if (sqlStatement instanceof CreateTablespaceStatement || sqlStatement instanceof AlterTablespaceStatement || sqlStatement instanceof DropTablespaceStatement) {
118             return new ShardingInstanceBroadcastRoutingEngine(database.getResourceMetaData());
119         }
120         Collection<String> tableNames = sqlStatementContext instanceof TableAvailable
121                 ? ((TableAvailable) sqlStatementContext).getTablesContext().getSimpleTables().stream().map(each -> each.getTableName().getIdentifier().getValue()).collect(Collectors.toSet())
122                 : Collections.emptyList();
123         Collection<String> shardingRuleTableNames = shardingRule.getShardingRuleTableNames(tableNames);
124         if (!tableNames.isEmpty() && shardingRuleTableNames.isEmpty()) {
125             return new ShardingIgnoreRoutingEngine();
126         }
127         return new ShardingTableBroadcastRoutingEngine(database, sqlStatementContext, shardingRuleTableNames);
128     }
129     
130     private static ShardingRouteEngine getCursorRouteEngine(final ShardingRule shardingRule, final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext,
131                                                             final HintValueContext hintValueContext, final ShardingConditions shardingConditions, final ConfigurationProperties props) {
132         if (sqlStatementContext instanceof CloseStatementContext && ((CloseStatementContext) sqlStatementContext).getSqlStatement().isCloseAll()) {
133             return new ShardingDatabaseBroadcastRoutingEngine();
134         }
135         Collection<String> tableNames = sqlStatementContext instanceof TableAvailable ? ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames() : Collections.emptyList();
136         Collection<String> logicTableNames = shardingRule.getShardingLogicTableNames(tableNames);
137         boolean allBindingTables = logicTableNames.size() > 1 && shardingRule.isAllBindingTables(database, sqlStatementContext, logicTableNames);
138         if (isShardingStandardQuery(shardingRule, logicTableNames, allBindingTables)) {
139             return new ShardingStandardRoutingEngine(getLogicTableName(shardingConditions, logicTableNames), shardingConditions, sqlStatementContext, hintValueContext, props);
140         }
141         return new ShardingIgnoreRoutingEngine();
142     }
143     
144     private static ShardingRouteEngine getDALRoutingEngine(final ShardingRule shardingRule, final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext,
145                                                            final ConnectionContext connectionContext) {
146         SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
147         if (sqlStatement instanceof MySQLUseStatement) {
148             return new ShardingIgnoreRoutingEngine();
149         }
150         if (sqlStatement instanceof SetStatement || sqlStatement instanceof ResetParameterStatement || sqlStatement instanceof MySQLShowDatabasesStatement || sqlStatement instanceof LoadStatement) {
151             return new ShardingDatabaseBroadcastRoutingEngine();
152         }
153         if (isResourceGroupStatement(sqlStatement)) {
154             return new ShardingInstanceBroadcastRoutingEngine(database.getResourceMetaData());
155         }
156         Collection<String> tableNames = sqlStatementContext instanceof TableAvailable ? ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames() : Collections.emptyList();
157         Collection<String> shardingRuleTableNames = shardingRule.getShardingRuleTableNames(tableNames);
158         if (!tableNames.isEmpty() && shardingRuleTableNames.isEmpty()) {
159             return new ShardingIgnoreRoutingEngine();
160         }
161         if (sqlStatement instanceof MySQLOptimizeTableStatement) {
162             return new ShardingTableBroadcastRoutingEngine(database, sqlStatementContext, shardingRuleTableNames);
163         }
164         if (sqlStatement instanceof AnalyzeTableStatement) {
165             return shardingRuleTableNames.isEmpty() ? new ShardingDatabaseBroadcastRoutingEngine()
166                     : new ShardingTableBroadcastRoutingEngine(database, sqlStatementContext, shardingRuleTableNames);
167         }
168         if (!shardingRuleTableNames.isEmpty()) {
169             return new ShardingUnicastRoutingEngine(sqlStatementContext, shardingRuleTableNames, connectionContext);
170         }
171         return new ShardingDataSourceGroupBroadcastRoutingEngine();
172     }
173     
174     private static boolean isResourceGroupStatement(final SQLStatement sqlStatement) {
175         // TODO add dropResourceGroupStatement, alterResourceGroupStatement
176         return sqlStatement instanceof MySQLCreateResourceGroupStatement || sqlStatement instanceof MySQLSetResourceGroupStatement;
177     }
178     
179     private static ShardingRouteEngine getDCLRoutingEngine(final ShardingRule shardingRule, final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext) {
180         if (isDCLForSingleTable(sqlStatementContext)) {
181             Collection<String> tableNames = sqlStatementContext instanceof TableAvailable ? ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames() : Collections.emptyList();
182             Collection<String> shardingRuleTableNames = shardingRule.getShardingRuleTableNames(tableNames);
183             return shardingRuleTableNames.isEmpty() ? new ShardingIgnoreRoutingEngine() : new ShardingTableBroadcastRoutingEngine(database, sqlStatementContext, shardingRuleTableNames);
184         }
185         return new ShardingInstanceBroadcastRoutingEngine(database.getResourceMetaData());
186     }
187     
188     private static boolean isDCLForSingleTable(final SQLStatementContext sqlStatementContext) {
189         if (sqlStatementContext instanceof TableAvailable) {
190             TableAvailable tableSegmentsAvailable = (TableAvailable) sqlStatementContext;
191             return 1 == tableSegmentsAvailable.getTablesContext().getSimpleTables().size()
192                     && !"*".equals(tableSegmentsAvailable.getTablesContext().getSimpleTables().iterator().next().getTableName().getIdentifier().getValue());
193         }
194         return false;
195     }
196     
197     private static ShardingRouteEngine getDQLRoutingEngine(final ShardingRule shardingRule, final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext,
198                                                            final HintValueContext hintValueContext, final ShardingConditions shardingConditions, final ConfigurationProperties props,
199                                                            final ConnectionContext connectionContext) {
200         Collection<String> tableNames = sqlStatementContext instanceof TableAvailable ? ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames() : Collections.emptyList();
201         if (sqlStatementContext.getSqlStatement() instanceof DMLStatement && shardingConditions.isAlwaysFalse() || tableNames.isEmpty()) {
202             return new ShardingUnicastRoutingEngine(sqlStatementContext, tableNames, connectionContext);
203         }
204         Collection<String> shardingLogicTableNames = shardingRule.getShardingLogicTableNames(tableNames);
205         if (shardingLogicTableNames.isEmpty()) {
206             return new ShardingIgnoreRoutingEngine();
207         }
208         return getDQLRouteEngineForShardingTable(shardingRule, database, sqlStatementContext, hintValueContext, shardingConditions, props, shardingLogicTableNames);
209     }
210     
211     private static ShardingRouteEngine getDQLRouteEngineForShardingTable(final ShardingRule shardingRule, final ShardingSphereDatabase database,
212                                                                          final SQLStatementContext sqlStatementContext, final HintValueContext hintValueContext,
213                                                                          final ShardingConditions shardingConditions, final ConfigurationProperties props, final Collection<String> tableNames) {
214         boolean allBindingTables = tableNames.size() > 1 && shardingRule.isAllBindingTables(database, sqlStatementContext, tableNames);
215         if (isShardingStandardQuery(shardingRule, tableNames, allBindingTables)) {
216             return new ShardingStandardRoutingEngine(getLogicTableName(shardingConditions, tableNames), shardingConditions, sqlStatementContext, hintValueContext, props);
217         }
218         // TODO config for cartesian set
219         return new ShardingComplexRoutingEngine(shardingConditions, sqlStatementContext, hintValueContext, props, tableNames);
220     }
221     
222     private static String getLogicTableName(final ShardingConditions shardingConditions, final Collection<String> tableNames) {
223         if (shardingConditions.getConditions().isEmpty()) {
224             return tableNames.iterator().next();
225         }
226         ShardingCondition shardingCondition = shardingConditions.getConditions().iterator().next();
227         return shardingCondition.getValues().isEmpty() ? tableNames.iterator().next() : shardingCondition.getValues().iterator().next().getTableName();
228     }
229     
230     private static boolean isShardingStandardQuery(final ShardingRule shardingRule, final Collection<String> tableNames, final boolean allBindingTables) {
231         return 1 == tableNames.size() && shardingRule.isAllShardingTables(tableNames) || allBindingTables;
232     }
