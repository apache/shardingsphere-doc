21  import lombok.ToString;

23  import org.apache.shardingsphere.db.protocol.opengauss.packet.command.OpenGaussCommandPacketType;

25  import org.apache.shardingsphere.db.protocol.postgresql.constant.PostgreSQLValueFormat;
26  import org.apache.shardingsphere.db.protocol.postgresql.packet.command.query.extended.PostgreSQLColumnType;
27  import org.apache.shardingsphere.db.protocol.postgresql.packet.command.query.extended.bind.protocol.PostgreSQLBinaryProtocolValue;
28  import org.apache.shardingsphere.db.protocol.postgresql.packet.command.query.extended.bind.protocol.PostgreSQLBinaryProtocolValueFactory;
29  import org.apache.shardingsphere.db.protocol.postgresql.packet.command.query.extended.bind.protocol.PostgreSQLTextTimeUtils;
30  import org.apache.shardingsphere.db.protocol.postgresql.packet.command.query.extended.bind.protocol.PostgreSQLTextTimestampUtils;
31  import org.apache.shardingsphere.db.protocol.postgresql.payload.PostgreSQLPacketPayload;
32  
33  import java.math.BigDecimal;
34  import java.sql.Date;
35  import java.util.ArrayList;
36  import java.util.List;
37  
38  /**
39   * Batch bind packet for openGauss.
40   */
41  @Getter
42  @ToString
43  public final class OpenGaussComBatchBindPacket extends OpenGaussCommandPacket {
44      
45      private final PostgreSQLPacketPayload payload;
46      
47      private final int batchNum;
48      
49      private final String statementId;
50      
51      private final List<Integer> parameterFormats;
52      
53      private final List<PostgreSQLValueFormat> resultFormats;
54      
55      private final int eachGroupParametersCount;
56      
57      public OpenGaussComBatchBindPacket(final PostgreSQLPacketPayload payload) {
58          this.payload = payload;
59          payload.readInt4();
60          batchNum = payload.readInt4();
61          payload.readStringNul();
62          statementId = payload.readStringNul();
63          int parameterFormatCount = payload.readInt2();
64          parameterFormats = new ArrayList<>(parameterFormatCount);
65          for (int i = 0; i < parameterFormatCount; i++) {
66              parameterFormats.add(payload.readInt2());
67          }
68          int resultFormatsLength = payload.readInt2();
69          resultFormats = new ArrayList<>(resultFormatsLength);
70          for (int i = 0; i < resultFormatsLength; i++) {
71              resultFormats.add(PostgreSQLValueFormat.valueOf(payload.readInt2()));
72          }
73          eachGroupParametersCount = payload.readInt2();
74      }
75      
76      /**
77       * Read parameter sets from payload.
78       *
79       * @param parameterTypes types of parameters
80       * @return parameter sets
81       */
82      public List<List<Object>> readParameterSets(final List<PostgreSQLColumnType> parameterTypes) {
83          List<List<Object>> result = new ArrayList<>(batchNum);
84          for (int i = 0; i < batchNum; i++) {
85              result.add(readOneGroupOfParameters(parameterTypes));
86          }
87          payload.skipReserved(payload.getByteBuf().readableBytes());
88          return result;
89      }
90      
91      private List<Object> readOneGroupOfParameters(final List<PostgreSQLColumnType> parameterTypes) {
92          List<Object> result = new ArrayList<>(eachGroupParametersCount);
93          for (int paramIndex = 0; paramIndex < eachGroupParametersCount; paramIndex++) {
94              int paramValueLength = payload.readInt4();
95              if (-1 == paramValueLength) {
96                  result.add(null);
97                  continue;
98              }
99              Object parameterValue = isTextParameterValue(parameterFormats, paramIndex)
100                     ? getTextParameters(payload, paramValueLength, parameterTypes.get(paramIndex))
101                     : getBinaryParameters(payload, paramValueLength, parameterTypes.get(paramIndex));
102             result.add(parameterValue);
103         }
104         return result;
105     }
106     
107     private boolean isTextParameterValue(final List<Integer> paramFormats, final int paramIndex) {
108         return paramFormats.isEmpty() || 0 == paramFormats.get(paramIndex % paramFormats.size());
109     }
110     
111     private Object getTextParameters(final PostgreSQLPacketPayload payload, final int paramValueLength, final PostgreSQLColumnType paramType) {
112         String value = payload.getByteBuf().readCharSequence(paramValueLength, payload.getCharset()).toString();
113         return getTextParameters(value, paramType);
114     }
115     
116     private Object getTextParameters(final String textValue, final PostgreSQLColumnType columnType) {
117         switch (columnType) {
118             case POSTGRESQL_TYPE_BOOL:
119                 return Boolean.valueOf(textValue);
120             case POSTGRESQL_TYPE_INT2:
121             case POSTGRESQL_TYPE_INT4:
122                 return Integer.parseInt(textValue);
123             case POSTGRESQL_TYPE_INT8:
124                 return Long.parseLong(textValue);
125             case POSTGRESQL_TYPE_FLOAT4:
126                 return Float.parseFloat(textValue);
127             case POSTGRESQL_TYPE_FLOAT8:
128                 return Double.parseDouble(textValue);
129             case POSTGRESQL_TYPE_NUMERIC:
130                 try {
131                     return Integer.parseInt(textValue);
132                 } catch (final NumberFormatException ignored) {
133                 }
134                 try {
135                     return Long.parseLong(textValue);
136                 } catch (final NumberFormatException ignored) {
137                 }
138                 return new BigDecimal(textValue);
139             case POSTGRESQL_TYPE_DATE:
140                 return Date.valueOf(textValue);
141             case POSTGRESQL_TYPE_TIME:
142                 return PostgreSQLTextTimeUtils.parse(textValue);
143             case POSTGRESQL_TYPE_TIMESTAMP:
144             case POSTGRESQL_TYPE_TIMESTAMPTZ:
145                 return PostgreSQLTextTimestampUtils.parse(textValue);
146             default:
147                 return textValue;
148         }
149     }
150     
151     private Object getBinaryParameters(final PostgreSQLPacketPayload payload, final int paramValueLength, final PostgreSQLColumnType columnType) {
152         PostgreSQLBinaryProtocolValue binaryProtocolValue = PostgreSQLBinaryProtocolValueFactory.getBinaryProtocolValue(columnType);
153         return binaryProtocolValue.read(payload, paramValueLength);
154     }
155     
156     @Override
157     public void write(final PostgreSQLPacketPayload payload) {
158     }
159     
160     @Override
161     public OpenGaussIdentifierTag getIdentifier() {
162         return OpenGaussCommandPacketType.BATCH_BIND_COMMAND;
163     }
164 }
