52  /**
53   * Table assert.
54   */
55  @NoArgsConstructor(access = AccessLevel.PRIVATE)
56  public final class TableAssert {
57      
58      /**
59       * Assert actual table segments is correct with expected tables.
60       *
61       * @param assertContext assert context
62       * @param actual actual tables
63       * @param expected expected tables
64       * @throws UnsupportedOperationException unsupported operation exception
65       */
66      public static void assertIs(final SQLCaseAssertContext assertContext, final TableSegment actual, final ExpectedTable expected) {
67          if (actual instanceof JoinTableSegment) {
68              assertIs(assertContext, (JoinTableSegment) actual, expected.getJoinTable());
69          } else if (actual instanceof SimpleTableSegment) {
70              assertIs(assertContext, (SimpleTableSegment) actual, expected.getSimpleTable());
71          } else if (actual instanceof SubqueryTableSegment) {
72              assertIs(assertContext, (SubqueryTableSegment) actual, expected.getSubqueryTable());
73          } else if (actual instanceof XmlTableSegment) {
74              assertIs(assertContext, (XmlTableSegment) actual, expected.getXmlTable());
75          } else {
76              throw new UnsupportedOperationException(String.format("Unsupported table segment type `%s`.", actual.getClass()));
77          }
78      }
79      
80      /**
81       * Assert actual xml table segment is correct with expected xml table.
82       *
83       * @param assertContext assert context
84       * @param actual actual xml table
85       * @param expected expected xml table
86       */
87      private static void assertIs(final SQLCaseAssertContext assertContext, final XmlTableSegment actual, final ExpectedXmlTable expected) {
88          assertXmlTableFunction(assertContext, actual.getXmlTableFunction(), expected.getXmlTableFunction());
89          if (null != actual.getXmlTableFunctionAlias()) {
90              assertThat(assertContext.getText("Xml table function alias assertion error"), actual.getXmlTableFunctionAlias(), is(expected.getXmlTableFunctionAlias()));
91          }
92      }
93      
94      /**
95       * Assert actual table segment is correct with expected table.
96       *
97       * @param assertContext assert context
98       * @param actual actual table
99       * @param expected expected table
100      */
101     public static void assertIs(final SQLCaseAssertContext assertContext, final SimpleTableSegment actual, final ExpectedSimpleTable expected) {
102         IdentifierValueAssert.assertIs(assertContext, actual.getTableName().getIdentifier(), expected, "Table");
103         assertThat(assertContext.getText("Table alias assertion error: "), actual.getAliasName().orElse(null), is(expected.getAlias()));
104         if (null == expected.getOwner()) {
105             assertFalse(actual.getOwner().isPresent(), assertContext.getText("Actual owner should not exist."));
106         } else {
107             assertTrue(actual.getOwner().isPresent(), assertContext.getText("Actual owner should exist."));
108             OwnerAssert.assertIs(assertContext, actual.getOwner().get(), expected.getOwner());
109         }
110         SQLSegmentAssert.assertIs(assertContext, actual, expected);
111     }
112     
113     /**
114      * Assert subquery expression.
115      *
116      * @param assertContext assert context
117      * @param actual actual subquery segment
118      * @param expected expected subquery expression
119      */
120     public static void assertIs(final SQLCaseAssertContext assertContext, final SubqueryTableSegment actual, final ExpectedSubqueryTable expected) {
121         SelectStatementAssert.assertIs(assertContext, actual.getSubquery().getSelect(), expected.getSubquery().getSelectTestCases());
122         assertThat(assertContext.getText("Table alias assertion error: "), actual.getAliasName().orElse(null), is(expected.getAlias()));
123     }
124     
125     /**
126      * Assert join table.
127      *
128      * @param assertContext assert context
129      * @param actual actual JoinTableSegment
130      * @param expected expected ExpectedJoinTable
131      */
132     public static void assertIs(final SQLCaseAssertContext assertContext, final JoinTableSegment actual, final ExpectedJoinTable expected) {
133         assertIs(assertContext, actual.getLeft(), expected.getLeft());
134         assertIs(assertContext, actual.getRight(), expected.getRight());
135         ExpressionAssert.assertExpression(assertContext, actual.getCondition(), expected.getOnCondition());
136         assertJoinType(assertContext, actual.getJoinType(), expected.getJoinType());
137         assertThat(assertContext.getText("Natural should be the same."), actual.isNatural(), is(expected.isNatural()));
138         assertThat(assertContext.getText("Column size assertion error: "), actual.getUsing().size(), is(expected.getUsingColumns().size()));
139         int count = 0;
140         for (ExpectedColumn each : expected.getUsingColumns()) {
141             ColumnAssert.assertIs(assertContext, actual.getUsing().get(count), each);
142             count++;
143         }
144     }
145     
146     /**
147      * Assert actual table segments is correct with expected tables.
148      *
149      * @param assertContext assert context
150      * @param actual actual tables
151      * @param expected expected tables
152      */
153     public static void assertIs(final SQLCaseAssertContext assertContext, final Collection<SimpleTableSegment> actual, final List<ExpectedSimpleTable> expected) {
154         assertThat(assertContext.getText("Tables size assertion error: "), actual.size(), is(expected.size()));
155         int count = 0;
156         for (SimpleTableSegment each : actual) {
157             assertIs(assertContext, each, expected.get(count));
158             count++;
159         }
160     }
161     
162     /**
163      * Assert actual simple table segments with expected simple tables.
164      *
165      * @param assertContext assert context
166      * @param actualTables actual simple tables
167      * @param expectedTables expected simple tables
168      */
169     public static void assertIs(final SQLCaseAssertContext assertContext, final List<SimpleTableSegment> actualTables, final List<ExpectedSimpleTable> expectedTables) {
170         assertThat(assertContext.getText("tables size should be the same."),
171                 actualTables.size(), is(expectedTables.size()));
172         for (int i = 0; i < actualTables.size(); i++) {
173             assertIs(assertContext, actualTables.get(i), expectedTables.get(i));
174         }
175     }
176     
177     private static void assertJoinType(final SQLCaseAssertContext assertContext, final String actual, final String expected) {
178         if (Strings.isNullOrEmpty(expected)) {
179             assertTrue(Strings.isNullOrEmpty(actual), assertContext.getText("Actual join-type should not exist."));
180         } else {
181             assertThat(assertContext.getText("Actual join-type should exist."), actual, is(expected));
182         }
183     }
184     
185     /**
186      * Assert actual xml table function segment is correct with expected xml table function.
187      *
188      * @param assertContext assert context
189      * @param actual actual xml table function
190      * @param expected expected xml table function
191      */
192     private static void assertXmlTableFunction(final SQLCaseAssertContext assertContext, final XmlTableFunctionSegment actual, final ExpectedXmlTableFunction expected) {
193         assertThat(assertContext.getText("Function name assertion error"), actual.getFunctionName(), is(expected.getFunctionName()));
194         assertThat(assertContext.getText("Function text assert error"), actual.getText(), is(expected.getText()));
195     }
