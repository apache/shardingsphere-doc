18  package org.apache.shardingsphere.mask.rule;
19  
20  import lombok.Getter;
21  import org.apache.shardingsphere.infra.config.rule.RuleConfiguration;
22  import org.apache.shardingsphere.infra.rule.identifier.scope.DatabaseRule;
23  import org.apache.shardingsphere.infra.rule.identifier.type.ColumnContainedRule;
24  import org.apache.shardingsphere.infra.rule.identifier.type.TableContainedRule;
25  import org.apache.shardingsphere.infra.util.spi.type.typed.TypedSPILoader;
26  import org.apache.shardingsphere.mask.api.config.MaskRuleConfiguration;
27  import org.apache.shardingsphere.mask.spi.MaskAlgorithm;
28  
29  import java.util.Collection;
30  import java.util.LinkedHashMap;
31  import java.util.Map;
32  import java.util.Optional;
33  
34  /**
35   * Mask rule.
36   */
37  @SuppressWarnings("rawtypes")
38  public final class MaskRule implements DatabaseRule, TableContainedRule, ColumnContainedRule {
39      
40      @Getter
41      private final RuleConfiguration configuration;
42      
43      private final Map<String, MaskAlgorithm> maskAlgorithms = new LinkedHashMap<>();
44      
45      private final Map<String, MaskTable> tables = new LinkedHashMap<>();
46      
47      public MaskRule(final MaskRuleConfiguration ruleConfig) {
48          configuration = ruleConfig;
49          ruleConfig.getMaskAlgorithms().forEach((key, value) -> maskAlgorithms.put(key, TypedSPILoader.getService(MaskAlgorithm.class, value.getType(), value.getProps())));
50          ruleConfig.getTables().forEach(each -> tables.put(each.getName().toLowerCase(), new MaskTable(each)));
51      }
52      
53      /**
54       * Find mask algorithm.
55       *
56       * @param logicTable logic table name
57       * @param logicColumn logic column name
58       * @return maskAlgorithm
59       */
60      public Optional<MaskAlgorithm> findMaskAlgorithm(final String logicTable, final String logicColumn) {
61          String lowerCaseLogicTable = logicTable.toLowerCase();
62          return tables.containsKey(lowerCaseLogicTable) ? tables.get(lowerCaseLogicTable).findMaskAlgorithmName(logicColumn).map(maskAlgorithms::get) : Optional.empty();
63      }
64      
65      @Override
66      public Collection<String> getTables() {
67          return tables.keySet();
68      }
69      
70      @Override
71      public String getType() {
72          return MaskRule.class.getSimpleName();
73      }
74  }
