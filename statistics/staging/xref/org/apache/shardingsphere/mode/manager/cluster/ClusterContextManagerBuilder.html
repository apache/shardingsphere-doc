51  /**
52   * Cluster context manager builder.
53   */
54  public final class ClusterContextManagerBuilder implements ContextManagerBuilder {
55      
56      @Override
57      public ContextManager build(final ContextManagerBuilderParameter param, final EventBusContext eventBusContext) throws SQLException {
58          ClusterPersistRepository repository = getClusterPersistRepository((ClusterPersistRepositoryConfiguration) param.getModeConfiguration().getRepository());
59          RegistryCenter registryCenter = new RegistryCenter(eventBusContext, repository, param.getInstanceMetaData(), param.getDatabaseConfigs());
60          InstanceContext instanceContext = buildInstanceContext(registryCenter, param, eventBusContext);
61          if (registryCenter.getRepository() instanceof InstanceContextAware) {
62              ((InstanceContextAware) registryCenter.getRepository()).setInstanceContext(instanceContext);
63          }
64          MetaDataPersistService metaDataPersistService = new MetaDataPersistService(repository);
65          MetaDataContexts metaDataContexts = MetaDataContextsFactory.create(metaDataPersistService, param, instanceContext, registryCenter.getQualifiedDataSourceStatusService().loadStatus());
66          ContextManager result = new ContextManager(metaDataContexts, instanceContext);
67          setContextManagerAware(result);
68          createSubscribers(eventBusContext, repository);
69          registerOnline(registryCenter, param, result);
70          setClusterState(repository, result);
71          return result;
72      }
73      
74      private ClusterPersistRepository getClusterPersistRepository(final ClusterPersistRepositoryConfiguration config) {
75          ShardingSpherePreconditions.checkNotNull(config, MissingRequiredClusterRepositoryConfigurationException::new);
76          ClusterPersistRepository result = TypedSPILoader.getService(ClusterPersistRepository.class, config.getType(), config.getProps());
77          result.init(config);
78          return result;
79      }
80      
81      private InstanceContext buildInstanceContext(final RegistryCenter registryCenter, final ContextManagerBuilderParameter param, final EventBusContext eventBusContext) {
82          return new InstanceContext(new ComputeNodeInstance(param.getInstanceMetaData()), new ClusterWorkerIdGenerator(registryCenter, param.getInstanceMetaData()),
83                  param.getModeConfiguration(), new ClusterModeContextManager(), new GlobalLockContext(registryCenter.getGlobalLockPersistService()), eventBusContext);
84      }
85      
86      private void setContextManagerAware(final ContextManager contextManager) {
87          ((ContextManagerAware) contextManager.getInstanceContext().getModeContextManager()).setContextManager(contextManager);
88      }
89      
90      // TODO remove the method, only keep ZooKeeper's events, remove all decouple events
91      private void createSubscribers(final EventBusContext eventBusContext, final ClusterPersistRepository repository) {
92          eventBusContext.register(new ComputeNodeStatusSubscriber(repository));
93          eventBusContext.register(new ClusterStatusSubscriber(repository));
94          eventBusContext.register(new QualifiedDataSourceStatusSubscriber(repository));
95          eventBusContext.register(new ClusterProcessSubscriber(repository, eventBusContext));
96          eventBusContext.register(new ShardingSphereSchemaDataRegistrySubscriber(repository));
97      }
98      
99      private void registerOnline(final RegistryCenter registryCenter, final ContextManagerBuilderParameter param, final ContextManager contextManager) {
100         registryCenter.onlineInstance(contextManager.getInstanceContext().getInstance());
101         contextManager.getInstanceContext().getInstance().setLabels(param.getLabels());
102         contextManager.getInstanceContext().getAllClusterInstances().addAll(registryCenter.getComputeNodeStatusService().loadAllComputeNodeInstances());
103         new ClusterEventSubscriberRegistry(contextManager, registryCenter).register();
104     }
105     
106     private void setClusterState(final ClusterPersistRepository repository, final ContextManager contextManager) {
107         ClusterStatusService clusterStatusService = new ClusterStatusService(repository);
108         Optional<ClusterState> clusterState = clusterStatusService.load();
109         if (clusterState.isPresent()) {
110             contextManager.updateClusterState(clusterState.get());
111         } else {
112             clusterStatusService.persist(ClusterState.OK);
113         }
114     }
115     
116     @Override
117     public String getType() {
118         return "Cluster";
119     }
120 }
