47  /**
48   * ShardingSphere data source.
49   */
50  public final class ShardingSphereDataSource extends AbstractDataSourceAdapter implements AutoCloseable {
51      
52      private final String databaseName;
53      
54      private final ContextManager contextManager;
55      
56      public ShardingSphereDataSource(final String databaseName, final ModeConfiguration modeConfig) throws SQLException {
57          this.databaseName = databaseName;
58          contextManager = createContextManager(databaseName, modeConfig, new LinkedHashMap<>(), new LinkedList<>(), new Properties());
59          contextManagerInitializedCallback(databaseName, contextManager);
60      }
61      
62      public ShardingSphereDataSource(final String databaseName, final ModeConfiguration modeConfig, final Map<String, DataSource> dataSourceMap,
63                                      final Collection<RuleConfiguration> ruleConfigs, final Properties props) throws SQLException {
64          this.databaseName = databaseName;
65          contextManager = createContextManager(databaseName, modeConfig, dataSourceMap, ruleConfigs, null == props ? new Properties() : props);
66          contextManagerInitializedCallback(databaseName, contextManager);
67      }
68      
69      private ContextManager createContextManager(final String databaseName, final ModeConfiguration modeConfig, final Map<String, DataSource> dataSourceMap,
70                                                  final Collection<RuleConfiguration> ruleConfigs, final Properties props) throws SQLException {
71          InstanceMetaData instanceMetaData = TypedSPILoader.getService(InstanceMetaDataBuilder.class, "JDBC").build(-1);
72          Collection<RuleConfiguration> globalRuleConfigs = ruleConfigs.stream().filter(GlobalRuleConfiguration.class::isInstance).collect(Collectors.toList());
73          Collection<RuleConfiguration> databaseRuleConfigs = new LinkedList<>(ruleConfigs);
74          databaseRuleConfigs.removeAll(globalRuleConfigs);
75          ContextManagerBuilderParameter param = new ContextManagerBuilderParameter(modeConfig, Collections.singletonMap(databaseName,
76                  new DataSourceProvidedDatabaseConfiguration(dataSourceMap, databaseRuleConfigs)), Collections.emptyMap(), globalRuleConfigs, props, Collections.emptyList(), instanceMetaData, false);
77          return TypedSPILoader.getService(ContextManagerBuilder.class, null == modeConfig ? null : modeConfig.getType()).build(param);
78      }
79      
80      private void contextManagerInitializedCallback(final String databaseName, final ContextManager contextManager) {
81          for (ContextManagerLifecycleListener each : ShardingSphereServiceLoader.getServiceInstances(ContextManagerLifecycleListener.class)) {
82              try {
83                  each.onInitialized(databaseName, contextManager);
84                  // CHECKSTYLE:OFF
85              } catch (final RuntimeException ignored) {
86                  // CHECKSTYLE:ON
87              }
88          }
89      }
90      
91      @Override
92      public Connection getConnection() {
93          return DriverStateContext.getConnection(databaseName, contextManager);
94      }
95      
96      @Override
97      public Connection getConnection(final String username, final String password) {
98          return getConnection();
99      }
100     
101     /**
102      * Close data sources.
103      *
104      * @param dataSourceNames data source names to be closed
105      * @throws SQLException SQL exception
106      */
107     // TODO Replace public to private?
108     public void close(final Collection<String> dataSourceNames) throws SQLException {
109         Map<String, DataSource> dataSourceMap = contextManager.getDataSourceMap(databaseName);
110         for (String each : dataSourceNames) {
111             close(dataSourceMap.get(each));
112         }
113         contextManager.close();
114     }
115     
116     private void close(final DataSource dataSource) throws SQLException {
117         if (dataSource instanceof AutoCloseable) {
118             try {
119                 ((AutoCloseable) dataSource).close();
120                 // CHECKSTYLE:OFF
121             } catch (final Exception ex) {
122                 // CHECKSTYLE:ON
123                 throw new SQLException(ex);
124             }
125         }
126     }
127     
128     @Override
129     public void close() throws SQLException {
130         contextManagerDestroyedCallback(databaseName);
131         close(contextManager.getDataSourceMap(databaseName).keySet());
132     }
133     
134     private void contextManagerDestroyedCallback(final String databaseName) {
135         for (ContextManagerLifecycleListener each : ShardingSphereServiceLoader.getServiceInstances(ContextManagerLifecycleListener.class)) {
136             try {
137                 each.onDestroyed(databaseName, InstanceType.JDBC);
138                 // CHECKSTYLE:OFF
139             } catch (final RuntimeException ignored) {
140                 // CHECKSTYLE:ON
141             }
142         }
143     }
144     
145     @Override
146     public int getLoginTimeout() throws SQLException {
147         Map<String, DataSource> dataSourceMap = contextManager.getDataSourceMap(databaseName);
148         return dataSourceMap.isEmpty() ? 0 : dataSourceMap.values().iterator().next().getLoginTimeout();
149     }
150     
151     @Override
152     public void setLoginTimeout(final int seconds) throws SQLException {
153         for (DataSource each : contextManager.getDataSourceMap(databaseName).values()) {
154             each.setLoginTimeout(seconds);
155         }
156     }
157 }
