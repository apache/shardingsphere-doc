42  import static org.mockito.Mockito.mock;

44  class SQLRouteEngineTest {
45      
46      @Test
47      void assertRouteSuccess() {
48          QueryContext queryContext = new QueryContext(mock(CommonSQLStatementContext.class), "SELECT 1", Collections.emptyList());
49          ShardingSphereRuleMetaData ruleMetaData = new ShardingSphereRuleMetaData(Collections.singleton(new RouteRuleFixture()));
50          ShardingSphereDatabase database = new ShardingSphereDatabase("logic_schema",
51                  mock(DatabaseType.class), mock(ShardingSphereResourceMetaData.class, RETURNS_DEEP_STUBS), ruleMetaData, Collections.emptyMap());
52          SQLRouteEngine sqlRouteEngine = new SQLRouteEngine(Collections.singleton(new RouteRuleFixture()), new ConfigurationProperties(new Properties()));
53          RouteContext actual = sqlRouteEngine.route(new ConnectionContext(), queryContext, mock(ShardingSphereRuleMetaData.class), database);
54          assertThat(actual.getRouteUnits().size(), is(1));
55          RouteUnit routeUnit = actual.getRouteUnits().iterator().next();
56          assertThat(routeUnit.getDataSourceMapper().getLogicName(), is("ds"));
57          assertThat(routeUnit.getDataSourceMapper().getActualName(), is("ds_0"));
58          assertTrue(routeUnit.getTableMappers().isEmpty());
59      }
60      
61      @Test
62      void assertRouteFailure() {
63          QueryContext queryContext = new QueryContext(mock(CommonSQLStatementContext.class), "SELECT 1", Collections.emptyList());
64          ShardingSphereRuleMetaData ruleMetaData = new ShardingSphereRuleMetaData(Collections.singleton(new RouteRuleFixture()));
65          ShardingSphereDatabase database = new ShardingSphereDatabase("logic_schema",
66                  mock(DatabaseType.class), mock(ShardingSphereResourceMetaData.class, RETURNS_DEEP_STUBS), ruleMetaData, Collections.emptyMap());
67          assertThrows(UnsupportedOperationException.class, () -> new SQLRouteEngine(Collections.singleton(new RouteFailureRuleFixture()),
68                  new ConfigurationProperties(new Properties())).route(new ConnectionContext(), queryContext, mock(ShardingSphereRuleMetaData.class), database));
69      }
70  }
