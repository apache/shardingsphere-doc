18  package org.apache.shardingsphere.mode.path.rule.item;
19  
20  import org.apache.shardingsphere.mode.path.rule.root.RuleRootNodePath;
21  
22  import java.util.regex.Matcher;
23  import java.util.regex.Pattern;
24  
25  /**
26   * Unique rule item node path.
27   */
28  public final class UniqueRuleItemNodePath {
29      
30      private static final String VERSIONS = "/versions/\\d+$";
31      
32      private static final String ACTIVE_VERSION = "/active_version$";
33      
34      private final String parentNode;
35      
36      private final String type;
37      
38      private final Pattern pathPattern;
39      
40      private final Pattern activeVersionPathPattern;
41      
42      public UniqueRuleItemNodePath(final RuleRootNodePath ruleRootNodePath, final String type) {
43          parentNode = null;
44          this.type = type;
45          pathPattern = Pattern.compile(ruleRootNodePath.getNodePrefix() + type + VERSIONS);
46          activeVersionPathPattern = Pattern.compile(ruleRootNodePath.getNodePrefix() + type + ACTIVE_VERSION);
47      }
48      
49      public UniqueRuleItemNodePath(final RuleRootNodePath ruleRootNodePath, final String parentNode, final String type) {
50          this.parentNode = parentNode;
51          this.type = type;
52          pathPattern = Pattern.compile(ruleRootNodePath.getNodePrefix() + parentNode + "/" + type + VERSIONS);
53          activeVersionPathPattern = Pattern.compile(ruleRootNodePath.getNodePrefix() + parentNode + "/" + type + ACTIVE_VERSION);
54      }
55      
56      /**
57       * Get path.
58       *
59       * @return path
60       */
61      public String getPath() {
62          return null == parentNode ? String.join("/", type) : String.join("/", parentNode, type);
63      }
64      
65      /**
66       * Judge whether is validated rule item path.
67       *
68       * @param path path to be judged
69       * @return is validated rule item path or not
70       */
71      public boolean isValidatedPath(final String path) {
72          return pathPattern.matcher(path).find();
73      }
74      
75      /**
76       * Judge whether active version path.
77       * 
78       * @param path path to be judged
79       * @return is active version path or not
80       */
81      public boolean isActiveVersionPath(final String path) {
82          Matcher matcher = activeVersionPathPattern.matcher(path);
83          return matcher.find();
84      }
85  }
