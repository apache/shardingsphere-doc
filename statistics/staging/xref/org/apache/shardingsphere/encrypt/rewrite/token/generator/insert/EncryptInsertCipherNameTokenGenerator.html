44  /**
45   * Insert cipher column name token generator for encrypt.
46   */
47  @HighFrequencyInvocation
48  @RequiredArgsConstructor
49  @Setter
50  public final class EncryptInsertCipherNameTokenGenerator implements CollectionSQLTokenGenerator<InsertStatementContext> {
51      
52      private final EncryptRule encryptRule;
53      
54      @Override
55      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
56          if (!(sqlStatementContext instanceof InsertStatementContext)) {
57              return false;
58          }
59          Optional<InsertColumnsSegment> insertColumnsSegment = ((InsertStatementContext) sqlStatementContext).getSqlStatement().getInsertColumns();
60          return insertColumnsSegment.isPresent() && !insertColumnsSegment.get().getColumns().isEmpty();
61      }
62      
63      @Override
64      public Collection<SQLToken> generateSQLTokens(final InsertStatementContext insertStatementContext) {
65          Optional<InsertColumnsSegment> insertColumnsSegment = insertStatementContext.getSqlStatement().getInsertColumns();
66          Preconditions.checkState(insertColumnsSegment.isPresent());
67          Collection<ColumnSegment> insertColumns = insertColumnsSegment.get().getColumns();
68          if (null != insertStatementContext.getInsertSelectContext()) {
69              Collection<Projection> projections = insertStatementContext.getInsertSelectContext().getSelectStatementContext().getProjectionsContext().getExpandProjections();
70              ShardingSpherePreconditions.checkState(insertColumns.size() == projections.size(), () -> new UnsupportedSQLOperationException("Column count doesn't match value count."));
71              ShardingSpherePreconditions.checkState(InsertSelectColumnsEncryptorComparator.isSame(insertColumns, projections, encryptRule),
72                      () -> new UnsupportedSQLOperationException("Can not use different encryptor in insert select columns"));
73          }
74          EncryptTable encryptTable = encryptRule.getEncryptTable(insertStatementContext.getSqlStatement().getTable().getTableName().getIdentifier().getValue());
75          Collection<SQLToken> result = new LinkedList<>();
76          for (ColumnSegment each : insertColumns) {
77              String columnName = each.getIdentifier().getValue();
78              if (encryptTable.isEncryptColumn(columnName)) {
79                  Collection<Projection> projections =
80                          Collections.singleton(new ColumnProjection(null, encryptTable.getEncryptColumn(columnName).getCipher().getName(), null, insertStatementContext.getDatabaseType()));
81                  result.add(new SubstitutableColumnNameToken(each.getStartIndex(), each.getStopIndex(), projections, insertStatementContext.getDatabaseType()));
82              }
83          }
84          return result;
85      }
86  }
