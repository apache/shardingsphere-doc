57  @ExtendWith(AutoMockExtension.class)
58  @StaticMockSettings(ShardingSphereDatabase.class)
59  @MockitoSettings(strictness = Strictness.LENIENT)
60  class ShardingSphereMetaDataTest {
61      
62      @Test
63      void assertAddDatabase() {
64          GlobalRule globalRule = mock(GlobalRule.class);
65          ShardingSphereDatabase database = mockDatabase(mock(ResourceMetaData.class, RETURNS_DEEP_STUBS), new MockedDataSource(), globalRule);
66          DatabaseType databaseType = mock(DatabaseType.class);
67          ConfigurationProperties configProps = new ConfigurationProperties(new Properties());
68          when(ShardingSphereDatabase.create("foo_db", databaseType, configProps)).thenReturn(database);
69          Map<String, ShardingSphereDatabase> databases = new HashMap<>(Collections.singletonMap("foo_db", database));
70          ShardingSphereMetaData metaData = new ShardingSphereMetaData(databases, mock(ResourceMetaData.class), new RuleMetaData(Collections.singleton(globalRule)), configProps);
71          metaData.addDatabase("foo_db", databaseType, configProps);
72          assertThat(metaData.getDatabases(), is(databases));
73          verify(globalRule).refresh(databases, GlobalRuleChangedType.DATABASE_CHANGED);
74      }
75      
76      @Test
77      void assertDropDatabase() {
78          ResourceMetaData resourceMetaData = mock(ResourceMetaData.class, RETURNS_DEEP_STUBS);
79          GlobalRule globalRule = mock(GlobalRule.class);
80          MockedDataSource dataSource = new MockedDataSource();
81          ShardingSphereRule databaseRule = mock(ShardingSphereRule.class);
82          when(databaseRule.getAttributes()).thenReturn(new RuleAttributes());
83          ShardingSphereMetaData metaData = new ShardingSphereMetaData(new HashMap<>(Collections.singletonMap("foo_db", mockDatabase(resourceMetaData, dataSource, databaseRule))),
84                  mock(ResourceMetaData.class), new RuleMetaData(Collections.singleton(globalRule)), new ConfigurationProperties(new Properties()));
85          metaData.dropDatabase("foo_db");
86          assertTrue(metaData.getDatabases().isEmpty());
87          Awaitility.await().pollDelay(10L, TimeUnit.MILLISECONDS).until(dataSource::isClosed);
88          assertTrue(dataSource.isClosed());
89          verify(globalRule).refresh(metaData.getDatabases(), GlobalRuleChangedType.DATABASE_CHANGED);
90      }
91      
92      @Test
93      void assertContainsDatabase() {
94          ShardingSphereRule globalRule = mock(ShardingSphereRule.class);
95          ShardingSphereDatabase database = mockDatabase(mock(ResourceMetaData.class, RETURNS_DEEP_STUBS), new MockedDataSource(), globalRule);
96          Map<String, ShardingSphereDatabase> databases = new HashMap<>(Collections.singletonMap("foo_db", database));
97          ConfigurationProperties configProps = new ConfigurationProperties(new Properties());
98          ShardingSphereMetaData metaData = new ShardingSphereMetaData(databases, mock(ResourceMetaData.class), new RuleMetaData(Collections.singleton(globalRule)), configProps);
99          assertTrue(metaData.containsDatabase("foo_db"));
100     }
101     
102     @Test
103     void assertGetDatabase() {
104         ShardingSphereRule globalRule = mock(ShardingSphereRule.class);
105         ShardingSphereDatabase database = mockDatabase(mock(ResourceMetaData.class, RETURNS_DEEP_STUBS), new MockedDataSource(), globalRule);
106         Map<String, ShardingSphereDatabase> databases = new HashMap<>(Collections.singletonMap("foo_db", database));
107         ConfigurationProperties configProps = new ConfigurationProperties(new Properties());
108         ShardingSphereMetaData metaData = new ShardingSphereMetaData(databases, mock(ResourceMetaData.class), new RuleMetaData(Collections.singleton(globalRule)), configProps);
109         assertThat(metaData.getDatabase("foo_db"), is(database));
110     }
111     
112     private ShardingSphereDatabase mockDatabase(final ResourceMetaData resourceMetaData, final DataSource dataSource, final ShardingSphereRule rule) {
113         ShardingSphereDatabase result = mock(ShardingSphereDatabase.class);
114         when(result.getName()).thenReturn("foo_db");
115         when(result.getResourceMetaData()).thenReturn(resourceMetaData);
116         DataSourcePoolProperties dataSourcePoolProps = mock(DataSourcePoolProperties.class, RETURNS_DEEP_STUBS);
117         when(dataSourcePoolProps.getConnectionPropertySynonyms().getStandardProperties()).thenReturn(Maps.of("url", "jdbc:mock://127.0.0.1/foo_ds", "username", "test"));
118         StorageUnit storageUnit = new StorageUnit(mock(StorageNode.class), dataSourcePoolProps, dataSource);
119         when(result.getResourceMetaData().getStorageUnits()).thenReturn(Collections.singletonMap("foo_db", storageUnit));
120         when(result.getRuleMetaData()).thenReturn(new RuleMetaData(Collections.singleton(rule)));
121         return result;
122     }
