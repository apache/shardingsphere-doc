93      @Test
94      void assertPreValidateWhenInsertMultiTables() {
95          SQLStatementContext<InsertStatement> sqlStatementContext = createInsertStatementContext(Collections.singletonList(1), createInsertStatement());
96          Collection<String> tableNames = sqlStatementContext.getTablesContext().getTableNames();
97          when(shardingRule.isAllShardingTables(tableNames)).thenReturn(false);
98          when(shardingRule.tableRuleExists(tableNames)).thenReturn(true);
99          assertThrows(DMLWithMultipleShardingTablesException.class, () -> new ShardingInsertStatementValidator(shardingConditions).preValidate(shardingRule,
100                 sqlStatementContext, Collections.emptyList(), mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS), mock(ConfigurationProperties.class)));
101     }

103     private InsertStatementContext createInsertStatementContext(final List<Object> params, final InsertStatement insertStatement) {
104         ShardingSphereDatabase database = mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS);
105         when(database.getSchema(DefaultDatabase.LOGIC_NAME)).thenReturn(mock(ShardingSphereSchema.class));
106         ShardingSphereMetaData metaData =
107                 new ShardingSphereMetaData(Collections.singletonMap(DefaultDatabase.LOGIC_NAME, database), mock(ShardingSphereRuleMetaData.class), mock(ConfigurationProperties.class));
108         return new InsertStatementContext(metaData, params, insertStatement, DefaultDatabase.LOGIC_NAME);
109     }

220     private void mockShardingRuleForUpdateShardingColumn() {
221         TableRule tableRule = mock(TableRule.class);
222         when(tableRule.getActualDataSourceNames()).thenReturn(Arrays.asList("ds_0", "ds_1"));
223         when(tableRule.getActualTableNames("ds_1")).thenReturn(Collections.singletonList("user"));
224         when(shardingRule.findShardingColumn("id", "user")).thenReturn(Optional.of("id"));
225         when(shardingRule.getTableRule("user")).thenReturn(tableRule);
226         StandardShardingStrategyConfiguration databaseStrategyConfig = mock(StandardShardingStrategyConfiguration.class);
227         when(databaseStrategyConfig.getShardingColumn()).thenReturn("id");
228         when(databaseStrategyConfig.getShardingAlgorithmName()).thenReturn("database_inline");
229         when(shardingRule.getDatabaseShardingStrategyConfiguration(tableRule)).thenReturn(databaseStrategyConfig);
230         when(shardingRule.getShardingAlgorithms()).thenReturn(Collections.singletonMap("database_inline",
231                 TypedSPILoader.getService(ShardingAlgorithm.class, "INLINE", PropertiesBuilder.build(new Property("algorithm-expression", "ds_${id % 2}")))));
232     }

270     private InsertStatement createInsertStatement() {
271         MySQLInsertStatement result = new MySQLInsertStatement();
272         result.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("user"))));
273         ColumnSegment columnSegment = new ColumnSegment(0, 0, new IdentifierValue("id"));
274         List<ColumnSegment> columnSegments = new LinkedList<>();
275         columnSegments.add(columnSegment);
276         AssignmentSegment assignmentSegment = new ColumnAssignmentSegment(0, 0, columnSegments, new ParameterMarkerExpressionSegment(0, 0, 0));
277         result.setOnDuplicateKeyColumns(new OnDuplicateKeyColumnsSegment(0, 0, Collections.singletonList(assignmentSegment)));
278         Collection<ColumnSegment> columns = new LinkedList<>();
279         columns.add(columnSegment);
280         result.setInsertColumns(new InsertColumnsSegment(0, 0, columns));
281         return result;
282     }
283     
284     private InsertStatement createInsertSelectStatement() {
285         InsertStatement result = createInsertStatement();
286         SelectStatement selectStatement = new MySQLSelectStatement();
287         selectStatement.setProjections(new ProjectionsSegment(0, 0));
288         result.setInsertSelect(new SubquerySegment(0, 0, selectStatement));
289         return result;
290     }
