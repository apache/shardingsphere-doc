47  /**
48   * Statistics collect job.
49   */
50  @RequiredArgsConstructor
51  @Slf4j
52  public final class StatisticsCollectJob implements SimpleJob {
53      
54      private final ContextManager contextManager;
55      
56      @Override
57      public void execute(final ShardingContext shardingContext) {
58          try {
59              if (contextManager.getMetaDataContexts().getMetaData().getTemporaryProps().getValue(TemporaryConfigurationPropertyKey.PROXY_META_DATA_COLLECTOR_ENABLED)) {
60                  ShardingSphereStatistics statistics = contextManager.getMetaDataContexts().getStatistics();
61                  ShardingSphereMetaData metaData = contextManager.getMetaDataContexts().getMetaData();
62                  ShardingSphereStatistics changedStatistics = new ShardingSphereStatistics();
63                  statistics.getDatabaseData().forEach((key, value) -> {
64                      if (metaData.containsDatabase(key)) {
65                          collectForDatabase(key, value, metaData.getDatabases(), changedStatistics);
66                      }
67                  });
68                  compareUpdateAndSendEvent(statistics, changedStatistics, metaData.getDatabases());
69              }
70              // CHECKSTYLE:OFF
71          } catch (final Exception ex) {
72              // CHECKSTYLE:ON
73              log.error("Collect data error", ex);
74          }
75      }
76      
77      private void collectForDatabase(final String databaseName, final ShardingSphereDatabaseData databaseData,
78                                      final Map<String, ShardingSphereDatabase> databases, final ShardingSphereStatistics statistics) {
79          databaseData.getSchemaData().forEach((key, value) -> {
80              if (databases.get(databaseName.toLowerCase()).containsSchema(key)) {
81                  collectForSchema(databaseName, key, value, databases, statistics);
82              }
83          });
84      }
85      
86      private void collectForSchema(final String databaseName, final String schemaName, final ShardingSphereSchemaData schemaData,
87                                    final Map<String, ShardingSphereDatabase> databases, final ShardingSphereStatistics statistics) {
88          schemaData.getTableData().forEach((key, value) -> {
89              if (databases.get(databaseName.toLowerCase()).getSchema(schemaName).containsTable(key)) {
90                  collectForTable(databaseName, schemaName, databases.get(databaseName).getSchema(schemaName).getTable(key), databases, statistics);
91              }
92          });
93      }
94      
95      private void collectForTable(final String databaseName, final String schemaName, final ShardingSphereTable table,
96                                   final Map<String, ShardingSphereDatabase> databases, final ShardingSphereStatistics statistics) {
97          Optional<ShardingSphereStatisticsCollector> dataCollector = TypedSPILoader.findService(ShardingSphereStatisticsCollector.class, table.getName());
98          if (!dataCollector.isPresent()) {
99              return;
100         }
101         Optional<ShardingSphereTableData> tableData = Optional.empty();
102         try {
103             tableData = dataCollector.get().collect(databaseName, table, databases, contextManager.getMetaDataContexts().getMetaData().getGlobalRuleMetaData());
104             // CHECKSTYLE:OFF
105         } catch (final Exception ex) {
106             // CHECKSTYLE:ON
107             log.error(String.format("Collect %s.%s.%s data failed", databaseName, schemaName, table.getName()), ex);
108         }
109         tableData.ifPresent(optional -> statistics.getDatabaseData().computeIfAbsent(databaseName.toLowerCase(), key -> new ShardingSphereDatabaseData())
110                 .getSchemaData().computeIfAbsent(schemaName, key -> new ShardingSphereSchemaData()).getTableData().put(table.getName().toLowerCase(), optional));
111     }
112     
113     private void compareUpdateAndSendEvent(final ShardingSphereStatistics statistics, final ShardingSphereStatistics changedStatistics,
114                                            final Map<String, ShardingSphereDatabase> databases) {
115         changedStatistics.getDatabaseData().forEach((key, value) -> compareUpdateAndSendEventForDatabase(key, statistics.getDatabaseData().get(key), value, statistics,
116                 databases.get(key.toLowerCase())));
117     }
118     
119     private void compareUpdateAndSendEventForDatabase(final String databaseName, final ShardingSphereDatabaseData databaseData, final ShardingSphereDatabaseData changedDatabaseData,
120                                                       final ShardingSphereStatistics statistics, final ShardingSphereDatabase database) {
121         changedDatabaseData.getSchemaData().forEach((key, value) -> compareUpdateAndSendEventForSchema(databaseName, key, databaseData.getSchemaData().get(key), value, statistics,
122                 database.getSchema(key)));
123     }
124     
125     private void compareUpdateAndSendEventForSchema(final String databaseName, final String schemaName, final ShardingSphereSchemaData schemaData,
126                                                     final ShardingSphereSchemaData changedSchemaData, final ShardingSphereStatistics statistics, final ShardingSphereSchema schema) {
127         changedSchemaData.getTableData().forEach((key, value) -> compareUpdateAndSendEventForTable(databaseName, schemaName, schemaData.getTableData().get(key), value, statistics,
128                 schema.getTable(key)));
129     }
130     
131     private void compareUpdateAndSendEventForTable(final String databaseName, final String schemaName, final ShardingSphereTableData tableData,
132                                                    final ShardingSphereTableData changedTableData, final ShardingSphereStatistics statistics, final ShardingSphereTable table) {
133         if (tableData.equals(changedTableData)) {
134             return;
135         }
136         statistics.getDatabaseData().get(databaseName).getSchemaData().get(schemaName).getTableData().put(changedTableData.getName().toLowerCase(), changedTableData);
137         ShardingSphereSchemaDataAlteredEvent event = getShardingSphereSchemaDataAlteredEvent(databaseName, schemaName, tableData, changedTableData, table);
138         contextManager.getComputeNodeInstanceContext().getEventBusContext().post(event);
139     }
140     
141     private ShardingSphereSchemaDataAlteredEvent getShardingSphereSchemaDataAlteredEvent(final String databaseName, final String schemaName, final ShardingSphereTableData tableData,
142                                                                                          final ShardingSphereTableData changedTableData, final ShardingSphereTable table) {
143         ShardingSphereSchemaDataAlteredEvent result = new ShardingSphereSchemaDataAlteredEvent(databaseName, schemaName, tableData.getName());
144         Map<String, ShardingSphereRowData> tableDataMap = tableData.getRows().stream().collect(Collectors.toMap(ShardingSphereRowData::getUniqueKey, Function.identity()));
145         Map<String, ShardingSphereRowData> changedTableDataMap = changedTableData.getRows().stream().collect(Collectors.toMap(ShardingSphereRowData::getUniqueKey, Function.identity()));
146         YamlShardingSphereRowDataSwapper swapper = new YamlShardingSphereRowDataSwapper(new ArrayList<>(table.getColumnValues()));
147         for (Entry<String, ShardingSphereRowData> entry : changedTableDataMap.entrySet()) {
148             if (!tableDataMap.containsKey(entry.getKey())) {
149                 result.getAddedRows().add(swapper.swapToYamlConfiguration(entry.getValue()));
150             } else if (!tableDataMap.get(entry.getKey()).equals(entry.getValue())) {
151                 result.getUpdatedRows().add(swapper.swapToYamlConfiguration(entry.getValue()));
152             }
153         }
154         for (Entry<String, ShardingSphereRowData> entry : tableDataMap.entrySet()) {
155             if (!changedTableDataMap.containsKey(entry.getKey())) {
156                 result.getDeletedRows().add(swapper.swapToYamlConfiguration(entry.getValue()));
157             }
158         }
159         return result;
160     }
