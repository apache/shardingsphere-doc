40  import org.mockito.quality.Strictness;

42  import static org.hamcrest.CoreMatchers.is;
43  import static org.hamcrest.MatcherAssert.assertThat;
44  import static org.junit.jupiter.api.Assertions.assertThrows;
45  import static org.junit.jupiter.api.Assertions.assertNotNull;
46  import static org.mockito.Mockito.verify;
47  import static org.mockito.Mockito.when;
48  
49  @ExtendWith(MockitoExtension.class)
50  @MockitoSettings(strictness = Strictness.LENIENT)
51  class MySQLNegotiateHandlerTest {
52      
53      private static final String USER_NAME = "username";
54      
55      private static final String PASSWORD = "password";
56      
57      @Mock
58      private Promise<Object> authResultCallback;
59      
60      @Mock
61      private ChannelHandlerContext channelHandlerContext;
62      
63      @Mock
64      private Channel channel;
65      
66      @Mock
67      private ChannelPipeline pipeline;
68      
69      private MySQLNegotiateHandler mysqlNegotiateHandler;
70      
71      @BeforeEach
72      void setUp() {
73          when(channelHandlerContext.channel()).thenReturn(channel);
74          when(channel.pipeline()).thenReturn(pipeline);
75          mysqlNegotiateHandler = new MySQLNegotiateHandler(USER_NAME, PASSWORD, authResultCallback);
76      }
77      
78      @Test
79      void assertChannelReadHandshakeInitPacket() throws ReflectiveOperationException {
80          MySQLHandshakePacket handshakePacket = new MySQLHandshakePacket(0, new MySQLAuthenticationPluginData(new byte[8], new byte[12]));
81          handshakePacket.setAuthPluginName(MySQLAuthenticationMethod.NATIVE);
82          mysqlNegotiateHandler.channelRead(channelHandlerContext, handshakePacket);
83          verify(channel).writeAndFlush(ArgumentMatchers.any(MySQLHandshakeResponse41Packet.class));
84          ServerInfo serverInfo = (ServerInfo) Plugins.getMemberAccessor().get(MySQLNegotiateHandler.class.getDeclaredField("serverInfo"), mysqlNegotiateHandler);
85          assertNotNull(serverInfo);
86          assertThat(serverInfo.getServerVersion().getMajor(), is(5));
87          assertThat(serverInfo.getServerVersion().getMinor(), is(7));
88          assertThat(serverInfo.getServerVersion().getSeries(), is(22));
89      }
90      
91      @Test
92      void assertChannelReadOkPacket() throws ReflectiveOperationException {
93          MySQLOKPacket okPacket = new MySQLOKPacket(0);
94          ServerInfo serverInfo = new ServerInfo();
95          Plugins.getMemberAccessor().set(MySQLNegotiateHandler.class.getDeclaredField("serverInfo"), mysqlNegotiateHandler, serverInfo);
96          mysqlNegotiateHandler.channelRead(channelHandlerContext, okPacket);
97          verify(pipeline).remove(mysqlNegotiateHandler);
98          verify(authResultCallback).setSuccess(serverInfo);
99      }
100     
101     @Test
102     void assertChannelReadErrorPacket() {
103         MySQLErrPacket errorPacket = new MySQLErrPacket(MySQLVendorError.ER_NO_DB_ERROR);
104         assertThrows(RuntimeException.class, () -> mysqlNegotiateHandler.channelRead(channelHandlerContext, errorPacket));
105     }
106 }
