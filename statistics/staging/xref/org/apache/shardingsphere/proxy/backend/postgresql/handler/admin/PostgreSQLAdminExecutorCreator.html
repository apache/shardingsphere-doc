45  /**
46   * Database admin executor creator for PostgreSQL.
47   */
48  public final class PostgreSQLAdminExecutorCreator implements DatabaseAdminExecutorCreator {
49      
50      private static final String PG_CLASS = "pg_class";
51      
52      private static final String PG_NAMESPACE = "pg_namespace";
53      
54      private static final Collection<String> KERNEL_SUPPORTED_TABLES = Arrays.asList(PG_NAMESPACE, PG_CLASS);
55      
56      @Override
57      public Optional<DatabaseAdminExecutor> create(final SQLStatementContext sqlStatementContext) {
58          SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
59          if (sqlStatement instanceof ShowStatement) {
60              return Optional.of(new PostgreSQLShowVariableExecutor((ShowStatement) sqlStatement));
61          }
62          return Optional.empty();
63      }
64      
65      @Override
66      public Optional<DatabaseAdminExecutor> create(final SQLStatementContext sqlStatementContext, final String sql, final String databaseName, final List<Object> parameters) {
67          SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
68          if (sqlStatement instanceof SelectStatement) {
69              Collection<String> selectedTableNames = getSelectedTableNames((SelectStatement) sqlStatement);
70              if (!selectedTableNames.isEmpty() && KERNEL_SUPPORTED_TABLES.containsAll(selectedTableNames)) {
71                  return Optional.empty();
72              }
73              if (!selectedTableNames.isEmpty() && (SystemSchemaManager.isSystemTable("postgresql", "information_schema", selectedTableNames)
74                      || SystemSchemaManager.isSystemTable("postgresql", "pg_catalog", selectedTableNames))) {
75                  return Optional.of(new DefaultDatabaseMetaDataExecutor(sql, parameters));
76              }
77          }
78          if (sqlStatement instanceof SetStatement) {
79              return Optional.of(new PostgreSQLSetVariableAdminExecutor((SetStatement) sqlStatement));
80          }
81          if (sqlStatement instanceof ResetParameterStatement) {
82              return Optional.of(new PostgreSQLResetVariableAdminExecutor((ResetParameterStatement) sqlStatement));
83          }
84          return Optional.empty();
85      }
86      
87      private Collection<String> getSelectedTableNames(final SelectStatement sqlStatement) {
88          TableExtractor extractor = new TableExtractor();
89          extractor.extractTablesFromSelect(sqlStatement);
90          List<TableSegment> extracted = new LinkedList<>(extractor.getTableContext());
91          for (TableSegment each : extractor.getTableContext()) {
92              if (each instanceof SubqueryTableSegment) {
93                  TableExtractor subExtractor = new TableExtractor();
94                  subExtractor.extractTablesFromSelect(((SubqueryTableSegment) each).getSubquery().getSelect());
95                  extracted.addAll(subExtractor.getTableContext());
96              }
97          }
98          List<String> result = new ArrayList<>(extracted.size());
99          for (TableSegment each : extracted) {
100             if (each instanceof SimpleTableSegment) {
101                 result.add(((SimpleTableSegment) each).getTableName().getIdentifier().getValue());
102             }
103         }
104         return result;
105     }
106     
107     @Override
108     public String getDatabaseType() {
109         return "PostgreSQL";
110     }
111 }
