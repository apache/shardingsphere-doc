18  package org.apache.shardingsphere.infra.database.core.type;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.infra.database.spi.DatabaseType;
23  import org.apache.shardingsphere.infra.util.exception.ShardingSpherePreconditions;
24  import org.apache.shardingsphere.infra.util.spi.type.typed.TypedSPILoader;
25  
26  import javax.sql.DataSource;
27  import java.sql.Connection;
28  import java.sql.SQLException;
29  import java.util.Arrays;
30  import java.util.Collection;
31  import java.util.Collections;
32  import java.util.HashSet;
33  import java.util.Map;
34  import java.util.Map.Entry;
35  
36  /**
37   * Database type checker.
38   */
39  @NoArgsConstructor(access = AccessLevel.PRIVATE)
40  public final class DatabaseTypeChecker {
41      
42      private static final Collection<String> MOCKED_URL_PREFIXES = new HashSet<>(Arrays.asList("jdbc:fixture", "jdbc:mock", "mock:jdbc"));
43      
44      private static final Collection<String> UNSUPPORTED_URL_PREFIXES = Collections.singletonList("jdbc:mysql:aws");
45      
46      private static final Collection<DatabaseType> SUPPORTED_STORAGE_TYPES = new HashSet<>(8, 1F);
47      
48      private static volatile boolean isChecked;
49      
50      static {
51          Arrays.asList("MySQL", "PostgreSQL", "openGauss", "Oracle", "SQLServer", "H2", "MariaDB")
52                  .forEach(each -> TypedSPILoader.findService(DatabaseType.class, each).ifPresent(SUPPORTED_STORAGE_TYPES::add));
53      }
54      
55      /**
56       * Check supported storage types.
57       *
58       * @param dataSources  data sources
59       * @param databaseName database name
60       * @param storageTypes storage types
61       * @throws SQLException SQL exception
62       */
63      public static void checkSupportedStorageTypes(final Map<String, DataSource> dataSources, final String databaseName, final Map<String, DatabaseType> storageTypes) throws SQLException {
64          if (isChecked || dataSources.isEmpty()) {
65              return;
66          }
67          for (Entry<String, DataSource> entry : dataSources.entrySet()) {
68              try (Connection connection = entry.getValue().getConnection()) {
69                  String url = connection.getMetaData().getURL();
70                  if (MOCKED_URL_PREFIXES.stream().anyMatch(url::startsWith)) {
71                      return;
72                  }
73                  ShardingSpherePreconditions.checkState(UNSUPPORTED_URL_PREFIXES.stream()
74                          .noneMatch(url::startsWith), () -> new UnsupportedStorageTypeException(databaseName, entry.getKey()));
75              }
76          }
77          storageTypes.forEach((key, value) -> ShardingSpherePreconditions.checkState(SUPPORTED_STORAGE_TYPES.stream()
78                  .anyMatch(each -> each.getClass().equals(value.getClass())), () -> new UnsupportedStorageTypeException(databaseName, key)));
79          isChecked = true;
80      }
81      
82      /**
83       * Check supported storage type.
84       *
85       * @param url URL
86       * @param dataSourceName data source name
87       */
88      public static void checkSupportedStorageType(final String url, final String dataSourceName) {
89          if (MOCKED_URL_PREFIXES.stream().anyMatch(url::startsWith)) {
90              return;
91          }
92          ShardingSpherePreconditions.checkState(UNSUPPORTED_URL_PREFIXES.stream()
93                  .noneMatch(url::startsWith), () -> new UnsupportedStorageTypeException(dataSourceName));
94          ShardingSpherePreconditions.checkState(SUPPORTED_STORAGE_TYPES.stream()
95                  .flatMap(storageType -> storageType.getJdbcUrlPrefixes().stream())
96                  .anyMatch(url::startsWith), () -> new UnsupportedStorageTypeException(dataSourceName));
97      }
98  }
