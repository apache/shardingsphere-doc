1480     public ASTNode visitCursorForLoopStatement(final CursorForLoopStatementContext ctx) {
1481         SQLStatement relatedCursorStatement;
1482         String cursorName = null;
1483         if (null != ctx.select()) {
1484             relatedCursorStatement = visitSelect0(ctx.select());
1485         } else {
1486             cursorName = null == ctx.cursor().variableName().identifier()
1487                     ? new StringLiteralValue(ctx.cursor().getText()).getValue()
1488                     : new IdentifierValue(ctx.cursor().getText()).getValue();
1489             relatedCursorStatement = getCursorStatements().get(cursorName);
1490         }
1491         increaseCursorForLoopLevel();
1492         for (StatementContext each : ctx.statement()) {
1493             visit(each);
1494         }
1495         Set<SQLStatement> sqlStatements = getTempCursorForLoopStatements().remove(getCursorForLoopLevel());
1496         CursorForLoopStatementSegment cursorForLoopStatementSegment = new CursorForLoopStatementSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(),
1497                 new IdentifierValue(ctx.record().getText()).getValue(), cursorName, relatedCursorStatement, null == sqlStatements ? Collections.emptyList() : sqlStatements);
1498         getCursorForLoopStatementSegments().add(cursorForLoopStatementSegment);
1499         decreaseCursorForLoopLevel();
1500         return defaultResult();
1501     }
