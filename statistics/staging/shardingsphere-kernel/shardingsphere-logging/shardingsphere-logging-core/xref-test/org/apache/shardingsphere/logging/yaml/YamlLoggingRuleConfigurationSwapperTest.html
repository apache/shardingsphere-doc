18  package org.apache.shardingsphere.logging.yaml;
19  
20  import org.apache.shardingsphere.logging.config.LoggingRuleConfiguration;
21  import org.apache.shardingsphere.logging.logger.ShardingSphereAppender;
22  import org.apache.shardingsphere.logging.logger.ShardingSphereLogger;
23  import org.apache.shardingsphere.logging.yaml.config.YamlLoggingRuleConfiguration;
24  import org.apache.shardingsphere.logging.yaml.swapper.YamlAppenderSwapper;
25  import org.apache.shardingsphere.logging.yaml.swapper.YamlLoggerSwapper;
26  import org.apache.shardingsphere.logging.yaml.swapper.YamlLoggingRuleConfigurationSwapper;
27  import org.junit.jupiter.api.Test;
28  
29  import java.util.Collections;
30  
31  import static org.hamcrest.CoreMatchers.is;
32  import static org.hamcrest.MatcherAssert.assertThat;
33  
34  class YamlLoggingRuleConfigurationSwapperTest {
35      
36      private final YamlLoggingRuleConfigurationSwapper swapper = new YamlLoggingRuleConfigurationSwapper();
37      
38      private final YamlLoggerSwapper loggerSwapper = new YamlLoggerSwapper();
39      
40      private final YamlAppenderSwapper appenderSwapper = new YamlAppenderSwapper();
41      
42      @Test
43      void swapToYamlConfiguration() {
44          YamlLoggingRuleConfiguration yamlLoggingRuleConfiguration = swapper.swapToYamlConfiguration(createLoggingRuleConfiguration());
45          assertThat(yamlLoggingRuleConfiguration.getLoggers().size(), is(1));
46          assertThat(yamlLoggingRuleConfiguration.getAppenders().size(), is(1));
47      }
48      
49      private LoggingRuleConfiguration createLoggingRuleConfiguration() {
50          return new LoggingRuleConfiguration(Collections.singleton(new ShardingSphereLogger("ROOT", "INFO", true, "console")),
51                  Collections.singleton(new ShardingSphereAppender("console", "ch.qos.logback.core.ConsoleAppender", "[%-5level] %d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %logger{36} - %msg%n")));
52      }
53      
54      @Test
55      void swapToObject() {
56          LoggingRuleConfiguration loggingRuleConfiguration = swapper.swapToObject(createYamlLoggingRuleConfiguration());
57          assertThat(loggingRuleConfiguration.getLoggers().size(), is(1));
58          assertThat(loggingRuleConfiguration.getAppenders().size(), is(1));
59      }
60      
61      private YamlLoggingRuleConfiguration createYamlLoggingRuleConfiguration() {
62          YamlLoggingRuleConfiguration result = new YamlLoggingRuleConfiguration();
63          result.setLoggers(Collections.singleton(loggerSwapper.swapToYamlConfiguration(new ShardingSphereLogger("ROOT", "INFO", true, "console"))));
64          result.setAppenders(Collections.singleton(appenderSwapper.swapToYamlConfiguration(
65                  new ShardingSphereAppender("console", "ch.qos.logback.core.ConsoleAppender", "[%-5level] %d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %logger{36} - %msg%n"))));
66          return result;
67      }
68  }
