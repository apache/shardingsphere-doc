46  class SimpleTableSegmentBinderTest {
47      
48      private final DatabaseType databaseType = TypedSPILoader.getService(DatabaseType.class, "FIXTURE");
49      
50      @Test
51      void assertBind() {
52          SimpleTableSegment simpleTableSegment = new SimpleTableSegment(new TableNameSegment(0, 10, new IdentifierValue("t_order")));
53          ShardingSphereMetaData metaData = createMetaData();
54          Map<String, TableSegmentBinderContext> tableBinderContexts = new CaseInsensitiveMap<>();
55          SimpleTableSegment actual = SimpleTableSegmentBinder.bind(simpleTableSegment, metaData, DefaultDatabase.LOGIC_NAME, databaseType, tableBinderContexts);
56          assertThat(actual.getTableName().getOriginalDatabase().getValue(), is(DefaultDatabase.LOGIC_NAME));
57          assertThat(actual.getTableName().getOriginalSchema().getValue(), is(DefaultDatabase.LOGIC_NAME));
58          assertTrue(tableBinderContexts.containsKey("t_order"));
59          assertThat(tableBinderContexts.get("t_order").getProjectionSegments().size(), is(3));
60          assertTrue(tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("user_id") instanceof ColumnProjectionSegment);
61          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("user_id")).getColumn().getOriginalDatabase().getValue(),
62                  is(DefaultDatabase.LOGIC_NAME));
63          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("user_id")).getColumn().getOriginalSchema().getValue(),
64                  is(DefaultDatabase.LOGIC_NAME));
65          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("user_id")).getColumn().getOriginalTable().getValue(), is("t_order"));
66          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("user_id")).getColumn().getOriginalColumn().getValue(), is("user_id"));
67          assertTrue(tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("order_id") instanceof ColumnProjectionSegment);
68          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("order_id")).getColumn().getOriginalDatabase().getValue(),
69                  is(DefaultDatabase.LOGIC_NAME));
70          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("order_id")).getColumn().getOriginalSchema().getValue(),
71                  is(DefaultDatabase.LOGIC_NAME));
72          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("order_id")).getColumn().getOriginalTable().getValue(), is("t_order"));
73          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("order_id")).getColumn().getOriginalColumn().getValue(), is("order_id"));
74          assertTrue(tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("status") instanceof ColumnProjectionSegment);
75          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("status")).getColumn().getOriginalDatabase().getValue(),
76                  is(DefaultDatabase.LOGIC_NAME));
77          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("status")).getColumn().getOriginalSchema().getValue(),
78                  is(DefaultDatabase.LOGIC_NAME));
79          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("status")).getColumn().getOriginalTable().getValue(), is("t_order"));
80          assertThat(((ColumnProjectionSegment) tableBinderContexts.get("t_order").getProjectionSegmentByColumnLabel("status")).getColumn().getOriginalColumn().getValue(), is("status"));
81      }
82      
83      @Test
84      void assertBindWithSchemaForMySQL() {
85          SimpleTableSegment simpleTableSegment = new SimpleTableSegment(new TableNameSegment(0, 10, new IdentifierValue("t_order")));
86          simpleTableSegment.setOwner(new OwnerSegment(0, 0, new IdentifierValue("sharding_db")));
87          ShardingSphereMetaData metaData = createMetaData();
88          Map<String, TableSegmentBinderContext> tableBinderContexts = new CaseInsensitiveMap<>();
89          SimpleTableSegment actual = SimpleTableSegmentBinder.bind(simpleTableSegment, metaData, DefaultDatabase.LOGIC_NAME, databaseType, tableBinderContexts);
90          assertThat(actual.getTableName().getOriginalDatabase().getValue(), is("sharding_db"));
91          assertThat(actual.getTableName().getOriginalSchema().getValue(), is("sharding_db"));
92      }
93      
94      @Test
95      void assertBindWithoutSchemaForMySQL() {
96          SimpleTableSegment simpleTableSegment = new SimpleTableSegment(new TableNameSegment(0, 10, new IdentifierValue("t_order")));
97          ShardingSphereMetaData metaData = createMetaData();
98          Map<String, TableSegmentBinderContext> tableBinderContexts = new CaseInsensitiveMap<>();
99          SimpleTableSegment actual = SimpleTableSegmentBinder.bind(simpleTableSegment, metaData, DefaultDatabase.LOGIC_NAME, databaseType, tableBinderContexts);
100         assertThat(actual.getTableName().getOriginalDatabase().getValue(), is(DefaultDatabase.LOGIC_NAME));
101         assertThat(actual.getTableName().getOriginalSchema().getValue(), is(DefaultDatabase.LOGIC_NAME));
102     }
103     
104     private ShardingSphereMetaData createMetaData() {
105         ShardingSphereSchema schema = mock(ShardingSphereSchema.class, RETURNS_DEEP_STUBS);
106         when(schema.getTable("t_order").getColumnValues()).thenReturn(Arrays.asList(
107                 new ShardingSphereColumn("order_id", Types.INTEGER, true, false, false, true, false),
108                 new ShardingSphereColumn("user_id", Types.INTEGER, false, false, false, true, false),
109                 new ShardingSphereColumn("status", Types.INTEGER, false, false, false, true, false)));
110         when(schema.getTable("pg_database").getColumnValues()).thenReturn(Arrays.asList(
111                 new ShardingSphereColumn("datname", Types.VARCHAR, false, false, false, true, false),
112                 new ShardingSphereColumn("datdba", Types.VARCHAR, false, false, false, true, false)));
113         ShardingSphereMetaData result = mock(ShardingSphereMetaData.class, RETURNS_DEEP_STUBS);
114         when(result.getDatabase(DefaultDatabase.LOGIC_NAME).getSchema(DefaultDatabase.LOGIC_NAME)).thenReturn(schema);
115         when(result.getDatabase("sharding_db").getSchema("sharding_db")).thenReturn(schema);
116         when(result.getDatabase(DefaultDatabase.LOGIC_NAME).getSchema("public")).thenReturn(schema);
117         when(result.getDatabase("sharding_db").getSchema("test")).thenReturn(schema);
118         return result;
119     }
