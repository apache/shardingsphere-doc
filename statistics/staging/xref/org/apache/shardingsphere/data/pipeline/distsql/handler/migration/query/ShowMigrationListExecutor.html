20  import org.apache.shardingsphere.data.pipeline.core.context.PipelineContextKey;
21  import org.apache.shardingsphere.data.pipeline.core.job.service.PipelineJobManager;
22  import org.apache.shardingsphere.data.pipeline.migration.distsql.statement.queryable.ShowMigrationListStatement;
23  import org.apache.shardingsphere.data.pipeline.scenario.migration.MigrationJobType;
24  import org.apache.shardingsphere.distsql.handler.engine.query.DistSQLQueryExecutor;
25  import org.apache.shardingsphere.infra.instance.metadata.InstanceType;
26  import org.apache.shardingsphere.infra.merge.result.impl.local.LocalDataQueryResultRow;
27  import org.apache.shardingsphere.mode.manager.ContextManager;
28  
29  import java.util.Arrays;
30  import java.util.Collection;
31  import java.util.stream.Collectors;
32  
33  /**
34   * Show migration list executor.
35   */
36  public final class ShowMigrationListExecutor implements DistSQLQueryExecutor<ShowMigrationListStatement> {
37      
38      private final PipelineJobManager pipelineJobManager = new PipelineJobManager(new MigrationJobType());
39      
40      @Override
41      public Collection<String> getColumnNames(final ShowMigrationListStatement sqlStatement) {
42          return Arrays.asList("id", "tables", "job_item_count", "active", "create_time", "stop_time");
43      }
44      
45      @Override
46      public Collection<LocalDataQueryResultRow> getRows(final ShowMigrationListStatement sqlStatement, final ContextManager contextManager) {
47          return pipelineJobManager.getJobInfos(new PipelineContextKey(InstanceType.PROXY)).stream().map(each -> new LocalDataQueryResultRow(each.getJobMetaData().getJobId(), each.getTableName(),
48                  each.getJobMetaData().getJobItemCount(), each.getJobMetaData().isActive(), each.getJobMetaData().getCreateTime(), each.getJobMetaData().getStopTime())).collect(Collectors.toList());
49      }
50      
51      @Override
52      public Class<ShowMigrationListStatement> getType() {
53          return ShowMigrationListStatement.class;
54      }
55  }
