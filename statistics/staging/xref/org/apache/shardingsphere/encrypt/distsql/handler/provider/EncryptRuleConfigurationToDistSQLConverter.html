18  package org.apache.shardingsphere.encrypt.distsql.handler.provider;

20  import com.google.common.base.Strings;
21  import org.apache.shardingsphere.distsql.handler.constant.DistSQLConstants;
22  import org.apache.shardingsphere.distsql.handler.engine.query.ral.convert.AlgorithmDistSQLConverter;
23  import org.apache.shardingsphere.distsql.handler.engine.query.ral.convert.RuleConfigurationToDistSQLConverter;
24  import org.apache.shardingsphere.encrypt.api.config.EncryptRuleConfiguration;
25  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptColumnItemRuleConfiguration;
26  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptColumnRuleConfiguration;
27  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptTableRuleConfiguration;
28  import org.apache.shardingsphere.encrypt.distsql.handler.constant.EncryptDistSQLConstants;
29  import org.apache.shardingsphere.infra.algorithm.core.config.AlgorithmConfiguration;
30  
31  import java.util.Collection;
32  import java.util.Iterator;
33  import java.util.Map;
34  
35  /**
36   * Encrypt rule configuration to DistSQL converter.
37   */
38  public final class EncryptRuleConfigurationToDistSQLConverter implements RuleConfigurationToDistSQLConverter<EncryptRuleConfiguration> {
39      
40      @Override
41      public String convert(final EncryptRuleConfiguration ruleConfig) {
42          if (ruleConfig.getTables().isEmpty()) {
43              return "";
44          }
45          StringBuilder result = new StringBuilder(EncryptDistSQLConstants.CREATE_ENCRYPT);
46          Iterator<EncryptTableRuleConfiguration> iterator = ruleConfig.getTables().iterator();
47          while (iterator.hasNext()) {
48              EncryptTableRuleConfiguration tableRuleConfig = iterator.next();
49              result.append(String.format(EncryptDistSQLConstants.ENCRYPT, tableRuleConfig.getName(), getEncryptColumns(tableRuleConfig.getColumns(), ruleConfig.getEncryptors())));
50              if (iterator.hasNext()) {
51                  result.append(DistSQLConstants.COMMA).append(System.lineSeparator());
52              }
53          }
54          result.append(DistSQLConstants.SEMI);
55          return result.toString();
56      }
57      
58      private String getEncryptColumns(final Collection<EncryptColumnRuleConfiguration> ruleConfigs, final Map<String, AlgorithmConfiguration> encryptors) {
59          StringBuilder result = new StringBuilder();
60          Iterator<EncryptColumnRuleConfiguration> iterator = ruleConfigs.iterator();
61          while (iterator.hasNext()) {
62              EncryptColumnRuleConfiguration columnRuleConfig = iterator.next();
63              result.append(String.format(EncryptDistSQLConstants.ENCRYPT_COLUMN, columnRuleConfig.getName(), getColumns(columnRuleConfig), getEncryptAlgorithms(columnRuleConfig, encryptors)));
64              if (iterator.hasNext()) {
65                  result.append(DistSQLConstants.COMMA).append(System.lineSeparator());
66              }
67          }
68          return result.toString();
69      }
70      
71      private String getColumns(final EncryptColumnRuleConfiguration ruleConfig) {
72          StringBuilder result = new StringBuilder();
73          String cipherColumnName = ruleConfig.getCipher().getName();
74          if (!Strings.isNullOrEmpty(cipherColumnName)) {
75              result.append(String.format(EncryptDistSQLConstants.CIPHER, cipherColumnName));
76          }
77          if (ruleConfig.getAssistedQuery().isPresent()) {
78              result.append(DistSQLConstants.COMMA).append(' ').append(String.format(EncryptDistSQLConstants.ASSISTED_QUERY_COLUMN, ruleConfig.getAssistedQuery().get().getName()));
79          }
80          if (ruleConfig.getLikeQuery().isPresent()) {
81              result.append(DistSQLConstants.COMMA).append(' ').append(String.format(EncryptDistSQLConstants.LIKE_QUERY_COLUMN, ruleConfig.getLikeQuery().get().getName()));
82          }
83          return result.toString();
84      }
85      
86      private String getEncryptAlgorithms(final EncryptColumnRuleConfiguration ruleConfig, final Map<String, AlgorithmConfiguration> encryptors) {
87          StringBuilder result = new StringBuilder();
88          String cipherEncryptorName = ruleConfig.getCipher().getEncryptorName();
89          String assistedQueryEncryptorName = ruleConfig.getAssistedQuery().map(EncryptColumnItemRuleConfiguration::getEncryptorName).orElse("");
90          String likeQueryEncryptorName = ruleConfig.getLikeQuery().map(EncryptColumnItemRuleConfiguration::getEncryptorName).orElse("");
91          if (!Strings.isNullOrEmpty(cipherEncryptorName)) {
92              result.append(String.format(EncryptDistSQLConstants.ENCRYPT_ALGORITHM, AlgorithmDistSQLConverter.getAlgorithmType(encryptors.get(cipherEncryptorName))));
93          }
94          if (!Strings.isNullOrEmpty(assistedQueryEncryptorName)) {
95              result.append(DistSQLConstants.COMMA).append(' ')
96                      .append(String.format(EncryptDistSQLConstants.ASSISTED_QUERY_ALGORITHM, AlgorithmDistSQLConverter.getAlgorithmType(encryptors.get(assistedQueryEncryptorName))));
97          }
98          if (!Strings.isNullOrEmpty(likeQueryEncryptorName)) {
99              result.append(DistSQLConstants.COMMA).append(' ')
100                     .append(String.format(EncryptDistSQLConstants.LIKE_QUERY_ALGORITHM, AlgorithmDistSQLConverter.getAlgorithmType(encryptors.get(likeQueryEncryptorName))));
101         }
102         return result.toString();
103     }
104     
105     @Override
106     public Class<EncryptRuleConfiguration> getType() {
107         return EncryptRuleConfiguration.class;
108     }
109 }
