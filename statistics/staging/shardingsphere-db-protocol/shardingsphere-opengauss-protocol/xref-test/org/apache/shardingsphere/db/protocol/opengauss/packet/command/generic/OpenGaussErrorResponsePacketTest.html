18  package org.apache.shardingsphere.db.protocol.opengauss.packet.command.generic;
19  
20  import org.apache.shardingsphere.db.protocol.postgresql.packet.identifier.PostgreSQLMessagePacketType;
21  import org.apache.shardingsphere.db.protocol.postgresql.payload.PostgreSQLPacketPayload;
22  import org.junit.jupiter.api.Test;
23  import org.opengauss.util.ServerErrorMessage;
24  
25  import static org.hamcrest.CoreMatchers.is;
26  import static org.hamcrest.MatcherAssert.assertThat;
27  import static org.mockito.Mockito.mock;
28  import static org.mockito.Mockito.verify;
29  
30  class OpenGaussErrorResponsePacketTest {
31      
32      @Test
33      void assertWritePacketFromServerErrorMessage() {
34          String encodedMessage = "SFATAL\0C3D000\0Mdatabase \"test\" does not exist\0c-1\0Ddetail\0Hhint\0P1\0p2\0qinternal query\0Wwhere\0Ffile\0L3\0Rroutine\0a0.0.0.0:1";
35          OpenGaussErrorResponsePacket packet = new OpenGaussErrorResponsePacket(new ServerErrorMessage(encodedMessage));
36          PostgreSQLPacketPayload payload = mock(PostgreSQLPacketPayload.class);
37          packet.write(payload);
38          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_SEVERITY);
39          verify(payload).writeStringNul("FATAL");
40          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_CODE);
41          verify(payload).writeStringNul("3D000");
42          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_MESSAGE);
43          verify(payload).writeStringNul("database \"test\" does not exist");
44          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_ERRORCODE);
45          verify(payload).writeStringNul("-1");
46          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_DETAIL);
47          verify(payload).writeStringNul("detail");
48          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_HINT);
49          verify(payload).writeStringNul("hint");
50          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_POSITION);
51          verify(payload).writeStringNul("1");
52          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_INTERNAL_POSITION);
53          verify(payload).writeStringNul("2");
54          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_INTERNAL_QUERY);
55          verify(payload).writeStringNul("internal query");
56          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_WHERE);
57          verify(payload).writeStringNul("where");
58          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_FILE);
59          verify(payload).writeStringNul("file");
60          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_LINE);
61          verify(payload).writeStringNul("3");
62          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_ROUTINE);
63          verify(payload).writeStringNul("routine");
64          verify(payload).writeInt1(0);
65          assertThat(packet.getIdentifier(), is(PostgreSQLMessagePacketType.ERROR_RESPONSE));
66      }
67      
68      @Test
69      void assertWritePacketFromSeverityAndMessage() {
70          OpenGaussErrorResponsePacket packet = new OpenGaussErrorResponsePacket("FATAL", "3D000", "database \"test\" does not exist");
71          PostgreSQLPacketPayload payload = mock(PostgreSQLPacketPayload.class);
72          packet.write(payload);
73          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_SEVERITY);
74          verify(payload).writeStringNul("FATAL");
75          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_CODE);
76          verify(payload).writeStringNul("3D000");
77          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_MESSAGE);
78          verify(payload).writeStringNul("database \"test\" does not exist");
79          verify(payload).writeInt1(OpenGaussErrorResponsePacket.FIELD_TYPE_ERRORCODE);
80          verify(payload).writeStringNul("0");
81      }
82  }
