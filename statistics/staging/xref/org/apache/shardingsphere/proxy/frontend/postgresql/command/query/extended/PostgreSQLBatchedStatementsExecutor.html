68  /**
69   * Batched statements executor for PostgreSQL.
70   */
71  public final class PostgreSQLBatchedStatementsExecutor {
72      
73      private final KernelProcessor kernelProcessor = new KernelProcessor();
74      
75      private final JDBCExecutor jdbcExecutor;
76      
77      private final ConnectionSession connectionSession;
78      
79      private final MetaDataContexts metaDataContexts;
80      
81      private final PostgreSQLServerPreparedStatement preparedStatement;
82      
83      private final Map<ExecutionUnit, List<List<Object>>> executionUnitParams = new HashMap<>();
84      
85      private final ExecutionContext anyExecutionContext;
86      
87      private ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext;
88      
89      public PostgreSQLBatchedStatementsExecutor(final ConnectionSession connectionSession, final PostgreSQLServerPreparedStatement preparedStatement, final List<List<Object>> parameterSets) {
90          jdbcExecutor = new JDBCExecutor(BackendExecutorContext.getInstance().getExecutorEngine(), connectionSession.getConnectionContext());
91          this.connectionSession = connectionSession;
92          metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
93          this.preparedStatement = preparedStatement;
94          Iterator<List<Object>> parameterSetsIterator = parameterSets.iterator();
95          SQLStatementContext sqlStatementContext = null;
96          ExecutionContext executionContext = null;
97          if (parameterSetsIterator.hasNext()) {
98              List<Object> firstGroupOfParam = parameterSetsIterator.next();
99              sqlStatementContext = createSQLStatementContext(firstGroupOfParam, preparedStatement.getHintValueContext());
100             executionContext = createExecutionContext(createQueryContext(sqlStatementContext, firstGroupOfParam, preparedStatement.getHintValueContext()));
101             for (ExecutionUnit each : executionContext.getExecutionUnits()) {
102                 executionUnitParams.computeIfAbsent(each, unused -> new LinkedList<>()).add(each.getSqlUnit().getParameters());
103             }
104         }
105         anyExecutionContext = executionContext;
106         prepareForRestOfParametersSet(parameterSetsIterator, sqlStatementContext, preparedStatement.getHintValueContext());
107     }
108     
109     private SQLStatementContext createSQLStatementContext(final List<Object> params, final HintValueContext hintValueContext) {
110         return new SQLBindEngine(metaDataContexts.getMetaData(), connectionSession.getUsedDatabaseName(), hintValueContext).bind(preparedStatement.getSqlStatementContext().getSqlStatement(), params);
111     }
112     
113     private void prepareForRestOfParametersSet(final Iterator<List<Object>> paramSetsIterator, final SQLStatementContext sqlStatementContext, final HintValueContext hintValueContext) {
114         while (paramSetsIterator.hasNext()) {
115             List<Object> eachGroupOfParam = paramSetsIterator.next();
116             if (sqlStatementContext instanceof ParameterAware) {
117                 ((ParameterAware) sqlStatementContext).setUpParameters(eachGroupOfParam);
118             }
119             ExecutionContext eachExecutionContext = createExecutionContext(createQueryContext(sqlStatementContext, eachGroupOfParam, hintValueContext));
120             for (ExecutionUnit each : eachExecutionContext.getExecutionUnits()) {
121                 executionUnitParams.computeIfAbsent(each, unused -> new LinkedList<>()).add(each.getSqlUnit().getParameters());
122             }
123         }
124     }
125     
126     private QueryContext createQueryContext(final SQLStatementContext sqlStatementContext, final List<Object> params, final HintValueContext hintValueContext) {
127         return new QueryContext(sqlStatementContext, preparedStatement.getSql(), params, hintValueContext, connectionSession.getConnectionContext(), metaDataContexts.getMetaData());
128     }
129     
130     private ExecutionContext createExecutionContext(final QueryContext queryContext) {
131         RuleMetaData globalRuleMetaData = metaDataContexts.getMetaData().getGlobalRuleMetaData();
132         ShardingSphereDatabase currentDatabase = metaDataContexts.getMetaData().getDatabase(connectionSession.getUsedDatabaseName());
133         SQLAuditEngine.audit(queryContext, globalRuleMetaData, currentDatabase);
134         return kernelProcessor.generateExecutionContext(queryContext, globalRuleMetaData, metaDataContexts.getMetaData().getProps(), connectionSession.getConnectionContext());
135     }
136     
137     /**
138      * Execute batch.
139      *
140      * @return inserted rows
141      * @throws SQLException SQL exception
142      */
143     public int executeBatch() throws SQLException {
144         connectionSession.getDatabaseConnectionManager().handleAutoCommit();
145         addBatchedParametersToPreparedStatements();
146         return executeBatchedPreparedStatements();
147     }
148     
149     private void addBatchedParametersToPreparedStatements() throws SQLException {
150         Collection<ShardingSphereRule> rules = metaDataContexts.getMetaData().getDatabase(connectionSession.getUsedDatabaseName()).getRuleMetaData().getRules();
151         DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine = new DriverExecutionPrepareEngine<>(JDBCDriverType.PREPARED_STATEMENT,
152                 metaDataContexts.getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY),
153                 connectionSession.getDatabaseConnectionManager(), (JDBCBackendStatement) connectionSession.getStatementManager(),
154                 new StatementOption(false), rules, metaDataContexts.getMetaData().getDatabase(connectionSession.getUsedDatabaseName()).getResourceMetaData().getStorageUnits());
155         executionGroupContext = prepareEngine.prepare(connectionSession.getUsedDatabaseName(), anyExecutionContext.getRouteContext(), executionUnitParams.keySet(),
156                 new ExecutionGroupReportContext(connectionSession.getProcessId(), connectionSession.getUsedDatabaseName(), connectionSession.getConnectionContext().getGrantee()));
157         for (ExecutionGroup<JDBCExecutionUnit> eachGroup : executionGroupContext.getInputGroups()) {
158             for (JDBCExecutionUnit each : eachGroup.getInputs()) {
159                 prepareJDBCExecutionUnit(each);
160             }
161         }
162     }
163     
164     private void prepareJDBCExecutionUnit(final JDBCExecutionUnit jdbcExecutionUnit) throws SQLException {
165         PreparedStatement preparedStatement = (PreparedStatement) jdbcExecutionUnit.getStorageResource();
166         for (List<Object> eachGroupParam : executionUnitParams.getOrDefault(jdbcExecutionUnit.getExecutionUnit(), Collections.emptyList())) {
167             ListIterator<Object> params = eachGroupParam.listIterator();
168             while (params.hasNext()) {
169                 int paramIndex = params.nextIndex() + 1;
170                 Object value = params.next();
171                 if (value instanceof PostgreSQLTypeUnspecifiedSQLParameter) {
172                     value = value.toString();
173                 }
174                 preparedStatement.setObject(paramIndex, value);
175             }
176             preparedStatement.addBatch();
177         }
178     }
179     
180     private int executeBatchedPreparedStatements() throws SQLException {
181         boolean isExceptionThrown = SQLExecutorExceptionHandler.isExceptionThrown();
182         ShardingSphereDatabase database = metaDataContexts.getMetaData().getDatabase(connectionSession.getUsedDatabaseName());
183         DatabaseType protocolType = database.getProtocolType();
184         JDBCExecutorCallback<int[]> callback =
185                 new BatchedStatementsJDBCExecutorCallback(protocolType, database.getResourceMetaData(), preparedStatement.getSqlStatementContext().getSqlStatement(), isExceptionThrown);
186         List<int[]> executeResults = jdbcExecutor.execute(executionGroupContext, callback);
187         int result = 0;
188         for (int[] eachResult : executeResults) {
189             for (int each : eachResult) {
190                 result += each;
191             }
192         }
193         return result;
194     }
195     
196     private static final class BatchedStatementsJDBCExecutorCallback extends JDBCExecutorCallback<int[]> {
197         
198         private BatchedStatementsJDBCExecutorCallback(final DatabaseType protocolType, final ResourceMetaData resourceMetaData, final SQLStatement sqlStatement,
199                                                       final boolean isExceptionThrown) {
200             super(protocolType, resourceMetaData, sqlStatement, isExceptionThrown);
201         }
202         
203         @Override
204         protected int[] executeSQL(final String sql, final Statement statement, final ConnectionMode connectionMode, final DatabaseType storageType) throws SQLException {
205             try {
206                 return statement.executeBatch();
207             } finally {
208                 statement.close();
209             }
210         }
211         
212         @SuppressWarnings("OptionalContainsCollection")
213         @Override
214         protected Optional<int[]> getSaneResult(final SQLStatement sqlStatement, final SQLException ex) {
215             return Optional.empty();
216         }
217     }
