50  /**
51   * Data match data consistency calculate algorithm.
52   */
53  @SPIDescription("Match raw data of records.")
54  @Slf4j
55  public final class DataMatchDataConsistencyCalculateAlgorithm extends AbstractStreamingDataConsistencyCalculateAlgorithm {
56      
57      private static final String CHUNK_SIZE_KEY = "chunk-size";
58      
59      private static final int DEFAULT_CHUNK_SIZE = 1000;
60      
61      private int chunkSize;
62      
63      @Override
64      public void init(final Properties props) {
65          chunkSize = getChunkSize(props);
66      }
67      
68      private int getChunkSize(final Properties props) {
69          int result;
70          try {
71              result = Integer.parseInt(props.getProperty(CHUNK_SIZE_KEY, Integer.toString(DEFAULT_CHUNK_SIZE)));
72          } catch (final NumberFormatException ignore) {
73              log.warn("'chunk-size' is not a valid number, use default value {}", DEFAULT_CHUNK_SIZE);
74              return DEFAULT_CHUNK_SIZE;
75          }
76          if (result <= 0) {
77              log.warn("Invalid 'chunk-size': {}, use default value {}", result, DEFAULT_CHUNK_SIZE);
78              return DEFAULT_CHUNK_SIZE;
79          }
80          return result;
81      }
82      
83      @Override
84      public Optional<DataConsistencyCalculatedResult> calculateChunk(final DataConsistencyCalculateParameter param) {
85          CalculationContext calculationContext = getOrCreateCalculationContext(param);
86          try {
87              Collection<Collection<Object>> records = new LinkedList<>();
88              Object maxUniqueKeyValue = null;
89              ColumnValueReaderEngine columnValueReaderEngine = new ColumnValueReaderEngine(param.getDatabaseType());
90              ResultSet resultSet = calculationContext.getResultSet();
91              while (resultSet.next()) {
92                  ShardingSpherePreconditions.checkState(!isCanceling(), () -> new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName()));
93                  ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
94                  int columnCount = resultSetMetaData.getColumnCount();
95                  Collection<Object> columnRecord = new LinkedList<>();
96                  for (int columnIndex = 1; columnIndex <= columnCount; columnIndex++) {
97                      columnRecord.add(columnValueReaderEngine.read(resultSet, resultSetMetaData, columnIndex));
98                  }
99                  records.add(columnRecord);
100                 maxUniqueKeyValue = columnValueReaderEngine.read(resultSet, resultSetMetaData, param.getUniqueKey().getOrdinalPosition());
101                 if (records.size() == chunkSize) {
102                     break;
103                 }
104             }
105             if (records.isEmpty()) {
106                 calculationContext.close();
107             }
108             return records.isEmpty() ? Optional.empty() : Optional.of(new DataMatchCalculatedResult(maxUniqueKeyValue, records));
109         } catch (final PipelineSQLException ex) {
110             calculationContext.close();
111             throw ex;
112             // CHECKSTYLE:OFF
113         } catch (final SQLException | RuntimeException ex) {
114             // CHECKSTYLE:ON
115             calculationContext.close();
116             throw new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), ex);
117         }
118     }
119     
120     private CalculationContext getOrCreateCalculationContext(final DataConsistencyCalculateParameter param) {
121         CalculationContext result = (CalculationContext) param.getCalculationContext();
122         if (null != result) {
123             return result;
124         }
125         try {
126             result = createCalculationContext(param);
127             fulfillCalculationContext(result, param);
128             // CHECKSTYLE:OFF
129         } catch (final SQLException | RuntimeException ex) {
130             // CHECKSTYLE:ON
131             QuietlyCloser.close(result);
132             throw new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), ex);
133         }
134         return result;
135     }
136     
137     private CalculationContext createCalculationContext(final DataConsistencyCalculateParameter param) throws SQLException {
138         Connection connection = param.getDataSource().getConnection();
139         CalculationContext result = new CalculationContext(connection);
140         param.setCalculationContext(result);
141         return result;
142     }
143     
144     private void fulfillCalculationContext(final CalculationContext calculationContext, final DataConsistencyCalculateParameter param) throws SQLException {
145         String sql = getQuerySQL(param);
146         PreparedStatement preparedStatement = JDBCStreamQueryBuilder.build(param.getDatabaseType(), calculationContext.getConnection(), sql);
147         setCurrentStatement(preparedStatement);
148         if (!(param.getDatabaseType() instanceof MySQLDatabaseType)) {
149             preparedStatement.setFetchSize(chunkSize);
150         }
151         calculationContext.setPreparedStatement(preparedStatement);
152         Object tableCheckPosition = param.getTableCheckPosition();
153         if (null != tableCheckPosition) {
154             preparedStatement.setObject(1, tableCheckPosition);
155         }
156         ResultSet resultSet = preparedStatement.executeQuery();
157         calculationContext.setResultSet(resultSet);
158     }
159     
160     private String getQuerySQL(final DataConsistencyCalculateParameter param) {
161         if (null == param.getUniqueKey()) {
162             throw new UnsupportedOperationException("Data consistency of DATA_MATCH type not support table without unique key and primary key now");
163         }
164         PipelineDataConsistencyCalculateSQLBuilder pipelineSQLBuilder = new PipelineDataConsistencyCalculateSQLBuilder(param.getDatabaseType());
165         Collection<String> columnNames = param.getColumnNames().isEmpty() ? Collections.singleton("*") : param.getColumnNames();
166         boolean firstQuery = null == param.getTableCheckPosition();
167         return pipelineSQLBuilder.buildQueryAllOrderingSQL(param.getSchemaName(), param.getLogicTableName(), columnNames, param.getUniqueKey().getName(), firstQuery);
168     }
169     
170     @Override
171     public String getType() {
172         return "DATA_MATCH";
173     }
174     
175     @Override
176     public Collection<DatabaseType> getSupportedDatabaseTypes() {
177         return ShardingSphereServiceLoader.getServiceInstances(DatabaseType.class);
178     }
179     
180     @RequiredArgsConstructor
181     private static final class CalculationContext implements AutoCloseable {
182         
183         @Getter
184         private final Connection connection;
185         
186         private final AtomicReference<PreparedStatement> preparedStatement = new AtomicReference<>();
187         
188         private final AtomicReference<ResultSet> resultSet = new AtomicReference<>();
189         
190         /**
191          * Get result set.
192          *
193          * @return result set
194          */
195         public ResultSet getResultSet() {
196             return resultSet.get();
197         }
198         
199         /**
200          * Set prepared statement.
201          *
202          * @param preparedStatement prepared statement
203          */
204         public void setPreparedStatement(final PreparedStatement preparedStatement) {
205             this.preparedStatement.set(preparedStatement);
206         }
207         
208         /**
209          * Set result set.
210          *
211          * @param resultSet result set
212          */
213         public void setResultSet(final ResultSet resultSet) {
214             this.resultSet.set(resultSet);
215         }
216         
217         @Override
218         public void close() {
219             QuietlyCloser.close(resultSet.get());
220             QuietlyCloser.close(preparedStatement.get());
221             QuietlyCloser.close(connection);
222         }
223     }
