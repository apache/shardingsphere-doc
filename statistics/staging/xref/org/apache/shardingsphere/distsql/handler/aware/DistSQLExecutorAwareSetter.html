18  package org.apache.shardingsphere.distsql.handler.aware;

20  import lombok.RequiredArgsConstructor;
21  import org.apache.shardingsphere.distsql.handler.engine.DistSQLConnectionContext;
22  import org.apache.shardingsphere.distsql.handler.engine.update.rdl.rule.spi.global.GlobalRuleDefinitionExecutor;
23  import org.apache.shardingsphere.distsql.statement.DistSQLStatement;
24  import org.apache.shardingsphere.distsql.statement.rql.rule.global.ShowGlobalRulesStatement;
25  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
26  import org.apache.shardingsphere.infra.exception.generic.UnsupportedSQLOperationException;
27  import org.apache.shardingsphere.infra.exception.dialect.exception.syntax.database.NoDatabaseSelectedException;
28  import org.apache.shardingsphere.infra.metadata.database.ShardingSphereDatabase;
29  import org.apache.shardingsphere.infra.rule.ShardingSphereRule;
30  import org.apache.shardingsphere.mode.manager.ContextManager;
31  
32  import java.util.Optional;
33  
34  /**
35   * DistSQL executor aware setter.
36   */
37  @RequiredArgsConstructor
38  public final class DistSQLExecutorAwareSetter {
39      
40      private final Object executor;
41      
42      /**
43       * Set aware context.
44       *
45       * @param contextManager context manager
46       * @param database database
47       * @param connectionContext connection context
48       * @param sqlStatement DistSQL statement
49       */
50      @SuppressWarnings("rawtypes")
51      public void set(final ContextManager contextManager, final ShardingSphereDatabase database, final DistSQLConnectionContext connectionContext, final DistSQLStatement sqlStatement) {
52          if (executor instanceof DistSQLExecutorDatabaseAware) {
53              ShardingSpherePreconditions.checkNotNull(database, NoDatabaseSelectedException::new);
54              ((DistSQLExecutorDatabaseAware) executor).setDatabase(database);
55          }
56          if (executor instanceof DistSQLExecutorRuleAware) {
57              if (!(sqlStatement instanceof ShowGlobalRulesStatement) && !(executor instanceof GlobalRuleDefinitionExecutor)) {
58                  ShardingSpherePreconditions.checkNotNull(database, NoDatabaseSelectedException::new);
59              }
60              setRule((DistSQLExecutorRuleAware) executor, contextManager, database);
61          }
62          if (executor instanceof DistSQLExecutorConnectionContextAware) {
63              ((DistSQLExecutorConnectionContextAware) executor).setConnectionContext(connectionContext);
64          }
65      }
66      
67      @SuppressWarnings({"rawtypes", "unchecked"})
68      private void setRule(final DistSQLExecutorRuleAware executor, final ContextManager contextManager, final ShardingSphereDatabase database) throws UnsupportedSQLOperationException {
69          Optional<ShardingSphereRule> rule = findRule(contextManager, database, executor.getRuleClass());
70          ShardingSpherePreconditions.checkState(rule.isPresent(), () -> new UnsupportedSQLOperationException(String.format("The current database has no `%s` rules", executor.getRuleClass())));
71          executor.setRule(rule.get());
72      }
73      
74      private Optional<ShardingSphereRule> findRule(final ContextManager contextManager, final ShardingSphereDatabase database, final Class<ShardingSphereRule> ruleClass) {
75          Optional<ShardingSphereRule> globalRule = contextManager.getMetaDataContexts().getMetaData().getGlobalRuleMetaData().findSingleRule(ruleClass);
76          return globalRule.isPresent() ? globalRule : database.getRuleMetaData().findSingleRule(ruleClass);
77      }
78  }
