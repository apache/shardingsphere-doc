25  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.expr.BinaryOperationExpression;
26  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.expr.ExistsSubqueryExpression;
27  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.expr.ExpressionSegment;
28  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.expr.InExpression;
29  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.expr.ListExpression;

31  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.expr.subquery.SubquerySegment;

35  import org.apache.shardingsphere.sql.parser.sql.common.segment.generic.table.JoinTableSegment;
36  import org.apache.shardingsphere.sql.parser.sql.common.segment.generic.table.SubqueryTableSegment;
37  import org.apache.shardingsphere.sql.parser.sql.common.segment.generic.table.TableSegment;
38  import org.apache.shardingsphere.sql.parser.sql.common.statement.dml.SelectStatement;

40  import java.util.Collection;
41  import java.util.LinkedList;
42  import java.util.List;
43  
44  /**
45   * Subquery extract utility class.
46   */
47  @NoArgsConstructor(access = AccessLevel.PRIVATE)
48  public final class SubqueryExtractUtils {
49      
50      /**
51       * Get subquery segment from SelectStatement.
52       *
53       * @param selectStatement SelectStatement
54       * @return subquery segment collection
55       */
56      public static Collection<SubquerySegment> getSubquerySegments(final SelectStatement selectStatement) {
57          List<SubquerySegment> result = new LinkedList<>();
58          extractSubquerySegments(result, selectStatement);
59          return result;
60      }
61      
62      private static void extractSubquerySegments(final List<SubquerySegment> result, final SelectStatement selectStatement) {
63          extractSubquerySegmentsFromProjections(result, selectStatement.getProjections());
64          extractSubquerySegmentsFromTableSegment(result, selectStatement.getFrom());
65          if (selectStatement.getWhere().isPresent()) {
66              extractSubquerySegmentsFromExpression(result, selectStatement.getWhere().get().getExpr());
67          }
68          if (selectStatement.getCombine().isPresent()) {
69              CombineSegment combineSegment = selectStatement.getCombine().get();
70              extractSubquerySegments(result, combineSegment.getLeft());
71              extractSubquerySegments(result, combineSegment.getRight());
72          }
73      }
74      
75      private static void extractSubquerySegmentsFromProjections(final List<SubquerySegment> result, final ProjectionsSegment projections) {
76          if (null == projections || projections.getProjections().isEmpty()) {
77              return;
78          }
79          for (ProjectionSegment each : projections.getProjections()) {
80              if (!(each instanceof SubqueryProjectionSegment)) {
81                  continue;
82              }
83              SubquerySegment subquery = ((SubqueryProjectionSegment) each).getSubquery();
84              subquery.setSubqueryType(SubqueryType.PROJECTION_SUBQUERY);
85              result.add(subquery);
86              extractSubquerySegments(result, subquery.getSelect());
87          }
88      }
89      
90      private static void extractSubquerySegmentsFromTableSegment(final List<SubquerySegment> result, final TableSegment tableSegment) {
91          if (null == tableSegment) {
92              return;
93          }
94          if (tableSegment instanceof SubqueryTableSegment) {
95              SubquerySegment subquery = ((SubqueryTableSegment) tableSegment).getSubquery();
96              subquery.setSubqueryType(SubqueryType.TABLE_SUBQUERY);
97              result.add(subquery);
98              extractSubquerySegments(result, subquery.getSelect());
99          }
100         if (tableSegment instanceof JoinTableSegment) {
101             extractSubquerySegmentsFromTableSegment(result, ((JoinTableSegment) tableSegment).getLeft());
102             extractSubquerySegmentsFromTableSegment(result, ((JoinTableSegment) tableSegment).getRight());
103         }
104     }
105     
106     private static void extractSubquerySegmentsFromExpression(final List<SubquerySegment> result, final ExpressionSegment expressionSegment) {
107         if (expressionSegment instanceof SubqueryExpressionSegment) {
108             SubquerySegment subquery = ((SubqueryExpressionSegment) expressionSegment).getSubquery();
109             subquery.setSubqueryType(SubqueryType.PREDICATE_SUBQUERY);
110             result.add(subquery);
111             extractSubquerySegments(result, subquery.getSelect());
112         }
113         if (expressionSegment instanceof ListExpression) {
114             for (ExpressionSegment each : ((ListExpression) expressionSegment).getItems()) {
115                 extractSubquerySegmentsFromExpression(result, each);
116             }
117         }
118         if (expressionSegment instanceof BinaryOperationExpression) {
119             extractSubquerySegmentsFromExpression(result, ((BinaryOperationExpression) expressionSegment).getLeft());
120             extractSubquerySegmentsFromExpression(result, ((BinaryOperationExpression) expressionSegment).getRight());
121         }
122         if (expressionSegment instanceof InExpression) {
123             extractSubquerySegmentsFromExpression(result, ((InExpression) expressionSegment).getLeft());
124             extractSubquerySegmentsFromExpression(result, ((InExpression) expressionSegment).getRight());
125         }
126         if (expressionSegment instanceof BetweenExpression) {
127             extractSubquerySegmentsFromExpression(result, ((BetweenExpression) expressionSegment).getBetweenExpr());
128             extractSubquerySegmentsFromExpression(result, ((BetweenExpression) expressionSegment).getAndExpr());
129         }
130         if (expressionSegment instanceof ExistsSubqueryExpression) {
131             SubquerySegment subquery = ((ExistsSubqueryExpression) expressionSegment).getSubquery();
132             subquery.setSubqueryType(SubqueryType.EXISTS_SUBQUERY);
133             result.add(subquery);
134             extractSubquerySegments(result, subquery.getSelect());
135         }
136     }
137 }
