43  class EncryptMetaDataReviseEngineTest {
44      
45      private static final String TABLE_NAME = "t_encrypt";
46      
47      @Test
48      void assertRevise() {
49          Map<String, SchemaMetaData> schemaMetaData = Collections.singletonMap(
50                  DefaultDatabase.LOGIC_NAME, new SchemaMetaData(DefaultDatabase.LOGIC_NAME, Collections.singleton(createTableMetaData())));
51          TableMetaData actual = new MetaDataReviseEngine(Collections.singleton(mockEncryptRule())).revise(
52                  schemaMetaData, mock(GenericSchemaBuilderMaterial.class)).get(DefaultDatabase.LOGIC_NAME).getTables().iterator().next();
53          assertThat(actual.getColumns().size(), is(2));
54          Iterator<ColumnMetaData> columns = actual.getColumns().iterator();
55          assertThat(columns.next().getName(), is("id"));
56          assertThat(columns.next().getName(), is("pwd"));
57      }
58      
59      private EncryptRule mockEncryptRule() {
60          EncryptRule result = mock(EncryptRule.class);
61          EncryptTable encryptTable = mock(EncryptTable.class);
62          when(result.findEncryptTable(TABLE_NAME)).thenReturn(Optional.of(encryptTable));
63          when(encryptTable.getAssistedQueryColumns()).thenReturn(Collections.emptyList());
64          when(encryptTable.getLikeQueryColumns()).thenReturn(Collections.singletonList("pwd_like"));
65          when(encryptTable.getPlainColumns()).thenReturn(Collections.singleton("pwd_plain"));
66          when(encryptTable.isCipherColumn("pwd_cipher")).thenReturn(true);
67          when(encryptTable.getLogicColumnByCipherColumn("pwd_cipher")).thenReturn("pwd");
68          when(encryptTable.getLogicColumnByPlainColumn("pwd_plain")).thenReturn("pwd");
69          return result;
70      }
71      
72      private TableMetaData createTableMetaData() {
73          Collection<ColumnMetaData> columns = Arrays.asList(new ColumnMetaData("id", Types.INTEGER, true, true, true, true, false),
74                  new ColumnMetaData("pwd_cipher", Types.VARCHAR, false, false, true, true, false),
75                  new ColumnMetaData("pwd_plain", Types.VARCHAR, false, false, true, true, false),
76                  new ColumnMetaData("pwd_like", Types.VARCHAR, false, false, true, true, false));
77          return new TableMetaData(TABLE_NAME, columns, Collections.emptyList(), Collections.emptyList());
78      }
79  }
