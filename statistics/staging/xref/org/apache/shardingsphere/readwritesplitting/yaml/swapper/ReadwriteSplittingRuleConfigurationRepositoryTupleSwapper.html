40  /**
41   * Readwrite-splitting rule configuration repository tuple swapper.
42   */
43  public final class ReadwriteSplittingRuleConfigurationRepositoryTupleSwapper implements RepositoryTupleSwapper<ReadwriteSplittingRuleConfiguration, YamlReadwriteSplittingRuleConfiguration> {
44      
45      private final RuleNodePath readwriteSplittingRuleNodePath = new ReadwriteSplittingRuleNodePathProvider().getRuleNodePath();
46      
47      @Override
48      public Collection<RepositoryTuple> swapToRepositoryTuples(final YamlReadwriteSplittingRuleConfiguration yamlRuleConfig) {
49          Collection<RepositoryTuple> result = new LinkedList<>();
50          for (Entry<String, YamlAlgorithmConfiguration> entry : yamlRuleConfig.getLoadBalancers().entrySet()) {
51              result.add(new RepositoryTuple(readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.LOAD_BALANCERS).getPath(entry.getKey()),
52                      YamlEngine.marshal(entry.getValue())));
53          }
54          for (Entry<String, YamlReadwriteSplittingDataSourceGroupRuleConfiguration> entry : yamlRuleConfig.getDataSourceGroups().entrySet()) {
55              result.add(new RepositoryTuple(
56                      readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.DATA_SOURCES).getPath(entry.getKey()), YamlEngine.marshal(entry.getValue())));
57          }
58          return result;
59      }
60      
61      @Override
62      public Optional<YamlReadwriteSplittingRuleConfiguration> swapToObject(final Collection<RepositoryTuple> repositoryTuples) {
63          List<RepositoryTuple> validRepositoryTuples = repositoryTuples.stream().filter(each -> readwriteSplittingRuleNodePath.getRoot().isValidatedPath(each.getKey())).collect(Collectors.toList());
64          if (validRepositoryTuples.isEmpty()) {
65              return Optional.empty();
66          }
67          YamlReadwriteSplittingRuleConfiguration yamlRuleConfig = new YamlReadwriteSplittingRuleConfiguration();
68          Map<String, YamlReadwriteSplittingDataSourceGroupRuleConfiguration> dataSourceGroups = new LinkedHashMap<>();
69          Map<String, YamlAlgorithmConfiguration> loadBalancers = new LinkedHashMap<>();
70          for (RepositoryTuple each : validRepositoryTuples) {
71              readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.DATA_SOURCES).getName(each.getKey())
72                      .ifPresent(optional -> dataSourceGroups.put(optional, YamlEngine.unmarshal(each.getValue(), YamlReadwriteSplittingDataSourceGroupRuleConfiguration.class)));
73              readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.LOAD_BALANCERS).getName(each.getKey())
74                      .ifPresent(optional -> loadBalancers.put(optional, YamlEngine.unmarshal(each.getValue(), YamlAlgorithmConfiguration.class)));
75          }
76          yamlRuleConfig.setDataSourceGroups(dataSourceGroups);
77          yamlRuleConfig.setLoadBalancers(loadBalancers);
78          return Optional.of(yamlRuleConfig);
79      }
80      
81      @Override
82      public Class<ReadwriteSplittingRuleConfiguration> getTypeClass() {
83          return ReadwriteSplittingRuleConfiguration.class;
84      }
85      
86      @Override
87      public String getRuleTagName() {
88          return "READWRITE_SPLITTING";
89      }
90      
91      @Override
92      public int getOrder() {
93          return ReadwriteSplittingOrder.ORDER;
94      }
95  }
