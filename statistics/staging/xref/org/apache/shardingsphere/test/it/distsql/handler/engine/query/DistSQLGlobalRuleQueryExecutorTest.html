22  import org.apache.shardingsphere.distsql.handler.engine.query.DistSQLQueryExecuteEngine;
23  import org.apache.shardingsphere.distsql.statement.DistSQLStatement;
24  import org.apache.shardingsphere.infra.config.rule.scope.GlobalRuleConfiguration;
25  import org.apache.shardingsphere.infra.merge.result.impl.local.LocalDataQueryResultRow;

29  
30  import java.sql.SQLException;
31  import java.util.ArrayList;
32  import java.util.Collection;
33  import java.util.Collections;
34  
35  import static org.apache.shardingsphere.test.matcher.ShardingSphereAssertionMatchers.deepEqual;
36  import static org.hamcrest.MatcherAssert.assertThat;
37  import static org.mockito.Mockito.RETURNS_DEEP_STUBS;
38  import static org.mockito.Mockito.mock;
39  import static org.mockito.Mockito.when;
40  
41  @RequiredArgsConstructor
42  public abstract class DistSQLGlobalRuleQueryExecutorTest {
43      
44      private final GlobalRule mockedRule;
45      
46      protected void assertQueryResultRows(final GlobalRuleConfiguration ruleConfig, final DistSQLStatement sqlStatement, final Collection<LocalDataQueryResultRow> expected) throws SQLException {
47          DistSQLQueryExecuteEngine engine = new DistSQLQueryExecuteEngine(sqlStatement, null, mockContextManager(ruleConfig), mock(DistSQLConnectionContext.class));
48          engine.executeQuery();
49          Collection<LocalDataQueryResultRow> actual = new ArrayList<>(engine.getRows());
50          assertThat(actual, deepEqual(new ArrayList<>(expected)));
51      }
52      
53      private ContextManager mockContextManager(final GlobalRuleConfiguration ruleConfig) {
54          ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
55          when(mockedRule.getConfiguration()).thenReturn(ruleConfig);
56          when(result.getMetaDataContexts().getMetaData().getGlobalRuleMetaData()).thenReturn(new RuleMetaData(Collections.singleton(mockedRule)));
57          return result;
58      }
59  }
