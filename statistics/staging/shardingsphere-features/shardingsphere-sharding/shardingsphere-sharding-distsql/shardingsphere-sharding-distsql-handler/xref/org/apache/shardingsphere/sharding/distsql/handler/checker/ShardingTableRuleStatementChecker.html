112      *

114      * @param currentRuleConfig current rule configuration
115      * @return binding table groups are valid or not

117     public static boolean isValidBindingTableGroups(final Collection<ShardingTableReferenceRuleConfiguration> bindingTableGroups, final ShardingRuleConfiguration currentRuleConfig) {
118         ShardingRuleConfiguration toBeCheckedRuleConfig = createToBeCheckedShardingRuleConfiguration(currentRuleConfig);
119         toBeCheckedRuleConfig.setBindingTableGroups(bindingTableGroups);
120         Collection<String> dataSourceNames = getRequiredDataSource(toBeCheckedRuleConfig);
121         dataSourceNames.addAll(getRequiredDataSource(currentRuleConfig));
122         return check(toBeCheckedRuleConfig, dataSourceNames);
123     }

125     private static void check(final ShardingSphereDatabase database, final Collection<AbstractTableRuleSegment> rules, final boolean ifNotExists, final ShardingRuleConfiguration currentRuleConfig,
126                               final boolean isCreated) {
127         String databaseName = database.getName();
128         checkTables(databaseName, rules, currentRuleConfig, isCreated, ifNotExists);
129         checkDataSources(databaseName, rules, database);
130         checkKeyGenerators(rules, currentRuleConfig);
131         checkAuditors(rules);
132         Map<String, List<AbstractTableRuleSegment>> groupedTableRule = groupingByClassType(rules);
133         checkAutoTableRule(groupedTableRule.getOrDefault(AutoTableRuleSegment.class.getSimpleName(), Collections.emptyList()));
134         checkTableRule(databaseName, groupedTableRule.getOrDefault(TableRuleSegment.class.getSimpleName(), Collections.emptyList()));
135         if (!isCreated) {
136             checkBindingTableRules(rules, currentRuleConfig);
137         }
138     }

140     private static boolean check(final ShardingRuleConfiguration checkedConfig, final Collection<String> dataSourceNames) {
141         Collection<String> allDataSourceNames = getDataSourceNames(checkedConfig.getTables(), checkedConfig.getAutoTables(), dataSourceNames);
142         Map<String, ShardingAlgorithm> shardingAlgorithms = new HashMap<>(checkedConfig.getShardingAlgorithms().size(), 1);
143         Map<String, TableRule> tableRules = new HashMap<>();
144         checkedConfig.getShardingAlgorithms().forEach((key, value) -> shardingAlgorithms.put(key, TypedSPILoader.getService(ShardingAlgorithm.class, value.getType(), value.getProps())));
145         tableRules.putAll(createTableRules(checkedConfig.getTables(), checkedConfig.getDefaultKeyGenerateStrategy(), allDataSourceNames));
146         tableRules.putAll(createAutoTableRules(checkedConfig.getAutoTables(), shardingAlgorithms, checkedConfig.getDefaultKeyGenerateStrategy(), allDataSourceNames));
147         Collection<String> broadcastTables = createBroadcastTables(checkedConfig.getBroadcastTables());
148         ShardingStrategyConfiguration defaultDatabaseShardingStrategyConfig = null == checkedConfig.getDefaultDatabaseShardingStrategy()
149                 ? new NoneShardingStrategyConfiguration()
150                 : checkedConfig.getDefaultDatabaseShardingStrategy();
151         ShardingStrategyConfiguration defaultTableShardingStrategyConfig = null == checkedConfig.getDefaultTableShardingStrategy()
152                 ? new NoneShardingStrategyConfiguration()
153                 : checkedConfig.getDefaultTableShardingStrategy();
154         return isValidBindingTableConfiguration(tableRules, new BindingTableCheckedConfiguration(allDataSourceNames, shardingAlgorithms, checkedConfig.getBindingTableGroups(), broadcastTables,
155                 defaultDatabaseShardingStrategyConfig, defaultTableShardingStrategyConfig, checkedConfig.getDefaultShardingColumn()));
156     }

158     private static void checkDataSources(final String databaseName, final Collection<AbstractTableRuleSegment> rules, final ShardingSphereDatabase database) {
159         Collection<String> requiredDataSource = getRequiredDataSources(rules);
160         Collection<String> notExistedDataSources = database.getResourceMetaData().getNotExistedDataSources(requiredDataSource);
161         Collection<String> logicDataSources = getLogicDataSources(database);
162         notExistedDataSources.removeIf(logicDataSources::contains);
163         ShardingSpherePreconditions.checkState(notExistedDataSources.isEmpty(), () -> new MissingRequiredStorageUnitsException(databaseName, notExistedDataSources));
164     }
165     
166     private static Collection<String> getLogicDataSources(final ShardingSphereDatabase database) {
167         return database.getRuleMetaData().findRules(DataSourceContainedRule.class).stream()
168                 .map(each -> each.getDataSourceMapper().keySet()).flatMap(Collection::stream).collect(Collectors.toCollection(LinkedHashSet::new));
169     }

171     private static Collection<String> getRequiredDataSources(final Collection<AbstractTableRuleSegment> rules) {
172         return rules.stream().map(AbstractTableRuleSegment::getDataSourceNodes).flatMap(Collection::stream)
173                 .map(ShardingTableRuleStatementChecker::parseDateSource).map(ShardingTableRuleStatementChecker::getDataSourceNames).flatMap(Collection::stream).collect(Collectors.toList());
174     }
175     
176     private static Collection<String> parseDateSource(final String dateSource) {
177         return new InlineExpressionParser(dateSource).splitAndEvaluate();
178     }

180     private static Collection<String> getDataSourceNames(final Collection<String> actualDataNodes) {
181         Collection<String> result = new HashSet<>();
182         for (String each : actualDataNodes) {
183             result.add(isValidDataNode(each) ? new DataNode(each).getDataSourceName() : each);
184         }
185         return result;
186     }
187     
188     private static Collection<String> getDataSourceNames(final Collection<ShardingTableRuleConfiguration> tableRuleConfigs,
189                                                          final Collection<ShardingAutoTableRuleConfiguration> autoTableRuleConfigs, final Collection<String> dataSourceNames) {
190         if (tableRuleConfigs.isEmpty() && autoTableRuleConfigs.isEmpty()) {
191             return dataSourceNames;
192         }
193         if (tableRuleConfigs.stream().map(ShardingTableRuleConfiguration::getActualDataNodes).anyMatch(each -> null == each || each.isEmpty())) {
194             return dataSourceNames;
195         }
196         Collection<String> result = new LinkedHashSet<>();
197         tableRuleConfigs.forEach(each -> result.addAll(getDataSourceNames(each)));
198         autoTableRuleConfigs.forEach(each -> result.addAll(new InlineExpressionParser(each.getActualDataSources()).splitAndEvaluate()));
199         return result;
200     }

202     private static Collection<String> getDataSourceNames(final ShardingTableRuleConfiguration shardingTableRuleConfig) {
203         return new InlineExpressionParser(shardingTableRuleConfig.getActualDataNodes()).splitAndEvaluate().stream().map(each -> new DataNode(each).getDataSourceName()).collect(Collectors.toList());
204     }

206     private static boolean isValidDataNode(final String dataNodeStr) {
207         return dataNodeStr.contains(DELIMITER) && 2 == Splitter.on(DELIMITER).omitEmptyStrings().splitToList(dataNodeStr).size();
208     }

210     private static void checkTables(final String databaseName, final Collection<AbstractTableRuleSegment> rules, final ShardingRuleConfiguration currentRuleConfig, final boolean isCreate,
211                                     final boolean ifNotExists) {
212         Collection<String> requiredTables = rules.stream().map(AbstractTableRuleSegment::getLogicTable).collect(Collectors.toList());
213         Collection<String> duplicatedRuleNames = getDuplicatedRuleNames(requiredTables);
214         ShardingSpherePreconditions.checkState(duplicatedRuleNames.isEmpty(), () -> new DuplicateRuleException("sharding", databaseName, duplicatedRuleNames));
215         Collection<String> currentShardingTables = null == currentRuleConfig ? Collections.emptyList() : getCurrentShardingTables(currentRuleConfig);
216         if (isCreate) {
217             if (!ifNotExists) {
218                 duplicatedRuleNames.addAll(getDuplicatedRuleNames(requiredTables, currentShardingTables));
219                 ShardingSpherePreconditions.checkState(duplicatedRuleNames.isEmpty(), () -> new DuplicateRuleException("sharding", databaseName, duplicatedRuleNames));
220             }
221         } else {
222             Collection<String> notExistsRules = getNotExistsRules(requiredTables, currentShardingTables);
223             ShardingSpherePreconditions.checkState(notExistsRules.isEmpty(), () -> new MissingRequiredRuleException("sharding", databaseName, notExistsRules));
224         }
225     }
226     
227     private static Collection<String> getDuplicatedRuleNames(final Collection<String> collection) {
228         Collection<String> duplicate = collection.stream().collect(Collectors.groupingBy(String::toLowerCase, Collectors.counting())).entrySet().stream()
229                 .filter(each -> each.getValue() > 1).map(Entry::getKey).collect(Collectors.toSet());
230         return collection.stream().filter(each -> containsIgnoreCase(duplicate, each)).collect(Collectors.toSet());
231     }
232     
233     private static Collection<String> getDuplicatedRuleNames(final Collection<String> require, final Collection<String> current) {
234         return require.stream().filter(each -> containsIgnoreCase(current, each)).collect(Collectors.toSet());
235     }
236     
237     private static Set<String> getNotExistsRules(final Collection<String> require, final Collection<String> current) {
238         return require.stream().filter(each -> !containsIgnoreCase(current, each)).collect(Collectors.toSet());
239     }
240     
241     private static boolean containsIgnoreCase(final Collection<String> collection, final String str) {
242         return collection.stream().anyMatch(each -> each.equalsIgnoreCase(str));
243     }

245     private static Collection<String> getCurrentShardingTables(final ShardingRuleConfiguration currentRuleConfig) {
246         Collection<String> result = new LinkedList<>();
247         result.addAll(currentRuleConfig.getTables().stream().map(ShardingTableRuleConfiguration::getLogicTable).collect(Collectors.toSet()));
248         result.addAll(currentRuleConfig.getAutoTables().stream().map(ShardingAutoTableRuleConfiguration::getLogicTable).collect(Collectors.toSet()));
249         return result;

252     private static void checkKeyGenerators(final Collection<AbstractTableRuleSegment> rules, final ShardingRuleConfiguration currentRuleConfig) {
253         Set<String> notExistKeyGenerator = new LinkedHashSet<>(rules.size());
254         Set<AlgorithmSegment> requiredKeyGenerators = new LinkedHashSet<>(rules.size());
255         rules.stream().map(AbstractTableRuleSegment::getKeyGenerateStrategySegment).filter(Objects::nonNull)
256                 .peek(each -> each.getKeyGenerateAlgorithmName()
257                         .filter(optional -> null == currentRuleConfig || !currentRuleConfig.getKeyGenerators().containsKey(optional)).ifPresent(notExistKeyGenerator::add))
258                 .filter(each -> !each.getKeyGenerateAlgorithmName().isPresent()).filter(each -> Objects.nonNull(each.getKeyGenerateAlgorithmSegment()))
259                 .forEach(each -> requiredKeyGenerators.add(each.getKeyGenerateAlgorithmSegment()));
260         ShardingSpherePreconditions.checkState(notExistKeyGenerator.isEmpty(), () -> new MissingRequiredAlgorithmException("key generator", notExistKeyGenerator));
261         requiredKeyGenerators.forEach(each -> TypedSPILoader.checkService(KeyGenerateAlgorithm.class, each.getName(), each.getProps()));
262     }

264     private static void checkAuditors(final Collection<AbstractTableRuleSegment> rules) {
265         Collection<AuditStrategySegment> auditStrategySegments = rules.stream().map(AbstractTableRuleSegment::getAuditStrategySegment).filter(Objects::nonNull).collect(Collectors.toList());
266         Set<AlgorithmSegment> requiredAuditors = new LinkedHashSet<>();
267         for (AuditStrategySegment each : auditStrategySegments) {
268             requiredAuditors.addAll(each.getAuditorSegments().stream()
269                     .map(ShardingAuditorSegment::getAlgorithmSegment).collect(Collectors.toList()));
270         }
271         requiredAuditors.forEach(each -> TypedSPILoader.checkService(ShardingAuditAlgorithm.class, each.getName(), each.getProps()));
272     }

274     private static void checkAutoTableRule(final Collection<AbstractTableRuleSegment> rules) {
275         Collection<AutoTableRuleSegment> autoTableRules = rules.stream().map(each -> (AutoTableRuleSegment) each).collect(Collectors.toList());
276         Optional<AlgorithmSegment> anyAutoTableRule = autoTableRules.stream().map(AutoTableRuleSegment::getShardingAlgorithmSegment).filter(Objects::nonNull).findAny();
277         if (anyAutoTableRule.isPresent()) {
278             checkAutoTableShardingAlgorithms(autoTableRules);
279         }
280     }

282     private static void checkAutoTableShardingAlgorithms(final Collection<AutoTableRuleSegment> rules) {
283         rules.forEach(each -> {
284             ShardingSpherePreconditions.checkState(TypedSPILoader.findService(
285                     ShardingAlgorithm.class, each.getShardingAlgorithmSegment().getName(), each.getShardingAlgorithmSegment().getProps()).isPresent(),
286                     () -> new InvalidAlgorithmConfigurationException("sharding", each.getShardingAlgorithmSegment().getName()));
287             ShardingAlgorithm shardingAlgorithm = TypedSPILoader.getService(ShardingAlgorithm.class, each.getShardingAlgorithmSegment().getName(), each.getShardingAlgorithmSegment().getProps());
288             ShardingSpherePreconditions.checkState(shardingAlgorithm instanceof ShardingAutoTableAlgorithm, () -> new InvalidAlgorithmConfigurationException("sharding", shardingAlgorithm.getType(),
289                     String.format("auto sharding algorithm is required for rule `%s`", each.getLogicTable())));
290         });
291     }

293     private static void checkTableRule(final String databaseName, final Collection<AbstractTableRuleSegment> rules) {
294         Collection<TableRuleSegment> tableRules = rules.stream().map(each -> (TableRuleSegment) each).collect(Collectors.toList());
295         Optional<ShardingStrategySegment> anyTableRule = tableRules.stream().map(each -> Arrays.asList(each.getDatabaseStrategySegment(), each.getTableStrategySegment()))
296                 .flatMap(Collection::stream).filter(Objects::nonNull).findAny();
297         if (anyTableRule.isPresent()) {
298             checkStrategy(databaseName, tableRules);
299         }
300     }

302     private static void checkStrategy(final String databaseName, final Collection<TableRuleSegment> rules) {
303         for (TableRuleSegment each : rules) {
304             Optional<ShardingStrategySegment> databaseStrategySegment = Optional.ofNullable(each.getDatabaseStrategySegment());
305             if (databaseStrategySegment.isPresent() && !databaseStrategySegment.get().getType().equalsIgnoreCase("none")) {
306                 AlgorithmSegment databaseShardingAlgorithm = databaseStrategySegment.get().getShardingAlgorithm();
307                 if (null != databaseShardingAlgorithm) {
308                     ShardingAlgorithm shardingAlgorithm = TypedSPILoader.getService(ShardingAlgorithm.class, databaseShardingAlgorithm.getName(), databaseShardingAlgorithm.getProps());
309                     ShardingSpherePreconditions.checkState(!(shardingAlgorithm instanceof ShardingAutoTableAlgorithm),
310                             () -> new InvalidAlgorithmConfigurationException("sharding", shardingAlgorithm.getType(),
311                                     String.format("auto sharding algorithm cannot be used to create a table in Table mode `%s`", each.getLogicTable())));
312                 }
313                 ShardingSpherePreconditions.checkState(!isInvalidStrategy(each.getDatabaseStrategySegment()),
314                         () -> new InvalidAlgorithmConfigurationException(databaseName, null == databaseShardingAlgorithm ? null : databaseShardingAlgorithm.getName()));
315             }
316             Optional<ShardingStrategySegment> tableStrategySegment = Optional.ofNullable(each.getTableStrategySegment());
317             if (tableStrategySegment.isPresent() && !tableStrategySegment.get().getType().equalsIgnoreCase("none")) {
318                 AlgorithmSegment tableShardingAlgorithm = tableStrategySegment.get().getShardingAlgorithm();
319                 if (null != tableShardingAlgorithm) {
320                     ShardingAlgorithm shardingAlgorithm = TypedSPILoader.getService(ShardingAlgorithm.class, tableShardingAlgorithm.getName(), tableShardingAlgorithm.getProps());
321                     ShardingSpherePreconditions.checkState(!(shardingAlgorithm instanceof ShardingAutoTableAlgorithm),
322                             () -> new InvalidAlgorithmConfigurationException("sharding", shardingAlgorithm.getType(),
323                                     String.format("auto sharding algorithm cannot be used to create a table in Table mode `%s`", each.getLogicTable())));
324                 }
325                 ShardingSpherePreconditions.checkState(!isInvalidStrategy(each.getTableStrategySegment()),
326                         () -> new InvalidAlgorithmConfigurationException(databaseName, null == tableShardingAlgorithm ? null : tableShardingAlgorithm.getName()));
327             }
328         }
329     }

332         return !ShardingStrategyType.getValueOf(shardingStrategySegment.getType()).isValid(shardingStrategySegment.getShardingColumn()) || null == shardingStrategySegment.getShardingAlgorithm();

339     private static void checkBindingTableRules(final Collection<AbstractTableRuleSegment> rules, final ShardingRuleConfiguration currentRuleConfig) {
340         if (null == currentRuleConfig || currentRuleConfig.getBindingTableGroups().isEmpty()) {
341             return;
342         }
343         Collection<String> bindingTables = getCurrentBindingTables(currentRuleConfig);
344         if (bindingTables.size() <= 1) {
345             return;
346         }
347         ShardingRuleConfiguration toBeAlteredRuleConfig = ShardingTableRuleStatementConverter.convert(rules);
348         Collection<String> toBeAlteredLogicTableNames = getAlteredLogicalTableNames(toBeAlteredRuleConfig);
349         Collection<String> toBeAlteredBindingTableNames = toBeAlteredLogicTableNames.stream().filter(bindingTables::contains).collect(Collectors.toSet());
350         if (toBeAlteredBindingTableNames.isEmpty()) {
351             return;
352         }
353         ShardingRuleConfiguration toBeCheckedRuleConfig = createToBeCheckedShardingRuleConfiguration(currentRuleConfig);
354         removeRuleConfiguration(toBeCheckedRuleConfig, toBeAlteredRuleConfig);
355         addRuleConfiguration(toBeCheckedRuleConfig, toBeAlteredRuleConfig);
356         Collection<String> dataSourceNames = getRequiredDataSource(toBeCheckedRuleConfig);
357         dataSourceNames.addAll(getRequiredDataSource(toBeAlteredRuleConfig));
358         ShardingSpherePreconditions.checkState(check(toBeCheckedRuleConfig, dataSourceNames),
359                 () -> new InvalidRuleConfigurationException("sharding table", toBeAlteredLogicTableNames, Collections.singleton("invalid binding table configuration.")));
360     }

362     private static ShardingRuleConfiguration createToBeCheckedShardingRuleConfiguration(final ShardingRuleConfiguration currentRuleConfig) {
363         ShardingRuleConfiguration result = new ShardingRuleConfiguration();
364         result.setTables(new LinkedList<>(currentRuleConfig.getTables()));
365         result.setAutoTables(new LinkedList<>(currentRuleConfig.getAutoTables()));
366         result.setBindingTableGroups(new LinkedList<>(currentRuleConfig.getBindingTableGroups()));
367         result.setBroadcastTables(new LinkedList<>(currentRuleConfig.getBroadcastTables()));
368         result.setDefaultTableShardingStrategy(currentRuleConfig.getDefaultTableShardingStrategy());
369         result.setDefaultDatabaseShardingStrategy(currentRuleConfig.getDefaultDatabaseShardingStrategy());
370         result.setDefaultKeyGenerateStrategy(currentRuleConfig.getDefaultKeyGenerateStrategy());
371         result.setDefaultShardingColumn(currentRuleConfig.getDefaultShardingColumn());
372         result.setShardingAlgorithms(new LinkedHashMap<>(currentRuleConfig.getShardingAlgorithms()));
373         result.setKeyGenerators(new LinkedHashMap<>(currentRuleConfig.getKeyGenerators()));
374         result.setAuditors(new LinkedHashMap<>(currentRuleConfig.getAuditors()));
375         return result;
376     }

378     private static Collection<String> getRequiredDataSource(final ShardingRuleConfiguration config) {
379         Collection<String> result = new LinkedHashSet<>();
380         result.addAll(config.getAutoTables().stream().map(ShardingAutoTableRuleConfiguration::getActualDataSources)
381                 .map(each -> Splitter.on(",").trimResults().splitToList(each)).flatMap(Collection::stream).collect(Collectors.toSet()));
382         result.addAll(config.getTables().stream().map(each -> new InlineExpressionParser(each.getActualDataNodes()).splitAndEvaluate())
383                 .flatMap(Collection::stream).distinct().map(each -> new DataNode(each).getDataSourceName()).collect(Collectors.toSet()));
384         return result;
385     }

387     private static boolean isValidBindingTableConfiguration(final Map<String, TableRule> tableRules, final BindingTableCheckedConfiguration checkedConfig) {
388         for (ShardingTableReferenceRuleConfiguration each : checkedConfig.getBindingTableGroups()) {
389             Collection<String> bindingTables = Splitter.on(",").trimResults().splitToList(each.getReference().toLowerCase());
390             if (bindingTables.size() <= 1) {
391                 return false;
392             }
393             Iterator<String> iterator = bindingTables.iterator();
394             TableRule sampleTableRule = getTableRule(iterator.next(), checkedConfig.getDataSourceNames(), tableRules, checkedConfig.getBroadcastTables());
395             while (iterator.hasNext()) {
396                 TableRule tableRule = getTableRule(iterator.next(), checkedConfig.getDataSourceNames(), tableRules, checkedConfig.getBroadcastTables());
397                 if (!isValidActualDataSourceName(sampleTableRule, tableRule) || !isValidActualTableName(sampleTableRule, tableRule)) {
398                     return false;
399                 }
400                 if (!isValidShardingAlgorithm(sampleTableRule, tableRule, true, checkedConfig) || !isValidShardingAlgorithm(sampleTableRule, tableRule, false, checkedConfig)) {
401                     return false;
402                 }
403             }
404         }
405         return true;
406     }

408     private static boolean isValidActualDataSourceName(final TableRule sampleTableRule, final TableRule tableRule) {
409         return sampleTableRule.getActualDataSourceNames().equals(tableRule.getActualDataSourceNames());
410     }
411     
412     private static boolean isValidActualTableName(final TableRule sampleTableRule, final TableRule tableRule) {
413         for (String each : sampleTableRule.getActualDataSourceNames()) {
414             Collection<String> sampleActualTableNames =
415                     sampleTableRule.getActualTableNames(each).stream().map(actualTableName -> actualTableName.replace(sampleTableRule.getTableDataNode().getPrefix(), "")).collect(Collectors.toSet());
416             Collection<String> actualTableNames =
417                     tableRule.getActualTableNames(each).stream().map(optional -> optional.replace(tableRule.getTableDataNode().getPrefix(), "")).collect(Collectors.toSet());
418             if (!sampleActualTableNames.equals(actualTableNames)) {
419                 return false;
420             }
421         }
422         return true;
423     }

425     private static boolean isValidShardingAlgorithm(final TableRule sampleTableRule, final TableRule tableRule, final boolean databaseAlgorithm, final BindingTableCheckedConfiguration checkedConfig) {
426         return getAlgorithmExpression(sampleTableRule, databaseAlgorithm, checkedConfig).equals(getAlgorithmExpression(tableRule, databaseAlgorithm, checkedConfig));
427     }

429     private static Optional<String> getAlgorithmExpression(final TableRule tableRule, final boolean databaseAlgorithm, final BindingTableCheckedConfiguration checkedConfig) {
430         ShardingStrategyConfiguration shardingStrategyConfig = databaseAlgorithm
431                 ? null == tableRule.getDatabaseShardingStrategyConfig() ? checkedConfig.getDefaultDatabaseShardingStrategyConfig() : tableRule.getDatabaseShardingStrategyConfig()
432                 : null == tableRule.getTableShardingStrategyConfig() ? checkedConfig.getDefaultTableShardingStrategyConfig() : tableRule.getTableShardingStrategyConfig();
433         ShardingAlgorithm shardingAlgorithm = checkedConfig.getShardingAlgorithms().get(shardingStrategyConfig.getShardingAlgorithmName());
434         String dataNodePrefix = databaseAlgorithm ? tableRule.getDataSourceDataNode().getPrefix() : tableRule.getTableDataNode().getPrefix();
435         String shardingColumn = getShardingColumn(shardingStrategyConfig, checkedConfig.getDefaultShardingColumn());
436         return null == shardingAlgorithm ? Optional.empty() : shardingAlgorithm.getAlgorithmStructure(dataNodePrefix, shardingColumn);
437     }

439     private static String getShardingColumn(final ShardingStrategyConfiguration shardingStrategyConfig, final String defaultShardingColumn) {
440         String shardingColumn = defaultShardingColumn;
441         if (shardingStrategyConfig instanceof ComplexShardingStrategyConfiguration) {
442             shardingColumn = ((ComplexShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumns();
443         }
444         if (shardingStrategyConfig instanceof StandardShardingStrategyConfiguration) {
445             shardingColumn = ((StandardShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumn();
446         }
447         return null == shardingColumn ? "" : shardingColumn;
448     }

450     private static TableRule getTableRule(final String logicTableName, final Collection<String> dataSourceNames, final Map<String, TableRule> tableRules, final Collection<String> broadcastTables) {
451         TableRule result = tableRules.get(logicTableName);
452         if (null != result) {
453             return result;
454         }
455         if (broadcastTables.contains(logicTableName)) {
456             return new TableRule(dataSourceNames, logicTableName);
457         }
458         throw new ShardingTableRuleNotFoundException(Collections.singleton(logicTableName));
459     }

473     private static TableRule createAutoTableRule(final KeyGenerateStrategyConfiguration defaultKeyGenerateStrategyConfig, final ShardingAutoTableRuleConfiguration autoTableRuleConfig,
474                                                  final Map<String, ShardingAlgorithm> shardingAlgorithms, final Collection<String> dataSourceNames) {
475         ShardingAlgorithm shardingAlgorithm = shardingAlgorithms.get(autoTableRuleConfig.getShardingStrategy().getShardingAlgorithmName());
476         ShardingSpherePreconditions.checkState(shardingAlgorithm instanceof ShardingAutoTableAlgorithm,
477                 () -> new ShardingAlgorithmClassImplementationException(autoTableRuleConfig.getShardingStrategy().getShardingAlgorithmName(), ShardingAutoTableAlgorithm.class));
478         return new TableRule(autoTableRuleConfig, dataSourceNames, (ShardingAutoTableAlgorithm) shardingAlgorithm, getDefaultGenerateKeyColumn(defaultKeyGenerateStrategyConfig));
479     }

491     private static void removeRuleConfiguration(final ShardingRuleConfiguration currentRuleConfig, final ShardingRuleConfiguration toBeAlteredRuleConfig) {
492         Collection<String> toBeAlteredLogicTableNames = getAlteredLogicalTableNames(toBeAlteredRuleConfig);
493         toBeAlteredLogicTableNames.forEach(each -> {
494             currentRuleConfig.getTables().removeIf(table -> table.getLogicTable().equalsIgnoreCase(each));
495             currentRuleConfig.getAutoTables().removeIf(table -> table.getLogicTable().equalsIgnoreCase(each));
496         });
497     }
498     
499     private static void addRuleConfiguration(final ShardingRuleConfiguration currentRuleConfig, final ShardingRuleConfiguration toBeAlteredRuleConfig) {
500         currentRuleConfig.getTables().addAll(toBeAlteredRuleConfig.getTables());
501         currentRuleConfig.getAutoTables().addAll(toBeAlteredRuleConfig.getAutoTables());
502         currentRuleConfig.getShardingAlgorithms().putAll(toBeAlteredRuleConfig.getShardingAlgorithms());
503         currentRuleConfig.getKeyGenerators().putAll(toBeAlteredRuleConfig.getKeyGenerators());
504     }

506     private static Collection<String> getAlteredLogicalTableNames(final ShardingRuleConfiguration toBeAlteredRuleConfig) {
507         Collection<String> result = toBeAlteredRuleConfig.getTables().stream().map(ShardingTableRuleConfiguration::getLogicTable).collect(Collectors.toList());
508         result.addAll(toBeAlteredRuleConfig.getAutoTables().stream().map(ShardingAutoTableRuleConfiguration::getLogicTable).collect(Collectors.toList()));
509         return result;
510     }

512     private static Collection<String> getCurrentBindingTables(final ShardingRuleConfiguration currentRuleConfig) {
513         Collection<String> result = new LinkedHashSet<>();
514         currentRuleConfig.getBindingTableGroups().forEach(each -> result.addAll(Splitter.on(",").trimResults().splitToList(each.getReference())));
515         return result;
