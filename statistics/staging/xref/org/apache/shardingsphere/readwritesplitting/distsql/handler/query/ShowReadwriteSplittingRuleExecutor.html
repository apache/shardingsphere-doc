42  /**
43   * Show readwrite-splitting rule executor.
44   */
45  @Setter
46  public final class ShowReadwriteSplittingRuleExecutor implements DistSQLQueryExecutor<ShowReadwriteSplittingRulesStatement>, DistSQLExecutorRuleAware<ReadwriteSplittingRule> {
47      
48      private ReadwriteSplittingRule rule;
49      
50      @Override
51      public Collection<String> getColumnNames(final ShowReadwriteSplittingRulesStatement sqlStatement) {
52          return Arrays.asList("name", "write_storage_unit_name", "read_storage_unit_names", "transactional_read_query_strategy", "load_balancer_type", "load_balancer_props");
53      }
54      
55      @Override
56      public Collection<LocalDataQueryResultRow> getRows(final ShowReadwriteSplittingRulesStatement sqlStatement, final ContextManager contextManager) {
57          Collection<LocalDataQueryResultRow> result = new LinkedList<>();
58          Map<String, Map<String, String>> exportableDataSourceMap = getExportableDataSourceMap(rule);
59          ReadwriteSplittingRuleConfiguration ruleConfig = rule.getConfiguration();
60          ruleConfig.getDataSourceGroups().forEach(each -> {
61              LocalDataQueryResultRow dataItem = buildDataItem(exportableDataSourceMap, each, getLoadBalancers(ruleConfig));
62              if (null == sqlStatement.getRuleName() || sqlStatement.getRuleName().equalsIgnoreCase(each.getName())) {
63                  result.add(dataItem);
64              }
65          });
66          return result;
67      }
68      
69      @SuppressWarnings("unchecked")
70      private Map<String, Map<String, String>> getExportableDataSourceMap(final ReadwriteSplittingRule rule) {
71          return (Map<String, Map<String, String>>) rule.getAttributes().getAttribute(ExportableRuleAttribute.class).getExportData().get(ExportableConstants.EXPORT_STATIC_READWRITE_SPLITTING_RULE);
72      }
73      
74      private LocalDataQueryResultRow buildDataItem(final Map<String, Map<String, String>> exportableDataSourceMap,
75                                                    final ReadwriteSplittingDataSourceGroupRuleConfiguration dataSourceGroupRuleConfig, final Map<String, AlgorithmConfiguration> loadBalancers) {
76          String name = dataSourceGroupRuleConfig.getName();
77          Map<String, String> exportDataSources = exportableDataSourceMap.get(name);
78          Optional<AlgorithmConfiguration> loadBalancer = Optional.ofNullable(loadBalancers.get(dataSourceGroupRuleConfig.getLoadBalancerName()));
79          return new LocalDataQueryResultRow(name,
80                  getWriteDataSourceName(dataSourceGroupRuleConfig, exportDataSources),
81                  getReadDataSourceNames(dataSourceGroupRuleConfig, exportDataSources),
82                  dataSourceGroupRuleConfig.getTransactionalReadQueryStrategy().name(),
83                  loadBalancer.map(AlgorithmConfiguration::getType).orElse(null),
84                  loadBalancer.map(AlgorithmConfiguration::getProps).orElse(null));
85      }
86      
87      private Map<String, AlgorithmConfiguration> getLoadBalancers(final ReadwriteSplittingRuleConfiguration ruleConfig) {
88          Map<String, AlgorithmConfiguration> loadBalancers = ruleConfig.getLoadBalancers();
89          return null == loadBalancers ? Collections.emptyMap() : loadBalancers;
90      }
91      
92      private String getWriteDataSourceName(final ReadwriteSplittingDataSourceGroupRuleConfiguration dataSourceGroupRuleConfig, final Map<String, String> exportDataSources) {
93          return null == exportDataSources ? dataSourceGroupRuleConfig.getWriteDataSourceName() : exportDataSources.get(ExportableItemConstants.PRIMARY_DATA_SOURCE_NAME);
94      }
95      
96      private String getReadDataSourceNames(final ReadwriteSplittingDataSourceGroupRuleConfiguration dataSourceGroupRuleConfig, final Map<String, String> exportDataSources) {
97          return null == exportDataSources ? Joiner.on(",").join(dataSourceGroupRuleConfig.getReadDataSourceNames()) : exportDataSources.get(ExportableItemConstants.REPLICA_DATA_SOURCE_NAMES);
98      }
99      
100     @Override
101     public Class<ReadwriteSplittingRule> getRuleClass() {
102         return ReadwriteSplittingRule.class;
103     }
104     
105     @Override
106     public Class<ShowReadwriteSplittingRulesStatement> getType() {
107         return ShowReadwriteSplittingRulesStatement.class;
108     }
109 }
