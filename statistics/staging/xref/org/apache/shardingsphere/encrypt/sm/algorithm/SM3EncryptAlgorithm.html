18  package org.apache.shardingsphere.encrypt.sm.algorithm;
19  
20  import org.apache.shardingsphere.encrypt.api.encrypt.standard.StandardEncryptAlgorithm;
21  import org.apache.shardingsphere.encrypt.exception.algorithm.EncryptAlgorithmInitializationException;
22  import org.apache.shardingsphere.encrypt.api.context.EncryptContext;
23  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
24  import org.bouncycastle.crypto.digests.SM3Digest;
25  import org.bouncycastle.jce.provider.BouncyCastleProvider;
26  import org.bouncycastle.pqc.math.linearalgebra.ByteUtils;
27  
28  import java.nio.charset.StandardCharsets;
29  import java.security.Security;
30  import java.util.Properties;
31  
32  /**
33   * SM3 encrypt algorithm.
34   */
35  public final class SM3EncryptAlgorithm implements StandardEncryptAlgorithm<Object, String> {
36      
37      static {
38          Security.addProvider(new BouncyCastleProvider());
39      }
40      
41      private static final String SM3_SALT = "sm3-salt";
42      
43      private static final int SALT_LENGTH = 8;
44      
45      private byte[] sm3Salt;
46      
47      @Override
48      public void init(final Properties props) {
49          sm3Salt = createSm3Salt(props);
50      }
51      
52      private byte[] createSm3Salt(final Properties props) {
53          String salt = null == props.getProperty(SM3_SALT) ? "" : String.valueOf(props.getProperty(SM3_SALT));
54          ShardingSpherePreconditions.checkState(salt.isEmpty() || SALT_LENGTH == salt.length(),
55                  () -> new EncryptAlgorithmInitializationException("SM3", "Salt should be either blank or better " + SALT_LENGTH + " bytes long."));
56          return salt.isEmpty() ? new byte[0] : salt.getBytes(StandardCharsets.UTF_8);
57      }
58      
59      @Override
60      public String encrypt(final Object plainValue, final EncryptContext encryptContext) {
61          return null == plainValue ? null : ByteUtils.toHexString(digest(String.valueOf(plainValue).getBytes(StandardCharsets.UTF_8), sm3Salt));
62      }
63      
64      @Override
65      public Object decrypt(final String cipherValue, final EncryptContext encryptContext) {
66          return cipherValue;
67      }
68      
69      private byte[] digest(final byte[] input, final byte[] salt) {
70          SM3Digest sm3Digest = new SM3Digest();
71          byte[] updateByte = concat(input, salt);
72          sm3Digest.update(updateByte, 0, updateByte.length);
73          byte[] result = new byte[sm3Digest.getDigestSize()];
74          sm3Digest.doFinal(result, 0);
75          return result;
76      }
77      
78      private byte[] concat(final byte[] input, final byte[] salt) {
79          int inputLength = input.length;
80          int saltLength = salt.length;
81          byte[] result = new byte[inputLength + saltLength];
82          System.arraycopy(input, 0, result, 0, inputLength);
83          System.arraycopy(salt, 0, result, inputLength, saltLength);
84          return result;
85      }
86      
87      @Override
88      public String getType() {
89          return "SM3";
90      }
91  }
