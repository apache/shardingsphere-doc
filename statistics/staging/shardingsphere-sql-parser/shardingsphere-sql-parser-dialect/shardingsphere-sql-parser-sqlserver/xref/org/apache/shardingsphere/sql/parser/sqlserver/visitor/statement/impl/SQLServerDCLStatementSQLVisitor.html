89      @Override
90      public ASTNode visitGrant(final GrantContext ctx) {
91          SQLServerGrantStatement result = new SQLServerGrantStatement();
92          if (null != ctx.grantClassPrivilegesClause()) {
93              for (SimpleTableSegment each : getTableFromGrantPrivilegeClause(ctx.grantClassPrivilegesClause())) {
94                  result.getTables().add(each);
95              }
96              if (null != ctx.grantClassPrivilegesClause().classPrivileges().columnNames()) {
97                  for (ColumnNamesContext each : ctx.grantClassPrivilegesClause().classPrivileges().columnNames()) {
98                      result.getColumns().addAll(((CollectionValue<ColumnSegment>) visit(each)).getValue());
99                  }
100             }
101         }
102         if (null != ctx.grantClassTypePrivilegesClause()) {
103             for (SimpleTableSegment each : getTableFromGrantTypePrivilegeClause(ctx.grantClassTypePrivilegesClause())) {
104                 result.getTables().add(each);
105             }
106         }
107         return result;
108     }
109     
110     private Collection<SimpleTableSegment> getTableFromGrantPrivilegeClause(final GrantClassPrivilegesClauseContext ctx) {
111         Collection<SimpleTableSegment> result = new LinkedList<>();
112         if (null != ctx.onClassClause()) {
113             if (null != ctx.onClassClause().classItem() && null != ctx.onClassClause().classItem().OBJECT()) {
114                 result.add((SimpleTableSegment) visit(ctx.onClassClause().securable()));
115             } else if (null != ctx.classPrivileges().privilegeType().get(0).objectPermission()) {
116                 result.add((SimpleTableSegment) visit(ctx.onClassClause().securable()));
117             } else if (null != ctx.classPrivileges().privilegeType().get(0).PRIVILEGES()) {
118                 result.add((SimpleTableSegment) visit(ctx.onClassClause().securable()));
119             }
120         }
121         return result;
122     }
123     
124     private Collection<SimpleTableSegment> getTableFromGrantTypePrivilegeClause(final GrantClassTypePrivilegesClauseContext ctx) {
125         Collection<SimpleTableSegment> result = new LinkedList<>();
126         if (null != ctx.onClassTypeClause() && null != ctx.onClassTypeClause().classType() && null != ctx.onClassTypeClause().classType().OBJECT()) {
127             result.add((SimpleTableSegment) visit(ctx.onClassTypeClause().securable()));
128         }
129         return result;
130     }

175     @Override
176     public ASTNode visitSecurable(final SecurableContext ctx) {
177         SimpleTableSegment result = new SimpleTableSegment(new TableNameSegment(ctx.name().getStart().getStartIndex(), ctx.name().getStop().getStopIndex(), (IdentifierValue) visit(ctx.name())));
178         OwnerContext owner = ctx.owner();
179         if (null != owner) {
180             result.setOwner(new OwnerSegment(owner.getStart().getStartIndex(), owner.getStop().getStopIndex(), (IdentifierValue) visit(owner.identifier())));
181         }
182         return result;
183     }

185     @Override
186     public ASTNode visitCreateUser(final CreateUserContext ctx) {
187         SQLServerCreateUserStatement result = new SQLServerCreateUserStatement();
188         if (null != ctx.createUserLoginClause()) {
189             result.getUsers().add((UserSegment) visit(ctx.createUserLoginClause().userName()));
190         } else if (null != ctx.createUserWindowsPrincipalClause()) {
191             result.getUsers().add((UserSegment) visit(ctx.createUserWindowsPrincipalClause()));
192         } else if (null != ctx.createUserLoginWindowsPrincipalClause()) {
193             result.getUsers().add((UserSegment) visit(ctx.createUserLoginWindowsPrincipalClause()));
194         } else if (null != ctx.createUserWithoutLoginClause()) {
195             result.getUsers().add((UserSegment) visit(ctx.createUserWithoutLoginClause().userName()));
196         } else if (null != ctx.createUserFromExternalProviderClause()) {
197             result.getUsers().add((UserSegment) visit(ctx.createUserFromExternalProviderClause().userName()));
198         } else if (null != ctx.createUserWithDefaultSchema()) {
199             result.getUsers().add((UserSegment) visit(ctx.createUserWithDefaultSchema().userName()));
200         } else if (null != ctx.createUserWithAzureActiveDirectoryPrincipalClause()) {
201             result.getUsers().add((UserSegment) visit(ctx.createUserWithAzureActiveDirectoryPrincipalClause().azureActiveDirectoryPrincipal().userName()));
202         } else {
203             result.getUsers().add((UserSegment) visit(ctx.userName()));
204         }
205         return result;
206     }

208     @Override
209     public ASTNode visitUserName(final UserNameContext ctx) {
210         UserSegment result = new UserSegment();
211         String user = ((IdentifierValue) visit(ctx.ignoredNameIdentifier())).getValue();
212         result.setUser(user);
213         return result;
214     }
215     
216     @Override
217     public ASTNode visitIgnoredNameIdentifier(final IgnoredNameIdentifierContext ctx) {
218         int identifierCount = ctx.identifier().size();
219         return 1 == identifierCount ? (IdentifierValue) visit(ctx.identifier(0)) : new IdentifierValue(ctx.getText());
220     }
221     
222     @Override
223     public ASTNode visitCreateUserWindowsPrincipalClause(final CreateUserWindowsPrincipalClauseContext ctx) {
224         if (null != ctx.windowsPrincipal()) {
225             return visit(ctx.windowsPrincipal().userName());
226         }
227         if (null != ctx.azureActiveDirectoryPrincipal()) {
228             return visit(ctx.azureActiveDirectoryPrincipal().userName());
229         }
230         return visit(ctx.userName());
231     }
232     
233     @Override
234     public ASTNode visitCreateUserLoginWindowsPrincipalClause(final CreateUserLoginWindowsPrincipalClauseContext ctx) {
235         return null != ctx.userName() ? (UserSegment) visit(ctx.userName()) : (UserSegment) visit(ctx.windowsPrincipal(0));
236     }

341     @Override
342     public ASTNode visitSetUser(final SetUserContext ctx) {
343         SQLServerSetUserStatement result = new SQLServerSetUserStatement();
344         if (null != ctx.stringLiterals()) {
345             UserSegment userSegment = new UserSegment();
346             userSegment.setUser(((StringLiteralValue) visit(ctx.stringLiterals())).getValue());
347             userSegment.setStartIndex(ctx.stringLiterals().start.getStartIndex());
348             userSegment.setStopIndex(ctx.stringLiterals().stop.getStopIndex());
349             result.setUser(userSegment);
350         }
351         return result;
352     }
