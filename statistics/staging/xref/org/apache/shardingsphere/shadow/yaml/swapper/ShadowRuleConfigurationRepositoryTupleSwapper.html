44  /**
45   * Shadow rule configuration repository tuple swapper.
46   */
47  public final class ShadowRuleConfigurationRepositoryTupleSwapper implements RepositoryTupleConfigurationSwapper<ShadowRuleConfiguration> {
48      
49      private final YamlShadowTableConfigurationSwapper tableSwapper = new YamlShadowTableConfigurationSwapper();
50      
51      private final YamlAlgorithmConfigurationSwapper algorithmSwapper = new YamlAlgorithmConfigurationSwapper();
52      
53      private final RuleNodePath shadowRuleNodePath = new ShadowRuleNodePathProvider().getRuleNodePath();
54      
55      @Override
56      public Collection<RepositoryTuple> swapToRepositoryTuples(final ShadowRuleConfiguration data) {
57          Collection<RepositoryTuple> result = new LinkedList<>();
58          for (Entry<String, AlgorithmConfiguration> entry : data.getShadowAlgorithms().entrySet()) {
59              result.add(new RepositoryTuple(shadowRuleNodePath.getNamedItem(ShadowRuleNodePathProvider.ALGORITHMS).getPath(entry.getKey()),
60                      YamlEngine.marshal(algorithmSwapper.swapToYamlConfiguration(entry.getValue()))));
61          }
62          if (!Strings.isNullOrEmpty(data.getDefaultShadowAlgorithmName())) {
63              result.add(new RepositoryTuple(shadowRuleNodePath.getUniqueItem(ShadowRuleNodePathProvider.DEFAULT_ALGORITHM).getPath(), data.getDefaultShadowAlgorithmName()));
64          }
65          for (ShadowDataSourceConfiguration each : data.getDataSources()) {
66              result.add(new RepositoryTuple(shadowRuleNodePath.getNamedItem(ShadowRuleNodePathProvider.DATA_SOURCES).getPath(each.getName()),
67                      YamlEngine.marshal(swapToDataSourceYamlConfiguration(each))));
68          }
69          for (Entry<String, ShadowTableConfiguration> entry : data.getTables().entrySet()) {
70              result.add(new RepositoryTuple(shadowRuleNodePath.getNamedItem(ShadowRuleNodePathProvider.TABLES).getPath(entry.getKey()),
71                      YamlEngine.marshal(tableSwapper.swapToYamlConfiguration(entry.getValue()))));
72          }
73          return result;
74      }
75      
76      private YamlShadowDataSourceConfiguration swapToDataSourceYamlConfiguration(final ShadowDataSourceConfiguration data) {
77          YamlShadowDataSourceConfiguration result = new YamlShadowDataSourceConfiguration();
78          result.setProductionDataSourceName(data.getProductionDataSourceName());
79          result.setShadowDataSourceName(data.getShadowDataSourceName());
80          return result;
81      }
82      
83      @Override
84      public Optional<ShadowRuleConfiguration> swapToObject(final Collection<RepositoryTuple> repositoryTuples) {
85          List<RepositoryTuple> validRepositoryTuples = repositoryTuples.stream().filter(each -> shadowRuleNodePath.getRoot().isValidatedPath(each.getKey())).collect(Collectors.toList());
86          if (validRepositoryTuples.isEmpty()) {
87              return Optional.empty();
88          }
89          ShadowRuleConfiguration result = new ShadowRuleConfiguration();
90          for (RepositoryTuple each : validRepositoryTuples) {
91              shadowRuleNodePath.getNamedItem(ShadowRuleNodePathProvider.DATA_SOURCES).getName(each.getKey())
92                      .ifPresent(optional -> result.getDataSources().add(swapDataSource(optional, YamlEngine.unmarshal(each.getValue(), YamlShadowDataSourceConfiguration.class))));
93              shadowRuleNodePath.getNamedItem(ShadowRuleNodePathProvider.TABLES).getName(each.getKey())
94                      .ifPresent(optional -> result.getTables().put(optional, tableSwapper.swapToObject(YamlEngine.unmarshal(each.getValue(), YamlShadowTableConfiguration.class))));
95              shadowRuleNodePath.getNamedItem(ShadowRuleNodePathProvider.ALGORITHMS).getName(each.getKey())
96                      .ifPresent(optional -> result.getShadowAlgorithms().put(optional, algorithmSwapper.swapToObject(YamlEngine.unmarshal(each.getValue(), YamlAlgorithmConfiguration.class))));
97              if (shadowRuleNodePath.getUniqueItem(ShadowRuleNodePathProvider.DEFAULT_ALGORITHM).isValidatedPath(each.getKey())) {
98                  result.setDefaultShadowAlgorithmName(each.getValue());
99              }
100         }
101         return Optional.of(result);
102     }
103     
104     private ShadowDataSourceConfiguration swapDataSource(final String name, final YamlShadowDataSourceConfiguration yamlConfig) {
105         return new ShadowDataSourceConfiguration(name, yamlConfig.getProductionDataSourceName(), yamlConfig.getShadowDataSourceName());
106     }
107     
108     @Override
109     public Class<ShadowRuleConfiguration> getTypeClass() {
110         return ShadowRuleConfiguration.class;
111     }
112     
113     @Override
114     public String getRuleTagName() {
115         return "SHADOW";
116     }
117     
118     @Override
119     public int getOrder() {
120         return ShadowOrder.ORDER;
121     }
122 }
