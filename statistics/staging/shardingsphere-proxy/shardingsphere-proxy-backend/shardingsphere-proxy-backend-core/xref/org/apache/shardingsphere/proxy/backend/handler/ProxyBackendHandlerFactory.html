68  /**
69   * Proxy backend handler factory.
70   */
71  @NoArgsConstructor(access = AccessLevel.PRIVATE)
72  public final class ProxyBackendHandlerFactory {
73      
74      /**
75       * Create new instance of backend handler.
76       *
77       * @param databaseType database type
78       * @param sql SQL to be executed
79       * @param connectionSession connection session
80       * @return created instance
81       * @throws SQLException SQL exception
82       */
83      public static ProxyBackendHandler newInstance(final DatabaseType databaseType, final String sql, final ConnectionSession connectionSession) throws SQLException {
84          if (Strings.isNullOrEmpty(SQLUtils.trimComment(sql))) {
85              return new SkipBackendHandler(new EmptyStatement());
86          }
87          SQLParserRule sqlParserRule = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getGlobalRuleMetaData().getSingleRule(SQLParserRule.class);
88          SQLStatement sqlStatement = sqlParserRule.getSQLParserEngine(getProtocolType(databaseType, connectionSession).getType()).parse(sql, false);
89          return newInstance(databaseType, sql, sqlStatement, connectionSession, new HintValueContext());
90      }
91      
92      /**
93       * Create new instance of backend handler.
94       *
95       * @param databaseType database type
96       * @param sql SQL to be executed
97       * @param sqlStatement SQL statement
98       * @param connectionSession connection session
99       * @param hintValueContext hint query context
100      * @return created instance
101      * @throws SQLException SQL exception
102      */
103     public static ProxyBackendHandler newInstance(final DatabaseType databaseType, final String sql, final SQLStatement sqlStatement,
104                                                   final ConnectionSession connectionSession, final HintValueContext hintValueContext) throws SQLException {
105         if (sqlStatement instanceof EmptyStatement) {
106             return new SkipBackendHandler(sqlStatement);
107         }
108         SQLStatementContext<?> sqlStatementContext = sqlStatement instanceof DistSQLStatement ? new DistSQLStatementContext((DistSQLStatement) sqlStatement)
109                 : SQLStatementContextFactory.newInstance(ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData(), sqlStatement, connectionSession.getDefaultDatabaseName());
110         QueryContext queryContext = new QueryContext(sqlStatementContext, sql, Collections.emptyList(), hintValueContext);
111         connectionSession.setQueryContext(queryContext);
112         return newInstance(databaseType, queryContext, connectionSession, false);
113     }
114     
115     /**
116      * Create new instance of backend handler.
117      *
118      * @param databaseType database type
119      * @param queryContext query context
120      * @param connectionSession connection session
121      * @param preferPreparedStatement use prepared statement as possible
122      * @return created instance
123      * @throws SQLException SQL exception
124      */
125     @SuppressWarnings("unchecked")
126     public static ProxyBackendHandler newInstance(final DatabaseType databaseType, final QueryContext queryContext, final ConnectionSession connectionSession,
127                                                   final boolean preferPreparedStatement) throws SQLException {
128         SQLStatementContext<?> sqlStatementContext = queryContext.getSqlStatementContext();
129         SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
130         databaseType.handleRollbackOnly(connectionSession.getTransactionStatus().isRollbackOnly(), sqlStatement);
131         checkUnsupportedSQLStatement(sqlStatement);
132         checkClusterState(sqlStatement);
133         if (sqlStatement instanceof EmptyStatement) {
134             return new SkipBackendHandler(sqlStatement);
135         }
136         if (sqlStatement instanceof DistSQLStatement) {
137             checkUnsupportedDistSQLStatementInTransaction(sqlStatement, connectionSession);
138             return DistSQLBackendHandlerFactory.newInstance((DistSQLStatement) sqlStatement, connectionSession);
139         }
140         String sql = queryContext.getSql();
141         handleAutoCommit(sqlStatement, connectionSession);
142         if (sqlStatement instanceof TCLStatement) {
143             return TransactionBackendHandlerFactory.newInstance((SQLStatementContext<TCLStatement>) sqlStatementContext, sql, connectionSession);
144         }
145         Optional<ProxyBackendHandler> backendHandler = DatabaseAdminBackendHandlerFactory.newInstance(databaseType, sqlStatementContext, connectionSession, sql);
146         if (backendHandler.isPresent()) {
147             return backendHandler.get();
148         }
149         Optional<ExtraProxyBackendHandler> extraHandler = findExtraProxyBackendHandler(sqlStatement);
150         if (extraHandler.isPresent()) {
151             return extraHandler.get();
152         }
153         Optional<ProxyBackendHandler> databaseOperateHandler = findDatabaseOperateBackendHandler(sqlStatement, connectionSession);
154         if (databaseOperateHandler.isPresent()) {
155             return databaseOperateHandler.get();
156         }
157         String databaseName = sqlStatementContext.getTablesContext().getDatabaseName().isPresent()
158                 ? sqlStatementContext.getTablesContext().getDatabaseName().get()
159                 : connectionSession.getDatabaseName();
160         AuthorityRule authorityRule = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getGlobalRuleMetaData().getSingleRule(AuthorityRule.class);
161         ShardingSphereDatabase database = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getDatabase(databaseName);
162         new AuthorityChecker(authorityRule, connectionSession.getGrantee()).checkPrivileges(databaseName, sqlStatementContext.getSqlStatement());
163         SQLAuditEngine.audit(sqlStatementContext, queryContext.getParameters(),
164                 ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getGlobalRuleMetaData(), database, connectionSession.getGrantee());
165         backendHandler = DatabaseAdminBackendHandlerFactory.newInstance(databaseType, sqlStatementContext, connectionSession);
166         return backendHandler.orElseGet(() -> DatabaseBackendHandlerFactory.newInstance(queryContext, connectionSession, preferPreparedStatement));
167     }
168     
169     private static void checkUnsupportedDistSQLStatementInTransaction(final SQLStatement sqlStatement, final ConnectionSession connectionSession) {
170         ShardingSpherePreconditions.checkState(!connectionSession.getTransactionStatus().isInTransaction() || isSupportedDistSQLStatementInTransaction(sqlStatement),
171                 () -> new UnsupportedSQLOperationException("Non-query DistSQL is not supported within a transaction"));
172     }
173     
174     private static boolean isSupportedDistSQLStatementInTransaction(final SQLStatement sqlStatement) {
175         return sqlStatement instanceof RQLStatement || sqlStatement instanceof QueryableRALStatement || sqlStatement instanceof RULStatement;
176     }
177     
178     private static DatabaseType getProtocolType(final DatabaseType defaultDatabaseType, final ConnectionSession connectionSession) {
179         String databaseName = connectionSession.getDatabaseName();
180         return Strings.isNullOrEmpty(databaseName) || !ProxyContext.getInstance().databaseExists(databaseName)
181                 ? defaultDatabaseType
182                 : ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getDatabase(databaseName).getProtocolType();
183     }
184     
185     private static void handleAutoCommit(final SQLStatement sqlStatement, final ConnectionSession connectionSession) {
186         if (AutoCommitUtils.needOpenTransaction(sqlStatement)) {
187             connectionSession.getBackendConnection().handleAutoCommit();
188         }
189     }
190     
191     private static Optional<ExtraProxyBackendHandler> findExtraProxyBackendHandler(final SQLStatement sqlStatement) {
192         for (ExtraProxyBackendHandler each : ShardingSphereServiceLoader.getServiceInstances(ExtraProxyBackendHandler.class)) {
193             if (each.accept(sqlStatement)) {
194                 return Optional.of(each);
195             }
196         }
197         return Optional.empty();
198     }
199     
200     private static Optional<ProxyBackendHandler> findDatabaseOperateBackendHandler(final SQLStatement sqlStatement, final ConnectionSession connectionSession) {
201         if (sqlStatement instanceof CreateDatabaseStatement || sqlStatement instanceof DropDatabaseStatement) {
202             return Optional.of(DatabaseOperateBackendHandlerFactory.newInstance(sqlStatement, connectionSession));
203         }
204         return Optional.empty();
205     }
206     
207     private static void checkUnsupportedSQLStatement(final SQLStatement sqlStatement) {
208         if (sqlStatement instanceof DCLStatement || sqlStatement instanceof FlushStatement || sqlStatement instanceof MySQLShowCreateUserStatement) {
209             throw new UnsupportedSQLOperationException("Unsupported operation");
210         }
211     }
212     
213     private static void checkClusterState(final SQLStatement sqlStatement) {
214         ClusterState clusterCurrentState = ProxyContext.getInstance().getContextManager().getClusterStateContext().getCurrentState();
215         if (ClusterState.OK != clusterCurrentState) {
216             TypedSPILoader.getService(ProxyClusterState.class, clusterCurrentState.name()).check(sqlStatement);
217         }
218     }
