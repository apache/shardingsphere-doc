40  import java.sql.SQLException;

42  /**
43   * PostgreSQL dialect exception mapper.
44   */
45  public final class PostgreSQLDialectExceptionMapper implements SQLDialectExceptionMapper {
46      
47      private static final String FATAL_SEVERITY = "FATAL";
48      
49      private static final String ERROR_SEVERITY = "ERROR";
50      
51      @Override
52      public SQLException convert(final SQLDialectException sqlDialectException) {
53          if (sqlDialectException instanceof UnknownDatabaseException) {
54              return new PostgreSQLException(new ServerErrorMessage(FATAL_SEVERITY, PostgreSQLVendorError.INVALID_CATALOG_NAME, ((UnknownDatabaseException) sqlDialectException).getDatabaseName()));
55          }
56          if (sqlDialectException instanceof DatabaseCreateExistsException) {
57              return new PostgreSQLException(new ServerErrorMessage(FATAL_SEVERITY, PostgreSQLVendorError.DUPLICATE_DATABASE, ((DatabaseCreateExistsException) sqlDialectException).getDatabaseName()));
58          }
59          if (sqlDialectException instanceof TableExistsException) {
60              return new PostgreSQLException(new ServerErrorMessage(ERROR_SEVERITY, PostgreSQLVendorError.DUPLICATE_TABLE, ((TableExistsException) sqlDialectException).getTableName()));
61          }
62          if (sqlDialectException instanceof InTransactionException) {
63              return new PostgreSQLException(new ServerErrorMessage(ERROR_SEVERITY, PostgreSQLVendorError.TRANSACTION_STATE_INVALID));
64          }
65          if (sqlDialectException instanceof InsertColumnsAndValuesMismatchedException) {
66              return new PostgreSQLException(new ServerErrorMessage(ERROR_SEVERITY,
67                      PostgreSQLVendorError.WRONG_VALUE_COUNT_ON_ROW, ((InsertColumnsAndValuesMismatchedException) sqlDialectException).getMismatchedRowNumber()));
68          }
69          if (sqlDialectException instanceof InvalidParameterValueException) {
70              InvalidParameterValueException cause = (InvalidParameterValueException) sqlDialectException;
71              return new PostgreSQLException(new ServerErrorMessage(ERROR_SEVERITY, PostgreSQLVendorError.INVALID_PARAMETER_VALUE, cause.getParameterName(), cause.getParameterValue()));
72          }
73          if (sqlDialectException instanceof TooManyConnectionsException) {
74              return new PostgreSQLException(new ServerErrorMessage(ERROR_SEVERITY, PostgreSQLVendorError.DATA_SOURCE_REJECTED_CONNECTION_ATTEMPT));
75          }
76          if (sqlDialectException instanceof UnknownUsernameException) {
77              return new PostgreSQLException(new ServerErrorMessage(
78                      FATAL_SEVERITY, PostgreSQLVendorError.INVALID_AUTHORIZATION_SPECIFICATION, ((UnknownUsernameException) sqlDialectException).getUsername()));
79          }
80          if (sqlDialectException instanceof InvalidPasswordException) {
81              return new PostgreSQLException(new ServerErrorMessage(FATAL_SEVERITY, PostgreSQLVendorError.INVALID_PASSWORD, ((InvalidPasswordException) sqlDialectException).getUsername()));
82          }
83          if (sqlDialectException instanceof PrivilegeNotGrantedException) {
84              PrivilegeNotGrantedException cause = (PrivilegeNotGrantedException) sqlDialectException;
85              return new PostgreSQLException(new ServerErrorMessage(FATAL_SEVERITY, PostgreSQLVendorError.PRIVILEGE_NOT_GRANTED, cause.getUsername(), cause.getDatabaseName()));
86          }
87          if (sqlDialectException instanceof EmptyUsernameException) {
88              return new PostgreSQLException(new ServerErrorMessage(FATAL_SEVERITY, PostgreSQLVendorError.NO_USERNAME));
89          }
90          if (sqlDialectException instanceof ProtocolViolationException) {
91              ProtocolViolationException cause = (ProtocolViolationException) sqlDialectException;
92              return new PostgreSQLException(new ServerErrorMessage(FATAL_SEVERITY, PostgreSQLVendorError.PROTOCOL_VIOLATION, cause.getExpectedMessageType(), cause.getActualMessageType()));
93          }
94          if (sqlDialectException instanceof ColumnNotFoundException) {
95              ColumnNotFoundException cause = (ColumnNotFoundException) sqlDialectException;
96              return new PostgreSQLException(new ServerErrorMessage(FATAL_SEVERITY, PostgreSQLVendorError.UNDEFINED_COLUMN, cause.getTableName(), cause.getColumnName()));
97          }
98          return new PostgreSQLException(sqlDialectException.getMessage(), PostgreSQLState.UNEXPECTED_ERROR.getValue());
99      }
100     
101     @Override
102     public String getDatabaseType() {
103         return "PostgreSQL";
104     }
105 }
