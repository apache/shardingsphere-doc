66  /**
67   * MySQL Connector.
68   */
69  @RequiredArgsConstructor
70  @Slf4j
71  public final class MySQLClient {
72      
73      private final ConnectInfo connectInfo;
74      
75      private EventLoopGroup eventLoopGroup;
76      
77      private Channel channel;
78      
79      private Promise<Object> responseCallback;
80      
81      private final ArrayBlockingQueue<List<AbstractBinlogEvent>> blockingEventQueue = new ArrayBlockingQueue<>(2500);
82      
83      private ServerInfo serverInfo;
84      
85      private volatile boolean running = true;
86      
87      private final AtomicInteger reconnectTimes = new AtomicInteger();
88      
89      private final boolean decodeWithTX;
90      
91      /**
92       * Connect to MySQL.
93       */
94      public synchronized void connect() {
95          eventLoopGroup = new NioEventLoopGroup(1);
96          responseCallback = new DefaultPromise<>(eventLoopGroup.next());
97          channel = new Bootstrap()
98                  .group(eventLoopGroup)
99                  .channel(NioSocketChannel.class)
100                 .option(ChannelOption.AUTO_READ, true)
101                 .option(ChannelOption.CONNECT_TIMEOUT_MILLIS, 5000)
102                 .handler(new ChannelInitializer<SocketChannel>() {
103                     
104                     @Override
105                     protected void initChannel(final SocketChannel socketChannel) {
106                         socketChannel.attr(MySQLConstants.MYSQL_SEQUENCE_ID).set(new AtomicInteger());
107                         socketChannel.pipeline().addLast(new ChannelAttrInitializer());
108                         socketChannel.pipeline().addLast(new PacketCodec(new MySQLPacketCodecEngine()));
109                         socketChannel.pipeline().addLast(new MySQLSequenceIdInboundHandler());
110                         socketChannel.pipeline().addLast(new MySQLNegotiatePackageDecoder());
111                         socketChannel.pipeline().addLast(new MySQLCommandPacketDecoder());
112                         socketChannel.pipeline().addLast(new MySQLNegotiateHandler(connectInfo.getUsername(), connectInfo.getPassword(), responseCallback));
113                         socketChannel.pipeline().addLast(new MySQLCommandResponseHandler());
114                     }
115                 }).connect(connectInfo.getHost(), connectInfo.getPort()).channel();
116         serverInfo = waitExpectedResponse(ServerInfo.class).orElse(null);
117         running = true;
118     }
119     
120     /**
121      * Execute command.
122      *
123      * @param queryString query string
124      * @return true if execute successfully, otherwise false
125      */
126     public synchronized boolean execute(final String queryString) {
127         responseCallback = new DefaultPromise<>(eventLoopGroup.next());
128         MySQLComQueryPacket comQueryPacket = new MySQLComQueryPacket(queryString);
129         resetSequenceID();
130         channel.writeAndFlush(comQueryPacket);
131         return waitExpectedResponse(MySQLOKPacket.class).isPresent();
132     }
133     
134     /**
135      * Execute update.
136      *
137      * @param queryString query string
138      * @return affected rows
139      * @throws PipelineInternalException if could not get MySQL OK packet
140      */
141     public synchronized int executeUpdate(final String queryString) {
142         responseCallback = new DefaultPromise<>(eventLoopGroup.next());
143         MySQLComQueryPacket comQueryPacket = new MySQLComQueryPacket(queryString);
144         resetSequenceID();
145         channel.writeAndFlush(comQueryPacket);
146         Optional<MySQLOKPacket> packet = waitExpectedResponse(MySQLOKPacket.class);
147         if (!packet.isPresent()) {
148             throw new PipelineInternalException("Could not get MySQL OK packet");
149         }
150         return (int) packet.get().getAffectedRows();
151     }
152     
153     /**
154      * Execute query.
155      *
156      * @param queryString query string
157      * @return result set
158      * @throws PipelineInternalException if getting MySQL packet failed
159      */
160     public synchronized InternalResultSet executeQuery(final String queryString) {
161         responseCallback = new DefaultPromise<>(eventLoopGroup.next());
162         MySQLComQueryPacket comQueryPacket = new MySQLComQueryPacket(queryString);
163         resetSequenceID();
164         channel.writeAndFlush(comQueryPacket);
165         Optional<InternalResultSet> result = waitExpectedResponse(InternalResultSet.class);
166         if (!result.isPresent()) {
167             throw new PipelineInternalException("Could not get MySQL FieldCount/ColumnDefinition/TextResultSetRow packet");
168         }
169         return result.get();
170     }
171     
172     /**
173      * Start dump binlog.
174      *
175      * @param binlogFileName binlog file name
176      * @param binlogPosition binlog position
177      */
178     public synchronized void subscribe(final String binlogFileName, final long binlogPosition) {
179         initDumpConnectSession();
180         registerSlave();
181         dumpBinlog(binlogFileName, binlogPosition, queryChecksumLength());
182         log.info("subscribe binlog file: {}, position: {}", binlogFileName, binlogPosition);
183     }
184     
185     private void initDumpConnectSession() {
186         if (serverInfo.getServerVersion().greaterThanOrEqualTo(5, 6, 0)) {
187             execute("SET @MASTER_BINLOG_CHECKSUM= @@GLOBAL.BINLOG_CHECKSUM");
188         }
189     }
190     
191     private void registerSlave() {
192         responseCallback = new DefaultPromise<>(eventLoopGroup.next());
193         InetSocketAddress localAddress = (InetSocketAddress) channel.localAddress();
194         MySQLComRegisterSlaveCommandPacket packet = new MySQLComRegisterSlaveCommandPacket(
195                 connectInfo.getServerId(), localAddress.getHostName(), connectInfo.getUsername(), connectInfo.getPassword(), localAddress.getPort());
196         resetSequenceID();
197         channel.writeAndFlush(packet);
198         waitExpectedResponse(MySQLOKPacket.class);
199     }
200     
201     private int queryChecksumLength() {
202         if (!serverInfo.getServerVersion().greaterThanOrEqualTo(5, 6, 0)) {
203             return 0;
204         }
205         InternalResultSet resultSet = executeQuery("SELECT @@GLOBAL.BINLOG_CHECKSUM");
206         String checksumType = resultSet.getFieldValues().get(0).getData().iterator().next().toString();
207         switch (checksumType.toUpperCase()) {
208             case "NONE":
209                 return 0;
210             case "CRC32":
211                 return 4;
212             default:
213                 throw new UnsupportedSQLOperationException(checksumType);
214         }
215     }
216     
217     private void dumpBinlog(final String binlogFileName, final long binlogPosition, final int checksumLength) {
218         responseCallback = null;
219         channel.pipeline().remove(MySQLCommandPacketDecoder.class);
220         channel.pipeline().remove(MySQLCommandResponseHandler.class);
221         String tableKey = String.join(":", connectInfo.getHost(), String.valueOf(connectInfo.getPort()));
222         channel.pipeline().addLast(new MySQLBinlogEventPacketDecoder(checksumLength, GlobalTableMapEventMapping.getTableMapEventMap(tableKey), decodeWithTX));
223         channel.pipeline().addLast(new MySQLBinlogEventHandler(getLastBinlogEvent(binlogFileName, binlogPosition)));
224         resetSequenceID();
225         channel.writeAndFlush(new MySQLComBinlogDumpCommandPacket((int) binlogPosition, connectInfo.getServerId(), binlogFileName));
226     }
227     
228     private AbstractBinlogEvent getLastBinlogEvent(final String binlogFileName, final long binlogPosition) {
229         PlaceholderEvent result = new PlaceholderEvent();
230         result.setFileName(binlogFileName);
231         result.setPosition(binlogPosition);
232         return result;
233     }
234     
235     private void resetSequenceID() {
236         channel.attr(MySQLConstants.MYSQL_SEQUENCE_ID).get().set(0);
237     }
238     
239     /**
240      * Poll binlog event.
241      *
242      * @return binlog event
243      */
244     public synchronized List<AbstractBinlogEvent> poll() {
245         if (!running) {
246             return Collections.emptyList();
247         }
248         try {
249             List<AbstractBinlogEvent> result = blockingEventQueue.poll(100L, TimeUnit.MILLISECONDS);
250             return null == result ? Collections.emptyList() : result;
251         } catch (final InterruptedException ignored) {
252             Thread.currentThread().interrupt();
253             return Collections.emptyList();
254         }
255     }
256     
257     @SuppressWarnings("unchecked")
258     private <T> Optional<T> waitExpectedResponse(final Class<T> type) {
259         try {
260             Object response = responseCallback.get(5L, TimeUnit.SECONDS);
261             if (null == response) {
262                 return Optional.empty();
263             }
264             if (type.equals(response.getClass())) {
265                 return Optional.of((T) response);
266             }
267             if (response instanceof MySQLErrPacket) {
268                 throw new PipelineInternalException(((MySQLErrPacket) response).getErrorMessage());
269             }
270             throw new PipelineInternalException("unexpected response type");
271         } catch (final InterruptedException ex) {
272             Thread.currentThread().interrupt();
273             throw new PipelineInternalException(ex);
274         } catch (final ExecutionException | TimeoutException ex) {
275             throw new PipelineInternalException(ex);
276         }
277     }
278     
279     /**
280      * Close netty channel.
281      *
282      * @return channel future
283      */
284     public Optional<ChannelFuture> closeChannel() {
285         if (null == channel || !channel.isOpen()) {
286             return Optional.empty();
287         }
288         running = false;
289         ChannelFuture future = channel.close();
290         if (null != eventLoopGroup) {
291             eventLoopGroup.shutdownGracefully();
292         }
293         return Optional.of(future);
294     }
295     
296     private final class MySQLCommandResponseHandler extends ChannelInboundHandlerAdapter {
297         
298         @Override
299         public void channelRead(final ChannelHandlerContext ctx, final Object msg) {
300             if (null != responseCallback) {
301                 responseCallback.setSuccess(msg);
302             }
303         }
304         
305         @Override
306         public void exceptionCaught(final ChannelHandlerContext ctx, final Throwable cause) {
307             if (null != responseCallback) {
308                 responseCallback.setFailure(cause);
309                 log.error("MySQLCommandResponseHandler protocol resolution error", cause);
310             }
311         }
312     }
313     
314     private final class MySQLBinlogEventHandler extends ChannelInboundHandlerAdapter {
315         
316         private final AtomicReference<AbstractBinlogEvent> lastBinlogEvent;
317         
318         MySQLBinlogEventHandler(final AbstractBinlogEvent lastBinlogEvent) {
319             this.lastBinlogEvent = new AtomicReference<>(lastBinlogEvent);
320         }
321         
322         @SuppressWarnings("unchecked")
323         @Override
324         public void channelRead(final ChannelHandlerContext ctx, final Object msg) throws Exception {
325             if (!running) {
326                 return;
327             }
328             reconnectTimes.set(0);
329             if (msg instanceof List) {
330                 List<AbstractBinlogEvent> records = (List<AbstractBinlogEvent>) msg;
331                 if (records.isEmpty()) {
332                     log.warn("The records is empty");
333                     return;
334                 }
335                 lastBinlogEvent.set(records.get(records.size() - 1));
336                 blockingEventQueue.put(records);
337                 return;
338             }
339             if (msg instanceof AbstractBinlogEvent) {
340                 lastBinlogEvent.set((AbstractBinlogEvent) msg);
341                 blockingEventQueue.put(Collections.singletonList(lastBinlogEvent.get()));
342             }
343         }
344         
345         @Override
346         public void channelInactive(final ChannelHandlerContext ctx) throws Exception {
347             log.warn("MySQL binlog channel inactive");
348             if (!running) {
349                 return;
350             }
351             reconnect();
352         }
353         
354         @Override
355         public void exceptionCaught(final ChannelHandlerContext ctx, final Throwable cause) {
356             String fileName = null == lastBinlogEvent.get() ? null : lastBinlogEvent.get().getFileName();
357             Long position = null == lastBinlogEvent.get() ? null : lastBinlogEvent.get().getPosition();
358             log.error("MySQLBinlogEventHandler protocol resolution error, file name:{}, position:{}", fileName, position, cause);
359         }
360         
361         private void reconnect() throws ExecutionException, InterruptedException, TimeoutException {
362             Optional<ChannelFuture> future = closeChannel();
363             if (future.isPresent()) {
364                 future.get().get(1L, TimeUnit.SECONDS);
365             }
366             if (reconnectTimes.incrementAndGet() > 3) {
367                 log.warn("Exceeds the maximum number of retry times, last binlog event:{}", lastBinlogEvent);
368                 return;
369             }
370             connect();
371             log.info("Reconnect times {}", reconnectTimes.get());
372             subscribe(lastBinlogEvent.get().getFileName(), lastBinlogEvent.get().getPosition());
373         }
374     }
