18  package org.apache.shardingsphere.infra.database.core.metadata.data.loader.type;

20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.infra.database.core.metadata.data.model.ColumnMetaData;
23  import org.apache.shardingsphere.infra.database.core.metadata.database.DialectDatabaseMetaData;
24  import org.apache.shardingsphere.infra.database.core.spi.DatabaseTypedSPILoader;
25  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
26  
27  import java.sql.Connection;
28  import java.sql.ResultSet;
29  import java.sql.SQLException;
30  import java.sql.Statement;
31  import java.util.ArrayList;
32  import java.util.Collection;
33  import java.util.HashSet;
34  import java.util.LinkedList;
35  import java.util.List;
36  import java.util.Objects;
37  import java.util.stream.Collectors;
38  
39  /**
40   * Column meta data loader.
41   */
42  @NoArgsConstructor(access = AccessLevel.PRIVATE)
43  public final class ColumnMetaDataLoader {
44      
45      private static final String COLUMN_NAME = "COLUMN_NAME";
46      
47      private static final String DATA_TYPE = "DATA_TYPE";
48      
49      private static final String TABLE_NAME = "TABLE_NAME";
50      
51      /**
52       * Load column meta data list.
53       *
54       * @param connection connection
55       * @param tableNamePattern table name pattern
56       * @param databaseType database type
57       * @return column meta data list
58       * @throws SQLException SQL exception
59       */
60      public static Collection<ColumnMetaData> load(final Connection connection, final String tableNamePattern, final DatabaseType databaseType) throws SQLException {
61          Collection<ColumnMetaData> result = new LinkedList<>();
62          Collection<String> primaryKeys = loadPrimaryKeys(connection, tableNamePattern);
63          List<String> columnNames = new ArrayList<>();
64          List<Integer> columnTypes = new ArrayList<>();
65          List<Boolean> primaryKeyFlags = new ArrayList<>();
66          List<Boolean> caseSensitiveFlags = new ArrayList<>();
67          try (ResultSet resultSet = connection.getMetaData().getColumns(connection.getCatalog(), connection.getSchema(), tableNamePattern, "%")) {
68              while (resultSet.next()) {
69                  String tableName = resultSet.getString(TABLE_NAME);
70                  if (Objects.equals(tableNamePattern, tableName)) {
71                      String columnName = resultSet.getString(COLUMN_NAME);
72                      columnTypes.add(resultSet.getInt(DATA_TYPE));
73                      primaryKeyFlags.add(primaryKeys.contains(columnName));
74                      columnNames.add(columnName);
75                  }
76              }
77          }
78          try (Statement statement = connection.createStatement(); ResultSet resultSet = statement.executeQuery(generateEmptyResultSQL(tableNamePattern, columnNames, databaseType))) {
79              for (int i = 0; i < columnNames.size(); i++) {
80                  boolean generated = resultSet.getMetaData().isAutoIncrement(i + 1);
81                  caseSensitiveFlags.add(resultSet.getMetaData().isCaseSensitive(resultSet.findColumn(columnNames.get(i))));
82                  result.add(new ColumnMetaData(columnNames.get(i), columnTypes.get(i), primaryKeyFlags.get(i), generated, caseSensitiveFlags.get(i), true, false));
83              }
84          }
85          return result;
86      }
87      
88      private static String generateEmptyResultSQL(final String table, final List<String> columnNames, final DatabaseType databaseType) {
89          DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, databaseType);
90          String wrappedColumnNames = columnNames.stream().map(each -> dialectDatabaseMetaData.getQuoteCharacter().wrap(each)).collect(Collectors.joining(","));
91          return String.format("SELECT %s FROM %s WHERE 1 != 1", wrappedColumnNames, dialectDatabaseMetaData.getQuoteCharacter().wrap(table));
92      }
93      
94      private static Collection<String> loadPrimaryKeys(final Connection connection, final String table) throws SQLException {
95          Collection<String> result = new HashSet<>();
96          try (ResultSet resultSet = connection.getMetaData().getPrimaryKeys(connection.getCatalog(), connection.getSchema(), table)) {
97              while (resultSet.next()) {
98                  result.add(resultSet.getString(COLUMN_NAME));
99              }
100         }
101         return result;
102     }
103 }
