18  package org.apache.shardingsphere.infra.util.close;
19  
20  import org.apache.shardingsphere.infra.exception.core.external.sql.type.wrapper.SQLWrapperException;
21  import org.junit.jupiter.api.Test;
22  
23  import javax.sql.DataSource;
24  import java.sql.SQLException;
25  import java.util.Arrays;
26  
27  import static org.junit.jupiter.api.Assertions.assertThrows;
28  import static org.mockito.Mockito.doThrow;
29  import static org.mockito.Mockito.mock;
30  import static org.mockito.Mockito.times;
31  import static org.mockito.Mockito.verify;
32  import static org.mockito.Mockito.withSettings;
33  
34  class DataSourcesCloserTest {
35      
36      @Test
37      void assertClose() throws Exception {
38          DataSource dataSource0 = mock(DataSource.class, withSettings().extraInterfaces(AutoCloseable.class));
39          DataSource dataSource1 = mock(DataSource.class, withSettings().extraInterfaces(AutoCloseable.class));
40          doThrow(new SQLException("test")).when((AutoCloseable) dataSource1).close();
41          DataSource dataSource2 = mock(DataSource.class, withSettings().extraInterfaces(AutoCloseable.class));
42          assertThrows(SQLWrapperException.class, () -> DataSourcesCloser.close(Arrays.asList(dataSource0, dataSource1, dataSource2)));
43          verify((AutoCloseable) dataSource0, times(1)).close();
44          verify((AutoCloseable) dataSource1, times(1)).close();
45          verify((AutoCloseable) dataSource2, times(1)).close();
46      }
47  }
