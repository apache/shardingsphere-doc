46   * Meta data loader for MySQL.

48  public final class MySQLMetaDataLoader implements DialectMetaDataLoader {
49      
50      private static final String ORDER_BY_ORDINAL_POSITION = " ORDER BY ORDINAL_POSITION";
51      
52      private static final String TABLE_META_DATA_NO_ORDER =
53              "SELECT TABLE_NAME, COLUMN_NAME, DATA_TYPE, COLUMN_KEY, EXTRA, COLLATION_NAME, ORDINAL_POSITION, COLUMN_TYPE, IS_NULLABLE FROM information_schema.columns "
54                      + "WHERE TABLE_SCHEMA=?";
55      
56      private static final String TABLE_META_DATA_SQL = TABLE_META_DATA_NO_ORDER + ORDER_BY_ORDINAL_POSITION;
57      
58      private static final String TABLE_META_DATA_SQL_IN_TABLES = TABLE_META_DATA_NO_ORDER + " AND TABLE_NAME IN (%s)" + ORDER_BY_ORDINAL_POSITION;
59      
60      private static final String INDEX_META_DATA_SQL = "SELECT TABLE_NAME, INDEX_NAME, NON_UNIQUE, COLUMN_NAME FROM information_schema.statistics "
61              + "WHERE TABLE_SCHEMA=? and TABLE_NAME IN (%s) ORDER BY NON_UNIQUE, INDEX_NAME, SEQ_IN_INDEX";
62      
63      private static final String CONSTRAINT_META_DATA_SQL = "SELECT CONSTRAINT_NAME, TABLE_NAME, REFERENCED_TABLE_NAME FROM information_schema.KEY_COLUMN_USAGE "
64              + "WHERE TABLE_NAME IN (%s) AND REFERENCED_TABLE_SCHEMA IS NOT NULL";
65      
66      private static final String VIEW_META_DATA_SQL = "SELECT TABLE_NAME FROM information_schema.VIEWS WHERE TABLE_SCHEMA=? AND TABLE_NAME IN (%s)";
67      
68      @Override
69      public Collection<SchemaMetaData> load(final MetaDataLoaderMaterial material) throws SQLException {
70          Collection<TableMetaData> tableMetaDataList = new LinkedList<>();
71          Map<String, Collection<ColumnMetaData>> columnMetaDataMap = loadColumnMetaDataMap(material.getDataSource(), material.getActualTableNames());
72          Collection<String> viewNames = columnMetaDataMap.isEmpty() ? Collections.emptySet() : loadViewNames(material.getDataSource(), columnMetaDataMap.keySet());
73          Map<String, Collection<IndexMetaData>> indexMetaDataMap = columnMetaDataMap.isEmpty() ? Collections.emptyMap() : loadIndexMetaData(material.getDataSource(), columnMetaDataMap.keySet());
74          Map<String, Collection<ConstraintMetaData>> constraintMetaDataMap =
75                  columnMetaDataMap.isEmpty() ? Collections.emptyMap() : loadConstraintMetaDataMap(material.getDataSource(), columnMetaDataMap.keySet());
76          for (Entry<String, Collection<ColumnMetaData>> entry : columnMetaDataMap.entrySet()) {
77              Collection<IndexMetaData> indexMetaDataList = indexMetaDataMap.getOrDefault(entry.getKey(), Collections.emptyList());
78              Collection<ConstraintMetaData> constraintMetaDataList = constraintMetaDataMap.getOrDefault(entry.getKey(), Collections.emptyList());
79              tableMetaDataList.add(
80                      new TableMetaData(entry.getKey(), entry.getValue(), indexMetaDataList, constraintMetaDataList, viewNames.contains(entry.getKey()) ? TableType.VIEW : TableType.TABLE));
81          }
82          return Collections.singletonList(new SchemaMetaData(material.getDefaultSchemaName(), tableMetaDataList));
83      }
84      
85      private Collection<String> loadViewNames(final DataSource dataSource, final Collection<String> tableNames) throws SQLException {
86          Collection<String> result = new LinkedList<>();
87          try (
88                  Connection connection = dataSource.getConnection();
89                  PreparedStatement preparedStatement = connection.prepareStatement(getViewMetaDataSQL(tableNames))) {
90              String databaseName = "".equals(connection.getCatalog()) ? GlobalDataSourceRegistry.getInstance().getCachedDatabaseTables().get(tableNames.iterator().next()) : connection.getCatalog();
91              preparedStatement.setString(1, databaseName);
92              try (ResultSet resultSet = preparedStatement.executeQuery()) {
93                  while (resultSet.next()) {
94                      result.add(resultSet.getString(1));
95                  }
96              }
97          }
98          return result;
99      }
100     
101     private String getViewMetaDataSQL(final Collection<String> tableNames) {
102         return String.format(VIEW_META_DATA_SQL, tableNames.stream().map(each -> String.format("'%s'", each)).collect(Collectors.joining(",")));
103     }
104     
105     private Map<String, Collection<ConstraintMetaData>> loadConstraintMetaDataMap(final DataSource dataSource, final Collection<String> tables) throws SQLException {
106         Map<String, Collection<ConstraintMetaData>> result = new LinkedHashMap<>();
107         try (
108                 Connection connection = dataSource.getConnection();
109                 PreparedStatement preparedStatement = connection.prepareStatement(getConstraintMetaDataSQL(tables))) {
110             try (ResultSet resultSet = preparedStatement.executeQuery()) {
111                 while (resultSet.next()) {
112                     String constraintName = resultSet.getString("CONSTRAINT_NAME");
113                     String tableName = resultSet.getString("TABLE_NAME");
114                     String referencedTableName = resultSet.getString("REFERENCED_TABLE_NAME");
115                     if (!result.containsKey(tableName)) {
116                         result.put(tableName, new LinkedList<>());
117                     }
118                     result.get(tableName).add(new ConstraintMetaData(constraintName, referencedTableName));
119                 }
120             }
121         }
122         return result;
123     }
124     
125     private String getConstraintMetaDataSQL(final Collection<String> tableNames) {
126         return String.format(CONSTRAINT_META_DATA_SQL, tableNames.stream().map(each -> String.format("'%s'", each)).collect(Collectors.joining(",")));
127     }
128     
129     private Map<String, Collection<ColumnMetaData>> loadColumnMetaDataMap(final DataSource dataSource, final Collection<String> tables) throws SQLException {
130         Map<String, Collection<ColumnMetaData>> result = new HashMap<>();
131         try (
132                 Connection connection = dataSource.getConnection();
133                 PreparedStatement preparedStatement = connection.prepareStatement(getTableMetaDataSQL(tables))) {
134             Map<String, Integer> dataTypes = new DataTypeLoader().load(connection.getMetaData(), getType());
135             String databaseName = "".equals(connection.getCatalog()) ? GlobalDataSourceRegistry.getInstance().getCachedDatabaseTables().get(tables.iterator().next()) : connection.getCatalog();
136             preparedStatement.setString(1, databaseName);
137             try (ResultSet resultSet = preparedStatement.executeQuery()) {
138                 while (resultSet.next()) {
139                     String tableName = resultSet.getString("TABLE_NAME");
140                     ColumnMetaData columnMetaData = loadColumnMetaData(dataTypes, resultSet);
141                     if (!result.containsKey(tableName)) {
142                         result.put(tableName, new LinkedList<>());
143                     }
144                     result.get(tableName).add(columnMetaData);
145                 }
146             }
147         }
148         return result;
149     }
150     
151     private ColumnMetaData loadColumnMetaData(final Map<String, Integer> dataTypeMap, final ResultSet resultSet) throws SQLException {
152         String columnName = resultSet.getString("COLUMN_NAME");
153         String dataType = resultSet.getString("DATA_TYPE");
154         boolean primaryKey = "PRI".equalsIgnoreCase(resultSet.getString("COLUMN_KEY"));
155         String extra = resultSet.getString("EXTRA");
156         boolean generated = "auto_increment".equals(extra);
157         String collationName = resultSet.getString("COLLATION_NAME");
158         boolean caseSensitive = null != collationName && !collationName.endsWith("_ci");
159         boolean visible = !"INVISIBLE".equalsIgnoreCase(extra);
160         boolean unsigned = resultSet.getString("COLUMN_TYPE").toUpperCase().contains("UNSIGNED");
161         boolean nullable = "YES".equals(resultSet.getString("IS_NULLABLE"));
162         return new ColumnMetaData(columnName, dataTypeMap.get(dataType), primaryKey, generated, caseSensitive, visible, unsigned, nullable);
163     }
164     
165     private String getTableMetaDataSQL(final Collection<String> tables) {
166         return tables.isEmpty() ? TABLE_META_DATA_SQL : String.format(TABLE_META_DATA_SQL_IN_TABLES, tables.stream().map(each -> String.format("'%s'", each)).collect(Collectors.joining(",")));
167     }
168     
169     private Map<String, Collection<IndexMetaData>> loadIndexMetaData(final DataSource dataSource, final Collection<String> tableNames) throws SQLException {
170         Map<String, Map<String, IndexMetaData>> tableToIndex = new HashMap<>();
171         try (
172                 Connection connection = dataSource.getConnection();
173                 PreparedStatement preparedStatement = connection.prepareStatement(getIndexMetaDataSQL(tableNames))) {
174             String databaseName = "".equals(connection.getCatalog()) ? GlobalDataSourceRegistry.getInstance().getCachedDatabaseTables().get(tableNames.iterator().next()) : connection.getCatalog();
175             preparedStatement.setString(1, databaseName);
176             try (ResultSet resultSet = preparedStatement.executeQuery()) {
177                 while (resultSet.next()) {
178                     String indexName = resultSet.getString("INDEX_NAME");
179                     String tableName = resultSet.getString("TABLE_NAME");
180                     if (!tableToIndex.containsKey(tableName)) {
181                         tableToIndex.put(tableName, new HashMap<>());
182                     }
183                     Map<String, IndexMetaData> indexMap = tableToIndex.get(tableName);
184                     if (indexMap.containsKey(indexName)) {
185                         indexMap.get(indexName).getColumns().add(resultSet.getString("COLUMN_NAME"));
186                     } else {
187                         IndexMetaData indexMetaData = new IndexMetaData(indexName);
188                         indexMetaData.getColumns().add(resultSet.getString("COLUMN_NAME"));
189                         indexMetaData.setUnique("0".equals(resultSet.getString("NON_UNIQUE")));
190                         indexMap.put(indexName, indexMetaData);
191                     }
192                 }
193             }
194         }
195         Map<String, Collection<IndexMetaData>> result = new HashMap<>(tableToIndex.size(), 1F);
196         for (Entry<String, Map<String, IndexMetaData>> entry : tableToIndex.entrySet()) {
197             result.put(entry.getKey(), entry.getValue().values());
198         }
199         return result;
200     }
201     
202     private String getIndexMetaDataSQL(final Collection<String> tableNames) {
203         return String.format(INDEX_META_DATA_SQL, tableNames.stream().map(each -> String.format("'%s'", each)).collect(Collectors.joining(",")));
204     }

206     @Override
207     public String getDatabaseType() {
208         return "MySQL";
209     }
