35  import java.util.Collection;

37  import java.util.Map.Entry;
38  import java.util.stream.Collectors;

40  /**
41   * Configuration changed subscriber.
42   */
43  @SuppressWarnings("UnstableApiUsage")
44  public final class ConfigurationChangedSubscriber {
45      
46      private final RegistryCenter registryCenter;
47      
48      private final ContextManager contextManager;
49      
50      public ConfigurationChangedSubscriber(final RegistryCenter registryCenter, final ContextManager contextManager) {
51          this.registryCenter = registryCenter;
52          this.contextManager = contextManager;
53          contextManager.getInstanceContext().getEventBusContext().register(this);
54          disableDataSources();
55      }
56      
57      /**
58       * Renew data source units configuration.
59       *
60       * @param event data source changed event.
61       */
62      @Subscribe
63      public synchronized void renew(final DataSourceUnitsChangedEvent event) {
64          contextManager.alterDataSourceUnitsConfiguration(event.getDatabaseName(), event.getDataSourcePropertiesMap());
65          disableDataSources();
66      }
67      
68      /**
69       * Renew data source nodes configuration.
70       *
71       * @param event data source changed event.
72       */
73      @Subscribe
74      public synchronized void renew(final DataSourceNodesChangedEvent event) {
75          contextManager.alterDataSourceNodesConfiguration(event.getDatabaseName(), event.getDataSourcePropertiesMap());
76          disableDataSources();
77      }
78      
79      /**
80       * Renew rule configurations.
81       *
82       * @param event rule configurations changed event
83       */
84      @Subscribe
85      public synchronized void renew(final RuleConfigurationsChangedEvent event) {
86          contextManager.alterRuleConfiguration(event.getDatabaseName(), event.getRuleConfigs());
87          disableDataSources();
88      }
89      
90      /**
91       * Renew global rule configurations.
92       *
93       * @param event global rule configurations changed event
94       */
95      @Subscribe
96      public synchronized void renew(final GlobalRuleConfigurationsChangedEvent event) {
97          contextManager.alterGlobalRuleConfiguration(event.getRuleConfigs());
98          disableDataSources();
99      }
100     
101     /**
102      * Renew properties.
103      *
104      * @param event properties changed event
105      */
106     @Subscribe
107     public synchronized void renew(final PropertiesChangedEvent event) {
108         contextManager.alterProperties(event.getProps());
109     }
110     
111     private void disableDataSources() {
112         Map<String, StorageNodeDataSource> storageNodes = getDisabledDataSources();
113         for (Entry<String, ShardingSphereDatabase> entry : contextManager.getMetaDataContexts().getMetaData().getDatabases().entrySet()) {
114             entry.getValue().getRuleMetaData().findRules(StaticDataSourceContainedRule.class).forEach(each -> disableDataSources(entry.getKey(), each, storageNodes));
115         }
116     }
117     
118     private void disableDataSources(final String databaseName, final StaticDataSourceContainedRule rule, final Map<String, StorageNodeDataSource> storageNodes) {
119         for (Entry<String, StorageNodeDataSource> entry : storageNodes.entrySet()) {
120             QualifiedDatabase database = new QualifiedDatabase(entry.getKey());
121             if (!database.getDatabaseName().equals(databaseName)) {
122                 continue;
123             }
124             disableDataSources(entry.getValue(), rule, database);
125         }
126     }
127     
128     private void disableDataSources(final StorageNodeDataSource storageNodeDataSource, final StaticDataSourceContainedRule rule, final QualifiedDatabase database) {
129         for (Entry<String, Collection<String>> entry : rule.getDataSourceMapper().entrySet()) {
130             if (!database.getGroupName().equals(entry.getKey())) {
131                 continue;
132             }
133             entry.getValue().forEach(each -> rule.updateStatus(new StorageNodeDataSourceChangedEvent(database, storageNodeDataSource)));
134         }
135     }
136     
137     private Map<String, StorageNodeDataSource> getDisabledDataSources() {
138         return registryCenter.getStorageNodeStatusService().loadStorageNodes().entrySet()
139                 .stream().filter(entry -> DataSourceState.DISABLED == entry.getValue().getStatus()).collect(Collectors.toMap(Entry::getKey, Entry::getValue));
140     }
141 }
