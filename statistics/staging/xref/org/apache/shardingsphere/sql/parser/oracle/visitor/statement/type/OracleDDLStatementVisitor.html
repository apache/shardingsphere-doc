267     @SuppressWarnings("unchecked")
268     @Override
269     public ASTNode visitCreateTable(final CreateTableContext ctx) {
270         OracleCreateTableStatement result = new OracleCreateTableStatement();
271         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
272         if (null != ctx.createDefinitionClause()) {
273             CollectionValue<CreateDefinitionSegment> createDefinitions = (CollectionValue<CreateDefinitionSegment>) visit(ctx.createDefinitionClause());
274             for (CreateDefinitionSegment each : createDefinitions.getValue()) {
275                 if (each instanceof ColumnDefinitionSegment) {
276                     result.getColumnDefinitions().add((ColumnDefinitionSegment) each);
277                 } else if (each instanceof ConstraintDefinitionSegment) {
278                     result.getConstraintDefinitions().add((ConstraintDefinitionSegment) each);
279                 }
280             }
281         }
282         return result;
283     }
284     
285     @Override
286     public ASTNode visitCreateDefinitionClause(final CreateDefinitionClauseContext ctx) {
287         CollectionValue<CreateDefinitionSegment> result = new CollectionValue<>();
288         if (null == ctx.createRelationalTableClause()) {
289             return result;
290         }
291         for (RelationalPropertyContext each : ctx.createRelationalTableClause().relationalProperties().relationalProperty()) {
292             if (null != each.columnDefinition()) {
293                 result.getValue().add((ColumnDefinitionSegment) visit(each.columnDefinition()));
294             }
295             if (null != each.outOfLineConstraint()) {
296                 result.getValue().add((ConstraintDefinitionSegment) visit(each.outOfLineConstraint()));
297             }
298             if (null != each.outOfLineRefConstraint()) {
299                 result.getValue().add((ConstraintDefinitionSegment) visit(each.outOfLineRefConstraint()));
300             }
301         }
302         return result;
303     }
304     
305     @Override
306     public ASTNode visitColumnDefinition(final ColumnDefinitionContext ctx) {
307         ColumnSegment column = (ColumnSegment) visit(ctx.columnName());
308         DataTypeSegment dataType = (DataTypeSegment) visit(ctx.dataType());
309         boolean isPrimaryKey = ctx.inlineConstraint().stream().anyMatch(each -> null != each.primaryKey());
310         boolean isNotNull = ctx.inlineConstraint().stream().anyMatch(each -> null != each.NOT() && null != each.NULL());
311         ColumnDefinitionSegment result = new ColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), column, dataType, isPrimaryKey, isNotNull);
312         for (InlineConstraintContext each : ctx.inlineConstraint()) {
313             if (null != each.referencesClause()) {
314                 result.getReferencedTables().add((SimpleTableSegment) visit(each.referencesClause().tableName()));
315             }
316         }
317         if (null != ctx.inlineRefConstraint()) {
318             result.getReferencedTables().add((SimpleTableSegment) visit(ctx.inlineRefConstraint().tableName()));
319         }
320         return result;
321     }
322     
323     @SuppressWarnings("unchecked")
324     @Override
325     public ASTNode visitOutOfLineConstraint(final OutOfLineConstraintContext ctx) {
326         ConstraintDefinitionSegment result = new ConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
327         if (null != ctx.constraintName()) {
328             result.setConstraintName((ConstraintSegment) visit(ctx.constraintName()));
329         }
330         if (null != ctx.primaryKey()) {
331             result.getPrimaryKeyColumns().addAll(((CollectionValue<ColumnSegment>) visit(ctx.columnNames())).getValue());
332         }
333         if (null != ctx.UNIQUE()) {
334             result.getIndexColumns().addAll(((CollectionValue<ColumnSegment>) visit(ctx.columnNames())).getValue());
335         }
336         if (null != ctx.referencesClause()) {
337             result.setReferencedTable((SimpleTableSegment) visit(ctx.referencesClause().tableName()));
338         }
339         return result;
340     }

342     @Override
343     public ASTNode visitOutOfLineRefConstraint(final OutOfLineRefConstraintContext ctx) {
344         ConstraintDefinitionSegment result = new ConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
345         if (null != ctx.constraintName()) {
346             result.setConstraintName((ConstraintSegment) visit(ctx.constraintName()));
347         }
348         if (null != ctx.referencesClause()) {
349             result.setReferencedTable((SimpleTableSegment) visit(ctx.referencesClause().tableName()));
350         }
351         return result;
352     }

354     @SuppressWarnings("unchecked")
355     @Override
356     public ASTNode visitAlterTable(final AlterTableContext ctx) {
357         OracleAlterTableStatement result = new OracleAlterTableStatement();
358         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
359         if (null != ctx.alterDefinitionClause()) {
360             for (AlterDefinitionSegment each : ((CollectionValue<AlterDefinitionSegment>) visit(ctx.alterDefinitionClause())).getValue()) {
361                 if (each instanceof AddColumnDefinitionSegment) {
362                     result.getAddColumnDefinitions().add((AddColumnDefinitionSegment) each);
363                 } else if (each instanceof ModifyColumnDefinitionSegment) {
364                     result.getModifyColumnDefinitions().add((ModifyColumnDefinitionSegment) each);
365                 } else if (each instanceof DropColumnDefinitionSegment) {
366                     result.getDropColumnDefinitions().add((DropColumnDefinitionSegment) each);
367                 } else if (each instanceof AddConstraintDefinitionSegment) {
368                     result.getAddConstraintDefinitions().add((AddConstraintDefinitionSegment) each);
369                 } else if (each instanceof ModifyConstraintDefinitionSegment) {
370                     result.getModifyConstraintDefinitions().add((ModifyConstraintDefinitionSegment) each);
371                 } else if (each instanceof DropConstraintDefinitionSegment) {
372                     result.getDropConstraintDefinitions().add((DropConstraintDefinitionSegment) each);
373                 }
374             }
375         }
376         return result;
377     }

379     @SuppressWarnings("unchecked")
380     @Override
381     public ASTNode visitAlterDefinitionClause(final AlterDefinitionClauseContext ctx) {
382         CollectionValue<AlterDefinitionSegment> result = new CollectionValue<>();
383         if (null != ctx.columnClauses()) {
384             for (OperateColumnClauseContext each : ctx.columnClauses().operateColumnClause()) {
385                 if (null != each.addColumnSpecification()) {
386                     result.getValue().addAll(((CollectionValue<AddColumnDefinitionSegment>) visit(each.addColumnSpecification())).getValue());
387                 }
388                 if (null != each.modifyColumnSpecification()) {
389                     result.getValue().add((ModifyColumnDefinitionSegment) visit(each.modifyColumnSpecification()));
390                 }
391                 if (null != each.dropColumnClause()) {
392                     result.getValue().add((DropColumnDefinitionSegment) visit(each.dropColumnClause()));
393                 }
394             }
395         }
396         if (null != ctx.constraintClauses()) {
397             // TODO Support rename constraint
398             ConstraintClausesContext constraintClausesContext = ctx.constraintClauses();
399             if (null != constraintClausesContext.addConstraintSpecification()) {
400                 result.combine((CollectionValue<AlterDefinitionSegment>) visit(constraintClausesContext.addConstraintSpecification()));
401             }
402             if (null != constraintClausesContext.modifyConstraintClause()) {
403                 result.getValue().add((AlterDefinitionSegment) visit(constraintClausesContext.modifyConstraintClause()));
404             }
405             for (DropConstraintClauseContext each : constraintClausesContext.dropConstraintClause()) {
406                 if (null != each.constraintName()) {
407                     result.getValue().add((AlterDefinitionSegment) visit(each));
408                 }
409             }
410         }
411         return result;
412     }

414     @Override
415     public ASTNode visitAddColumnSpecification(final AddColumnSpecificationContext ctx) {
416         CollectionValue<AddColumnDefinitionSegment> result = new CollectionValue<>();
417         for (ColumnOrVirtualDefinitionContext each : ctx.columnOrVirtualDefinitions().columnOrVirtualDefinition()) {
418             if (null != each.columnDefinition()) {
419                 AddColumnDefinitionSegment addColumnDefinition = new AddColumnDefinitionSegment(
420                         each.columnDefinition().getStart().getStartIndex(), each.columnDefinition().getStop().getStopIndex(),
421                         Collections.singletonList((ColumnDefinitionSegment) visit(each.columnDefinition())));
422                 result.getValue().add(addColumnDefinition);
423             }
424         }
425         return result;
426     }

428     @Override
429     public ASTNode visitModifyColumnSpecification(final ModifyColumnSpecificationContext ctx) {
430         // TODO handle no columnDefinition and multiple columnDefinitions
431         ColumnDefinitionSegment columnDefinition = null;
432         for (ModifyColPropertiesContext each : ctx.modifyColProperties()) {
433             columnDefinition = (ColumnDefinitionSegment) visit(each);
434         }
435         return new ModifyColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnDefinition);
436     }

441         DataTypeSegment dataType = (DataTypeSegment) visit(ctx.dataType());
442         // TODO visit pk and reference table
443         return new ColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), column, dataType, false, false);
444     }

446     @Override
447     public ASTNode visitDropColumnSpecification(final DropColumnSpecificationContext ctx) {
448         Collection<ColumnSegment> columns = new LinkedList<>();
449         if (null != ctx.columnOrColumnList().columnName()) {
450             columns.add((ColumnSegment) visit(ctx.columnOrColumnList().columnName()));
451         } else {
452             for (ColumnNameContext each : ctx.columnOrColumnList().columnNames().columnName()) {
453                 columns.add((ColumnSegment) visit(each));
454             }
455         }
456         return new DropColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columns);
457     }

459     @Override
460     public ASTNode visitAddConstraintSpecification(final AddConstraintSpecificationContext ctx) {
461         CollectionValue<AddConstraintDefinitionSegment> result = new CollectionValue<>();
462         for (OutOfLineConstraintContext each : ctx.outOfLineConstraint()) {
463             result.getValue().add(new AddConstraintDefinitionSegment(each.getStart().getStartIndex(), each.getStop().getStopIndex(), (ConstraintDefinitionSegment) visit(each)));
464         }
465         if (null != ctx.outOfLineRefConstraint()) {
466             result.getValue().add(new AddConstraintDefinitionSegment(ctx.outOfLineRefConstraint().getStart().getStartIndex(), ctx.outOfLineRefConstraint().getStop().getStopIndex(),
467                     (ConstraintDefinitionSegment) visit(ctx.outOfLineRefConstraint())));
468         }
469         return result;
470     }

646     public ASTNode visitAssociateStatistics(final AssociateStatisticsContext ctx) {
647         OracleAssociateStatisticsStatement result = new OracleAssociateStatisticsStatement();
648         if (null != ctx.columnAssociation()) {
649             for (TableNameContext each : ctx.columnAssociation().tableName()) {
650                 result.getTables().add((SimpleTableSegment) visit(each));
651             }
652             for (ColumnNameContext each : ctx.columnAssociation().columnName()) {
653                 result.getColumns().add((ColumnSegment) visit(each));
654             }
655         }
656         if (null != ctx.functionAssociation()) {
657             for (IndexNameContext each : ctx.functionAssociation().indexName()) {
658                 result.getIndexes().add((IndexSegment) visit(each));
659             }
660             for (FunctionContext each : ctx.functionAssociation().function()) {
661                 result.getFunctions().add((FunctionSegment) visit(each));
662             }
663             for (PackageNameContext each : ctx.functionAssociation().packageName()) {
664                 result.getPackages().add((PackageSegment) visit(each));
665             }
666             for (TypeNameContext each : ctx.functionAssociation().typeName()) {
667                 result.getTypes().add((TypeSegment) visit(each));
668             }
669             for (IndexTypeNameContext each : ctx.functionAssociation().indexTypeName()) {
670                 result.getIndexTypes().add((IndexTypeSegment) visit(each));
671             }
672         }
673         return result;
674     }
