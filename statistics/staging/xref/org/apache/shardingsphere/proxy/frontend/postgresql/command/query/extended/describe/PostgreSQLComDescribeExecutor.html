77   * Command describe for PostgreSQL.
78   */
79  @RequiredArgsConstructor
80  public final class PostgreSQLComDescribeExecutor implements CommandExecutor {
81      
82      private static final String ANONYMOUS_COLUMN_NAME = "?column?";
83      
84      private final PortalContext portalContext;
85      
86      private final PostgreSQLComDescribePacket packet;
87      
88      private final ConnectionSession connectionSession;
89      
90      @Override
91      public Collection<DatabasePacket> execute() throws SQLException {
92          switch (packet.getType()) {
93              case 'S':
94                  return describePreparedStatement();
95              case 'P':
96                  return Collections.singleton(portalContext.get(packet.getName()).describe());
97              default:
98                  throw new UnsupportedSQLOperationException("Unsupported describe type: " + packet.getType());
99          }
100     }
101     
102     private List<DatabasePacket> describePreparedStatement() throws SQLException {
103         List<DatabasePacket> result = new ArrayList<>(2);
104         PostgreSQLServerPreparedStatement preparedStatement = connectionSession.getServerPreparedStatementRegistry().getPreparedStatement(packet.getName());
105         result.add(preparedStatement.describeParameters());
106         Optional<PostgreSQLPacket> rowDescription = preparedStatement.describeRows();
107         if (rowDescription.isPresent()) {
108             result.add(rowDescription.get());
109         } else {
110             tryDescribePreparedStatement(preparedStatement);
111             preparedStatement.describeRows().ifPresent(result::add);
112         }
113         return result;
114     }
115     
116     private void tryDescribePreparedStatement(final PostgreSQLServerPreparedStatement preparedStatement) throws SQLException {
117         if (preparedStatement.getSqlStatementContext().getSqlStatement() instanceof InsertStatement) {
118             describeInsertStatementByDatabaseMetaData(preparedStatement);
119         } else {
120             tryDescribePreparedStatementByJDBC(preparedStatement);
121         }
122     }
123     
124     private void describeInsertStatementByDatabaseMetaData(final PostgreSQLServerPreparedStatement preparedStatement) {
125         InsertStatement insertStatement = (InsertStatement) preparedStatement.getSqlStatementContext().getSqlStatement();
126         Collection<Integer> unspecifiedTypeParameterIndexes = getUnspecifiedTypeParameterIndexes(preparedStatement);
127         Optional<ReturningSegment> returningSegment = insertStatement.getReturningSegment();
128         if (insertStatement.getParameterMarkerSegments().isEmpty() && unspecifiedTypeParameterIndexes.isEmpty() && !returningSegment.isPresent()) {
129             return;
130         }
131         String logicTableName = insertStatement.getTable().getTableName().getIdentifier().getValue();
132         ShardingSphereTable table = getTableFromMetaData(connectionSession.getUsedDatabaseName(), insertStatement, logicTableName);
133         List<String> columnNamesOfInsert = getColumnNamesOfInsertStatement(insertStatement, table);
134         preparedStatement.setRowDescription(returningSegment.<PostgreSQLPacket>map(returning -> describeReturning(returning, table))
135                 .orElseGet(PostgreSQLNoDataPacket::getInstance));
136         int parameterMarkerIndex = 0;
137         for (InsertValuesSegment each : insertStatement.getValues()) {
138             ListIterator<ExpressionSegment> listIterator = each.getValues().listIterator();
139             for (int columnIndex = listIterator.nextIndex(); listIterator.hasNext(); columnIndex = listIterator.nextIndex()) {
140                 ExpressionSegment value = listIterator.next();
141                 if (!(value instanceof ParameterMarkerExpressionSegment)) {
142                     continue;
143                 }
144                 if (!unspecifiedTypeParameterIndexes.contains(parameterMarkerIndex)) {
145                     parameterMarkerIndex++;
146                     continue;
147                 }
148                 String columnName = columnNamesOfInsert.get(columnIndex);
149                 ShardingSphereColumn column = table.getColumn(columnName);
150                 ShardingSpherePreconditions.checkNotNull(column, () -> new ColumnNotFoundException(logicTableName, columnName));
151                 preparedStatement.getParameterTypes().set(parameterMarkerIndex++, PostgreSQLColumnType.valueOfJDBCType(column.getDataType()));
152             }
153         }
154     }
155     
156     private Collection<Integer> getUnspecifiedTypeParameterIndexes(final PostgreSQLServerPreparedStatement preparedStatement) {
157         Collection<Integer> result = new HashSet<>();
158         ListIterator<PostgreSQLColumnType> parameterTypesListIterator = preparedStatement.getParameterTypes().listIterator();
159         for (int index = parameterTypesListIterator.nextIndex(); parameterTypesListIterator.hasNext(); index = parameterTypesListIterator.nextIndex()) {
160             if (PostgreSQLColumnType.UNSPECIFIED == parameterTypesListIterator.next()) {
161                 result.add(index);
162             }
163         }
164         return result;
165     }
166     
167     private ShardingSphereTable getTableFromMetaData(final String databaseName, final InsertStatement insertStatement, final String logicTableName) {
168         ShardingSphereDatabase database = ProxyContext.getInstance().getContextManager().getDatabase(databaseName);
169         String schemaName = insertStatement.getTable().getOwner().map(optional -> optional.getIdentifier()
170                 .getValue()).orElseGet(() -> new DatabaseTypeRegistry(database.getProtocolType()).getDefaultSchemaName(databaseName));
171         return database.getSchema(schemaName).getTable(logicTableName);
172     }
173     
174     private List<String> getColumnNamesOfInsertStatement(final InsertStatement insertStatement, final ShardingSphereTable table) {
175         return insertStatement.getColumns().isEmpty() ? table.getColumnNames() : insertStatement.getColumns().stream().map(each -> each.getIdentifier().getValue()).collect(Collectors.toList());
176     }
177     
178     private PostgreSQLRowDescriptionPacket describeReturning(final ReturningSegment returningSegment, final ShardingSphereTable table) {
179         Collection<PostgreSQLColumnDescription> result = new LinkedList<>();
180         for (ProjectionSegment each : returningSegment.getProjections().getProjections()) {
181             if (each instanceof ShorthandProjectionSegment) {
182                 table.getColumnValues().stream().map(column -> new PostgreSQLColumnDescription(column.getName(), 0, column.getDataType(), estimateColumnLength(column.getDataType()), ""))
183                         .forEach(result::add);
184             }
185             if (each instanceof ColumnProjectionSegment) {
186                 ColumnProjectionSegment segment = (ColumnProjectionSegment) each;
187                 String columnName = segment.getColumn().getIdentifier().getValue();
188                 ShardingSphereColumn column = table.containsColumn(columnName) ? table.getColumn(columnName) : generateDefaultColumn(segment);
189                 String alias = segment.getAliasName().orElseGet(column::getName);
190                 result.add(new PostgreSQLColumnDescription(alias, 0, column.getDataType(), estimateColumnLength(column.getDataType()), ""));
191             }
192             if (each instanceof ExpressionProjectionSegment) {
193                 result.add(convertExpressionToDescription((ExpressionProjectionSegment) each));
194             }
195         }
196         return new PostgreSQLRowDescriptionPacket(result);
197     }
198     
199     private ShardingSphereColumn generateDefaultColumn(final ColumnProjectionSegment segment) {
200         return new ShardingSphereColumn(segment.getColumn().getIdentifier().getValue(), Types.VARCHAR, false, false, false, true, false, false);
201     }
202     
203     private PostgreSQLColumnDescription convertExpressionToDescription(final ExpressionProjectionSegment expressionProjectionSegment) {
204         ExpressionSegment expressionSegment = expressionProjectionSegment.getExpr();
205         String columnName = expressionProjectionSegment.getAliasName().orElse(ANONYMOUS_COLUMN_NAME);
206         if (expressionSegment instanceof LiteralExpressionSegment) {
207             Object value = ((LiteralExpressionSegment) expressionSegment).getLiterals();
208             if (value instanceof String) {
209                 return new PostgreSQLColumnDescription(columnName, 0, Types.VARCHAR, estimateColumnLength(Types.VARCHAR), "");
210             }
211             if (value instanceof Integer) {
212                 return new PostgreSQLColumnDescription(columnName, 0, Types.INTEGER, estimateColumnLength(Types.INTEGER), "");
213             }
214             if (value instanceof Long) {
215                 return new PostgreSQLColumnDescription(columnName, 0, Types.BIGINT, estimateColumnLength(Types.BIGINT), "");
216             }
217             if (value instanceof Number) {
218                 return new PostgreSQLColumnDescription(columnName, 0, Types.NUMERIC, estimateColumnLength(Types.NUMERIC), "");
219             }
220         }
221         return new PostgreSQLColumnDescription(columnName, 0, Types.VARCHAR, estimateColumnLength(Types.VARCHAR), "");
222     }
223     
224     private int estimateColumnLength(final int jdbcType) {
225         switch (jdbcType) {
226             case Types.SMALLINT:
227                 return 2;
228             case Types.INTEGER:
229                 return 4;
230             case Types.BIGINT:
231                 return 8;
232             default:
233                 return -1;
234         }
235     }

237     private void tryDescribePreparedStatementByJDBC(final PostgreSQLServerPreparedStatement logicPreparedStatement) throws SQLException {
238         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
239         String databaseName = connectionSession.getUsedDatabaseName();
240         SQLStatementContext sqlStatementContext = new SQLBindEngine(metaDataContexts.getMetaData(), connectionSession.getCurrentDatabaseName(), logicPreparedStatement.getHintValueContext())
241                 .bind(logicPreparedStatement.getSqlStatementContext().getSqlStatement(), Collections.emptyList());
242         QueryContext queryContext = new QueryContext(sqlStatementContext, logicPreparedStatement.getSql(), Collections.emptyList(), logicPreparedStatement.getHintValueContext(),
243                 connectionSession.getConnectionContext(), metaDataContexts.getMetaData());
244         ExecutionContext executionContext = new KernelProcessor().generateExecutionContext(
245                 queryContext, metaDataContexts.getMetaData().getGlobalRuleMetaData(), metaDataContexts.getMetaData().getProps(), connectionSession.getConnectionContext());
246         ExecutionUnit executionUnitSample = executionContext.getExecutionUnits().iterator().next();
247         ProxyDatabaseConnectionManager databaseConnectionManager = connectionSession.getDatabaseConnectionManager();
248         Connection connection = databaseConnectionManager.getConnections(databaseName, executionUnitSample.getDataSourceName(), 0, 1, ConnectionMode.CONNECTION_STRICTLY).iterator().next();
249         try (PreparedStatement actualPreparedStatement = connection.prepareStatement(executionUnitSample.getSqlUnit().getSql())) {
250             populateParameterTypes(logicPreparedStatement, actualPreparedStatement);
251             populateColumnTypes(logicPreparedStatement, actualPreparedStatement);
252         }
253     }
254     
255     private void populateParameterTypes(final PostgreSQLServerPreparedStatement logicPreparedStatement, final PreparedStatement actualPreparedStatement) throws SQLException {
256         if (0 == logicPreparedStatement.getSqlStatementContext().getSqlStatement().getParameterCount()
257                 || logicPreparedStatement.getParameterTypes().stream().noneMatch(each -> PostgreSQLColumnType.UNSPECIFIED == each)) {
258             return;
259         }
260         ParameterMetaData parameterMetaData = actualPreparedStatement.getParameterMetaData();
261         for (int i = 0; i < logicPreparedStatement.getSqlStatementContext().getSqlStatement().getParameterCount(); i++) {
262             if (PostgreSQLColumnType.UNSPECIFIED == logicPreparedStatement.getParameterTypes().get(i)) {
263                 logicPreparedStatement.getParameterTypes().set(i, PostgreSQLColumnType.valueOfJDBCType(parameterMetaData.getParameterType(i + 1), parameterMetaData.getParameterTypeName(i + 1)));
264             }
265         }
266     }
267     
268     private void populateColumnTypes(final PostgreSQLServerPreparedStatement logicPreparedStatement, final PreparedStatement actualPreparedStatement) throws SQLException {
269         if (logicPreparedStatement.describeRows().isPresent()) {
270             return;
271         }
272         ResultSetMetaData resultSetMetaData = actualPreparedStatement.getMetaData();
273         if (null == resultSetMetaData) {
274             logicPreparedStatement.setRowDescription(PostgreSQLNoDataPacket.getInstance());
275             return;
276         }
277         List<PostgreSQLColumnDescription> columnDescriptions = new ArrayList<>(resultSetMetaData.getColumnCount());
278         for (int columnIndex = 1; columnIndex <= resultSetMetaData.getColumnCount(); columnIndex++) {
279             String columnName = resultSetMetaData.getColumnName(columnIndex);
280             int columnType = resultSetMetaData.getColumnType(columnIndex);
281             int columnLength = resultSetMetaData.getColumnDisplaySize(columnIndex);
282             String columnTypeName = resultSetMetaData.getColumnTypeName(columnIndex);
283             columnDescriptions.add(new PostgreSQLColumnDescription(columnName, columnIndex, columnType, columnLength, columnTypeName));
284         }
285         logicPreparedStatement.setRowDescription(new PostgreSQLRowDescriptionPacket(columnDescriptions));
286     }
