119     @Override
120     public ASTNode visitCreateShardingTableRule(final CreateShardingTableRuleContext ctx) {
121         Collection<AbstractTableRuleSegment> tableRuleSegments = ctx.shardingTableRuleDefinition().stream()
122                 .map(each -> (AbstractTableRuleSegment) visit(each)).filter(Objects::nonNull).collect(Collectors.toList());
123         return new CreateShardingTableRuleStatement(null != ctx.ifNotExists(), tableRuleSegments);
124     }

126     @Override
127     public ASTNode visitCreateShardingTableReferenceRule(final CreateShardingTableReferenceRuleContext ctx) {
128         return new CreateShardingTableReferenceRuleStatement(null != ctx.ifNotExists(), getTableReferenceRuleSegments(ctx.tableReferenceRuleDefinition()));
129     }
130     
131     private Collection<TableReferenceRuleSegment> getTableReferenceRuleSegments(final List<TableReferenceRuleDefinitionContext> ctx) {
132         Collection<TableReferenceRuleSegment> result = new LinkedList<>();
133         for (TableReferenceRuleDefinitionContext each : ctx) {
134             String ruleName = getIdentifierValue(each.ruleName());
135             String reference = each.tableName().stream().map(this::getIdentifierValue).collect(Collectors.joining(","));
136             result.add(new TableReferenceRuleSegment(ruleName, reference));
137         }
138         return result;
139     }

163     @Override
164     public ASTNode visitCreateDefaultShardingStrategy(final CreateDefaultShardingStrategyContext ctx) {
165         ShardingStrategyContext shardingStrategyContext = ctx.shardingStrategy();
166         String defaultType = new IdentifierValue(ctx.type.getText()).getValue();
167         String strategyType = getIdentifierValue(shardingStrategyContext.strategyType());
168         if ("none".equalsIgnoreCase(strategyType)) {
169             return new CreateDefaultShardingStrategyStatement(null != ctx.ifNotExists(), defaultType, "none", null, null);
170         }
171         AlgorithmSegment algorithmSegment = null == shardingStrategyContext.shardingAlgorithm().algorithmDefinition()
172                 ? null
173                 : (AlgorithmSegment) visitAlgorithmDefinition(shardingStrategyContext.shardingAlgorithm().algorithmDefinition());
174         String shardingColumn = null == ctx.shardingStrategy().shardingColumnDefinition() ? null : buildShardingColumn(ctx.shardingStrategy().shardingColumnDefinition());
175         return new CreateDefaultShardingStrategyStatement(null != ctx.ifNotExists(), defaultType, strategyType, shardingColumn, algorithmSegment);
176     }

178     @Override
179     public ASTNode visitAlterDefaultShardingStrategy(final AlterDefaultShardingStrategyContext ctx) {
180         String defaultType = new IdentifierValue(ctx.type.getText()).getValue();
181         String strategyType = getIdentifierValue(ctx.shardingStrategy().strategyType());
182         if ("none".equalsIgnoreCase(strategyType)) {
183             return new AlterDefaultShardingStrategyStatement(defaultType, "none", null, null);
184         }
185         AlgorithmSegment algorithmSegment = null == ctx.shardingStrategy().shardingAlgorithm().algorithmDefinition()
186                 ? null
187                 : (AlgorithmSegment) visitAlgorithmDefinition(ctx.shardingStrategy().shardingAlgorithm().algorithmDefinition());
188         String shardingColumn = null == ctx.shardingStrategy().shardingColumnDefinition() ? null : buildShardingColumn(ctx.shardingStrategy().shardingColumnDefinition());
189         return new AlterDefaultShardingStrategyStatement(defaultType, strategyType, shardingColumn, algorithmSegment);
190     }

224     @Override
225     public ASTNode visitShardingTableRule(final ShardingTableRuleContext ctx) {
226         KeyGenerateStrategySegment keyGenerateSegment = null == ctx.keyGenerateDefinition() ? null : (KeyGenerateStrategySegment) visit(ctx.keyGenerateDefinition());
227         AuditStrategySegment auditStrategySegment = null == ctx.auditDefinition() ? null : (AuditStrategySegment) visitAuditDefinition(getIdentifierValue(ctx.tableName()), ctx.auditDefinition());
228         TableRuleSegment result = new TableRuleSegment(getIdentifierValue(ctx.tableName()), getDataNodes(ctx.dataNodes()), keyGenerateSegment, auditStrategySegment);
229         Optional.ofNullable(ctx.tableStrategy()).ifPresent(optional -> result.setTableStrategySegment((ShardingStrategySegment) visit(ctx.tableStrategy().shardingStrategy())));
230         Optional.ofNullable(ctx.databaseStrategy()).ifPresent(optional -> result.setDatabaseStrategySegment((ShardingStrategySegment) visit(ctx.databaseStrategy().shardingStrategy())));
231         return result;
232     }

234     private ASTNode visitAuditDefinition(final String tableName, final AuditDefinitionContext ctx) {
235         if (null == ctx) {
236             return null;
237         }
238         Collection<ShardingAuditorSegment> shardingAuditorSegments = new LinkedList<>();
239         int index = 0;
240         for (SingleAuditDefinitionContext each : ctx.multiAuditDefinition().singleAuditDefinition()) {
241             String algorithmTypeName = getIdentifierValue(each.algorithmDefinition().algorithmTypeName());
242             String auditorName = String.format("%s_%s_%s", tableName, algorithmTypeName, index++).toLowerCase();
243             shardingAuditorSegments.add(new ShardingAuditorSegment(auditorName, (AlgorithmSegment) visit(each.algorithmDefinition())));
244         }
245         return new AuditStrategySegment(shardingAuditorSegments, Boolean.parseBoolean(getIdentifierValue(ctx.auditAllowHintDisable())));
246     }

254     @Override
255     public ASTNode visitShardingAutoTableRule(final ShardingAutoTableRuleContext ctx) {
256         AutoTableRuleSegment result = new AutoTableRuleSegment(getIdentifierValue(ctx.tableName()), getResources(ctx.storageUnits()));
257         Optional.ofNullable(ctx.keyGenerateDefinition()).ifPresent(optional -> result.setKeyGenerateStrategySegment((KeyGenerateStrategySegment) visit(ctx.keyGenerateDefinition())));
258         Optional.ofNullable(ctx.auditDefinition()).ifPresent(optional -> result.setAuditStrategySegment((AuditStrategySegment) visitAuditDefinition(getIdentifierValue(ctx.tableName()),
259                 ctx.auditDefinition())));
260         Optional.ofNullable(ctx.autoShardingColumnDefinition()).ifPresent(optional -> result.setShardingColumn(buildShardingColumn(ctx.autoShardingColumnDefinition())));
261         Optional.ofNullable(ctx.algorithmDefinition()).ifPresent(optional -> result.setShardingAlgorithmSegment((AlgorithmSegment) visit(ctx.algorithmDefinition())));
262         return result;
263     }

270     @Override
271     public ASTNode visitShardingStrategy(final ShardingStrategyContext ctx) {
272         String strategyType = getIdentifierValue(ctx.strategyType());
273         if ("none".equalsIgnoreCase(strategyType)) {
274             return new ShardingStrategySegment(strategyType, null, null);
275         }
276         AlgorithmSegment algorithmSegment = null != ctx.shardingAlgorithm().algorithmDefinition() ? (AlgorithmSegment) visitAlgorithmDefinition(ctx.shardingAlgorithm().algorithmDefinition()) : null;
277         String shardingColumn = null != ctx.shardingColumnDefinition() ? buildShardingColumn(ctx.shardingColumnDefinition()) : null;
278         return new ShardingStrategySegment(strategyType, shardingColumn, algorithmSegment);
279     }

294     private String getIdentifierValue(final ParseTree context) {
295         return null == context ? null : new IdentifierValue(context.getText()).getValue();
296     }
297     
298     private String getIdentifierValueForDataNodes(final ParseTree context) {
299         if (null == context) {
300             return null;
301         }
302         String value = new IdentifierValue(context.getText(), "[]'").getValue();
303         return value.startsWith("'") ? value.substring(1, value.length() - 1) : value.trim();
304     }
305     
306     private Properties getProperties(final PropertiesDefinitionContext ctx) {
307         Properties result = new Properties();
308         if (null == ctx || null == ctx.properties()) {
309             return result;
310         }
311         for (PropertyContext each : ctx.properties().property()) {
312             result.setProperty(IdentifierValue.getQuotedContent(each.key.getText()), IdentifierValue.getQuotedContent(each.value.getText()));
313         }
314         return result;
315     }
