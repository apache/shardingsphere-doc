72  @ExtendWith(AutoMockExtension.class)
73  @StaticMockSettings(ProxyContext.class)
74  @MockitoSettings(strictness = Strictness.LENIENT)
75  class PostgreSQLBatchedStatementsExecutorTest {
76      
77      @Mock
78      private BackendConnection backendConnection;
79      
80      @Mock
81      private JDBCBackendStatement backendStatement;
82      
83      @Test
84      void assertExecuteBatch() throws SQLException {
85          Connection connection = mock(Connection.class, RETURNS_DEEP_STUBS);
86          when(connection.getMetaData().getURL()).thenReturn("jdbc:postgresql://127.0.0.1/db");
87          when(backendConnection.getConnections(nullable(String.class), anyInt(), any(ConnectionMode.class))).thenReturn(Collections.singletonList(connection));
88          PreparedStatement preparedStatement = mock(PreparedStatement.class);
89          when(preparedStatement.getConnection()).thenReturn(connection);
90          when(preparedStatement.executeBatch()).thenReturn(new int[]{1, 1, 1});
91          when(backendStatement.createStorageResource(any(ExecutionUnit.class), eq(connection), any(ConnectionMode.class), any(StatementOption.class), nullable(DatabaseType.class)))
92                  .thenReturn(preparedStatement);
93          ContextManager contextManager = mockContextManager();
94          ConnectionSession connectionSession = mockConnectionSession();
95          PostgreSQLServerPreparedStatement postgreSQLPreparedStatement = new PostgreSQLServerPreparedStatement("insert into t (id, col) values (?, ?)", mockInsertStatementContext(),
96                  Arrays.asList(PostgreSQLColumnType.POSTGRESQL_TYPE_INT4, PostgreSQLColumnType.POSTGRESQL_TYPE_VARCHAR));
97          List<List<Object>> parameterSets = Arrays.asList(Arrays.asList(1, new PostgreSQLTypeUnspecifiedSQLParameter("foo")),
98                  Arrays.asList(2, new PostgreSQLTypeUnspecifiedSQLParameter("bar")), Arrays.asList(3, new PostgreSQLTypeUnspecifiedSQLParameter("baz")));
99          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
100         PostgreSQLBatchedStatementsExecutor actual = new PostgreSQLBatchedStatementsExecutor(connectionSession, postgreSQLPreparedStatement, parameterSets);
101         prepareExecutionUnitParameters(actual, parameterSets);
102         int actualUpdated = actual.executeBatch();
103         assertThat(actualUpdated, is(3));
104         InOrder inOrder = inOrder(preparedStatement);
105         for (List<Object> each : parameterSets) {
106             inOrder.verify(preparedStatement).setObject(1, each.get(0));
107             inOrder.verify(preparedStatement).setObject(2, each.get(1).toString());
108             inOrder.verify(preparedStatement).addBatch();
109         }
110     }
111     
112     private InsertStatementContext mockInsertStatementContext() {
113         PostgreSQLInsertStatement insertStatement = mock(PostgreSQLInsertStatement.class, RETURNS_DEEP_STUBS);
114         when(insertStatement.getTable().getTableName().getIdentifier().getValue()).thenReturn("t");
115         InsertStatementContext result = mock(InsertStatementContext.class);
116         when(result.getSqlStatement()).thenReturn(insertStatement);
117         return result;
118     }
119     
120     private ContextManager mockContextManager() {
121         ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
122         when(result.getMetaDataContexts().getMetaData().getProps().getValue(ConfigurationPropertyKey.KERNEL_EXECUTOR_SIZE)).thenReturn(1);
123         when(result.getMetaDataContexts().getMetaData().getProps().getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY)).thenReturn(1);
124         when(result.getMetaDataContexts().getMetaData().getProps().getValue(ConfigurationPropertyKey.SQL_SHOW)).thenReturn(false);
125         ShardingSphereDatabase database = mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS);
126         when(database.getResourceMetaData().getStorageTypes()).thenReturn(Collections.singletonMap("ds_0", new PostgreSQLDatabaseType()));
127         when(database.getResourceMetaData().getAllInstanceDataSourceNames()).thenReturn(Collections.singletonList("ds_0"));
128         when(database.getRuleMetaData()).thenReturn(new ShardingSphereRuleMetaData(Collections.emptyList()));
129         when(result.getMetaDataContexts().getMetaData().getDatabase("db")).thenReturn(database);
130         ShardingSphereRuleMetaData globalRuleMetaData = new ShardingSphereRuleMetaData(Arrays.asList(new SQLTranslatorRule(new DefaultSQLTranslatorRuleConfigurationBuilder().build()),
131                 new LoggingRule(new DefaultLoggingRuleConfigurationBuilder().build())));
132         when(result.getMetaDataContexts().getMetaData().getGlobalRuleMetaData()).thenReturn(globalRuleMetaData);
133         return result;
134     }
135     
136     private ConnectionSession mockConnectionSession() {
137         ConnectionSession result = mock(ConnectionSession.class);
138         when(result.getDatabaseName()).thenReturn("db");
139         when(result.getBackendConnection()).thenReturn(backendConnection);
140         when(result.getStatementManager()).thenReturn(backendStatement);
141         when(result.getConnectionContext()).thenReturn(new ConnectionContext());
142         return result;
143     }

145     @SuppressWarnings("unchecked")
146     @SneakyThrows(ReflectiveOperationException.class)
147     private void prepareExecutionUnitParameters(final PostgreSQLBatchedStatementsExecutor target, final List<List<Object>> parameterSets) {
148         ((Map<ExecutionUnit, List<List<Object>>>) Plugins.getMemberAccessor().get(PostgreSQLBatchedStatementsExecutor.class.getDeclaredField("executionUnitParams"), target))
149                 .replaceAll((k, v) -> parameterSets);
150     }
