44  /**
45   * Inventory task splitter.
46   */
47  @RequiredArgsConstructor
48  @Slf4j
49  public final class InventoryTaskSplitter {
50      
51      private final PipelineDataSource sourceDataSource;
52      
53      private final InventoryDumperContext dumperContext;
54      
55      private final ImporterConfiguration importerConfig;
56      
57      /**
58       * Split inventory data to multi-tasks.
59       *
60       * @param jobItemContext job item context
61       * @return split inventory data task
62       */
63      public List<InventoryTask> split(final TransmissionJobItemContext jobItemContext) {
64          List<InventoryTask> result = new LinkedList<>();
65          long startTimeMillis = System.currentTimeMillis();
66          TransmissionProcessContext processContext = jobItemContext.getJobProcessContext();
67          InventoryDumperContextSplitter dumperContextSplitter = new InventoryDumperContextSplitter(sourceDataSource, dumperContext);
68          for (InventoryDumperContext each : dumperContextSplitter.split(jobItemContext)) {
69              AtomicReference<IngestPosition> position = new AtomicReference<>(each.getCommonContext().getPosition());
70              PipelineChannel channel = InventoryChannelCreator.create(processContext.getProcessConfiguration().getStreamChannel(), importerConfig.getBatchSize(), position);
71              InventoryDataRecordPositionCreator positionCreator = each.hasUniqueKey() ? new UniqueKeyInventoryDataRecordPositionCreator() : new PlaceholderInventoryDataRecordPositionCreator();
72              Dumper dumper = new InventoryDumper(each, channel, sourceDataSource, jobItemContext.getSourceMetaDataLoader(), positionCreator);
73              Importer importer = new SingleChannelConsumerImporter(channel, importerConfig.getBatchSize(), 3000L, jobItemContext.getSink(), jobItemContext);
74              result.add(new InventoryTask(PipelineTaskUtils.generateInventoryTaskId(each),
75                      processContext.getInventoryDumperExecuteEngine(), processContext.getInventoryImporterExecuteEngine(), dumper, importer, position));
76          }
77          log.info("Split inventory tasks cost {} ms", System.currentTimeMillis() - startTimeMillis);
78          return result;
79      }
80  }
