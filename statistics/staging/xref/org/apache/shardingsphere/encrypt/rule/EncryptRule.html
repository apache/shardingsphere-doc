52  public final class EncryptRule implements DatabaseRule, TableContainedRule {
53      
54      @Getter
55      private final RuleConfiguration configuration;
56      
57      @SuppressWarnings("rawtypes")
58      private final Map<String, StandardEncryptAlgorithm> standardEncryptors = new LinkedHashMap<>();
59      
60      @SuppressWarnings("rawtypes")
61      private final Map<String, LikeEncryptAlgorithm> likeEncryptors = new LinkedHashMap<>();
62      
63      @SuppressWarnings("rawtypes")
64      private final Map<String, AssistedEncryptAlgorithm> assistedEncryptors = new LinkedHashMap<>();
65      
66      private final Map<String, EncryptTable> tables = new LinkedHashMap<>();
67      
68      private final TableNamesMapper tableNamesMapper = new TableNamesMapper();
69      
70      public EncryptRule(final EncryptRuleConfiguration ruleConfig) {
71          configuration = ruleConfig;
72          ruleConfig.getEncryptors().forEach((key, value) -> putAllEncryptors(key, TypedSPILoader.getService(EncryptAlgorithm.class, value.getType(), value.getProps())));
73          for (EncryptTableRuleConfiguration each : ruleConfig.getTables()) {
74              each.getColumns().forEach(this::checkEncryptAlgorithmType);
75              tables.put(each.getName().toLowerCase(), new EncryptTable(each));
76              tableNamesMapper.put(each.getName());
77          }
78      }
79      
80      /**
81       * Encrypt rule constructor.
82       * 
83       * @deprecated deprecated by compatible encrypt rule configuration
84       */
85      @Deprecated
86      public EncryptRule(final CompatibleEncryptRuleConfiguration ruleConfig) {
87          configuration = ruleConfig;
88          ruleConfig.getEncryptors().forEach((key, value) -> putAllEncryptors(key, TypedSPILoader.getService(EncryptAlgorithm.class, value.getType(), value.getProps())));
89          for (EncryptTableRuleConfiguration each : ruleConfig.getTables()) {
90              each.getColumns().forEach(this::checkEncryptAlgorithmType);
91              tables.put(each.getName().toLowerCase(), new EncryptTable(each));
92              tableNamesMapper.put(each.getName());
93          }
94      }
95      
96      @SuppressWarnings("rawtypes")
97      private void putAllEncryptors(final String encryptorName, final EncryptAlgorithm algorithm) {
98          if (algorithm instanceof StandardEncryptAlgorithm) {
99              standardEncryptors.put(encryptorName, (StandardEncryptAlgorithm) algorithm);
100         }
101         if (algorithm instanceof LikeEncryptAlgorithm) {
102             likeEncryptors.put(encryptorName, (LikeEncryptAlgorithm) algorithm);
103         }
104         if (algorithm instanceof AssistedEncryptAlgorithm) {
105             assistedEncryptors.put(encryptorName, (AssistedEncryptAlgorithm) algorithm);
106         }
107     }
108     
109     private void checkEncryptAlgorithmType(final EncryptColumnRuleConfiguration columnRuleConfig) {
110         ShardingSpherePreconditions.checkState(standardEncryptors.containsKey(columnRuleConfig.getCipher().getEncryptorName()),
111                 () -> new MismatchedEncryptAlgorithmTypeException("Cipher", columnRuleConfig.getCipher().getEncryptorName(), StandardEncryptAlgorithm.class.getSimpleName()));
112         columnRuleConfig.getLikeQuery().ifPresent(optional -> ShardingSpherePreconditions.checkState(likeEncryptors.containsKey(optional.getEncryptorName()),
113                 () -> new MismatchedEncryptAlgorithmTypeException("Like query", optional.getEncryptorName(), LikeEncryptAlgorithm.class.getSimpleName())));
114         columnRuleConfig.getAssistedQuery().ifPresent(optional -> ShardingSpherePreconditions.checkState(assistedEncryptors.containsKey(optional.getEncryptorName()),
115                 () -> new MismatchedEncryptAlgorithmTypeException("Assisted query", optional.getEncryptorName(), AssistedEncryptAlgorithm.class.getSimpleName())));
116     }

175     /**
176      * Encrypt.
177      *
178      * @param databaseName database name
179      * @param schemaName schema name
180      * @param tableName table name
181      * @param logicColumnName logic column name
182      * @param originalValues original values
183      * @return encrypted values
184      */
185     public List<Object> encrypt(final String databaseName, final String schemaName, final String tableName, final String logicColumnName, final List<Object> originalValues) {
186         @SuppressWarnings("rawtypes")
187         Optional<StandardEncryptAlgorithm> encryptor = findStandardEncryptor(tableName, logicColumnName);
188         ShardingSpherePreconditions.checkState(encryptor.isPresent(), () -> new MissingEncryptorException(tableName, logicColumnName, "STANDARD"));
189         EncryptContext context = EncryptContextBuilder.build(databaseName, schemaName, tableName, logicColumnName);
190         return encrypt(encryptor.get(), originalValues, context);
191     }
192     
193     @SuppressWarnings("unchecked")
194     private List<Object> encrypt(@SuppressWarnings("rawtypes") final StandardEncryptAlgorithm encryptor, final List<Object> originalValues, final EncryptContext context) {
195         List<Object> result = new LinkedList<>();
196         for (Object each : originalValues) {
197             result.add(null == each ? null : encryptor.encrypt(each, context));
198         }
199         return result;
200     }
