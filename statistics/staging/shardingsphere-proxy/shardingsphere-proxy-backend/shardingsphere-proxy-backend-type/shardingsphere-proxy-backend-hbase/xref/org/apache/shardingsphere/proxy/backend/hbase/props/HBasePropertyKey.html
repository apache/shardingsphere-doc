18  package org.apache.shardingsphere.proxy.backend.hbase.props;
19  
20  import lombok.Getter;
21  import lombok.RequiredArgsConstructor;
22  import org.apache.shardingsphere.infra.util.props.TypedPropertyKey;
23  
24  /**
25   * Typed property key of HBase.
26   */
27  @RequiredArgsConstructor
28  @Getter
29  public enum HBasePropertyKey implements TypedPropertyKey {
30      
31      /**
32       * Background threads are periodically executed to update metadata.
33       */
34      META_REFRESHER_INTERVAL_SECONDS("meta-refresher-interval-seconds", String.valueOf(300), int.class),
35      
36      /**
37       * Maximum number of scanned rows.
38       */
39      MAX_SCAN_LIMIT_SIZE("max-scan-limit-size", String.valueOf(5000), long.class),
40      
41      /**
42       * Warm up thread num.
43       */
44      WARM_UP_THREAD_NUM("warm-up-thread-num", String.valueOf(1), int.class),
45      
46      /**
47       * Is sync warm up.
48       */
49      IS_SYNC_WARM_UP("is-sync-warm-up", String.valueOf(true), boolean.class),
50      
51      /**
52       * HBase execute out time.
53       */
54      EXECUTE_TIME_OUT("execute-time-out", String.valueOf(2000), Long.class);
55      
56      private final String key;
57      
58      private final String defaultValue;
59      
60      private final Class<?> type;
61  }
