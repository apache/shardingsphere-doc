18  package org.apache.shardingsphere.infra.rewrite.sql.token.common.generator.generic;
19  
20  import org.apache.shardingsphere.infra.binder.context.statement.SQLStatementContext;
21  import org.apache.shardingsphere.infra.binder.context.type.IndexAvailable;
22  import org.apache.shardingsphere.infra.binder.context.type.RemoveAvailable;
23  import org.apache.shardingsphere.infra.binder.context.type.TableAvailable;
24  import org.apache.shardingsphere.infra.database.core.metadata.database.DialectDatabaseMetaData;
25  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
26  import org.apache.shardingsphere.infra.database.core.type.DatabaseTypeRegistry;
27  import org.apache.shardingsphere.infra.rewrite.sql.token.common.generator.CollectionSQLTokenGenerator;
28  import org.apache.shardingsphere.infra.rewrite.sql.token.common.pojo.SQLToken;
29  import org.apache.shardingsphere.infra.rewrite.sql.token.common.pojo.generic.RemoveToken;
30  import org.apache.shardingsphere.sql.parser.statement.core.segment.SQLSegment;
31  import org.apache.shardingsphere.sql.parser.statement.core.segment.ddl.index.IndexSegment;
32  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.OwnerSegment;
33  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.table.SimpleTableSegment;
34  
35  import java.util.Collection;
36  import java.util.LinkedList;
37  import java.util.stream.Collectors;
38  
39  /**
40   * Remove token generator.
41   */
42  public final class RemoveTokenGenerator implements CollectionSQLTokenGenerator<SQLStatementContext> {
43      
44      @Override
45      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
46          if (sqlStatementContext instanceof RemoveAvailable && !((RemoveAvailable) sqlStatementContext).getRemoveSegments().isEmpty()) {
47              return true;
48          }
49          if (sqlStatementContext instanceof TableAvailable && !((TableAvailable) sqlStatementContext).getTablesContext().getSimpleTables().isEmpty()) {
50              return true;
51          }
52          return sqlStatementContext instanceof IndexAvailable && !((IndexAvailable) sqlStatementContext).getIndexes().isEmpty();
53      }
54      
55      @Override
56      public Collection<SQLToken> generateSQLTokens(final SQLStatementContext sqlStatementContext) {
57          Collection<SQLToken> result = new LinkedList<>();
58          if (sqlStatementContext instanceof RemoveAvailable && !((RemoveAvailable) sqlStatementContext).getRemoveSegments().isEmpty()) {
59              result.addAll(generateRemoveAvailableSQLTokens(((RemoveAvailable) sqlStatementContext).getRemoveSegments()));
60          }
61          if (sqlStatementContext instanceof TableAvailable && !((TableAvailable) sqlStatementContext).getTablesContext().getSimpleTables().isEmpty()) {
62              result.addAll(generateTableAvailableSQLTokens((TableAvailable) sqlStatementContext, sqlStatementContext.getDatabaseType()));
63          }
64          if (sqlStatementContext instanceof IndexAvailable && !((IndexAvailable) sqlStatementContext).getIndexes().isEmpty()) {
65              result.addAll(generateIndexAvailableSQLTokens((IndexAvailable) sqlStatementContext));
66          }
67          return result;
68      }
69      
70      private Collection<RemoveToken> generateRemoveAvailableSQLTokens(final Collection<SQLSegment> removeSegments) {
71          return removeSegments.stream().map(each -> new RemoveToken(each.getStartIndex(), each.getStopIndex())).collect(Collectors.toList());
72      }
73      
74      private Collection<RemoveToken> generateTableAvailableSQLTokens(final TableAvailable tableAvailable, final DatabaseType databaseType) {
75          Collection<RemoveToken> result = new LinkedList<>();
76          for (SimpleTableSegment each : tableAvailable.getTablesContext().getSimpleTables()) {
77              if (!each.getOwner().isPresent()) {
78                  continue;
79              }
80              DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(databaseType).getDialectDatabaseMetaData();
81              OwnerSegment ownerSegment = each.getOwner().get();
82              if (dialectDatabaseMetaData.getDefaultSchema().isPresent()) {
83                  ownerSegment.getOwner().ifPresent(optional -> result.add(new RemoveToken(optional.getStartIndex(), ownerSegment.getStartIndex() - 1)));
84              } else {
85                  result.add(new RemoveToken(ownerSegment.getStartIndex(), each.getTableName().getStartIndex() - 1));
86              }
87          }
88          return result;
89      }
90      
91      private Collection<RemoveToken> generateIndexAvailableSQLTokens(final IndexAvailable indexAvailable) {
92          Collection<RemoveToken> result = new LinkedList<>();
93          for (IndexSegment each : indexAvailable.getIndexes()) {
94              if (!each.getOwner().isPresent()) {
95                  continue;
96              }
97              OwnerSegment owner = each.getOwner().get();
98              int startIndex = owner.getOwner().isPresent() ? owner.getOwner().get().getStartIndex() : owner.getStartIndex();
99              int stopIndex = owner.getOwner().isPresent() ? owner.getStartIndex() - 1 : each.getStartIndex() - 1;
100             result.add(new RemoveToken(startIndex, stopIndex));
101         }
102         return result;
103     }
104 }
