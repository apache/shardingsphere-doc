52      @Test
53      void assertBindWithAlias() {
54          JoinTableSegment joinTableSegment = mock(JoinTableSegment.class);
55          SimpleTableSegment leftTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")));
56          leftTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("o")));
57          SimpleTableSegment rightTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order_item")));
58          rightTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("i")));
59          when(joinTableSegment.getLeft()).thenReturn(leftTable);
60          when(joinTableSegment.getRight()).thenReturn(rightTable);
61          ShardingSphereMetaData metaData = createMetaData();
62          Map<String, TableSegmentBinderContext> tableBinderContexts = new CaseInsensitiveMap<>();
63          JoinTableSegment actual = JoinTableSegmentBinder.bind(joinTableSegment, metaData, DefaultDatabase.LOGIC_NAME, databaseType, tableBinderContexts);
64          assertTrue(actual.getLeft() instanceof SimpleTableSegment);
65          assertThat(((SimpleTableSegment) actual.getLeft()).getTableName().getOriginalDatabase().getValue(), is(DefaultDatabase.LOGIC_NAME));
66          assertThat(((SimpleTableSegment) actual.getLeft()).getTableName().getOriginalSchema().getValue(), is(DefaultDatabase.LOGIC_NAME));
67          assertTrue(actual.getRight() instanceof SimpleTableSegment);
68          assertThat(((SimpleTableSegment) actual.getRight()).getTableName().getOriginalDatabase().getValue(), is(DefaultDatabase.LOGIC_NAME));
69          assertThat(((SimpleTableSegment) actual.getRight()).getTableName().getOriginalSchema().getValue(), is(DefaultDatabase.LOGIC_NAME));
70          assertThat(actual.getJoinTableProjectionSegments().size(), is(7));
71          assertTrue(tableBinderContexts.containsKey("o"));
72          assertTrue(tableBinderContexts.containsKey("i"));
73      }

97      void assertBindWithNaturalJoin() {
98          JoinTableSegment joinTableSegment = mock(JoinTableSegment.class);
99          SimpleTableSegment leftTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")));
100         leftTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("o")));
101         SimpleTableSegment rightTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order_item")));
102         rightTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("i")));
103         when(joinTableSegment.getLeft()).thenReturn(leftTable);
104         when(joinTableSegment.getRight()).thenReturn(rightTable);
105         when(joinTableSegment.isNatural()).thenReturn(true);
106         when(joinTableSegment.getJoinType()).thenReturn(JoinType.RIGHT.name());
107         ShardingSphereMetaData metaData = createMetaData();

115         assertThat(((SimpleTableSegment) actual.getRight()).getTableName().getOriginalSchema().getValue(), is(DefaultDatabase.LOGIC_NAME));

146     private ShardingSphereMetaData createMetaData() {
147         ShardingSphereSchema schema = mock(ShardingSphereSchema.class, RETURNS_DEEP_STUBS);
148         when(schema.getTable("t_order").getColumnValues()).thenReturn(Arrays.asList(
149                 new ShardingSphereColumn("order_id", Types.INTEGER, true, false, false, true, false),
150                 new ShardingSphereColumn("user_id", Types.INTEGER, false, false, false, true, false),
151                 new ShardingSphereColumn("status", Types.INTEGER, false, false, false, true, false)));
152         when(schema.getTable("t_order_item").getColumnValues()).thenReturn(Arrays.asList(
153                 new ShardingSphereColumn("item_id", Types.INTEGER, true, false, false, true, false),
154                 new ShardingSphereColumn("order_id", Types.INTEGER, false, false, false, true, false),
155                 new ShardingSphereColumn("user_id", Types.INTEGER, false, false, false, true, false),
156                 new ShardingSphereColumn("status", Types.INTEGER, false, false, false, true, false)));
157         ShardingSphereMetaData result = mock(ShardingSphereMetaData.class, RETURNS_DEEP_STUBS);
158         when(result.getDatabase(DefaultDatabase.LOGIC_NAME).getSchema(DefaultDatabase.LOGIC_NAME)).thenReturn(schema);
159         return result;
160     }
