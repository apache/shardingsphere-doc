20  import io.netty.channel.Channel;
21  import lombok.Getter;
22  import org.apache.shardingsphere.db.protocol.codec.DatabasePacketCodecEngine;
23  import org.apache.shardingsphere.db.protocol.mysql.codec.MySQLPacketCodecEngine;
24  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLConstants;
25  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLServerInfo;
26  import org.apache.shardingsphere.db.protocol.mysql.packet.MySQLPacket;
27  import org.apache.shardingsphere.infra.config.props.ConfigurationPropertyKey;
28  import org.apache.shardingsphere.proxy.backend.context.ProxyContext;
29  import org.apache.shardingsphere.proxy.backend.session.ConnectionSession;
30  import org.apache.shardingsphere.proxy.frontend.authentication.AuthenticationEngine;
31  import org.apache.shardingsphere.proxy.frontend.command.CommandExecuteEngine;
32  import org.apache.shardingsphere.proxy.frontend.context.FrontendContext;
33  import org.apache.shardingsphere.proxy.frontend.mysql.authentication.MySQLAuthenticationEngine;
34  import org.apache.shardingsphere.proxy.frontend.mysql.command.MySQLCommandExecuteEngine;
35  import org.apache.shardingsphere.proxy.frontend.mysql.command.query.binary.MySQLStatementIDGenerator;
36  import org.apache.shardingsphere.proxy.frontend.spi.DatabaseProtocolFrontendEngine;
37  
38  import java.util.concurrent.atomic.AtomicInteger;
39  
40  /**
41   * Frontend engine for MySQL.
42   */
43  @Getter
44  public final class MySQLFrontendEngine implements DatabaseProtocolFrontendEngine {
45      
46      private final FrontendContext frontendContext = new MySQLFrontendContext();
47      
48      private final AuthenticationEngine authenticationEngine = new MySQLAuthenticationEngine();
49      
50      private final CommandExecuteEngine commandExecuteEngine = new MySQLCommandExecuteEngine();
51      
52      private final DatabasePacketCodecEngine<MySQLPacket> codecEngine = new MySQLPacketCodecEngine();
53      
54      public MySQLFrontendEngine() {
55          MySQLServerInfo.setDefaultMysqlVersion(ProxyContext.getInstance()
56                  .getContextManager().getMetaDataContexts().getMetaData().getProps().getValue(ConfigurationPropertyKey.PROXY_MYSQL_DEFAULT_VERSION));
57      }
58      
59      @Override
60      public void initChannel(final Channel channel) {
61          channel.attr(MySQLConstants.MYSQL_SEQUENCE_ID).set(new AtomicInteger());
62      }
63      
64      @Override
65      public void setDatabaseVersion(final String databaseName, final String databaseVersion) {
66          MySQLServerInfo.setServerVersion(databaseName, databaseVersion);
67      }
68      
69      @Override
70      public void release(final ConnectionSession connectionSession) {
71          MySQLStatementIDGenerator.getInstance().unregisterConnection(connectionSession.getConnectionId());
72      }
73      
74      @Override
75      public void handleException(final ConnectionSession connectionSession, final Exception exception) {
76      }
77      
78      @Override
79      public String getType() {
80          return "MySQL";
81      }
82  }
