59  @ExtendWith(AutoMockExtension.class)
60  @StaticMockSettings(PostgreSQLPositionInitializer.class)
61  class PostgreSQLWALDumperTest {
62      
63      @Mock
64      private PostgreSQLLogicalReplication logicalReplication;
65      
66      @Mock
67      private PgConnection pgConnection;
68      
69      @Mock
70      private PGReplicationStream pgReplicationStream;
71      
72      private WALPosition position;
73      
74      private DumperConfiguration dumperConfig;
75      
76      private PostgreSQLWALDumper walDumper;
77      
78      private MultiplexMemoryPipelineChannel channel;
79      
80      private final PipelineDataSourceManager dataSourceManager = new DefaultPipelineDataSourceManager();
81      
82      @BeforeEach
83      void setUp() {
84          position = new WALPosition(new PostgreSQLLogSequenceNumber(LogSequenceNumber.valueOf(100L)));
85          channel = new MultiplexMemoryPipelineChannel(1, 10000, new EmptyAckCallback());
86          String jdbcUrl = "jdbc:h2:mem:test;DB_CLOSE_DELAY=-1;DATABASE_TO_UPPER=false;MODE=PostgreSQL";
87          String username = "root";
88          String password = "root";
89          createTable(jdbcUrl, username, password);
90          dumperConfig = createDumperConfiguration(jdbcUrl, username, password);
91          walDumper = new PostgreSQLWALDumper(dumperConfig, position, channel, new StandardPipelineTableMetaDataLoader(dataSourceManager.getDataSource(dumperConfig.getDataSourceConfig())));
92      }
93      
94      private static void createTable(final String jdbcUrl, final String username, final String password) {
95          String sql = "CREATE TABLE t_order_0 (order_id INT NOT NULL, user_id INT NOT NULL, status VARCHAR(45) NULL, PRIMARY KEY (order_id))";
96          try (
97                  Connection connection = DriverManager.getConnection(jdbcUrl, username, password);
98                  Statement statement = connection.createStatement()) {
99              statement.execute(sql);
100         } catch (final SQLException ex) {
101             throw new RuntimeException("Init table failed.", ex);
102         }
103     }
104     
105     private DumperConfiguration createDumperConfiguration(final String jdbcUrl, final String username, final String password) {
106         DumperConfiguration result = new DumperConfiguration();
107         result.setJobId("0101123456");
108         result.setDataSourceConfig(new StandardPipelineDataSourceConfiguration(jdbcUrl, username, password));
109         result.setTableNameMap(Collections.singletonMap(new ActualTableName("t_order_0"), new LogicTableName("t_order")));
110         result.setTableNameSchemaNameMapping(new TableNameSchemaNameMapping(Collections.emptyMap()));
111         return result;
112     }
113     
114     @AfterEach
115     void tearDown() {
116         dataSourceManager.close();
117     }
118     
119     @Test
120     void assertStart() throws SQLException, ReflectiveOperationException {
121         StandardPipelineDataSourceConfiguration dataSourceConfig = (StandardPipelineDataSourceConfiguration) dumperConfig.getDataSourceConfig();
122         try {
123             Plugins.getMemberAccessor().set(PostgreSQLWALDumper.class.getDeclaredField("logicalReplication"), walDumper, logicalReplication);
124             when(logicalReplication.createConnection(dataSourceConfig)).thenReturn(pgConnection);
125             when(pgConnection.unwrap(PgConnection.class)).thenReturn(pgConnection);
126             when(PostgreSQLPositionInitializer.getUniqueSlotName(eq(pgConnection), anyString())).thenReturn("0101123456");
127             when(logicalReplication.createReplicationStream(pgConnection, PostgreSQLPositionInitializer.getUniqueSlotName(pgConnection, ""), position.getLogSequenceNumber()))
128                     .thenReturn(pgReplicationStream);
129             ByteBuffer data = ByteBuffer.wrap("table public.t_order_0: DELETE: order_id[integer]:1".getBytes());
130             when(pgReplicationStream.readPending()).thenReturn(null).thenReturn(data).thenThrow(new SQLException(""));
131             when(pgReplicationStream.getLastReceiveLSN()).thenReturn(LogSequenceNumber.valueOf(101L));
132             // TODO NPE occurred here
133             walDumper.start();
134         } catch (final IngestException ignored) {
135         }
136         assertThat(channel.fetchRecords(100, 0).size(), is(1));
137     }
138 }
