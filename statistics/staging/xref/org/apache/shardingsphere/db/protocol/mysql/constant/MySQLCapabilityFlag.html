18  package org.apache.shardingsphere.db.protocol.mysql.constant;
19  
20  import lombok.Getter;
21  import lombok.RequiredArgsConstructor;
22  
23  /**
24   * Capability flag for MySQL.
25   * 
26   * @see <a href="https://dev.mysql.com/doc/dev/mysql-server/latest/group__group__cs__capabilities__flags.html">Capabilities Flags</a>
27   */
28  @RequiredArgsConstructor
29  @Getter
30  public enum MySQLCapabilityFlag {
31      
32      CLIENT_LONG_PASSWORD(0x00000001),
33      
34      CLIENT_FOUND_ROWS(0x00000002),
35      
36      CLIENT_LONG_FLAG(0x00000004),
37      
38      CLIENT_CONNECT_WITH_DB(0x00000008),
39      
40      CLIENT_NO_SCHEMA(0x00000010),
41      
42      CLIENT_COMPRESS(0x00000020),
43      
44      CLIENT_ODBC(0x00000040),
45      
46      CLIENT_LOCAL_FILES(0x00000080),
47      
48      CLIENT_IGNORE_SPACE(0x00000100),
49      
50      CLIENT_PROTOCOL_41(0x00000200),
51      
52      CLIENT_INTERACTIVE(0x00000400),
53      
54      CLIENT_SSL(0x00000800),
55      
56      CLIENT_IGNORE_SIGPIPE(0x00001000),
57      
58      CLIENT_TRANSACTIONS(0x00002000),
59      
60      CLIENT_RESERVED(0x00004000),
61      
62      CLIENT_SECURE_CONNECTION(0x00008000),
63      
64      CLIENT_MULTI_STATEMENTS(0x00010000),
65      
66      CLIENT_MULTI_RESULTS(0x00020000),
67      
68      CLIENT_PS_MULTI_RESULTS(0x00040000),
69      
70      CLIENT_PLUGIN_AUTH(0x00080000),
71      
72      CLIENT_CONNECT_ATTRS(0x00100000),
73      
74      CLIENT_PLUGIN_AUTH_LENENC_CLIENT_DATA(0x00200000),
75      
76      CLIENT_CAN_HANDLE_EXPIRED_PASSWORDS(0x00400000),
77      
78      CLIENT_SESSION_TRACK(0x00800000),
79      
80      CLIENT_DEPRECATE_EOF(0x01000000);
81      
82      private final int value;
83      
84      /**
85       * Get handshake capability flags lower bit.
86       *
87       * @return handshake capability flags lower bit
88       */
89      public static int calculateHandshakeCapabilityFlagsLower() {
90          return calculateCapabilityFlags(CLIENT_LONG_PASSWORD, CLIENT_FOUND_ROWS, CLIENT_LONG_FLAG, CLIENT_CONNECT_WITH_DB, CLIENT_ODBC, CLIENT_IGNORE_SPACE,
91                  CLIENT_PROTOCOL_41, CLIENT_INTERACTIVE, CLIENT_IGNORE_SIGPIPE, CLIENT_TRANSACTIONS, CLIENT_SECURE_CONNECTION) & 0x0000ffff;
92      }
93      
94      /**
95       * Get handshake capability flags upper bit.
96       *
97       * @return handshake capability flags upper bit
98       */
99      public static int calculateHandshakeCapabilityFlagsUpper() {
100         return calculateCapabilityFlags(CLIENT_MULTI_STATEMENTS, CLIENT_PLUGIN_AUTH) >> 16;
101     }
102     
103     /**
104      * Calculate capability flags.
105      *
106      * @param capabilities single capabilities of need to be calculated
107      * @return combined capabilities
108      */
109     public static int calculateCapabilityFlags(final MySQLCapabilityFlag... capabilities) {
110         int result = 0;
111         for (MySQLCapabilityFlag each : capabilities) {
112             result |= each.value;
113         }
114         return result;
115     }
116 }
