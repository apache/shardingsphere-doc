56      @Test
57      void assertGetSubquerySegmentsInWhere() {
58          SelectStatement subquerySelectStatement = mock(SelectStatement.class);
59          when(subquerySelectStatement.getFrom()).thenReturn(Optional.of(new SimpleTableSegment(new TableNameSegment(73, 99, new IdentifierValue("t_order")))));
60          ProjectionsSegment subqueryProjections = new ProjectionsSegment(59, 66);
61          when(subquerySelectStatement.getProjections()).thenReturn(subqueryProjections);
62          subqueryProjections.getProjections().add(new ColumnProjectionSegment(new ColumnSegment(59, 66, new IdentifierValue("order_id"))));
63          ColumnSegment subqueryWhereLeft = new ColumnSegment(87, 92, new IdentifierValue("status"));
64          LiteralExpressionSegment subqueryWhereRight = new LiteralExpressionSegment(96, 99, "OK");
65          WhereSegment subqueryWhereSegment = new WhereSegment(81, 99, new BinaryOperationExpression(87, 99, subqueryWhereLeft, subqueryWhereRight, "=", "status = 'OK'"));
66          when(subquerySelectStatement.getWhere()).thenReturn(Optional.of(subqueryWhereSegment));
67          SelectStatement selectStatement = mock(SelectStatement.class);
68          when(selectStatement.getFrom()).thenReturn(Optional.of(new SimpleTableSegment(new TableNameSegment(21, 32, new IdentifierValue("t_order_item")))));
69          when(selectStatement.getProjections()).thenReturn(new ProjectionsSegment(7, 14));
70          selectStatement.getProjections().getProjections().add(new ColumnProjectionSegment(new ColumnSegment(7, 14, new IdentifierValue("order_id"))));
71          ColumnSegment left = new ColumnSegment(40, 47, new IdentifierValue("order_id"));
72          SubqueryExpressionSegment right = new SubqueryExpressionSegment(new SubquerySegment(51, 100, subquerySelectStatement, ""));
73          WhereSegment whereSegment = new WhereSegment(34, 100, new BinaryOperationExpression(40, 100, left, right, "=", "order_id = (SELECT order_id FROM t_order WHERE status = 'OK')"));
74          when(selectStatement.getWhere()).thenReturn(Optional.of(whereSegment));
75          Collection<SubquerySegment> actual = SubqueryExtractUtils.getSubquerySegments(selectStatement, true);
76          assertThat(actual.size(), is(1));
77          assertThat(actual.iterator().next(), is(right.getSubquery()));
78          assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, false).size(), is(1));
79      }

121     @Test
122     void assertGetSubquerySegmentsInFrom2() {
123         SelectStatement subqueryLeftSelectStatement = mock(SelectStatement.class);
124         when(subqueryLeftSelectStatement.getFrom()).thenReturn(Optional.of(new SimpleTableSegment(new TableNameSegment(65, 71, new IdentifierValue("t_order")))));
125         ColumnSegment leftColumnSegment = new ColumnSegment(79, 84, new IdentifierValue("status"));
126         LiteralExpressionSegment leftLiteralExpressionSegment = new LiteralExpressionSegment(88, 91, "OK");
127         BinaryOperationExpression leftStatusCondition = new BinaryOperationExpression(79, 91, leftColumnSegment, leftLiteralExpressionSegment, "=", "status = OK");
128         when(subqueryLeftSelectStatement.getWhere()).thenReturn(Optional.of(new WhereSegment(73, 91, leftStatusCondition)));
129         ProjectionsSegment subqueryLeftProjections = new ProjectionsSegment(34, 58);
130         when(subqueryLeftSelectStatement.getProjections()).thenReturn(subqueryLeftProjections);
131         subqueryLeftProjections.getProjections().add(new ColumnProjectionSegment(new ColumnSegment(34, 41, new IdentifierValue("order_id"))));
132         subqueryLeftProjections.getProjections().add(new AggregationProjectionSegment(44, 51, AggregationType.COUNT, "COUNT(*)"));
133         SelectStatement subqueryRightSelectStatement = mock(SelectStatement.class);
134         when(subqueryRightSelectStatement.getFrom()).thenReturn(Optional.of(new SimpleTableSegment(new TableNameSegment(143, 154, new IdentifierValue("t_order_item")))));
135         ColumnSegment rightColumnSegment = new ColumnSegment(162, 167, new IdentifierValue("status"));
136         LiteralExpressionSegment rightLiteralExpressionSegment = new LiteralExpressionSegment(171, 174, "OK");
137         BinaryOperationExpression rightStatusCondition = new BinaryOperationExpression(162, 174, rightColumnSegment, rightLiteralExpressionSegment, "=", "status = OK");
138         when(subqueryRightSelectStatement.getWhere()).thenReturn(Optional.of(new WhereSegment(156, 174, rightStatusCondition)));
139         ProjectionsSegment subqueryRightProjections = new ProjectionsSegment(112, 136);
140         when(subqueryRightSelectStatement.getProjections()).thenReturn(subqueryRightProjections);
141         subqueryRightProjections.getProjections().add(new ColumnProjectionSegment(new ColumnSegment(112, 119, new IdentifierValue("order_id"))));
142         subqueryRightProjections.getProjections().add(new AggregationProjectionSegment(122, 129, AggregationType.COUNT, "COUNT(*)"));
143         SelectStatement selectStatement = mock(SelectStatement.class);
144         ProjectionsSegment projections = new ProjectionsSegment(7, 19);
145         when(selectStatement.getProjections()).thenReturn(projections);
146         projections.getProjections().add(new ColumnProjectionSegment(new ColumnSegment(7, 11, new IdentifierValue("cnt"))));
147         projections.getProjections().add(new ColumnProjectionSegment(new ColumnSegment(14, 19, new IdentifierValue("cnt"))));
148         JoinTableSegment from = new JoinTableSegment();
149         from.setStartIndex(26);
150         from.setStopIndex(213);
151         ColumnSegment columnSegment1 = new ColumnSegment(190, 199, new IdentifierValue("order_id"));
152         ColumnSegment columnSegment2 = new ColumnSegment(203, 213, new IdentifierValue("order_id"));
153         BinaryOperationExpression orderIdCondition = new BinaryOperationExpression(190, 213, columnSegment1, columnSegment2, "=", "o.order_id = oi.order_id");
154         from.setCondition(orderIdCondition);
155         SubqueryTableSegment leftSubquerySegment = new SubqueryTableSegment(0, 0, new SubquerySegment(26, 92, subqueryLeftSelectStatement, ""));
156         SubqueryTableSegment rightSubquerySegment = new SubqueryTableSegment(0, 0, new SubquerySegment(104, 175, subqueryRightSelectStatement, ""));
157         from.setLeft(leftSubquerySegment);
158         from.setRight(rightSubquerySegment);
159         when(selectStatement.getFrom()).thenReturn(Optional.of(from));
160         Collection<SubquerySegment> actual = SubqueryExtractUtils.getSubquerySegments(selectStatement, true);
161         assertThat(actual.size(), is(2));
162         Iterator<SubquerySegment> iterator = actual.iterator();
163         assertThat(iterator.next(), is(leftSubquerySegment.getSubquery()));
164         assertThat(iterator.next(), is(rightSubquerySegment.getSubquery()));
165         assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, false).size(), is(2));
166     }

168     @Test
169     void assertGetSubquerySegmentsWithMultiNestedSubquery() {
170         SelectStatement selectStatement = mock(SelectStatement.class);
171         SubquerySegment subquerySelect = createSubquerySegment();
172         when(selectStatement.getFrom()).thenReturn(Optional.of(new SubqueryTableSegment(0, 0, subquerySelect)));
173         assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, true).size(), is(2));
174         assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, false).size(), is(1));
175     }

177     @Test
178     void assertGetSubquerySegmentsWithCombineSegment() {
179         SelectStatement selectStatement = mock(SelectStatement.class);
180         SubquerySegment left = new SubquerySegment(0, 0, mock(SelectStatement.class), "");
181         SubquerySegment right = createSubquerySegment();
182         when(selectStatement.getCombine()).thenReturn(Optional.of(new CombineSegment(0, 0, left, CombineType.UNION, right)));
183         assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, true).size(), is(3));
184         assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, false).size(), is(2));
185     }

187     private SubquerySegment createSubquerySegment() {
188         SelectStatement selectStatement = mock(SelectStatement.class);
189         ExpressionSegment left = new ColumnSegment(0, 0, new IdentifierValue("order_id"));
190         when(selectStatement.getWhere()).thenReturn(
191                 Optional.of(new WhereSegment(0, 0, new InExpression(0, 0, left, new SubqueryExpressionSegment(new SubquerySegment(0, 0, mock(SelectStatement.class), "")), false))));
192         return new SubquerySegment(0, 0, selectStatement, "");
193     }

195     @Test
196     void assertGetSubquerySegmentsFromProjectionFunctionParams() {
197         SelectStatement selectStatement = mock(SelectStatement.class);
198         ProjectionsSegment projections = new ProjectionsSegment(0, 0);
199         when(selectStatement.getProjections()).thenReturn(projections);
200         FunctionSegment functionSegment = new FunctionSegment(0, 0, "", "");
201         functionSegment.getParameters().add(new SubqueryExpressionSegment(new SubquerySegment(0, 0, mock(SelectStatement.class), "")));
202         ExpressionProjectionSegment expressionProjectionSegment = new ExpressionProjectionSegment(0, 0, "", functionSegment);
203         projections.getProjections().add(expressionProjectionSegment);
204         assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, true).size(), is(1));
205         assertThat(SubqueryExtractUtils.getSubquerySegments(selectStatement, false).size(), is(1));
206     }
