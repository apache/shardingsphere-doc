50  /**
51   * Record single table inventory calculator.
52   */
53  @HighFrequencyInvocation
54  @RequiredArgsConstructor
55  public final class RecordSingleTableInventoryCalculator extends AbstractStreamingSingleTableInventoryCalculator {
56      
57      private final int chunkSize;
58      
59      @Override
60      public Optional<SingleTableInventoryCalculatedResult> calculateChunk(final SingleTableInventoryCalculateParameter param) {
61          List<Map<String, Object>> records = calculateChunk(param, QueryType.RANGE_QUERY == param.getQueryType());
62          if (records.isEmpty()) {
63              return Optional.empty();
64          }
65          String firstUniqueKey = param.getFirstUniqueKey().getName();
66          if (QueryType.POINT_QUERY == param.getQueryType()) {
67              return convertRecordsToResult(records, firstUniqueKey);
68          }
69          if (records.size() == chunkSize) {
70              Object minUniqueKeyValue = DataConsistencyCheckUtils.getFirstUniqueKeyValue(records.get(0), firstUniqueKey);
71              removeLastRecords(records, param);
72              if (!records.isEmpty()) {
73                  updateQueryRangeLower(param, records, firstUniqueKey);
74                  return convertRecordsToResult(records, firstUniqueKey);
75              }
76              SingleTableInventoryCalculateParameter newParam = buildNewCalculateParameter(param, minUniqueKeyValue);
77              records = calculateChunk(newParam, false);
78              if (!records.isEmpty()) {
79                  updateQueryRangeLower(param, records, firstUniqueKey);
80                  return convertRecordsToResult(records, firstUniqueKey);
81              }
82              return Optional.empty();
83          }
84          updateQueryRangeLower(param, records, firstUniqueKey);
85          return convertRecordsToResult(records, firstUniqueKey);
86      }
87      
88      private List<Map<String, Object>> calculateChunk(final SingleTableInventoryCalculateParameter param, final boolean isRangeQuery) {
89          try (CalculationContext calculationContext = getOrCreateCalculationContext(param)) {
90              List<Map<String, Object>> result = new LinkedList<>();
91              InventoryColumnValueReaderEngine columnValueReaderEngine = new InventoryColumnValueReaderEngine(param.getDatabaseType());
92              ResultSet resultSet = calculationContext.getResultSet();
93              ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
94              while (resultSet.next()) {
95                  ShardingSpherePreconditions.checkState(!isCanceling(), () -> new PipelineJobCancelingException(
96                          "Calculate chunk canceled, schema name: %s, table name: %s", param.getSchemaName(), param.getLogicTableName()));
97                  Map<String, Object> columnRecord = new LinkedHashMap<>();
98                  for (int columnIndex = 1, columnCount = resultSetMetaData.getColumnCount(); columnIndex <= columnCount; columnIndex++) {
99                      columnRecord.put(resultSetMetaData.getColumnLabel(columnIndex), columnValueReaderEngine.read(resultSet, resultSetMetaData, columnIndex));
100                 }
101                 result.add(columnRecord);
102                 if (isRangeQuery && result.size() == chunkSize) {
103                     break;
104                 }
105             }
106             return result;
107         } catch (final PipelineSQLException | PipelineJobCancelingException ex) {
108             throw ex;
109             // CHECKSTYLE:OFF
110         } catch (final SQLException | RuntimeException ex) {
111             // CHECKSTYLE:ON
112             throw new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), ex);
113         }
114     }
115     
116     private CalculationContext getOrCreateCalculationContext(final SingleTableInventoryCalculateParameter param) {
117         CalculationContext result = (CalculationContext) param.getCalculationContext();
118         if (null != result && !result.isClosed()) {
119             return result;
120         }
121         try {
122             result = createCalculationContext(param);
123             fulfillCalculationContext(result, param);
124             // CHECKSTYLE:OFF
125         } catch (final SQLException | RuntimeException ex) {
126             // CHECKSTYLE:ON
127             QuietlyCloser.close(result);
128             throw new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), ex);
129         }
130         return result;
131     }
132     
133     private CalculationContext createCalculationContext(final SingleTableInventoryCalculateParameter param) throws SQLException {
134         Connection connection = param.getDataSource().getConnection();
135         CalculationContext result = new CalculationContext();
136         result.setConnection(connection);
137         param.setCalculationContext(result);
138         return result;
139     }
140     
141     private void fulfillCalculationContext(final CalculationContext calculationContext, final SingleTableInventoryCalculateParameter param) throws SQLException {
142         String sql = getQuerySQL(param);
143         PreparedStatement preparedStatement = JDBCStreamQueryBuilder.build(param.getDatabaseType(), calculationContext.getConnection(), sql, chunkSize);
144         setCurrentStatement(preparedStatement);
145         calculationContext.setPreparedStatement(preparedStatement);
146         setParameters(preparedStatement, param);
147         ResultSet resultSet = preparedStatement.executeQuery();
148         calculationContext.setResultSet(resultSet);
149     }
150     
151     private String getQuerySQL(final SingleTableInventoryCalculateParameter param) {
152         ShardingSpherePreconditions.checkNotNull(param.getFirstUniqueKey(),
153                 () -> new UnsupportedOperationException("Record inventory calculator does not support table without unique key and primary key now."));
154         PipelineDataConsistencyCalculateSQLBuilder pipelineSQLBuilder = new PipelineDataConsistencyCalculateSQLBuilder(param.getDatabaseType());
155         Collection<String> columnNames = param.getColumnNames().isEmpty() ? Collections.singleton("*") : param.getColumnNames();
156         switch (param.getQueryType()) {
157             case RANGE_QUERY:
158                 return pipelineSQLBuilder.buildQueryRangeOrderingSQL(param.getSchemaName(), param.getLogicTableName(),
159                         columnNames, param.getUniqueKeysNames(), param.getQueryRange(), param.getShardingColumnsNames());
160             case POINT_QUERY:
161                 return pipelineSQLBuilder.buildPointQuerySQL(param.getSchemaName(), param.getLogicTableName(), columnNames, param.getUniqueKeysNames(), param.getShardingColumnsNames());
162             default:
163                 throw new UnsupportedOperationException("Query type: " + param.getQueryType());
164         }
165     }
166     
167     private void setParameters(final PreparedStatement preparedStatement, final SingleTableInventoryCalculateParameter param) throws SQLException {
168         QueryType queryType = param.getQueryType();
169         if (queryType == QueryType.RANGE_QUERY) {
170             QueryRange queryRange = param.getQueryRange();
171             ShardingSpherePreconditions.checkNotNull(queryRange,
172                     () -> new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), new RuntimeException("Unique keys values range is null.")));
173             int parameterIndex = 1;
174             if (null != queryRange.getLower()) {
175                 preparedStatement.setObject(parameterIndex++, queryRange.getLower());
176             }
177             if (null != queryRange.getUpper()) {
178                 preparedStatement.setObject(parameterIndex++, queryRange.getUpper());
179             }
180             preparedStatement.setObject(parameterIndex, chunkSize);
181         } else if (queryType == QueryType.POINT_QUERY) {
182             Collection<Object> uniqueKeysValues = param.getUniqueKeysValues();
183             ShardingSpherePreconditions.checkNotNull(uniqueKeysValues,
184                     () -> new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), new RuntimeException("Unique keys values is null.")));
185             int parameterIndex = 1;
186             for (Object each : uniqueKeysValues) {
187                 preparedStatement.setObject(parameterIndex++, each);
188             }
189             if (null != param.getShardingColumnsNames() && !param.getShardingColumnsNames().isEmpty()) {
190                 List<Object> shardingColumnsValues = param.getShardingColumnsValues();
191                 ShardingSpherePreconditions.checkNotNull(shardingColumnsValues, () -> new PipelineTableDataConsistencyCheckLoadingFailedException(
192                         param.getSchemaName(), param.getLogicTableName(), new RuntimeException("Sharding columns values is null when names not empty.")));
193                 for (Object each : shardingColumnsValues) {
194                     preparedStatement.setObject(parameterIndex++, each);
195                 }
196             }
197         } else {
198             throw new UnsupportedOperationException("Query type: " + queryType);
199         }
200     }
201     
202     private void removeLastRecords(final List<Map<String, Object>> records, final SingleTableInventoryCalculateParameter param) {
203         Object minUniqueKeyValue = DataConsistencyCheckUtils.getFirstUniqueKeyValue(records.get(0), param.getFirstUniqueKey().getName());
204         Object maxUniqueKeyValue = DataConsistencyCheckUtils.getFirstUniqueKeyValue(records.get(records.size() - 1), param.getFirstUniqueKey().getName());
205         if (Objects.equals(minUniqueKeyValue, maxUniqueKeyValue)) {
206             records.clear();
207             return;
208         }
209         records.remove(records.size() - 1);
210         for (int i = records.size() - 1; i >= 0; i--) {
211             if (Objects.deepEquals(maxUniqueKeyValue, DataConsistencyCheckUtils.getFirstUniqueKeyValue(records.get(i), param.getFirstUniqueKey().getName()))) {
212                 records.remove(i);
213             } else {
214                 break;
215             }
216         }
217     }
218     
219     private Optional<SingleTableInventoryCalculatedResult> convertRecordsToResult(final List<Map<String, Object>> records, final String firstUniqueKey) {
220         Object maxUniqueKeyValue = DataConsistencyCheckUtils.getFirstUniqueKeyValue(records.get(records.size() - 1), firstUniqueKey);
221         return Optional.of(new RecordSingleTableInventoryCalculatedResult(maxUniqueKeyValue, records));
222     }
223     
224     private SingleTableInventoryCalculateParameter buildNewCalculateParameter(final SingleTableInventoryCalculateParameter param, final Object minUniqueKeyValue) {
225         SingleTableInventoryCalculateParameter result = new SingleTableInventoryCalculateParameter(param.getDataSource(), param.getTable(), param.getColumnNames(),
226                 Collections.singletonList(param.getFirstUniqueKey()), QueryType.POINT_QUERY);
227         result.setUniqueKeysValues(Collections.singletonList(minUniqueKeyValue));
228         result.setShardingColumnsNames(param.getShardingColumnsNames());
229         result.setShardingColumnsValues(param.getShardingColumnsValues());
230         return result;
231     }
232     
233     private void updateQueryRangeLower(final SingleTableInventoryCalculateParameter param, final List<Map<String, Object>> records, final String firstUniqueKey) {
234         Object maxUniqueKeyValue = DataConsistencyCheckUtils.getFirstUniqueKeyValue(records.get(records.size() - 1), firstUniqueKey);
235         param.setQueryRange(new QueryRange(maxUniqueKeyValue, false, param.getQueryRange().getUpper()));
236     }
