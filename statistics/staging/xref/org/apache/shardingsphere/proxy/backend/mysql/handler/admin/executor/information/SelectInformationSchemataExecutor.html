45  /**
46   * Schemata query executor, used to query the schemata table.
47   */
48  public final class SelectInformationSchemataExecutor extends DefaultDatabaseMetaDataExecutor {
49      
50      public static final String SCHEMA_NAME = "SCHEMA_NAME";
51      
52      public static final String DEFAULT_CHARACTER_SET_NAME = "DEFAULT_CHARACTER_SET_NAME";
53      
54      public static final String DEFAULT_COLLATION_NAME = "DEFAULT_COLLATION_NAME";
55      
56      public static final String CATALOG_NAME = "CATALOG_NAME";
57      
58      public static final String SQL_PATH = "SQL_PATH";
59      
60      public static final String DEFAULT_ENCRYPTION = "DEFAULT_ENCRYPTION";
61      
62      private static final Collection<String> SCHEMA_WITHOUT_DATA_SOURCE = new LinkedHashSet<>();
63      
64      private final SelectStatement sqlStatement;
65      
66      private String schemaNameAlias = SCHEMA_NAME;
67      
68      private boolean queryDatabase;
69      
70      public SelectInformationSchemataExecutor(final SelectStatement sqlStatement, final String sql, final List<Object> parameters) {
71          super(sql, parameters);
72          this.sqlStatement = sqlStatement;
73      }
74      
75      @Override
76      protected void postProcess() {
77          removeDuplicatedRow();
78      }
79      
80      private void removeDuplicatedRow() {
81          if (queryDatabase) {
82              Collection<Map<String, Object>> reservedRow = getRows().stream()
83                      .collect(Collectors.groupingBy(each -> Optional.ofNullable(each.get(schemaNameAlias)), Collectors.toCollection(LinkedList::new)))
84                      .values().stream().map(LinkedList::getFirst).collect(Collectors.toList());
85              reservedRow.forEach(each -> getRows().removeIf(row -> !getRows().contains(each)));
86          }
87      }
88      
89      @Override
90      protected Collection<String> getDatabaseNames(final ConnectionSession connectionSession) {
91          Collection<String> databaseNames = ProxyContext.getInstance().getAllDatabaseNames().stream()
92                  .filter(each -> isAuthorized(each, connectionSession.getConnectionContext().getGrantee())).collect(Collectors.toList());
93          SCHEMA_WITHOUT_DATA_SOURCE.addAll(databaseNames.stream().filter(each -> !hasDataSource(each)).collect(Collectors.toSet()));
94          Collection<String> result = databaseNames.stream().filter(AbstractDatabaseMetaDataExecutor::hasDataSource).collect(Collectors.toList());
95          if (!SCHEMA_WITHOUT_DATA_SOURCE.isEmpty()) {
96              fillSchemasWithoutDataSource();
97          }
98          return result;
99      }
100     
101     private void fillSchemasWithoutDataSource() {
102         if (SCHEMA_WITHOUT_DATA_SOURCE.isEmpty()) {
103             return;
104         }
105         Map<String, String> defaultRowData = getTheDefaultRowData();
106         SCHEMA_WITHOUT_DATA_SOURCE.forEach(each -> {
107             Map<String, Object> row = new LinkedHashMap<>(defaultRowData);
108             row.replace(SCHEMA_NAME, each);
109             getRows().add(row);
110         });
111         SCHEMA_WITHOUT_DATA_SOURCE.clear();
112     }
113     
114     private Map<String, String> getTheDefaultRowData() {
115         Map<String, String> result;
116         Collection<ProjectionSegment> projections = sqlStatement.getProjections().getProjections();
117         if (projections.stream().anyMatch(ShorthandProjectionSegment.class::isInstance)) {
118             result = Stream.of(CATALOG_NAME, SCHEMA_NAME, DEFAULT_CHARACTER_SET_NAME, DEFAULT_COLLATION_NAME, SQL_PATH, DEFAULT_ENCRYPTION).collect(Collectors.toMap(each -> each, each -> ""));
119         } else {
120             result = projections.stream().map(each -> ((ColumnProjectionSegment) each).getColumn().getIdentifier())
121                     .map(each -> each.getValue().toUpperCase()).collect(Collectors.toMap(each -> each, each -> ""));
122         }
123         return result;
124     }
125     
126     @Override
127     protected void preProcess(final String databaseName, final Map<String, Object> rows, final Map<String, String> alias) throws SQLException {
128         ResourceMetaData resourceMetaData = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getDatabase(databaseName).getResourceMetaData();
129         Collection<String> catalogs = getCatalogs(resourceMetaData);
130         schemaNameAlias = alias.getOrDefault(SCHEMA_NAME, "");
131         String rowValue = rows.getOrDefault(schemaNameAlias, "").toString();
132         queryDatabase = !rowValue.isEmpty();
133         if (catalogs.contains(rowValue)) {
134             rows.replace(schemaNameAlias, databaseName);
135         } else {
136             rows.clear();
137         }
138     }
139     
140     private Collection<String> getCatalogs(final ResourceMetaData resourceMetaData) throws SQLException {
141         Collection<String> result = new HashSet<>(resourceMetaData.getStorageUnits().size(), 1F);
142         for (Entry<String, StorageUnit> entry : resourceMetaData.getStorageUnits().entrySet()) {
143             try (Connection connection = entry.getValue().getDataSource().getConnection()) {
144                 result.add(connection.getCatalog());
145             }
146         }
147         return result;
148     }
149 }
