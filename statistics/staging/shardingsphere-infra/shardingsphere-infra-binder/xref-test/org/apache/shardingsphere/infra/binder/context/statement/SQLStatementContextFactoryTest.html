67  class SQLStatementContextFactoryTest {
68      
69      @Test
70      void assertSQLStatementContextCreatedWhenSQLStatementInstanceOfSelectStatement() {
71          ProjectionsSegment projectionsSegment = new ProjectionsSegment(0, 0);
72          MySQLSelectStatement selectStatement = new MySQLSelectStatement();
73          selectStatement.setLimit(new LimitSegment(0, 10, null, null));
74          selectStatement.setProjections(projectionsSegment);
75          SQLStatementContext sqlStatementContext = new SQLBindEngine(mockMetaData(), DefaultDatabase.LOGIC_NAME).bind(selectStatement, Collections.emptyList());
76          assertThat(sqlStatementContext, instanceOf(SelectStatementContext.class));
77      }
78      
79      @Test
80      void assertSQLStatementContextCreatedWhenSQLStatementInstanceOfMySQLInsertStatement() {
81          MySQLInsertStatement insertStatement = new MySQLInsertStatement();
82          List<ColumnSegment> columnSegments = new LinkedList<>();
83          columnSegments.add(new ColumnSegment(0, 0, new IdentifierValue("IdentifierValue")));
84          AssignmentSegment assignment = new ColumnAssignmentSegment(0, 0, columnSegments, null);
85          insertStatement.setSetAssignment(new SetAssignmentSegment(0, 0,
86                  Collections.singleton(assignment)));
87          assertSQLStatementContextCreatedWhenSQLStatementInstanceOfInsertStatement(insertStatement);
88      }
89      
90      @Test
91      void assertSQLStatementContextCreatedWhenSQLStatementInstanceOfOracleInsertStatement() {
92          assertSQLStatementContextCreatedWhenSQLStatementInstanceOfInsertStatement(new OracleInsertStatement());
93      }
94      
95      @Test

98      }

110     private void assertSQLStatementContextCreatedWhenSQLStatementInstanceOfInsertStatement(final InsertStatement insertStatement) {
111         insertStatement.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("tbl"))));
112         SQLStatementContext sqlStatementContext = new SQLBindEngine(mockMetaData(), DefaultDatabase.LOGIC_NAME).bind(insertStatement, Collections.emptyList());
113         assertThat(sqlStatementContext, instanceOf(InsertStatementContext.class));
114     }
115     
116     @Test
117     void assertSQLStatementContextCreatedWhenSQLStatementNotInstanceOfSelectStatementAndInsertStatement() {
118         SQLStatementContext sqlStatementContext = new SQLBindEngine(mockMetaData(), DefaultDatabase.LOGIC_NAME).bind(mock(MySQLStatement.class), Collections.emptyList());
119         assertThat(sqlStatementContext, instanceOf(CommonSQLStatementContext.class));
120     }
121     
122     @Test
123     void assertNewInstanceForCursorStatement() {
124         OpenGaussCursorStatement cursorStatement = mock(OpenGaussCursorStatement.class, RETURNS_DEEP_STUBS);
125         MySQLSelectStatement selectStatement = mock(MySQLSelectStatement.class, RETURNS_DEEP_STUBS);
126         when(selectStatement.getProjections().isDistinctRow()).thenReturn(false);
127         when(cursorStatement.getSelect()).thenReturn(selectStatement);
128         SQLStatementContext actual = new SQLBindEngine(mockMetaData(), DefaultDatabase.LOGIC_NAME).bind(cursorStatement, Collections.emptyList());
129         assertThat(actual, instanceOf(CursorStatementContext.class));
130     }

150     private ShardingSphereMetaData mockMetaData() {
151         Map<String, ShardingSphereDatabase> databases = Collections.singletonMap(DefaultDatabase.LOGIC_NAME, mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS));
152         return new ShardingSphereMetaData(databases, mock(ShardingSphereResourceMetaData.class), mock(ShardingSphereRuleMetaData.class), mock(ConfigurationProperties.class));
153     }
