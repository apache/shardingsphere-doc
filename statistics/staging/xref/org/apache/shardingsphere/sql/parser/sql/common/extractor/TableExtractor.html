65  @Getter
66  public final class TableExtractor {
67      
68      private final Collection<SimpleTableSegment> rewriteTables = new LinkedList<>();
69      
70      private final Collection<TableSegment> tableContext = new LinkedList<>();
71      
72      private final Collection<JoinTableSegment> joinTables = new LinkedList<>();
73      
74      /**
75       * Extract table that should be rewritten from select statement.
76       *
77       * @param selectStatement select statement
78       */
79      public void extractTablesFromSelect(final SelectStatement selectStatement) {
80          if (selectStatement.getCombine().isPresent()) {
81              CombineSegment combineSegment = selectStatement.getCombine().get();
82              extractTablesFromSelect(combineSegment.getLeft().getSelect());
83              extractTablesFromSelect(combineSegment.getRight().getSelect());
84          }
85          if (selectStatement.getFrom().isPresent() && !selectStatement.getCombine().isPresent()) {
86              extractTablesFromTableSegment(selectStatement.getFrom().get());
87          }
88          selectStatement.getWhere().ifPresent(optional -> extractTablesFromExpression(optional.getExpr()));
89          if (null != selectStatement.getProjections() && !selectStatement.getCombine().isPresent()) {
90              extractTablesFromProjections(selectStatement.getProjections());
91          }
92          selectStatement.getGroupBy().ifPresent(optional -> extractTablesFromOrderByItems(optional.getGroupByItems()));
93          selectStatement.getOrderBy().ifPresent(optional -> extractTablesFromOrderByItems(optional.getOrderByItems()));
94          selectStatement.getHaving().ifPresent(optional -> extractTablesFromExpression(optional.getExpr()));
95          selectStatement.getWithSegment().ifPresent(optional -> extractTablesFromCTEs(optional.getCommonTableExpressions()));
96          selectStatement.getLock().ifPresent(this::extractTablesFromLock);
97      }
98      
99      private void extractTablesFromCTEs(final Collection<CommonTableExpressionSegment> commonTableExpressionSegments) {
100         for (CommonTableExpressionSegment each : commonTableExpressionSegments) {
101             extractTablesFromSelect(each.getSubquery().getSelect());
102         }
103     }
104     
105     private void extractTablesFromTableSegment(final TableSegment tableSegment) {
106         if (tableSegment instanceof SimpleTableSegment) {
107             tableContext.add(tableSegment);
108             rewriteTables.add((SimpleTableSegment) tableSegment);
109         }
110         if (tableSegment instanceof SubqueryTableSegment) {
111             tableContext.add(tableSegment);
112             TableExtractor tableExtractor = new TableExtractor();
113             tableExtractor.extractTablesFromSelect(((SubqueryTableSegment) tableSegment).getSubquery().getSelect());
114             rewriteTables.addAll(tableExtractor.rewriteTables);
115             joinTables.addAll(tableExtractor.joinTables);
116         }
117         if (tableSegment instanceof JoinTableSegment) {
118             joinTables.add((JoinTableSegment) tableSegment);
119             extractTablesFromJoinTableSegment((JoinTableSegment) tableSegment);
120         }
121         if (tableSegment instanceof DeleteMultiTableSegment) {
122             DeleteMultiTableSegment deleteMultiTableSegment = (DeleteMultiTableSegment) tableSegment;
123             rewriteTables.addAll(deleteMultiTableSegment.getActualDeleteTables());
124             extractTablesFromTableSegment(deleteMultiTableSegment.getRelationTable());
125         }
126     }
127     
128     private void extractTablesFromJoinTableSegment(final JoinTableSegment tableSegment) {
129         extractTablesFromTableSegment(tableSegment.getLeft());
130         extractTablesFromTableSegment(tableSegment.getRight());
131         extractTablesFromExpression(tableSegment.getCondition());
132     }
133     
134     private void extractTablesFromExpression(final ExpressionSegment expressionSegment) {
135         if (expressionSegment instanceof ColumnSegment && ((ColumnSegment) expressionSegment).getOwner().isPresent() && needRewrite(((ColumnSegment) expressionSegment).getOwner().get())) {
136             OwnerSegment ownerSegment = ((ColumnSegment) expressionSegment).getOwner().get();
137             rewriteTables.add(new SimpleTableSegment(new TableNameSegment(ownerSegment.getStartIndex(), ownerSegment.getStopIndex(), ownerSegment.getIdentifier())));
138         }
139         if (expressionSegment instanceof ListExpression) {
140             for (ExpressionSegment each : ((ListExpression) expressionSegment).getItems()) {
141                 extractTablesFromExpression(each);
142             }
143         }
144         if (expressionSegment instanceof ExistsSubqueryExpression) {
145             extractTablesFromSelect(((ExistsSubqueryExpression) expressionSegment).getSubquery().getSelect());
146         }
147         if (expressionSegment instanceof BetweenExpression) {
148             extractTablesFromExpression(((BetweenExpression) expressionSegment).getLeft());
149             extractTablesFromExpression(((BetweenExpression) expressionSegment).getBetweenExpr());
150             extractTablesFromExpression(((BetweenExpression) expressionSegment).getAndExpr());
151         }
152         if (expressionSegment instanceof InExpression) {
153             extractTablesFromExpression(((InExpression) expressionSegment).getLeft());
154             extractTablesFromExpression(((InExpression) expressionSegment).getRight());
155         }
156         if (expressionSegment instanceof SubqueryExpressionSegment) {
157             extractTablesFromSelect(((SubqueryExpressionSegment) expressionSegment).getSubquery().getSelect());
158         }
159         if (expressionSegment instanceof BinaryOperationExpression) {
160             extractTablesFromExpression(((BinaryOperationExpression) expressionSegment).getLeft());
161             extractTablesFromExpression(((BinaryOperationExpression) expressionSegment).getRight());
162         }
163         if (expressionSegment instanceof MatchAgainstExpression) {
164             for (ColumnSegment each : ((MatchAgainstExpression) expressionSegment).getColumns()) {
165                 extractTablesFromExpression(each);
166             }
167         }
168         if (expressionSegment instanceof FunctionSegment) {
169             for (ExpressionSegment each : ((FunctionSegment) expressionSegment).getParameters()) {
170                 extractTablesFromExpression(each);
171             }
172         }
173     }
174     
175     private void extractTablesFromProjections(final ProjectionsSegment projections) {
176         for (ProjectionSegment each : projections.getProjections()) {
177             if (each instanceof SubqueryProjectionSegment) {
178                 extractTablesFromSelect(((SubqueryProjectionSegment) each).getSubquery().getSelect());
179             } else if (each instanceof OwnerAvailable) {
180                 if (((OwnerAvailable) each).getOwner().isPresent() && needRewrite(((OwnerAvailable) each).getOwner().get())) {
181                     OwnerSegment ownerSegment = ((OwnerAvailable) each).getOwner().get();
182                     rewriteTables.add(createSimpleTableSegment(ownerSegment));
183                 }
184             } else if (each instanceof ColumnProjectionSegment) {
185                 if (((ColumnProjectionSegment) each).getColumn().getOwner().isPresent() && needRewrite(((ColumnProjectionSegment) each).getColumn().getOwner().get())) {
186                     OwnerSegment ownerSegment = ((ColumnProjectionSegment) each).getColumn().getOwner().get();
187                     rewriteTables.add(createSimpleTableSegment(ownerSegment));
188                 }
189             } else if (each instanceof AggregationProjectionSegment) {
190                 ((AggregationProjectionSegment) each).getParameters().forEach(this::extractTablesFromExpression);
191             } else if (each instanceof ExpressionProjectionSegment) {
192                 extractTablesFromExpression(((ExpressionProjectionSegment) each).getExpr());
193             }
194         }
195     }
196     
197     private SimpleTableSegment createSimpleTableSegment(final OwnerSegment ownerSegment) {
198         SimpleTableSegment result = new SimpleTableSegment(new TableNameSegment(ownerSegment.getStartIndex(), ownerSegment.getStopIndex(), ownerSegment.getIdentifier()));
199         ownerSegment.getOwner().ifPresent(result::setOwner);
200         return result;
201     }
202     
203     private void extractTablesFromOrderByItems(final Collection<OrderByItemSegment> orderByItems) {
204         for (OrderByItemSegment each : orderByItems) {
205             if (each instanceof ColumnOrderByItemSegment) {
206                 Optional<OwnerSegment> owner = ((ColumnOrderByItemSegment) each).getColumn().getOwner();
207                 if (owner.isPresent() && needRewrite(owner.get())) {
208                     rewriteTables.add(new SimpleTableSegment(new TableNameSegment(owner.get().getStartIndex(), owner.get().getStopIndex(), owner.get().getIdentifier())));
209                 }
210             }
211         }
212     }
213     
214     private void extractTablesFromLock(final LockSegment lockSegment) {
215         rewriteTables.addAll(lockSegment.getTables());
216     }
217     
218     /**
219      * Extract table that should be rewritten from delete statement.
220      *
221      * @param deleteStatement delete statement
222      */
223     public void extractTablesFromDelete(final DeleteStatement deleteStatement) {
224         extractTablesFromTableSegment(deleteStatement.getTable());
225         if (deleteStatement.getWhere().isPresent()) {
226             extractTablesFromExpression(deleteStatement.getWhere().get().getExpr());
227         }
228     }
229     
230     /**
231      * Extract table that should be rewritten from insert statement.
232      *
233      * @param insertStatement insert statement
234      */
235     public void extractTablesFromInsert(final InsertStatement insertStatement) {
236         if (null != insertStatement.getTable()) {
237             extractTablesFromTableSegment(insertStatement.getTable());
238         }
239         if (!insertStatement.getColumns().isEmpty()) {
240             for (ColumnSegment each : insertStatement.getColumns()) {
241                 extractTablesFromExpression(each);
242             }
243         }
244         insertStatement.getOnDuplicateKeyColumns().ifPresent(optional -> extractTablesFromAssignmentItems(optional.getColumns()));
245         if (insertStatement.getInsertSelect().isPresent()) {
246             extractTablesFromSelect(insertStatement.getInsertSelect().get().getSelect());
247         }
248     }
249     
250     private void extractTablesFromAssignmentItems(final Collection<ColumnAssignmentSegment> assignmentItems) {
251         assignmentItems.forEach(each -> extractTablesFromColumnSegments(each.getColumns()));
252     }
253     
254     private void extractTablesFromColumnSegments(final Collection<ColumnSegment> columnSegments) {
255         columnSegments.forEach(each -> {
256             if (each.getOwner().isPresent() && needRewrite(each.getOwner().get())) {
257                 OwnerSegment ownerSegment = each.getOwner().get();
258                 rewriteTables.add(new SimpleTableSegment(new TableNameSegment(ownerSegment.getStartIndex(), ownerSegment.getStopIndex(), ownerSegment.getIdentifier())));
259             }
260         });
261     }
262     
263     /**
264      * Extract table that should be rewritten from update statement.
265      *
266      * @param updateStatement update statement.
267      */
268     public void extractTablesFromUpdate(final UpdateStatement updateStatement) {
269         extractTablesFromTableSegment(updateStatement.getTable());
270         updateStatement.getSetAssignment().getAssignments().forEach(each -> extractTablesFromExpression(each.getColumns().get(0)));
271         if (updateStatement.getWhere().isPresent()) {
272             extractTablesFromExpression(updateStatement.getWhere().get().getExpr());
273         }
274     }
275     
276     /**
277      * Check if the table needs to be overwritten.
278      *
279      * @param owner owner
280      * @return boolean
281      */
282     public boolean needRewrite(final OwnerSegment owner) {
283         for (TableSegment each : tableContext) {
284             if (owner.getIdentifier().getValue().equalsIgnoreCase(each.getAliasName().orElse(null))) {
285                 return false;
286             }
287         }
288         return true;
289     }
290     
291     /**
292      * Extract the tables that should exist from routine body segment.
293      *
294      * @param routineBody routine body segment
295      * @return the tables that should exist
296      */
297     public Collection<SimpleTableSegment> extractExistTableFromRoutineBody(final RoutineBodySegment routineBody) {
298         Collection<SimpleTableSegment> result = new LinkedList<>();
299         for (ValidStatementSegment each : routineBody.getValidStatements()) {
300             if (each.getAlterTable().isPresent()) {
301                 result.add(each.getAlterTable().get().getTable());
302             }
303             if (each.getDropTable().isPresent()) {
304                 result.addAll(each.getDropTable().get().getTables());
305             }
306             if (each.getTruncate().isPresent()) {
307                 result.addAll(each.getTruncate().get().getTables());
308             }
309             result.addAll(extractExistTableFromDMLStatement(each));
310         }
311         return result;
312     }
313     
314     private Collection<SimpleTableSegment> extractExistTableFromDMLStatement(final ValidStatementSegment validStatementSegment) {
315         if (validStatementSegment.getInsert().isPresent()) {
316             extractTablesFromInsert(validStatementSegment.getInsert().get());
317         } else if (validStatementSegment.getReplace().isPresent()) {
318             extractTablesFromInsert(validStatementSegment.getReplace().get());
319         } else if (validStatementSegment.getUpdate().isPresent()) {
320             extractTablesFromUpdate(validStatementSegment.getUpdate().get());
321         } else if (validStatementSegment.getDelete().isPresent()) {
322             extractTablesFromDelete(validStatementSegment.getDelete().get());
323         } else if (validStatementSegment.getSelect().isPresent()) {
324             extractTablesFromSelect(validStatementSegment.getSelect().get());
325         }
326         return rewriteTables;
327     }
328     
329     /**
330      * Extract the tables that should not exist from routine body segment.
331      *
332      * @param routineBody routine body segment
333      * @return the tables that should not exist
334      */
335     public Collection<SimpleTableSegment> extractNotExistTableFromRoutineBody(final RoutineBodySegment routineBody) {
336         Collection<SimpleTableSegment> result = new LinkedList<>();
337         for (ValidStatementSegment each : routineBody.getValidStatements()) {
338             Optional<CreateTableStatement> createTable = each.getCreateTable();
339             if (createTable.isPresent() && !createTable.get().isIfNotExists()) {
340                 result.add(createTable.get().getTable());
341             }
342         }
343         return result;
344     }
345     
346     /**
347      * Extract table that should be rewritten from SQL statement.
348      *
349      * @param sqlStatement SQL statement
350      */
351     public void extractTablesFromSQLStatement(final SQLStatement sqlStatement) {
352         if (sqlStatement instanceof SelectStatement) {
353             extractTablesFromSelect((SelectStatement) sqlStatement);
354         } else if (sqlStatement instanceof InsertStatement) {
355             extractTablesFromInsert((InsertStatement) sqlStatement);
356         } else if (sqlStatement instanceof UpdateStatement) {
357             extractTablesFromUpdate((UpdateStatement) sqlStatement);
358         } else if (sqlStatement instanceof DeleteStatement) {
359             extractTablesFromDelete((DeleteStatement) sqlStatement);
360         }
361     }
362     
363     /**
364      * Extract table that should be rewritten from create view statement.
365      *
366      * @param createViewStatement create view statement
367      */
368     public void extractTablesFromCreateViewStatement(final CreateViewStatement createViewStatement) {
369         tableContext.add(createViewStatement.getView());
370         rewriteTables.add(createViewStatement.getView());
371         extractTablesFromSelect(createViewStatement.getSelect());
372     }
