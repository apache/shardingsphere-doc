18  package org.apache.shardingsphere.infra.exception.core;
19  
20  import com.google.common.base.Strings;
21  import lombok.AccessLevel;
22  import lombok.NoArgsConstructor;
23  
24  import java.util.Collection;
25  import java.util.Map;
26  import java.util.function.Supplier;
27  
28  /**
29   * ShardingSphere preconditions.
30   */
31  @NoArgsConstructor(access = AccessLevel.PRIVATE)
32  public final class ShardingSpherePreconditions {
33      
34      /**
35       * Ensures the truth of an expression involving the state of the calling instance.
36       *
37       * @param <T> type of exception
38       * @param expectedExpression expected expression
39       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
40       * @throws T exception to be thrown
41       */
42      public static <T extends Throwable> void checkState(final boolean expectedExpression, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
43          if (!expectedExpression) {
44              throw exceptionSupplierIfUnexpected.get();
45          }
46      }
47      
48      /**
49       * Ensures that an object reference passed as a parameter to the calling method is not null.
50       *
51       * @param <T> type of exception
52       * @param reference object reference to be checked
53       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
54       * @throws T exception to be thrown
55       */
56      public static <T extends Throwable> void checkNotNull(final Object reference, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
57          if (null == reference) {
58              throw exceptionSupplierIfUnexpected.get();
59          }
60      }
61      
62      /**
63       * Ensures that a string passed as a parameter to the calling method is not empty.
64       *
65       * @param <T> type of exception
66       * @param value string to be checked
67       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
68       * @throws T exception to be thrown
69       */
70      public static <T extends Throwable> void checkNotEmpty(final String value, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
71          if (Strings.isNullOrEmpty(value)) {
72              throw exceptionSupplierIfUnexpected.get();
73          }
74      }
75      
76      /**
77       * Ensures that a collection passed as a parameter to the calling method is not empty.
78       *
79       * @param <T> type of exception
80       * @param values collection to be checked
81       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
82       * @throws T exception to be thrown
83       */
84      public static <T extends Throwable> void checkNotEmpty(final Collection<?> values, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
85          if (values.isEmpty()) {
86              throw exceptionSupplierIfUnexpected.get();
87          }
88      }
89      
90      /**
91       * Ensures that a collection passed as a parameter to the calling method is not empty.
92       *
93       * @param <T> type of exception
94       * @param map map to be checked
95       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
96       * @throws T exception to be thrown
97       */
98      public static <T extends Throwable> void checkNotEmpty(final Map<?, ?> map, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
99          if (map.isEmpty()) {
100             throw exceptionSupplierIfUnexpected.get();
101         }
102     }
103     
104     /**
105      * Ensures that a collection passed as a parameter to the calling method must empty.
106      *
107      * @param <T> type of exception
108      * @param values collection to be checked
109      * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
110      * @throws T exception to be thrown
111      */
112     public static <T extends Throwable> void checkMustEmpty(final Collection<?> values, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
113         if (!values.isEmpty()) {
114             throw exceptionSupplierIfUnexpected.get();
115         }
116     }
117     
118     /**
119      * Ensures that a collection passed as a parameter to the calling method must empty.
120      *
121      * @param <T> type of exception
122      * @param map map to be checked
123      * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
124      * @throws T exception to be thrown
125      */
126     public static <T extends Throwable> void checkMustEmpty(final Map<?, ?> map, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
127         if (!map.isEmpty()) {
128             throw exceptionSupplierIfUnexpected.get();
129         }
130     }
131 }
