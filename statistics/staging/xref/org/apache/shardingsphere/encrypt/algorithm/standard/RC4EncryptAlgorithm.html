18  package org.apache.shardingsphere.encrypt.algorithm.standard;
19  
20  import org.apache.commons.codec.binary.Base64;
21  import org.apache.shardingsphere.encrypt.api.context.EncryptContext;
22  import org.apache.shardingsphere.encrypt.api.encrypt.standard.StandardEncryptAlgorithm;
23  import org.apache.shardingsphere.encrypt.exception.algorithm.EncryptAlgorithmInitializationException;
24  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
25  
26  import java.nio.charset.StandardCharsets;
27  import java.util.Properties;
28  
29  /**
30   * RC4 encrypt algorithm.
31   */
32  public final class RC4EncryptAlgorithm implements StandardEncryptAlgorithm<Object, String> {
33      
34      private static final String RC4_KEY = "rc4-key-value";
35      
36      private static final int KEY_MIN_LENGTH = 5;
37      
38      private static final int SBOX_LENGTH = 256;
39      
40      private byte[] key;
41      
42      @Override
43      public void init(final Properties props) {
44          key = getKey(props);
45      }
46      
47      private byte[] getKey(final Properties props) {
48          byte[] result = props.getProperty(RC4_KEY, "").getBytes(StandardCharsets.UTF_8);
49          ShardingSpherePreconditions.checkState(KEY_MIN_LENGTH <= result.length && SBOX_LENGTH > result.length,
50                  () -> new EncryptAlgorithmInitializationException(getType(), "Key length has to be between " + KEY_MIN_LENGTH + " and " + (SBOX_LENGTH - 1)));
51          return result;
52      }
53      
54      @Override
55      public String encrypt(final Object plainValue, final EncryptContext encryptContext) {
56          return null == plainValue ? null : Base64.encodeBase64String(crypt(String.valueOf(plainValue).getBytes(StandardCharsets.UTF_8)));
57      }
58      
59      @Override
60      public Object decrypt(final String cipherValue, final EncryptContext encryptContext) {
61          return null == cipherValue ? null : new String(crypt(Base64.decodeBase64(cipherValue)), StandardCharsets.UTF_8);
62      }
63      
64      /*
65       * @see <a href="http://en.wikipedia.org/wiki/RC4#Pseudo-random_generation_algorithm_.28PRGA.29">Pseudo-random generation algorithm</a>
66       */
67      private byte[] crypt(final byte[] message) {
68          int[] sBox = getSBox();
69          byte[] result = new byte[message.length];
70          int i = 0;
71          int j = 0;
72          for (int n = 0; n < message.length; n++) {
73              i = (i + 1) % SBOX_LENGTH;
74              j = (j + sBox[i]) % SBOX_LENGTH;
75              swap(i, j, sBox);
76              int rand = sBox[(sBox[i] + sBox[j]) % SBOX_LENGTH];
77              result[n] = (byte) (rand ^ message[n]);
78          }
79          return result;
80      }
81      
82      /*
83       * @see <a href="http://en.wikipedia.org/wiki/RC4#Key-scheduling_algorithm_.28KSA.29">Wikipedia. Init sBox</a>
84       */
85      private int[] getSBox() {
86          int[] result = new int[SBOX_LENGTH];
87          int j = 0;
88          for (int i = 0; i < SBOX_LENGTH; i++) {
89              result[i] = i;
90          }
91          for (int i = 0; i < SBOX_LENGTH; i++) {
92              j = (j + result[i] + (key[i % key.length]) & 0xFF) % SBOX_LENGTH;
93              swap(i, j, result);
94          }
95          return result;
96      }
97      
98      private void swap(final int i, final int j, final int[] sBox) {
99          int temp = sBox[i];
100         sBox[i] = sBox[j];
101         sBox[j] = temp;
102     }
103     
104     @Override
105     public String getType() {
106         return "RC4";
107     }
108 }
