49  class SingleSQLFederationDeciderTest {
50      
51      @Test
52      void assertDecideWhenNotContainsSingleTable() {
53          SingleSQLFederationDecider federationDecider = new SingleSQLFederationDecider();
54          SelectStatementContext select = createStatementContext();
55          QueryContext queryContext = new QueryContext(select, "", Collections.emptyList());
56          SQLFederationDeciderContext actual = new SQLFederationDeciderContext();
57          federationDecider.decide(actual, queryContext, mock(ShardingSphereRuleMetaData.class), createDatabase(), mock(SingleRule.class), new ConfigurationProperties(new Properties()));
58          assertTrue(actual.getDataNodes().isEmpty());
59          assertFalse(actual.isUseSQLFederation());
60      }
61      
62      @Test
63      void assertDecideWhenAllSingleTablesInSameDataSource() {
64          Collection<QualifiedTable> qualifiedTables = Arrays.asList(new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order"), new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order_item"));
65          SingleRule rule = createSingleRule(qualifiedTables);
66          when(rule.isSingleTablesInSameDataSource(qualifiedTables)).thenReturn(true);
67          SelectStatementContext select = createStatementContext();
68          QueryContext queryContext = new QueryContext(select, "", Collections.emptyList());
69          SQLFederationDeciderContext actual = new SQLFederationDeciderContext();
70          SingleSQLFederationDecider federationDecider = new SingleSQLFederationDecider();
71          federationDecider.decide(actual, queryContext, mock(ShardingSphereRuleMetaData.class), createDatabase(), rule, new ConfigurationProperties(new Properties()));
72          assertThat(actual.getDataNodes().size(), is(2));
73          assertFalse(actual.isUseSQLFederation());
74      }
75      
76      @Test
77      void assertDecideWhenAllSingleTablesNotInSameDataSource() {
78          Collection<QualifiedTable> qualifiedTables = Arrays.asList(new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order"), new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order_item"));
79          SingleRule rule = createSingleRule(qualifiedTables);
80          when(rule.isSingleTablesInSameDataSource(qualifiedTables)).thenReturn(false);
81          SelectStatementContext select = createStatementContext();
82          QueryContext queryContext = new QueryContext(select, "", Collections.emptyList());
83          SQLFederationDeciderContext actual = new SQLFederationDeciderContext();
84          SingleSQLFederationDecider federationDecider = new SingleSQLFederationDecider();
85          federationDecider.decide(actual, queryContext, mock(ShardingSphereRuleMetaData.class), createDatabase(), rule, new ConfigurationProperties(new Properties()));
86          assertThat(actual.getDataNodes().size(), is(2));
87          assertTrue(actual.isUseSQLFederation());
88      }
89      
90      @Test
91      void assertDecideWhenAllTablesInSameDataSource() {
92          Collection<QualifiedTable> qualifiedTables = Arrays.asList(new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order"), new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order_item"));
93          SingleRule rule = createSingleRule(qualifiedTables);
94          when(rule.isSingleTablesInSameDataSource(qualifiedTables)).thenReturn(true);
95          SelectStatementContext select = createStatementContext();
96          QueryContext queryContext = new QueryContext(select, "", Collections.emptyList());
97          SQLFederationDeciderContext actual = new SQLFederationDeciderContext();
98          actual.getDataNodes().add(new DataNode("ds_0", "t_user"));
99          SingleSQLFederationDecider federationDecider = new SingleSQLFederationDecider();
100         federationDecider.decide(actual, queryContext, mock(ShardingSphereRuleMetaData.class), createDatabase(), rule, new ConfigurationProperties(new Properties()));
101         assertThat(actual.getDataNodes().size(), is(3));
102         assertFalse(actual.isUseSQLFederation());
103     }
104     
105     @Test
106     void assertDecideWhenAllTablesNotInSameDataSource() {
107         Collection<QualifiedTable> qualifiedTables = Arrays.asList(new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order"), new QualifiedTable(DefaultDatabase.LOGIC_NAME, "t_order_item"));
108         SingleRule rule = createSingleRule(qualifiedTables);
109         when(rule.isSingleTablesInSameDataSource(qualifiedTables)).thenReturn(true);
110         SelectStatementContext select = createStatementContext();
111         QueryContext queryContext = new QueryContext(select, "", Collections.emptyList());
112         SQLFederationDeciderContext actual = new SQLFederationDeciderContext();
113         actual.getDataNodes().add(new DataNode("ds_1", "t_user"));
114         SingleSQLFederationDecider federationDecider = new SingleSQLFederationDecider();
115         federationDecider.decide(actual, queryContext, mock(ShardingSphereRuleMetaData.class), createDatabase(), rule, new ConfigurationProperties(new Properties()));
116         assertThat(actual.getDataNodes().size(), is(3));
117         assertTrue(actual.isUseSQLFederation());
118     }
119     
120     private static SingleRule createSingleRule(final Collection<QualifiedTable> qualifiedTables) {
121         SingleRule result = mock(SingleRule.class);
122         when(result.getSingleTableNames(any())).thenReturn(qualifiedTables);
123         when(result.findSingleTableDataNode(DefaultDatabase.LOGIC_NAME, "t_order")).thenReturn(Optional.of(new DataNode("ds_0", "t_order")));
124         when(result.findSingleTableDataNode(DefaultDatabase.LOGIC_NAME, "t_order_item")).thenReturn(Optional.of(new DataNode("ds_0", "t_order_item")));
125         return result;
126     }
127     
128     private static SelectStatementContext createStatementContext() {
129         SelectStatementContext result = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
130         when(result.getDatabaseType()).thenReturn(new MySQLDatabaseType());
131         return result;
132     }
133     
134     private static ShardingSphereDatabase createDatabase() {
135         ShardingSphereDatabase result = mock(ShardingSphereDatabase.class);
136         when(result.getName()).thenReturn(DefaultDatabase.LOGIC_NAME);
137         when(result.getSchema(DefaultDatabase.LOGIC_NAME)).thenReturn(mock(ShardingSphereSchema.class));
138         return result;
139     }
