64      /**
65       * Create storage resource.
66       *
67       * @param dataSourcePropsMap data source properties map
68       * @param cacheEnabled cache enabled
69       * @return created storage resource
70       */
71      public static StorageResource createStorageResource(final Map<String, DataSourceProperties> dataSourcePropsMap, final boolean cacheEnabled) {
72          Map<String, DataSource> storageNodes = new LinkedHashMap<>();
73          Map<String, StorageUnit> storageUnits = new LinkedHashMap<>();
74          for (Entry<String, DataSourceProperties> entry : dataSourcePropsMap.entrySet()) {
75              StorageNodeProperties storageNodeProperties = getStorageNodeProperties(entry.getKey(), entry.getValue());
76              if (storageNodes.containsKey(storageNodeProperties.getName())) {
77                  appendStorageUnit(storageUnits, storageNodeProperties, entry.getKey(), entry.getValue());
78                  continue;
79              }
80              DataSource dataSource;
81              try {
82                  dataSource = create(entry.getKey(), entry.getValue(), cacheEnabled);
83                  // CHECKSTYLE:OFF
84              } catch (final RuntimeException ex) {
85                  // CHECKSTYLE:ON
86                  if (!cacheEnabled) {
87                      storageNodes.values().stream().map(DataSourcePoolDestroyer::new).forEach(DataSourcePoolDestroyer::asyncDestroy);
88                  }
89                  throw ex;
90              }
91              storageNodes.put(storageNodeProperties.getName(), dataSource);
92              appendStorageUnit(storageUnits, storageNodeProperties, entry.getKey(), entry.getValue());
93          }
94          return new StorageResource(storageNodes, storageUnits);
95      }

97      /**
98       * Create storage resource without data source.
99       *
100      * @param dataSourcePropsMap data source properties map
101      * @return created storage resource
102      */
103     public static StorageResourceWithProperties createStorageResourceWithoutDataSource(final Map<String, DataSourceProperties> dataSourcePropsMap) {
104         Map<String, DataSource> storageNodes = new LinkedHashMap<>();
105         Map<String, StorageUnit> storageUnits = new LinkedHashMap<>();
106         Map<String, DataSourceProperties> dataSourcePropertiesMap = new LinkedHashMap<>();
107         for (Entry<String, DataSourceProperties> entry : dataSourcePropsMap.entrySet()) {
108             StorageNodeProperties storageNodeProperties = getStorageNodeProperties(entry.getKey(), entry.getValue());
109             if (storageNodes.containsKey(storageNodeProperties.getName())) {
110                 appendStorageUnit(storageUnits, storageNodeProperties, entry.getKey(), entry.getValue());
111                 continue;
112             }
113             storageNodes.put(storageNodeProperties.getName(), null);
114             appendStorageUnit(storageUnits, storageNodeProperties, entry.getKey(), entry.getValue());
115             dataSourcePropertiesMap.put(storageNodeProperties.getName(), entry.getValue());
116         }
117         return new StorageResourceWithProperties(storageNodes, storageUnits, dataSourcePropertiesMap);
118     }
119     
120     private static void appendStorageUnit(final Map<String, StorageUnit> storageUnits, final StorageNodeProperties storageNodeProperties,
121                                           final String unitName, final DataSourceProperties dataSourceProps) {
122         String url = dataSourceProps.getConnectionPropertySynonyms().getStandardProperties().get("url").toString();
123         storageUnits.put(unitName, getStorageUnit(storageNodeProperties, unitName, url));
124     }
125     
126     private static StorageUnit getStorageUnit(final StorageNodeProperties storageNodeProperties, final String unitName, final String url) {
127         DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, storageNodeProperties.getDatabaseType());
128         return dialectDatabaseMetaData.isInstanceConnectionAvailable()
129                 ? new StorageUnit(unitName, storageNodeProperties.getName(), storageNodeProperties.getDatabase(), url)
130                 : new StorageUnit(unitName, storageNodeProperties.getName(), url);
131     }
132     
133     private static StorageNodeProperties getStorageNodeProperties(final String dataSourceName, final DataSourceProperties dataSourceProperties) {
134         Map<String, Object> standardProperties = dataSourceProperties.getConnectionPropertySynonyms().getStandardProperties();
135         String url = standardProperties.get("url").toString();
136         String username = standardProperties.get("username").toString();
137         DatabaseType databaseType = DatabaseTypeFactory.get(url);
138         return getStorageNodeProperties(dataSourceName, dataSourceProperties, url, username, databaseType);
139     }
140     
141     private static StorageNodeProperties getStorageNodeProperties(final String dataSourceName, final DataSourceProperties dataSourceProperties,
142                                                                   final String url, final String username, final DatabaseType databaseType) {
143         try {
144             JdbcUrl jdbcUrl = new StandardJdbcUrlParser().parse(url);
145             DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, databaseType);
146             String nodeName = dialectDatabaseMetaData.isInstanceConnectionAvailable() ? generateStorageNodeName(jdbcUrl.getHostname(), jdbcUrl.getPort(), username) : dataSourceName;
147             return new StorageNodeProperties(nodeName, databaseType, dataSourceProperties, jdbcUrl.getDatabase());
148         } catch (final UnrecognizedDatabaseURLException ex) {
149             return new StorageNodeProperties(dataSourceName, databaseType, dataSourceProperties, null);
150         }
151     }
152     
153     private static String generateStorageNodeName(final String hostname, final int port, final String username) {
154         return String.format("%s_%s_%s", hostname, port, username);
155     }

167     /**
168      * Create data sources.
169      *
170      * @param dataSourcePropsMap data source properties map
171      * @param cacheEnabled cache enabled
172      * @return created data sources
173      */
174     public static Map<String, DataSource> create(final Map<String, DataSourceProperties> dataSourcePropsMap, final boolean cacheEnabled) {
175         Map<String, DataSource> result = new LinkedHashMap<>();
176         for (Entry<String, DataSourceProperties> entry : dataSourcePropsMap.entrySet()) {
177             DataSource dataSource;
178             try {
179                 dataSource = create(entry.getKey(), entry.getValue(), cacheEnabled);
180                 // CHECKSTYLE:OFF
181             } catch (final RuntimeException ex) {
182                 // CHECKSTYLE:ON
183                 if (!cacheEnabled) {
184                     result.values().stream().map(DataSourcePoolDestroyer::new).forEach(DataSourcePoolDestroyer::asyncDestroy);
185                 }
186                 throw ex;
187             }
188             result.put(entry.getKey(), dataSource);
189         }
190         return result;
191     }

193     /**
194      * Create data source.
195      *
196      * @param dataSourceProps data source properties
197      * @return created data source
198      */
199     public static DataSource create(final DataSourceProperties dataSourceProps) {
200         DataSource result = createDataSource(dataSourceProps.getDataSourceClassName());
201         Optional<DataSourcePoolMetaData> poolMetaData = TypedSPILoader.findService(DataSourcePoolMetaData.class, dataSourceProps.getDataSourceClassName());
202         DataSourceReflection dataSourceReflection = new DataSourceReflection(result);
203         if (poolMetaData.isPresent()) {
204             setDefaultFields(dataSourceReflection, poolMetaData.get());
205             setConfiguredFields(dataSourceProps, dataSourceReflection, poolMetaData.get());
206             appendJdbcUrlProperties(dataSourceProps.getCustomDataSourceProperties(), result, poolMetaData.get());
207             dataSourceReflection.addDefaultDataSourceProperties();
208         } else {
209             setConfiguredFields(dataSourceProps, dataSourceReflection);
210         }
211         return result;
212     }
213     
214     /**
215      * Create data source.
216      *
217      * @param dataSourceName data source name
218      * @param dataSourceProps data source properties
219      * @param cacheEnabled cache enabled
220      * @return created data source
221      */
222     public static DataSource create(final String dataSourceName, final DataSourceProperties dataSourceProps, final boolean cacheEnabled) {
223         DataSource result = create(dataSourceProps);
224         if (cacheEnabled && !GlobalDataSourceRegistry.getInstance().getCachedDataSources().containsKey(dataSourceName)) {
225             GlobalDataSourceRegistry.getInstance().getCachedDataSources().put(dataSourceName, result);
226         }
227         return result;
228     }
229     
230     @SneakyThrows(ReflectiveOperationException.class)
231     private static DataSource createDataSource(final String dataSourceClassName) {
232         return (DataSource) Class.forName(dataSourceClassName).getConstructor().newInstance();
233     }
234     
235     private static void setDefaultFields(final DataSourceReflection dataSourceReflection, final DataSourcePoolMetaData poolMetaData) {
236         for (Entry<String, Object> entry : poolMetaData.getDefaultProperties().entrySet()) {
237             dataSourceReflection.setField(entry.getKey(), entry.getValue());
238         }
239     }
240     
241     private static void setConfiguredFields(final DataSourceProperties dataSourceProps, final DataSourceReflection dataSourceReflection) {
242         for (Entry<String, Object> entry : dataSourceProps.getAllLocalProperties().entrySet()) {
243             dataSourceReflection.setField(entry.getKey(), entry.getValue());
244         }
245     }
246     
247     private static void setConfiguredFields(final DataSourceProperties dataSourceProps, final DataSourceReflection dataSourceReflection, final DataSourcePoolMetaData poolMetaData) {
248         for (Entry<String, Object> entry : dataSourceProps.getAllLocalProperties().entrySet()) {
249             String fieldName = entry.getKey();
250             Object fieldValue = entry.getValue();
251             if (isValidProperty(fieldName, fieldValue, poolMetaData) && !fieldName.equals(poolMetaData.getFieldMetaData().getJdbcUrlPropertiesFieldName())) {
252                 dataSourceReflection.setField(fieldName, fieldValue);
253             }
254         }
255     }
256     
257     private static boolean isValidProperty(final String key, final Object value, final DataSourcePoolMetaData poolMetaData) {
258         return !poolMetaData.getInvalidProperties().containsKey(key) || null == value || !value.equals(poolMetaData.getInvalidProperties().get(key));
259     }
260     
261     @SuppressWarnings("unchecked")
262     private static void appendJdbcUrlProperties(final CustomDataSourceProperties customDataSourceProps, final DataSource targetDataSource, final DataSourcePoolMetaData poolMetaData) {
263         String jdbcUrlPropertiesFieldName = poolMetaData.getFieldMetaData().getJdbcUrlPropertiesFieldName();
264         if (null != jdbcUrlPropertiesFieldName && customDataSourceProps.getProperties().containsKey(jdbcUrlPropertiesFieldName)) {
265             Map<String, Object> jdbcUrlProps = (Map<String, Object>) customDataSourceProps.getProperties().get(jdbcUrlPropertiesFieldName);
266             DataSourcePoolMetaDataReflection dataSourcePoolMetaDataReflection = new DataSourcePoolMetaDataReflection(targetDataSource, poolMetaData.getFieldMetaData());
267             for (Entry<String, Object> entry : jdbcUrlProps.entrySet()) {
268                 dataSourcePoolMetaDataReflection.getJdbcConnectionProperties().ifPresent(optional -> optional.setProperty(entry.getKey(), entry.getValue().toString()));
269             }
270         }
271     }
