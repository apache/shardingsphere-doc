68  /**
69   * Proxy backend handler factory.
70   */
71  @NoArgsConstructor(access = AccessLevel.PRIVATE)
72  public final class ProxyBackendHandlerFactory {
73      
74      /**
75       * Create new instance of backend handler.
76       *
77       * @param databaseType database type
78       * @param sql SQL to be executed
79       * @param sqlStatement SQL statement
80       * @param connectionSession connection session
81       * @param hintValueContext hint query context
82       * @return created instance
83       * @throws SQLException SQL exception
84       */
85      public static ProxyBackendHandler newInstance(final DatabaseType databaseType, final String sql, final SQLStatement sqlStatement,
86                                                    final ConnectionSession connectionSession, final HintValueContext hintValueContext) throws SQLException {
87          if (sqlStatement instanceof EmptyStatement) {
88              return new SkipBackendHandler(sqlStatement);
89          }
90          SQLStatementContext sqlStatementContext = sqlStatement instanceof DistSQLStatement ? new DistSQLStatementContext((DistSQLStatement) sqlStatement)
91                  : new SQLBindEngine(ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData(), connectionSession.getCurrentDatabaseName(), hintValueContext).bind(sqlStatement,
92                          Collections.emptyList());
93          QueryContext queryContext = new QueryContext(sqlStatementContext, sql, Collections.emptyList(), hintValueContext, connectionSession.getConnectionContext(),
94                  ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData());
95          connectionSession.setQueryContext(queryContext);
96          return newInstance(databaseType, queryContext, connectionSession, false);
97      }
98      
99      /**
100      * Create new instance of backend handler.
101      *
102      * @param databaseType database type
103      * @param queryContext query context
104      * @param connectionSession connection session
105      * @param preferPreparedStatement use prepared statement as possible
106      * @return created instance
107      * @throws SQLException SQL exception
108      */
109     public static ProxyBackendHandler newInstance(final DatabaseType databaseType, final QueryContext queryContext,
110                                                   final ConnectionSession connectionSession, final boolean preferPreparedStatement) throws SQLException {
111         SQLStatementContext sqlStatementContext = queryContext.getSqlStatementContext();
112         SQLStatement sqlStatement = sqlStatementContext.getSqlStatement();
113         allowExecutingWhenTransactionalError(databaseType, connectionSession, sqlStatement);
114         checkUnsupportedSQLStatement(sqlStatement);
115         checkClusterState(sqlStatement);
116         if (sqlStatement instanceof EmptyStatement) {
117             return new SkipBackendHandler(sqlStatement);
118         }
119         if (sqlStatement instanceof DistSQLStatement) {
120             checkUnsupportedDistSQLStatementInTransaction(sqlStatement, connectionSession);
121             return DistSQLBackendHandlerFactory.newInstance((DistSQLStatement) sqlStatement, connectionSession);
122         }
123         String sql = queryContext.getSql();
124         handleAutoCommit(sqlStatement, connectionSession);
125         if (sqlStatement instanceof TCLStatement) {
126             return TransactionBackendHandlerFactory.newInstance(sqlStatementContext, sql, connectionSession);
127         }
128         Optional<ProxyBackendHandler> backendHandler = DatabaseAdminBackendHandlerFactory.newInstance(databaseType, sqlStatementContext, connectionSession, sql, queryContext.getParameters());
129         if (backendHandler.isPresent()) {
130             return backendHandler.get();
131         }
132         Optional<ProxyBackendHandler> databaseOperateHandler = findDatabaseOperateBackendHandler(sqlStatement, connectionSession);
133         if (databaseOperateHandler.isPresent()) {
134             return databaseOperateHandler.get();
135         }
136         String databaseName = sqlStatementContext instanceof TableAvailable && ((TableAvailable) sqlStatementContext).getTablesContext().getDatabaseName().isPresent()
137                 ? ((TableAvailable) sqlStatementContext).getTablesContext().getDatabaseName().get()
138                 : connectionSession.getUsedDatabaseName();
139         if (null == databaseName) {
140             return DatabaseBackendHandlerFactory.newInstance(queryContext, connectionSession, preferPreparedStatement);
141         }
142         Grantee grantee = connectionSession.getConnectionContext().getGrantee();
143         ShardingSphereMetaData metaData = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData();
144         ShardingSphereDatabase database = metaData.getDatabase(databaseName);
145         for (SQLExecutionChecker each : ShardingSphereServiceLoader.getServiceInstances(SQLExecutionChecker.class)) {
146             each.check(metaData, grantee, queryContext, database);
147         }
148         return DatabaseAdminBackendHandlerFactory.newInstance(databaseType, sqlStatementContext, connectionSession)
149                 .orElseGet(() -> DatabaseBackendHandlerFactory.newInstance(queryContext, connectionSession, preferPreparedStatement));
150     }
151     
152     private static void allowExecutingWhenTransactionalError(final DatabaseType databaseType, final ConnectionSession connectionSession, final SQLStatement sqlStatement) throws SQLException {
153         if (!connectionSession.getConnectionContext().getTransactionContext().isExceptionOccur()) {
154             return;
155         }
156         Optional<TransactionalErrorAllowedSQLStatementHandler> allowedSQLStatementHandler = DatabaseTypedSPILoader.findService(TransactionalErrorAllowedSQLStatementHandler.class, databaseType);
157         if (allowedSQLStatementHandler.isPresent()) {
158             allowedSQLStatementHandler.get().judgeContinueToExecute(sqlStatement);
159         }
160     }
161     
162     private static void checkUnsupportedDistSQLStatementInTransaction(final SQLStatement sqlStatement, final ConnectionSession connectionSession) {
163         ShardingSpherePreconditions.checkState(!connectionSession.getTransactionStatus().isInTransaction() || isSupportedDistSQLStatementInTransaction(sqlStatement),
164                 () -> new UnsupportedSQLOperationException("Non-query DistSQL is not supported within a transaction"));
165     }
166     
167     private static boolean isSupportedDistSQLStatementInTransaction(final SQLStatement sqlStatement) {
168         return sqlStatement instanceof RQLStatement || sqlStatement instanceof QueryableRALStatement || sqlStatement instanceof RULStatement;
169     }
170     
171     private static void handleAutoCommit(final SQLStatement sqlStatement, final ConnectionSession connectionSession) {
172         if (AutoCommitUtils.needOpenTransaction(sqlStatement)) {
173             connectionSession.getDatabaseConnectionManager().handleAutoCommit();
174         }
175     }
176     
177     private static Optional<ProxyBackendHandler> findDatabaseOperateBackendHandler(final SQLStatement sqlStatement, final ConnectionSession connectionSession) {
178         if (sqlStatement instanceof CreateDatabaseStatement || sqlStatement instanceof DropDatabaseStatement) {
179             return Optional.of(DatabaseOperateBackendHandlerFactory.newInstance(sqlStatement, connectionSession));
180         }
181         return Optional.empty();
182     }
183     
184     private static void checkUnsupportedSQLStatement(final SQLStatement sqlStatement) {
185         if (sqlStatement instanceof DCLStatement || sqlStatement instanceof FlushStatement || sqlStatement instanceof MySQLShowCreateUserStatement || sqlStatement instanceof RenameTableStatement) {
186             throw new UnsupportedSQLOperationException(String.format("unsupported SQL statement `%s`", sqlStatement.getClass().getSimpleName()));
187         }
188     }
189     
190     private static void checkClusterState(final SQLStatement sqlStatement) {
191         ClusterState clusterCurrentState = ProxyContext.getInstance().getContextManager().getStateContext().getClusterState();
192         if (ClusterState.OK != clusterCurrentState) {
193             TypedSPILoader.getService(ProxyClusterState.class, clusterCurrentState.name()).check(sqlStatement);
194         }
195     }
