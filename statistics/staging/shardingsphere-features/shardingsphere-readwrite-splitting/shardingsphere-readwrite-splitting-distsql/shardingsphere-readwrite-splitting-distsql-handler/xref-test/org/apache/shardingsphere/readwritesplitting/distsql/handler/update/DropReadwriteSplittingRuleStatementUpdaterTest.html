55  @ExtendWith(MockitoExtension.class)
56  class DropReadwriteSplittingRuleStatementUpdaterTest {
57      
58      private final DropReadwriteSplittingRuleStatementUpdater updater = new DropReadwriteSplittingRuleStatementUpdater();
59      
60      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
61      private ShardingSphereDatabase database;
62      
63      @Test
64      void assertCheckSQLStatementWithoutCurrentRule() {
65          assertThrows(MissingRequiredRuleException.class, () -> updater.checkSQLStatement(database, createSQLStatement(), null));
66      }
67      
68      @Test
69      void assertCheckSQLStatementWithoutToBeDroppedRule() throws RuleDefinitionViolationException {
70          assertThrows(MissingRequiredRuleException.class,
71                  () -> updater.checkSQLStatement(database, createSQLStatement(), new ReadwriteSplittingRuleConfiguration(Collections.emptyList(), Collections.emptyMap())));
72      }
73      
74      @Test
75      void assertCheckSQLStatementWithIfExists() throws RuleDefinitionViolationException {
76          updater.checkSQLStatement(database, new DropReadwriteSplittingRuleStatement(true, Collections.singleton("readwrite_ds")),
77                  new ReadwriteSplittingRuleConfiguration(Collections.emptyList(), Collections.emptyMap()));
78          updater.checkSQLStatement(database, new DropReadwriteSplittingRuleStatement(true, Collections.singleton("readwrite_ds")), null);
79      }
80      
81      @Test
82      void assertCheckSQLStatementWithInUsed() throws RuleDefinitionViolationException {
83          DataSourceContainedRule dataSourceContainedRule = mock(DataSourceContainedRule.class);
84          Map<String, Collection<DataSourceRoleInfo>> dataSourceMapper = Collections.singletonMap("foo_ds", Collections.singleton(new DataSourceRoleInfo("readwrite_ds", DataSourceRole.PRIMARY)));
85          when(dataSourceContainedRule.getDataSourceMapper()).thenReturn(dataSourceMapper);
86          when(database.getRuleMetaData().findRules(DataSourceContainedRule.class)).thenReturn(Collections.singleton(dataSourceContainedRule));
87          DataNodeContainedRule dataNodeContainedRule = mock(DataNodeContainedRule.class);
88          when(dataNodeContainedRule.getAllDataNodes()).thenReturn(Collections.singletonMap("foo_ds", Collections.singleton(new DataNode("readwrite_ds.tbl"))));
89          when(database.getRuleMetaData().findRules(DataNodeContainedRule.class)).thenReturn(Collections.singleton(dataNodeContainedRule));
90          assertThrows(RuleInUsedException.class, () -> updater.checkSQLStatement(database, createSQLStatement(), createCurrentRuleConfiguration()));
91      }
92      
93      @Test
94      void assertUpdateCurrentRuleConfiguration() {
95          ReadwriteSplittingRuleConfiguration ruleConfig = createCurrentRuleConfiguration();
96          assertTrue(updater.updateCurrentRuleConfiguration(createSQLStatement(), ruleConfig));
97          assertThat(ruleConfig.getLoadBalancers().size(), is(1));
98      }
99      
100     @Test
101     void assertUpdateCurrentRuleConfigurationWithInUsedLoadBalancer() {
102         ReadwriteSplittingRuleConfiguration ruleConfig = createMultipleCurrentRuleConfigurations();
103         assertFalse(updater.updateCurrentRuleConfiguration(createSQLStatement(), ruleConfig));
104         assertThat(ruleConfig.getLoadBalancers().size(), is(1));
105     }
106     
107     @Test
108     void assertUpdateCurrentRuleConfigurationWithoutLoadBalancerName() {
109         ReadwriteSplittingRuleConfiguration ruleConfig = createCurrentRuleConfigurationWithoutLoadBalancerName();
110         assertTrue(updater.updateCurrentRuleConfiguration(createSQLStatement(), ruleConfig));
111         assertThat(ruleConfig.getLoadBalancers().size(), is(1));
112     }
113     
114     private DropReadwriteSplittingRuleStatement createSQLStatement() {
115         return new DropReadwriteSplittingRuleStatement(false, Collections.singleton("readwrite_ds"));
116     }
117     
118     private ReadwriteSplittingRuleConfiguration createCurrentRuleConfiguration() {
119         ReadwriteSplittingDataSourceRuleConfiguration dataSourceRuleConfig = new ReadwriteSplittingDataSourceRuleConfiguration("readwrite_ds",
120                 new StaticReadwriteSplittingStrategyConfiguration("", Collections.emptyList()), null, "TEST");
121         Map<String, AlgorithmConfiguration> loadBalancers = Collections.singletonMap("readwrite_ds", new AlgorithmConfiguration("TEST", new Properties()));
122         return new ReadwriteSplittingRuleConfiguration(new LinkedList<>(Collections.singleton(dataSourceRuleConfig)), loadBalancers);
123     }
124     
125     private ReadwriteSplittingRuleConfiguration createCurrentRuleConfigurationWithoutLoadBalancerName() {
126         ReadwriteSplittingDataSourceRuleConfiguration dataSourceRuleConfig = new ReadwriteSplittingDataSourceRuleConfiguration("readwrite_ds",
127                 new StaticReadwriteSplittingStrategyConfiguration("", Collections.emptyList()), null, null);
128         Map<String, AlgorithmConfiguration> loadBalancers = Collections.singletonMap("readwrite_ds", new AlgorithmConfiguration("TEST", new Properties()));
129         return new ReadwriteSplittingRuleConfiguration(new LinkedList<>(Collections.singleton(dataSourceRuleConfig)), loadBalancers);
130     }
131     
132     private ReadwriteSplittingRuleConfiguration createMultipleCurrentRuleConfigurations() {
133         ReadwriteSplittingDataSourceRuleConfiguration fooDataSourceRuleConfig = new ReadwriteSplittingDataSourceRuleConfiguration("foo_ds",
134                 new StaticReadwriteSplittingStrategyConfiguration("", Collections.emptyList()), null, "TEST");
135         ReadwriteSplittingDataSourceRuleConfiguration barDataSourceRuleConfig = new ReadwriteSplittingDataSourceRuleConfiguration("bar_ds",
136                 new StaticReadwriteSplittingStrategyConfiguration("", Collections.emptyList()), null, "TEST");
137         Map<String, AlgorithmConfiguration> loadBalancers = Collections.singletonMap("foo_ds", new AlgorithmConfiguration("TEST", new Properties()));
138         return new ReadwriteSplittingRuleConfiguration(new LinkedList<>(Arrays.asList(fooDataSourceRuleConfig, barDataSourceRuleConfig)), loadBalancers);
139     }
