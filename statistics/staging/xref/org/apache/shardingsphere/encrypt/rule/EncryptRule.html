25  import org.apache.shardingsphere.encrypt.exception.metadata.MismatchedEncryptAlgorithmTypeException;

33  import java.util.Collection;
34  import java.util.LinkedHashMap;
35  import java.util.Map;
36  import java.util.Map.Entry;
37  import java.util.Optional;
38  import java.util.stream.Collectors;

40  /**
41   * Encrypt rule.
42   */
43  public final class EncryptRule implements DatabaseRule {
44      
45      private final String databaseName;
46      
47      @Getter
48      private final EncryptRuleConfiguration configuration;
49      
50      private final Map<String, EncryptTable> tables;
51      
52      @Getter
53      private final RuleAttributes attributes;
54      
55      public EncryptRule(final String databaseName, final EncryptRuleConfiguration ruleConfig) {
56          this.databaseName = databaseName;
57          configuration = ruleConfig;
58          tables = new LinkedHashMap<>();
59          Map<String, EncryptAlgorithm> encryptors = ruleConfig.getEncryptors().entrySet().stream()
60                  .collect(Collectors.toMap(Entry::getKey, entry -> TypedSPILoader.getService(EncryptAlgorithm.class, entry.getValue().getType(), entry.getValue().getProps())));
61          for (EncryptTableRuleConfiguration each : ruleConfig.getTables()) {
62              each.getColumns().forEach(columnRuleConfig -> checkEncryptorType(columnRuleConfig, encryptors));
63              tables.put(each.getName().toLowerCase(), new EncryptTable(each, encryptors));
64          }
65          attributes = new RuleAttributes(new EncryptTableMapperRuleAttribute(ruleConfig.getTables()));
66      }
67      
68      private void checkEncryptorType(final EncryptColumnRuleConfiguration columnRuleConfig, final Map<String, EncryptAlgorithm> encryptors) {
69          ShardingSpherePreconditions.checkState(encryptors.containsKey(columnRuleConfig.getCipher().getEncryptorName())
70                  && encryptors.get(columnRuleConfig.getCipher().getEncryptorName()).getMetaData().isSupportDecrypt(),
71                  () -> new MismatchedEncryptAlgorithmTypeException(databaseName, "Cipher", columnRuleConfig.getCipher().getEncryptorName(), "decrypt"));
72          columnRuleConfig.getAssistedQuery().ifPresent(optional -> ShardingSpherePreconditions.checkState(encryptors.containsKey(optional.getEncryptorName())
73                  && encryptors.get(optional.getEncryptorName()).getMetaData().isSupportEquivalentFilter(),
74                  () -> new MismatchedEncryptAlgorithmTypeException(databaseName, "Assisted query", columnRuleConfig.getCipher().getEncryptorName(), "equivalent filter")));
75          columnRuleConfig.getLikeQuery().ifPresent(optional -> ShardingSpherePreconditions.checkState(encryptors.containsKey(optional.getEncryptorName())
76                  && encryptors.get(optional.getEncryptorName()).getMetaData().isSupportLike(),
77                  () -> new MismatchedEncryptAlgorithmTypeException(databaseName, "Like query", columnRuleConfig.getCipher().getEncryptorName(), "like")));
78      }
79      
80      /**
81       * Get all table names.
82       *
83       * @return all table names
84       */
85      public Collection<String> getAllTableNames() {
86          return tables.keySet();
87      }
88      
89      /**
90       * Find encrypt table.
91       * 
92       * @param tableName table name
93       * @return encrypt table
94       */
95      public Optional<EncryptTable> findEncryptTable(final String tableName) {
96          return Optional.ofNullable(tables.get(tableName.toLowerCase()));
97      }
98      
99      /**
100      * Get encrypt table.
101      *
102      * @param tableName table name
103      * @return encrypt table
104      */
105     public EncryptTable getEncryptTable(final String tableName) {
106         Optional<EncryptTable> encryptTable = findEncryptTable(tableName);
107         ShardingSpherePreconditions.checkState(encryptTable.isPresent(), () -> new EncryptTableNotFoundException(tableName));
108         return encryptTable.get();
109     }
110 }
