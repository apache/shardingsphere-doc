41  class InsertValueContextTest {
42      
43      @Test
44      void assertInstanceConstructedOk() throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
45          InsertValueContext insertValueContext = new InsertValueContext(Collections.emptyList(), Collections.emptyList(), 0);
46          assertThat(insertValueContext.getValueExpressions(), is(
47                  Plugins.getMemberAccessor().invoke(InsertValueContext.class.getDeclaredMethod("getValueExpressions", Collection.class), insertValueContext, Collections.emptyList())));
48          assertThat(insertValueContext.getParameters(), is(
49                  Plugins.getMemberAccessor().invoke(InsertValueContext.class.getDeclaredMethod("getParameters", List.class, int.class), insertValueContext, Collections.emptyList(), 0)));
50      }
51      
52      @Test
53      void assertGetLiteralValueWithParameterMarker() {
54          Collection<ExpressionSegment> assignments = makeParameterMarkerExpressionSegment();
55          String parameterValue = "test";
56          InsertValueContext insertValueContext = new InsertValueContext(assignments, Collections.singletonList(parameterValue), 0);
57          Optional<Object> valueFromInsertValueContext = insertValueContext.getLiteralValue(0);
58          assertTrue(valueFromInsertValueContext.isPresent());
59          assertThat(valueFromInsertValueContext.get(), is(parameterValue));
60      }
61      
62      @Test
63      void assertGetLiteralValueWhenParameterIsNull() {
64          Collection<ExpressionSegment> assignments = makeParameterMarkerExpressionSegment();
65          int parametersOffset = 0;
66          InsertValueContext insertValueContext = new InsertValueContext(assignments, Collections.singletonList(null), parametersOffset);
67          Optional<Object> literalValue = insertValueContext.getLiteralValue(0);
68          assertThat(false, is(literalValue.isPresent()));
69      }
70      
71      private Collection<ExpressionSegment> makeParameterMarkerExpressionSegment() {
72          return Collections.singleton(new ParameterMarkerExpressionSegment(0, 10, 5));
73      }
74      
75      @Test
76      void assertGetLiteralValueWhenLiteralExpressionSegment() {
77          Object literalObject = new Object();
78          Collection<ExpressionSegment> assignments = makeLiteralExpressionSegment(literalObject);
79          InsertValueContext insertValueContext = new InsertValueContext(assignments, Collections.emptyList(), 0);
80          Optional<Object> valueFromInsertValueContext = insertValueContext.getLiteralValue(0);
81          assertTrue(valueFromInsertValueContext.isPresent());
82          assertThat(valueFromInsertValueContext.get(), is(literalObject));
83      }
84      
85      private Collection<ExpressionSegment> makeLiteralExpressionSegment(final Object literalObject) {
86          return Collections.singleton(new LiteralExpressionSegment(0, 10, literalObject));
87      }
88      
89      @Test
90      void assertGetParameterCount() {
91          Collection<ExpressionSegment> expressions = Arrays.asList(
92                  new LiteralExpressionSegment(0, 10, null),
93                  new ExpressionProjectionSegment(0, 10, ""),
94                  new ParameterMarkerExpressionSegment(0, 10, 5),
95                  new BinaryOperationExpression(0, 0, new ColumnSegment(0, 0, new IdentifierValue("")), new ParameterMarkerExpressionSegment(0, 10, 5), "=", ""));
96          InsertValueContext insertValueContext = new InsertValueContext(expressions, Arrays.asList("", ""), 0);
97          assertThat(insertValueContext.getParameterCount(), is(2));
98      }
99  }
