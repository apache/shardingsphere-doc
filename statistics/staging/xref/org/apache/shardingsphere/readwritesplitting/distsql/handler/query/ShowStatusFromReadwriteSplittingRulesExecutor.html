50  /**
51   * Show status from readwrite-splitting rules executor.
52   */
53  @Setter
54  public final class ShowStatusFromReadwriteSplittingRulesExecutor implements DistSQLQueryExecutor<ShowStatusFromReadwriteSplittingRulesStatement>, DistSQLExecutorDatabaseAware {
55      
56      private ShardingSphereDatabase database;
57      
58      @Override
59      public Collection<String> getColumnNames(final ShowStatusFromReadwriteSplittingRulesStatement sqlStatement) {
60          return Arrays.asList("storage_unit", "status");
61      }
62      
63      @Override
64      public Collection<LocalDataQueryResultRow> getRows(final ShowStatusFromReadwriteSplittingRulesStatement sqlStatement, final ContextManager contextManager) {
65          Collection<String> allReadResources = getAllReadResources(sqlStatement.getGroupName());
66          Map<String, StorageNodeDataSource> persistentReadResources = getPersistentReadResources(database.getName(), contextManager);
67          return buildRows(allReadResources, persistentReadResources);
68      }
69      
70      @SuppressWarnings("unchecked")
71      private Collection<String> getAllReadResources(final String groupName) {
72          Collection<String> exportKeys = Arrays.asList(ExportableConstants.EXPORT_STATIC_READWRITE_SPLITTING_RULE, ExportableConstants.EXPORT_DYNAMIC_READWRITE_SPLITTING_RULE);
73          Map<String, Object> exportMap = database.getRuleMetaData().findRules(ExportableRule.class).stream()
74                  .filter(each -> new RuleExportEngine(each).containExportableKey(exportKeys)).findFirst().map(each -> new RuleExportEngine(each).export(exportKeys)).orElse(Collections.emptyMap());
75          Map<String, Map<String, String>> allReadwriteRuleMap = exportMap.values().stream().map(each -> ((Map<String, Map<String, String>>) each).entrySet())
76                  .flatMap(Collection::stream).collect(Collectors.toMap(Entry::getKey, Entry::getValue, (oldValue, currentValue) -> currentValue, LinkedHashMap::new));
77          if (!Strings.isNullOrEmpty(groupName)) {
78              allReadwriteRuleMap = allReadwriteRuleMap.entrySet().stream().filter(each -> groupName.equalsIgnoreCase(each.getKey()))
79                      .collect(Collectors.toMap(Entry::getKey, Entry::getValue, (oldValue, currentValue) -> currentValue, LinkedHashMap::new));
80          }
81          return allReadwriteRuleMap.values().stream().map(each -> each.get(ExportableItemConstants.REPLICA_DATA_SOURCE_NAMES)).filter(each -> null != each && !each.isEmpty())
82                  .map(this::deconstructString).flatMap(Collection::stream).collect(Collectors.toCollection(LinkedHashSet::new));
83      }
84      
85      private Map<String, StorageNodeDataSource> getPersistentReadResources(final String databaseName, final ContextManager contextManager) {
86          if (!contextManager.getInstanceContext().isCluster()) {
87              return Collections.emptyMap();
88          }
89          Map<String, StorageNodeDataSource> storageNodes = new StorageNodeStatusService(contextManager.getMetaDataContexts().getPersistService().getRepository()).loadStorageNodes();
90          Map<String, StorageNodeDataSource> result = new HashMap<>();
91          storageNodes.entrySet().stream().filter(entry -> StorageNodeRole.MEMBER == entry.getValue().getRole()).forEach(entry -> {
92              QualifiedDatabase qualifiedDatabase = new QualifiedDatabase(entry.getKey());
93              if (databaseName.equalsIgnoreCase(qualifiedDatabase.getDatabaseName())) {
94                  result.put(qualifiedDatabase.getDataSourceName(), entry.getValue());
95              }
96          });
97          return result;
98      }
99      
100     private Collection<LocalDataQueryResultRow> buildRows(final Collection<String> readResources, final Map<String, StorageNodeDataSource> persistentReadResources) {
101         Map<DataSourceState, Map<String, StorageNodeDataSource>> persistentReadResourceGroup = persistentReadResources.entrySet().stream()
102                 .collect(Collectors.groupingBy(each -> each.getValue().getStatus(), Collectors.toMap(Entry::getKey, Entry::getValue)));
103         Map<String, StorageNodeDataSource> disabledReadResources = persistentReadResourceGroup.getOrDefault(DataSourceState.DISABLED, Collections.emptyMap());
104         Map<String, StorageNodeDataSource> enabledReadResources = persistentReadResourceGroup.getOrDefault(DataSourceState.ENABLED, Collections.emptyMap());
105         readResources.removeIf(disabledReadResources::containsKey);
106         readResources.addAll(enabledReadResources.keySet());
107         readResources.addAll(disabledReadResources.keySet());
108         return readResources.stream().map(each -> buildRow(each, disabledReadResources.get(each))).collect(Collectors.toList());
109     }
110     
111     private List<String> deconstructString(final String str) {
112         return new LinkedList<>(Arrays.asList(str.split(",")));
113     }
114     
115     private LocalDataQueryResultRow buildRow(final String resource, final StorageNodeDataSource storageNodeDataSource) {
116         return null == storageNodeDataSource ? new LocalDataQueryResultRow(resource, DataSourceState.ENABLED) : new LocalDataQueryResultRow(resource, storageNodeDataSource.getStatus());
117     }
118     
119     @Override
120     public Class<ShowStatusFromReadwriteSplittingRulesStatement> getType() {
121         return ShowStatusFromReadwriteSplittingRulesStatement.class;
122     }
123 }
