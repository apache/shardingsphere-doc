80  /**
81   * Proxy SQL Executor.
82   */
83  public final class ProxySQLExecutor {
84      
85      private final String type;
86      
87      private final ProxyDatabaseConnectionManager databaseConnectionManager;
88      
89      private final ProxyJDBCExecutor regularExecutor;
90      
91      private final RawExecutor rawExecutor;
92      
93      @Getter
94      private final SQLFederationEngine sqlFederationEngine;
95      
96      private final Collection<TransactionHook> transactionHooks = ShardingSphereServiceLoader.getServiceInstances(TransactionHook.class);
97      
98      public ProxySQLExecutor(final String type, final ProxyDatabaseConnectionManager databaseConnectionManager, final DatabaseConnector databaseConnector, final QueryContext queryContext) {
99          this.type = type;
100         this.databaseConnectionManager = databaseConnectionManager;
101         ExecutorEngine executorEngine = BackendExecutorContext.getInstance().getExecutorEngine();
102         ConnectionContext connectionContext = databaseConnectionManager.getConnectionSession().getConnectionContext();
103         JDBCExecutor jdbcExecutor = new JDBCExecutor(executorEngine, connectionContext);
104         regularExecutor = new ProxyJDBCExecutor(type, databaseConnectionManager.getConnectionSession(), databaseConnector, jdbcExecutor);
105         rawExecutor = new RawExecutor(executorEngine, connectionContext);
106         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
107         String currentDatabaseName = databaseConnectionManager.getConnectionSession().getCurrentDatabaseName();
108         String currentSchemaName = getSchemaName(queryContext.getSqlStatementContext(), metaDataContexts.getMetaData().getDatabase(currentDatabaseName));
109         sqlFederationEngine = new SQLFederationEngine(currentDatabaseName, currentSchemaName, metaDataContexts.getMetaData(), metaDataContexts.getStatistics(), jdbcExecutor);
110     }
111     
112     private String getSchemaName(final SQLStatementContext sqlStatementContext, final ShardingSphereDatabase database) {
113         String defaultSchemaName = new DatabaseTypeRegistry(sqlStatementContext.getDatabaseType()).getDefaultSchemaName(database.getName());
114         return sqlStatementContext instanceof TableAvailable
115                 ? ((TableAvailable) sqlStatementContext).getTablesContext().getSchemaName().orElse(defaultSchemaName)
116                 : defaultSchemaName;
117     }
118     
119     /**
120      * Check execute prerequisites.
121      *
122      * @param executionContext execution context
123      */
124     public void checkExecutePrerequisites(final ExecutionContext executionContext) {
125         ShardingSpherePreconditions.checkState(isValidExecutePrerequisites(executionContext), () -> new TableModifyInTransactionException(getTableName(executionContext)));
126     }
127     
128     private boolean isValidExecutePrerequisites(final ExecutionContext executionContext) {
129         return !isExecuteDDLInXATransaction(executionContext.getSqlStatementContext().getSqlStatement())
130                 && !isExecuteDDLInPostgreSQLOpenGaussTransaction(executionContext.getSqlStatementContext().getSqlStatement());
131     }
132     
133     private boolean isExecuteDDLInXATransaction(final SQLStatement sqlStatement) {
134         TransactionType transactionType = TransactionUtils.getTransactionType(databaseConnectionManager.getConnectionSession().getConnectionContext().getTransactionContext());
135         TransactionStatus transactionStatus = databaseConnectionManager.getConnectionSession().getTransactionStatus();
136         return TransactionType.XA == transactionType && transactionStatus.isInTransaction() && isUnsupportedDDLStatement(sqlStatement);
137     }
138     
139     private boolean isExecuteDDLInPostgreSQLOpenGaussTransaction(final SQLStatement sqlStatement) {
140         // TODO implement DDL statement commit/rollback in PostgreSQL/openGauss transaction
141         boolean isPostgreSQLOpenGaussStatement = isPostgreSQLOrOpenGaussStatement(sqlStatement);
142         boolean isSupportedStatement = isSupportedSQLStatement(sqlStatement);
143         return sqlStatement instanceof DDLStatement
144                 && !isSupportedStatement && isPostgreSQLOpenGaussStatement && databaseConnectionManager.getConnectionSession().getTransactionStatus().isInTransaction();
145     }
146     
147     private boolean isSupportedSQLStatement(final SQLStatement sqlStatement) {
148         return isCursorStatement(sqlStatement) || sqlStatement instanceof TruncateStatement;
149     }
150     
151     private boolean isCursorStatement(final SQLStatement sqlStatement) {
152         return sqlStatement instanceof OpenGaussCursorStatement
153                 || sqlStatement instanceof CloseStatement || sqlStatement instanceof MoveStatement || sqlStatement instanceof FetchStatement;
154     }
155     
156     private boolean isUnsupportedDDLStatement(final SQLStatement sqlStatement) {
157         if (isPostgreSQLOrOpenGaussStatement(sqlStatement) && isSupportedSQLStatement(sqlStatement)) {
158             return false;
159         }
160         return sqlStatement instanceof DDLStatement;
161     }
162     
163     private boolean isPostgreSQLOrOpenGaussStatement(final SQLStatement sqlStatement) {
164         return sqlStatement instanceof PostgreSQLStatement || sqlStatement instanceof OpenGaussStatement;
165     }
166     
167     private String getTableName(final ExecutionContext executionContext) {
168         return executionContext.getSqlStatementContext() instanceof TableAvailable && !((TableAvailable) executionContext.getSqlStatementContext()).getTablesContext().getSimpleTables().isEmpty()
169                 ? ((TableAvailable) executionContext.getSqlStatementContext()).getTablesContext().getSimpleTables().iterator().next().getTableName().getIdentifier().getValue()
170                 : "unknown_table";
171     }
172     
173     /**
174      * Execute SQL.
175      *
176      * @param executionContext execution context
177      * @return execute results
178      * @throws SQLException SQL exception
179      */
180     public List<ExecuteResult> execute(final ExecutionContext executionContext) throws SQLException {
181         String databaseName = databaseConnectionManager.getConnectionSession().getUsedDatabaseName();
182         Collection<ShardingSphereRule> rules = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getDatabase(databaseName).getRuleMetaData().getRules();
183         int maxConnectionsSizePerQuery = ProxyContext.getInstance()
184                 .getContextManager().getMetaDataContexts().getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
185         boolean isReturnGeneratedKeys = executionContext.getSqlStatementContext().getSqlStatement() instanceof MySQLInsertStatement;
186         return hasRawExecutionRule(rules) ? rawExecute(executionContext, rules, maxConnectionsSizePerQuery)
187                 : useDriverToExecute(executionContext, rules, maxConnectionsSizePerQuery, isReturnGeneratedKeys, SQLExecutorExceptionHandler.isExceptionThrown());
188     }
189     
190     private boolean hasRawExecutionRule(final Collection<ShardingSphereRule> rules) {
191         for (ShardingSphereRule each : rules) {
192             if (each.getAttributes().findAttribute(RawExecutionRuleAttribute.class).isPresent()) {
193                 return true;
194             }
195         }
196         return false;
197     }
198     
199     private List<ExecuteResult> rawExecute(final ExecutionContext executionContext, final Collection<ShardingSphereRule> rules, final int maxConnectionsSizePerQuery) throws SQLException {
200         RawExecutionPrepareEngine prepareEngine = new RawExecutionPrepareEngine(maxConnectionsSizePerQuery, rules);
201         ExecutionGroupContext<RawSQLExecutionUnit> executionGroupContext;
202         try {
203             String databaseName = databaseConnectionManager.getConnectionSession().getUsedDatabaseName();
204             executionGroupContext = prepareEngine.prepare(databaseName, executionContext.getRouteContext(), executionContext.getExecutionUnits(),
205                     new ExecutionGroupReportContext(databaseConnectionManager.getConnectionSession().getProcessId(),
206                             databaseName, databaseConnectionManager.getConnectionSession().getConnectionContext().getGrantee()));
207         } catch (final SQLException ex) {
208             return getSaneExecuteResults(executionContext, ex);
209         }
210         // TODO handle query header
211         return rawExecutor.execute(executionGroupContext, executionContext.getQueryContext(), new RawSQLExecutorCallback());
212     }
213     
214     private List<ExecuteResult> useDriverToExecute(final ExecutionContext executionContext, final Collection<ShardingSphereRule> rules,
215                                                    final int maxConnectionsSizePerQuery, final boolean isReturnGeneratedKeys, final boolean isExceptionThrown) throws SQLException {
216         JDBCBackendStatement statementManager = (JDBCBackendStatement) databaseConnectionManager.getConnectionSession().getStatementManager();
217         String databaseName = databaseConnectionManager.getConnectionSession().getUsedDatabaseName();
218         DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine = new DriverExecutionPrepareEngine<>(
219                 type, maxConnectionsSizePerQuery, databaseConnectionManager, statementManager, new StatementOption(isReturnGeneratedKeys), rules,
220                 ProxyContext.getInstance().getContextManager().getDatabase(databaseName).getResourceMetaData().getStorageUnits());
221         ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext;
222         try {
223             executionGroupContext = prepareEngine.prepare(databaseName, executionContext.getRouteContext(), executionContext.getExecutionUnits(),
224                     new ExecutionGroupReportContext(databaseConnectionManager.getConnectionSession().getProcessId(),
225                             databaseName, databaseConnectionManager.getConnectionSession().getConnectionContext().getGrantee()));
226         } catch (final SQLException ex) {
227             return getSaneExecuteResults(executionContext, ex);
228         }
229         executeTransactionHooksBeforeExecuteSQL(databaseConnectionManager.getConnectionSession());
230         return regularExecutor.execute(executionContext.getQueryContext(), executionGroupContext, isReturnGeneratedKeys, isExceptionThrown);
231     }
232     
233     private void executeTransactionHooksBeforeExecuteSQL(final ConnectionSession connectionSession) throws SQLException {
234         if (!getTransactionContext(connectionSession).isInTransaction()) {
235             return;
236         }
237         for (TransactionHook each : transactionHooks) {
238             each.beforeExecuteSQL(connectionSession.getDatabaseConnectionManager().getCachedConnections().values(), getTransactionContext(connectionSession),
239                     connectionSession.getIsolationLevel().orElse(TransactionIsolationLevel.READ_COMMITTED));
240         }
241     }
242     
243     private TransactionConnectionContext getTransactionContext(final ConnectionSession connectionSession) {
244         return connectionSession.getDatabaseConnectionManager().getConnectionSession().getConnectionContext().getTransactionContext();
245     }
246     
247     private List<ExecuteResult> getSaneExecuteResults(final ExecutionContext executionContext, final SQLException originalException) throws SQLException {
248         DatabaseType databaseType = ProxyContext.getInstance().getContextManager().getDatabase(databaseConnectionManager.getConnectionSession().getUsedDatabaseName()).getProtocolType();
249         Optional<ExecuteResult> executeResult = new SaneQueryResultEngine(databaseType).getSaneQueryResult(executionContext.getSqlStatementContext().getSqlStatement(), originalException);
250         return executeResult.map(Collections::singletonList).orElseThrow(() -> originalException);
251     }
