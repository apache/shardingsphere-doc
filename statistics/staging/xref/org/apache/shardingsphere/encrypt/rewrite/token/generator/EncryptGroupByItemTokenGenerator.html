51  /**
52   * Group by item token generator for encrypt.
53   */
54  @HighFrequencyInvocation
55  @Setter
56  public final class EncryptGroupByItemTokenGenerator implements CollectionSQLTokenGenerator<SQLStatementContext>, EncryptRuleAware, SchemaMetaDataAware, DatabaseTypeAware {
57      
58      private EncryptRule encryptRule;
59      
60      private Map<String, ShardingSphereSchema> schemas;
61      
62      private ShardingSphereSchema defaultSchema;
63      
64      private DatabaseType databaseType;
65      
66      @Override
67      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
68          return sqlStatementContext instanceof SelectStatementContext && containsGroupByItem(sqlStatementContext);
69      }
70      
71      @Override
72      public Collection<SQLToken> generateSQLTokens(final SQLStatementContext sqlStatementContext) {
73          Collection<SQLToken> result = new LinkedHashSet<>();
74          ShardingSphereSchema schema = ((TableAvailable) sqlStatementContext).getTablesContext().getSchemaName().map(schemas::get).orElseGet(() -> defaultSchema);
75          for (OrderByItem each : getGroupByItems(sqlStatementContext)) {
76              if (each.getSegment() instanceof ColumnOrderByItemSegment) {
77                  ColumnSegment columnSegment = ((ColumnOrderByItemSegment) each.getSegment()).getColumn();
78                  Map<String, String> columnTableNames = ((TableAvailable) sqlStatementContext).getTablesContext().findTableNames(Collections.singleton(columnSegment), schema);
79                  result.addAll(generateSQLTokensWithColumnSegments(Collections.singleton(columnSegment), columnTableNames));
80              }
81          }
82          return result;
83      }
84      
85      private Collection<SubstitutableColumnNameToken> generateSQLTokensWithColumnSegments(final Collection<ColumnSegment> columnSegments, final Map<String, String> columnTableNames) {
86          Collection<SubstitutableColumnNameToken> result = new LinkedList<>();
87          for (ColumnSegment each : columnSegments) {
88              String tableName = columnTableNames.getOrDefault(each.getExpression(), "");
89              Optional<EncryptTable> encryptTable = encryptRule.findEncryptTable(tableName);
90              String columnName = each.getIdentifier().getValue();
91              if (!encryptTable.isPresent() || !encryptTable.get().isEncryptColumn(columnName)) {
92                  continue;
93              }
94              int startIndex = each.getOwner().isPresent() ? each.getOwner().get().getStopIndex() + 2 : each.getStartIndex();
95              int stopIndex = each.getStopIndex();
96              EncryptColumn encryptColumn = encryptTable.get().getEncryptColumn(columnName);
97              SubstitutableColumnNameToken encryptColumnNameToken = encryptColumn.getAssistedQuery()
98                      .map(optional -> new SubstitutableColumnNameToken(startIndex, stopIndex, createColumnProjections(optional.getName(), each.getIdentifier().getQuoteCharacter()), databaseType))
99                      .orElseGet(() -> new SubstitutableColumnNameToken(startIndex, stopIndex, createColumnProjections(encryptColumn.getCipher().getName(), each.getIdentifier().getQuoteCharacter()),
100                             databaseType));
101             result.add(encryptColumnNameToken);
102         }
103         return result;
104     }
105     
106     private Collection<OrderByItem> getGroupByItems(final SQLStatementContext sqlStatementContext) {
107         if (!(sqlStatementContext instanceof SelectStatementContext)) {
108             return Collections.emptyList();
109         }
110         SelectStatementContext statementContext = (SelectStatementContext) sqlStatementContext;
111         Collection<OrderByItem> result = new LinkedList<>(statementContext.getGroupByContext().getItems());
112         for (SelectStatementContext each : statementContext.getSubqueryContexts().values()) {
113             result.addAll(getGroupByItems(each));
114         }
115         return result;
116     }
117     
118     private boolean containsGroupByItem(final SQLStatementContext sqlStatementContext) {
119         if (!(sqlStatementContext instanceof SelectStatementContext)) {
120             return false;
121         }
122         SelectStatementContext statementContext = (SelectStatementContext) sqlStatementContext;
123         if (!statementContext.getGroupByContext().getItems().isEmpty()) {
124             return true;
125         }
126         for (SelectStatementContext each : statementContext.getSubqueryContexts().values()) {
127             if (containsGroupByItem(each)) {
128                 return true;
129             }
130         }
131         return false;
132     }
133     
134     private Collection<Projection> createColumnProjections(final String columnName, final QuoteCharacter quoteCharacter) {
135         return Collections.singleton(new ColumnProjection(null, new IdentifierValue(columnName, quoteCharacter), null, databaseType));
136     }
137 }
