91      private void assertCreateGenerateKeyContextWithoutGenerateKeyColumnConfiguration(final InsertStatement insertStatement) {
92          insertStatement.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("tbl1"))));
93          insertStatement.setInsertColumns(new InsertColumnsSegment(0, 0, Collections.singletonList(new ColumnSegment(0, 0, new IdentifierValue("id")))));
94          assertFalse(new GeneratedKeyContextEngine(insertStatement, schema).createGenerateKeyContext(Collections.emptyList(),

123     private void assertCreateGenerateKeyContextWhenCreateWithGenerateKeyColumnConfiguration(final InsertStatement insertStatement) {
124         insertStatement.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("tbl"))));
125         insertStatement.setInsertColumns(new InsertColumnsSegment(0, 0, Collections.singletonList(new ColumnSegment(0, 0, new IdentifierValue("id")))));
126         List<ExpressionSegment> expressionSegments = Collections.singletonList(new LiteralExpressionSegment(0, 0, 1));
127         insertStatement.getValues().add(new InsertValuesSegment(0, 0, expressionSegments));
128         Optional<GeneratedKeyContext> actual = new GeneratedKeyContextEngine(insertStatement, schema)
129                 .createGenerateKeyContext(Collections.singletonList("id"), Collections.singletonList(expressionSegments), Collections.singletonList(1));
130         assertTrue(actual.isPresent());
131         assertThat(actual.get().getGeneratedValues().size(), is(1));
132     }

159     private void assertCreateGenerateKeyContextWhenFind(final InsertStatement insertStatement) {
160         insertStatement.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("tbl"))));
161         insertStatement.setInsertColumns(new InsertColumnsSegment(0, 0, Collections.singletonList(new ColumnSegment(0, 0, new IdentifierValue("id")))));
162         insertStatement.getValues().add(new InsertValuesSegment(0, 0, Collections.singletonList(new ParameterMarkerExpressionSegment(1, 2, 0))));
163         insertStatement.getValues().add(new InsertValuesSegment(0, 0, Collections.singletonList(new LiteralExpressionSegment(1, 2, 100))));
164         insertStatement.getValues().add(new InsertValuesSegment(0, 0, Collections.singletonList(new LiteralExpressionSegment(1, 2, "value"))));
165         insertStatement.getValues().add(new InsertValuesSegment(0, 0, Collections.singletonList(new CommonExpressionSegment(1, 2, "ignored value"))));
166         List<List<ExpressionSegment>> valueExpressions = insertStatement.getValues().stream().map(InsertValuesSegment::getValues).collect(Collectors.toList());
167         Optional<GeneratedKeyContext> actual = new GeneratedKeyContextEngine(insertStatement, schema)
168                 .createGenerateKeyContext(Collections.singletonList("id"), valueExpressions, Collections.singletonList(1));
169         assertTrue(actual.isPresent());
170         assertThat(actual.get().getGeneratedValues().size(), is(3));
171         Iterator<Comparable<?>> generatedValuesIterator = actual.get().getGeneratedValues().iterator();
172         assertThat(generatedValuesIterator.next(), is(1));
173         assertThat(generatedValuesIterator.next(), is(100));
174         assertThat(generatedValuesIterator.next(), is("value"));
175         assertTrue(new GeneratedKeyContextEngine(insertStatement, schema).createGenerateKeyContext(Collections.emptyList(), Collections.emptyList(), Collections.singletonList(1)).isPresent());
176     }
