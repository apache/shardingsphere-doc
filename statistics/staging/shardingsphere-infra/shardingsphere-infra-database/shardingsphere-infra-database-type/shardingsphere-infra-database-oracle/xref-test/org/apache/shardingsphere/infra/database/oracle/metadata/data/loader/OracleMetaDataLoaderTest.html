80      @Test
81      void assertLoadCondition1() throws SQLException {
82          DataSource dataSource = mockDataSource();
83          ResultSet resultSet = mockTableMetaDataResultSet();
84          when(dataSource.getConnection().prepareStatement(ALL_TAB_COLUMNS_SQL_CONDITION1).executeQuery()).thenReturn(resultSet);
85          ResultSet indexResultSet = mockIndexMetaDataResultSet();
86          when(dataSource.getConnection().prepareStatement(ALL_INDEXES_SQL).executeQuery()).thenReturn(indexResultSet);
87          ResultSet primaryKeys = mockPrimaryKeysMetaDataResultSet();
88          when(dataSource.getConnection().prepareStatement(ALL_CONSTRAINTS_SQL_WITHOUT_TABLES).executeQuery()).thenReturn(primaryKeys);
89          when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(12);
90          when(dataSource.getConnection().getMetaData().getDatabaseMinorVersion()).thenReturn(2);
91          Collection<SchemaMetaData> actual = getDialectTableMetaDataLoader().load(
92                  new MetaDataLoaderMaterial(Collections.singleton("tbl"), "foo_ds", dataSource, new OracleDatabaseType(), "sharding_db"));
93          assertTableMetaDataMap(actual);
94          TableMetaData actualTableMetaData = actual.iterator().next().getTables().iterator().next();
95          Iterator<ColumnMetaData> columnsIterator = actualTableMetaData.getColumns().iterator();
96          assertThat(columnsIterator.next(), is(new ColumnMetaData("id", Types.INTEGER, false, true, true, true, false, false)));
97          assertThat(columnsIterator.next(), is(new ColumnMetaData("name", Types.VARCHAR, false, false, false, false, false, true)));
98      }

289     private void assertTableMetaDataMap(final Collection<SchemaMetaData> schemaMetaDataList) {
290         assertThat(schemaMetaDataList.size(), is(1));
291         TableMetaData actualTableMetaData = schemaMetaDataList.iterator().next().getTables().iterator().next();
292         assertThat(actualTableMetaData.getColumns().size(), is(3));
293         assertThat(actualTableMetaData.getIndexes().size(), is(1));
294         Iterator<IndexMetaData> indexesIterator = actualTableMetaData.getIndexes().iterator();
295         IndexMetaData indexMetaData = new IndexMetaData("id");
296         indexMetaData.setUnique(true);
297         assertThat(indexesIterator.next(), is(indexMetaData));
298     }
