184 @NoArgsConstructor

187     public MySQLDDLStatementSQLVisitor(final Properties props) {
188         super(props);
189     }

191     @Override
192     public ASTNode visitCreateView(final CreateViewContext ctx) {
193         MySQLCreateViewStatement result = new MySQLCreateViewStatement();
194         result.setView((SimpleTableSegment) visit(ctx.viewName()));
195         result.setViewDefinition(getOriginalText(ctx.select()));
196         result.setSelect((MySQLSelectStatement) visit(ctx.select()));
197         return result;
198     }
199     
200     @Override
201     public ASTNode visitAlterView(final AlterViewContext ctx) {
202         MySQLAlterViewStatement result = new MySQLAlterViewStatement();
203         result.setView((SimpleTableSegment) visit(ctx.viewName()));
204         result.setViewDefinition(getOriginalText(ctx.select()));
205         result.setSelect((MySQLSelectStatement) visit(ctx.select()));
206         return result;
207     }
208     
209     @SuppressWarnings("unchecked")
210     @Override
211     public ASTNode visitDropView(final DropViewContext ctx) {
212         MySQLDropViewStatement result = new MySQLDropViewStatement();
213         result.getViews().addAll(((CollectionValue<SimpleTableSegment>) visit(ctx.viewNames())).getValue());
214         return result;
215     }

217     @Override

221         result.setIfNotExists(null != ctx.ifNotExists());
222         return result;
223     }

228     }

230     @Override

232         MySQLDropDatabaseStatement result = new MySQLDropDatabaseStatement();
233         result.setDatabaseName(new IdentifierValue(ctx.schemaName().getText()).getValue());
234         result.setIfExists(null != ctx.ifExists());
235         return result;
236     }
237     
238     @SuppressWarnings("unchecked")
239     @Override
240     public ASTNode visitCreateTable(final CreateTableContext ctx) {
241         MySQLCreateTableStatement result = new MySQLCreateTableStatement(null != ctx.ifNotExists());
242         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
243         if (null != ctx.createDefinitionClause()) {
244             CollectionValue<CreateDefinitionSegment> createDefinitions = (CollectionValue<CreateDefinitionSegment>) visit(ctx.createDefinitionClause());
245             for (CreateDefinitionSegment each : createDefinitions.getValue()) {
246                 if (each instanceof ColumnDefinitionSegment) {
247                     result.getColumnDefinitions().add((ColumnDefinitionSegment) each);
248                 } else if (each instanceof ConstraintDefinitionSegment) {
249                     result.getConstraintDefinitions().add((ConstraintDefinitionSegment) each);
250                 }
251             }
252         }
253         return result;
254     }
255     
256     @Override
257     public ASTNode visitCreateDefinitionClause(final CreateDefinitionClauseContext ctx) {
258         CollectionValue<CreateDefinitionSegment> result = new CollectionValue<>();
259         for (TableElementContext each : ctx.tableElementList().tableElement()) {
260             if (null != each.columnDefinition()) {
261                 result.getValue().add((ColumnDefinitionSegment) visit(each.columnDefinition()));
262             }
263             if (null != each.tableConstraintDef()) {
264                 result.getValue().add((ConstraintDefinitionSegment) visit(each.tableConstraintDef()));
265             }
266         }
267         return result;
268     }
269     
270     @Override
271     public ASTNode visitCreateLikeClause(final CreateLikeClauseContext ctx) {
272         return visit(ctx.tableName());
273     }
274     
275     @SuppressWarnings("unchecked")
276     @Override
277     public ASTNode visitAlterTable(final AlterTableContext ctx) {
278         MySQLAlterTableStatement result = new MySQLAlterTableStatement();
279         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
280         if (null != ctx.alterTableActions() && null != ctx.alterTableActions().alterCommandList() && null != ctx.alterTableActions().alterCommandList().alterList()) {
281             for (AlterDefinitionSegment each : ((CollectionValue<AlterDefinitionSegment>) visit(ctx.alterTableActions().alterCommandList().alterList())).getValue()) {
282                 if (each instanceof AddColumnDefinitionSegment) {
283                     result.getAddColumnDefinitions().add((AddColumnDefinitionSegment) each);
284                 } else if (each instanceof ModifyColumnDefinitionSegment) {
285                     result.getModifyColumnDefinitions().add((ModifyColumnDefinitionSegment) each);
286                 } else if (each instanceof ChangeColumnDefinitionSegment) {
287                     result.getChangeColumnDefinitions().add((ChangeColumnDefinitionSegment) each);
288                 } else if (each instanceof DropColumnDefinitionSegment) {
289                     result.getDropColumnDefinitions().add((DropColumnDefinitionSegment) each);
290                 } else if (each instanceof AddConstraintDefinitionSegment) {
291                     result.getAddConstraintDefinitions().add((AddConstraintDefinitionSegment) each);
292                 } else if (each instanceof DropConstraintDefinitionSegment) {
293                     result.getDropConstraintDefinitions().add((DropConstraintDefinitionSegment) each);
294                 } else if (each instanceof RenameTableDefinitionSegment) {
295                     result.setRenameTable(((RenameTableDefinitionSegment) each).getRenameTable());
296                 } else if (each instanceof ConvertTableDefinitionSegment) {
297                     result.setConvertTableDefinition((ConvertTableDefinitionSegment) each);
298                 } else if (each instanceof DropIndexDefinitionSegment) {
299                     result.getDropIndexDefinitions().add((DropIndexDefinitionSegment) each);
300                 } else if (each instanceof RenameIndexDefinitionSegment) {
301                     result.getRenameIndexDefinitions().add((RenameIndexDefinitionSegment) each);
302                 }
303             }
304         }
305         return result;
306     }
307     
308     private ColumnDefinitionSegment generateColumnDefinitionSegment(final ColumnSegment column, final FieldDefinitionContext ctx) {
309         DataTypeSegment dataTypeSegment = (DataTypeSegment) visit(ctx.dataType());
310         boolean isPrimaryKey = ctx.columnAttribute().stream().anyMatch(each -> null != each.KEY() && null == each.UNIQUE());
311         // TODO parse not null
312         return new ColumnDefinitionSegment(column.getStartIndex(), ctx.getStop().getStopIndex(), column, dataTypeSegment, isPrimaryKey, false);
313     }
314     
315     @Override
316     public ASTNode visitAlterConstraint(final AlterConstraintContext ctx) {
317         return new ModifyConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), (ConstraintSegment) visit(ctx.constraintName()));
318     }
319     
320     @Override
321     public ASTNode visitAlterList(final AlterListContext ctx) {
322         CollectionValue<AlterDefinitionSegment> result = new CollectionValue<>();
323         if (ctx.alterListItem().isEmpty()) {
324             return result;
325         }
326         for (AlterListItemContext each : ctx.alterListItem()) {
327             if (each instanceof AddColumnContext) {
328                 result.getValue().add((AddColumnDefinitionSegment) visit(each));
329             }
330             if (each instanceof AlterConstraintContext || each instanceof AlterCheckContext) {
331                 result.getValue().add((AlterDefinitionSegment) visit(each));
332             }
333             if (each instanceof ChangeColumnContext) {
334                 result.getValue().add(generateModifyColumnDefinitionSegment((ChangeColumnContext) each));
335             }
336             if (each instanceof ModifyColumnContext) {
337                 result.getValue().add(generateModifyColumnDefinitionSegment((ModifyColumnContext) each));
338             }
339             if (each instanceof AlterTableDropContext) {
340                 AlterTableDropContext alterTableDrop = (AlterTableDropContext) each;
341                 if (null != alterTableDrop.CHECK() || null != alterTableDrop.CONSTRAINT()) {
342                     ConstraintSegment constraintSegment = new ConstraintSegment(alterTableDrop.identifier().getStart().getStartIndex(), alterTableDrop.identifier().getStop().getStopIndex(),
343                             (IdentifierValue) visit(alterTableDrop.identifier()));
344                     result.getValue().add(new DropConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), constraintSegment));
345                 } else if (null == alterTableDrop.KEY() && null == alterTableDrop.keyOrIndex()) {
346                     result.getValue().add(generateDropColumnDefinitionSegment(alterTableDrop));
347                 } else if (null != alterTableDrop.keyOrIndex()) {
348                     IndexSegment indexSegment = (IndexSegment) visit(alterTableDrop.indexName());
349                     result.getValue().add(new DropIndexDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), indexSegment));
350                 }
351             }
352             if (each instanceof AddTableConstraintContext) {
353                 result.getValue().add((AddConstraintDefinitionSegment) visit(each));
354             }
355             if (each instanceof AlterRenameTableContext) {
356                 result.getValue().add((RenameTableDefinitionSegment) visit(each));
357             }
358             if (each instanceof AlterConvertContext) {
359                 result.getValue().add((ConvertTableDefinitionSegment) visit(each));
360             }
361             if (each instanceof RenameIndexContext) {
362                 result.getValue().add((RenameIndexDefinitionSegment) visit(each));
363             }
364         }
365         return result;
366     }
367     
368     @Override
369     public ASTNode visitAlterConvert(final AlterConvertContext ctx) {
370         ConvertTableDefinitionSegment result = new ConvertTableDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), (CharsetNameSegment) visit(ctx.charsetName()));
371         if (null != ctx.collateClause()) {
372             result.setCollateValue((SimpleExpressionSegment) visit(ctx.collateClause()));
373         }

483     @Override
484     public ASTNode visitTableConstraintDef(final TableConstraintDefContext ctx) {
485         ConstraintDefinitionSegment result = new ConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
486         if (null != ctx.constraintClause() && null != ctx.constraintClause().constraintName()) {
487             result.setConstraintName((ConstraintSegment) visit(ctx.constraintClause().constraintName()));
488         }
489         if (null != ctx.KEY() && null != ctx.PRIMARY()) {
490             result.getPrimaryKeyColumns().addAll(((CollectionValue) visit(ctx.keyListWithExpression())).getValue());
491             return result;
492         }
493         if (null != ctx.FOREIGN()) {
494             result.setReferencedTable((SimpleTableSegment) visit(ctx.referenceDefinition()));
495             return result;
496         }
497         if (null != ctx.UNIQUE()) {
498             result.getIndexColumns().addAll(((CollectionValue) visit(ctx.keyListWithExpression())).getValue());
499             if (null != ctx.indexName()) {
500                 result.setIndexName((IndexSegment) visit(ctx.indexName()));
501             }
502             return result;
503         }
504         if (null != ctx.checkConstraint()) {
505             return result;
506         }
507         result.getIndexColumns().addAll(((CollectionValue) visit(ctx.keyListWithExpression())).getValue());
508         if (null != ctx.indexName()) {
509             result.setIndexName((IndexSegment) visit(ctx.indexName()));
510         }
511         return result;
512     }

677     private ValidStatementSegment createValidStatementSegment(final ValidStatementContext ctx) {
678         ValidStatementSegment result = new ValidStatementSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex());
679         SQLStatement sqlStatement = null;
680         if (null != ctx.createTable()) {
681             sqlStatement = (MySQLCreateTableStatement) visit(ctx.createTable());
682         } else if (null != ctx.alterTable()) {
683             sqlStatement = (MySQLAlterTableStatement) visit(ctx.alterTable());
684         } else if (null != ctx.dropTable()) {
685             sqlStatement = (MySQLDropTableStatement) visit(ctx.dropTable());
686         } else if (null != ctx.truncateTable()) {
687             sqlStatement = (MySQLTruncateStatement) visit(ctx.truncateTable());
688         } else if (null != ctx.insert()) {
689             sqlStatement = (MySQLInsertStatement) visit(ctx.insert());
690         } else if (null != ctx.replace()) {
691             sqlStatement = (MySQLInsertStatement) visit(ctx.replace());
692         } else if (null != ctx.update()) {
693             sqlStatement = (MySQLUpdateStatement) visit(ctx.update());
694         } else if (null != ctx.delete()) {
695             sqlStatement = (MySQLDeleteStatement) visit(ctx.delete());
696         } else if (null != ctx.select()) {
697             sqlStatement = (MySQLSelectStatement) visit(ctx.select());
698         }
699         result.setSqlStatement(sqlStatement);
700         return result;
701     }
