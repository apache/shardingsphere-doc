50  /**
51   * Single table inventory data consistency checker.
52   */
53  @Slf4j
54  @RequiredArgsConstructor
55  public final class SingleTableInventoryDataConsistencyChecker {
56      
57      private final String jobId;
58      
59      private final PipelineDataSourceWrapper sourceDataSource;
60      
61      private final PipelineDataSourceWrapper targetDataSource;
62      
63      private final SchemaTableName sourceTable;
64      
65      private final SchemaTableName targetTable;
66      
67      private final List<String> columnNames;
68      
69      private final PipelineColumnMetaData uniqueKey;
70      
71      private final JobRateLimitAlgorithm readRateLimitAlgorithm;
72      
73      private final ConsistencyCheckJobItemProgressContext progressContext;
74      
75      /**
76       * Data consistency check.
77       *
78       * @param calculateAlgorithm calculate algorithm
79       * @return data consistency check result
80       */
81      public DataConsistencyCheckResult check(final DataConsistencyCalculateAlgorithm calculateAlgorithm) {
82          ThreadFactory threadFactory = ExecutorThreadFactoryBuilder.build("job-" + getJobIdDigest(jobId) + "-check-%d");
83          ThreadPoolExecutor executor = new ThreadPoolExecutor(2, 2, 60, TimeUnit.SECONDS, new ArrayBlockingQueue<>(2), threadFactory);
84          try {
85              return check(calculateAlgorithm, executor);
86          } finally {
87              executor.shutdown();
88              executor.shutdownNow();
89          }
90      }
91      
92      private DataConsistencyCheckResult check(final DataConsistencyCalculateAlgorithm calculateAlgorithm, final ThreadPoolExecutor executor) {
93          String schemaName = sourceTable.getSchemaName().getOriginal();
94          String sourceTableName = sourceTable.getTableName().getOriginal();
95          Map<String, Object> tableCheckPositions = progressContext.getTableCheckPositions();
96          DataConsistencyCalculateParameter sourceParam = new DataConsistencyCalculateParameter(sourceDataSource,
97                  schemaName, sourceTableName, columnNames, sourceDataSource.getDatabaseType(), uniqueKey, tableCheckPositions.get(sourceTableName));
98          String targetTableName = targetTable.getTableName().getOriginal();
99          DataConsistencyCalculateParameter targetParam = new DataConsistencyCalculateParameter(targetDataSource,
100                 targetTable.getSchemaName().getOriginal(), targetTableName, columnNames, targetDataSource.getDatabaseType(), uniqueKey, tableCheckPositions.get(targetTableName));
101         Iterator<DataConsistencyCalculatedResult> sourceCalculatedResults = waitFuture(executor.submit(() -> calculateAlgorithm.calculate(sourceParam))).iterator();
102         Iterator<DataConsistencyCalculatedResult> targetCalculatedResults = waitFuture(executor.submit(() -> calculateAlgorithm.calculate(targetParam))).iterator();
103         try {
104             return check(sourceCalculatedResults, targetCalculatedResults, executor);
105         } finally {
106             QuietlyCloser.close(sourceParam.getCalculationContext());
107             QuietlyCloser.close(targetParam.getCalculationContext());
108         }
109     }
110     
111     private DataConsistencyCheckResult check(final Iterator<DataConsistencyCalculatedResult> sourceCalculatedResults,
112                                              final Iterator<DataConsistencyCalculatedResult> targetCalculatedResults, final ThreadPoolExecutor executor) {
113         long sourceRecordsCount = 0;
114         long targetRecordsCount = 0;
115         boolean contentMatched = true;
116         while (sourceCalculatedResults.hasNext() && targetCalculatedResults.hasNext()) {
117             if (null != readRateLimitAlgorithm) {
118                 readRateLimitAlgorithm.intercept(JobOperationType.SELECT, 1);
119             }
120             DataConsistencyCalculatedResult sourceCalculatedResult = waitFuture(executor.submit(sourceCalculatedResults::next));
121             DataConsistencyCalculatedResult targetCalculatedResult = waitFuture(executor.submit(targetCalculatedResults::next));
122             sourceRecordsCount += sourceCalculatedResult.getRecordsCount();
123             targetRecordsCount += targetCalculatedResult.getRecordsCount();
124             contentMatched = Objects.equals(sourceCalculatedResult, targetCalculatedResult);
125             if (!contentMatched) {
126                 log.info("content matched false, jobId={}, sourceTable={}, targetTable={}, uniqueKey={}", jobId, sourceTable, targetTable, uniqueKey);
127                 break;
128             }
129             if (sourceCalculatedResult.getMaxUniqueKeyValue().isPresent()) {
130                 progressContext.getTableCheckPositions().put(sourceTable.getTableName().getOriginal(), sourceCalculatedResult.getMaxUniqueKeyValue().get());
131             }
132             if (targetCalculatedResult.getMaxUniqueKeyValue().isPresent()) {
133                 progressContext.getTableCheckPositions().put(targetTable.getTableName().getOriginal(), targetCalculatedResult.getMaxUniqueKeyValue().get());
134             }
135             progressContext.onProgressUpdated(new PipelineJobProgressUpdatedParameter(sourceCalculatedResult.getRecordsCount()));
136         }
137         if (sourceCalculatedResults.hasNext()) {
138             // TODO Refactor DataConsistencyCalculatedResult to represent inaccurate number
139             return new DataConsistencyCheckResult(new DataConsistencyCountCheckResult(sourceRecordsCount + 1, targetRecordsCount), new DataConsistencyContentCheckResult(false));
140         }
141         if (targetCalculatedResults.hasNext()) {
142             return new DataConsistencyCheckResult(new DataConsistencyCountCheckResult(sourceRecordsCount, targetRecordsCount + 1), new DataConsistencyContentCheckResult(false));
143         }
144         return new DataConsistencyCheckResult(new DataConsistencyCountCheckResult(sourceRecordsCount, targetRecordsCount), new DataConsistencyContentCheckResult(contentMatched));
145     }
146     
147     // TODO use digest (crc32, murmurhash)
148     private String getJobIdDigest(final String jobId) {
149         return jobId.length() <= 6 ? jobId : jobId.substring(0, 6);
150     }
151     
152     private <T> T waitFuture(final Future<T> future) {
153         try {
154             return future.get();
155         } catch (final InterruptedException ex) {
156             Thread.currentThread().interrupt();
157             throw new SQLWrapperException(new SQLException(ex));
158         } catch (final ExecutionException ex) {
159             if (ex.getCause() instanceof PipelineSQLException) {
160                 throw (PipelineSQLException) ex.getCause();
161             }
162             throw new SQLWrapperException(new SQLException(ex));
163         }
164     }
