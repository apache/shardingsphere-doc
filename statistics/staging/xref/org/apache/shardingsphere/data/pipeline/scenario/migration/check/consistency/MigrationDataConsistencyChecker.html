62  /**
63   * Data consistency checker for migration job.
64   */
65  @Slf4j
66  public final class MigrationDataConsistencyChecker implements PipelineDataConsistencyChecker {
67      
68      private final MigrationJobConfiguration jobConfig;
69      
70      private final JobRateLimitAlgorithm readRateLimitAlgorithm;
71      
72      private final ConsistencyCheckJobItemProgressContext progressContext;
73      
74      private final AtomicReference<TableInventoryChecker> currentTableInventoryChecker = new AtomicReference<>();
75      
76      private final AtomicBoolean canceling = new AtomicBoolean(false);
77      
78      public MigrationDataConsistencyChecker(final MigrationJobConfiguration jobConfig, final TransmissionProcessContext processContext,
79                                             final ConsistencyCheckJobItemProgressContext progressContext) {
80          this.jobConfig = jobConfig;
81          readRateLimitAlgorithm = null == processContext ? null : processContext.getReadRateLimitAlgorithm();
82          this.progressContext = progressContext;
83      }
84      
85      @Override
86      public Map<String, TableDataConsistencyCheckResult> check(final String algorithmType, final Properties algorithmProps) {
87          List<String> sourceTableNames = new LinkedList<>();
88          jobConfig.getJobShardingDataNodes().forEach(each -> each.getEntries().forEach(entry -> entry.getDataNodes()
89                  .forEach(dataNode -> sourceTableNames.add(DataNodeUtils.formatWithSchema(dataNode)))));
90          progressContext.setRecordsCount(getRecordsCount());
91          progressContext.getTableNames().addAll(sourceTableNames);
92          progressContext.onProgressUpdated(new PipelineJobProgressUpdatedParameter(0));
93          Map<CaseInsensitiveQualifiedTable, TableDataConsistencyCheckResult> result = new LinkedHashMap<>();
94          try (
95                  PipelineDataSourceManager dataSourceManager = new PipelineDataSourceManager();
96                  TableDataConsistencyChecker tableChecker = TableDataConsistencyCheckerFactory.newInstance(algorithmType, algorithmProps)) {
97              for (JobDataNodeLine each : jobConfig.getJobShardingDataNodes()) {
98                  if (checkTableInventoryDataUnmatchedAndBreak(each, tableChecker, result, dataSourceManager)) {
99                      return result.entrySet().stream().collect(Collectors.toMap(entry -> entry.getKey().toString(), Entry::getValue));
100                 }
101             }
102         }
103         return result.entrySet().stream().collect(Collectors.toMap(entry -> entry.getKey().toString(), Entry::getValue));
104     }
105     
106     private long getRecordsCount() {
107         Map<Integer, TransmissionJobItemProgress> jobProgress = new TransmissionJobManager(new MigrationJobType()).getJobProgress(jobConfig);
108         return jobProgress.values().stream().filter(Objects::nonNull).mapToLong(TransmissionJobItemProgress::getInventoryRecordsCount).sum();
109     }
110     
111     private boolean checkTableInventoryDataUnmatchedAndBreak(final JobDataNodeLine jobDataNodeLine, final TableDataConsistencyChecker tableChecker,
112                                                              final Map<CaseInsensitiveQualifiedTable, TableDataConsistencyCheckResult> checkResultMap,
113                                                              final PipelineDataSourceManager dataSourceManager) {
114         for (JobDataNodeEntry entry : jobDataNodeLine.getEntries()) {
115             for (DataNode each : entry.getDataNodes()) {
116                 TableDataConsistencyCheckResult checkResult = checkSingleTableInventoryData(entry.getLogicTableName(), each, tableChecker, dataSourceManager);
117                 checkResultMap.put(new CaseInsensitiveQualifiedTable(each.getSchemaName(), each.getTableName()), checkResult);
118                 if (!checkResult.isMatched() && tableChecker.isBreakOnInventoryCheckNotMatched()) {
119                     log.info("Unmatched on table '{}', ignore left tables", DataNodeUtils.formatWithSchema(each));
120                     return true;
121                 }
122             }
123         }
124         return false;
125     }
126     
127     private TableDataConsistencyCheckResult checkSingleTableInventoryData(final String targetTableName, final DataNode dataNode,
128                                                                           final TableDataConsistencyChecker tableChecker, final PipelineDataSourceManager dataSourceManager) {
129         CaseInsensitiveQualifiedTable sourceTable = new CaseInsensitiveQualifiedTable(dataNode.getSchemaName(), dataNode.getTableName());
130         CaseInsensitiveQualifiedTable targetTable = new CaseInsensitiveQualifiedTable(dataNode.getSchemaName(), targetTableName);
131         PipelineDataSourceWrapper sourceDataSource = dataSourceManager.getDataSource(jobConfig.getSources().get(dataNode.getDataSourceName()));
132         PipelineDataSourceWrapper targetDataSource = dataSourceManager.getDataSource(jobConfig.getTarget());
133         PipelineTableMetaDataLoader metaDataLoader = new StandardPipelineTableMetaDataLoader(sourceDataSource);
134         PipelineTableMetaData tableMetaData = metaDataLoader.getTableMetaData(dataNode.getSchemaName(), dataNode.getTableName());
135         ShardingSpherePreconditions.checkNotNull(tableMetaData, () -> new PipelineTableDataConsistencyCheckLoadingFailedException(dataNode.getSchemaName(), dataNode.getTableName()));
136         List<String> columnNames = tableMetaData.getColumnNames();
137         List<PipelineColumnMetaData> uniqueKeys = PipelineTableMetaDataUtils.getUniqueKeyColumns(
138                 sourceTable.getSchemaName().toString(), sourceTable.getTableName().toString(), metaDataLoader);
139         TableInventoryCheckParameter param = new TableInventoryCheckParameter(
140                 jobConfig.getJobId(), sourceDataSource, targetDataSource, sourceTable, targetTable, columnNames, uniqueKeys, readRateLimitAlgorithm, progressContext);
141         TableInventoryChecker tableInventoryChecker = tableChecker.buildTableInventoryChecker(param);
142         currentTableInventoryChecker.set(tableInventoryChecker);
143         TableDataConsistencyCheckResult result = tableInventoryChecker.checkSingleTableInventoryData();
144         currentTableInventoryChecker.set(null);
145         return result;
146     }
147     
148     @Override
149     public void cancel() {
150         canceling.set(true);
151         Optional.ofNullable(currentTableInventoryChecker.get()).ifPresent(TableInventoryChecker::cancel);
152     }
153     
154     @Override
155     public boolean isCanceling() {
156         return canceling.get();
157     }
158 }
