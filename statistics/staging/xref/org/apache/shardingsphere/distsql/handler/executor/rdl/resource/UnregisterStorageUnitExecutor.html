37  import java.sql.SQLException;

40  import java.util.LinkedList;
41  import java.util.Map;
42  import java.util.stream.Collectors;

44  /**
45   * Unregister storage unit executor.
46   */
47  @Setter
48  @Slf4j
49  public final class UnregisterStorageUnitExecutor implements DistSQLUpdateExecutor<UnregisterStorageUnitStatement>, DistSQLExecutorDatabaseAware {
50      
51      private ShardingSphereDatabase database;
52      
53      @Override
54      public void executeUpdate(final UnregisterStorageUnitStatement sqlStatement, final ContextManager contextManager) {
55          if (!sqlStatement.isIfExists()) {
56              checkExisted(sqlStatement.getStorageUnitNames());
57          }
58          checkInUsed(sqlStatement);
59          try {
60              contextManager.getInstanceContext().getModeContextManager().unregisterStorageUnits(database.getName(), sqlStatement.getStorageUnitNames());
61          } catch (final SQLException | ShardingSphereServerException ex) {
62              throw new StorageUnitsOperateException("unregister", sqlStatement.getStorageUnitNames(), ex);
63          }
64      }
65      
66      private void checkExisted(final Collection<String> storageUnitNames) {
67          Map<String, StorageUnit> storageUnits = database.getResourceMetaData().getStorageUnits();
68          Collection<String> notExistedStorageUnits = storageUnitNames.stream().filter(each -> !storageUnits.containsKey(each)).collect(Collectors.toList());
69          ShardingSpherePreconditions.checkMustEmpty(notExistedStorageUnits, () -> new MissingRequiredStorageUnitsException(database.getName(), notExistedStorageUnits));
70      }
71      
72      private void checkInUsed(final UnregisterStorageUnitStatement sqlStatement) {
73          Map<String, Collection<Class<? extends ShardingSphereRule>>> inUsedStorageUnits = database.getRuleMetaData().getInUsedStorageUnitNameAndRulesMap();
74          Collection<String> inUsedStorageUnitNames = inUsedStorageUnits.keySet();
75          inUsedStorageUnitNames.retainAll(sqlStatement.getStorageUnitNames());
76          if (inUsedStorageUnitNames.isEmpty()) {
77              return;
78          }
79          Collection<Class<ShardingSphereRule>> ignoreUsageCheckRules = getIgnoreUsageCheckRules(sqlStatement);
80          String firstResource = inUsedStorageUnitNames.iterator().next();
81          ShardingSpherePreconditions.checkNotEmpty(ignoreUsageCheckRules, () -> new InUsedStorageUnitException(firstResource, inUsedStorageUnits.get(firstResource)));
82          checkInUsedIgnoreTables(new HashSet<>(inUsedStorageUnitNames), inUsedStorageUnits, ignoreUsageCheckRules);
83      }
84      
85      @SuppressWarnings({"unchecked", "rawtypes"})
86      private Collection<Class<ShardingSphereRule>> getIgnoreUsageCheckRules(final UnregisterStorageUnitStatement sqlStatement) {
87          Collection<Class<ShardingSphereRule>> result = new LinkedList<>();
88          for (StorageUnitDefinitionProcessor each : ShardingSphereServiceLoader.getServiceInstances(StorageUnitDefinitionProcessor.class)) {
89              if (each.ignoreUsageCheckOnUnregister(sqlStatement)) {
90                  result.add(each.getRuleClass());
91              }
92          }
93          return result;
94      }
95      
96      private void checkInUsedIgnoreTables(final Collection<String> inUsedResourceNames, final Map<String, Collection<Class<? extends ShardingSphereRule>>> inUsedStorageUnits,
97                                           final Collection<Class<ShardingSphereRule>> ignoreShardingSphereRules) {
98          for (String each : inUsedResourceNames) {
99              Collection<Class<? extends ShardingSphereRule>> inUsedRules = inUsedStorageUnits.get(each);
100             ignoreShardingSphereRules.forEach(inUsedRules::remove);
101             ShardingSpherePreconditions.checkMustEmpty(inUsedRules, () -> new InUsedStorageUnitException(each, inUsedRules));
102         }
103     }
104     
105     @Override
106     public Class<UnregisterStorageUnitStatement> getType() {
107         return UnregisterStorageUnitStatement.class;
108     }
109 }
