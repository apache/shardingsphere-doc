36  
37  /**
38   * TCL statement visitor for Oracle.
39   */
40  public final class OracleTCLStatementVisitor extends OracleStatementVisitor implements TCLStatementVisitor {
41      
42      @Override
43      public ASTNode visitSetTransaction(final SetTransactionContext ctx) {
44          return new OracleSetTransactionStatement();
45      }
46      
47      @Override
48      public ASTNode visitCommit(final CommitContext ctx) {
49          return new OracleCommitStatement();
50      }
51      
52      @Override
53      public ASTNode visitRollback(final RollbackContext ctx) {
54          OracleRollbackStatement result = new OracleRollbackStatement();
55          if (null != ctx.savepointClause().savepointName()) {
56              result.setSavepointName(((IdentifierValue) visit(ctx.savepointClause().savepointName())).getValue());
57          }
58          return result;
59      }
60      
61      @Override
62      public ASTNode visitSavepoint(final SavepointContext ctx) {
63          OracleSavepointStatement result = new OracleSavepointStatement();
64          result.setSavepointName(((IdentifierValue) visit(ctx.savepointName())).getValue());
65          return result;
66      }
67      
68      @Override
69      public ASTNode visitSetConstraints(final SetConstraintsContext ctx) {
70          return new OracleSetConstraintsStatement();
71      }
72      
73      @Override
74      public ASTNode visitLock(final LockContext ctx) {
75          return new OracleLockStatement();
76      }
77  }
