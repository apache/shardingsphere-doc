114     public DatabaseConnector(final String driverType, final ShardingSphereDatabase database, final QueryContext queryContext, final ProxyDatabaseConnectionManager databaseConnectionManager) {
115         SQLStatementContext sqlStatementContext = queryContext.getSqlStatementContext();
116         failedIfBackendNotReady(databaseConnectionManager.getConnectionSession(), sqlStatementContext);
117         this.driverType = driverType;
118         this.database = database;
119         this.queryContext = queryContext;
120         this.selectContainsEnhancedTable = sqlStatementContext instanceof SelectStatementContext && ((SelectStatementContext) sqlStatementContext).isContainsEnhancedTable();
121         this.databaseConnectionManager = databaseConnectionManager;
122         if (sqlStatementContext instanceof CursorAvailable) {
123             prepareCursorStatementContext((CursorAvailable) sqlStatementContext, databaseConnectionManager.getConnectionSession());
124         }
125         proxySQLExecutor = new ProxySQLExecutor(driverType, databaseConnectionManager, this, queryContext);
126     }
127     
128     private void failedIfBackendNotReady(final ConnectionSession connectionSession, final SQLStatementContext sqlStatementContext) {
129         ShardingSphereDatabase database = ProxyContext.getInstance().getDatabase(connectionSession.getDatabaseName());
130         boolean isSystemSchema = SystemSchemaUtils.containsSystemSchema(sqlStatementContext.getDatabaseType(), sqlStatementContext.getTablesContext().getSchemaNames(), database);
131         ShardingSpherePreconditions.checkState(isSystemSchema || database.containsDataSource(), () -> new StorageUnitNotExistedException(connectionSession.getDatabaseName()));
132         if (!isSystemSchema && !database.isComplete()) {
133             throw new RuleNotExistedException(connectionSession.getDatabaseName());
134         }
135     }

155     /**
156      * Execute to database.
157      *
158      * @return backend response
159      * @throws SQLException SQL exception
160      */
161     @Override
162     public ResponseHeader execute() throws SQLException {
163         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
164         if (proxySQLExecutor.getSqlFederationEngine().decide(queryContext.getSqlStatementContext(), queryContext.getParameters(), database, metaDataContexts.getMetaData().getGlobalRuleMetaData())) {
165             ResultSet resultSet = doExecuteFederation(queryContext, metaDataContexts);
166             return processExecuteFederation(resultSet, metaDataContexts);
167         }
168         Collection<ExecutionContext> executionContexts = generateExecutionContexts();
169         return isNeedImplicitCommitTransaction(executionContexts) ? doExecuteWithImplicitCommitTransaction(executionContexts) : doExecute(executionContexts);
170     }
171     
172     private Collection<ExecutionContext> generateExecutionContexts() {
173         Collection<ExecutionContext> result = new LinkedList<>();
174         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
175         ExecutionContext executionContext = new KernelProcessor().generateExecutionContext(queryContext, database, metaDataContexts.getMetaData().getGlobalRuleMetaData(),
176                 metaDataContexts.getMetaData().getProps(), databaseConnectionManager.getConnectionSession().getConnectionContext());
177         result.add(executionContext);
178         // TODO support logical SQL optimize to generate multiple logical SQL
179         return result;
180     }
181     
182     private boolean isNeedImplicitCommitTransaction(final Collection<ExecutionContext> executionContexts) {
183         TransactionStatus transactionStatus = databaseConnectionManager.getConnectionSession().getTransactionStatus();
184         if (!TransactionType.isDistributedTransaction(transactionStatus.getTransactionType()) || transactionStatus.isInTransaction()) {
185             return false;
186         }
187         if (1 == executionContexts.size()) {
188             SQLStatement sqlStatement = executionContexts.iterator().next().getSqlStatementContext().getSqlStatement();
189             return isWriteDMLStatement(sqlStatement) && executionContexts.iterator().next().getExecutionUnits().size() > 1;
190         }
191         return executionContexts.stream().anyMatch(each -> isWriteDMLStatement(each.getSqlStatementContext().getSqlStatement()));
192     }
193     
194     private boolean isWriteDMLStatement(final SQLStatement sqlStatement) {
195         return sqlStatement instanceof DMLStatement && !(sqlStatement instanceof SelectStatement);
196     }
197     
198     private ResponseHeader doExecuteWithImplicitCommitTransaction(final Collection<ExecutionContext> executionContexts) throws SQLException {
199         ResponseHeader result;
200         BackendTransactionManager transactionManager = new BackendTransactionManager(databaseConnectionManager);
201         try {
202             transactionManager.begin();
203             result = doExecute(executionContexts);
204             transactionManager.commit();
205             // CHECKSTYLE:OFF
206         } catch (final RuntimeException ex) {
207             // CHECKSTYLE:ON
208             transactionManager.rollback();
209             String databaseName = databaseConnectionManager.getConnectionSession().getDatabaseName();
210             throw SQLExceptionTransformEngine.toSQLException(ex, ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getDatabase(databaseName).getProtocolType());
211         }
212         return result;
213     }
214     
215     private ResponseHeader doExecute(final Collection<ExecutionContext> executionContexts) throws SQLException {
216         ResponseHeader result = null;
217         for (ExecutionContext each : executionContexts) {
218             ResponseHeader responseHeader = doExecute(each);
219             if (null == result) {
220                 result = responseHeader;
221             }
222         }
223         return result;
224     }
225     
226     @SuppressWarnings({"unchecked", "rawtypes"})
227     private ResponseHeader doExecute(final ExecutionContext executionContext) throws SQLException {
228         if (executionContext.getExecutionUnits().isEmpty()) {
229             return new UpdateResponseHeader(executionContext.getSqlStatementContext().getSqlStatement());
230         }
231         proxySQLExecutor.checkExecutePrerequisites(executionContext);
232         List result = proxySQLExecutor.execute(executionContext);
233         refreshMetaData(executionContext);
234         Object executeResultSample = result.iterator().next();
235         return executeResultSample instanceof QueryResult ? processExecuteQuery(executionContext, result, (QueryResult) executeResultSample) : processExecuteUpdate(executionContext, result);
236     }
237     
238     private ResultSet doExecuteFederation(final QueryContext queryContext, final MetaDataContexts metaDataContexts) {
239         boolean isReturnGeneratedKeys = queryContext.getSqlStatementContext().getSqlStatement() instanceof MySQLInsertStatement;
240         ShardingSphereDatabase database = metaDataContexts.getMetaData().getDatabase(databaseConnectionManager.getConnectionSession().getDatabaseName());
241         DatabaseType protocolType = database.getProtocolType();
242         ProxyJDBCExecutorCallback callback = ProxyJDBCExecutorCallbackFactory.newInstance(driverType, protocolType, database.getResourceMetaData(),
243                 queryContext.getSqlStatementContext().getSqlStatement(), this, isReturnGeneratedKeys, SQLExecutorExceptionHandler.isExceptionThrown(), true);
244         DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine = createDriverExecutionPrepareEngine(isReturnGeneratedKeys, metaDataContexts);
245         SQLFederationExecutorContext context = new SQLFederationExecutorContext(false, queryContext, metaDataContexts.getMetaData());
246         return proxySQLExecutor.getSqlFederationEngine().executeQuery(prepareEngine, callback, context);
247     }
248     
249     private DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> createDriverExecutionPrepareEngine(final boolean isReturnGeneratedKeys, final MetaDataContexts metaData) {
250         int maxConnectionsSizePerQuery = metaData.getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
251         JDBCBackendStatement statementManager = (JDBCBackendStatement) databaseConnectionManager.getConnectionSession().getStatementManager();
252         return new DriverExecutionPrepareEngine<>(driverType, maxConnectionsSizePerQuery, databaseConnectionManager, statementManager,
253                 new StatementOption(isReturnGeneratedKeys), metaData.getMetaData().getDatabase(databaseConnectionManager.getConnectionSession().getDatabaseName()).getRuleMetaData().getRules(),
254                 metaData.getMetaData().getDatabase(databaseConnectionManager.getConnectionSession().getDatabaseName()).getResourceMetaData().getStorageTypes());
255     }
256     
257     private ResponseHeader processExecuteFederation(final ResultSet resultSet, final MetaDataContexts metaDataContexts) throws SQLException {
258         int columnCount = resultSet.getMetaData().getColumnCount();
259         queryHeaders = new ArrayList<>(columnCount);
260         ShardingSphereDatabase database = metaDataContexts.getMetaData().getDatabase(databaseConnectionManager.getConnectionSession().getDatabaseName());
261         QueryHeaderBuilderEngine queryHeaderBuilderEngine = new QueryHeaderBuilderEngine(null == database ? null : database.getProtocolType());
262         for (int columnIndex = 1; columnIndex <= columnCount; columnIndex++) {
263             queryHeaders.add(queryHeaderBuilderEngine.build(new JDBCQueryResultMetaData(resultSet.getMetaData()), database, columnIndex));
264         }
265         mergedResult = new IteratorStreamMergedResult(Collections.singletonList(new JDBCStreamQueryResult(resultSet)));
266         return new QueryResponseHeader(queryHeaders);
267     }
268     
269     private void prepareCursorStatementContext(final CursorAvailable statementContext, final ConnectionSession connectionSession) {
270         if (statementContext.getCursorName().isPresent()) {
271             String cursorName = statementContext.getCursorName().get().getIdentifier().getValue().toLowerCase();
272             prepareCursorStatementContext(statementContext, connectionSession, cursorName);
273         }
274         if (statementContext instanceof CloseStatementContext && ((CloseStatementContext) statementContext).getSqlStatement().isCloseAll()) {
275             connectionSession.getConnectionContext().clearCursorConnectionContext();
276         }
277     }
278     
279     private void prepareCursorStatementContext(final CursorAvailable statementContext, final ConnectionSession connectionSession, final String cursorName) {
280         if (statementContext instanceof CursorStatementContext) {
281             connectionSession.getConnectionContext().getCursorContext().getCursorDefinitions().put(cursorName, (CursorStatementContext) statementContext);
282         }
283         if (statementContext instanceof CursorDefinitionAware) {
284             CursorStatementContext cursorStatementContext = (CursorStatementContext) connectionSession.getConnectionContext().getCursorContext().getCursorDefinitions().get(cursorName);
285             Preconditions.checkArgument(null != cursorStatementContext, "Cursor %s does not exist.", cursorName);
286             ((CursorDefinitionAware) statementContext).setUpCursorDefinition(cursorStatementContext);
287         }
288         if (statementContext instanceof CloseStatementContext) {
289             connectionSession.getConnectionContext().getCursorContext().removeCursor(cursorName);
290         }
291     }

293     private void refreshMetaData(final ExecutionContext executionContext) throws SQLException {
294         ContextManager contextManager = ProxyContext.getInstance().getContextManager();
295         new MetaDataRefreshEngine(contextManager.getInstanceContext().getModeContextManager(), database,
296                 contextManager.getMetaDataContexts().getMetaData().getProps()).refresh(executionContext.getSqlStatementContext(), executionContext.getRouteContext().getRouteUnits());
297     }
298     
299     private QueryResponseHeader processExecuteQuery(final ExecutionContext executionContext, final List<QueryResult> queryResults, final QueryResult queryResultSample) throws SQLException {
300         queryHeaders = createQueryHeaders(executionContext, queryResultSample);
301         mergedResult = mergeQuery(executionContext.getSqlStatementContext(), queryResults);
302         return new QueryResponseHeader(queryHeaders);
303     }
304     
305     private List<QueryHeader> createQueryHeaders(final ExecutionContext executionContext, final QueryResult queryResultSample) throws SQLException {
306         int columnCount = getColumnCount(executionContext, queryResultSample);
307         List<QueryHeader> result = new ArrayList<>(columnCount);
308         QueryHeaderBuilderEngine queryHeaderBuilderEngine = new QueryHeaderBuilderEngine(database.getProtocolType());
309         for (int columnIndex = 1; columnIndex <= columnCount; columnIndex++) {
310             result.add(createQueryHeader(queryHeaderBuilderEngine, executionContext, queryResultSample, database, columnIndex));
311         }
312         return result;
313     }
314     
315     private int getColumnCount(final ExecutionContext executionContext, final QueryResult queryResultSample) throws SQLException {
316         return selectContainsEnhancedTable && hasSelectExpandProjections(executionContext.getSqlStatementContext())
317                 ? ((SelectStatementContext) executionContext.getSqlStatementContext()).getProjectionsContext().getExpandProjections().size()
318                 : queryResultSample.getMetaData().getColumnCount();
319     }
320     
321     private boolean hasSelectExpandProjections(final SQLStatementContext sqlStatementContext) {
322         return sqlStatementContext instanceof SelectStatementContext && !((SelectStatementContext) sqlStatementContext).getProjectionsContext().getExpandProjections().isEmpty();
323     }
324     
325     private QueryHeader createQueryHeader(final QueryHeaderBuilderEngine queryHeaderBuilderEngine, final ExecutionContext executionContext,
326                                           final QueryResult queryResultSample, final ShardingSphereDatabase database, final int columnIndex) throws SQLException {
327         return selectContainsEnhancedTable && hasSelectExpandProjections(executionContext.getSqlStatementContext())
328                 ? queryHeaderBuilderEngine.build(((SelectStatementContext) executionContext.getSqlStatementContext()).getProjectionsContext(), queryResultSample.getMetaData(), database, columnIndex)
329                 : queryHeaderBuilderEngine.build(queryResultSample.getMetaData(), database, columnIndex);
330     }
331     
332     private MergedResult mergeQuery(final SQLStatementContext sqlStatementContext, final List<QueryResult> queryResults) throws SQLException {
333         MergeEngine mergeEngine = new MergeEngine(database, ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getProps(),
334                 databaseConnectionManager.getConnectionSession().getConnectionContext());
335         return mergeEngine.merge(queryResults, sqlStatementContext);
336     }
337     
338     private UpdateResponseHeader processExecuteUpdate(final ExecutionContext executionContext, final Collection<UpdateResult> updateResults) {
339         Optional<GeneratedKeyContext> generatedKeyContext = executionContext.getSqlStatementContext() instanceof InsertStatementContext
340                 ? ((InsertStatementContext) executionContext.getSqlStatementContext()).getGeneratedKeyContext()
341                 : Optional.empty();
342         Collection<Comparable<?>> autoIncrementGeneratedValues =
343                 generatedKeyContext.filter(GeneratedKeyContext::isSupportAutoIncrement).map(GeneratedKeyContext::getGeneratedValues).orElseGet(Collections::emptyList);
344         UpdateResponseHeader result = new UpdateResponseHeader(executionContext.getSqlStatementContext().getSqlStatement(), updateResults, autoIncrementGeneratedValues);
345         mergeUpdateCount(executionContext.getSqlStatementContext(), result);
346         return result;
347     }
348     
349     private void mergeUpdateCount(final SQLStatementContext sqlStatementContext, final UpdateResponseHeader response) {
350         if (isNeedAccumulate(sqlStatementContext)) {
351             response.mergeUpdateCount();
352         }
353     }
354     
355     private boolean isNeedAccumulate(final SQLStatementContext sqlStatementContext) {
356         Optional<DataNodeContainedRule> dataNodeContainedRule = database.getRuleMetaData().findSingleRule(DataNodeContainedRule.class);
357         return dataNodeContainedRule.isPresent() && dataNodeContainedRule.get().isNeedAccumulate(sqlStatementContext.getTablesContext().getTableNames());
358     }

371     /**
372      * Get query response row.
373      *
374      * @return query response row
375      * @throws SQLException SQL exception
376      */
377     @Override
378     public QueryResponseRow getRowData() throws SQLException {
379         List<QueryResponseCell> cells = new ArrayList<>(queryHeaders.size());
380         for (int columnIndex = 1; columnIndex <= queryHeaders.size(); columnIndex++) {
381             Object data = mergedResult.getValue(columnIndex, Object.class);
382             cells.add(new QueryResponseCell(queryHeaders.get(columnIndex - 1).getColumnType(), data, queryHeaders.get(columnIndex - 1).getColumnTypeName()));
383         }
384         return new QueryResponseRow(cells);
385     }

387     /**
388      * Close database connector.
389      *
390      * @throws SQLException SQL exception
391      */
392     @Override
393     public void close() throws SQLException {
394         Collection<SQLException> result = new LinkedList<>();
395         result.addAll(closeResultSets());
396         result.addAll(closeStatements());
397         closeSQLFederationEngine().ifPresent(result::add);
398         if (result.isEmpty()) {
399             return;
400         }
401         SQLException ex = new SQLException();
402         result.forEach(ex::setNextException);
403         throw ex;
404     }
405     
406     private Collection<SQLException> closeResultSets() {
407         Collection<SQLException> result = new LinkedList<>();
408         for (ResultSet each : cachedResultSets) {
409             try {
410                 each.close();
411             } catch (final SQLException ex) {
412                 result.add(ex);
413             }
414         }
415         cachedResultSets.clear();
416         return result;
417     }
418     
419     private Collection<SQLException> closeStatements() {
420         Collection<SQLException> result = new LinkedList<>();
421         for (Statement each : cachedStatements) {
422             try {
423                 each.cancel();
424                 each.close();
425             } catch (final SQLException ex) {
426                 result.add(ex);
427             }
428         }
429         cachedStatements.clear();
430         return result;
431     }
432     
433     private Optional<SQLException> closeSQLFederationEngine() {
434         if (null != proxySQLExecutor.getSqlFederationEngine()) {
435             try {
436                 proxySQLExecutor.getSqlFederationEngine().close();
437             } catch (final SQLException ex) {
438                 return Optional.of(ex);
439             }
440         }
441         return Optional.empty();
442     }
