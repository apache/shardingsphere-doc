41  class MySQLTableVisitorTest {
42      
43      @ParameterizedTest(name = "{0}")
44      @ArgumentsSource(TestCaseArgumentsProvider.class)
45      void assertSQLStats(final String caseId, final String inputSql, final int tableNum, final int columnNum) {
46          CodePointBuffer buffer = CodePointBuffer.withChars(CharBuffer.wrap(inputSql.toCharArray()));
47          MySQLLexer lexer = new MySQLLexer(CodePointCharStream.fromBuffer(buffer));
48          MySQLParser parser = new MySQLParser(new CommonTokenStream(lexer));
49          ParseTree tree = ((ParseASTNode) parser.parse()).getRootNode();
50          MySQLSQLStatVisitor visitor = new MySQLSQLStatVisitor();
51          SQLStats sqlStats = visitor.visit(tree);
52          assertThat(sqlStats.getTables().keySet().size(), is(tableNum));
53          assertThat(sqlStats.getColumns().keySet().size(), is(columnNum));
54      }
55      
56      private static class TestCaseArgumentsProvider implements ArgumentsProvider {
57          
58          @Override
59          public Stream<? extends Arguments> provideArguments(final ExtensionContext extensionContext) {
60              return Stream.of(Arguments.of("select_with_union", "select a+1 as b, name n from table1 join table2 where id=1 and name='lu';", 2, 3),
61                      Arguments.of("select_item_nums", "select id, name, age, sex, ss, yy from table1 where id=1", 1, 6),
62                      Arguments.of("select_with_subquery", "select id, name, age, count(*) as n, (select id, name, age, sex from table2 where id=2) as sid, yyyy from table1 where id=1", 2, 5),
63                      Arguments.of("select_where_num", "select id, name, age, sex, ss, yy from table1 where id=1 and name=1 and a=1 and b=2 and c=4 and d=3", 1, 10),
64                      Arguments.of("alter_table", "ALTER TABLE t_order ADD column4 DATE, ADD column5 DATETIME, engine ss max_rows 10,min_rows 2, ADD column6 TIMESTAMP, ADD column7 TIME;", 1, 4),
65                      Arguments.of("create_table", "CREATE TABLE IF NOT EXISTS `runoob_tbl`(\n"
66                              + "`runoob_id` INT UNSIGNED AUTO_INCREMENT,\n"
67                              + "`runoob_title` VARCHAR(100) NOT NULL,\n"
68                              + "`runoob_author` VARCHAR(40) NOT NULL,\n"
69                              + "`runoob_test` NATIONAL CHAR(40),\n"
70                              + "`submission_date` DATE,\n"
71                              + "PRIMARY KEY ( `runoob_id` )\n"
72                              + ")ENGINE=InnoDB DEFAULT CHARSET=utf8;",
73                              1, 5));
74          }
75      }
