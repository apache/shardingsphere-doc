18  package org.apache.shardingsphere.sqlfederation.executor.utils;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
23  import org.apache.shardingsphere.infra.executor.sql.execute.result.query.impl.driver.jdbc.type.util.ResultSetUtils;
24  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereColumn;
25  import org.apache.shardingsphere.sqlfederation.optimizer.metadata.util.SQLFederationDataTypeUtils;
26  
27  import java.sql.SQLFeatureNotSupportedException;
28  import java.util.HashMap;
29  import java.util.List;
30  import java.util.Map;
31  import java.util.Optional;
32  
33  /**
34   * Enumerator utilities.
35   */
36  @NoArgsConstructor(access = AccessLevel.PRIVATE)
37  public final class EnumeratorUtils {
38      
39      /**
40       * Create column types.
41       *
42       * @param columns columns
43       * @param databaseType database type
44       * @return column types
45       */
46      public static Map<Integer, Class<?>> createColumnTypes(final List<ShardingSphereColumn> columns, final DatabaseType databaseType) {
47          Map<Integer, Class<?>> result = new HashMap<>(columns.size(), 1F);
48          for (int index = 0; index < columns.size(); index++) {
49              int finalIndex = index;
50              getSQLTypeClass(columns, databaseType, index).ifPresent(optional -> result.put(finalIndex, optional));
51          }
52          return result;
53      }
54      
55      private static Optional<Class<?>> getSQLTypeClass(final List<ShardingSphereColumn> columns, final DatabaseType databaseType, final int index) {
56          try {
57              return Optional.of(SQLFederationDataTypeUtils.getSqlTypeClass(databaseType, columns.get(index)));
58          } catch (final IllegalArgumentException ex) {
59              return Optional.empty();
60          }
61      }
62      
63      /**
64       * Convert to target type.
65       *
66       * @param columnTypes column types
67       * @param rows rows
68       * @return target type
69       */
70      public static Object[] convertToTargetType(final Map<Integer, Class<?>> columnTypes, final Object[] rows) {
71          Object[] result = new Object[rows.length];
72          for (int index = 0; index < rows.length; index++) {
73              if (columnTypes.containsKey(index)) {
74                  result[index] = convertValue(rows, columnTypes, index);
75              }
76          }
77          return result;
78      }
79      
80      private static Object convertValue(final Object[] rows, final Map<Integer, Class<?>> columnTypes, final int index) {
81          try {
82              return ResultSetUtils.convertValue(rows[index], columnTypes.get(index));
83          } catch (final SQLFeatureNotSupportedException ex) {
84              return rows[index];
85          }
86      }
87  }
