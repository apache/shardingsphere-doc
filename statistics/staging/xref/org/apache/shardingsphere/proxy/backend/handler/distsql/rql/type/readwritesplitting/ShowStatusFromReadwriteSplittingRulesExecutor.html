52  /**
53   * Show status from readwrite-splitting rules executor.
54   */
55  // TODO move to readwritesplitting module
56  @Setter
57  public final class ShowStatusFromReadwriteSplittingRulesExecutor
58          implements
59              DatabaseAwareRQLExecutor<ShowStatusFromReadwriteSplittingRulesStatement>,
60              MetaDataAwareRQLExecutor<ShowStatusFromReadwriteSplittingRulesStatement> {
61      
62      private ShardingSphereDatabase database;
63      
64      private MetaDataContexts metaDataContexts;
65      
66      @Override
67      public Collection<String> getColumnNames() {
68          return Arrays.asList("storage_unit", "status");
69      }
70      
71      @Override
72      public Collection<LocalDataQueryResultRow> getRows(final ShowStatusFromReadwriteSplittingRulesStatement sqlStatement) {
73          Collection<String> allReadResources = getAllReadResources(sqlStatement.getGroupName());
74          Map<String, StorageNodeDataSource> persistentReadResources = getPersistentReadResources(database.getName(), metaDataContexts.getPersistService());
75          return buildRows(allReadResources, persistentReadResources);
76      }
77      
78      private Collection<String> getAllReadResources(final String groupName) {
79          Collection<String> exportKeys = Arrays.asList(ExportableConstants.EXPORT_STATIC_READWRITE_SPLITTING_RULE, ExportableConstants.EXPORT_DYNAMIC_READWRITE_SPLITTING_RULE);
80          Map<String, Object> exportMap = database.getRuleMetaData().findRules(ExportableRule.class).stream()
81                  .filter(each -> new RuleExportEngine(each).containExportableKey(exportKeys)).findFirst().map(each -> new RuleExportEngine(each).export(exportKeys)).orElse(Collections.emptyMap());
82          Map<String, Map<String, String>> allReadwriteRuleMap = exportMap.values().stream().map(each -> ((Map<String, Map<String, String>>) each).entrySet())
83                  .flatMap(Collection::stream).collect(Collectors.toMap(Entry::getKey, Entry::getValue, (oldValue, currentValue) -> currentValue, LinkedHashMap::new));
84          if (!Strings.isNullOrEmpty(groupName)) {
85              allReadwriteRuleMap = allReadwriteRuleMap.entrySet().stream().filter(each -> groupName.equalsIgnoreCase(each.getKey()))
86                      .collect(Collectors.toMap(Entry::getKey, Entry::getValue, (oldValue, currentValue) -> currentValue, LinkedHashMap::new));
87          }
88          return allReadwriteRuleMap.values().stream().map(each -> each.get(ExportableItemConstants.REPLICA_DATA_SOURCE_NAMES)).filter(each -> null != each && !each.isEmpty())
89                  .map(this::deconstructString).flatMap(Collection::stream).collect(Collectors.toCollection(LinkedHashSet::new));
90      }
91      
92      private Map<String, StorageNodeDataSource> getPersistentReadResources(final String databaseName, final MetaDataBasedPersistService persistService) {
93          if (null == persistService || null == persistService.getRepository() || !(persistService.getRepository() instanceof ClusterPersistRepository)) {
94              return Collections.emptyMap();
95          }
96          Map<String, StorageNodeDataSource> storageNodes = new StorageNodeStatusService(persistService.getRepository()).loadStorageNodes();
97          Map<String, StorageNodeDataSource> result = new HashMap<>();
98          storageNodes.entrySet().stream().filter(entry -> StorageNodeRole.MEMBER == entry.getValue().getRole()).forEach(entry -> {
99              QualifiedDatabase qualifiedDatabase = new QualifiedDatabase(entry.getKey());
100             if (databaseName.equalsIgnoreCase(qualifiedDatabase.getDatabaseName())) {
101                 result.put(qualifiedDatabase.getDataSourceName(), entry.getValue());
102             }
103         });
104         return result;
105     }
106     
107     private Collection<LocalDataQueryResultRow> buildRows(final Collection<String> readResources, final Map<String, StorageNodeDataSource> persistentReadResources) {
108         Map<DataSourceState, Map<String, StorageNodeDataSource>> persistentReadResourceGroup = persistentReadResources.entrySet().stream()
109                 .collect(Collectors.groupingBy(each -> each.getValue().getStatus(), Collectors.toMap(Entry::getKey, Entry::getValue)));
110         Map<String, StorageNodeDataSource> disabledReadResources = persistentReadResourceGroup.getOrDefault(DataSourceState.DISABLED, Collections.emptyMap());
111         Map<String, StorageNodeDataSource> enabledReadResources = persistentReadResourceGroup.getOrDefault(DataSourceState.ENABLED, Collections.emptyMap());
112         readResources.removeIf(disabledReadResources::containsKey);
113         readResources.addAll(enabledReadResources.keySet());
114         readResources.addAll(disabledReadResources.keySet());
115         return readResources.stream().map(each -> buildRow(each, disabledReadResources.get(each))).collect(Collectors.toList());
116     }
117     
118     private List<String> deconstructString(final String str) {
119         return new LinkedList<>(Arrays.asList(str.split(",")));
120     }
121     
122     private LocalDataQueryResultRow buildRow(final String resource, final StorageNodeDataSource storageNodeDataSource) {
123         return null == storageNodeDataSource ? new LocalDataQueryResultRow(resource, DataSourceState.ENABLED.name()) : new LocalDataQueryResultRow(resource, storageNodeDataSource.getStatus().name());
124     }
125     
126     @Override
127     public Class<ShowStatusFromReadwriteSplittingRulesStatement> getType() {
128         return ShowStatusFromReadwriteSplittingRulesStatement.class;
129     }
130 }
