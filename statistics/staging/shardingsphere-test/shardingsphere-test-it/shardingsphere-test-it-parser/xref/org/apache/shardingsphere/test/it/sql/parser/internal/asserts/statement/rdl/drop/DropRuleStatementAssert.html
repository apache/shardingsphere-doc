62  /**
63   * Drop RDL statement assert.
64   */
65  @NoArgsConstructor(access = AccessLevel.PRIVATE)
66  public final class DropRuleStatementAssert {
67      
68      /**
69       * Assert drop RDL statement is correct with expected parser result.
70       *
71       * @param assertContext assert context
72       * @param actual actual drop RDL statement
73       * @param expected expected drop RDL statement test case
74       */
75      public static void assertIs(final SQLCaseAssertContext assertContext, final DropRuleStatement actual, final SQLParserTestCase expected) {
76          if (actual instanceof DropDefaultShardingStrategyStatement) {
77              DropDefaultShardingStrategyStatementAssert.assertIs(assertContext, (DropDefaultShardingStrategyStatement) actual, (DropDefaultShardingStrategyStatementTestCase) expected);
78          } else if (actual instanceof DropEncryptRuleStatement) {
79              DropEncryptRuleStatementAssert.assertIs(assertContext, (DropEncryptRuleStatement) actual, (DropEncryptRuleStatementTestCase) expected);
80          } else if (actual instanceof DropReadwriteSplittingRuleStatement) {
81              DropReadwriteSplittingRuleStatementAssert.assertIs(assertContext, (DropReadwriteSplittingRuleStatement) actual, (DropReadwriteSplittingRuleStatementTestCase) expected);
82          } else if (actual instanceof DropShardingTableReferenceRuleStatement) {
83              DropShardingTableReferenceRulesStatementAssert.assertIs(assertContext, (DropShardingTableReferenceRuleStatement) actual, (DropShardingTableReferenceRuleStatementTestCase) expected);
84          } else if (actual instanceof DropBroadcastTableRuleStatement) {
85              DropBroadcastTableRuleStatementAssert.assertIs(assertContext, (DropBroadcastTableRuleStatement) actual, (DropBroadcastTableRuleStatementTestCase) expected);
86          } else if (actual instanceof DropShardingTableRuleStatement) {
87              DropShardingTableRuleStatementAssert.assertIs(assertContext, (DropShardingTableRuleStatement) actual, (DropShardingTableRuleStatementTestCase) expected);
88          } else if (actual instanceof DropShadowRuleStatement) {
89              DropShadowRuleStatementAssert.assertIs(assertContext, (DropShadowRuleStatement) actual, (DropShadowRuleStatementTestCase) expected);
90          } else if (actual instanceof DropShadowAlgorithmStatement) {
91              DropShadowAlgorithmStatementAssert.assertIs(assertContext, (DropShadowAlgorithmStatement) actual, (DropShadowAlgorithmStatementTestCase) expected);
92          } else if (actual instanceof DropShardingAlgorithmStatement) {
93              DropShardingAlgorithmStatementAssert.assertIs(assertContext, (DropShardingAlgorithmStatement) actual, (DropShardingAlgorithmStatementTestCase) expected);
94          } else if (actual instanceof DropShardingKeyGeneratorStatement) {
95              DropShardingKeyGeneratorStatementAssert.assertIs(assertContext, (DropShardingKeyGeneratorStatement) actual, (DropShardingKeyGeneratorStatementTestCase) expected);
96          } else if (actual instanceof DropShardingAuditorStatement) {
97              DropShardingAuditorStatementAssert.assertIs(assertContext, (DropShardingAuditorStatement) actual, (DropShardingAuditorStatementTestCase) expected);
98          } else if (actual instanceof DropMaskRuleStatement) {
99              DropMaskRuleStatementAssert.assertIs(assertContext, (DropMaskRuleStatement) actual, (DropMaskRuleStatementTestCase) expected);
100         }
101     }
