58  /**
59   * MySQL incremental dumper.
60   */
61  @HighFrequencyInvocation
62  @Slf4j
63  public final class MySQLIncrementalDumper extends AbstractPipelineLifecycleRunnable implements IncrementalDumper {
64      
65      private final IncrementalDumperContext dumperContext;
66      
67      private final MySQLBinlogPosition binlogPosition;
68      
69      private final PipelineTableMetaDataLoader metaDataLoader;
70      
71      private final PipelineChannel channel;
72      
73      private final MySQLBinlogClient client;
74      
75      private final String catalog;
76      
77      public MySQLIncrementalDumper(final IncrementalDumperContext dumperContext, final IngestPosition binlogPosition, final PipelineChannel channel, final PipelineTableMetaDataLoader metaDataLoader) {
78          this.dumperContext = dumperContext;
79          this.binlogPosition = (MySQLBinlogPosition) binlogPosition;
80          this.channel = channel;
81          this.metaDataLoader = metaDataLoader;
82          StandardPipelineDataSourceConfiguration pipelineDataSourceConfig = (StandardPipelineDataSourceConfiguration) dumperContext.getCommonContext().getDataSourceConfig();
83          DatabaseType databaseType = TypedSPILoader.getService(DatabaseType.class, "MySQL");
84          ConnectionProperties connectionProps = DatabaseTypedSPILoader.getService(ConnectionPropertiesParser.class, databaseType).parse(pipelineDataSourceConfig.getUrl(), null, null);
85          ConnectInfo connectInfo = new ConnectInfo(
86                  generateServerId(), connectionProps.getHostname(), connectionProps.getPort(), pipelineDataSourceConfig.getUsername(), pipelineDataSourceConfig.getPassword());
87          log.info("incremental dump, jdbcUrl={}, serverId={}, hostname={}, port={}", pipelineDataSourceConfig.getUrl(), connectInfo.getServerId(), connectInfo.getHost(), connectInfo.getPort());
88          client = new MySQLBinlogClient(connectInfo, dumperContext.isDecodeWithTX());
89          catalog = connectionProps.getCatalog();
90      }
91      
92      private int generateServerId() {
93          int result = hashCode();
94          return Integer.MIN_VALUE == result ? Integer.MAX_VALUE : Math.abs(result);
95      }
96      
97      @Override
98      protected void runBlocking() {
99          client.connect();
100         client.subscribe(binlogPosition.getFilename(), binlogPosition.getPosition());
101         while (isRunning()) {
102             handleEvents(client.poll());
103         }
104     }
105     
106     private void handleEvents(final List<AbstractBinlogEvent> events) {
107         List<Record> dataRecords = new LinkedList<>();
108         for (AbstractBinlogEvent each : events) {
109             dataRecords.addAll(handleEvent(each));
110         }
111         if (!dataRecords.isEmpty()) {
112             channel.push(dataRecords);
113         }
114     }
115     
116     private List<? extends Record> handleEvent(final AbstractBinlogEvent event) {
117         if (!(event instanceof AbstractRowsEvent)) {
118             return Collections.singletonList(createPlaceholderRecord(event));
119         }
120         AbstractRowsEvent rowsEvent = (AbstractRowsEvent) event;
121         if (!rowsEvent.getDatabaseName().equals(catalog) || !dumperContext.getCommonContext().getTableNameMapper().containsTable(rowsEvent.getTableName())) {
122             return Collections.singletonList(createPlaceholderRecord(event));
123         }
124         PipelineTableMetaData tableMetaData = getPipelineTableMetaData(rowsEvent.getTableName());
125         if (event instanceof WriteRowsEvent) {
126             return handleWriteRowsEvent((WriteRowsEvent) event, tableMetaData);
127         }
128         if (event instanceof UpdateRowsEvent) {
129             return handleUpdateRowsEvent((UpdateRowsEvent) event, tableMetaData);
130         }
131         if (event instanceof DeleteRowsEvent) {
132             return handleDeleteRowsEvent((DeleteRowsEvent) event, tableMetaData);
133         }
134         return Collections.emptyList();
135     }
136     
137     private PlaceholderRecord createPlaceholderRecord(final AbstractBinlogEvent event) {
138         PlaceholderRecord result = new PlaceholderRecord(new MySQLBinlogPosition(event.getFileName(), event.getPosition()));
139         result.setCommitTime(event.getTimestamp() * 1000L);
140         return result;
141     }
142     
143     private PipelineTableMetaData getPipelineTableMetaData(final String actualTableName) {
144         CaseInsensitiveIdentifier logicTableName = dumperContext.getCommonContext().getTableNameMapper().getLogicTableName(actualTableName);
145         return metaDataLoader.getTableMetaData(dumperContext.getCommonContext().getTableAndSchemaNameMapper().getSchemaName(logicTableName), actualTableName);
146     }
147     
148     private List<DataRecord> handleWriteRowsEvent(final WriteRowsEvent event, final PipelineTableMetaData tableMetaData) {
149         List<DataRecord> result = new LinkedList<>();
150         for (Serializable[] each : event.getAfterRows()) {
151             DataRecord dataRecord = createDataRecord(PipelineSQLOperationType.INSERT, event, each.length);
152             for (int i = 0; i < each.length; i++) {
153                 PipelineColumnMetaData columnMetaData = tableMetaData.getColumnMetaData(i + 1);
154                 dataRecord.addColumn(new Column(columnMetaData.getName(), MySQLBinlogDataHandler.handle(columnMetaData, each[i]), true, columnMetaData.isUniqueKey()));
155             }
156             result.add(dataRecord);
157         }
158         return result;
159     }
160     
161     private List<DataRecord> handleUpdateRowsEvent(final UpdateRowsEvent event, final PipelineTableMetaData tableMetaData) {
162         List<DataRecord> result = new LinkedList<>();
163         for (int i = 0; i < event.getBeforeRows().size(); i++) {
164             Serializable[] beforeValues = event.getBeforeRows().get(i);
165             Serializable[] afterValues = event.getAfterRows().get(i);
166             DataRecord dataRecord = createDataRecord(PipelineSQLOperationType.UPDATE, event, beforeValues.length);
167             for (int j = 0; j < beforeValues.length; j++) {
168                 PipelineColumnMetaData columnMetaData = tableMetaData.getColumnMetaData(j + 1);
169                 Serializable oldValue = MySQLBinlogDataHandler.handle(columnMetaData, beforeValues[j]);
170                 Serializable newValue = MySQLBinlogDataHandler.handle(columnMetaData, afterValues[j]);
171                 boolean updated = !Objects.deepEquals(newValue, oldValue);
172                 dataRecord.addColumn(new Column(columnMetaData.getName(), oldValue, newValue, updated, columnMetaData.isUniqueKey()));
173             }
174             result.add(dataRecord);
175         }
176         return result;
177     }
178     
179     private List<DataRecord> handleDeleteRowsEvent(final DeleteRowsEvent event, final PipelineTableMetaData tableMetaData) {
180         List<DataRecord> result = new LinkedList<>();
181         for (Serializable[] each : event.getBeforeRows()) {
182             DataRecord dataRecord = createDataRecord(PipelineSQLOperationType.DELETE, event, each.length);
183             for (int i = 0, length = each.length; i < length; i++) {
184                 PipelineColumnMetaData columnMetaData = tableMetaData.getColumnMetaData(i + 1);
185                 dataRecord.addColumn(new Column(columnMetaData.getName(), MySQLBinlogDataHandler.handle(columnMetaData, each[i]), null, true, columnMetaData.isUniqueKey()));
186             }
187             result.add(dataRecord);
188         }
189         return result;
190     }
191     
192     private DataRecord createDataRecord(final PipelineSQLOperationType type, final AbstractRowsEvent rowsEvent, final int columnCount) {
193         String tableName = dumperContext.getCommonContext().getTableNameMapper().getLogicTableName(rowsEvent.getTableName()).toString();
194         IngestPosition binlogPosition = new MySQLBinlogPosition(rowsEvent.getFileName(), rowsEvent.getPosition());
195         DataRecord result = new DataRecord(type, tableName, binlogPosition, columnCount);
196         result.setActualTableName(rowsEvent.getTableName());
197         result.setCommitTime(rowsEvent.getTimestamp() * 1000L);
198         return result;
199     }
200     
201     @Override
202     protected void doStop() {
203         if (null != client) {
204             client.closeChannel();
205         }
206     }
