46  /**
47   * Show sharding table rules executor.
48   */
49  public final class ShowShardingTableRuleExecutor implements RQLExecutor<ShowShardingTableRulesStatement> {
50      
51      private ShardingRuleConfiguration shardingRuleConfig;
52      
53      @Override
54      public Collection<LocalDataQueryResultRow> getRows(final ShardingSphereDatabase database, final ShowShardingTableRulesStatement sqlStatement) {
55          Optional<ShardingRule> rule = database.getRuleMetaData().findSingleRule(ShardingRule.class);
56          if (!rule.isPresent()) {
57              return Collections.emptyList();
58          }
59          ShardingRuleConfiguration config = (ShardingRuleConfiguration) rule.get().getConfiguration();
60          String tableName = sqlStatement.getTableName();
61          Iterator<ShardingTableRuleConfiguration> tables;
62          Iterator<ShardingAutoTableRuleConfiguration> autoTables;
63          if (Objects.isNull(tableName)) {
64              tables = config.getTables().iterator();
65              autoTables = config.getAutoTables().iterator();
66          } else {
67              tables = config.getTables().stream().filter(each -> tableName.equalsIgnoreCase(each.getLogicTable())).collect(Collectors.toList()).iterator();
68              autoTables = config.getAutoTables().stream().filter(each -> tableName.equalsIgnoreCase(each.getLogicTable())).collect(Collectors.toList()).iterator();
69          }
70          shardingRuleConfig = config;
71          Collection<LocalDataQueryResultRow> result = new LinkedList<>();
72          while (tables.hasNext()) {
73              result.add(buildTableRowData(tables.next()));
74          }
75          while (autoTables.hasNext()) {
76              result.add(buildAutoTableRowData(autoTables.next()));
77          }
78          return result;
79      }
80      
81      @Override
82      public Collection<String> getColumnNames() {
83          return Arrays.asList("table", "actual_data_nodes", "actual_data_sources", "database_strategy_type", "database_sharding_column", "database_sharding_algorithm_type",
84                  "database_sharding_algorithm_props", "table_strategy_type", "table_sharding_column", "table_sharding_algorithm_type", "table_sharding_algorithm_props",
85                  "key_generate_column", "key_generator_type", "key_generator_props", "auditor_types", "allow_hint_disable");
86      }
87      
88      private LocalDataQueryResultRow buildTableRowData(final ShardingTableRuleConfiguration shardingTableRuleConfig) {
89          Optional<ShardingStrategyConfiguration> databaseShardingStrategyConfig = getDatabaseShardingStrategy(shardingTableRuleConfig);
90          Optional<ShardingStrategyConfiguration> tableShardingStrategyConfig = getTableShardingStrategy(shardingTableRuleConfig.getTableShardingStrategy());
91          return new LocalDataQueryResultRow(shardingTableRuleConfig.getLogicTable(), shardingTableRuleConfig.getActualDataNodes(), "",
92                  databaseShardingStrategyConfig.map(this::getStrategyType).orElse(""), databaseShardingStrategyConfig.map(this::getShardingColumn).orElse(""),
93                  databaseShardingStrategyConfig.map(this::getAlgorithmType).orElse(""), databaseShardingStrategyConfig.map(this::getAlgorithmProperties).orElse(""),
94                  tableShardingStrategyConfig.map(this::getStrategyType).orElse(""), tableShardingStrategyConfig.map(this::getShardingColumn).orElse(""),
95                  tableShardingStrategyConfig.map(this::getAlgorithmType).orElse(""), tableShardingStrategyConfig.map(this::getAlgorithmProperties).orElse(""),
96                  getKeyGenerateColumn(shardingTableRuleConfig.getKeyGenerateStrategy()), getKeyGeneratorType(shardingTableRuleConfig.getKeyGenerateStrategy()),
97                  getKeyGeneratorProps(shardingTableRuleConfig.getKeyGenerateStrategy()), getAuditorTypes(shardingTableRuleConfig.getAuditStrategy()),
98                  getAllowHintDisable(shardingTableRuleConfig.getAuditStrategy()));
99      }
100     
101     private LocalDataQueryResultRow buildAutoTableRowData(final ShardingAutoTableRuleConfiguration shardingAutoTableRuleConfig) {
102         Optional<ShardingStrategyConfiguration> tableShardingStrategyConfig = getTableShardingStrategy(shardingAutoTableRuleConfig.getShardingStrategy());
103         return new LocalDataQueryResultRow(shardingAutoTableRuleConfig.getLogicTable(), "", shardingAutoTableRuleConfig.getActualDataSources(), "", "", "", "",
104                 tableShardingStrategyConfig.map(this::getStrategyType).orElse(""), tableShardingStrategyConfig.map(this::getShardingColumn).orElse(""),
105                 tableShardingStrategyConfig.map(this::getAlgorithmType).orElse(""), tableShardingStrategyConfig.map(this::getAlgorithmProperties).orElse(""),
106                 getKeyGenerateColumn(shardingAutoTableRuleConfig.getKeyGenerateStrategy()), getKeyGeneratorType(shardingAutoTableRuleConfig.getKeyGenerateStrategy()),
107                 getKeyGeneratorProps(shardingAutoTableRuleConfig.getKeyGenerateStrategy()), getAuditorTypes(shardingAutoTableRuleConfig.getAuditStrategy()),
108                 getAllowHintDisable(shardingAutoTableRuleConfig.getAuditStrategy()));
109     }
110     
111     private String getShardingColumn(final ShardingStrategyConfiguration shardingStrategyConfig) {
112         if (shardingStrategyConfig instanceof StandardShardingStrategyConfiguration) {
113             return ((StandardShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumn();
114         }
115         if (shardingStrategyConfig instanceof ComplexShardingStrategyConfiguration) {
116             return ((ComplexShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumns();
117         }
118         return "";
119     }
120     
121     private String getAlgorithmType(final ShardingStrategyConfiguration shardingStrategyConfig) {
122         return shardingStrategyConfig instanceof NoneShardingStrategyConfiguration ? "" : getAlgorithmConfiguration(shardingStrategyConfig.getShardingAlgorithmName()).getType();
123     }
124     
125     private String getAlgorithmProperties(final ShardingStrategyConfiguration shardingStrategyConfig) {
126         return shardingStrategyConfig instanceof NoneShardingStrategyConfiguration
127                 ? ""
128                 : PropertiesConverter.convert(getAlgorithmConfiguration(shardingStrategyConfig.getShardingAlgorithmName()).getProps());
129     }
130     
131     private Optional<ShardingStrategyConfiguration> getDatabaseShardingStrategy(final ShardingTableRuleConfiguration shardingTableRuleConfig) {
132         return null == shardingTableRuleConfig.getDatabaseShardingStrategy()
133                 ? Optional.ofNullable(shardingRuleConfig.getDefaultDatabaseShardingStrategy())
134                 : Optional.ofNullable(shardingTableRuleConfig.getDatabaseShardingStrategy());
135     }
136     
137     private AlgorithmConfiguration getAlgorithmConfiguration(final String algorithmName) {
138         return shardingRuleConfig.getShardingAlgorithms().get(algorithmName);
139     }
140     
141     private String getStrategyType(final ShardingStrategyConfiguration shardingStrategyConfig) {
142         return shardingStrategyConfig.getType();
143     }
144     
145     private Optional<ShardingStrategyConfiguration> getTableShardingStrategy(final ShardingStrategyConfiguration shardingStrategyConfig) {
146         return null == shardingStrategyConfig ? Optional.ofNullable(shardingRuleConfig.getDefaultTableShardingStrategy()) : Optional.of(shardingStrategyConfig);
147     }
148     
149     private String getKeyGenerateColumn(final KeyGenerateStrategyConfiguration keyGenerateStrategyConfig) {
150         return getKeyGenerateStrategyConfiguration(keyGenerateStrategyConfig).isPresent() ? getKeyGenerateStrategyConfiguration(keyGenerateStrategyConfig).get().getColumn() : "";
151     }
152     
153     private String getKeyGeneratorType(final KeyGenerateStrategyConfiguration originalKeyGenerateStrategyConfig) {
154         Optional<KeyGenerateStrategyConfiguration> keyGenerateStrategyConfig = getKeyGenerateStrategyConfiguration(originalKeyGenerateStrategyConfig);
155         return keyGenerateStrategyConfig.isPresent() ? shardingRuleConfig.getKeyGenerators().get(keyGenerateStrategyConfig.get().getKeyGeneratorName()).getType() : "";
156     }
157     
158     private String getKeyGeneratorProps(final KeyGenerateStrategyConfiguration keyGenerateStrategyConfig) {
159         return getKeyGenerateStrategyConfiguration(keyGenerateStrategyConfig)
160                 .map(optional -> PropertiesConverter.convert(shardingRuleConfig.getKeyGenerators().get(optional.getKeyGeneratorName()).getProps())).orElse("");
161     }
162     
163     private Optional<KeyGenerateStrategyConfiguration> getKeyGenerateStrategyConfiguration(final KeyGenerateStrategyConfiguration keyGenerateStrategyConfig) {
164         return null == keyGenerateStrategyConfig ? Optional.ofNullable(shardingRuleConfig.getDefaultKeyGenerateStrategy()) : Optional.of(keyGenerateStrategyConfig);
165     }
166     
167     private String getAuditorTypes(final ShardingAuditStrategyConfiguration shardingAuditStrategyConfig) {
168         Optional<ShardingAuditStrategyConfiguration> auditStrategyConfig = getShardingAuditStrategyConfiguration(shardingAuditStrategyConfig);
169         Collection<String> auditorTypes = new LinkedList<>();
170         if (auditStrategyConfig.isPresent()) {
171             for (String each : auditStrategyConfig.get().getAuditorNames()) {
172                 auditorTypes.add(shardingRuleConfig.getAuditors().get(each).getType());
173             }
174         }
175         return !auditorTypes.isEmpty() ? String.join(",", auditorTypes) : "";
176     }
177     
178     private String getAllowHintDisable(final ShardingAuditStrategyConfiguration shardingAuditStrategyConfig) {
179         return getShardingAuditStrategyConfiguration(shardingAuditStrategyConfig).isPresent()
180                 ? Boolean.valueOf(getShardingAuditStrategyConfiguration(shardingAuditStrategyConfig).get().isAllowHintDisable()).toString()
181                 : "";
182     }
183     
184     private Optional<ShardingAuditStrategyConfiguration> getShardingAuditStrategyConfiguration(final ShardingAuditStrategyConfiguration shardingAuditStrategyConfig) {
185         return null == shardingAuditStrategyConfig ? Optional.ofNullable(shardingRuleConfig.getDefaultAuditStrategy()) : Optional.of(shardingAuditStrategyConfig);
186     }
187     
188     @Override
189     public String getType() {
190         return ShowShardingTableRulesStatement.class.getName();
191     }
