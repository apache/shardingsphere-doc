39  /**
40   * Show encrypt rule executor.
41   */
42  @Setter
43  public final class ShowEncryptRuleExecutor implements DistSQLQueryExecutor<ShowEncryptRulesStatement>, DistSQLExecutorRuleAware<EncryptRule> {
44      
45      private EncryptRule rule;
46      
47      @Override
48      public Collection<String> getColumnNames(final ShowEncryptRulesStatement sqlStatement) {
49          return Arrays.asList("table", "logic_column", "cipher_column",
50                  "assisted_query_column", "like_query_column", "encryptor_type", "encryptor_props", "assisted_query_type", "assisted_query_props", "like_query_type", "like_query_props");
51      }
52      
53      @Override
54      public Collection<LocalDataQueryResultRow> getRows(final ShowEncryptRulesStatement sqlStatement, final ContextManager contextManager) {
55          return rule.getConfiguration().getTables().stream().filter(each -> null == sqlStatement.getTableName() || each.getName().equals(sqlStatement.getTableName()))
56                  .map(each -> buildColumnData(each, rule.getConfiguration().getEncryptors())).flatMap(Collection::stream).collect(Collectors.toList());
57      }
58      
59      private Collection<LocalDataQueryResultRow> buildColumnData(final EncryptTableRuleConfiguration tableRuleConfig, final Map<String, AlgorithmConfiguration> encryptors) {
60          Collection<LocalDataQueryResultRow> result = new LinkedList<>();
61          for (EncryptColumnRuleConfiguration each : tableRuleConfig.getColumns()) {
62              AlgorithmConfiguration encryptorAlgorithmConfig = encryptors.get(each.getCipher().getEncryptorName());
63              Optional<AlgorithmConfiguration> assistedQueryEncryptorAlgorithmConfig = each.getAssistedQuery().map(optional -> encryptors.get(optional.getEncryptorName()));
64              Optional<AlgorithmConfiguration> likeQueryEncryptorAlgorithmConfig = each.getLikeQuery().map(optional -> encryptors.get(optional.getEncryptorName()));
65              result.add(new LocalDataQueryResultRow(
66                      tableRuleConfig.getName(),
67                      each.getName(),
68                      each.getCipher().getName(),
69                      each.getAssistedQuery().map(EncryptColumnItemRuleConfiguration::getName),
70                      each.getLikeQuery().map(EncryptColumnItemRuleConfiguration::getName),
71                      encryptorAlgorithmConfig.getType(),
72                      encryptorAlgorithmConfig.getProps(),
73                      assistedQueryEncryptorAlgorithmConfig.map(AlgorithmConfiguration::getType),
74                      assistedQueryEncryptorAlgorithmConfig.map(AlgorithmConfiguration::getProps),
75                      likeQueryEncryptorAlgorithmConfig.map(AlgorithmConfiguration::getType),
76                      likeQueryEncryptorAlgorithmConfig.map(AlgorithmConfiguration::getProps)));
77          }
78          return result;
79      }
80      
81      @Override
82      public Class<EncryptRule> getRuleClass() {
83          return EncryptRule.class;
84      }
85      
86      @Override
87      public Class<ShowEncryptRulesStatement> getType() {
88          return ShowEncryptRulesStatement.class;
89      }
90  }
