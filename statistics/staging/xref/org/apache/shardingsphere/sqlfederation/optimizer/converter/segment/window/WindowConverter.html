20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.calcite.sql.SqlIdentifier;
23  import org.apache.calcite.sql.SqlLiteral;
24  import org.apache.calcite.sql.SqlNode;
25  import org.apache.calcite.sql.SqlNodeList;
26  import org.apache.calcite.sql.SqlWindow;
27  import org.apache.calcite.sql.parser.SqlParserPos;
28  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.WindowItemSegment;
29  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.WindowSegment;
30  import org.apache.shardingsphere.sqlfederation.optimizer.converter.segment.expression.ExpressionConverter;
31  
32  import java.util.Collection;
33  import java.util.LinkedList;
34  import java.util.Optional;
35  
36  /**
37   * Window converter.
38   */
39  @NoArgsConstructor(access = AccessLevel.PRIVATE)
40  public final class WindowConverter {
41      
42      /**
43       * Convert window segment to sql node list.
44       * 
45       * @param segment window segment
46       * @return sql node list
47       */
48      public static Optional<SqlNodeList> convert(final WindowSegment segment) {
49          Collection<SqlWindow> sqlWindows = new LinkedList<>();
50          for (WindowItemSegment each : segment.getItemSegments()) {
51              SqlIdentifier sqlIdentifier = null == each.getWindowName() ? new SqlIdentifier("", SqlParserPos.ZERO) : new SqlIdentifier(each.getWindowName().getValue(), SqlParserPos.ZERO);
52              Collection<SqlNode> partitionNodes = new LinkedList<>();
53              each.getPartitionListSegments().forEach(expressionSegment -> ExpressionConverter.convert(expressionSegment).ifPresent(partitionNodes::add));
54              SqlNodeList partitionList = new SqlNodeList(partitionNodes, SqlParserPos.ZERO);
55              SqlNodeList orderList = new SqlNodeList(SqlParserPos.ZERO);
56              SqlWindow sqlWindow = new SqlWindow(SqlParserPos.ZERO, sqlIdentifier, null, partitionList, orderList, SqlLiteral.createBoolean(false, SqlParserPos.ZERO), null, null, null);
57              sqlWindows.add(sqlWindow);
58          }
59          SqlNodeList result = new SqlNodeList(sqlWindows, SqlParserPos.ZERO);
60          return Optional.of(result);
61      }
62  }
