43  /**
44   * Command execute engine for openGauss.
45   */
46  public final class OpenGaussCommandExecuteEngine implements CommandExecuteEngine {
47      
48      private final PostgreSQLCommandExecuteEngine postgresqlCommandExecuteEngine = new PostgreSQLCommandExecuteEngine();
49      
50      @Override
51      public CommandPacketType getCommandPacketType(final PacketPayload payload) {
52          return OpenGaussCommandPacketType.valueOf(payload.getByteBuf().getByte(payload.getByteBuf().readerIndex()));
53      }
54      
55      @Override
56      public CommandPacket getCommandPacket(final PacketPayload payload, final CommandPacketType type, final ConnectionSession connectionSession) {
57          return OpenGaussCommandPacketFactory.newInstance(type, (PostgreSQLPacketPayload) payload);
58      }
59      
60      @Override
61      public CommandExecutor getCommandExecutor(final CommandPacketType type, final CommandPacket packet, final ConnectionSession connectionSession) throws SQLException {
62          PortalContext portalContext = PostgreSQLPortalContextRegistry.getInstance().get(connectionSession.getConnectionId());
63          return OpenGaussCommandExecutorFactory.newInstance(type, (PostgreSQLCommandPacket) packet, connectionSession, portalContext);
64      }
65      
66      @Override
67      public PostgreSQLPacket getErrorPacket(final Exception cause) {
68          return OpenGaussErrorPacketFactory.newInstance(cause);
69      }
70      
71      @Override
72      public Optional<DatabasePacket> getOtherPacket(final ConnectionSession connectionSession) {
73          return postgresqlCommandExecuteEngine.getOtherPacket(connectionSession);
74      }
75      
76      @Override
77      public void writeQueryData(final ChannelHandlerContext context,
78                                 final ProxyDatabaseConnectionManager databaseConnectionManager, final QueryCommandExecutor queryCommandExecutor, final int headerPackagesCount) throws SQLException {
79          postgresqlCommandExecuteEngine.writeQueryData(context, databaseConnectionManager, queryCommandExecutor, headerPackagesCount);
80      }
81  }
