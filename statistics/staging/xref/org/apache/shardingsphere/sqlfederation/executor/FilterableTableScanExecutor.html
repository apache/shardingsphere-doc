121     @Override
122     public Enumerable<Object> executeScalar(final ShardingSphereTable table, final ScanNodeExecutorContext scanContext) {
123         return new AbstractEnumerable<Object>() {
124             
125             @Override
126             public Enumerator<Object> enumerator() {
127                 return new EmptyRowEnumerator<>();
128             }
129         };
130     }
131     
132     @Override
133     public Enumerable<Object[]> execute(final ShardingSphereTable table, final ScanNodeExecutorContext scanContext) {
134         String databaseName = executorContext.getDatabaseName().toLowerCase();
135         String schemaName = executorContext.getSchemaName().toLowerCase();
136         DatabaseType databaseType = DatabaseTypeEngine.getTrunkDatabaseType(optimizerContext.getParserContext(databaseName).getDatabaseType().getType());
137         if (databaseType.getSystemSchemas().contains(schemaName)) {
138             return executeByShardingSphereData(databaseName, schemaName, table);
139         }
140         SqlString sqlString = createSQLString(table, (FilterableScanNodeExecutorContext) scanContext, SQLDialectFactory.getSQLDialect(databaseType));
141         SQLFederationExecutorContext federationContext = executorContext.getFederationContext();
142         QueryContext queryContext = createQueryContext(federationContext.getMetaData(), sqlString, databaseType, federationContext.getQueryContext().isUseCache());
143         ShardingSphereDatabase database = federationContext.getMetaData().getDatabase(databaseName);
144         // TODO need to get session context
145         ExecutionContext context = new KernelProcessor().generateExecutionContext(queryContext, database, globalRuleMetaData, executorContext.getProps(), new ConnectionContext());
146         if (federationContext.isPreview()) {
147             federationContext.getExecutionUnits().addAll(context.getExecutionUnits());
148             return createEmptyEnumerable();
149         }
150         return execute(databaseType, queryContext, database, context);
151     }
152     
153     private AbstractEnumerable<Object[]> execute(final DatabaseType databaseType, final QueryContext queryContext, final ShardingSphereDatabase database, final ExecutionContext context) {
154         try {
155             ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext =
156                     prepareEngine.prepare(context.getRouteContext(), context.getExecutionUnits(), new ExecutionGroupReportContext(database.getName()));
157             setParameters(executionGroupContext.getInputGroups());
158             processEngine.executeSQL(executionGroupContext, context.getQueryContext());
159             List<QueryResult> queryResults = execute(executionGroupContext, databaseType);
160             // TODO need to get session context
161             MergeEngine mergeEngine = new MergeEngine(database, executorContext.getProps(), new ConnectionContext());
162             MergedResult mergedResult = mergeEngine.merge(queryResults, queryContext.getSqlStatementContext());
163             Collection<Statement> statements = getStatements(executionGroupContext.getInputGroups());
164             return createEnumerable(mergedResult, queryResults.get(0).getMetaData(), statements);
165         } catch (final SQLException ex) {
166             throw new SQLWrapperException(ex);
167         } finally {
168             processEngine.completeSQLExecution();
169         }
170     }
171     
172     private List<QueryResult> execute(final ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext, final DatabaseType databaseType) throws SQLException {
173         Collection<QueryResult> queryResults = jdbcExecutor.execute(executionGroupContext, callback).stream().map(each -> (QueryResult) each).collect(Collectors.toList());
174         List<QueryResult> result = new LinkedList<>();
175         for (QueryResult each : queryResults) {
176             QueryResult queryResult = each instanceof JDBCStreamQueryResult
177                     ? new JDBCMemoryQueryResult(((JDBCStreamQueryResult) each).getResultSet(), databaseType)
178                     : each;
179             result.add(queryResult);
180         }
181         return result;
182     }
183     
184     private Enumerable<Object[]> executeByShardingSphereData(final String databaseName, final String schemaName, final ShardingSphereTable table) {
185         Optional<ShardingSphereTableData> tableData = Optional.ofNullable(data.getDatabaseData().get(databaseName)).map(optional -> optional.getSchemaData().get(schemaName))
186                 .map(ShardingSphereSchemaData::getTableData).map(shardingSphereData -> shardingSphereData.get(table.getName()));
187         return tableData.map(this::createMemoryEnumerator).orElseGet(this::createEmptyEnumerable);
188     }
189     
190     private Enumerable<Object[]> createMemoryEnumerator(final ShardingSphereTableData tableData) {
191         return new AbstractEnumerable<Object[]>() {
192             
193             @Override
194             public Enumerator<Object[]> enumerator() {
195                 return new MemoryEnumerator<>(tableData.getRows());
196             }
197         };
198     }
199     
200     private Collection<Statement> getStatements(final Collection<ExecutionGroup<JDBCExecutionUnit>> inputGroups) {
201         Collection<Statement> result = new LinkedList<>();
202         for (ExecutionGroup<JDBCExecutionUnit> each : inputGroups) {
203             for (JDBCExecutionUnit executionUnit : each.getInputs()) {
204                 result.add(executionUnit.getStorageResource());
205             }
206         }
207         return result;
208     }
209     
210     private SqlString createSQLString(final ShardingSphereTable table, final FilterableScanNodeExecutorContext scanContext, final SqlDialect sqlDialect) {
211         return new RelToSqlConverter(sqlDialect).visitRoot(createRelNode(table, scanContext)).asStatement().toSqlString(sqlDialect);
212     }
213     
214     private void setParameters(final Collection<ExecutionGroup<JDBCExecutionUnit>> inputGroups) {
215         for (ExecutionGroup<JDBCExecutionUnit> each : inputGroups) {
216             for (JDBCExecutionUnit executionUnit : each.getInputs()) {
217                 if (!(executionUnit.getStorageResource() instanceof PreparedStatement)) {
218                     continue;
219                 }
220                 setParameters((PreparedStatement) executionUnit.getStorageResource(), executionUnit.getExecutionUnit().getSqlUnit().getParameters());
221             }
222         }
223     }
224     
225     @SneakyThrows(SQLException.class)
226     private void setParameters(final PreparedStatement preparedStatement, final List<Object> params) {
227         for (int i = 0; i < params.size(); i++) {
228             preparedStatement.setObject(i + 1, params.get(i));
229         }
230     }
231     
232     private RelNode createRelNode(final ShardingSphereTable table, final FilterableScanNodeExecutorContext scanContext) {
233         String databaseName = executorContext.getDatabaseName();
234         String schemaName = executorContext.getSchemaName();
235         CalciteConnectionConfig connectionConfig = new CalciteConnectionConfigImpl(optimizerContext.getParserContext(databaseName).getDialectProps());
236         ShardingSphereDatabase database = executorContext.getFederationContext().getMetaData().getDatabase(databaseName);
237         CalciteCatalogReader catalogReader = SQLFederationPlannerUtils.createCatalogReader(schemaName,
238                 new FilterableSchema(schemaName, database.getSchema(schemaName), database.getProtocolType(), JAVA_TYPE_FACTORY, null), JAVA_TYPE_FACTORY, connectionConfig);
239         RelOptCluster relOptCluster = RelOptCluster.create(SQLFederationPlannerUtils.createVolcanoPlanner(), new RexBuilder(JAVA_TYPE_FACTORY));
240         RelBuilder builder = RelFactories.LOGICAL_BUILDER.create(relOptCluster, catalogReader).scan(table.getName()).filter(scanContext.getFilterValues());
241         if (null != scanContext.getProjects()) {
242             builder.project(createProjections(scanContext.getProjects(), builder, table.getColumnNames()));
243         }
244         return builder.build();
245     }
246     
247     private Collection<RexNode> createProjections(final int[] projects, final RelBuilder relBuilder, final List<String> columnNames) {
248         Collection<RexNode> result = new LinkedList<>();
249         for (int each : projects) {
250             result.add(relBuilder.field(columnNames.get(each)));
251         }
252         return result;
253     }
254     
255     private AbstractEnumerable<Object[]> createEnumerable(final MergedResult mergedResult, final QueryResultMetaData metaData, final Collection<Statement> statements) throws SQLException {
256         // TODO remove getRows when mergedResult support JDBC first method
257         Collection<Object[]> rows = getRows(mergedResult, metaData);
258         return new AbstractEnumerable<Object[]>() {
259             
260             @Override
261             public Enumerator<Object[]> enumerator() {
262                 return new SQLFederationRowEnumerator<>(rows, statements);
263             }
264         };
265     }
266     
267     private Collection<Object[]> getRows(final MergedResult mergedResult, final QueryResultMetaData metaData) throws SQLException {
268         Collection<Object[]> result = new LinkedList<>();
269         while (mergedResult.next()) {
270             Object[] currentRow = new Object[metaData.getColumnCount()];
271             for (int i = 0; i < metaData.getColumnCount(); i++) {
272                 currentRow[i] = mergedResult.getValue(i + 1, Object.class);
273             }
274             result.add(currentRow);
275         }
276         return result;
277     }
278     
279     private QueryContext createQueryContext(final ShardingSphereMetaData metaData, final SqlString sqlString, final DatabaseType databaseType, final boolean useCache) {
280         String sql = sqlString.getSql().replace("\n", " ");
281         SQLStatement sqlStatement = new SQLStatementParserEngine(databaseType.getType(),
282                 optimizerContext.getSqlParserRule().getSqlStatementCache(), optimizerContext.getSqlParserRule().getParseTreeCache(),
283                 optimizerContext.getSqlParserRule().isSqlCommentParseEnabled()).parse(sql, useCache);
284         List<Object> params = getParameters(sqlString.getDynamicParameters());
285         SQLStatementContext<?> sqlStatementContext = SQLStatementContextFactory.newInstance(metaData, params, sqlStatement, executorContext.getDatabaseName());
286         return new QueryContext(sqlStatementContext, sql, params, new HintValueContext(), useCache);
287     }
288     
289     private List<Object> getParameters(final List<Integer> paramIndexes) {
290         if (null == paramIndexes) {
291             return Collections.emptyList();
292         }
293         List<Object> result = new ArrayList<>();
294         for (Integer each : paramIndexes) {
295             result.add(executorContext.getFederationContext().getQueryContext().getParameters().get(each));
296         }
297         return result;
298     }
299     
300     private AbstractEnumerable<Object[]> createEmptyEnumerable() {
301         return new AbstractEnumerable<Object[]>() {
302             
303             @Override
304             public Enumerator<Object[]> enumerator() {
305                 return new EmptyRowEnumerator<>();
306             }
307         };
308     }
