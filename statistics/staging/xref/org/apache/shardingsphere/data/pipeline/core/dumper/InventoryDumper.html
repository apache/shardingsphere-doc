66  /**
67   * Inventory dumper.
68   */
69  @Slf4j
70  public final class InventoryDumper extends AbstractLifecycleExecutor implements Dumper {
71      
72      @Getter(AccessLevel.PROTECTED)
73      private final InventoryDumperConfiguration dumperConfig;
74      
75      private final PipelineChannel channel;
76      
77      private final DataSource dataSource;
78      
79      private final PipelineInventoryDumpSQLBuilder inventoryDumpSQLBuilder;
80      
81      private final ColumnValueReaderEngine columnValueReaderEngine;
82      
83      private final PipelineTableMetaDataLoader metaDataLoader;
84      
85      private final AtomicReference<Statement> dumpStatement = new AtomicReference<>();
86      
87      public InventoryDumper(final InventoryDumperConfiguration dumperConfig, final PipelineChannel channel, final DataSource dataSource, final PipelineTableMetaDataLoader metaDataLoader) {
88          this.dumperConfig = dumperConfig;
89          this.channel = channel;
90          this.dataSource = dataSource;
91          DatabaseType databaseType = dumperConfig.getDataSourceConfig().getDatabaseType();
92          inventoryDumpSQLBuilder = new PipelineInventoryDumpSQLBuilder(databaseType);
93          columnValueReaderEngine = new ColumnValueReaderEngine(databaseType);
94          this.metaDataLoader = metaDataLoader;
95      }
96      
97      @Override
98      protected void runBlocking() {
99          IngestPosition position = dumperConfig.getPosition();
100         if (position instanceof FinishedPosition) {
101             log.info("Ignored because of already finished.");
102             return;
103         }
104         PipelineTableMetaData tableMetaData = metaDataLoader.getTableMetaData(dumperConfig.getSchemaName(new LogicTableName(dumperConfig.getLogicTableName())), dumperConfig.getActualTableName());
105         try (Connection connection = dataSource.getConnection()) {
106             dump(tableMetaData, connection);
107         } catch (final SQLException ex) {
108             log.error("Inventory dump, ex caught, msg={}.", ex.getMessage());
109             throw new IngestException("Inventory dump failed on " + dumperConfig.getActualTableName(), ex);
110         }
111     }
112     
113     @SuppressWarnings("MagicConstant")
114     private void dump(final PipelineTableMetaData tableMetaData, final Connection connection) throws SQLException {
115         int batchSize = dumperConfig.getBatchSize();
116         DatabaseType databaseType = dumperConfig.getDataSourceConfig().getDatabaseType();
117         if (null != dumperConfig.getTransactionIsolation()) {
118             connection.setTransactionIsolation(dumperConfig.getTransactionIsolation());
119         }
120         try (PreparedStatement preparedStatement = JDBCStreamQueryBuilder.build(databaseType, connection, buildInventoryDumpSQL())) {
121             dumpStatement.set(preparedStatement);
122             if (!(databaseType instanceof MySQLDatabaseType)) {
123                 preparedStatement.setFetchSize(batchSize);
124             }
125             setParameters(preparedStatement);
126             try (ResultSet resultSet = preparedStatement.executeQuery()) {
127                 int rowCount = 0;
128                 JobRateLimitAlgorithm rateLimitAlgorithm = dumperConfig.getRateLimitAlgorithm();
129                 ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
130                 List<Record> dataRecords = new LinkedList<>();
131                 while (resultSet.next()) {
132                     if (dataRecords.size() >= batchSize) {
133                         channel.pushRecords(dataRecords);
134                         dataRecords = new LinkedList<>();
135                     }
136                     dataRecords.add(loadDataRecord(resultSet, resultSetMetaData, tableMetaData));
137                     ++rowCount;
138                     if (!isRunning()) {
139                         log.info("Broke because of inventory dump is not running.");
140                         break;
141                     }
142                     if (null != rateLimitAlgorithm && 0 == rowCount % batchSize) {
143                         rateLimitAlgorithm.intercept(JobOperationType.SELECT, 1);
144                     }
145                 }
146                 dataRecords.add(new FinishedRecord(new FinishedPosition()));
147                 channel.pushRecords(dataRecords);
148                 dumpStatement.set(null);
149                 log.info("Inventory dump done, rowCount={}", rowCount);
150             }
151         }
152     }
153     
154     private String buildInventoryDumpSQL() {
155         if (!Strings.isNullOrEmpty(dumperConfig.getQuerySQL())) {
156             return dumperConfig.getQuerySQL();
157         }
158         LogicTableName logicTableName = new LogicTableName(dumperConfig.getLogicTableName());
159         String schemaName = dumperConfig.getSchemaName(logicTableName);
160         if (!dumperConfig.hasUniqueKey()) {
161             return inventoryDumpSQLBuilder.buildFetchAllSQL(schemaName, dumperConfig.getActualTableName());
162         }
163         PrimaryKeyPosition<?> primaryKeyPosition = (PrimaryKeyPosition<?>) dumperConfig.getPosition();
164         PipelineColumnMetaData firstColumn = dumperConfig.getUniqueKeyColumns().get(0);
165         List<String> columnNames = dumperConfig.getColumnNames(logicTableName).orElse(Collections.singletonList("*"));
166         if (PipelineJdbcUtils.isIntegerColumn(firstColumn.getDataType()) || PipelineJdbcUtils.isStringColumn(firstColumn.getDataType())) {
167             if (null != primaryKeyPosition.getBeginValue() && null != primaryKeyPosition.getEndValue()) {
168                 return inventoryDumpSQLBuilder.buildDivisibleSQL(schemaName, dumperConfig.getActualTableName(), columnNames, firstColumn.getName());
169             }
170             if (null != primaryKeyPosition.getBeginValue() && null == primaryKeyPosition.getEndValue()) {
171                 return inventoryDumpSQLBuilder.buildUnlimitedDivisibleSQL(schemaName, dumperConfig.getActualTableName(), columnNames, firstColumn.getName());
172             }
173         }
174         return inventoryDumpSQLBuilder.buildIndivisibleSQL(schemaName, dumperConfig.getActualTableName(), columnNames, firstColumn.getName());
175     }
176     
177     private void setParameters(final PreparedStatement preparedStatement) throws SQLException {
178         if (!dumperConfig.hasUniqueKey()) {
179             return;
180         }
181         PipelineColumnMetaData firstColumn = dumperConfig.getUniqueKeyColumns().get(0);
182         PrimaryKeyPosition<?> position = (PrimaryKeyPosition<?>) dumperConfig.getPosition();
183         if (PipelineJdbcUtils.isIntegerColumn(firstColumn.getDataType()) && null != position.getBeginValue() && null != position.getEndValue()) {
184             preparedStatement.setObject(1, position.getBeginValue());
185             preparedStatement.setObject(2, position.getEndValue());
186             return;
187         }
188         if (PipelineJdbcUtils.isStringColumn(firstColumn.getDataType())) {
189             if (null != position.getBeginValue()) {
190                 preparedStatement.setObject(1, position.getBeginValue());
191             }
192             if (null != position.getEndValue()) {
193                 preparedStatement.setObject(2, position.getEndValue());
194             }
195         }
196     }
197     
198     private DataRecord loadDataRecord(final ResultSet resultSet, final ResultSetMetaData resultSetMetaData, final PipelineTableMetaData tableMetaData) throws SQLException {
199         int columnCount = resultSetMetaData.getColumnCount();
200         DataRecord result = new DataRecord(IngestDataChangeType.INSERT, dumperConfig.getLogicTableName(), newPosition(resultSet), columnCount);
201         List<String> insertColumnNames = Optional.ofNullable(dumperConfig.getInsertColumnNames()).orElse(Collections.emptyList());
202         ShardingSpherePreconditions.checkState(insertColumnNames.isEmpty() || insertColumnNames.size() == resultSetMetaData.getColumnCount(),
203                 () -> new PipelineInvalidParameterException("Insert colum names count not equals ResultSet column count"));
204         for (int i = 1; i <= columnCount; i++) {
205             String columnName = insertColumnNames.isEmpty() ? resultSetMetaData.getColumnName(i) : insertColumnNames.get(i - 1);
206             ShardingSpherePreconditions.checkNotNull(tableMetaData.getColumnMetaData(columnName), () -> new PipelineInvalidParameterException(String.format("Column name is %s", columnName)));
207             boolean isUniqueKey = tableMetaData.getColumnMetaData(columnName).isUniqueKey();
208             result.addColumn(new Column(columnName, columnValueReaderEngine.read(resultSet, resultSetMetaData, i), true, isUniqueKey));
209         }
210         return result;
211     }
212     
213     private IngestPosition newPosition(final ResultSet resultSet) throws SQLException {
214         return dumperConfig.hasUniqueKey()
215                 ? PrimaryKeyPositionFactory.newInstance(resultSet.getObject(dumperConfig.getUniqueKeyColumns().get(0).getName()), ((PrimaryKeyPosition<?>) dumperConfig.getPosition()).getEndValue())
216                 : new PlaceholderPosition();
217     }
218     
219     @Override
220     protected void doStop() {
221         PipelineJdbcUtils.cancelStatement(dumpStatement.get());
222     }
