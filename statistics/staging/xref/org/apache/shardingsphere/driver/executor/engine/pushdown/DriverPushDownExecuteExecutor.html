44  /**
45   * Driver push down execute executor.
46   */
47  public final class DriverPushDownExecuteExecutor {
48      
49      private final DriverJDBCPushDownExecuteExecutor jdbcPushDownExecutor;
50      
51      private final DriverRawPushDownExecuteExecutor rawPushDownExecutor;
52      
53      public DriverPushDownExecuteExecutor(final ShardingSphereConnection connection, final ShardingSphereMetaData metaData, final Grantee grantee,
54                                           final JDBCExecutor jdbcExecutor, final RawExecutor rawExecutor) {
55          jdbcPushDownExecutor = new DriverJDBCPushDownExecuteExecutor(connection, metaData, grantee, jdbcExecutor);
56          rawPushDownExecutor = new DriverRawPushDownExecuteExecutor(connection, metaData, grantee, rawExecutor);
57      }
58      
59      /**
60       * Execute.
61       * 
62       * @param database database
63       * @param executionContext execution context
64       * @param prepareEngine prepare engine
65       * @param executeCallback statement execute callback
66       * @param addCallback statement add callback
67       * @param replayCallback statement replay callback
68       * @return is query
69       * @throws SQLException SQL exception
70       */
71      @SuppressWarnings("rawtypes")
72      public boolean execute(final ShardingSphereDatabase database, final ExecutionContext executionContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
73                             final StatementExecuteCallback executeCallback, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
74          return database.getRuleMetaData().getAttributes(RawExecutionRuleAttribute.class).isEmpty()
75                  ? jdbcPushDownExecutor.execute(database, executionContext, prepareEngine, executeCallback, addCallback, replayCallback)
76                  : rawPushDownExecutor.execute(database, executionContext);
77      }
78      
79      /**
80       * Get result set.
81       *
82       * @param database database
83       * @param sqlStatementContext SQL statement context
84       * @param statement statement
85       * @param statements statements
86       * @return result set
87       * @throws SQLException SQL exception
88       */
89      public Optional<ResultSet> getResultSet(final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext,
90                                              final Statement statement, final List<? extends Statement> statements) throws SQLException {
91          return database.getRuleMetaData().getAttributes(RawExecutionRuleAttribute.class).isEmpty()
92                  ? jdbcPushDownExecutor.getResultSet(database, sqlStatementContext, statement, statements)
93                  : Optional.empty();
94      }
95  }
