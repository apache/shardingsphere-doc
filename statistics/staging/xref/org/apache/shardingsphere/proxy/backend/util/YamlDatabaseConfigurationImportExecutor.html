64  /**
65   * Yaml database configuration import executor.
66   */
67  public final class YamlDatabaseConfigurationImportExecutor {
68      
69      private final YamlProxyDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlProxyDataSourceConfigurationSwapper();
70      
71      private final DistSQLDataSourcePoolPropertiesValidator validateHandler = new DistSQLDataSourcePoolPropertiesValidator();
72      
73      /**
74       * Import proxy database from yaml configuration.
75       *
76       * @param yamlConfig yaml proxy database configuration
77       */
78      public void importDatabaseConfiguration(final YamlProxyDatabaseConfiguration yamlConfig) {
79          String databaseName = yamlConfig.getDatabaseName();
80          checkDatabase(databaseName);
81          addDatabase(databaseName);
82          if (null == yamlConfig.getDataSources() || yamlConfig.getDataSources().isEmpty()) {
83              return;
84          }
85          importDataSources(databaseName, yamlConfig.getDataSources());
86          try {
87              importRules(databaseName, yamlConfig.getRules());
88          } catch (final ShardingSphereSQLException ex) {
89              dropDatabase(databaseName);
90              throw ex;
91          }
92      }
93      
94      private void checkDatabase(final String databaseName) {
95          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(databaseName), MissingRequiredDatabaseException::new);
96          ShardingSpherePreconditions.checkState(!ProxyContext.getInstance().databaseExists(databaseName), () -> new DatabaseCreateExistsException(databaseName));
97      }
98      
99      private void addDatabase(final String databaseName) {
100         ContextManager contextManager = ProxyContext.getInstance().getContextManager();
101         contextManager.getInstanceContext().getModeContextManager().createDatabase(databaseName);
102         DatabaseType protocolType = DatabaseTypeEngine.getProtocolType(Collections.emptyMap(), contextManager.getMetaDataContexts().getMetaData().getProps());
103         contextManager.getMetaDataContexts().getMetaData().addDatabase(databaseName, protocolType, contextManager.getMetaDataContexts().getMetaData().getProps());
104     }
105     
106     private void importDataSources(final String databaseName, final Map<String, YamlProxyDataSourceConfiguration> yamlDataSourceMap) {
107         Map<String, DataSourcePoolProperties> propsMap = new LinkedHashMap<>(yamlDataSourceMap.size(), 1F);
108         for (Entry<String, YamlProxyDataSourceConfiguration> entry : yamlDataSourceMap.entrySet()) {
109             DataSourceConfiguration dataSourceConfig = dataSourceConfigSwapper.swap(entry.getValue());
110             propsMap.put(entry.getKey(), DataSourcePoolPropertiesCreator.create(dataSourceConfig));
111         }
112         validateHandler.validate(propsMap);
113         try {
114             ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().registerStorageUnits(databaseName, propsMap);
115         } catch (final SQLException ex) {
116             throw new StorageUnitsOperateException("import", propsMap.keySet(), ex);
117         }
118         Map<String, StorageUnit> storageUnits = ProxyContext.getInstance().getContextManager()
119                 .getMetaDataContexts().getMetaData().getDatabase(databaseName).getResourceMetaData().getStorageUnits();
120         Map<String, StorageNode> toBeAddedStorageNode = StorageUnitNodeMapCreator.create(propsMap);
121         for (Entry<String, DataSourcePoolProperties> entry : propsMap.entrySet()) {
122             storageUnits.put(entry.getKey(), new StorageUnit(toBeAddedStorageNode.get(entry.getKey()), entry.getValue(), DataSourcePoolCreator.create(entry.getValue())));
123         }
124     }
125     
126     private void importRules(final String databaseName, final Collection<YamlRuleConfiguration> yamlRuleConfigs) {
127         if (null == yamlRuleConfigs || yamlRuleConfigs.isEmpty()) {
128             return;
129         }
130         Collection<RuleConfiguration> ruleConfigs = new LinkedList<>();
131         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
132         ShardingSphereDatabase database = metaDataContexts.getMetaData().getDatabase(databaseName);
133         swapToRuleConfigs(yamlRuleConfigs).values().forEach(each -> addRule(ruleConfigs, each, database));
134         metaDataContexts.getPersistService().getDatabaseRulePersistService().persist(metaDataContexts.getMetaData().getDatabase(databaseName).getName(), ruleConfigs);
135     }
136     
137     private void addRule(final Collection<RuleConfiguration> ruleConfigs, final RuleConfiguration ruleConfig, final ShardingSphereDatabase database) {
138         RuleConfigurationCheckEngine.check(ruleConfig, database);
139         ruleConfigs.add(ruleConfig);
140         database.getRuleMetaData().getRules().add(buildRule(ruleConfig, database));
141     }
142     
143     @SuppressWarnings({"unchecked", "rawtypes"})
144     private ShardingSphereRule buildRule(final RuleConfiguration ruleConfig, final ShardingSphereDatabase database) {
145         DatabaseRuleBuilder ruleBuilder = OrderedSPILoader.getServices(DatabaseRuleBuilder.class, Collections.singleton(ruleConfig)).get(ruleConfig);
146         Map<String, DataSource> dataSources = database.getResourceMetaData().getStorageUnits().entrySet().stream()
147                 .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new));
148         InstanceContext instanceContext = ProxyContext.getInstance().getContextManager().getInstanceContext();
149         return ruleBuilder.build(ruleConfig, database.getName(), database.getProtocolType(), dataSources, database.getRuleMetaData().getRules(), instanceContext);
150     }
151     
152     @SuppressWarnings({"rawtypes", "unchecked"})
153     private Map<Integer, RuleConfiguration> swapToRuleConfigs(final Collection<YamlRuleConfiguration> yamlRuleConfigs) {
154         Map<Integer, RuleConfiguration> result = new TreeMap<>(Comparator.reverseOrder());
155         for (YamlRuleConfiguration each : yamlRuleConfigs) {
156             YamlRuleConfigurationSwapper swapper = OrderedSPILoader.getServicesByClass(YamlRuleConfigurationSwapper.class, Collections.singleton(each.getRuleConfigurationType()))
157                     .get(each.getRuleConfigurationType());
158             result.put(swapper.getOrder(), (RuleConfiguration) swapper.swapToObject(each));
159         }
160         return result;
161     }
162     
163     private void dropDatabase(final String databaseName) {
164         ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().dropDatabase(databaseName);
165     }
