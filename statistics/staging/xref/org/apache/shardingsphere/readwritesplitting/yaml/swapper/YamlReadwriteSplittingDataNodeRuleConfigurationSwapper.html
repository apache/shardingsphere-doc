44  /**
45   * YAML readwrite-splitting data node rule configuration swapper.
46   */
47  public final class YamlReadwriteSplittingDataNodeRuleConfigurationSwapper implements YamlDataNodeRuleConfigurationSwapper<ReadwriteSplittingRuleConfiguration> {
48      
49      private final YamlAlgorithmConfigurationSwapper algorithmSwapper = new YamlAlgorithmConfigurationSwapper();
50      
51      private final RuleNodePath readwriteSplittingRuleNodePath = new ReadwriteSplittingRuleNodePathProvider().getRuleNodePath();
52      
53      @Override
54      public Collection<YamlDataNode> swapToDataNodes(final ReadwriteSplittingRuleConfiguration data) {
55          Collection<YamlDataNode> result = new LinkedList<>();
56          for (Entry<String, AlgorithmConfiguration> entry : data.getLoadBalancers().entrySet()) {
57              result.add(new YamlDataNode(readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.LOAD_BALANCERS).getPath(entry.getKey()),
58                      YamlEngine.marshal(algorithmSwapper.swapToYamlConfiguration(entry.getValue()))));
59          }
60          for (ReadwriteSplittingDataSourceGroupRuleConfiguration each : data.getDataSourceGroups()) {
61              result.add(new YamlDataNode(readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.DATA_SOURCES).getPath(each.getName()),
62                      YamlEngine.marshal(swapToYamlConfiguration(each))));
63          }
64          return result;
65      }
66      
67      private YamlReadwriteSplittingDataSourceGroupRuleConfiguration swapToYamlConfiguration(final ReadwriteSplittingDataSourceGroupRuleConfiguration dataSourceRuleGroupConfig) {
68          YamlReadwriteSplittingDataSourceGroupRuleConfiguration result = new YamlReadwriteSplittingDataSourceGroupRuleConfiguration();
69          result.setWriteDataSourceName(dataSourceRuleGroupConfig.getWriteDataSourceName());
70          result.setReadDataSourceNames(dataSourceRuleGroupConfig.getReadDataSourceNames());
71          result.setTransactionalReadQueryStrategy(dataSourceRuleGroupConfig.getTransactionalReadQueryStrategy().name());
72          result.setLoadBalancerName(dataSourceRuleGroupConfig.getLoadBalancerName());
73          return result;
74      }
75      
76      @Override
77      public Optional<ReadwriteSplittingRuleConfiguration> swapToObject(final Collection<YamlDataNode> dataNodes) {
78          List<YamlDataNode> validDataNodes = dataNodes.stream().filter(each -> readwriteSplittingRuleNodePath.getRoot().isValidatedPath(each.getKey())).collect(Collectors.toList());
79          if (validDataNodes.isEmpty()) {
80              return Optional.empty();
81          }
82          Collection<ReadwriteSplittingDataSourceGroupRuleConfiguration> dataSourceGroups = new LinkedList<>();
83          Map<String, AlgorithmConfiguration> loadBalancerMap = new LinkedHashMap<>();
84          for (YamlDataNode each : validDataNodes) {
85              readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.DATA_SOURCES).getName(each.getKey())
86                      .ifPresent(optional -> dataSourceGroups.add(swapDataSourceGroup(optional, YamlEngine.unmarshal(each.getValue(), YamlReadwriteSplittingDataSourceGroupRuleConfiguration.class))));
87              readwriteSplittingRuleNodePath.getNamedItem(ReadwriteSplittingRuleNodePathProvider.LOAD_BALANCERS).getName(each.getKey())
88                      .ifPresent(optional -> loadBalancerMap.put(optional, algorithmSwapper.swapToObject(YamlEngine.unmarshal(each.getValue(), YamlAlgorithmConfiguration.class))));
89          }
90          return Optional.of(new ReadwriteSplittingRuleConfiguration(dataSourceGroups, loadBalancerMap));
91      }
92      
93      private ReadwriteSplittingDataSourceGroupRuleConfiguration swapDataSourceGroup(final String name, final YamlReadwriteSplittingDataSourceGroupRuleConfiguration yamlDataSourceRuleConfig) {
94          return new ReadwriteSplittingDataSourceGroupRuleConfiguration(name, yamlDataSourceRuleConfig.getWriteDataSourceName(), yamlDataSourceRuleConfig.getReadDataSourceNames(),
95                  getTransactionalReadQueryStrategy(yamlDataSourceRuleConfig), yamlDataSourceRuleConfig.getLoadBalancerName());
96      }
97      
98      private TransactionalReadQueryStrategy getTransactionalReadQueryStrategy(final YamlReadwriteSplittingDataSourceGroupRuleConfiguration yamlDataSourceGroupRuleConfig) {
99          return Strings.isNullOrEmpty(yamlDataSourceGroupRuleConfig.getTransactionalReadQueryStrategy())
100                 ? TransactionalReadQueryStrategy.DYNAMIC
101                 : TransactionalReadQueryStrategy.valueOf(yamlDataSourceGroupRuleConfig.getTransactionalReadQueryStrategy());
102     }
103     
104     @Override
105     public Class<ReadwriteSplittingRuleConfiguration> getTypeClass() {
106         return ReadwriteSplittingRuleConfiguration.class;
107     }
108     
109     @Override
110     public String getRuleTagName() {
111         return "READWRITE_SPLITTING";
112     }
113     
114     @Override
115     public int getOrder() {
116         return ReadwriteSplittingOrder.ORDER;
117     }
118 }
