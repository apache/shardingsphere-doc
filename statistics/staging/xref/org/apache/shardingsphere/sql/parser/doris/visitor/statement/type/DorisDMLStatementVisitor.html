57  /**
58   * DML statement visitor for Doris.
59   */
60  public final class DorisDMLStatementVisitor extends DorisStatementVisitor implements DMLStatementVisitor {
61      
62      @Override
63      public ASTNode visitCall(final CallContext ctx) {
64          List<ExpressionSegment> params = new ArrayList<>(ctx.expr().size());
65          ctx.expr().forEach(each -> params.add((ExpressionSegment) visit(each)));
66          String procedureName = ctx.identifier().getText();
67          if (null != ctx.owner()) {
68              procedureName = ctx.owner().getText() + "." + procedureName;
69          }
70          return new DorisCallStatement(procedureName, params);
71      }
72      
73      @Override
74      public ASTNode visitDoStatement(final DoStatementContext ctx) {
75          List<ExpressionSegment> params = new ArrayList<>(ctx.expr().size());
76          ctx.expr().forEach(each -> params.add((ExpressionSegment) visit(each)));
77          return new DorisDoStatement(params);
78      }
79      
80      @Override
81      public ASTNode visitHandlerStatement(final HandlerStatementContext ctx) {
82          return new DorisHandlerStatement();
83      }
84      
85      @Override
86      public ASTNode visitImportStatement(final ImportStatementContext ctx) {
87          return new DorisImportStatement();
88      }
89      
90      @Override
91      public ASTNode visitLoadStatement(final LoadStatementContext ctx) {
92          return null == ctx.loadDataStatement() ? visit(ctx.loadXmlStatement()) : visit(ctx.loadDataStatement());
93      }
94      
95      @Override
96      public ASTNode visitLoadDataStatement(final LoadDataStatementContext ctx) {
97          return new DorisLoadDataStatement((SimpleTableSegment) visit(ctx.tableName()));
98      }
99      
100     @Override
101     public ASTNode visitLoadXmlStatement(final LoadXmlStatementContext ctx) {
102         return new DorisLoadXMLStatement((SimpleTableSegment) visit(ctx.tableName()));
103     }
104     
105     @Override
106     public ASTNode visitIndexHint(final IndexHintContext ctx) {
107         Collection<String> indexNames = new LinkedList<>();
108         if (null != ctx.indexNameList()) {
109             ctx.indexNameList().indexName().forEach(each -> indexNames.add(each.getText()));
110         }
111         String useType;
112         if (null != ctx.USE()) {
113             useType = ctx.USE().getText();
114         } else if (null != ctx.IGNORE()) {
115             useType = ctx.IGNORE().getText();
116         } else {
117             useType = ctx.FORCE().getText();
118         }
119         IndexHintSegment result = new IndexHintSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), indexNames, useType,
120                 null == ctx.INDEX() ? ctx.KEY().getText() : ctx.INDEX().getText(), getOriginalText(ctx));
121         if (null != ctx.indexHintClause().FOR()) {
122             String hintScope;
123             if (null != ctx.indexHintClause().JOIN()) {
124                 hintScope = "JOIN";
125             } else if (null != ctx.indexHintClause().ORDER()) {
126                 hintScope = "ORDER BY";
127             } else {
128                 hintScope = "GROUP BY";
129             }
130             result.setHintScope(hintScope);
131         }
132         return result;
133     }
134     
135     @Override
136     public ASTNode visitWindowClause(final WindowClauseContext ctx) {
137         WindowSegment result = new WindowSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
138         for (WindowItemContext each : ctx.windowItem()) {
139             result.getItemSegments().add((WindowItemSegment) visit(each));
140         }
141         return result;
142     }
143     
144     @Override
145     public ASTNode visitWindowItem(final WindowItemContext ctx) {
146         WindowItemSegment result = new WindowItemSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
147         result.setWindowName(new IdentifierValue(ctx.identifier().getText()));
148         WindowItemSegment windowItemSegment = (WindowItemSegment) visit(ctx.windowSpecification());
149         result.setPartitionListSegments(windowItemSegment.getPartitionListSegments());
150         result.setOrderBySegment(windowItemSegment.getOrderBySegment());
151         result.setFrameClause(windowItemSegment.getFrameClause());
152         return result;
153     }
154     
155     @Override
156     public ASTNode visitWindowSpecification(final WindowSpecificationContext ctx) {
157         WindowItemSegment result = new WindowItemSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex());
158         if (null != ctx.PARTITION()) {
159             result.setPartitionListSegments(getExpressions(ctx.expr()));
160         }
161         if (null != ctx.orderByClause()) {
162             result.setOrderBySegment((OrderBySegment) visit(ctx.orderByClause()));
163         }
164         if (null != ctx.frameClause()) {
165             result.setFrameClause(new CommonExpressionSegment(ctx.frameClause().start.getStartIndex(), ctx.frameClause().stop.getStopIndex(), ctx.frameClause().getText()));
166         }
167         if (null != ctx.identifier()) {
168             result.setWindowName(new IdentifierValue(ctx.identifier().getText()));
169         }
170         return result;
171     }
172     
173     @Override
174     public ASTNode visitWindowFunction(final WindowFunctionContext ctx) {
175         super.visitWindowFunction(ctx);
176         FunctionSegment result = new FunctionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), ctx.funcName.getText(), getOriginalText(ctx));
177         if (null != ctx.NTILE()) {
178             result.getParameters().add((ExpressionSegment) visit(ctx.simpleExpr()));
179         }
180         if (null != ctx.LEAD() || null != ctx.LAG() || null != ctx.FIRST_VALUE() || null != ctx.LAST_VALUE()) {
181             result.getParameters().add((ExpressionSegment) visit(ctx.expr()));
182         }
183         if (null != ctx.NTH_VALUE()) {
184             result.getParameters().add((ExpressionSegment) visit(ctx.expr()));
185             result.getParameters().add((ExpressionSegment) visit(ctx.simpleExpr()));
186         }
187         result.setWindow((WindowItemSegment) visit(ctx.windowingClause()));
188         return result;
189     }
190     
191     @Override
192     public ASTNode visitWindowingClause(final WindowingClauseContext ctx) {
193         WindowItemSegment result = new WindowItemSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
194         if (null != ctx.windowName) {
195             result.setWindowName((IdentifierValue) visit(ctx.windowName));
196         }
197         if (null != ctx.windowSpecification()) {
198             WindowItemSegment windowItemSegment = (WindowItemSegment) visit(ctx.windowSpecification());
199             result.setPartitionListSegments(windowItemSegment.getPartitionListSegments());
200             result.setOrderBySegment(windowItemSegment.getOrderBySegment());
201             result.setFrameClause(windowItemSegment.getFrameClause());
202         }
203         return result;
204     }
