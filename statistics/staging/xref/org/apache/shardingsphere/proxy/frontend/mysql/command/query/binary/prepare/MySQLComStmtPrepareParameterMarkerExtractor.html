41  /**
42   * Parameter marker extractor for MySQL COM_STMT_PREPARE.
43   */
44  @NoArgsConstructor(access = AccessLevel.PRIVATE)
45  public final class MySQLComStmtPrepareParameterMarkerExtractor {
46      
47      /**
48       * TODO Support more statements and syntax.
49       * Find corresponding columns of parameter markers.
50       *
51       * @param sqlStatement SQL statement
52       * @param schema schema
53       * @return map parameter marker segment to column
54       */
55      public static Map<ParameterMarkerSegment, ShardingSphereColumn> findColumnsOfParameterMarkers(final SQLStatement sqlStatement, final ShardingSphereSchema schema) {
56          return sqlStatement instanceof InsertStatement ? findColumnsOfParameterMarkersForInsert((InsertStatement) sqlStatement, schema) : Collections.emptyMap();
57      }
58      
59      private static Map<ParameterMarkerSegment, ShardingSphereColumn> findColumnsOfParameterMarkersForInsert(final InsertStatement insertStatement, final ShardingSphereSchema schema) {
60          ShardingSphereTable table = schema.getTable(insertStatement.getTable().getTableName().getIdentifier().getValue());
61          List<String> columnNamesOfInsert = getColumnNamesOfInsertStatement(insertStatement, table);
62          Map<String, ShardingSphereColumn> columnsOfTable = table.getColumns();
63          Map<String, ShardingSphereColumn> caseInsensitiveColumnsOfTable = convertToCaseInsensitiveColumnsOfTable(columnsOfTable);
64          Map<ParameterMarkerSegment, ShardingSphereColumn> result = new LinkedHashMap<>(insertStatement.getParameterMarkerSegments().size(), 1);
65          for (InsertValuesSegment each : insertStatement.getValues()) {
66              ListIterator<ExpressionSegment> listIterator = each.getValues().listIterator();
67              for (int columnIndex = listIterator.nextIndex(); listIterator.hasNext(); columnIndex = listIterator.nextIndex()) {
68                  ExpressionSegment value = listIterator.next();
69                  if (!(value instanceof ParameterMarkerExpressionSegment)) {
70                      continue;
71                  }
72                  String columnName = columnNamesOfInsert.get(columnIndex);
73                  ShardingSphereColumn column = columnsOfTable.getOrDefault(columnName, caseInsensitiveColumnsOfTable.get(columnName));
74                  if (null != column) {
75                      result.put((ParameterMarkerSegment) value, column);
76                  }
77              }
78          }
79          return result;
80      }
81      
82      private static List<String> getColumnNamesOfInsertStatement(final InsertStatement insertStatement, final ShardingSphereTable table) {
83          return insertStatement.getColumns().isEmpty() ? new ArrayList<>(table.getColumns().keySet())
84                  : insertStatement.getColumns().stream().map(each -> each.getIdentifier().getValue()).collect(Collectors.toList());
85      }
86      
87      private static Map<String, ShardingSphereColumn> convertToCaseInsensitiveColumnsOfTable(final Map<String, ShardingSphereColumn> columns) {
88          Map<String, ShardingSphereColumn> result = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
89          result.putAll(columns);
90          return result;
91      }
92  }
