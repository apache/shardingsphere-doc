84      public ShardingSphereConnection(final String currentDatabaseName, final ContextManager contextManager, final String executorType) {
85          this.currentDatabaseName = currentDatabaseName;
86          this.contextManager = contextManager;
87          databaseConnectionManager = new DriverDatabaseConnectionManager(currentDatabaseName, contextManager);
88          processId = processEngine.connect(currentDatabaseName);
89          this.executorType = executorType;
90      }

92      /**
93       * Begin transaction if needed when auto commit is false.
94       *
95       * @throws SQLException SQL exception
96       */
97      public void beginTransactionIfNeededWhenAutoCommitFalse() throws SQLException {
98          if (autoCommit || databaseConnectionManager.getConnectionContext().getTransactionContext().isInTransaction()) {
99              return;
100         }
101         databaseConnectionManager.begin();
102     }

154     @Override
155     public boolean getAutoCommit() {
156         return autoCommit;
157     }
158     
159     @Override
160     public void setAutoCommit(final boolean autoCommit) throws SQLException {
161         this.autoCommit = autoCommit;
162         if (databaseConnectionManager.getConnectionTransaction().isLocalTransaction()) {
163             processLocalTransaction();
164         } else {
165             processDistributedTransaction();
166         }
167     }
168     
169     private void processLocalTransaction() throws SQLException {
170         databaseConnectionManager.setAutoCommit(autoCommit);
171         TransactionConnectionContext transactionContext = databaseConnectionManager.getConnectionContext().getTransactionContext();
172         if (autoCommit && transactionContext.isInTransaction()) {
173             transactionContext.close();
174             return;
175         }
176         if (!autoCommit && !transactionContext.isInTransaction()) {
177             transactionContext.beginTransaction(String.valueOf(contextManager.getMetaDataContexts().getMetaData().getGlobalRuleMetaData().getSingleRule(TransactionRule.class).getDefaultType()));
178         }
179     }
180     
181     private void processDistributedTransaction() throws SQLException {
182         Optional<DistributedTransactionOperationType> operationType = databaseConnectionManager.getConnectionTransaction().getDistributedTransactionOperationType(autoCommit);
183         if (!operationType.isPresent()) {
184             return;
185         }
186         if (DistributedTransactionOperationType.BEGIN == operationType.get()) {
187             databaseConnectionManager.begin();
188         } else {
189             databaseConnectionManager.commit();
190         }
191     }

203     @Override
204     public void rollback(final Savepoint savepoint) throws SQLException {
205         checkClose();
206         ShardingSpherePreconditions.checkState(databaseConnectionManager.getConnectionTransaction().isHoldTransaction(autoCommit) || !isSchemaSupportedDatabaseType(),
207                 () -> new SQLFeatureNotSupportedException("ROLLBACK TO SAVEPOINT can only be used in transaction blocks"));
208         databaseConnectionManager.rollback(savepoint);
209     }
210     
211     @Override
212     public Savepoint setSavepoint(final String name) throws SQLException {
213         checkClose();
214         ShardingSpherePreconditions.checkState(databaseConnectionManager.getConnectionTransaction().isHoldTransaction(autoCommit) || !isSchemaSupportedDatabaseType(),
215                 () -> new SQLFeatureNotSupportedException("Savepoint can only be used in transaction blocks"));
216         return databaseConnectionManager.setSavepoint(name);
217     }
218     
219     @Override
220     public Savepoint setSavepoint() throws SQLException {
221         checkClose();
222         ShardingSpherePreconditions.checkState(databaseConnectionManager.getConnectionTransaction().isHoldTransaction(autoCommit) || !isSchemaSupportedDatabaseType(),
223                 () -> new SQLFeatureNotSupportedException("Savepoint can only be used in transaction blocks"));
224         return databaseConnectionManager.setSavepoint();
225     }
226     
227     @Override
228     public void releaseSavepoint(final Savepoint savepoint) throws SQLException {
229         checkClose();
230         ShardingSpherePreconditions.checkState(databaseConnectionManager.getConnectionTransaction().isHoldTransaction(autoCommit) || !isSchemaSupportedDatabaseType(),
231                 () -> new SQLFeatureNotSupportedException("RELEASE SAVEPOINT can only be used in transaction blocks"));
232         if (databaseConnectionManager.getConnectionTransaction().isHoldTransaction(autoCommit)) {
233             databaseConnectionManager.releaseSavepoint(savepoint);
234         }
235     }
236     
237     private void checkClose() throws SQLException {
238         ShardingSpherePreconditions.checkState(!isClosed(), () -> new ConnectionClosedException().toSQLException());
239     }
240     
241     private boolean isSchemaSupportedDatabaseType() {
242         DatabaseType databaseType = contextManager.getMetaDataContexts().getMetaData().getDatabase(currentDatabaseName).getProtocolType();
243         return new DatabaseTypeRegistry(databaseType).getDialectDatabaseMetaData().getDefaultSchema().isPresent();
244     }

298     @Override
299     public boolean isClosed() {
300         return closed;
301     }
302     
303     @Override
304     public void close() throws SQLException {
305         if (databaseConnectionManager.getConnectionTransaction().isInDistributedTransaction(databaseConnectionManager.getConnectionContext().getTransactionContext())) {
306             databaseConnectionManager.getConnectionTransaction().rollback();
307         }
308         closed = true;
309         processEngine.disconnect(processId);
310         try {
311             forceExecuteTemplate.execute(statementManagers, StatementManager::close);
312         } finally {
313             statementManagers.clear();
314             databaseConnectionManager.close();
315         }
316     }
