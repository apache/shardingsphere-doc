697      * @return visitor rule
698      * @throws SQLASTVisitorException SQL AST visitor exception
699      */
700     public static SQLVisitorRule valueOf(final Class<? extends ParseTree> parseTreeClass) {
701         String parseTreeClassName = parseTreeClass.getSimpleName();
702         for (SQLVisitorRule each : values()) {
703             if (each.getContextName().equals(parseTreeClassName)) {
704                 return each;
705             }
706         }
