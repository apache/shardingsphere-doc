51  /**
52   * ShardingSphere statistics refresh engine.
53   */
54  @RequiredArgsConstructor
55  @Slf4j
56  public final class ShardingSphereStatisticsRefreshEngine {
57      
58      private static final ExecutorService EXECUTOR_SERVICE = Executors.newSingleThreadScheduledExecutor(ExecutorThreadFactoryBuilder.build("statistics-collect-%d"));
59      
60      private final ContextManager contextManager;
61      
62      private final GlobalLockContext globalLockContext;
63      
64      /**
65       * Async refresh.
66       */
67      public void asyncRefresh() {
68          EXECUTOR_SERVICE.execute(this::refresh);
69      }
70      
71      /**
72       * Refresh.
73       */
74      public void refresh() {
75          try {
76              if (contextManager.getMetaDataContexts().getMetaData().getTemporaryProps().getValue(TemporaryConfigurationPropertyKey.PROXY_META_DATA_COLLECTOR_ENABLED)) {
77                  collectAndRefresh();
78              }
79              // CHECKSTYLE:OFF
80          } catch (final Exception ex) {
81              // CHECKSTYLE:ON
82              log.error("Collect data error", ex);
83          }
84      }
85      
86      private void collectAndRefresh() {
87          GlobalLockDefinition lockDefinition = new GlobalLockDefinition(GlobalLockNames.STATISTICS.getLockName());
88          if (globalLockContext.tryLock(lockDefinition, 5000L)) {
89              try {
90                  ShardingSphereStatistics statistics = contextManager.getMetaDataContexts().getStatistics();
91                  ShardingSphereMetaData metaData = contextManager.getMetaDataContexts().getMetaData();
92                  ShardingSphereStatistics changedStatistics = new ShardingSphereStatistics();
93                  for (Entry<String, ShardingSphereDatabaseData> entry : statistics.getDatabaseData().entrySet()) {
94                      if (metaData.containsDatabase(entry.getKey())) {
95                          collectForDatabase(entry.getKey(), entry.getValue(), metaData, changedStatistics);
96                      }
97                  }
98                  compareAndUpdate(changedStatistics);
99              } finally {
100                 globalLockContext.unlock(lockDefinition);
101             }
102         }
103     }
104     
105     private void collectForDatabase(final String databaseName, final ShardingSphereDatabaseData databaseData,
106                                     final ShardingSphereMetaData metaData, final ShardingSphereStatistics statistics) {
107         for (Entry<String, ShardingSphereSchemaData> entry : databaseData.getSchemaData().entrySet()) {
108             if (metaData.getDatabase(databaseName).containsSchema(entry.getKey())) {
109                 collectForSchema(databaseName, entry.getKey(), entry.getValue(), metaData, statistics);
110             }
111         }
112     }
113     
114     private void collectForSchema(final String databaseName, final String schemaName, final ShardingSphereSchemaData schemaData,
115                                   final ShardingSphereMetaData metaData, final ShardingSphereStatistics statistics) {
116         for (Entry<String, ShardingSphereTableData> entry : schemaData.getTableData().entrySet()) {
117             if (metaData.getDatabase(databaseName).getSchema(schemaName).containsTable(entry.getKey())) {
118                 collectForTable(databaseName, schemaName, metaData.getDatabase(databaseName).getSchema(schemaName).getTable(entry.getKey()), metaData, statistics);
119             }
120         }
121     }
122     
123     private void collectForTable(final String databaseName, final String schemaName, final ShardingSphereTable table,
124                                  final ShardingSphereMetaData metaData, final ShardingSphereStatistics statistics) {
125         Optional<ShardingSphereStatisticsCollector> dataCollector = TypedSPILoader.findService(ShardingSphereStatisticsCollector.class, table.getName());
126         Optional<ShardingSphereTableData> tableData = Optional.empty();
127         if (dataCollector.isPresent()) {
128             try {
129                 tableData = dataCollector.get().collect(databaseName, table, metaData.getDatabases(), contextManager.getMetaDataContexts().getMetaData().getGlobalRuleMetaData());
130                 // CHECKSTYLE:OFF
131             } catch (final Exception ex) {
132                 // CHECKSTYLE:ON
133                 log.error(String.format("Collect %s.%s.%s data failed", databaseName, schemaName, table.getName()), ex);
134             }
135         }
136         ShardingSphereDatabaseData databaseData = statistics.containsDatabase(databaseName) ? statistics.getDatabase(databaseName) : new ShardingSphereDatabaseData();
137         ShardingSphereSchemaData schemaData = databaseData.containsSchema(schemaName) ? databaseData.getSchema(schemaName) : new ShardingSphereSchemaData();
138         tableData.ifPresent(optional -> schemaData.putTable(table.getName(), optional));
139         databaseData.putSchema(schemaName, schemaData);
140         statistics.putDatabase(databaseName, databaseData);
141     }
142     
143     private void compareAndUpdate(final ShardingSphereStatistics changedStatistics) {
144         ShardingSphereMetaData metaData = contextManager.getMetaDataContexts().getMetaData();
145         ShardingSphereStatistics statistics = contextManager.getMetaDataContexts().getStatistics();
146         for (Entry<String, ShardingSphereDatabaseData> entry : changedStatistics.getDatabaseData().entrySet()) {
147             compareAndUpdateForDatabase(entry.getKey(), statistics.getDatabase(entry.getKey()), entry.getValue(), statistics, metaData.getDatabase(entry.getKey()));
148         }
149         for (Entry<String, ShardingSphereDatabaseData> entry : statistics.getDatabaseData().entrySet()) {
150             if (!changedStatistics.containsDatabase(entry.getKey())) {
151                 statistics.dropDatabase(entry.getKey());
152                 contextManager.getPersistServiceFacade().getMetaDataPersistService().getShardingSphereDataPersistService().delete(entry.getKey());
153             }
154         }
155     }
156     
157     private void compareAndUpdateForDatabase(final String databaseName, final ShardingSphereDatabaseData databaseData, final ShardingSphereDatabaseData changedDatabaseData,
158                                              final ShardingSphereStatistics statistics, final ShardingSphereDatabase database) {
159         for (Entry<String, ShardingSphereSchemaData> entry : changedDatabaseData.getSchemaData().entrySet()) {
160             compareAndUpdateForSchema(databaseName, entry.getKey(), databaseData.getSchema(entry.getKey()), entry.getValue(), statistics, database.getSchema(entry.getKey()));
161         }
162     }
163     
164     private void compareAndUpdateForSchema(final String databaseName, final String schemaName, final ShardingSphereSchemaData schemaData,
165                                            final ShardingSphereSchemaData changedSchemaData, final ShardingSphereStatistics statistics, final ShardingSphereSchema schema) {
166         for (Entry<String, ShardingSphereTableData> entry : changedSchemaData.getTableData().entrySet()) {
167             compareAndUpdateForTable(databaseName, schemaName, schemaData.getTable(entry.getKey()), entry.getValue(), statistics, schema.getTable(entry.getKey()));
168         }
169     }
170     
171     private void compareAndUpdateForTable(final String databaseName, final String schemaName, final ShardingSphereTableData tableData,
172                                           final ShardingSphereTableData changedTableData, final ShardingSphereStatistics statistics, final ShardingSphereTable table) {
173         if (!tableData.equals(changedTableData)) {
174             statistics.getDatabase(databaseName).getSchema(schemaName).putTable(changedTableData.getName(), changedTableData);
175             AlteredShardingSphereDatabaseData alteredShardingSphereDatabaseData = createAlteredShardingSphereDatabaseData(databaseName, schemaName, tableData, changedTableData, table);
176             contextManager.getPersistServiceFacade().getMetaDataPersistService().getShardingSphereDataPersistService().update(alteredShardingSphereDatabaseData);
177         }
178     }
179     
180     private AlteredShardingSphereDatabaseData createAlteredShardingSphereDatabaseData(final String databaseName, final String schemaName, final ShardingSphereTableData tableData,
181                                                                                       final ShardingSphereTableData changedTableData, final ShardingSphereTable table) {
182         AlteredShardingSphereDatabaseData result = new AlteredShardingSphereDatabaseData(databaseName, schemaName, tableData.getName());
183         Map<String, ShardingSphereRowData> tableDataMap = tableData.getRows().stream().collect(Collectors.toMap(ShardingSphereRowData::getUniqueKey, Function.identity()));
184         Map<String, ShardingSphereRowData> changedTableDataMap = changedTableData.getRows().stream().collect(Collectors.toMap(ShardingSphereRowData::getUniqueKey, Function.identity()));
185         YamlShardingSphereRowDataSwapper swapper = new YamlShardingSphereRowDataSwapper(new ArrayList<>(table.getColumnValues()));
186         for (Entry<String, ShardingSphereRowData> entry : changedTableDataMap.entrySet()) {
187             if (!tableDataMap.containsKey(entry.getKey())) {
188                 result.getAddedRows().add(swapper.swapToYamlConfiguration(entry.getValue()));
189             } else if (!tableDataMap.get(entry.getKey()).equals(entry.getValue())) {
190                 result.getUpdatedRows().add(swapper.swapToYamlConfiguration(entry.getValue()));
191             }
192         }
193         for (Entry<String, ShardingSphereRowData> entry : tableDataMap.entrySet()) {
194             if (!changedTableDataMap.containsKey(entry.getKey())) {
195                 result.getDeletedRows().add(swapper.swapToYamlConfiguration(entry.getValue()));
196             }
197         }
198         return result;
199     }
