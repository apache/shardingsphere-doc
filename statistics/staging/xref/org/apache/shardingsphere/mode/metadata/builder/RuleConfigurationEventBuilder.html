20  import com.google.common.base.Strings;
21  import org.apache.shardingsphere.mode.path.rule.RuleNodePath;
22  import org.apache.shardingsphere.mode.path.rule.item.NamedRuleItemNodePath;
23  import org.apache.shardingsphere.mode.path.rule.item.UniqueRuleItemNodePath;
24  import org.apache.shardingsphere.infra.rule.event.GovernanceEvent;
25  import org.apache.shardingsphere.infra.spi.ShardingSphereServiceLoader;
26  import org.apache.shardingsphere.mode.event.DataChangedEvent;
27  import org.apache.shardingsphere.mode.event.DataChangedEvent.Type;
28  import org.apache.shardingsphere.mode.event.NamedRuleItemChangedEventCreator;
29  import org.apache.shardingsphere.mode.event.UniqueRuleItemChangedEventCreator;
30  import org.apache.shardingsphere.mode.spi.RuleNodePathProvider;
31  
32  import java.util.Map.Entry;
33  import java.util.Optional;
34  
35  /**
36   * Rule configuration event builder.
37   */
38  public final class RuleConfigurationEventBuilder {
39      
40      /**
41       * Build rule changed event.
42       *
43       * @param databaseName database name
44       * @param event data changed event
45       * @return rule changed event
46       */
47      public Optional<GovernanceEvent> build(final String databaseName, final DataChangedEvent event) {
48          for (RuleNodePathProvider each : ShardingSphereServiceLoader.getServiceInstances(RuleNodePathProvider.class)) {
49              Optional<GovernanceEvent> result = build(each.getRuleNodePath(), databaseName, event);
50              if (result.isPresent()) {
51                  return result;
52              }
53          }
54          return Optional.empty();
55      }
56      
57      private Optional<GovernanceEvent> build(final RuleNodePath ruleNodePath, final String databaseName, final DataChangedEvent event) {
58          if (!ruleNodePath.getRoot().isValidatedPath(event.getKey()) || Type.DELETED != event.getType() && Strings.isNullOrEmpty(event.getValue())) {
59              return Optional.empty();
60          }
61          for (Entry<String, NamedRuleItemNodePath> entry : ruleNodePath.getNamedItems().entrySet()) {
62              Optional<String> itemName = entry.getValue().getNameByActiveVersion(event.getKey());
63              if (itemName.isPresent()) {
64                  return Optional.of(new NamedRuleItemChangedEventCreator().create(databaseName, itemName.get(), event, ruleNodePath.getRoot().getRuleType() + "." + entry.getKey()));
65              }
66          }
67          for (Entry<String, UniqueRuleItemNodePath> entry : ruleNodePath.getUniqueItems().entrySet()) {
68              if (entry.getValue().isActiveVersionPath(event.getKey())) {
69                  return Optional.of(new UniqueRuleItemChangedEventCreator().create(databaseName, event, ruleNodePath.getRoot().getRuleType() + "." + entry.getKey()));
70              }
71          }
72          return Optional.empty();
73      }
74  }
