42  class SQLRouteCountAdviceTest {
43      
44      private final MetricConfiguration config = new MetricConfiguration("routed_sql_total", MetricCollectorType.COUNTER, null, Collections.singletonList("type"), Collections.emptyMap());
45      
46      private final SQLRouteCountAdvice advice = new SQLRouteCountAdvice();
47      
48      @AfterEach
49      void reset() {
50          ((MetricsCollectorFixture) MetricsCollectorRegistry.get(config, "FIXTURE")).reset();
51      }
52      
53      @Test
54      void assertInsertRoute() {
55          QueryContext queryContext = new QueryContext(new CommonSQLStatementContext<>(new MySQLInsertStatement()), "", Collections.emptyList());
56          assertRoute(queryContext, "INSERT=1");
57      }
58      
59      @Test
60      void assertUpdateRoute() {
61          QueryContext queryContext = new QueryContext(new CommonSQLStatementContext<>(new MySQLUpdateStatement()), "", Collections.emptyList());
62          assertRoute(queryContext, "UPDATE=1");
63      }
64      
65      @Test
66      void assertDeleteRoute() {
67          QueryContext queryContext = new QueryContext(new CommonSQLStatementContext<>(new MySQLDeleteStatement()), "", Collections.emptyList());
68          assertRoute(queryContext, "DELETE=1");
69      }
70      
71      @Test
72      void assertSelectRoute() {
73          QueryContext queryContext = new QueryContext(new CommonSQLStatementContext<>(new MySQLSelectStatement()), "", Collections.emptyList());
74          assertRoute(queryContext, "SELECT=1");
75      }
76      
77      void assertRoute(final QueryContext queryContext, final String expected) {
78          advice.beforeMethod(new TargetAdviceObjectFixture(), mock(Method.class), new Object[]{new ConnectionContext(), queryContext}, "FIXTURE");
79          assertThat(MetricsCollectorRegistry.get(config, "FIXTURE").toString(), is(expected));
80      }
81  }
