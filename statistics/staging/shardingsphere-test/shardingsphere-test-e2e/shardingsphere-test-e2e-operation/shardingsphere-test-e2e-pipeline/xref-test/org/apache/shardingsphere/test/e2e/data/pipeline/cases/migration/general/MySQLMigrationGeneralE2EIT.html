48  @PipelineE2ESettings(database = @PipelineE2EDatabaseSettings(type = "MySQL", scenarioFiles = "env/scenario/general/mysql.xml"))
49  @Slf4j
50  class MySQLMigrationGeneralE2EIT extends AbstractMigrationE2EIT {
51      
52      private static final String SOURCE_TABLE_NAME = "t_order_copy";
53      
54      private static final String TARGET_TABLE_NAME = "t_order";
55      
56      @ParameterizedTest(name = "{0}")
57      @EnabledIf("isEnabled")
58      @ArgumentsSource(PipelineE2ETestCaseArgumentsProvider.class)
59      void assertMigrationSuccess(final PipelineTestParameter testParam) throws SQLException, InterruptedException {
60          try (PipelineContainerComposer containerComposer = new PipelineContainerComposer(testParam, new MigrationJobType())) {
61              addMigrationProcessConfig(containerComposer);
62              containerComposer.createSourceOrderTable(SOURCE_TABLE_NAME);
63              containerComposer.createSourceOrderItemTable();
64              addMigrationSourceResource(containerComposer);
65              addMigrationTargetResource(containerComposer);
66              createTargetOrderTableRule(containerComposer);
67              createTargetOrderTableEncryptRule(containerComposer);
68              createTargetOrderItemTableRule(containerComposer);
69              Pair<List<Object[]>, List<Object[]>> dataPair = PipelineCaseHelper.generateFullInsertData(containerComposer.getDatabaseType(), PipelineContainerComposer.TABLE_INIT_ROW_COUNT);
70              log.info("init data begin: {}", LocalDateTime.now());
71              DataSourceExecuteUtils.execute(containerComposer.getSourceDataSource(), containerComposer.getExtraSQLCommand().getFullInsertOrder(SOURCE_TABLE_NAME), dataPair.getLeft());
72              DataSourceExecuteUtils.execute(containerComposer.getSourceDataSource(), containerComposer.getExtraSQLCommand().getFullInsertOrderItem(), dataPair.getRight());
73              log.info("init data end: {}", LocalDateTime.now());
74              startMigration(containerComposer, SOURCE_TABLE_NAME, TARGET_TABLE_NAME);
75              startMigration(containerComposer, "t_order_item", "t_order_item");
76              String orderJobId = getJobIdByTableName(containerComposer, "ds_0." + SOURCE_TABLE_NAME);
77              containerComposer.waitJobPrepareSuccess(String.format("SHOW MIGRATION STATUS '%s'", orderJobId));
78              containerComposer.startIncrementTask(
79                      new E2EIncrementalTask(containerComposer.getSourceDataSource(), SOURCE_TABLE_NAME, new SnowflakeKeyGenerateAlgorithm(), containerComposer.getDatabaseType(), 30));
80              assertMigrationSuccessById(containerComposer, orderJobId, "DATA_MATCH");
81              String orderItemJobId = getJobIdByTableName(containerComposer, "ds_0.t_order_item");
82              assertMigrationSuccessById(containerComposer, orderItemJobId, "DATA_MATCH");
83              Thread.sleep(2000L);
84              assertMigrationSuccessById(containerComposer, orderItemJobId, "CRC32_MATCH");
85              for (String each : listJobId(containerComposer)) {
86                  commitMigrationByJobId(containerComposer, each);
87              }
88              List<String> lastJobIds = listJobId(containerComposer);
89              assertTrue(lastJobIds.isEmpty());
90              containerComposer.proxyExecuteWithLog("REFRESH TABLE METADATA", 2);
91              containerComposer.assertGreaterThanOrderTableInitRows(PipelineContainerComposer.TABLE_INIT_ROW_COUNT, "");
92          }
93      }
94      
95      private void assertMigrationSuccessById(final PipelineContainerComposer containerComposer, final String jobId, final String algorithmType) throws SQLException, InterruptedException {
96          List<Map<String, Object>> jobStatus = containerComposer.waitIncrementTaskFinished(String.format("SHOW MIGRATION STATUS '%s'", jobId));
97          for (Map<String, Object> each : jobStatus) {
98              assertTrue(Integer.parseInt(each.get("processed_records_count").toString()) > 0);
99              assertThat(Integer.parseInt(each.get("inventory_finished_percentage").toString()), is(100));
100         }
101         assertCheckMigrationSuccess(containerComposer, jobId, algorithmType);
102     }
103     
104     private static boolean isEnabled() {
105         return PipelineE2ECondition.isEnabled(new MySQLDatabaseType());
106     }
