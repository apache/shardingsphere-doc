18  package org.apache.shardingsphere.mode.manager.cluster.coordinator.registry.nodes.compute.online.subscriber;
19  
20  import com.google.common.eventbus.Subscribe;
21  import lombok.RequiredArgsConstructor;
22  import org.apache.shardingsphere.infra.instance.ComputeNodeInstance;
23  import org.apache.shardingsphere.infra.util.eventbus.EventSubscriber;
24  import org.apache.shardingsphere.mode.manager.ContextManager;
25  import org.apache.shardingsphere.mode.manager.cluster.coordinator.registry.nodes.compute.online.event.InstanceOfflineEvent;
26  import org.apache.shardingsphere.mode.manager.cluster.coordinator.registry.nodes.compute.online.event.InstanceOnlineEvent;
27  
28  /**
29   * Compute node online subscriber.
30   */
31  @RequiredArgsConstructor
32  public final class ComputeNodeOnlineSubscriber implements EventSubscriber {
33      
34      private final ContextManager contextManager;
35      
36      /**
37       * Renew instance list.
38       *
39       * @param event compute node online event
40       */
41      @Subscribe
42      public synchronized void renew(final InstanceOnlineEvent event) {
43          contextManager.getComputeNodeInstanceContext().addComputeNodeInstance(contextManager.getPersistServiceFacade()
44                  .getComputeNodePersistService().loadComputeNodeInstance(event.getInstanceMetaData()));
45      }
46      
47      /**
48       * Renew instance list.
49       *
50       * @param event compute node offline event
51       */
52      @Subscribe
53      public synchronized void renew(final InstanceOfflineEvent event) {
54          contextManager.getComputeNodeInstanceContext().deleteComputeNodeInstance(new ComputeNodeInstance(event.getInstanceMetaData()));
55      }
56  }
