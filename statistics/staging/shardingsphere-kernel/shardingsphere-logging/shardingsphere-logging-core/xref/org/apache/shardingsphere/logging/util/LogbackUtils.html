18  package org.apache.shardingsphere.logging.util;
19  
20  import ch.qos.logback.classic.Level;
21  import ch.qos.logback.classic.Logger;
22  import ch.qos.logback.classic.LoggerContext;
23  import ch.qos.logback.classic.encoder.PatternLayoutEncoder;
24  import ch.qos.logback.classic.spi.ILoggingEvent;
25  import ch.qos.logback.core.Appender;
26  import ch.qos.logback.core.FileAppender;
27  import ch.qos.logback.core.OutputStreamAppender;
28  import ch.qos.logback.core.util.DynamicClassLoadingException;
29  import ch.qos.logback.core.util.IncompatibleClassException;
30  import ch.qos.logback.core.util.OptionHelper;
31  import lombok.SneakyThrows;
32  import org.apache.shardingsphere.logging.config.LoggingRuleConfiguration;
33  import org.apache.shardingsphere.logging.logger.ShardingSphereAppender;
34  import org.apache.shardingsphere.logging.logger.ShardingSphereLogger;
35  import org.slf4j.LoggerFactory;
36  
37  import java.util.Collection;
38  import java.util.Iterator;
39  import java.util.Optional;
40  
41  /**
42   * Logback utility class.
43   */
44  public final class LogbackUtils {
45      
46      /**
47       * Refresh logger context with logging rule.
48       *
49       * @param loggingRuleConfiguration logging global rule configuration
50       */
51      public static void refreshLogger(final LoggingRuleConfiguration loggingRuleConfiguration) {
52          LoggerContext loggerContext = (LoggerContext) LoggerFactory.getILoggerFactory();
53          LogbackUtils.configLoggers(loggingRuleConfiguration, loggerContext);
54          LogbackUtils.startRootLogger(loggerContext);
55      }
56      
57      private static void configLoggers(final LoggingRuleConfiguration loggingRuleConfiguration, final LoggerContext loggerContext) {
58          Collection<ShardingSphereLogger> loggers = loggingRuleConfiguration.getLoggers();
59          loggers.forEach(each -> {
60              Logger logger = loggerContext.getLogger(each.getLoggerName());
61              logger.setLevel(Level.valueOf(each.getLevel()));
62              logger.setAdditive(each.getAdditivity());
63              LogbackUtils.addAppender(logger, loggingRuleConfiguration, each.getAppenderName());
64          });
65      }
66      
67      @SneakyThrows({IncompatibleClassException.class, DynamicClassLoadingException.class})
68      @SuppressWarnings("unchecked")
69      private static void addAppender(final Logger logger, final LoggingRuleConfiguration loggingRuleConfiguration, final String appenderName) {
70          if (null == appenderName) {
71              return;
72          }
73          Optional<ShardingSphereAppender> shardingSphereAppenderOptional = loggingRuleConfiguration.getAppenders().stream().filter(each -> appenderName.equals(each.getAppenderName())).findFirst();
74          if (shardingSphereAppenderOptional.isPresent()) {
75              ShardingSphereAppender shardingSphereAppender = shardingSphereAppenderOptional.get();
76              Appender<ILoggingEvent> appender = (Appender<ILoggingEvent>) OptionHelper.instantiateByClassName(shardingSphereAppender.getAppenderClass(), Appender.class, logger.getLoggerContext());
77              appender.setContext(logger.getLoggerContext());
78              appender.setName(appenderName);
79              LogbackUtils.addEncoder(appender, shardingSphereAppender);
80              appender.start();
81              logger.detachAndStopAllAppenders();
82              logger.addAppender(appender);
83          }
84      }
85      
86      private static void addEncoder(final Appender<ILoggingEvent> appender, final ShardingSphereAppender shardingSphereAppender) {
87          if (appender instanceof OutputStreamAppender) {
88              OutputStreamAppender<ILoggingEvent> outputStreamAppender = (OutputStreamAppender<ILoggingEvent>) appender;
89              PatternLayoutEncoder patternLayoutEncoder = new PatternLayoutEncoder();
90              patternLayoutEncoder.setContext(appender.getContext());
91              patternLayoutEncoder.setPattern(shardingSphereAppender.getPattern());
92              outputStreamAppender.setEncoder(patternLayoutEncoder);
93              LogbackUtils.setFileOutput(outputStreamAppender, shardingSphereAppender);
94              patternLayoutEncoder.start();
95          }
96      }
97      
98      private static void setFileOutput(final OutputStreamAppender<ILoggingEvent> outputStreamAppender, final ShardingSphereAppender shardingSphereAppender) {
99          if (outputStreamAppender instanceof FileAppender) {
100             FileAppender<ILoggingEvent> fileAppender = (FileAppender<ILoggingEvent>) outputStreamAppender;
101             fileAppender.setFile(shardingSphereAppender.getFile());
102         }
103     }
104     
105     private static void startRootLogger(final LoggerContext loggerContext) {
106         Logger rootLogger = loggerContext.getLogger(Logger.ROOT_LOGGER_NAME);
107         Iterator<Appender<ILoggingEvent>> appenderIterator = rootLogger.iteratorForAppenders();
108         while (appenderIterator.hasNext()) {
109             appenderIterator.next().start();
110         }
111     }
112 }
