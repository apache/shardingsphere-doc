111     public PipelineContainerComposer(final PipelineTestParameter testParam, final JobType jobType) {
112         databaseType = testParam.getDatabaseType();
113         containerComposer = PipelineE2EEnvironment.getInstance().getItEnvType() == PipelineEnvTypeEnum.DOCKER
114                 ? new DockerContainerComposer(testParam.getDatabaseType(), testParam.getStorageContainerImage(), testParam.getStorageContainerCount())
115                 : new NativeContainerComposer(testParam.getDatabaseType());
116         if (PipelineE2EEnvironment.getInstance().getItEnvType() == PipelineEnvTypeEnum.DOCKER) {
117             DockerStorageContainer storageContainer = ((DockerContainerComposer) containerComposer).getStorageContainers().get(0);
118             username = storageContainer.getUsername();
119             password = storageContainer.getPassword();
120         } else {
121             username = PipelineE2EEnvironment.getInstance().getActualDataSourceUsername(databaseType);
122             password = PipelineE2EEnvironment.getInstance().getActualDataSourcePassword(databaseType);
123         }
124         extraSQLCommand = JAXB.unmarshal(Objects.requireNonNull(PipelineContainerComposer.class.getClassLoader().getResource(testParam.getScenario())), ExtraSQLCommand.class);
125         containerComposer.start();
126         sourceDataSource = StorageContainerUtils.generateDataSource(appendExtraParameter(getActualJdbcUrlTemplate(DS_0, false)), username, password);
127         proxyDataSource = StorageContainerUtils.generateDataSource(
128                 appendExtraParameter(containerComposer.getProxyJdbcUrl(PROXY_DATABASE)), ProxyContainerConstants.USERNAME, ProxyContainerConstants.PASSWORD);
129         init(jobType);
130     }

132     @SneakyThrows(SQLException.class)
133     private void init(final JobType jobType) {
134         String jdbcUrl = containerComposer.getProxyJdbcUrl(DatabaseTypeUtils.isPostgreSQL(databaseType) || DatabaseTypeUtils.isOpenGauss(databaseType) ? "postgres" : "");
135         try (Connection connection = DriverManager.getConnection(jdbcUrl, ProxyContainerConstants.USERNAME, ProxyContainerConstants.PASSWORD)) {
136             cleanUpPipelineJobs(connection, jobType);
137             cleanUpProxyDatabase(connection);
138             createProxyDatabase(connection);
139         }
140         cleanUpDataSource();
141     }
142     
143     private void cleanUpPipelineJobs(final Connection connection, final JobType jobType) throws SQLException {
144         if (PipelineEnvTypeEnum.NATIVE != PipelineE2EEnvironment.getInstance().getItEnvType()) {
145             return;
146         }
147         String jobTypeName = jobType.getTypeName();
148         for (Map<String, Object> each : queryJobs(connection, jobTypeName)) {
149             String jobId = each.get("id").toString();
150             Map<String, Object> jobInfo = queryForListWithLog(String.format("SHOW %s STATUS '%s'", jobTypeName, jobId)).get(0);
151             String status = jobInfo.get("status").toString();
152             if (JobStatus.FINISHED.name().equals(status)) {
153                 connection.createStatement().execute(String.format("COMMIT %s '%s'", jobTypeName, jobId));
154             } else {
155                 connection.createStatement().execute(String.format("ROLLBACK %s '%s'", jobTypeName, jobId));
156             }
157         }
158     }
159     
160     private List<Map<String, Object>> queryJobs(final Connection connection, final String jobTypeName) {
161         try {
162             return transformResultSetToList(connection.createStatement().executeQuery(String.format("SHOW %s LIST", jobTypeName)));
163         } catch (final SQLException ex) {
164             log.warn("{} execute failed, message {}", String.format("SHOW %s LIST", jobTypeName), ex.getMessage());
165             return Collections.emptyList();
166         }
167     }

169     @SneakyThrows(InterruptedException.class)
170     private void cleanUpProxyDatabase(final Connection connection) {
171         if (PipelineEnvTypeEnum.NATIVE != PipelineE2EEnvironment.getInstance().getItEnvType()) {
172             return;
173         }
174         try {
175             connection.createStatement().execute(String.format("DROP DATABASE IF EXISTS %s", PROXY_DATABASE));
176             Thread.sleep(2000L);
177         } catch (final SQLException ex) {
178             log.warn("Drop proxy database failed, error={}", ex.getMessage());
179         }
180     }

190     private void cleanUpDataSource() {
191         if (PipelineEnvTypeEnum.NATIVE != PipelineE2EEnvironment.getInstance().getItEnvType()) {
192             return;
193         }
194         for (String each : Arrays.asList(DS_0, DS_1, DS_2, DS_3, DS_4)) {
195             containerComposer.cleanUpDatabase(each);
196         }
197     }

199     /**
200      * Append extra parameter.
201      * 
202      * @param jdbcUrl JDBC URL
203      * @return appended JDBC URL
204      */
205     public String appendExtraParameter(final String jdbcUrl) {
206         if (DatabaseTypeUtils.isMySQL(databaseType)) {
207             return new JdbcUrlAppender().appendQueryProperties(jdbcUrl, PropertiesBuilder.build(new Property("rewriteBatchedStatements", Boolean.TRUE.toString())));
208         }
209         if (DatabaseTypeUtils.isPostgreSQL(databaseType) || DatabaseTypeUtils.isOpenGauss(databaseType)) {
210             return new JdbcUrlAppender().appendQueryProperties(jdbcUrl, PropertiesBuilder.build(new Property("stringtype", "unspecified")));
211         }
212         return jdbcUrl;
213     }

242      * 
243      * @param databaseName database name
244      * @param isInContainer is in container
245      * @param storageContainerIndex storage container index
246      * @return actual JDBC URL template

248     public String getActualJdbcUrlTemplate(final String databaseName, final boolean isInContainer, final int storageContainerIndex) {
249         if (PipelineEnvTypeEnum.DOCKER == PipelineE2EEnvironment.getInstance().getItEnvType()) {
250             DockerStorageContainer storageContainer = ((DockerContainerComposer) containerComposer).getStorageContainers().get(storageContainerIndex);
251             return isInContainer
252                     ? DataSourceEnvironment.getURL(getDatabaseType(), storageContainer.getNetworkAliases().get(0), storageContainer.getExposedPort(), databaseName)
253                     : storageContainer.getJdbcUrl(databaseName);
254         }
255         return DataSourceEnvironment.getURL(getDatabaseType(), "127.0.0.1", PipelineE2EEnvironment.getInstance().getActualDataSourceDefaultPort(databaseType), databaseName);
256     }

363      * 
364      * @param distSQL dist SQL
365      */
366     @SneakyThrows(InterruptedException.class)
367     public void waitJobPrepareSuccess(final String distSQL) {
368         for (int i = 0; i < 5; i++) {
369             List<Map<String, Object>> jobStatus = queryForListWithLog(distSQL);
370             Set<String> statusSet = jobStatus.stream().map(each -> String.valueOf(each.get("status"))).collect(Collectors.toSet());
371             if (statusSet.contains(JobStatus.PREPARING.name()) || statusSet.contains(JobStatus.RUNNING.name())) {
372                 Thread.sleep(2000L);
373             }
374         }
375     }

377     /**
378      * Query for list with log.
379      * 
380      * @param sql SQL
381      * @return query result
382      * @throws RuntimeException runtime exception
383      */
384     @SneakyThrows(InterruptedException.class)
385     public List<Map<String, Object>> queryForListWithLog(final String sql) {
386         int retryNumber = 0;
387         while (retryNumber <= 3) {
388             try (Connection connection = proxyDataSource.getConnection()) {
389                 ResultSet resultSet = connection.createStatement().executeQuery(sql);
390                 return transformResultSetToList(resultSet);
391             } catch (final SQLException ex) {
392                 log.error("Data access error.", ex);
393             }
394             Thread.sleep(3000L);
395             retryNumber++;
396         }
397         throw new RuntimeException("Can not get result from proxy.");
398     }

400     /**
401      * Transform result set to list.
402      * 
403      * @param resultSet result set
404      * @return transformed result
405      * @throws SQLException SQL exception
406      */
407     public List<Map<String, Object>> transformResultSetToList(final ResultSet resultSet) throws SQLException {
408         ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
409         int columns = resultSetMetaData.getColumnCount();
410         List<Map<String, Object>> result = new ArrayList<>();
411         while (resultSet.next()) {
412             Map<String, Object> row = new HashMap<>();
413             for (int i = 1; i <= columns; i++) {
414                 row.put(resultSetMetaData.getColumnLabel(i).toLowerCase(), resultSet.getObject(i));
415             }
416             result.add(row);
417         }
418         return result;
419     }

432      * Wait increment task finished.
433      * 
434      * @param distSQL dist SQL
435      * @return result
436      * @throws InterruptedException interrupted exception
437      */
438     // TODO use DAO to query via DistSQL
439     public List<Map<String, Object>> waitIncrementTaskFinished(final String distSQL) throws InterruptedException {
440         if (null != increaseTaskThread) {
441             TimeUnit.SECONDS.timedJoin(increaseTaskThread, 30);
442         }
443         for (int i = 0; i < 10; i++) {
444             List<Map<String, Object>> listJobStatus = queryForListWithLog(distSQL);
445             log.info("show status result: {}", listJobStatus);
446             Set<String> actualStatus = new HashSet<>();
447             Collection<Integer> incrementalIdleSecondsList = new LinkedList<>();
448             for (Map<String, Object> each : listJobStatus) {
449                 assertTrue(Strings.isNullOrEmpty(each.get("error_message").toString()), "error_message is not null");
450                 actualStatus.add(each.get("status").toString());
451                 String incrementalIdleSeconds = each.get("incremental_idle_seconds").toString();
452                 incrementalIdleSecondsList.add(Strings.isNullOrEmpty(incrementalIdleSeconds) ? 0 : Integer.parseInt(incrementalIdleSeconds));
453             }
454             if (Collections.min(incrementalIdleSecondsList) <= 5) {
455                 Thread.sleep(3000L);
456                 continue;
457             }
458             if (actualStatus.size() == 1 && actualStatus.contains(JobStatus.EXECUTE_INCREMENTAL_TASK.name())) {
459                 return listJobStatus;
460             }
461         }
462         return Collections.emptyList();
463     }

524     /**
525      * Generate ShardingSphere data source from proxy.
526      * 
527      * @return ShardingSphere data source
528      * @throws SQLException SQL exception
529      */
530     // TODO proxy support for some fields still needs to be optimized, such as binary of MySQL, after these problems are optimized, Proxy dataSource can be used.
531     public DataSource generateShardingSphereDataSourceFromProxy() throws SQLException {
532         Awaitility.await().atMost(5, TimeUnit.SECONDS).pollInterval(1, TimeUnit.SECONDS).until(() -> !getYamlRootConfig().getRules().isEmpty());
533         YamlRootConfiguration rootConfig = getYamlRootConfig();
534         if (PipelineEnvTypeEnum.DOCKER == PipelineE2EEnvironment.getInstance().getItEnvType()) {
535             DockerStorageContainer storageContainer = ((DockerContainerComposer) containerComposer).getStorageContainers().get(0);
536             String sourceUrl = String.join(":", storageContainer.getNetworkAliases().get(0), Integer.toString(storageContainer.getExposedPort()));
537             String targetUrl = String.join(":", storageContainer.getHost(), Integer.toString(storageContainer.getMappedPort()));
538             for (Map<String, Object> each : rootConfig.getDataSources().values()) {
539                 each.put("url", each.get("url").toString().replaceFirst(sourceUrl, targetUrl));
540             }
541         }
542         for (Map<String, Object> each : rootConfig.getDataSources().values()) {
543             each.put("dataSourceClassName", "com.zaxxer.hikari.HikariDataSource");
544         }
545         return YamlShardingSphereDataSourceFactory.createDataSourceWithoutCache(rootConfig);
546     }
