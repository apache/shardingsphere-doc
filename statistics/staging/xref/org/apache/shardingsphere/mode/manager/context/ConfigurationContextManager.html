72      /**
73       * Register storage unit.
74       *
75       * @param databaseName database name
76       * @param propsMap data source pool properties map
77       */
78      public synchronized void registerStorageUnit(final String databaseName, final Map<String, DataSourcePoolProperties> propsMap) {
79          try {
80              closeStaleRules(databaseName);
81              SwitchingResource switchingResource =
82                      new ResourceSwitchManager().registerStorageUnit(metaDataContexts.get().getMetaData().getDatabase(databaseName).getResourceMetaData(), propsMap);
83              buildNewMetaDataContext(databaseName, switchingResource, false);
84          } catch (final SQLException ex) {
85              log.error("Alter database: {} register storage unit failed", databaseName, ex);
86          }
87      }

101         } catch (final SQLException ex) {

109      * @param databaseName database name
110      * @param storageUnitName storage unit name

112     public synchronized void unregisterStorageUnit(final String databaseName, final String storageUnitName) {
113         try {
114             closeStaleRules(databaseName);

123     private void buildNewMetaDataContext(final String databaseName, final SwitchingResource switchingResource, final boolean isDropConfig) throws SQLException {
124         metaDataContexts.get().getMetaData().getDatabases().putAll(renewDatabase(metaDataContexts.get().getMetaData().getDatabase(databaseName), switchingResource));
125         MetaDataContexts reloadMetaDataContexts = createMetaDataContexts(databaseName, false, switchingResource, null);
126         reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getSchemas().forEach((schemaName, schema) -> reloadMetaDataContexts.getPersistService().getDatabaseMetaDataService()
127                 .persist(reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getName(), schemaName, schema));
128         Optional.ofNullable(reloadMetaDataContexts.getStatistics().getDatabaseData().get(databaseName))
129                 .ifPresent(optional -> optional.getSchemaData().forEach((schemaName, schemaData) -> reloadMetaDataContexts.getPersistService().getShardingSphereDataPersistService()
130                         .persist(databaseName, schemaName, schemaData, metaDataContexts.get().getMetaData().getDatabases())));
131         alterSchemaMetaData(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName), metaDataContexts.get().getMetaData().getDatabase(databaseName), isDropConfig);
132         metaDataContexts.set(reloadMetaDataContexts);
133         metaDataContexts.get().getMetaData().getDatabases().putAll(newShardingSphereDatabase(metaDataContexts.get().getMetaData().getDatabase(databaseName)));
134         switchingResource.closeStaleDataSources();
135     }

138      * Alter rule configuration.
139      *
140      * @param databaseName database name
141      * @param ruleConfigs rule configurations
142      */
143     public synchronized void alterRuleConfiguration(final String databaseName, final Collection<RuleConfiguration> ruleConfigs) {
144         try {
145             // TODO consider rename this method to alterDatabaseRuleConfiguration
146             closeStaleRules(databaseName);
147             MetaDataContexts reloadMetaDataContexts = createMetaDataContextsWhenRuleChanged(databaseName, false, null, ruleConfigs);
148             alterSchemaMetaData(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName), metaDataContexts.get().getMetaData().getDatabase(databaseName), false);
149             metaDataContexts.set(reloadMetaDataContexts);
150             metaDataContexts.get().getMetaData().getDatabase(databaseName).getSchemas().putAll(newShardingSphereSchemas(metaDataContexts.get().getMetaData().getDatabase(databaseName)));
151         } catch (final SQLException ex) {
152             log.error("Alter database: {} rule configurations failed", databaseName, ex);
153         }
154     }

156     /**
157      * Alter rule configuration.
158      *
159      * @param databaseName database name
160      * @param ruleConfig rule configurations
161      */
162     public synchronized void alterRuleConfiguration(final String databaseName, final RuleConfiguration ruleConfig) {
163         try {
164             ShardingSphereDatabase database = metaDataContexts.get().getMetaData().getDatabase(databaseName);
165             Collection<ShardingSphereRule> rules = new LinkedList<>(database.getRuleMetaData().getRules());
166             rules.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
167             rules.addAll(DatabaseRulesBuilder.build(databaseName, database.getProtocolType(),
168                     database.getResourceMetaData().getStorageUnits().entrySet().stream()
169                             .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
170                     database.getRuleMetaData().getRules(), ruleConfig, instanceContext));
171             refreshMetadata(databaseName, database, rules, false);
172         } catch (final SQLException ex) {
173             log.error("Alter database: {} rule configurations failed", databaseName, ex);
174         }
175     }

177     /**
178      * Drop rule configuration.
179      *
180      * @param databaseName database name
181      * @param ruleConfig rule configurations
182      */
183     public synchronized void dropRuleConfiguration(final String databaseName, final RuleConfiguration ruleConfig) {
184         try {
185             ShardingSphereDatabase database = metaDataContexts.get().getMetaData().getDatabase(databaseName);
186             Collection<ShardingSphereRule> rules = new LinkedList<>(database.getRuleMetaData().getRules());
187             rules.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
188             if (isNotEmptyConfig(ruleConfig)) {
189                 rules.addAll(DatabaseRulesBuilder.build(databaseName, database.getProtocolType(),
190                         database.getResourceMetaData().getStorageUnits().entrySet().stream()
191                                 .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
192                         database.getRuleMetaData().getRules(), ruleConfig, instanceContext));
193             }
194             refreshMetadata(databaseName, database, rules, true);
195         } catch (final SQLException ex) {
196             log.error("Drop database: {} rule configurations failed", databaseName, ex);
197         }
198     }

200     private static boolean isNotEmptyConfig(final RuleConfiguration ruleConfig) {
201         return !((DatabaseRuleConfiguration) ruleConfig).isEmpty();
202     }

204     private void refreshMetadata(final String databaseName, final ShardingSphereDatabase database, final Collection<ShardingSphereRule> rules, final boolean isDropConfig) throws SQLException {
205         database.getRuleMetaData().getRules().clear();
206         database.getRuleMetaData().getRules().addAll(rules);
207         MetaDataContexts reloadMetaDataContexts = createMetaDataContextsByAlterRule(databaseName, database.getRuleMetaData().getConfigurations());
208         alterSchemaMetaData(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName), metaDataContexts.get().getMetaData().getDatabase(databaseName), isDropConfig);
209         metaDataContexts.set(reloadMetaDataContexts);
210         metaDataContexts.get().getMetaData().getDatabase(databaseName).getSchemas().putAll(newShardingSphereSchemas(metaDataContexts.get().getMetaData().getDatabase(databaseName)));
211     }

213     private MetaDataContexts createMetaDataContextsByAlterRule(final String databaseName, final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
214         Map<String, ShardingSphereDatabase> changedDatabases = createChangedDatabases(databaseName, false, null, ruleConfigs);
215         return newMetaDataContexts(new ShardingSphereMetaData(changedDatabases, metaDataContexts.get().getMetaData().getGlobalResourceMetaData(),
216                 metaDataContexts.get().getMetaData().getGlobalRuleMetaData(), metaDataContexts.get().getMetaData().getProps()));
217     }

219     /**
220      * Alter schema meta data.
221      *
222      * @param databaseName database name
223      * @param reloadDatabase reload database
224      * @param currentDatabase current database
225      * @param isDropConfig is drop configuration
226      */
227     public void alterSchemaMetaData(final String databaseName, final ShardingSphereDatabase reloadDatabase, final ShardingSphereDatabase currentDatabase, final boolean isDropConfig) {
228         Map<String, ShardingSphereSchema> toBeAlterSchemas = GenericSchemaManager.getToBeDeletedTablesBySchemas(reloadDatabase.getSchemas(), currentDatabase.getSchemas());
229         Map<String, ShardingSphereSchema> toBeAddedSchemas = GenericSchemaManager.getToBeAddedTablesBySchemas(reloadDatabase.getSchemas(), currentDatabase.getSchemas());
230         if (isDropConfig) {
231             toBeAddedSchemas.forEach((key, value) -> metaDataContexts.get().getPersistService().getDatabaseMetaDataService().persist(databaseName, key, value));
232         } else {
233             alterSchemaMetaData(databaseName, toBeAddedSchemas);
234         }
235         toBeAlterSchemas.forEach((key, value) -> metaDataContexts.get().getPersistService().getDatabaseMetaDataService().delete(databaseName, key, value));
236     }
237     
238     private void alterSchemaMetaData(final String databaseName, final Map<String, ShardingSphereSchema> toBeAddedSchemas) {
239         for (Entry<String, ShardingSphereSchema> entry : toBeAddedSchemas.entrySet()) {
240             if (entry.getValue().getTables().isEmpty() && entry.getValue().getViews().isEmpty()) {
241                 metaDataContexts.get().getPersistService().getDatabaseMetaDataService().addSchema(databaseName, entry.getKey());
242             }
243             metaDataContexts.get().getPersistService().getDatabaseMetaDataService().persist(databaseName, entry.getKey(), entry.getValue());
244         }
245     }

247     /**
248      * Renew ShardingSphere databases.
249      *
250      * @param database database
251      * @param resource resource
252      * @return ShardingSphere databases
253      */
254     public Map<String, ShardingSphereDatabase> renewDatabase(final ShardingSphereDatabase database, final SwitchingResource resource) {
255         Map<StorageNode, DataSource> newStorageNodes = getNewStorageNodes(database.getResourceMetaData().getDataSources(), resource);
256         Map<String, StorageUnit> newStorageUnits = getNewStorageUnits(database.getResourceMetaData().getStorageUnits(), resource);
257         return Collections.singletonMap(database.getName().toLowerCase(), new ShardingSphereDatabase(database.getName(), database.getProtocolType(),
258                 new ResourceMetaData(newStorageNodes, newStorageUnits), database.getRuleMetaData(), database.getSchemas()));
259     }
260     
261     private Map<StorageNode, DataSource> getNewStorageNodes(final Map<StorageNode, DataSource> currentStorageNodes, final SwitchingResource resource) {
262         Map<StorageNode, DataSource> result = new LinkedHashMap<>(currentStorageNodes.size(), 1F);
263         for (Entry<StorageNode, DataSource> entry : currentStorageNodes.entrySet()) {
264             if (!resource.getStaleDataSources().containsKey(entry.getKey())) {
265                 result.put(entry.getKey(), entry.getValue());
266             }
267         }
268         return result;
269     }
270     
271     private Map<String, StorageUnit> getNewStorageUnits(final Map<String, StorageUnit> currentStorageUnits, final SwitchingResource resource) {
272         Map<String, StorageUnit> result = new LinkedHashMap<>(currentStorageUnits.size(), 1F);
273         for (Entry<String, StorageUnit> entry : currentStorageUnits.entrySet()) {
274             if (!resource.getStaleStorageUnitNames().contains(entry.getKey())) {
275                 result.put(entry.getKey(), entry.getValue());
276             }
277         }
278         return result;
279     }

281     @SneakyThrows(Exception.class)
282     private void closeStaleRules(final String databaseName) {
283         for (ShardingSphereRule each : metaDataContexts.get().getMetaData().getDatabase(databaseName).getRuleMetaData().getRules()) {
284             if (each instanceof AutoCloseable) {
285                 ((AutoCloseable) each).close();
286             }
287         }
288     }

307     /**
308      * Create meta data contexts.
309      *
310      * @param databaseName database name
311      * @param internalLoadMetaData internal load meta data
312      * @param switchingResource switching resource
313      * @param ruleConfigs rule configs
314      * @return MetaDataContexts meta data contexts
315      * @throws SQLException SQL exception
316      */
317     public MetaDataContexts createMetaDataContexts(final String databaseName, final boolean internalLoadMetaData, final SwitchingResource switchingResource,
318                                                    final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
319         Map<String, ShardingSphereDatabase> changedDatabases = createChangedDatabases(databaseName, internalLoadMetaData, switchingResource, ruleConfigs);
320         ConfigurationProperties props = metaDataContexts.get().getMetaData().getProps();
321         RuleMetaData changedGlobalMetaData = new RuleMetaData(
322                 GlobalRulesBuilder.buildRules(metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getConfigurations(), changedDatabases, props));
323         return newMetaDataContexts(new ShardingSphereMetaData(changedDatabases, metaDataContexts.get().getMetaData().getGlobalResourceMetaData(), changedGlobalMetaData, props));
324     }

326     /**
327      * Create changed databases.
328      *
329      * @param databaseName database name
330      * @param internalLoadMetaData internal load meta data
331      * @param switchingResource switching resource
332      * @param ruleConfigs rule configs
333      * @return ShardingSphere databases
334      * @throws SQLException SQL exception
335      */
336     public synchronized Map<String, ShardingSphereDatabase> createChangedDatabases(final String databaseName, final boolean internalLoadMetaData,
337                                                                                    final SwitchingResource switchingResource, final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
338         Collection<RuleConfiguration> toBeCreatedRuleConfigs = null == ruleConfigs
339                 ? metaDataContexts.get().getMetaData().getDatabase(databaseName).getRuleMetaData().getConfigurations()
340                 : ruleConfigs;
341         DatabaseConfiguration toBeCreatedDatabaseConfig = getDatabaseConfiguration(
342                 metaDataContexts.get().getMetaData().getDatabase(databaseName).getResourceMetaData(), switchingResource, toBeCreatedRuleConfigs);
343         ShardingSphereDatabase changedDatabase = createChangedDatabase(metaDataContexts.get().getMetaData().getDatabase(databaseName).getName(), internalLoadMetaData,
344                 metaDataContexts.get().getPersistService(), toBeCreatedDatabaseConfig, metaDataContexts.get().getMetaData().getProps(), instanceContext);
345         Map<String, ShardingSphereDatabase> result = new LinkedHashMap<>(metaDataContexts.get().getMetaData().getDatabases());
346         result.put(databaseName.toLowerCase(), changedDatabase);
347         return result;
348     }
349     
350     private DatabaseConfiguration getDatabaseConfiguration(final ResourceMetaData resourceMetaData, final SwitchingResource switchingResource,
351                                                            final Collection<RuleConfiguration> toBeCreatedRuleConfigs) {
352         Map<String, DataSourcePoolProperties> propsMap = null == switchingResource
353                 ? resourceMetaData.getStorageUnits().entrySet().stream()
354                         .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSourcePoolProperties(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new))
355                 : switchingResource.getMergedDataSourcePoolPropertiesMap();
356         return new DataSourceProvidedDatabaseConfiguration(getMergedStorageNodeDataSources(resourceMetaData, switchingResource), toBeCreatedRuleConfigs, propsMap);
357     }
358     
359     private Map<StorageNode, DataSource> getMergedStorageNodeDataSources(final ResourceMetaData currentResourceMetaData, final SwitchingResource switchingResource) {
360         Map<StorageNode, DataSource> result = currentResourceMetaData.getDataSources();
361         if (null != switchingResource && !switchingResource.getNewDataSources().isEmpty()) {
362             result.putAll(switchingResource.getNewDataSources());
363         }
364         return result;
365     }

367     private ShardingSphereDatabase createChangedDatabase(final String databaseName, final boolean internalLoadMetaData, final MetaDataBasedPersistService persistService,
368                                                          final DatabaseConfiguration databaseConfig, final ConfigurationProperties props, final InstanceContext instanceContext) throws SQLException {
369         return internalLoadMetaData
370                 ? InternalMetaDataFactory.create(databaseName, persistService, databaseConfig, props, instanceContext)
371                 : ExternalMetaDataFactory.create(databaseName, databaseConfig, props, instanceContext);
372     }

374     private Map<String, ShardingSphereSchema> newShardingSphereSchemas(final ShardingSphereDatabase database) {
375         Map<String, ShardingSphereSchema> result = new LinkedHashMap<>(database.getSchemas().size(), 1F);
376         database.getSchemas().forEach((key, value) -> result.put(key, new ShardingSphereSchema(value.getTables(), value.getViews())));
377         return result;
378     }

392     /**
393      * Alter global rule configuration.
394      *
395      * @param ruleConfigs global rule configuration
396      */
397     public synchronized void alterGlobalRuleConfiguration(final Collection<RuleConfiguration> ruleConfigs) {
398         if (ruleConfigs.isEmpty()) {
399             return;
400         }
401         closeStaleGlobalRules();
402         RuleMetaData toBeChangedGlobalRuleMetaData = new RuleMetaData(
403                 GlobalRulesBuilder.buildRules(ruleConfigs, metaDataContexts.get().getMetaData().getDatabases(), metaDataContexts.get().getMetaData().getProps()));
404         ShardingSphereMetaData toBeChangedMetaData = new ShardingSphereMetaData(metaDataContexts.get().getMetaData().getDatabases(), metaDataContexts.get().getMetaData().getGlobalResourceMetaData(),
405                 toBeChangedGlobalRuleMetaData, metaDataContexts.get().getMetaData().getProps());
406         metaDataContexts.set(newMetaDataContexts(toBeChangedMetaData));
407     }
408     
409     /**
410      * Alter global rule configuration.
411      *
412      * @param ruleConfig global rule configuration
413      */
414     public synchronized void alterGlobalRuleConfiguration(final RuleConfiguration ruleConfig) {
415         if (null == ruleConfig) {
416             return;
417         }
418         Collection<ShardingSphereRule> rules = removeSingleGlobalRule(ruleConfig);
419         rules.addAll(GlobalRulesBuilder.buildSingleRules(ruleConfig, metaDataContexts.get().getMetaData().getDatabases(), metaDataContexts.get().getMetaData().getProps()));
420         metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getRules().clear();
421         metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getRules().addAll(rules);
422         ShardingSphereMetaData toBeChangedMetaData = new ShardingSphereMetaData(metaDataContexts.get().getMetaData().getDatabases(), metaDataContexts.get().getMetaData().getGlobalResourceMetaData(),
423                 metaDataContexts.get().getMetaData().getGlobalRuleMetaData(), metaDataContexts.get().getMetaData().getProps());
424         metaDataContexts.set(newMetaDataContexts(toBeChangedMetaData));
425     }
426     
427     private Collection<ShardingSphereRule> removeSingleGlobalRule(final RuleConfiguration ruleConfig) {
428         Collection<ShardingSphereRule> result = new LinkedList<>(metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getRules());
429         result.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
430         return result;
431     }
432     
433     @SneakyThrows(Exception.class)
434     private void closeStaleGlobalRules() {
435         for (ShardingSphereRule each : metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getRules()) {
436             if (each instanceof AutoCloseable) {
437                 ((AutoCloseable) each).close();
438             }
439         }
440     }

449                 metaDataContexts.get().getMetaData().getGlobalRuleMetaData(), new ConfigurationProperties(props));
450         metaDataContexts.set(newMetaDataContexts(toBeChangedMetaData));
