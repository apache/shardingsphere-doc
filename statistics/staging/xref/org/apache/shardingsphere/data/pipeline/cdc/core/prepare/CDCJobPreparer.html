42  /**
43   * CDC job preparer.
44   */
45  @Slf4j
46  public final class CDCJobPreparer {
47      
48      private final CDCJobAPI jobAPI = new CDCJobAPI();
49      
50      /**
51       * Do prepare work.
52       *
53       * @param jobItemContext job item context
54       */
55      public void initTasks(final CDCJobItemContext jobItemContext) {
56          Optional<InventoryIncrementalJobItemProgress> jobItemProgress = jobAPI.getJobItemProgress(jobItemContext.getJobId(), jobItemContext.getShardingItem());
57          if (!jobItemProgress.isPresent()) {
58              jobAPI.persistJobItemProgress(jobItemContext);
59          }
60          if (jobItemContext.isStopping()) {
61              PipelineJobCenter.stop(jobItemContext.getJobId());
62              return;
63          }
64          initIncrementalTasks(jobItemContext);
65          CDCJobConfiguration jobConfig = jobItemContext.getJobConfig();
66          if (jobConfig.isFull()) {
67              initInventoryTasks(jobItemContext);
68          }
69      }
70      
71      private void initInventoryTasks(final CDCJobItemContext jobItemContext) {
72          CDCTaskConfiguration taskConfig = jobItemContext.getTaskConfig();
73          // TODO channel requires a new implementation
74          InventoryDumperConfiguration inventoryDumperConfig = new InventoryDumperConfiguration(jobItemContext.getTaskConfig().getDumperConfig());
75          InventoryTaskSplitter inventoryTaskSplitter = new InventoryTaskSplitter(jobItemContext.getSourceDataSource(), inventoryDumperConfig, taskConfig.getImporterConfig());
76          List<InventoryTask> allInventoryTasks = inventoryTaskSplitter.splitInventoryData(jobItemContext);
77          jobItemContext.getInventoryTasks().addAll(allInventoryTasks);
78      }
79      
80      private void initIncrementalTasks(final CDCJobItemContext jobItemContext) {
81          PipelineChannelCreator pipelineChannelCreator = jobItemContext.getJobProcessContext().getPipelineChannelCreator();
82          CDCTaskConfiguration taskConfig = jobItemContext.getTaskConfig();
83          JobItemIncrementalTasksProgress initIncremental = null == jobItemContext.getInitProgress() ? null : jobItemContext.getInitProgress().getIncremental();
84          try {
85              taskConfig.getDumperConfig().setPosition(PipelineJobPreparerUtils.getIncrementalPosition(initIncremental, taskConfig.getDumperConfig(), jobItemContext.getDataSourceManager()));
86          } catch (final SQLException ex) {
87              throw new PrepareJobWithGetBinlogPositionException(jobItemContext.getJobId(), ex);
88          }
89          PipelineTableMetaDataLoader sourceMetaDataLoader = jobItemContext.getSourceMetaDataLoader();
90          ExecuteEngine incrementalExecuteEngine = jobItemContext.getJobProcessContext().getIncrementalExecuteEngine();
91          IncrementalTask incrementalTask = new IncrementalTask(taskConfig.getImporterConfig().getConcurrency(), taskConfig.getDumperConfig(), taskConfig.getImporterConfig(),
92                  pipelineChannelCreator, jobItemContext.getImporterConnector(), sourceMetaDataLoader, incrementalExecuteEngine, jobItemContext);
93          jobItemContext.getIncrementalTasks().add(incrementalTask);
94      }
95  }
