54  class TrafficRuleTest {
55      
56      @Test
57      void assertFindMatchedStrategyRuleWhenSQLHintMatch() {
58          TrafficRule trafficRule = new TrafficRule(createTrafficRuleConfig());
59          Optional<TrafficStrategyRule> actual = trafficRule.findMatchedStrategyRule(createQueryContext(true), false);
60          assertTrue(actual.isPresent());
61          assertThat(actual.get().getName(), is("sql_hint_traffic"));
62          assertThat(actual.get().getLabels(), is(new HashSet<>(Arrays.asList("OLTP", "OLAP"))));
63          assertThat(actual.get().getTrafficAlgorithm(), instanceOf(SQLHintTrafficAlgorithm.class));
64          assertThat(actual.get().getLoadBalancer(), instanceOf(RandomTrafficLoadBalanceAlgorithm.class));
65      }
66      
67      @Test
68      void assertFindMatchedStrategyRuleWhenSQLHintNotMatch() {
69          assertFalse(new TrafficRule(createTrafficRuleConfig()).findMatchedStrategyRule(createQueryContext(false), false).isPresent());
70      }
71      
72      @Test
73      void assertFindMatchedStrategyRuleWhenInTransaction() {
74          TrafficRule trafficRule = new TrafficRule(createTrafficRuleConfig());
75          Optional<TrafficStrategyRule> actual = trafficRule.findMatchedStrategyRule(createQueryContext(false), true);
76          assertTrue(actual.isPresent());
77          assertThat(actual.get().getName(), is("transaction_traffic"));
78          assertThat(actual.get().getLabels(), is(Collections.singleton("OLAP")));
79          assertThat(actual.get().getTrafficAlgorithm(), instanceOf(ProxyTrafficAlgorithm.class));
80          assertThat(actual.get().getLoadBalancer(), instanceOf(RandomTrafficLoadBalanceAlgorithm.class));
81      }
82      
83      @Test
84      void assertGetLabels() {
85          assertThat(new TrafficRule(createTrafficRuleConfig()).getLabels(), is(new HashSet<>(Arrays.asList("OLAP", "OLTP"))));
86      }
87      
88      @SuppressWarnings({"rawtypes", "unchecked"})
89      private QueryContext createQueryContext(final boolean includeComments) {
90          QueryContext result = mock(QueryContext.class);
91          MySQLSelectStatement sqlStatement = mock(MySQLSelectStatement.class);
92          when(sqlStatement.getCommentSegments()).thenReturn(includeComments ? Collections.singleton(new CommentSegment("/* SHARDINGSPHERE_HINT: USE_TRAFFIC=true */", 0, 0)) : Collections.emptyList());
93          when(sqlStatement.getProjections()).thenReturn(new ProjectionsSegment(0, 0));
94          SQLStatementContext statementContext =
95                  new SelectStatementContext(createShardingSphereMetaData(mock(ShardingSphereDatabase.class)), Collections.emptyList(), sqlStatement, DefaultDatabase.LOGIC_NAME);
96          when(result.getSqlStatementContext()).thenReturn(statementContext);
97          return result;
98      }
99      
100     private ShardingSphereMetaData createShardingSphereMetaData(final ShardingSphereDatabase database) {
101         return new ShardingSphereMetaData(Collections.singletonMap(DefaultDatabase.LOGIC_NAME, database), mock(ShardingSphereRuleMetaData.class), mock(ConfigurationProperties.class));
102     }
103     
104     private TrafficRuleConfiguration createTrafficRuleConfig() {
105         TrafficRuleConfiguration result = new TrafficRuleConfiguration();
106         result.getTrafficStrategies().add(new TrafficStrategyConfiguration("sql_hint_traffic", Arrays.asList("OLTP", "OLAP"), "sql_hint_match", "random"));
107         result.getTrafficStrategies().add(new TrafficStrategyConfiguration("transaction_traffic", Collections.singletonList("OLAP"), "transaction_algorithm", "random"));
108         result.getTrafficAlgorithms().put("sql_hint_match", createSQLHintTrafficAlgorithm());
109         result.getTrafficAlgorithms().put("transaction_algorithm", createTransactionTrafficAlgorithm());
110         result.getLoadBalancers().put("random", createLoadBalancer());
111         return result;
112     }
113     
114     private AlgorithmConfiguration createSQLHintTrafficAlgorithm() {
115         AlgorithmConfiguration result = mock(AlgorithmConfiguration.class);
116         when(result.getType()).thenReturn("SQL_HINT");
117         when(result.getProps()).thenReturn(PropertiesBuilder.build(new Property("traffic", Boolean.TRUE.toString())));
118         return result;
119     }
120     
121     private AlgorithmConfiguration createTransactionTrafficAlgorithm() {
122         AlgorithmConfiguration result = mock(AlgorithmConfiguration.class);
123         when(result.getType()).thenReturn("PROXY");
124         return result;
125     }
126     
127     private AlgorithmConfiguration createLoadBalancer() {
128         AlgorithmConfiguration result = mock(AlgorithmConfiguration.class);
129         when(result.getType()).thenReturn("RANDOM");
130         return result;
131     }
