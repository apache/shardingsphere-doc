20  import com.google.common.base.Strings;

22  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptColumnRuleConfiguration;
23  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptTableRuleConfiguration;
24  import org.apache.shardingsphere.encrypt.constant.EncryptOrder;
25  import org.apache.shardingsphere.encrypt.exception.metadata.EncryptAssistedQueryColumnNotFoundException;
26  import org.apache.shardingsphere.encrypt.exception.metadata.EncryptAssistedQueryEncryptorNotFoundException;
27  import org.apache.shardingsphere.encrypt.exception.metadata.EncryptCipherColumnNotFoundException;
28  import org.apache.shardingsphere.encrypt.exception.metadata.EncryptEncryptorNotFoundException;
29  import org.apache.shardingsphere.encrypt.exception.metadata.EncryptLikeQueryColumnNotFoundException;
30  import org.apache.shardingsphere.encrypt.exception.metadata.EncryptLikeQueryEncryptorNotFoundException;
31  import org.apache.shardingsphere.infra.config.rule.checker.RuleConfigurationChecker;
32  import org.apache.shardingsphere.infra.rule.ShardingSphereRule;

35  import javax.sql.DataSource;
36  import java.util.Collection;

39  /**
40   * Encrypt rule configuration checker.
41   */
42  public final class EncryptRuleConfigurationChecker implements RuleConfigurationChecker<EncryptRuleConfiguration> {
43      
44      @Override
45      public void check(final String databaseName, final EncryptRuleConfiguration config, final Map<String, DataSource> dataSourceMap, final Collection<ShardingSphereRule> rules) {
46          checkTableConfiguration(databaseName, config.getTables(), config.getEncryptors().keySet());
47      }
48      
49      private void checkTableConfiguration(final String databaseName, final Collection<EncryptTableRuleConfiguration> tables, final Collection<String> encryptors) {
50          for (EncryptTableRuleConfiguration each : tables) {
51              for (EncryptColumnRuleConfiguration column : each.getColumns()) {
52                  checkCipherColumnConfiguration(databaseName, encryptors, column);
53                  checkAssistColumnConfiguration(databaseName, encryptors, column);
54                  checkLikeColumnConfiguration(databaseName, encryptors, column);
55              }
56          }
57      }
58      
59      private void checkCipherColumnConfiguration(final String databaseName, final Collection<String> encryptors, final EncryptColumnRuleConfiguration column) {
60          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(column.getCipherColumn()), () -> new EncryptCipherColumnNotFoundException(column.getLogicColumn(), databaseName));
61          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(column.getEncryptorName()),
62                  () -> new EncryptEncryptorNotFoundException(String.format("Encryptor name of `%s` can not be null in database `%s`.", column.getLogicColumn(), databaseName)));
63          ShardingSpherePreconditions.checkState(encryptors.contains(column.getEncryptorName()),
64                  () -> new EncryptEncryptorNotFoundException(String.format("Can not find encryptor `%s` in database `%s`.", column.getEncryptorName(), databaseName)));
65      }
66      
67      private void checkAssistColumnConfiguration(final String databaseName, final Collection<String> encryptors, final EncryptColumnRuleConfiguration column) {
68          if (Strings.isNullOrEmpty(column.getAssistedQueryColumn()) && Strings.isNullOrEmpty(column.getAssistedQueryEncryptorName())) {
69              return;
70          }
71          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(column.getAssistedQueryColumn()), () -> new EncryptAssistedQueryColumnNotFoundException(column.getLogicColumn(), databaseName));
72          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(column.getAssistedQueryEncryptorName()), () -> new EncryptAssistedQueryEncryptorNotFoundException(
73                  String.format("Assisted query encryptor name of `%s` can not be null in database `%s`.", column.getLogicColumn(), databaseName)));
74          ShardingSpherePreconditions.checkState(encryptors.contains(column.getAssistedQueryEncryptorName()), () -> new EncryptAssistedQueryEncryptorNotFoundException(
75                  String.format("Can not find assisted query encryptor `%s` in database `%s`.", column.getAssistedQueryEncryptorName(), databaseName)));
76      }
77      
78      private void checkLikeColumnConfiguration(final String databaseName, final Collection<String> encryptors, final EncryptColumnRuleConfiguration column) {
79          if (Strings.isNullOrEmpty(column.getLikeQueryColumn()) && Strings.isNullOrEmpty(column.getLikeQueryEncryptorName())) {
80              return;
81          }
82          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(column.getLikeQueryColumn()), () -> new EncryptLikeQueryColumnNotFoundException(column.getLogicColumn(), databaseName));
83          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(column.getLikeQueryEncryptorName()),
84                  () -> new EncryptLikeQueryEncryptorNotFoundException(String.format("Like query encryptor name of `%s` can not be null in database `%s`.", column.getLogicColumn(), databaseName)));
85          ShardingSpherePreconditions.checkState(encryptors.contains(column.getLikeQueryEncryptorName()),
86                  () -> new EncryptLikeQueryEncryptorNotFoundException(String.format("Can not find like query encryptor `%s` in database `%s`.", column.getLikeQueryEncryptorName(), databaseName)));
87      }
88      
89      @Override
90      public int getOrder() {
91          return EncryptOrder.ORDER;
92      }
93      
94      @Override
95      public Class<EncryptRuleConfiguration> getTypeClass() {
96          return EncryptRuleConfiguration.class;
97      }
98  }
