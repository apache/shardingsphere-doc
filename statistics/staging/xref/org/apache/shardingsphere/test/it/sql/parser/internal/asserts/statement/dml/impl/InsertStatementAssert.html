59  /**
60   * Insert statement assert.
61   */
62  @NoArgsConstructor(access = AccessLevel.PRIVATE)
63  public final class InsertStatementAssert {
64      
65      /**
66       * Assert insert statement is correct with expected parser result.
67       *
68       * @param assertContext assert context
69       * @param actual actual insert statement
70       * @param expected expected insert statement test case
71       */
72      public static void assertIs(final SQLCaseAssertContext assertContext, final InsertStatement actual, final InsertStatementTestCase expected) {
73          assertTable(assertContext, actual, expected);
74          assertInsertColumnsClause(assertContext, actual, expected);
75          assertInsertValuesClause(assertContext, actual, expected);
76          assertSetClause(assertContext, actual, expected);
77          assertInsertSelectClause(assertContext, actual, expected);
78          assertOnDuplicateKeyColumns(assertContext, actual, expected);
79          assertWithClause(assertContext, actual, expected);
80          assertOutputClause(assertContext, actual, expected);
81          assertMultiTableInsertType(assertContext, actual, expected);
82          assertMultiTableInsertIntoClause(assertContext, actual, expected);
83          assertMultiTableConditionalIntoClause(assertContext, actual, expected);
84          assertReturningClause(assertContext, actual, expected);
85          assertInsertExecClause(assertContext, actual, expected);
86          assertWithTableHintClause(assertContext, actual, expected);
87          assertRowSetFunctionClause(assertContext, actual, expected);
88      }
89      
90      private static void assertTable(final SQLCaseAssertContext assertContext, final InsertStatement actual, final InsertStatementTestCase expected) {
91          if (null == expected.getTable()) {
92              assertNull(actual.getTable(), assertContext.getText("Actual table should not exist."));
93          } else {
94              TableAssert.assertIs(assertContext, actual.getTable(), expected.getTable());
95          }
96      }
97      
98      private static void assertInsertColumnsClause(final SQLCaseAssertContext assertContext, final InsertStatement actual, final InsertStatementTestCase expected) {
99          if (null == expected.getInsertColumnsClause()) {
100             assertFalse(actual.getInsertColumns().isPresent(), assertContext.getText("Actual insert columns segment should not exist."));
101         } else {
102             assertTrue(actual.getInsertColumns().isPresent(), assertContext.getText("Actual insert columns segment should exist."));
103             InsertColumnsClauseAssert.assertIs(assertContext, actual.getInsertColumns().get(), expected.getInsertColumnsClause());
104         }
105     }
106     
107     private static void assertInsertValuesClause(final SQLCaseAssertContext assertContext, final InsertStatement actual, final InsertStatementTestCase expected) {
108         if (null == expected.getInsertValuesClause()) {
109             assertTrue(actual.getValues().isEmpty(), assertContext.getText("Actual insert values segment should not exist."));
110         } else {
111             assertFalse(actual.getValues().isEmpty(), assertContext.getText("Actual insert values segment should exist."));
112             InsertValuesClauseAssert.assertIs(assertContext, actual.getValues(), expected.getInsertValuesClause());
113         }
114     }
115     
116     private static void assertSetClause(final SQLCaseAssertContext assertContext, final InsertStatement actual, final InsertStatementTestCase expected) {
117         Optional<SetAssignmentSegment> setAssignmentSegment = actual.getSetAssignment();
118         if (null == expected.getSetClause()) {
119             assertFalse(setAssignmentSegment.isPresent(), assertContext.getText("Actual set assignment segment should not exist."));
120         } else {
121             assertTrue(setAssignmentSegment.isPresent(), assertContext.getText("Actual set assignment segment should exist."));
122             SetClauseAssert.assertIs(assertContext, setAssignmentSegment.get(), expected.getSetClause());
123         }
124     }
125     
126     private static void assertInsertSelectClause(final SQLCaseAssertContext assertContext, final InsertStatement actual, final InsertStatementTestCase expected) {
127         if (null == expected.getSelectTestCase()) {
128             assertFalse(actual.getInsertSelect().isPresent(), assertContext.getText("Actual insert select segment should not exist."));
129         } else {
130             assertTrue(actual.getInsertSelect().isPresent(), assertContext.getText("Actual insert select segment should exist."));
131             ParameterMarkerAssert.assertCount(assertContext, actual.getInsertSelect().get().getSelect().getParameterCount(), expected.getSelectTestCase().getParameters().size());
132             SelectStatementAssert.assertIs(assertContext, actual.getInsertSelect().get().getSelect(), expected.getSelectTestCase());
133         }
134     }

166     private static void assertMultiTableInsertType(final SQLCaseAssertContext assertContext, final InsertStatement actual, final InsertStatementTestCase expected) {
167         Optional<MultiTableInsertType> multiTableInsertType = actual.getMultiTableInsertType();
168         if (null == expected.getMultiTableInsertType()) {
169             assertFalse(multiTableInsertType.isPresent(), assertContext.getText("Actual multi table insert type should not exist."));
170         } else {
171             assertTrue(multiTableInsertType.isPresent(), assertContext.getText("Actual multi table insert type should exist."));
172             assertThat(assertContext.getText(String.format("`%s`'s multiTableInsertType assertion error: ", actual.getClass().getSimpleName())), multiTableInsertType.get().name(),
173                     is(expected.getMultiTableInsertType().getMultiTableInsertType()));
174         }
