18  package org.apache.shardingsphere.infra.binder.context.segment.select.projection.impl;
19  
20  import com.google.common.base.Strings;
21  import lombok.EqualsAndHashCode;
22  import lombok.Getter;
23  import lombok.RequiredArgsConstructor;
24  import lombok.Setter;
25  import lombok.ToString;
26  import org.apache.shardingsphere.infra.binder.context.segment.select.projection.Projection;
27  import org.apache.shardingsphere.infra.binder.context.segment.select.projection.extractor.ProjectionIdentifierExtractEngine;
28  import org.apache.shardingsphere.infra.database.core.metadata.database.enums.QuoteCharacter;
29  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
30  import org.apache.shardingsphere.infra.database.mysql.type.MySQLDatabaseType;
31  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.ParenthesesSegment;
32  import org.apache.shardingsphere.sql.parser.statement.core.value.identifier.IdentifierValue;
33  
34  import java.util.Optional;
35  
36  /**
37   * Common projection.
38   */
39  @RequiredArgsConstructor
40  @Getter
41  @Setter
42  @EqualsAndHashCode(exclude = {"originalTable", "originalColumn"})
43  @ToString
44  public final class ColumnProjection implements Projection {
45      
46      private final IdentifierValue owner;
47      
48      private final IdentifierValue name;
49      
50      private final IdentifierValue alias;
51      
52      private final DatabaseType databaseType;
53      
54      private final ParenthesesSegment leftParentheses;
55      
56      private final ParenthesesSegment rightParentheses;
57      
58      private IdentifierValue originalTable;
59      
60      private IdentifierValue originalColumn;
61      
62      public ColumnProjection(final String owner, final String name, final String alias, final DatabaseType databaseType) {
63          this(null == owner ? null : new IdentifierValue(owner, QuoteCharacter.NONE), new IdentifierValue(name, QuoteCharacter.NONE),
64                  null == alias ? null : new IdentifierValue(alias, QuoteCharacter.NONE), databaseType, null, null);
65      }
66      
67      public ColumnProjection(final IdentifierValue owner, final IdentifierValue name, final IdentifierValue alias, final DatabaseType databaseType) {
68          this(owner, name, alias, databaseType, null, null);
69      }
70      
71      @Override
72      public String getColumnName() {
73          ProjectionIdentifierExtractEngine extractEngine = new ProjectionIdentifierExtractEngine(databaseType);
74          return databaseType instanceof MySQLDatabaseType ? extractEngine.getIdentifierValue(name) : getColumnLabel();
75      }
76      
77      @Override
78      public String getColumnLabel() {
79          ProjectionIdentifierExtractEngine extractEngine = new ProjectionIdentifierExtractEngine(databaseType);
80          return getAlias().isPresent() ? extractEngine.getIdentifierValue(getAlias().get()) : extractEngine.getIdentifierValue(name);
81      }
82      
83      @Override
84      public String getExpression() {
85          return null == owner ? name.getValue() : owner.getValue() + "." + name.getValue();
86      }
87      
88      @Override
89      public Optional<IdentifierValue> getAlias() {
90          return Optional.ofNullable(alias);
91      }
92      
93      /**
94       * Get owner.
95       *
96       * @return owner
97       */
98      public Optional<IdentifierValue> getOwner() {
99          return Optional.ofNullable(owner);
100     }
101     
102     /**
103      * Get original table.
104      *
105      * @return original table
106      */
107     public IdentifierValue getOriginalTable() {
108         if (null == originalTable || Strings.isNullOrEmpty(originalTable.getValue())) {
109             return null == owner ? new IdentifierValue("") : owner;
110         }
111         return originalTable;
112     }
113     
114     /**
115      * Get original column.
116      *
117      * @return original column
118      */
119     public IdentifierValue getOriginalColumn() {
120         return null == originalColumn || Strings.isNullOrEmpty(originalColumn.getValue()) ? name : originalColumn;
121     }
122     
123     /**
124      * Get left parentheses.
125      *
126      * @return left parentheses
127      */
128     public Optional<ParenthesesSegment> getLeftParentheses() {
129         return Optional.ofNullable(leftParentheses);
130     }
131     
132     /**
133      * Get right parentheses.
134      *
135      * @return right parentheses
136      */
137     public Optional<ParenthesesSegment> getRightParentheses() {
138         return Optional.ofNullable(rightParentheses);
139     }
140 }
