45  /**
46   * Sharding DML statement validator.
47   */
48  public abstract class ShardingDMLStatementValidator implements ShardingStatementValidator {
49      
50      /**
51       * Validate multiple table.
52       *
53       * @param shardingRule sharding rule
54       * @param sqlStatementContext sqlStatementContext
55       */
56      protected void validateMultipleTable(final ShardingRule shardingRule, final SQLStatementContext sqlStatementContext) {
57          Collection<String> tableNames = ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames();
58          boolean isAllShardingTables = shardingRule.isAllShardingTables(tableNames) && (1 == tableNames.size() || shardingRule.isAllBindingTables(tableNames));
59          boolean isAllSingleTables = !shardingRule.containsShardingTable(tableNames);
60          ShardingSpherePreconditions.checkState(isAllShardingTables || isAllSingleTables, () -> new DMLWithMultipleShardingTablesException(tableNames));
61      }
62      
63      /**
64       * Judge whether is same route context or not.
65       * 
66       * @param routeContext route context
67       * @param subRouteContext  sub route context
68       * @return whether is same route context or not
69       */
70      protected boolean isSameRouteContext(final RouteContext routeContext, final RouteContext subRouteContext) {
71          if (routeContext.getRouteUnits().size() != subRouteContext.getRouteUnits().size()) {
72              return false;
73          }
74          Iterator<RouteUnit> routeContextIterator = routeContext.getRouteUnits().iterator();
75          Iterator<RouteUnit> setAssignmentRouteContextIterator = subRouteContext.getRouteUnits().iterator();
76          while (routeContextIterator.hasNext()) {
77              RouteUnit routeUnit = routeContextIterator.next();
78              RouteUnit setAssignmentRouteUnit = setAssignmentRouteContextIterator.next();
79              if (!routeUnit.getDataSourceMapper().getLogicName().equals(setAssignmentRouteUnit.getDataSourceMapper().getLogicName())
80                      || !routeUnit.getDataSourceMapper().getActualName().equals(setAssignmentRouteUnit.getDataSourceMapper().getActualName())) {
81                  return false;
82              }
83              if (!isSameTableMapper(routeUnit.getTableMappers(), setAssignmentRouteUnit.getTableMappers())) {
84                  return false;
85              }
86          }
87          return true;
88      }
89      
90      private boolean isSameTableMapper(final Collection<RouteMapper> tableMappers, final Collection<RouteMapper> setAssignmentTableMappers) {
91          if (tableMappers.size() != setAssignmentTableMappers.size()) {
92              return false;
93          }
94          Iterator<RouteMapper> tableMapperIterator = tableMappers.iterator();
95          Iterator<RouteMapper> setAssignmentTableMapperIterator = setAssignmentTableMappers.iterator();
96          while (tableMapperIterator.hasNext()) {
97              RouteMapper routeMapper = tableMapperIterator.next();
98              RouteMapper setAssignmentRouteMapper = setAssignmentTableMapperIterator.next();
99              if (!routeMapper.getLogicName().equals(setAssignmentRouteMapper.getLogicName())
100                     || !routeMapper.getActualName().equals(setAssignmentRouteMapper.getActualName())) {
101                 return false;
102             }
103         }
104         return true;
105     }
106     
107     /**
108      * Create shardingConditions.
109      * 
110      * @param sqlStatementContext SQL statement context
111      * @param shardingRule shardingRule
112      * @param assignments assignments
113      * @param params parameters
114      * @return sharding conditions
115      */
116     @SuppressWarnings({"rawtypes", "unchecked"})
117     protected Optional<ShardingConditions> createShardingConditions(final SQLStatementContext sqlStatementContext, final ShardingRule shardingRule,
118                                                                     final Collection<ColumnAssignmentSegment> assignments, final List<Object> params) {
119         Collection<ShardingConditionValue> values = new LinkedList<>();
120         String tableName = ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames().iterator().next();
121         for (ColumnAssignmentSegment each : assignments) {
122             String shardingColumn = each.getColumns().get(0).getIdentifier().getValue();
123             if (shardingRule.findShardingColumn(shardingColumn, tableName).isPresent()) {
124                 Optional<Object> assignmentValue = getShardingColumnAssignmentValue(each, params);
125                 assignmentValue.ifPresent(optional -> values.add(new ListShardingConditionValue(shardingColumn, tableName, Collections.singletonList(optional))));
126             }
127         }
128         if (values.isEmpty()) {
129             return Optional.empty();
130         }
131         ShardingCondition shardingCondition = new ShardingCondition();
132         shardingCondition.getValues().addAll(values);
133         return Optional.of(new ShardingConditions(Collections.singletonList(shardingCondition), sqlStatementContext, shardingRule));
134     }
135     
136     private Optional<Object> getShardingColumnAssignmentValue(final ColumnAssignmentSegment assignmentSegment, final List<Object> params) {
137         ExpressionSegment segment = assignmentSegment.getValue();
138         int shardingSetAssignIndex = -1;
139         if (segment instanceof ParameterMarkerExpressionSegment) {
140             shardingSetAssignIndex = ((ParameterMarkerExpressionSegment) segment).getParameterMarkerIndex();
141         }
142         if (segment instanceof LiteralExpressionSegment) {
143             return Optional.of(((LiteralExpressionSegment) segment).getLiterals());
144         }
145         if (-1 == shardingSetAssignIndex || shardingSetAssignIndex > params.size() - 1) {
146             return Optional.empty();
147         }
148         return Optional.of(params.get(shardingSetAssignIndex));
149     }
