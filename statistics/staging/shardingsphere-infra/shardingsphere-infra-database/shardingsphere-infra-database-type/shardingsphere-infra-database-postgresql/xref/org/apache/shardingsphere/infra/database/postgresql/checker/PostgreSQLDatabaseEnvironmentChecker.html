18  package org.apache.shardingsphere.infra.database.postgresql.checker;
19  
20  import lombok.extern.slf4j.Slf4j;
21  import org.apache.shardingsphere.infra.database.core.checker.DialectDatabaseEnvironmentChecker;
22  import org.apache.shardingsphere.infra.database.core.checker.PrivilegeCheckType;
23  import org.apache.shardingsphere.infra.database.core.exception.CheckDatabaseEnvironmentFailedException;
24  import org.apache.shardingsphere.infra.database.core.exception.MissingRequiredPrivilegeException;
25  import org.apache.shardingsphere.infra.database.core.exception.MissingRequiredUserException;
26  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
27  
28  import javax.sql.DataSource;
29  import java.sql.Connection;
30  import java.sql.DatabaseMetaData;
31  import java.sql.PreparedStatement;
32  import java.sql.ResultSet;
33  import java.sql.SQLException;
34  import java.util.Collections;
35  
36  /**
37   * Database environment checker for PostgreSQL.
38   */
39  @Slf4j
40  public final class PostgreSQLDatabaseEnvironmentChecker implements DialectDatabaseEnvironmentChecker {
41      
42      private static final String SHOW_GRANTS_SQL = "SELECT * FROM pg_roles WHERE rolname = ?";
43      
44      @Override
45      public void checkPrivilege(final DataSource dataSource, final PrivilegeCheckType privilegeCheckType) {
46          if (PrivilegeCheckType.PIPELINE != privilegeCheckType) {
47              return;
48          }
49          try (
50                  Connection connection = dataSource.getConnection();
51                  PreparedStatement preparedStatement = connection.prepareStatement(SHOW_GRANTS_SQL)) {
52              DatabaseMetaData metaData = connection.getMetaData();
53              preparedStatement.setString(1, metaData.getUserName());
54              try (ResultSet resultSet = preparedStatement.executeQuery()) {
55                  String username = metaData.getUserName();
56                  ShardingSpherePreconditions.checkState(resultSet.next(), () -> new MissingRequiredUserException(username));
57                  String isSuperRole = resultSet.getString("rolsuper");
58                  String isReplicationRole = resultSet.getString("rolreplication");
59                  log.info("checkPrivilege: isSuperRole: {}, isReplicationRole: {}", isSuperRole, isReplicationRole);
60                  ShardingSpherePreconditions.checkState("t".equalsIgnoreCase(isSuperRole) || "t".equalsIgnoreCase(isReplicationRole),
61                          () -> new MissingRequiredPrivilegeException(Collections.singleton("REPLICATION")));
62              }
63          } catch (final SQLException ex) {
64              throw new CheckDatabaseEnvironmentFailedException(ex);
65          }
66      }
67      
68      @Override
69      public void checkVariable(final DataSource dataSource) {
70      }
71      
72      @Override
73      public String getDatabaseType() {
74          return "PostgreSQL";
75      }
76  }
