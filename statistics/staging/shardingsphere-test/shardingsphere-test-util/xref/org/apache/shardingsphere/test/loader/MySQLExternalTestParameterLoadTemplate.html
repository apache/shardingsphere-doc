18  package org.apache.shardingsphere.test.loader;
19  
20  import java.util.ArrayList;
21  import java.util.Collection;
22  import java.util.LinkedList;
23  import java.util.List;
24  import java.util.Objects;
25  
26  /**
27   * External test parameter load template for MySQL.
28   */
29  public final class MySQLExternalTestParameterLoadTemplate implements TestParameterLoadTemplate {
30      
31      private static final int DELIMITER_COMMAND_LENGTH = "DELIMITER".length();
32      
33      @Override
34      public Collection<ExternalSQLTestParameter> load(final String sqlCaseFileName, final List<String> sqlCaseFileContent,
35                                                       final List<String> resultFileContent, final String databaseType, final String reportType) {
36          Collection<ExternalSQLTestParameter> result = new LinkedList<>();
37          List<String> lines = new ArrayList<>();
38          String delimiter = ";";
39          for (int i = 0; i < sqlCaseFileContent.size(); i++) {
40              String line = sqlCaseFileContent.get(i).trim();
41              if (line.isEmpty() || lines.isEmpty() && SQLLineComment.isComment(line)) {
42                  continue;
43              }
44              if (lines.isEmpty() && line.toUpperCase().startsWith("DELIMITER")) {
45                  delimiter = getNewDelimiter(line, delimiter);
46                  continue;
47              }
48              lines.add(line);
49              if (line.endsWith(delimiter)) {
50                  if (resultFileContent.isEmpty() || existCorrectResultContent(resultFileContent, lines)) {
51                      String sqlCaseId = sqlCaseFileName + ":" + (i + 1);
52                      String sql = String.join("\n", lines);
53                      sql = sql.substring(0, sql.length() - delimiter.length());
54                      result.add(new ExternalSQLTestParameter(sqlCaseId, databaseType, sql, reportType));
55                  }
56                  lines.clear();
57              }
58          }
59          return result;
60      }
61      
62      private String getNewDelimiter(final String sql, final String delimiter) {
63          String newDelimiter = sql.substring(DELIMITER_COMMAND_LENGTH, sql.endsWith(delimiter) ? sql.length() - delimiter.length() : sql.length()).trim();
64          if (newDelimiter.startsWith("\"") && newDelimiter.endsWith("\"") || newDelimiter.startsWith("'") && newDelimiter.endsWith("'")) {
65              newDelimiter = newDelimiter.substring(1, newDelimiter.length() - 1);
66          }
67          return newDelimiter.isEmpty() ? delimiter : newDelimiter;
68      }
69      
70      private boolean existCorrectResultContent(final List<String> resultLines, final List<String> sqlLines) {
71          int nextLineIndex = findSQLNextLineIndex(resultLines, sqlLines);
72          return -1 != nextLineIndex && (nextLineIndex == resultLines.size() || !resultLines.get(nextLineIndex).contains("ERROR"));
73      }
74      
75      private int findSQLNextLineIndex(final List<String> resultLines, final List<String> sqlLines) {
76          int completedSQLIndex = 0;
77          for (int resultIndex = 0; resultIndex < resultLines.size(); resultIndex++) {
78              if (Objects.equals(sqlLines.get(completedSQLIndex), resultLines.get(resultIndex).trim())) {
79                  if (++completedSQLIndex == sqlLines.size()) {
80                      return resultIndex + 1;
81                  }
82              } else {
83                  completedSQLIndex = 0;
84              }
85          }
86          return -1;
87      }
88  }
