51  /**
52   * Single table inventory data consistency checker.
53   */
54  @Slf4j
55  @RequiredArgsConstructor
56  public final class SingleTableInventoryDataConsistencyChecker {
57      
58      private final String jobId;
59      
60      private final PipelineDataSourceWrapper sourceDataSource;
61      
62      private final PipelineDataSourceWrapper targetDataSource;
63      
64      private final SchemaTableName sourceTable;
65      
66      private final SchemaTableName targetTable;
67      
68      private final List<String> columnNames;
69      
70      private final PipelineColumnMetaData uniqueKey;
71      
72      private final JobRateLimitAlgorithm readRateLimitAlgorithm;
73      
74      private final ConsistencyCheckJobItemProgressContext progressContext;
75      
76      /**
77       * Data consistency check.
78       *
79       * @param calculateAlgorithm calculate algorithm
80       * @return data consistency check result
81       */
82      public DataConsistencyCheckResult check(final DataConsistencyCalculateAlgorithm calculateAlgorithm) {
83          ThreadFactory threadFactory = ExecutorThreadFactoryBuilder.build("job-" + getJobIdDigest(jobId) + "-check-%d");
84          ThreadPoolExecutor executor = new ThreadPoolExecutor(2, 2, 60, TimeUnit.SECONDS, new ArrayBlockingQueue<>(2), threadFactory);
85          try {
86              return check(calculateAlgorithm, executor);
87          } finally {
88              executor.shutdown();
89              executor.shutdownNow();
90          }
91      }
92      
93      private DataConsistencyCheckResult check(final DataConsistencyCalculateAlgorithm calculateAlgorithm, final ThreadPoolExecutor executor) {
94          String sourceDatabaseType = sourceDataSource.getDatabaseType().getType();
95          String targetDatabaseType = targetDataSource.getDatabaseType().getType();
96          String schemaName = sourceTable.getSchemaName().getOriginal();
97          String sourceTableName = sourceTable.getTableName().getOriginal();
98          Map<String, Object> tableCheckPositions = progressContext.getTableCheckPositions();
99          DataConsistencyCalculateParameter sourceParam = buildParameter(
100                 sourceDataSource, schemaName, sourceTableName, columnNames, sourceDatabaseType, targetDatabaseType, uniqueKey, tableCheckPositions.get(sourceTableName));
101         String targetTableName = targetTable.getTableName().getOriginal();
102         DataConsistencyCalculateParameter targetParam = buildParameter(targetDataSource, targetTable.getSchemaName().getOriginal(), targetTableName,
103                 columnNames, targetDatabaseType, sourceDatabaseType, uniqueKey, tableCheckPositions.get(targetTableName));
104         Iterator<DataConsistencyCalculatedResult> sourceCalculatedResults = waitFuture(executor.submit(() -> calculateAlgorithm.calculate(sourceParam))).iterator();
105         Iterator<DataConsistencyCalculatedResult> targetCalculatedResults = waitFuture(executor.submit(() -> calculateAlgorithm.calculate(targetParam))).iterator();
106         try {
107             return check0(sourceCalculatedResults, targetCalculatedResults, executor);
108         } finally {
109             CloseUtils.closeQuietly(sourceParam.getCalculationContext());
110             CloseUtils.closeQuietly(targetParam.getCalculationContext());
111         }
112     }
113     
114     private DataConsistencyCheckResult check0(final Iterator<DataConsistencyCalculatedResult> sourceCalculatedResults, final Iterator<DataConsistencyCalculatedResult> targetCalculatedResults,
115                                               final ThreadPoolExecutor executor) {
116         long sourceRecordsCount = 0;
117         long targetRecordsCount = 0;
118         boolean contentMatched = true;
119         while (sourceCalculatedResults.hasNext() && targetCalculatedResults.hasNext()) {
120             if (null != readRateLimitAlgorithm) {
121                 readRateLimitAlgorithm.intercept(JobOperationType.SELECT, 1);
122             }
123             DataConsistencyCalculatedResult sourceCalculatedResult = waitFuture(executor.submit(sourceCalculatedResults::next));
124             DataConsistencyCalculatedResult targetCalculatedResult = waitFuture(executor.submit(targetCalculatedResults::next));
125             sourceRecordsCount += sourceCalculatedResult.getRecordsCount();
126             targetRecordsCount += targetCalculatedResult.getRecordsCount();
127             contentMatched = Objects.equals(sourceCalculatedResult, targetCalculatedResult);
128             if (!contentMatched) {
129                 log.info("content matched false, jobId={}, sourceTable={}, targetTable={}, uniqueKey={}", jobId, sourceTable, targetTable, uniqueKey);
130                 break;
131             }
132             if (sourceCalculatedResult.getMaxUniqueKeyValue().isPresent()) {
133                 progressContext.getTableCheckPositions().put(sourceTable.getTableName().getOriginal(), sourceCalculatedResult.getMaxUniqueKeyValue().get());
134             }
135             if (targetCalculatedResult.getMaxUniqueKeyValue().isPresent()) {
136                 progressContext.getTableCheckPositions().put(targetTable.getTableName().getOriginal(), targetCalculatedResult.getMaxUniqueKeyValue().get());
137             }
138             progressContext.onProgressUpdated(new PipelineJobProgressUpdatedParameter(sourceCalculatedResult.getRecordsCount()));
139         }
140         return new DataConsistencyCheckResult(new DataConsistencyCountCheckResult(sourceRecordsCount, targetRecordsCount), new DataConsistencyContentCheckResult(contentMatched));
141     }
142     
143     // TODO use digest (crc32, murmurhash)
144     private String getJobIdDigest(final String jobId) {
145         return jobId.length() <= 6 ? jobId : jobId.substring(0, 6);
146     }
147     
148     private DataConsistencyCalculateParameter buildParameter(final PipelineDataSourceWrapper sourceDataSource,
149                                                              final String schemaName, final String tableName, final List<String> columnNames,
150                                                              final String sourceDatabaseType, final String targetDatabaseType, final PipelineColumnMetaData uniqueKey,
151                                                              final Object tableCheckPosition) {
152         return new DataConsistencyCalculateParameter(sourceDataSource, schemaName, tableName, columnNames, sourceDatabaseType, targetDatabaseType, uniqueKey, tableCheckPosition);
153     }
154     
155     private <T> T waitFuture(final Future<T> future) {
156         try {
157             return future.get();
158         } catch (final InterruptedException | ExecutionException ex) {
159             if (ex.getCause() instanceof PipelineSQLException) {
160                 throw (PipelineSQLException) ex.getCause();
161             }
162             throw new SQLWrapperException(new SQLException(ex));
163         }
164     }
