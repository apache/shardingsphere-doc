64  /**
65   * Updatable RAL statement assert.
66   */
67  @NoArgsConstructor(access = AccessLevel.PRIVATE)
68  public final class UpdatableRALStatementAssert {
69      
70      /**
71       * Assert updatable RAL statement is correct with expected parser result.
72       *
73       * @param assertContext assert context
74       * @param actual actual updatable RAL statement
75       * @param expected expected updatable RAL statement test case
76       */
77      public static void assertIs(final SQLCaseAssertContext assertContext, final UpdatableRALStatement actual, final SQLParserTestCase expected) {
78          if (actual instanceof LabelComputeNodeStatement) {
79              LabelComputeNodeStatementAssert.assertIs(assertContext, (LabelComputeNodeStatement) actual, (LabelComputeNodeStatementTestCase) expected);
80          } else if (actual instanceof UnlabelComputeNodeStatement) {
81              UnlabelComputeNodeStatementAssert.assertIs(assertContext, (UnlabelComputeNodeStatement) actual, (UnlabelComputeNodeStatementTestCase) expected);
82          } else if (actual instanceof AlterComputeNodeStatement) {
83              AlterComputeNodeStatementAssert.assertIs(assertContext, (AlterComputeNodeStatement) actual, (AlterComputeNodeStatementTestCase) expected);
84          } else if (actual instanceof SetDistVariableStatement) {
85              SetDistVariableStatementAssert.assertIs(assertContext, (SetDistVariableStatement) actual, (SetDistVariableStatementTestCase) expected);
86          } else if (actual instanceof RefreshTableMetaDataStatement) {
87              RefreshTableMetaDataStatementAssert.assertIs(assertContext, (RefreshTableMetaDataStatement) actual, (RefreshTableMetaDataStatementTestCase) expected);
88          } else if (actual instanceof AlterSQLParserRuleStatement) {
89              AlterSQLParserRuleStatementAssert.assertIs(assertContext, (AlterSQLParserRuleStatement) actual, (AlterSQLParserRuleStatementTestCase) expected);
90          } else if (actual instanceof AlterTrafficRuleStatement) {
91              AlterTrafficRuleStatementAssert.assertIs(assertContext, (AlterTrafficRuleStatement) actual, (AlterTrafficRuleStatementTestCase) expected);
92          } else if (actual instanceof ImportDatabaseConfigurationStatement) {
93              ImportDatabaseConfigurationStatementAssert.assertIs(assertContext, (ImportDatabaseConfigurationStatement) actual, (ImportDatabaseConfigurationStatementTestCase) expected);
94          } else if (actual instanceof ImportMetaDataStatement) {
95              ImportMetaDataStatementAssert.assertIs(assertContext, (ImportMetaDataStatement) actual, (ImportMetaDataStatementTestCase) expected);
96          } else if (actual instanceof AlterTransactionRuleStatement) {
97              ExistingAssert.assertIs(assertContext, actual, expected);
98          } else if (actual instanceof AlterReadwriteSplittingStorageUnitStatusStatement) {
99              AlterReadwriteSplittingStorageUnitStatusStatementAssert.assertIs(assertContext,
100                     (AlterReadwriteSplittingStorageUnitStatusStatement) actual, (AlterReadwriteSplittingStorageUnitStatusStatementTestCase) expected);
101         } else if (actual instanceof LockClusterStatement) {
102             LockClusterStatementAssert.assertIs(assertContext, (LockClusterStatement) actual, (LockClusterStatementTestCase) expected);
103         } else if (actual instanceof AlterGlobalClockRuleStatement) {
104             AlterGlobalClockRuleStatementAssert.assertIs(assertContext, (AlterGlobalClockRuleStatement) actual, (AlterGlobalClockRuleStatementTestCase) expected);
105         }
106     }
107 }
