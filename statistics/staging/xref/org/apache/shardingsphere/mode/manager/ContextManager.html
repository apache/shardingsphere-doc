76      
77      private final MetaDataContextManager metaDataContextManager;

79      public ContextManager(final MetaDataContexts metaDataContexts, final ComputeNodeInstanceContext computeNodeInstanceContext, final PersistRepository repository) {
80          this.metaDataContexts = new AtomicReference<>(metaDataContexts);
81          this.computeNodeInstanceContext = computeNodeInstanceContext;
82          metaDataContextManager = new MetaDataContextManager(this.metaDataContexts, computeNodeInstanceContext, repository);
83          persistServiceFacade = new PersistServiceFacade(repository, computeNodeInstanceContext.getModeConfiguration(), metaDataContextManager);
84          stateContext = new StateContext(persistServiceFacade.getStatePersistService().loadClusterState().orElse(ClusterState.OK));
85          executorEngine = ExecutorEngine.createExecutorEngineWithSize(metaDataContexts.getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.KERNEL_EXECUTOR_SIZE));
86          for (ContextManagerLifecycleListener each : ShardingSphereServiceLoader.getServiceInstances(ContextManagerLifecycleListener.class)) {
87              each.onInitialized(this);
88          }
89      }

109     /**
110      * Get database.
111      *
112      * @param name database name
113      * @return got database
114      */
115     public ShardingSphereDatabase getDatabase(final String name) {
116         ShardingSpherePreconditions.checkNotEmpty(name, NoDatabaseSelectedException::new);
117         ShardingSphereMetaData metaData = getMetaDataContexts().getMetaData();
118         ShardingSpherePreconditions.checkState(metaData.containsDatabase(name), () -> new UnknownDatabaseException(name));
119         return metaData.getDatabase(name);
120     }

132     /**
133      * Reload database meta data.
134      *
135      * @param database to be reloaded database
136      * @param force whether to force refresh table metadata
137      */
138     public void refreshDatabaseMetaData(final ShardingSphereDatabase database, final boolean force) {
139         try {
140             MetaDataContexts reloadedMetaDataContexts = createMetaDataContexts(database);
141             MetaDataPersistService persistService = persistServiceFacade.getMetaDataPersistService();
142             if (force) {
143                 metaDataContexts.set(reloadedMetaDataContexts);
144                 metaDataContexts.get().getMetaData().getDatabase(database.getName()).getSchemas()
145                         .forEach((schemaName, schema) -> persistService.getDatabaseMetaDataService().persistByAlterConfiguration(database.getName(), schemaName, schema));
146             } else {
147                 deletedSchemaNames(database.getName(), reloadedMetaDataContexts.getMetaData().getDatabase(database.getName()), database);
148                 metaDataContexts.set(reloadedMetaDataContexts);
149                 metaDataContexts.get().getMetaData().getDatabase(database.getName()).getSchemas()
150                         .forEach((schemaName, schema) -> persistService.getDatabaseMetaDataService().compareAndPersist(database.getName(), schemaName, schema));
151             }
152         } catch (final SQLException ex) {
153             log.error("Refresh database meta data: {} failed", database.getName(), ex);
154         }
155     }
156     
157     /**
158      * Reload table meta data.
159      *
160      * @param database to be reloaded database
161      */
162     public void refreshTableMetaData(final ShardingSphereDatabase database) {
163         try {
164             MetaDataContexts reloadedMetaDataContexts = createMetaDataContexts(database);
165             deletedSchemaNames(database.getName(), reloadedMetaDataContexts.getMetaData().getDatabase(database.getName()), database);
166             metaDataContexts.set(reloadedMetaDataContexts);
167             metaDataContexts.get().getMetaData().getDatabase(database.getName()).getSchemas()
168                     .forEach((schemaName, schema) -> persistServiceFacade.getMetaDataPersistService().getDatabaseMetaDataService().compareAndPersist(database.getName(), schemaName, schema));
169         } catch (final SQLException ex) {
170             log.error("Refresh table meta data: {} failed", database.getName(), ex);
171         }
172     }
173     
174     private MetaDataContexts createMetaDataContexts(final ShardingSphereDatabase database) throws SQLException {
175         MetaDataPersistService metaDataPersistService = persistServiceFacade.getMetaDataPersistService();
176         ConfigurationManager configurationManager = metaDataContextManager.getConfigurationManager();
177         Map<String, DataSourcePoolProperties> dataSourcePoolPropsFromRegCenter = metaDataPersistService.getDataSourceUnitService().load(database.getName());
178         SwitchingResource switchingResource = metaDataContextManager.getResourceSwitchManager().alterStorageUnit(database.getResourceMetaData(), dataSourcePoolPropsFromRegCenter);
179         Collection<RuleConfiguration> ruleConfigs = metaDataPersistService.getDatabaseRulePersistService().load(database.getName());
180         Map<String, ShardingSphereDatabase> changedDatabases = configurationManager.createChangedDatabases(database.getName(), false, switchingResource, ruleConfigs);
181         ConfigurationProperties props = new ConfigurationProperties(metaDataPersistService.getPropsService().load());
182         Collection<RuleConfiguration> globalRuleConfigs = metaDataPersistService.getGlobalRuleService().load();
183         RuleMetaData changedGlobalMetaData = new RuleMetaData(GlobalRulesBuilder.buildRules(globalRuleConfigs, changedDatabases, props));
184         MetaDataContexts result = MetaDataContextsFactory.create(metaDataPersistService,
185                 new ShardingSphereMetaData(changedDatabases, metaDataContexts.get().getMetaData().getGlobalResourceMetaData(), changedGlobalMetaData, props));
186         switchingResource.closeStaleDataSources();
187         return result;
188     }

190     /**
191      * Delete schema names.
192      *
193      * @param databaseName database name
194      * @param reloadDatabase reload database
195      * @param currentDatabase current database
196      */
197     public void deletedSchemaNames(final String databaseName, final ShardingSphereDatabase reloadDatabase, final ShardingSphereDatabase currentDatabase) {
198         GenericSchemaManager.getToBeDeletedSchemaNames(reloadDatabase.getSchemas(), currentDatabase.getSchemas()).keySet()
199                 .forEach(each -> persistServiceFacade.getMetaDataPersistService().getDatabaseMetaDataService().dropSchema(databaseName, each));
200     }
201     
202     /**
203      * Reload schema.
204      *
205      * @param database database
206      * @param schemaName to be reloaded schema name
207      * @param dataSourceName data source name
208      */
209     public void reloadSchema(final ShardingSphereDatabase database, final String schemaName, final String dataSourceName) {
210         try {
211             ShardingSphereSchema reloadedSchema = loadSchema(database, schemaName, dataSourceName);
212             if (reloadedSchema.getTables().isEmpty()) {
213                 database.dropSchema(schemaName);
214                 persistServiceFacade.getMetaDataPersistService().getDatabaseMetaDataService().dropSchema(database.getName(),
215                         schemaName);
216             } else {
217                 database.addSchema(schemaName, reloadedSchema);
218                 persistServiceFacade.getMetaDataPersistService().getDatabaseMetaDataService()
219                         .compareAndPersist(database.getName(), schemaName, reloadedSchema);
220             }
221         } catch (final SQLException ex) {
222             log.error("Reload meta data of database: {} schema: {} with data source: {} failed", database.getName(), schemaName, dataSourceName, ex);
223         }
224     }
225     
226     private ShardingSphereSchema loadSchema(final ShardingSphereDatabase database, final String schemaName, final String dataSourceName) throws SQLException {
227         database.reloadRules();
228         GenericSchemaBuilderMaterial material = new GenericSchemaBuilderMaterial(database.getProtocolType(),
229                 Collections.singletonMap(dataSourceName, database.getResourceMetaData().getStorageUnits().get(dataSourceName).getStorageType()),
230                 Collections.singletonMap(dataSourceName, database.getResourceMetaData().getStorageUnits().get(dataSourceName).getDataSource()),
231                 database.getRuleMetaData().getRules(), metaDataContexts.get().getMetaData().getProps(), schemaName);
232         ShardingSphereSchema result = GenericSchemaBuilder.build(material).get(schemaName);
233         result.getViews().putAll(persistServiceFacade.getMetaDataPersistService().getDatabaseMetaDataService().getViewMetaDataPersistService().load(database.getName(), schemaName));
234         return result;
235     }

237     /**
238      * Reload table.
239      *
240      * @param database database
241      * @param schemaName schema name
242      * @param tableName to be reloaded table name
243      */
244     public void reloadTable(final ShardingSphereDatabase database, final String schemaName, final String tableName) {
245         GenericSchemaBuilderMaterial material = new GenericSchemaBuilderMaterial(database.getProtocolType(),
246                 database.getResourceMetaData().getStorageUnits(), database.getRuleMetaData().getRules(), metaDataContexts.get().getMetaData().getProps(), schemaName);
247         try {
248             persistTable(database, schemaName, tableName, material);
249         } catch (final SQLException ex) {
250             log.error("Reload table: {} meta data of database: {} schema: {} failed", tableName, database.getName(), schemaName, ex);
251         }
252     }
253     
254     /**
255      * Reload table from single data source.
256      *
257      * @param database database
258      * @param schemaName schema name
259      * @param dataSourceName data source name
260      * @param tableName to be reloaded table name
261      */
262     public void reloadTable(final ShardingSphereDatabase database, final String schemaName, final String dataSourceName, final String tableName) {
263         StorageUnit storageUnit = database.getResourceMetaData().getStorageUnits().get(dataSourceName);
264         GenericSchemaBuilderMaterial material = new GenericSchemaBuilderMaterial(database.getProtocolType(),
265                 Collections.singletonMap(dataSourceName, storageUnit.getStorageType()), Collections.singletonMap(dataSourceName, storageUnit.getDataSource()),
266                 database.getRuleMetaData().getRules(), metaDataContexts.get().getMetaData().getProps(), schemaName);
267         try {
268             persistTable(database, schemaName, tableName, material);
269         } catch (final SQLException ex) {
270             log.error("Reload table: {} meta data of database: {} schema: {} with data source: {} failed", tableName, database.getName(), schemaName, dataSourceName, ex);
271         }
272     }
273     
274     private void persistTable(final ShardingSphereDatabase database, final String schemaName, final String tableName, final GenericSchemaBuilderMaterial material) throws SQLException {
275         ShardingSphereSchema schema = GenericSchemaBuilder.build(Collections.singleton(tableName), material).getOrDefault(schemaName, new ShardingSphereSchema());
276         persistServiceFacade.getMetaDataPersistService().getDatabaseMetaDataService().getTableMetaDataPersistService()
277                 .persist(database.getName(), schemaName, Collections.singletonMap(tableName, schema.getTable(tableName)));
278     }

280     /**
281      * Get pre-selected database name.
282      *
283      * @return pre-selected database name
284      */
285     public String getPreSelectedDatabaseName() {
286         return InstanceType.JDBC == computeNodeInstanceContext.getInstance().getMetaData().getType() ? metaDataContexts.get().getMetaData().getDatabases().keySet().iterator().next() : null;
287     }
288     
289     @Override
290     public void close() {
291         for (ContextManagerLifecycleListener each : ShardingSphereServiceLoader.getServiceInstances(ContextManagerLifecycleListener.class)) {
292             each.onDestroyed(this);
293         }
294         executorEngine.close();
295         metaDataContexts.get().close();
296         persistServiceFacade.getRepository().close();
297     }
