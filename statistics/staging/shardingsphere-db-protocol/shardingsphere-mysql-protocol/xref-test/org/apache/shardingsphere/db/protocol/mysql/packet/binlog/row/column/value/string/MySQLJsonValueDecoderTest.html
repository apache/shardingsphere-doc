240     private ByteBuf mockJsonObjectByteBuf(final List<JsonEntry> jsonEntries, final boolean isSmall) {
241         ByteBuf result = Unpooled.buffer();
242         result.writeByte(isSmall ? JsonValueTypes.SMALL_JSON_OBJECT : JsonValueTypes.LARGE_JSON_OBJECT);
243         result.writeBytes(mockJsonObjectByteBufValue(jsonEntries, isSmall));
244         return result;
245     }
246     
247     private ByteBuf mockJsonObjectByteBufValue(final List<JsonEntry> jsonEntries, final boolean isSmall) {
248         ByteBuf result = Unpooled.buffer();
249         writeInt(result, jsonEntries.size(), isSmall);
250         writeInt(result, 0, isSmall);
251         int startOffset = isSmall
252                 ? 1 + SMALL_JSON_INT_LENGTH + SMALL_JSON_INT_LENGTH + jsonEntries.size() * SMALL_JSON_KEY_META_DATA_LENGTH + jsonEntries.size() * SMALL_JSON_VALUE_META_DATA_LENGTH - 1
253                 : 1 + LARGE_JSON_INT_LENGTH + LARGE_JSON_INT_LENGTH + jsonEntries.size() * LARGE_JSON_KEY_META_DATA_LENGTH + jsonEntries.size() * LARGE_JSON_VALUE_META_DATA_LENGTH - 1;
254         ByteBuf keyByteBuf = writeKeys(result, jsonEntries, startOffset, isSmall);
255         startOffset += keyByteBuf.readableBytes();
256         ByteBuf valueByteBuf = writeValues(result, jsonEntries, startOffset, isSmall);
257         result.writeBytes(keyByteBuf);
258         result.writeBytes(valueByteBuf);
259         return result;
260     }

262     private ByteBuf mockJsonArrayByteBuf(final List<JsonEntry> jsonEntries, final boolean isSmall) {
263         ByteBuf result = Unpooled.buffer();
264         result.writeByte(isSmall ? JsonValueTypes.SMALL_JSON_ARRAY : JsonValueTypes.LARGE_JSON_ARRAY);
265         result.writeBytes(mockJsonArrayByteBufValue(jsonEntries, isSmall));
266         return result;

269     private ByteBuf mockJsonArrayByteBufValue(final List<JsonEntry> jsonEntries, final boolean isSmall) {
270         ByteBuf result = Unpooled.buffer();
271         writeInt(result, jsonEntries.size(), isSmall);
272         writeInt(result, 0, isSmall);
273         int startOffset = isSmall
274                 ? 1 + SMALL_JSON_INT_LENGTH + SMALL_JSON_INT_LENGTH + jsonEntries.size() * SMALL_JSON_VALUE_META_DATA_LENGTH - 1
275                 : 1 + LARGE_JSON_INT_LENGTH + LARGE_JSON_INT_LENGTH + jsonEntries.size() * LARGE_JSON_VALUE_META_DATA_LENGTH - 1;
276         ByteBuf valueByteBuf = writeValues(result, jsonEntries, startOffset, isSmall);
277         result.writeBytes(valueByteBuf);
278         return result;
279     }
280     
281     private void writeInt(final ByteBuf jsonByteBuf, final int value, final boolean isSmall) {
282         if (isSmall) {
283             jsonByteBuf.writeShortLE(value);
284         } else {
285             jsonByteBuf.writeIntLE(value);
286         }
287     }
288     
289     private ByteBuf writeKeys(final ByteBuf jsonByteBuf, final List<JsonEntry> jsonEntries, final int startOffset, final boolean isSmall) {
290         ByteBuf result = Unpooled.buffer();
291         for (JsonEntry each : jsonEntries) {
292             writeInt(jsonByteBuf, startOffset + result.readableBytes(), isSmall);
293             byte[] keyBytes = each.getKey().getBytes();
294             jsonByteBuf.writeShortLE(keyBytes.length);
295             result.writeBytes(keyBytes);
296         }
297         return result;
298     }
299     
300     private ByteBuf writeValues(final ByteBuf jsonByteBuf, final List<JsonEntry> jsonEntries, final int startOffset, final boolean isSmall) {
301         ByteBuf result = Unpooled.buffer();
302         for (JsonEntry each : jsonEntries) {
303             jsonByteBuf.writeByte(each.getType());
304             int offsetOrInlineValue = getOffsetOrInlineValue(startOffset, result.readableBytes(), each, isSmall);
305             writeInt(jsonByteBuf, offsetOrInlineValue, isSmall);
306             writeValueToByteBuf(each, result, isSmall);
307         }
308         return result;
309     }
310     
311     private int getOffsetOrInlineValue(final int startOffset, final int readableBytes, final JsonEntry jsonEntry, final boolean isSmall) {
312         switch (jsonEntry.getType()) {
313             case JsonValueTypes.INT16:
314             case JsonValueTypes.UINT16:
315                 return (int) jsonEntry.getValue();
316             case JsonValueTypes.LITERAL:
317                 return (byte) jsonEntry.getValue();
318             case JsonValueTypes.INT32:
319             case JsonValueTypes.UINT32:
320                 return isSmall ? startOffset + readableBytes : (int) jsonEntry.getValue();
321             default:
322                 return startOffset + readableBytes;
323         }
324     }
325     
326     private void writeValueToByteBuf(final JsonEntry jsonEntry, final ByteBuf byteBuf, final boolean isSmall) {
327         switch (jsonEntry.getType()) {
328             case JsonValueTypes.SMALL_JSON_OBJECT:
329                 byteBuf.writeBytes(mockJsonObjectByteBufValue((List<JsonEntry>) jsonEntry.getValue(), true));
330                 break;
331             case JsonValueTypes.LARGE_JSON_OBJECT:
332                 byteBuf.writeBytes(mockJsonObjectByteBufValue((List<JsonEntry>) jsonEntry.getValue(), false));
333                 break;
334             case JsonValueTypes.SMALL_JSON_ARRAY:
335                 byteBuf.writeBytes(mockJsonArrayByteBufValue((List<JsonEntry>) jsonEntry.getValue(), true));
336                 break;
337             case JsonValueTypes.LARGE_JSON_ARRAY:
338                 byteBuf.writeBytes(mockJsonArrayByteBufValue((List<JsonEntry>) jsonEntry.getValue(), false));
339                 break;
340             case JsonValueTypes.INT32:
341             case JsonValueTypes.UINT32:
342                 if (isSmall) {
343                     byteBuf.writeIntLE((int) jsonEntry.getValue());
344                 }
345                 break;
346             case JsonValueTypes.INT64:
347             case JsonValueTypes.UINT64:
348                 byteBuf.writeLongLE((long) jsonEntry.getValue());
349                 break;
350             case JsonValueTypes.DOUBLE:
351                 byteBuf.writeDoubleLE((double) jsonEntry.getValue());
352                 break;
353             case JsonValueTypes.STRING:
354                 writeString(byteBuf, (String) jsonEntry.getValue());
355                 break;
356             default:
357         }
358     }

360     private void writeString(final ByteBuf jsonByteBuf, final String value) {
361         byte[] result = codecDataLength(value.length());
362         jsonByteBuf.writeBytes(result, 0, result.length);
363         jsonByteBuf.writeBytes(value.getBytes());
364     }
365     
366     private byte[] codecDataLength(final int length) {
367         byte[] lengthData = new byte[32 / 7 + (0 == 32 % 7 ? 0 : 1)];
368         for (int i = 0; i < lengthData.length; i++) {
369             lengthData[i] = (byte) ((length >> (7 * i)) & 0x7f);
370         }
371         // compress
372         int index = lengthData.length - 1;
373         for (; index > 0; index--) {
374             if (0 != lengthData[index]) {
375                 break;
376             }
377         }
378         for (int i = 0; i < index; i++) {
379             lengthData[i] |= 0x80;
380         }
381         byte[] result = new byte[index + 1];
382         System.arraycopy(lengthData, 0, result, 0, index + 1);
383         return result;
384     }
