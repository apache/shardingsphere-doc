32  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.item.ShorthandProjectionSegment;

36  import java.util.List;

38  /**
39   * Projection token generator rewrite supported checker.
40   */
41  @HighFrequencyInvocation
42  @NoArgsConstructor(access = AccessLevel.PRIVATE)
43  public final class EncryptProjectionRewriteSupportedChecker {
44      
45      /**
46       * Check not contain encrypt projection in combine segment.
47       *
48       * @param encryptRule encrypt rule
49       * @param selectStatementContext select statement context
50       */
51      public static void checkNotContainEncryptProjectionInCombineSegment(final EncryptRule encryptRule, final SelectStatementContext selectStatementContext) {
52          ShardingSpherePreconditions.checkState(!containsEncryptProjectionInCombineSegment(encryptRule, selectStatementContext),
53                  () -> new UnsupportedSQLOperationException("Can not support encrypt projection in combine statement."));
54      }
55      
56      private static boolean containsEncryptProjectionInCombineSegment(final EncryptRule encryptRule, final SelectStatementContext selectStatementContext) {
57          if (!selectStatementContext.getSqlStatement().getCombine().isPresent()) {
58              return false;
59          }
60          CombineSegment combineSegment = selectStatementContext.getSqlStatement().getCombine().get();
61          List<Projection> leftProjections = selectStatementContext.getSubqueryContexts().get(combineSegment.getLeft().getStartIndex()).getProjectionsContext().getExpandProjections();
62          List<Projection> rightProjections = selectStatementContext.getSubqueryContexts().get(combineSegment.getRight().getStartIndex()).getProjectionsContext().getExpandProjections();
63          ShardingSpherePreconditions.checkState(leftProjections.size() == rightProjections.size(), () -> new UnsupportedSQLOperationException("Column projections must be same for combine statement"));
64          for (int i = 0; i < leftProjections.size(); i++) {
65              if (containsEncryptProjectionInCombineSegment(encryptRule, leftProjections.get(i), rightProjections.get(i))) {
66                  return true;
67              }
68          }
69          return false;
70      }
71      
72      private static boolean containsEncryptProjectionInCombineSegment(final EncryptRule encryptRule, final Projection leftProjection, final Projection rightProjection) {
73          ColumnSegmentBoundInfo leftColumnInfo = getColumnSegmentBoundInfo(leftProjection);
74          EncryptAlgorithm leftColumnEncryptor = encryptRule.findQueryEncryptor(leftColumnInfo.getOriginalTable().getValue(), leftColumnInfo.getOriginalColumn().getValue()).orElse(null);
75          ColumnSegmentBoundInfo rightColumnInfo = getColumnSegmentBoundInfo(rightProjection);
76          EncryptAlgorithm rightColumnEncryptor = encryptRule.findQueryEncryptor(rightColumnInfo.getOriginalTable().getValue(), rightColumnInfo.getOriginalColumn().getValue()).orElse(null);
77          return null != leftColumnEncryptor || null != rightColumnEncryptor;
78      }
79      
80      private static ColumnSegmentBoundInfo getColumnSegmentBoundInfo(final Projection projection) {
81          return projection instanceof ColumnProjection
82                  ? new ColumnSegmentBoundInfo(null, null, ((ColumnProjection) projection).getOriginalTable(), ((ColumnProjection) projection).getOriginalColumn())
83                  : new ColumnSegmentBoundInfo(new IdentifierValue(projection.getColumnLabel()));
84      }
85      
86      /**
87       * Check not contain encrypt shorthand expand with subquery statement.
88       *
89       * @param selectStatementContext select statement context
90       * @param projectionSegment projection segment
91       */
92      public static void checkNotContainEncryptShorthandExpandWithSubqueryStatement(final SelectStatementContext selectStatementContext, final ProjectionSegment projectionSegment) {
93          ShardingSpherePreconditions.checkState(!(projectionSegment instanceof ShorthandProjectionSegment) || !selectStatementContext.containsTableSubquery(),
94                  () -> new UnsupportedSQLOperationException("Can not support encrypt shorthand expand with subquery statement."));
95      }
96  }
