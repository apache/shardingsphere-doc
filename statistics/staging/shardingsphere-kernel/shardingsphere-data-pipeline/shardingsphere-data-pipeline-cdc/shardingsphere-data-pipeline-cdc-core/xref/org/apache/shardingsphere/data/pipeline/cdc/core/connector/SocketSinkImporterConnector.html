56  /**
57   * Socket sink importer connector.
58   */
59  @Slf4j
60  public final class SocketSinkImporterConnector implements ImporterConnector, AutoCloseable {
61      
62      private static final long DEFAULT_TIMEOUT_MILLISECONDS = 200L;
63      
64      private final Lock lock = new ReentrantLock();
65      
66      private final Condition condition = lock.newCondition();
67      
68      @Setter
69      private volatile boolean incrementalTaskRunning = true;
70      
71      private final ShardingSphereDatabase database;
72      
73      private final Channel channel;
74      
75      private final int jobShardingCount;
76      
77      private final Comparator<DataRecord> dataRecordComparator;
78      
79      private final Map<String, String> tableNameSchemaMap = new HashMap<>();
80      
81      private final Map<SocketSinkImporter, BlockingQueue<Record>> incrementalRecordMap = new ConcurrentHashMap<>();
82      
83      private final AtomicInteger runningIncrementalTaskCount = new AtomicInteger(0);
84      
85      private Thread incrementalImporterTask;
86      
87      public SocketSinkImporterConnector(final Channel channel, final ShardingSphereDatabase database, final int jobShardingCount, final Collection<String> schemaTableNames,
88                                         final Comparator<DataRecord> dataRecordComparator) {
89          this.channel = channel;
90          this.database = database;
91          this.jobShardingCount = jobShardingCount;
92          schemaTableNames.stream().filter(each -> each.contains(".")).forEach(each -> {
93              String[] split = each.split("\\.");
94              tableNameSchemaMap.put(split[1], split[0]);
95          });
96          this.dataRecordComparator = dataRecordComparator;
97      }
98      
99      @Override
100     public Object getConnector() {
101         return channel;
102     }
103     
104     /**
105      * Write data record into channel.
106      *
107      * @param recordList data records
108      * @param socketSinkImporter cdc importer
109      * @param importerType importer type
110      */
111     public void write(final List<Record> recordList, final SocketSinkImporter socketSinkImporter, final ImporterType importerType) {
112         if (recordList.isEmpty()) {
113             return;
114         }
115         if (ImporterType.INVENTORY == importerType || null == dataRecordComparator) {
116             Map<SocketSinkImporter, CDCAckPosition> importerDataRecordMap = new HashMap<>();
117             int dataRecordCount = (int) recordList.stream().filter(each -> each instanceof DataRecord).count();
118             Record lastRecord = recordList.get(recordList.size() - 1);
119             if (lastRecord instanceof FinishedRecord && 0 == dataRecordCount) {
120                 socketSinkImporter.ackWithLastDataRecord(new CDCAckPosition(lastRecord, 0));
121                 return;
122             }
123             importerDataRecordMap.put(socketSinkImporter, new CDCAckPosition(RecordUtils.getLastNormalRecord(recordList), dataRecordCount));
124             writeImmediately(recordList, importerDataRecordMap);
125         } else if (ImporterType.INCREMENTAL == importerType) {
126             writeIntoQueue(recordList, socketSinkImporter);
127         }
128     }
129     
130     private void writeImmediately(final List<? extends Record> recordList, final Map<SocketSinkImporter, CDCAckPosition> importerDataRecordMap) {
131         while (!channel.isWritable() && channel.isActive()) {
132             doAwait();
133         }
134         if (!channel.isActive()) {
135             return;
136         }
137         List<DataRecordResult.Record> records = new LinkedList<>();
138         for (Record each : recordList) {
139             if (!(each instanceof DataRecord)) {
140                 continue;
141             }
142             DataRecord dataRecord = (DataRecord) each;
143             records.add(DataRecordResultConvertUtils.convertDataRecordToRecord(database.getName(), tableNameSchemaMap.get(dataRecord.getTableName()), dataRecord));
144         }
145         String ackId = CDCAckHolder.getInstance().bindAckIdWithPosition(importerDataRecordMap);
146         DataRecordResult dataRecordResult = DataRecordResult.newBuilder().addAllRecord(records).setAckId(ackId).build();
147         channel.writeAndFlush(CDCResponseGenerator.succeedBuilder("").setDataRecordResult(dataRecordResult).build());
148     }
149     
150     private void doAwait() {
151         lock.lock();
152         try {
153             condition.await(DEFAULT_TIMEOUT_MILLISECONDS, TimeUnit.MILLISECONDS);
154         } catch (final InterruptedException ignored) {
155         } finally {
156             lock.unlock();
157         }
158     }
159     
160     @SneakyThrows(InterruptedException.class)
161     private void writeIntoQueue(final List<Record> dataRecords, final SocketSinkImporter socketSinkImporter) {
162         BlockingQueue<Record> blockingQueue = incrementalRecordMap.get(socketSinkImporter);
163         if (null == blockingQueue) {
164             log.warn("not find the queue to write");
165             return;
166         }
167         for (Record each : dataRecords) {
168             blockingQueue.put(each);
169         }
170     }
171     
172     /**
173      * Send incremental start event.
174      *
175      * @param socketSinkImporter socket sink importer
176      * @param batchSize batch size
177      */
178     public void sendIncrementalStartEvent(final SocketSinkImporter socketSinkImporter, final int batchSize) {
179         incrementalRecordMap.computeIfAbsent(socketSinkImporter, ignored -> new ArrayBlockingQueue<>(batchSize));
180         int count = runningIncrementalTaskCount.incrementAndGet();
181         if (count < jobShardingCount || null == dataRecordComparator) {
182             return;
183         }
184         log.debug("start CDC incremental importer");
185         if (null == incrementalImporterTask) {
186             incrementalImporterTask = new Thread(new CDCIncrementalImporterTask(batchSize));
187             incrementalImporterTask.start();
188         }
189     }
190     
191     /**
192      * Clean socket sink importer connector.
193      *
194      * @param socketSinkImporter CDC importer
195      */
196     public void clean(final SocketSinkImporter socketSinkImporter) {
197         incrementalRecordMap.remove(socketSinkImporter);
198         if (ImporterType.INCREMENTAL == socketSinkImporter.getImporterType()) {
199             incrementalTaskRunning = false;
200         }
201     }
202     
203     @Override
204     public String getType() {
205         return CDCSinkType.SOCKET.name();
206     }
207     
208     @Override
209     public void close() throws Exception {
210         channel.close();
211     }
212     
213     @RequiredArgsConstructor
214     private final class CDCIncrementalImporterTask implements Runnable {
215         
216         private final int batchSize;
217         
218         @SneakyThrows(InterruptedException.class)
219         @Override
220         public void run() {
221             while (incrementalTaskRunning) {
222                 Map<SocketSinkImporter, CDCAckPosition> cdcAckPositionMap = new HashMap<>();
223                 List<DataRecord> dataRecords = new LinkedList<>();
224                 for (int i = 0; i < batchSize; i++) {
225                     DataRecord minimumDataRecord = CDCDataRecordUtils.findMinimumDataRecordAndSavePosition(incrementalRecordMap, dataRecordComparator, cdcAckPositionMap);
226                     if (null == minimumDataRecord) {
227                         break;
228                     }
229                     dataRecords.add(minimumDataRecord);
230                 }
231                 if (dataRecords.isEmpty()) {
232                     Thread.sleep(200L);
233                 } else {
234                     writeImmediately(dataRecords, cdcAckPositionMap);
235                 }
236             }
237         }
238     }
