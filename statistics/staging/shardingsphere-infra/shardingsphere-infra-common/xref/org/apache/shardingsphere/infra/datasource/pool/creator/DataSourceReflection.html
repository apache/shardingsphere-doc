43  /**
44   * Data source reflection.
45   */
46  public final class DataSourceReflection {
47      
48      private static final Collection<Class<?>> GENERAL_CLASS_TYPES;
49      
50      private static final Collection<String> SKIPPED_PROPERTY_KEYS;
51      
52      private static final String IS_PREFIX = "is";
53      
54      private static final String GETTER_PREFIX = "get";
55      
56      private static final String SETTER_PREFIX = "set";
57      
58      private final DataSource dataSource;
59      
60      private final Method[] dataSourceMethods;
61      
62      static {
63          GENERAL_CLASS_TYPES = new HashSet<>(
64                  Arrays.asList(boolean.class, Boolean.class, int.class, Integer.class, long.class, Long.class, String.class, Collection.class, List.class, Properties.class));
65          SKIPPED_PROPERTY_KEYS = new HashSet<>(Arrays.asList("loginTimeout", "driverClassName"));
66      }
67      
68      public DataSourceReflection(final DataSource dataSource) {
69          this.dataSource = dataSource;
70          dataSourceMethods = dataSource.getClass().getMethods();
71      }
72      
73      /**
74       * Convert to properties.
75       * 
76       * @return properties converted from data source
77       */
78      public Map<String, Object> convertToProperties() {
79          Map<String, Object> getterProps = convertToProperties(GETTER_PREFIX);
80          Map<String, Object> isProps = convertToProperties(IS_PREFIX);
81          Map<String, Object> result = new LinkedHashMap<>(getterProps.size() + isProps.size(), 1F);
82          result.putAll(getterProps);
83          result.putAll(isProps);
84          return result;
85      }
86      
87      @SneakyThrows(ReflectiveOperationException.class)
88      private Map<String, Object> convertToProperties(final String prefix) {
89          Collection<Method> getterMethods = findAllGetterMethods(prefix);
90          Map<String, Object> result = new LinkedHashMap<>(getterMethods.size(), 1F);
91          for (Method each : getterMethods) {
92              String fieldName = getGetterFieldName(each, prefix);
93              if (GENERAL_CLASS_TYPES.contains(each.getReturnType()) && !SKIPPED_PROPERTY_KEYS.contains(fieldName)) {
94                  Object fieldValue = each.invoke(dataSource);
95                  if (null != fieldValue) {
96                      result.put(fieldName, fieldValue);
97                  }
98              }
99          }
100         return result;
101     }
102     
103     private Collection<Method> findAllGetterMethods(final String methodPrefix) {
104         Collection<Method> result = new HashSet<>(dataSourceMethods.length, 1F);
105         for (Method each : dataSourceMethods) {
106             if (each.getName().startsWith(methodPrefix) && 0 == each.getParameterTypes().length) {
107                 result.add(each);
108             }
109         }
110         return result;
111     }
112     
113     private String getGetterFieldName(final Method method, final String methodPrefix) {
114         return CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_CAMEL, method.getName().substring(methodPrefix.length()));
115     }
116     
117     /**
118      * Set data source field.
119      * 
120      * @param fieldName field name
121      * @param fieldValue field value
122      */
123     public void setField(final String fieldName, final Object fieldValue) {
124         if (null != fieldValue && !isSkippedProperty(fieldName)) {
125             findSetterMethod(fieldName).ifPresent(optional -> setField(optional, fieldValue));
126         }
127     }
128     
129     @SuppressWarnings("rawtypes")
130     @SneakyThrows(ReflectiveOperationException.class)
131     private void setField(final Method method, final Object fieldValue) {
132         Class<?> paramType = method.getParameterTypes()[0];
133         if (int.class == paramType || Integer.class == paramType) {
134             method.invoke(dataSource, Integer.parseInt(fieldValue.toString()));
135         } else if (long.class == paramType || Long.class == paramType) {
136             method.invoke(dataSource, Long.parseLong(fieldValue.toString()));
137         } else if (boolean.class == paramType || Boolean.class == paramType) {
138             method.invoke(dataSource, Boolean.parseBoolean(fieldValue.toString()));
139         } else if (String.class == paramType) {
140             method.invoke(dataSource, fieldValue.toString());
141         } else if (Properties.class == paramType) {
142             Properties props = new Properties();
143             props.putAll((Map) fieldValue);
144             method.invoke(dataSource, props);
145         } else {
146             method.invoke(dataSource, fieldValue);
147         }
148     }
149     
150     private boolean isSkippedProperty(final String key) {
151         return SKIPPED_PROPERTY_KEYS.contains(key);
152     }
153     
154     private Optional<Method> findSetterMethod(final String fieldName) {
155         String setterMethodName = SETTER_PREFIX + CaseFormat.LOWER_CAMEL.to(CaseFormat.UPPER_CAMEL, fieldName);
156         return Arrays.stream(dataSourceMethods).filter(each -> each.getName().equals(setterMethodName) && 1 == each.getParameterTypes().length).findFirst();
157     }
158     
159     /**
160      * Add default data source properties.
161      */
162     public void addDefaultDataSourceProperties() {
163         DataSourcePoolMetaDataReflection dataSourcePoolMetaDataReflection = new DataSourcePoolMetaDataReflection(dataSource,
164                 TypedSPILoader.findService(DataSourcePoolMetaData.class, dataSource.getClass().getName())
165                         .map(DataSourcePoolMetaData::getFieldMetaData).orElseGet(DefaultDataSourcePoolFieldMetaData::new));
166         Optional<String> jdbcUrl = dataSourcePoolMetaDataReflection.getJdbcUrl();
167         Optional<Properties> jdbcConnectionProps = dataSourcePoolMetaDataReflection.getJdbcConnectionProperties();
168         if (!jdbcUrl.isPresent() || !jdbcConnectionProps.isPresent()) {
169             return;
170         }
171         ConnectionProperties connectionProps = DatabaseTypedSPILoader.getService(ConnectionPropertiesParser.class, DatabaseTypeFactory.get(jdbcUrl.get())).parse(jdbcUrl.get(), null, null);
172         Properties queryProps = connectionProps.getQueryProperties();
173         for (Entry<Object, Object> entry : connectionProps.getDefaultQueryProperties().entrySet()) {
174             String defaultPropertyKey = entry.getKey().toString();
175             String defaultPropertyValue = entry.getValue().toString();
176             if (!containsDefaultProperty(defaultPropertyKey, jdbcConnectionProps.get(), queryProps)) {
177                 jdbcConnectionProps.get().setProperty(defaultPropertyKey, defaultPropertyValue);
178             }
179         }
180     }
181     
182     private boolean containsDefaultProperty(final String defaultPropertyKey, final Properties targetDataSourceProps, final Properties queryProps) {
183         return targetDataSourceProps.containsKey(defaultPropertyKey) || queryProps.containsKey(defaultPropertyKey);
184     }
