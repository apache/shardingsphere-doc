18  package org.apache.shardingsphere.test.e2e.driver;
19  
20  import com.zaxxer.hikari.HikariDataSource;
21  import org.h2.tools.RunScript;
22  import org.junit.jupiter.api.BeforeAll;
23  
24  import javax.sql.DataSource;
25  import java.io.InputStreamReader;
26  import java.sql.Connection;
27  import java.sql.SQLException;
28  import java.util.Arrays;
29  import java.util.HashMap;
30  import java.util.List;
31  import java.util.Map;
32  import java.util.Objects;
33  
34  // TODO merge in to sql-e2e
35  public abstract class AbstractDriverTest {
36      
37      private static final List<String> ACTUAL_DATA_SOURCE_NAMES = Arrays.asList("jdbc_0", "jdbc_1", "single_jdbc", "shadow_jdbc_0", "shadow_jdbc_1", "encrypt", "test_primary_ds", "test_replica_ds");
38      
39      private static final Map<String, DataSource> ACTUAL_DATA_SOURCES = new HashMap<>(ACTUAL_DATA_SOURCE_NAMES.size(), 1);
40      
41      @BeforeAll
42      static synchronized void initializeDataSource() throws SQLException {
43          for (String each : ACTUAL_DATA_SOURCE_NAMES) {
44              ACTUAL_DATA_SOURCES.put(each, buildDataSource(each));
45              initializeSchema(each);
46          }
47      }
48      
49      private static void initializeSchema(final String dataSourceName) throws SQLException {
50          try (Connection connection = ACTUAL_DATA_SOURCES.get(dataSourceName).getConnection()) {
51              if ("encrypt".equals(dataSourceName)) {
52                  RunScript.execute(connection, new InputStreamReader(Objects.requireNonNull(AbstractDriverTest.class.getClassLoader().getResourceAsStream("sql/jdbc_encrypt_init.sql"))));
53              } else if ("shadow_jdbc_0".equals(dataSourceName) || "shadow_jdbc_1".equals(dataSourceName)) {
54                  RunScript.execute(connection, new InputStreamReader(Objects.requireNonNull(AbstractDriverTest.class.getClassLoader().getResourceAsStream("sql/jdbc_shadow_init.sql"))));
55              } else if ("single_jdbc".equals(dataSourceName)) {
56                  RunScript.execute(connection, new InputStreamReader(Objects.requireNonNull(AbstractDriverTest.class.getClassLoader().getResourceAsStream("sql/single_jdbc_init.sql"))));
57              } else {
58                  RunScript.execute(connection, new InputStreamReader(Objects.requireNonNull(AbstractDriverTest.class.getClassLoader().getResourceAsStream("sql/jdbc_init.sql"))));
59              }
60          }
61      }
62      
63      private static DataSource buildDataSource(final String dataSourceName) {
64          HikariDataSource result = new HikariDataSource();
65          result.setDriverClassName("org.h2.Driver");
66          result.setJdbcUrl(String.format("jdbc:h2:mem:%s;DATABASE_TO_UPPER=false;MODE=MySQL", dataSourceName));
67          result.setUsername("sa");
68          result.setPassword("");
69          result.setMinimumIdle(0);
70          result.setMaximumPoolSize(50);
71          return result;
72      }
73      
74      protected static Map<String, DataSource> getActualDataSources() {
75          return ACTUAL_DATA_SOURCES;
76      }
77  }
