46  /**
47   * Socket sink importer.
48   */
49  @Slf4j
50  public final class SocketSinkImporter extends AbstractLifecycleExecutor implements Importer {
51      
52      @Getter(AccessLevel.PROTECTED)
53      private final ImporterConfiguration importerConfig;
54      
55      private final PipelineChannel channel;
56      
57      private final SocketSinkImporterConnector importerConnector;
58      
59      private final PipelineJobProgressListener jobProgressListener;
60      
61      @Getter
62      private final ImporterType importerType;
63      
64      private final JobRateLimitAlgorithm rateLimitAlgorithm;
65      
66      public SocketSinkImporter(final ImporterConfiguration importerConfig, final ImporterConnector importerConnector, final PipelineChannel channel,
67                                final PipelineJobProgressListener jobProgressListener, final ImporterType importerType) {
68          this.importerConfig = importerConfig;
69          rateLimitAlgorithm = null == importerConfig ? null : importerConfig.getRateLimitAlgorithm();
70          this.channel = channel;
71          this.importerConnector = (SocketSinkImporterConnector) importerConnector;
72          this.jobProgressListener = jobProgressListener;
73          this.importerType = importerType;
74      }
75      
76      @Override
77      protected void runBlocking() {
78          int batchSize = importerConfig.getBatchSize();
79          if (ImporterType.INCREMENTAL == importerType) {
80              importerConnector.sendIncrementalStartEvent(this, batchSize);
81          }
82          while (isRunning()) {
83              List<Record> records = channel.fetchRecords(batchSize, 500, TimeUnit.MILLISECONDS);
84              if (null != records && !records.isEmpty()) {
85                  List<Record> recordList = records.stream().filter(each -> !(each instanceof PlaceholderRecord)).collect(Collectors.toList());
86                  try {
87                      processDataRecords(recordList);
88                  } catch (final SQLException ex) {
89                      log.error("process data records failed", ex);
90                      throw new RuntimeException(ex);
91                  }
92                  if (FinishedRecord.class.equals(records.get(records.size() - 1).getClass())) {
93                      break;
94                  }
95              }
96          }
97      }
98      
99      private void processDataRecords(final List<Record> recordList) throws SQLException {
100         if (null == recordList || recordList.isEmpty()) {
101             return;
102         }
103         if (null != rateLimitAlgorithm) {
104             rateLimitAlgorithm.intercept(JobOperationType.INSERT, 1);
105         }
106         importerConnector.write(recordList, this, importerType);
107     }
108     
109     /**
110      * Ack with last data record.
111      *
112      * @param cdcAckPosition cdc ack position
113      */
114     public void ackWithLastDataRecord(final CDCAckPosition cdcAckPosition) {
115         channel.ack(Collections.singletonList(cdcAckPosition.getLastRecord()));
116         jobProgressListener.onProgressUpdated(new PipelineJobProgressUpdatedParameter(cdcAckPosition.getDataRecordCount()));
117     }
118     
119     @Override
120     protected void doStop() {
121         if (ImporterType.INCREMENTAL == importerType) {
122             importerConnector.clean(this);
123             CDCAckHolder.getInstance().cleanUp(this);
124         }
125     }
126 }
