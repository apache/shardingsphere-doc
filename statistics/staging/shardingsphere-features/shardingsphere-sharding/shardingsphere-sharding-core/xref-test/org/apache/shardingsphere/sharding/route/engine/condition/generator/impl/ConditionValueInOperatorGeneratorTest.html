45  class ConditionValueInOperatorGeneratorTest {
46      
47      private final ConditionValueInOperatorGenerator generator = new ConditionValueInOperatorGenerator();
48      
49      private final Column column = new Column("id", "tbl");
50      
51      private final TimeServiceRule timeServiceRule = new TimeServiceRule(new TimeServiceRuleConfiguration("System", new Properties()));
52      
53      @Test
54      void assertNowExpression() {
55          ListExpression listExpression = new ListExpression(0, 0);
56          listExpression.getItems().add(new CommonExpressionSegment(0, 0, "now()"));
57          InExpression inExpression = new InExpression(0, 0, null, listExpression, false);
58          Optional<ShardingConditionValue> shardingConditionValue = generator.generate(inExpression, column, new LinkedList<>(), timeServiceRule);
59          assertTrue(shardingConditionValue.isPresent());
60          assertThat(((ListShardingConditionValue<?>) shardingConditionValue.get()).getValues().iterator().next(), instanceOf(Date.class));
61          assertTrue(shardingConditionValue.get().getParameterMarkerIndexes().isEmpty());
62      }
63      
64      @SuppressWarnings("unchecked")
65      @Test
66      void assertGenerateConditionValueWithParameter() {
67          ColumnSegment left = new ColumnSegment(0, 0, new IdentifierValue("id"));
68          ListExpression right = new ListExpression(0, 0);
69          right.getItems().add(new ParameterMarkerExpressionSegment(0, 0, 0));
70          InExpression predicate = new InExpression(0, 0, left, right, false);
71          Optional<ShardingConditionValue> actual = generator.generate(predicate, column, Collections.singletonList(1), timeServiceRule);
72          assertTrue(actual.isPresent());
73          assertThat(actual.get(), instanceOf(ListShardingConditionValue.class));
74          ListShardingConditionValue<Integer> conditionValue = (ListShardingConditionValue<Integer>) actual.get();
75          assertThat(conditionValue.getTableName(), is("tbl"));
76          assertThat(conditionValue.getColumnName(), is("id"));
77          assertThat(conditionValue.getValues(), is(Collections.singletonList(1)));
78          assertThat(conditionValue.getParameterMarkerIndexes(), is(Collections.singletonList(0)));
79      }
80      
81      @Test
82      void assertGenerateConditionValueWithoutParameter() {
83          ColumnSegment left = new ColumnSegment(0, 0, new IdentifierValue("order_id"));
84          ListExpression right = new ListExpression(0, 0);
85          right.getItems().add(new ParameterMarkerExpressionSegment(0, 0, 0));
86          InExpression predicate = new InExpression(0, 0, left, right, false);
87          Optional<ShardingConditionValue> actual = generator.generate(predicate, column, new LinkedList<>(), timeServiceRule);
88          assertFalse(actual.isPresent());
89      }
90  }
