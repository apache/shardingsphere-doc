18  package org.apache.shardingsphere.db.protocol.mysql.packet.handshake;
19  
20  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLAuthenticationMethod;
21  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLCapabilityFlag;
22  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLServerInfo;
23  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLStatusFlag;
24  import org.apache.shardingsphere.db.protocol.mysql.payload.MySQLPacketPayload;
25  import org.junit.jupiter.api.Test;
26  import org.junit.jupiter.api.extension.ExtendWith;
27  import org.mockito.Mock;
28  import org.mockito.junit.jupiter.MockitoExtension;
29  
30  import static org.hamcrest.CoreMatchers.is;
31  import static org.hamcrest.MatcherAssert.assertThat;
32  import static org.mockito.Mockito.verify;
33  import static org.mockito.Mockito.when;
34  
35  @ExtendWith(MockitoExtension.class)
36  class MySQLHandshakePacketTest {
37      
38      @Mock
39      private MySQLPacketPayload payload;
40      
41      private final byte[] part1 = {106, 105, 55, 122, 117, 98, 115, 109};
42      
43      private final byte[] part2 = {68, 102, 53, 122, 65, 49, 84, 79, 85, 115, 116, 113};
44      
45      @Test
46      void assertNewWithPayload() {
47          when(payload.readInt1()).thenReturn(MySQLServerInfo.PROTOCOL_VERSION, MySQLServerInfo.DEFAULT_CHARSET.getId(), 0);
48          when(payload.readStringNul()).thenReturn(MySQLServerInfo.getDefaultServerVersion());
49          when(payload.readStringNulByBytes()).thenReturn(part1, part2);
50          when(payload.readInt4()).thenReturn(1000);
51          when(payload.readInt2()).thenReturn(
52                  MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsLower(), MySQLStatusFlag.SERVER_STATUS_AUTOCOMMIT.getValue(), MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsUpper());
53          MySQLHandshakePacket actual = new MySQLHandshakePacket(payload);
54          assertThat(actual.getServerVersion(), is(MySQLServerInfo.getDefaultServerVersion()));
55          assertThat(actual.getCapabilityFlagsLower(), is(MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsLower()));
56          assertThat(actual.getConnectionId(), is(1000));
57          assertThat(actual.getCharacterSet(), is(MySQLServerInfo.DEFAULT_CHARSET.getId()));
58          assertThat(actual.getStatusFlag(), is(MySQLStatusFlag.SERVER_STATUS_AUTOCOMMIT));
59          assertThat(actual.getCapabilityFlagsUpper(), is(MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsUpper()));
60          assertThat(actual.getAuthPluginData().getAuthenticationPluginDataPart1(), is(part1));
61          assertThat(actual.getAuthPluginData().getAuthenticationPluginDataPart2(), is(part2));
62          verify(payload).skipReserved(10);
63      }
64      
65      @Test
66      void assertNewWithClientPluginAuthPayload() {
67          when(payload.readInt1()).thenReturn(MySQLServerInfo.PROTOCOL_VERSION, MySQLServerInfo.DEFAULT_CHARSET.getId(), 0);
68          when(payload.readStringNul()).thenReturn(MySQLServerInfo.getDefaultServerVersion(), MySQLAuthenticationMethod.NATIVE.getMethodName());
69          when(payload.readStringNulByBytes()).thenReturn(part1, part2);
70          when(payload.readInt4()).thenReturn(1000);
71          when(payload.readInt2()).thenReturn(
72                  MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsLower(), MySQLStatusFlag.SERVER_STATUS_AUTOCOMMIT.getValue(), MySQLCapabilityFlag.CLIENT_PLUGIN_AUTH.getValue() >> 16);
73          MySQLHandshakePacket actual = new MySQLHandshakePacket(payload);
74          assertThat(actual.getServerVersion(), is(MySQLServerInfo.getDefaultServerVersion()));
75          assertThat(actual.getCapabilityFlagsLower(), is(MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsLower()));
76          assertThat(actual.getConnectionId(), is(1000));
77          assertThat(actual.getCharacterSet(), is(MySQLServerInfo.DEFAULT_CHARSET.getId()));
78          assertThat(actual.getStatusFlag(), is(MySQLStatusFlag.SERVER_STATUS_AUTOCOMMIT));
79          assertThat(actual.getCapabilityFlagsUpper(), is(MySQLCapabilityFlag.CLIENT_PLUGIN_AUTH.getValue() >> 16));
80          assertThat(actual.getAuthPluginData().getAuthenticationPluginDataPart1(), is(part1));
81          assertThat(actual.getAuthPluginData().getAuthenticationPluginDataPart2(), is(part2));
82          verify(payload).skipReserved(10);
83          assertThat(actual.getAuthPluginName(), is(MySQLAuthenticationMethod.NATIVE.getMethodName()));
84      }
85      
86      @Test
87      void assertWrite() {
88          MySQLAuthenticationPluginData authPluginData = new MySQLAuthenticationPluginData(part1, part2);
89          new MySQLHandshakePacket(1000, authPluginData).write(payload);
90          verify(payload).writeInt1(MySQLServerInfo.PROTOCOL_VERSION);
91          verify(payload).writeStringNul(MySQLServerInfo.getDefaultServerVersion());
92          verify(payload).writeInt4(1000);
93          verify(payload).writeStringNul(new String(authPluginData.getAuthenticationPluginDataPart1()));
94          verify(payload).writeInt2(MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsLower());
95          verify(payload).writeInt1(MySQLServerInfo.DEFAULT_CHARSET.getId());
96          verify(payload).writeInt2(MySQLStatusFlag.SERVER_STATUS_AUTOCOMMIT.getValue());
97          verify(payload).writeInt2(MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsUpper());
98          verify(payload).writeInt1(authPluginData.getAuthenticationPluginData().length + 1);
99          verify(payload).writeReserved(10);
100         verify(payload).writeStringNul(new String(authPluginData.getAuthenticationPluginDataPart2()));
101     }
102     
103     @Test
104     void assertWriteWithClientPluginAuth() {
105         MySQLAuthenticationPluginData authPluginData = new MySQLAuthenticationPluginData(part1, part2);
106         MySQLHandshakePacket actual = new MySQLHandshakePacket(1000, authPluginData);
107         actual.setAuthPluginName(MySQLAuthenticationMethod.NATIVE);
108         actual.write(payload);
109         verify(payload).writeInt1(MySQLServerInfo.PROTOCOL_VERSION);
110         verify(payload).writeStringNul(MySQLServerInfo.getDefaultServerVersion());
111         verify(payload).writeInt4(1000);
112         verify(payload).writeStringNul(new String(authPluginData.getAuthenticationPluginDataPart1()));
113         verify(payload).writeInt2(MySQLCapabilityFlag.calculateHandshakeCapabilityFlagsLower());
114         verify(payload).writeInt1(MySQLServerInfo.DEFAULT_CHARSET.getId());
115         verify(payload).writeInt2(MySQLStatusFlag.SERVER_STATUS_AUTOCOMMIT.getValue());
116         verify(payload).writeInt2(MySQLCapabilityFlag.CLIENT_PLUGIN_AUTH.getValue() >> 16);
117         verify(payload).writeInt1(authPluginData.getAuthenticationPluginData().length + 1);
118         verify(payload).writeReserved(10);
119         verify(payload).writeStringNul(new String(authPluginData.getAuthenticationPluginDataPart2()));
120         verify(payload).writeStringNul(MySQLAuthenticationMethod.NATIVE.getMethodName());
121     }
122 }
