18  package org.apache.shardingsphere.metadata.persist.node;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  
23  import java.util.Optional;
24  import java.util.regex.Matcher;
25  import java.util.regex.Pattern;
26  
27  /**
28   * Database meta data node.
29   */
30  @NoArgsConstructor(access = AccessLevel.PRIVATE)
31  public final class DatabaseMetaDataNode {
32      
33      private static final String ROOT_NODE = "metadata";
34      
35      private static final String SCHEMAS_NODE = "schemas";
36      
37      private static final String DATA_SOURCE_NODE = "data_sources";
38      
39      private static final String DATA_SOURCE_NODES_NODE = "nodes";
40      
41      private static final String DATA_SOURCE_UNITS_NODE = "units";
42      
43      private static final String RULE_NODE = "rules";
44      
45      private static final String TABLES_NODE = "tables";
46      
47      private static final String VIEWS_NODE = "views";
48      
49      private static final String ACTIVE_VERSION = "active_version";
50      
51      private static final String VERSIONS = "versions";
52      
53      /**
54       * Get meta data data source nodes path.
55       *
56       * @param databaseName database name
57       * @param version data source version
58       * @return data source path
59       */
60      public static String getMetaDataDataSourceUnitsPath(final String databaseName, final String version) {
61          return String.join("/", getFullMetaDataPath(databaseName, VERSIONS), version, DATA_SOURCE_NODE, DATA_SOURCE_UNITS_NODE);
62      }
63      
64      /**
65       * Get meta data data source units path.
66       *
67       * @param databaseName database name
68       * @param version data source version
69       * @return data source path
70       */
71      public static String getMetaDataDataSourceNodesPath(final String databaseName, final String version) {
72          return String.join("/", getFullMetaDataPath(databaseName, VERSIONS), version, DATA_SOURCE_NODE, DATA_SOURCE_NODES_NODE);
73      }
74      
75      /**
76       * Get meta data node path.
77       *
78       * @return meta data node path
79       */
80      public static String getMetaDataNodePath() {
81          return String.join("/", "", ROOT_NODE);
82      }
83      
84      /**
85       * Get database name path.
86       *
87       * @param databaseName database name
88       * @return database name path
89       */
90      public static String getDatabaseNamePath(final String databaseName) {
91          return String.join("/", getMetaDataNodePath(), databaseName);
92      }
93      
94      /**
95       * Get rule path.
96       *
97       * @param databaseName database name
98       * @param version rule version
99       * @return rule path
100      */
101     public static String getRulePath(final String databaseName, final String version) {
102         return String.join("/", getFullMetaDataPath(databaseName, VERSIONS), version, RULE_NODE);
103     }
104     
105     /**
106      * Get meta data tables path.
107      *
108      * @param databaseName database name
109      * @param schemaName schema name
110      * @return tables path
111      */
112     public static String getMetaDataTablesPath(final String databaseName, final String schemaName) {
113         return String.join("/", getMetaDataSchemaPath(databaseName, schemaName), TABLES_NODE);
114     }
115     
116     /**
117      * Get meta data views path.
118      *
119      * @param databaseName database name
120      * @param schemaName schema name
121      * @return views path
122      */
123     public static String getMetaDataViewsPath(final String databaseName, final String schemaName) {
124         return String.join("/", getMetaDataSchemaPath(databaseName, schemaName), VIEWS_NODE);
125     }
126     
127     /**
128      * Get schema path.
129      *
130      * @param databaseName database name
131      * @param schemaName schema name
132      * @return tables path
133      */
134     public static String getMetaDataSchemaPath(final String databaseName, final String schemaName) {
135         return String.join("/", getMetaDataSchemasPath(databaseName), schemaName);
136     }
137     
138     /**
139      * Get meta data schemas path.
140      *
141      * @param databaseName database name
142      * @return schemas path
143      */
144     public static String getMetaDataSchemasPath(final String databaseName) {
145         return String.join("/", getDatabaseNamePath(databaseName), SCHEMAS_NODE);
146     }
147     
148     /**
149      * Get table meta data path.
150      *
151      * @param databaseName database name
152      * @param schemaName schema name
153      * @param table table name
154      * @return table meta data path
155      */
156     public static String getTableMetaDataPath(final String databaseName, final String schemaName, final String table) {
157         return String.join("/", getMetaDataTablesPath(databaseName, schemaName), table);
158     }
159     
160     /**
161      * Get view meta data path.
162      *
163      * @param databaseName database name
164      * @param schemaName schema name
165      * @param view view name
166      * @return view meta data path
167      */
168     public static String getViewMetaDataPath(final String databaseName, final String schemaName, final String view) {
169         return String.join("/", getMetaDataViewsPath(databaseName, schemaName), view);
170     }
171     
172     private static String getFullMetaDataPath(final String databaseName, final String node) {
173         return String.join("/", "", ROOT_NODE, databaseName, node);
174     }
175     
176     /**
177      * Get database name.
178      *
179      * @param configNodeFullPath config node full path
180      * @return database name
181      */
182     public static Optional<String> getDatabaseName(final String configNodeFullPath) {
183         Pattern pattern = Pattern.compile(getMetaDataNodePath() + "/([\\w\\-]+)$", Pattern.CASE_INSENSITIVE);
184         Matcher matcher = pattern.matcher(configNodeFullPath);
185         return matcher.find() ? Optional.of(matcher.group(1)) : Optional.empty();
186     }
187     
188     /**
189      * Get schema name.
190      *
191      * @param configNodeFullPath config node full path
192      * @return schema name
193      */
194     public static Optional<String> getSchemaName(final String configNodeFullPath) {
195         Pattern pattern = Pattern.compile(getMetaDataNodePath() + "/([\\w\\-]+)/schemas/([\\w\\-]+)$", Pattern.CASE_INSENSITIVE);
196         Matcher matcher = pattern.matcher(configNodeFullPath);
197         return matcher.find() ? Optional.of(matcher.group(2)) : Optional.empty();
198     }
199     
200     /**
201      * Get database name by database path.
202      *
203      * @param databasePath database path
204      * @return database name
205      */
206     public static Optional<String> getDatabaseNameByDatabasePath(final String databasePath) {
207         Pattern pattern = Pattern.compile(getMetaDataNodePath() + "/([\\w\\-]+)?", Pattern.CASE_INSENSITIVE);
208         Matcher matcher = pattern.matcher(databasePath);
209         return matcher.find() ? Optional.of(matcher.group(1)) : Optional.empty();
210     }
211     
212     /**
213      * Get schema name.
214      *
215      * @param schemaPath schema path
216      * @return schema name
217      */
218     public static Optional<String> getSchemaNameBySchemaPath(final String schemaPath) {
219         Pattern pattern = Pattern.compile(getMetaDataNodePath() + "/([\\w\\-]+)/schemas/([\\w\\-]+)?", Pattern.CASE_INSENSITIVE);
220         Matcher matcher = pattern.matcher(schemaPath);
221         return matcher.find() ? Optional.of(matcher.group(2)) : Optional.empty();
222     }
223     
224     /**
225      * Get table meta data path.
226      *
227      * @param tableMetaDataPath table meta data path
228      * @return table name
229      */
230     public static Optional<String> getTableName(final String tableMetaDataPath) {
231         Pattern pattern = Pattern.compile(getMetaDataNodePath() + "/([\\w\\-]+)/([\\w\\-]+)/([\\w\\-]+)/tables" + "/([\\w\\-]+)$", Pattern.CASE_INSENSITIVE);
232         Matcher matcher = pattern.matcher(tableMetaDataPath);
233         return matcher.find() ? Optional.of(matcher.group(4)) : Optional.empty();
234     }

267     }

279     }
