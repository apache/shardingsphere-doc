53  /**
54   * Simple table segment binder.
55   */
56  @NoArgsConstructor(access = AccessLevel.PRIVATE)
57  public final class SimpleTableSegmentBinder {
58      
59      private static final Collection<String> SYSTEM_CATALOG_TABLES = new CaseInsensitiveSet<>(4, 1F);
60      
61      private static final String PG_CATALOG = "pg_catalog";
62      
63      static {
64          SYSTEM_CATALOG_TABLES.add("pg_database");
65          SYSTEM_CATALOG_TABLES.add("pg_tables");
66          SYSTEM_CATALOG_TABLES.add("pg_roles");
67          SYSTEM_CATALOG_TABLES.add("pg_settings");
68      }
69      
70      /**
71       * Bind simple table segment.
72       *
73       * @param segment simple table segment
74       * @param binderContext SQL statement binder context
75       * @param tableBinderContexts table binder contexts
76       * @return bound simple table segment
77       */
78      public static SimpleTableSegment bind(final SimpleTableSegment segment, final SQLStatementBinderContext binderContext, final Map<String, TableSegmentBinderContext> tableBinderContexts) {
79          fillPivotColumnNamesInBinderContext(segment, binderContext);
80          IdentifierValue databaseName = getDatabaseName(segment, binderContext);
81          ShardingSpherePreconditions.checkNotNull(databaseName.getValue(), NoDatabaseSelectedException::new);
82          IdentifierValue schemaName = getSchemaName(segment, binderContext);
83          IdentifierValue tableName = segment.getTableName().getIdentifier();
84          checkTableExists(binderContext, databaseName.getValue(), schemaName.getValue(), tableName.getValue());
85          ShardingSphereSchema schema = binderContext.getMetaData().getDatabase(databaseName.getValue()).getSchema(schemaName.getValue());
86          tableBinderContexts.putIfAbsent(
87                  (segment.getAliasName().orElseGet(tableName::getValue)).toLowerCase(), createSimpleTableBinderContext(segment, schema, databaseName, schemaName, binderContext));
88          TableNameSegment tableNameSegment = new TableNameSegment(segment.getTableName().getStartIndex(), segment.getTableName().getStopIndex(), tableName);
89          SimpleTableSegment result = new SimpleTableSegment(tableNameSegment);
90          segment.getOwner().ifPresent(result::setOwner);
91          segment.getAliasSegment().ifPresent(result::setAlias);
92          return result;
93      }
94      
95      private static void fillPivotColumnNamesInBinderContext(final SimpleTableSegment segment, final SQLStatementBinderContext binderContext) {
96          segment.getPivot().ifPresent(optional -> optional.getPivotColumns().forEach(each -> binderContext.getPivotColumnNames().add(each.getIdentifier().getValue().toLowerCase())));
97      }
98      
99      private static IdentifierValue getDatabaseName(final SimpleTableSegment segment, final SQLStatementBinderContext binderContext) {
100         DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(binderContext.getDatabaseType()).getDialectDatabaseMetaData();
101         Optional<OwnerSegment> owner = dialectDatabaseMetaData.getDefaultSchema().isPresent() ? segment.getOwner().flatMap(OwnerSegment::getOwner) : segment.getOwner();
102         return new IdentifierValue(owner.map(optional -> optional.getIdentifier().getValue()).orElse(binderContext.getCurrentDatabaseName()));
103     }
104     
105     private static IdentifierValue getSchemaName(final SimpleTableSegment segment, final SQLStatementBinderContext binderContext) {
106         if (segment.getOwner().isPresent()) {
107             return segment.getOwner().get().getIdentifier();
108         }
109         // TODO getSchemaName according to search path
110         DatabaseType databaseType = binderContext.getDatabaseType();
111         if ((databaseType instanceof PostgreSQLDatabaseType || databaseType instanceof OpenGaussDatabaseType) && SYSTEM_CATALOG_TABLES.contains(segment.getTableName().getIdentifier().getValue())) {
112             return new IdentifierValue(PG_CATALOG);
113         }
114         return new IdentifierValue(new DatabaseTypeRegistry(databaseType).getDefaultSchemaName(binderContext.getCurrentDatabaseName()));
115     }
116     
117     private static void checkTableExists(final SQLStatementBinderContext binderContext, final String databaseName, final String schemaName, final String tableName) {
118         if ("dual".equalsIgnoreCase(tableName)) {
119             return;
120         }
121         if (SystemSchemaManager.isSystemTable(schemaName, tableName)) {
122             return;
123         }
124         if (binderContext.getExternalTableBinderContexts().containsKey(tableName)) {
125             return;
126         }
127         ShardingSpherePreconditions.checkState(binderContext.getMetaData().containsDatabase(databaseName)
128                 && binderContext.getMetaData().getDatabase(databaseName).containsSchema(schemaName)
129                 && binderContext.getMetaData().getDatabase(databaseName).getSchema(schemaName).containsTable(tableName),
130                 () -> new TableNotFoundException(tableName));
131     }
132     
133     private static SimpleTableSegmentBinderContext createSimpleTableBinderContext(final SimpleTableSegment segment, final ShardingSphereSchema schema, final IdentifierValue databaseName,
134                                                                                   final IdentifierValue schemaName, final SQLStatementBinderContext binderContext) {
135         IdentifierValue tableName = segment.getTableName().getIdentifier();
136         if (!binderContext.getMetaData().getDatabase(databaseName.getValue()).getSchema(schemaName.getValue()).containsTable(tableName.getValue())) {
137             return new SimpleTableSegmentBinderContext(Collections.emptyList());
138         }
139         Collection<ProjectionSegment> projectionSegments = new LinkedList<>();
140         QuoteCharacter quoteCharacter = new DatabaseTypeRegistry(binderContext.getDatabaseType()).getDialectDatabaseMetaData().getQuoteCharacter();
141         for (ShardingSphereColumn each : schema.getTable(tableName.getValue()).getColumnValues()) {
142             ColumnProjectionSegment columnProjectionSegment = new ColumnProjectionSegment(createColumnSegment(segment, databaseName, schemaName, each, quoteCharacter, tableName));
143             columnProjectionSegment.setVisible(each.isVisible());
144             projectionSegments.add(columnProjectionSegment);
145         }
146         return new SimpleTableSegmentBinderContext(projectionSegments);
147     }
148     
149     private static ColumnSegment createColumnSegment(final SimpleTableSegment segment, final IdentifierValue databaseName, final IdentifierValue schemaName,
150                                                      final ShardingSphereColumn column, final QuoteCharacter quoteCharacter, final IdentifierValue tableName) {
151         ColumnSegment result = new ColumnSegment(0, 0, new IdentifierValue(column.getName(), quoteCharacter));
152         result.setOwner(new OwnerSegment(0, 0, segment.getAlias().orElse(tableName)));
153         result.setColumnBoundInfo(new ColumnSegmentBoundInfo(databaseName, schemaName, tableName, new IdentifierValue(column.getName(), quoteCharacter)));
154         return result;
155     }
