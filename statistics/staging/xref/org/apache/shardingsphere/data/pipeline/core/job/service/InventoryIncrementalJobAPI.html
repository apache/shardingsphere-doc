26  import org.apache.shardingsphere.data.pipeline.common.pojo.InventoryIncrementalJobItemInfo;
27  import org.apache.shardingsphere.data.pipeline.core.consistencycheck.ConsistencyCheckJobItemProgressContext;
28  import org.apache.shardingsphere.data.pipeline.core.consistencycheck.algorithm.DataConsistencyCalculateAlgorithm;
29  import org.apache.shardingsphere.data.pipeline.core.consistencycheck.result.DataConsistencyCheckResult;

31  import java.sql.SQLException;
32  import java.util.Collection;
33  import java.util.List;
34  import java.util.Map;
35  import java.util.Optional;
36  import java.util.Properties;
37  
38  /**
39   * Inventory incremental job API.
40   */
41  public interface InventoryIncrementalJobAPI extends PipelineJobAPI {
42      
43      /**
44       * Alter process configuration.
45       *
46       * @param contextKey context key
47       * @param processConfig process configuration
48       */
49      void alterProcessConfiguration(PipelineContextKey contextKey, PipelineProcessConfiguration processConfig);
50      
51      /**
52       * Show process configuration.
53       *
54       * @param contextKey context key
55       * @return process configuration, non-null
56       */
57      PipelineProcessConfiguration showProcessConfiguration(PipelineContextKey contextKey);
58      
59      /**
60       * Persist job offset info.
61       *
62       * @param jobId job ID
63       * @param jobOffsetInfo job offset info
64       */
65      void persistJobOffsetInfo(String jobId, JobOffsetInfo jobOffsetInfo);
66      
67      /**
68       * Get job offset info.
69       *
70       * @param jobId job ID
71       * @return job offset progress
72       */
73      JobOffsetInfo getJobOffsetInfo(String jobId);
74      
75      /**
76       * Get job progress.
77       *
78       * @param pipelineJobConfig job configuration
79       * @return each sharding item progress
80       */
81      Map<Integer, InventoryIncrementalJobItemProgress> getJobProgress(PipelineJobConfiguration pipelineJobConfig);
82      
83      @Override
84      Optional<InventoryIncrementalJobItemProgress> getJobItemProgress(String jobId, int shardingItem);
85      
86      /**
87       * Get job infos.
88       *
89       * @param jobId job ID
90       * @return job item infos
91       */
92      List<InventoryIncrementalJobItemInfo> getJobItemInfos(String jobId);
93      
94      /**
95       * List all data consistency check algorithms from SPI.
96       *
97       * @return data consistency check algorithms
98       */
99      Collection<DataConsistencyCheckAlgorithmInfo> listDataConsistencyCheckAlgorithms();
100     
101     /**
102      * Build data consistency calculate algorithm.
103      *
104      * @param algorithmType algorithm type
105      * @param algorithmProps algorithm properties
106      * @return calculate algorithm
107      */
108     DataConsistencyCalculateAlgorithm buildDataConsistencyCalculateAlgorithm(String algorithmType, Properties algorithmProps);
109     
110     /**
111      * Do data consistency check.
112      *
113      * @param pipelineJobConfig job configuration
114      * @param calculateAlgorithm calculate algorithm
115      * @param progressContext consistency check job item progress context
116      * @return each logic table check result
117      */
118     Map<String, DataConsistencyCheckResult> dataConsistencyCheck(PipelineJobConfiguration pipelineJobConfig, DataConsistencyCalculateAlgorithm calculateAlgorithm,
119                                                                  ConsistencyCheckJobItemProgressContext progressContext);
120     
121     /**
122      * Aggregate data consistency check results.
123      *
124      * @param jobId job ID
125      * @param checkResults check results
126      * @return check success or not
127      */
128     boolean aggregateDataConsistencyCheckResults(String jobId, Map<String, DataConsistencyCheckResult> checkResults);
129     
130     /**
131      * Commit pipeline job.
132      *
133      * @param jobId job ID
134      */
135     void commit(String jobId);
136     
137     /**
138      * Rollback pipeline job.
139      *
140      * @param jobId job ID
141      * @throws SQLException when rollback underlying database data
142      */
143     void rollback(String jobId) throws SQLException;
144 }
