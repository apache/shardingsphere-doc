46  @ExtendWith(AutoMockExtension.class)
47  @ConstructionMockSettings(BackendTransactionManager.class)
48  class MySQLComResetConnectionExecutorTest {
49      
50      @Test
51      void assertExecute() throws SQLException {
52          ConnectionSession connectionSession = mock(ConnectionSession.class);
53          BackendConnection backendConnection = mock(BackendConnection.class);
54          when(connectionSession.getBackendConnection()).thenReturn(backendConnection);
55          when(connectionSession.getTransactionStatus()).thenReturn(new TransactionStatus(TransactionType.LOCAL));
56          when(connectionSession.getServerPreparedStatementRegistry()).thenReturn(new ServerPreparedStatementRegistry());
57          int statementId = 1;
58          connectionSession.getServerPreparedStatementRegistry().addPreparedStatement(statementId, new MySQLServerPreparedStatement("", null, Collections.emptyList()));
59          Collection<DatabasePacket<?>> actual = new MySQLComResetConnectionExecutor(connectionSession).execute();
60          assertThat(actual.size(), is(1));
61          assertThat(actual.iterator().next(), instanceOf(MySQLOKPacket.class));
62          verify(connectionSession).setAutoCommit(true);
63          verify(connectionSession).setDefaultIsolationLevel(null);
64          verify(connectionSession).setIsolationLevel(null);
65          assertNull(connectionSession.getServerPreparedStatementRegistry().getPreparedStatement(statementId));
66      }
67  }
