21  import com.ecwid.consul.json.GsonFactory;

23  import com.ecwid.consul.v1.ConsulClient;
24  import com.ecwid.consul.v1.OperationException;
25  import com.ecwid.consul.v1.QueryParams;
26  import com.ecwid.consul.v1.Response;
27  import com.ecwid.consul.v1.kv.model.GetValue;
28  import com.ecwid.consul.v1.kv.model.PutParams;
29  import com.ecwid.consul.v1.session.model.NewSession;
30  import com.ecwid.consul.v1.session.model.Session.Behavior;
31  import com.google.common.base.Strings;
32  import com.google.common.reflect.TypeToken;
33  import org.apache.shardingsphere.mode.repository.cluster.consul.ShardingSphereConsulClient;
34  import org.apache.shardingsphere.mode.repository.cluster.consul.ShardingSphereQueryParams;
35  import org.apache.shardingsphere.mode.repository.cluster.consul.props.ConsulProperties;
36  import org.apache.shardingsphere.mode.repository.cluster.consul.props.ConsulPropertyKey;
37  import org.apache.shardingsphere.mode.repository.cluster.lock.DistributedLock;
38  
39  import java.util.List;
40  import java.util.concurrent.ScheduledThreadPoolExecutor;
41  import java.util.concurrent.TimeUnit;
42  
43  /**
44   * Consul distributed lock.
45   */
46  public final class ConsulDistributedLock implements DistributedLock {
47      
48      private static final String LOCK_PATH_PATTERN = "lock/%s";
49      
50      private static final String LOCK_VALUE = "LOCKED";
51      
52      private static final String UNLOCK_VALUE = "UNLOCKED";
53      
54      private static final ScheduledThreadPoolExecutor SESSION_FLUSH_EXECUTOR = new ScheduledThreadPoolExecutor(2);
55      
56      private final String lockPath;
57      
58      private final ConsulClient client;
59      
60      private final String timeToLiveSeconds;
61      
62      private final ThreadLocal<String> lockSessionId;
63      
64      public ConsulDistributedLock(final String lockKey, final ConsulClient client, final ConsulProperties props) {
65          lockPath = String.format(LOCK_PATH_PATTERN, lockKey);
66          this.client = client;
67          timeToLiveSeconds = props.getValue(ConsulPropertyKey.TIME_TO_LIVE_SECONDS);
68          lockSessionId = new ThreadLocal<>();
69      }
70      
71      @Override
72      public boolean tryLock(final long timeoutMillis) {
73          if (!Strings.isNullOrEmpty(lockSessionId.get())) {
74              return true;
75          }
76          PutParams putParams = new PutParams();
77          long remainingMillis = timeoutMillis;
78          while (true) {
79              String sessionId = createSessionId();
80              putParams.setAcquireSession(sessionId);
81              Response<Boolean> response = client.setKVValue(lockPath, LOCK_VALUE, putParams);
82              if (response.getValue()) {
83                  return tryLock(sessionId);
84              }
85              client.sessionDestroy(sessionId, null);
86              long waitingMillis = waitUntilRelease(response.getConsulIndex(), remainingMillis);
87              if (waitingMillis >= remainingMillis) {
88                  return false;
89              }
90              remainingMillis -= waitingMillis;
91          }
92      }
93      
94      private boolean tryLock(final String sessionId) {
95          lockSessionId.set(sessionId);
96          SESSION_FLUSH_EXECUTOR.scheduleAtFixedRate(() -> client.renewSession(sessionId, QueryParams.DEFAULT), 5L, 10L, TimeUnit.SECONDS);
97          return true;
98      }
99      
100     private String createSessionId() {
101         NewSession session = new NewSession();
102         session.setName(lockPath);
103         session.setTtl(timeToLiveSeconds);
104         session.setBehavior(Behavior.RELEASE);
105         return client.sessionCreate(session, null).getValue();
106     }
107     
108     private long waitUntilRelease(final long valueIndex, final long timeoutMillis) {
109         long currentIndex = valueIndex < 0 ? 0 : valueIndex;
110         long spentMillis = 0L;
111         long timeoutTime = System.currentTimeMillis() + timeoutMillis;
112         long remainingMillis = timeoutMillis;
113         while (true) {
114             long startTime = System.currentTimeMillis();
115             if (startTime >= timeoutTime) {
116                 return timeoutMillis;
117             }
118             Response<GetValue> response = getResponse(
119                     ((ShardingSphereConsulClient) client).getRawClient().makeGetRequest(String.format("/v1/kv/%s", lockPath), null, new ShardingSphereQueryParams(remainingMillis, currentIndex)));
120             spentMillis += System.currentTimeMillis() - startTime;
121             remainingMillis -= spentMillis;
122             Long index = response.getConsulIndex();
123             if (null != index && index >= currentIndex) {
124                 if (0 != currentIndex && (null == response.getValue() || null == response.getValue().getValue() || lockPath.equals(response.getValue().getKey()))) {
125                     return spentMillis;
126                 }
127                 currentIndex = index;
128                 continue;
129             }
130             if (null != index) {
131                 currentIndex = 0;
132             }
133         }
134     }
135     
136     private Response<GetValue> getResponse(final RawResponse rawResponse) {
137         if (200 == rawResponse.getStatusCode()) {
138             List<GetValue> value = GsonFactory.getGson().fromJson(rawResponse.getContent(), new TypeToken<List<GetValue>>() {
139                 
140                 private static final long serialVersionUID = -5065504617907914417L;
141                 
142             }.getType());
143             if (value.isEmpty()) {
144                 return new Response<>(null, rawResponse);
145             }
146             if (1 == value.size()) {
147                 return new Response<>(value.get(0), rawResponse);
148             }
149             throw new ConsulException("Strange response (list size=" + value.size() + ")");
150         }
151         if (404 == rawResponse.getStatusCode()) {
152             return new Response<>(null, rawResponse);
153         }
154         throw new OperationException(rawResponse);
155     }
156     
157     @Override
158     public void unlock() {
159         String sessionId = lockSessionId.get();
160         PutParams putParams = new PutParams();
161         putParams.setReleaseSession(sessionId);
162         try {
163             client.setKVValue(lockPath, UNLOCK_VALUE, putParams);
164             client.sessionDestroy(sessionId, null);
165         } finally {
166             lockSessionId.remove();
167         }
168     }
169 }
