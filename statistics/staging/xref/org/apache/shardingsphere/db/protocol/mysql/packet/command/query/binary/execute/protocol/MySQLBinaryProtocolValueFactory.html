18  package org.apache.shardingsphere.db.protocol.mysql.packet.command.query.binary.execute.protocol;
19  
20  import com.google.common.base.Preconditions;
21  import lombok.AccessLevel;
22  import lombok.NoArgsConstructor;
23  import org.apache.shardingsphere.db.protocol.binary.BinaryColumnType;
24  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLBinaryColumnType;
25  
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  /**
30   * Binary protocol value factory for MySQL.
31   */
32  @NoArgsConstructor(access = AccessLevel.PRIVATE)
33  public final class MySQLBinaryProtocolValueFactory {
34      
35      private static final Map<BinaryColumnType, MySQLBinaryProtocolValue> BINARY_PROTOCOL_VALUES = new HashMap<>();
36      
37      static {
38          setStringLenencBinaryProtocolValue();
39          setByteLenencBinaryProtocolValue();
40          setInt8BinaryProtocolValue();
41          setInt4BinaryProtocolValue();
42          setInt2BinaryProtocolValue();
43          setInt1BinaryProtocolValue();
44          setDoubleBinaryProtocolValue();
45          setFloatBinaryProtocolValue();
46          setDateBinaryProtocolValue();
47          setTimeBinaryProtocolValue();
48          setNullBinaryProtocolValue();
49      }
50      
51      private static void setStringLenencBinaryProtocolValue() {
52          MySQLStringLenencBinaryProtocolValue binaryProtocolValue = new MySQLStringLenencBinaryProtocolValue();
53          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.VARCHAR, binaryProtocolValue);
54          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.VAR_STRING, binaryProtocolValue);
55          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.ENUM, binaryProtocolValue);
56          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.SET, binaryProtocolValue);
57          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.GEOMETRY, binaryProtocolValue);
58          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.BIT, binaryProtocolValue);
59          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DECIMAL, binaryProtocolValue);
60          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.NEWDECIMAL, binaryProtocolValue);
61      }
62      
63      private static void setByteLenencBinaryProtocolValue() {
64          MySQLByteLenencBinaryProtocolValue binaryProtocolValue = new MySQLByteLenencBinaryProtocolValue();
65          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.STRING, binaryProtocolValue);
66          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.LONG_BLOB, binaryProtocolValue);
67          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.MEDIUM_BLOB, binaryProtocolValue);
68          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.BLOB, binaryProtocolValue);
69          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TINY_BLOB, binaryProtocolValue);
70      }
71      
72      private static void setInt8BinaryProtocolValue() {
73          MySQLInt8BinaryProtocolValue binaryProtocolValue = new MySQLInt8BinaryProtocolValue();
74          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.LONGLONG, binaryProtocolValue);
75      }
76      
77      private static void setInt4BinaryProtocolValue() {
78          MySQLInt4BinaryProtocolValue binaryProtocolValue = new MySQLInt4BinaryProtocolValue();
79          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.LONG, binaryProtocolValue);
80          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.INT24, binaryProtocolValue);
81      }
82      
83      private static void setInt2BinaryProtocolValue() {
84          MySQLInt2BinaryProtocolValue binaryProtocolValue = new MySQLInt2BinaryProtocolValue();
85          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.SHORT, binaryProtocolValue);
86          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.YEAR, binaryProtocolValue);
87      }
88      
89      private static void setInt1BinaryProtocolValue() {
90          MySQLInt1BinaryProtocolValue binaryProtocolValue = new MySQLInt1BinaryProtocolValue();
91          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TINY, binaryProtocolValue);
92      }
93      
94      private static void setDoubleBinaryProtocolValue() {
95          MySQLDoubleBinaryProtocolValue binaryProtocolValue = new MySQLDoubleBinaryProtocolValue();
96          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DOUBLE, binaryProtocolValue);
97      }
98      
99      private static void setFloatBinaryProtocolValue() {
100         MySQLFloatBinaryProtocolValue binaryProtocolValue = new MySQLFloatBinaryProtocolValue();
101         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.FLOAT, binaryProtocolValue);
102     }
103     
104     private static void setDateBinaryProtocolValue() {
105         MySQLDateBinaryProtocolValue binaryProtocolValue = new MySQLDateBinaryProtocolValue();
106         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DATE, binaryProtocolValue);
107         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DATETIME, binaryProtocolValue);
108         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TIMESTAMP, binaryProtocolValue);
109     }
110     
111     private static void setTimeBinaryProtocolValue() {
112         MySQLTimeBinaryProtocolValue binaryProtocolValue = new MySQLTimeBinaryProtocolValue();
113         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TIME, binaryProtocolValue);
114     }
115     
116     private static void setNullBinaryProtocolValue() {
117         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.NULL, null);
118     }
119     
120     /**
121      * Get binary protocol value.
122      *
123      * @param binaryColumnType binary column type
124      * @return binary protocol value
125      */
126     public static MySQLBinaryProtocolValue getBinaryProtocolValue(final BinaryColumnType binaryColumnType) {
127         Preconditions.checkArgument(BINARY_PROTOCOL_VALUES.containsKey(binaryColumnType), "Cannot find MySQL type '%s' in column type when process binary protocol value", binaryColumnType);
128         return BINARY_PROTOCOL_VALUES.get(binaryColumnType);
129     }
130 }
