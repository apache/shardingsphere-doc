18  package org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.ddl.index;
19  
20  import org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.ddl.PostgreSQLDDLTemplateExecutor;
21  import org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.template.PostgreSQLPipelineFreemarkerManager;
22  import org.postgresql.jdbc.PgArray;
23  
24  import java.sql.Connection;
25  import java.sql.SQLException;
26  import java.util.Collection;
27  import java.util.LinkedHashMap;
28  import java.util.LinkedList;
29  import java.util.Map;
30  
31  /**
32   * Index SQL generator for PostgreSQL.
33   */
34  public final class PostgreSQLIndexSQLGenerator {
35      
36      private static final Integer PG_INDEX_INCLUDE_VERSION = 11;
37      
38      private final PostgreSQLDDLTemplateExecutor templateExecutor;
39      
40      public PostgreSQLIndexSQLGenerator(final Connection connection, final int majorVersion, final int minorVersion) {
41          templateExecutor = new PostgreSQLDDLTemplateExecutor(connection, majorVersion, minorVersion);
42      }
43      
44      /**
45       * Generate create index SQL.
46       *
47       * @param context context
48       * @return generated SQL
49       * @throws SQLException SQL exception
50       */
51      public String generate(final Map<String, Object> context) throws SQLException {
52          StringBuilder result = new StringBuilder();
53          Collection<Map<String, Object>> indexNodes = getIndexNodes(context);
54          for (Map<String, Object> each : indexNodes) {
55              if (each.containsKey("is_inherited") && (Boolean) each.get("is_inherited")) {
56                  continue;
57              }
58              result.append(getIndexSql(context, each));
59          }
60          return result.toString().trim();
61      }
62      
63      private Collection<Map<String, Object>> getIndexNodes(final Map<String, Object> context) {
64          Map<String, Object> param = new LinkedHashMap<>();
65          param.put("tid", context.get("tid"));
66          return templateExecutor.executeByTemplate(param, "component/indexes/%s/nodes.ftl");
67      }
68      
69      private String getIndexSql(final Map<String, Object> context, final Map<String, Object> indexNode) throws SQLException {
70          Map<String, Object> indexData = getIndexData(context, indexNode);
71          appendColumnDetails(indexData, (Long) indexNode.get("oid"));
72          if (templateExecutor.getMajorVersion() >= PG_INDEX_INCLUDE_VERSION) {
73              appendIncludeDetails(indexData, (Long) indexNode.get("oid"));
74          }
75          return doGenerateIndexSql(indexData);
76      }
77      
78      private String doGenerateIndexSql(final Map<String, Object> indexData) {
79          String result = PostgreSQLPipelineFreemarkerManager.getSQLByVersion(indexData, "component/indexes/%s/create.ftl", templateExecutor.getMajorVersion(), templateExecutor.getMinorVersion());
80          result += System.lineSeparator();
81          result += PostgreSQLPipelineFreemarkerManager.getSQLByVersion(indexData, "component/indexes/%s/alter.ftl", templateExecutor.getMajorVersion(), templateExecutor.getMinorVersion());
82          return result;
83      }
84      
85      private Map<String, Object> getIndexData(final Map<String, Object> context, final Map<String, Object> indexNode) {
86          Collection<Map<String, Object>> indexProps = fetchIndexProperties(context, indexNode);
87          Map<String, Object> result = indexProps.iterator().next();
88          result.put("schema", context.get("schema"));
89          result.put("table", context.get("name"));
90          return result;
91      }
92      
93      private Collection<Map<String, Object>> fetchIndexProperties(final Map<String, Object> context, final Map<String, Object> indexNode) {
94          Map<String, Object> param = new LinkedHashMap<>();
95          param.put("did", context.get("did"));
96          param.put("tid", context.get("tid"));
97          param.put("idx", indexNode.get("oid"));
98          param.put("datlastsysoid", context.get("datlastsysoid"));
99          return templateExecutor.executeByTemplate(param, "component/indexes/%s/properties.ftl");
100     }
101     
102     private void appendColumnDetails(final Map<String, Object> indexData, final Long indexId) throws SQLException {
103         Collection<Map<String, Object>> columnDetails = fetchColumnDetails(indexId);
104         Collection<Map<String, Object>> columns = new LinkedList<>();
105         Collection<String> columnDisplays = new LinkedList<>();
106         for (Map<String, Object> each : columnDetails) {
107             columns.add(getColumnData(indexData, each));
108             columnDisplays.add(getColumnPropertyDisplayData(each, indexData));
109         }
110         indexData.put("columns", columns);
111         indexData.put("columns_csv", String.join(", ", columnDisplays));
112     }
113     
114     private Map<String, Object> getColumnData(final Map<String, Object> indexData, final Map<String, Object> columnDetail) throws SQLException {
115         Map<String, Object> result = new LinkedHashMap<>();
116         result.put("colname", columnDetail.get("attdef"));
117         result.put("collspcname", columnDetail.get("collnspname"));
118         result.put("op_class", columnDetail.get("opcname"));
119         if ("btree".equals(indexData.get("amname"))) {
120             result.put("sort_order", isSortOrder(columnDetail));
121             result.put("nulls", isNulls(columnDetail));
122         }
123         return result;
124     }
125     
126     private boolean isSortOrder(final Map<String, Object> columnDetail) throws SQLException {
127         if (null == columnDetail.get("options")) {
128             return false;
129         }
130         String[] options = (String[]) ((PgArray) columnDetail.get("options")).getArray();
131         return options.length > 0 && "DESC".equals(options[0]);
132     }
133     
134     private Object isNulls(final Map<String, Object> columnDetail) throws SQLException {
135         if (null == columnDetail.get("options")) {
136             return false;
137         }
138         String[] options = (String[]) ((PgArray) columnDetail.get("options")).getArray();
139         return options.length > 1 && options[1].split(" ").length > 1 && "FIRST".equals(options[1].split(" ")[1]);
140     }
141     
142     private Collection<Map<String, Object>> fetchColumnDetails(final Long indexId) {
143         Map<String, Object> param = new LinkedHashMap<>();
144         param.put("idx", indexId);
145         return templateExecutor.executeByTemplate(param, "component/indexes/%s/column_details.ftl");
146     }
147     
148     private String getColumnPropertyDisplayData(final Map<String, Object> columnDetail, final Map<String, Object> indexData) throws SQLException {
149         String result = (String) columnDetail.get("attdef");
150         if (null != columnDetail.get("collnspname")) {
151             result += " COLLATE " + columnDetail.get("collnspname");
152         }
153         if (null != columnDetail.get("opcname")) {
154             result += " " + columnDetail.get("opcname");
155         }
156         if ("btree".equals(indexData.get("amname"))) {
157             String[] options = (String[]) ((PgArray) columnDetail.get("options")).getArray();
158             if (options.length > 0) {
159                 result += " " + options[0];
160             }
161             if (options.length > 1) {
162                 result += " " + options[1];
163             }
164         }
165         return result;
166     }
167     
168     private void appendIncludeDetails(final Map<String, Object> indexData, final Long oid) {
169         Map<String, Object> param = new LinkedHashMap<>();
170         param.put("idx", oid);
171         Collection<Object> includes = new LinkedList<>();
172         for (Map<String, Object> each : templateExecutor.executeByTemplate(param, "component/indexes/%s/include_details.ftl")) {
173             includes.add(each.get("colname"));
174         }
175         indexData.put("include", includes);
176     }
177 }
