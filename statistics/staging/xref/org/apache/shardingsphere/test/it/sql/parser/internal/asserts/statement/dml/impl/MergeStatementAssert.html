47  import static org.junit.jupiter.api.Assertions.assertNull;
48  import static org.junit.jupiter.api.Assertions.assertTrue;

50  /**
51   * Merge statement assert.
52   */
53  @NoArgsConstructor(access = AccessLevel.PRIVATE)
54  public final class MergeStatementAssert {
55      
56      /**
57       * Assert merge statement is correct with expected parser result.
58       *
59       * @param assertContext assert context
60       * @param actual actual merge statement
61       * @param expected expected parser result
62       */
63      public static void assertIs(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
64          assertTable(assertContext, actual, expected);
65          assertExpression(assertContext, actual, expected);
66          assertSetClause(assertContext, actual, expected);
67          assertWhereClause(assertContext, actual, expected);
68          assertWithClause(assertContext, actual, expected);
69          assertWithTableHintClause(assertContext, actual, expected);
70          assertOutputClause(assertContext, actual, expected);
71          assertWhenAndThenSegments(assertContext, actual, expected);
72          assertIndexes(assertContext, actual, expected);
73      }
74      
75      private static void assertWithClause(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
76          Optional<WithSegment> withSegment = actual.getWithSegment();
77          if (null == expected.getWithClause()) {
78              assertFalse(withSegment.isPresent(), assertContext.getText("Actual with segment should not exist."));
79          } else {
80              assertTrue(withSegment.isPresent(), assertContext.getText("Actual with segment should exist."));
81              WithClauseAssert.assertIs(assertContext, withSegment.get(), expected.getWithClause());
82          }
83      }
84      
85      private static void assertWithTableHintClause(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
86          Optional<WithTableHintSegment> withTableHintSegment = actual.getWithTableHintSegment();
87          if (null == expected.getExpectedWithTableHintClause()) {
88              assertFalse(withTableHintSegment.isPresent(), assertContext.getText("Actual with table hint should not exist."));
89          } else {
90              assertTrue(withTableHintSegment.isPresent(), assertContext.getText("Actual with table hint segment should exist."));
91              WithTableHintClauseAssert.assertIs(assertContext, withTableHintSegment.get(), expected.getExpectedWithTableHintClause());
92          }
93      }
94      
95      private static void assertOutputClause(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
96          Optional<OutputSegment> outputSegment = actual.getOutputSegment();
97          if (null == expected.getOutputClause()) {
98              assertFalse(outputSegment.isPresent(), assertContext.getText("Actual output segment should not exist."));
99          } else {
100             assertTrue(outputSegment.isPresent(), assertContext.getText("Actual output segment should exist."));
101             OutputClauseAssert.assertIs(assertContext, outputSegment.get(), expected.getOutputClause());
102         }
103     }
104     
105     private static void assertWhenAndThenSegments(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
106         Collection<MergeWhenAndThenSegment> mergeWhenAndThenSegments = actual.getWhenAndThenSegments();
107         assertThat(assertContext.getText("merge when and then segment assertion error: "), mergeWhenAndThenSegments.size(), CoreMatchers.is(expected.getMergeWhenAndThenSegments().size()));
108         int count = 0;
109         for (MergeWhenAndThenSegment each : mergeWhenAndThenSegments) {
110             asserMergeWhenAndTheSegment(assertContext, each, expected.getMergeWhenAndThenSegments().get(count));
111             count++;
112         }
113     }
114     
115     private static void asserMergeWhenAndTheSegment(final SQLCaseAssertContext assertContext, final MergeWhenAndThenSegment actual, final ExpectedMergeWhenAndThenSegment expected) {
116         if (null == expected.getExpr()) {
117             assertNull(actual.getAndExpr(), assertContext.getText("Actual and expression should not exist."));
118         } else {
119             ExpressionAssert.assertExpression(assertContext, actual.getAndExpr(), expected.getExpr());
120         }
121         if (null == expected.getUpdateClause()) {
122             assertNull(actual.getUpdate(), assertContext.getText("Actual update statement should not exist."));
123         } else {
124             UpdateStatementAssert.assertIs(assertContext, actual.getUpdate(), expected.getUpdateClause());
125         }
126         if (null == expected.getInsertClause()) {
127             assertNull(actual.getInsert(), assertContext.getText("Actual insert statement should not exist."));
128         } else {
129             InsertStatementAssert.assertIs(assertContext, actual.getInsert(), expected.getInsertClause());
130         }
131     }
132     
133     private static void assertIndexes(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
134         Collection<IndexSegment> indexes = actual.getIndexes();
135         assertThat(assertContext.getText("index segment assertion error: "), indexes.size(), CoreMatchers.is(expected.getIndexs().size()));
136         int count = 0;
137         for (IndexSegment each : indexes) {
138             IndexAssert.assertIs(assertContext, each, expected.getIndexs().get(count));
139             count++;
140         }
141     }
142     
143     private static void assertTable(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
144         if (null == expected.getSource()) {
145             assertNull(actual.getSource(), assertContext.getText("Actual source should not exist."));
146         } else {
147             TableAssert.assertIs(assertContext, actual.getSource(), expected.getSource());
148         }
149         if (null == expected.getTarget()) {
150             assertNull(actual.getTarget(), assertContext.getText("Actual target should not exist."));
151         } else {
152             TableAssert.assertIs(assertContext, actual.getTarget(), expected.getTarget());
153         }
154     }
155     
156     private static void assertExpression(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
157         if (null == expected.getExpr()) {
158             assertNull(actual.getExpression(), assertContext.getText("Actual expression should not exist."));
159         } else {
160             ExpressionAssert.assertExpression(assertContext, actual.getExpression().getExpr(), expected.getExpr());
161         }
162     }
163     
164     private static void assertSetClause(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
165         if (null != expected.getUpdateClause()) {
166             assertTrue(actual.getUpdate().isPresent(), assertContext.getText("Actual merge update statement should exist."));
167             if (null == expected.getUpdateClause().getSetClause()) {
168                 assertNull(actual.getUpdate().get().getSetAssignment(), assertContext.getText("Actual assignment should not exist."));
169             } else {
170                 SetClauseAssert.assertIs(assertContext, actual.getUpdate().get().getSetAssignment(), expected.getUpdateClause().getSetClause());
171             }
172         }
173     }
174     
175     private static void assertWhereClause(final SQLCaseAssertContext assertContext, final MergeStatement actual, final MergeStatementTestCase expected) {
176         if (null != expected.getUpdateClause()) {
177             assertTrue(actual.getUpdate().isPresent(), assertContext.getText("Actual merge update statement should exist."));
178             if (null == expected.getUpdateClause().getWhereClause()) {
179                 assertFalse(actual.getUpdate().get().getWhere().isPresent(), assertContext.getText("Actual update where segment should not exist."));
180             } else {
181                 assertTrue(actual.getUpdate().get().getWhere().isPresent(), assertContext.getText("Actual update where segment should exist."));
182                 WhereClauseAssert.assertIs(assertContext, actual.getUpdate().get().getWhere().get(), expected.getUpdateClause().getWhereClause());
183             }
184         }
185         if (null != expected.getInsertClause() && null != expected.getInsertClause().getWhereClause() && actual.getInsert().orElse(null) instanceof OracleInsertStatement) {
186             assertTrue(actual.getInsert().isPresent(), assertContext.getText("Actual merge insert statement should exist."));
187             assertTrue(((OracleInsertStatement) actual.getInsert().get()).getWhere().isPresent(), assertContext.getText("Actual insert where segment should exist."));
188             WhereClauseAssert.assertIs(assertContext, ((OracleInsertStatement) actual.getInsert().get()).getWhere().get(), expected.getInsertClause().getWhereClause());
189         }
190     }
