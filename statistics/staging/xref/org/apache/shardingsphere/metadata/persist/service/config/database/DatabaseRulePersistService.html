20  import com.google.common.base.Strings;
21  import org.apache.shardingsphere.infra.config.rule.RuleConfiguration;
22  import org.apache.shardingsphere.infra.metadata.version.MetaDataVersion;
23  import org.apache.shardingsphere.infra.yaml.config.pojo.rule.YamlRuleConfiguration;
24  import org.apache.shardingsphere.infra.yaml.config.swapper.rule.YamlRuleConfigurationSwapperEngine;
25  import org.apache.shardingsphere.metadata.persist.node.metadata.DatabaseRuleMetaDataNode;
26  import org.apache.shardingsphere.metadata.persist.service.config.RepositoryTuplePersistService;
27  import org.apache.shardingsphere.mode.spi.PersistRepository;
28  import org.apache.shardingsphere.mode.tuple.RepositoryTuple;
29  import org.apache.shardingsphere.mode.tuple.RepositoryTupleSwapperEngine;
30  import org.apache.shardingsphere.mode.tuple.annotation.RepositoryTupleEntity;
31  
32  import java.util.Collection;
33  import java.util.Collections;
34  import java.util.LinkedList;
35  import java.util.List;
36  import java.util.Objects;
37  
38  /**
39   * Database rule persist service.
40   */
41  public final class DatabaseRulePersistService {
42      
43      private final PersistRepository repository;
44      
45      private final RepositoryTuplePersistService repositoryTuplePersistService;
46      
47      public DatabaseRulePersistService(final PersistRepository repository) {
48          this.repository = repository;
49          repositoryTuplePersistService = new RepositoryTuplePersistService(repository);
50      }
51      
52      /**
53       * Load configurations.
54       *
55       * @param databaseName database name
56       * @return configurations
57       */
58      public Collection<RuleConfiguration> load(final String databaseName) {
59          return new RepositoryTupleSwapperEngine().swapToRuleConfigurations(repositoryTuplePersistService.load(DatabaseRuleMetaDataNode.getRulesNode(databaseName)));
60      }
61      
62      /**
63       * Persist configurations.
64       *
65       * @param databaseName database name
66       * @param configs to be persisted configurations
67       * @return meta data versions
68       */
69      public Collection<MetaDataVersion> persist(final String databaseName, final Collection<RuleConfiguration> configs) {
70          Collection<MetaDataVersion> result = new LinkedList<>();
71          RepositoryTupleSwapperEngine repositoryTupleSwapperEngine = new RepositoryTupleSwapperEngine();
72          for (YamlRuleConfiguration each : new YamlRuleConfigurationSwapperEngine().swapToYamlRuleConfigurations(configs)) {
73              Collection<RepositoryTuple> repositoryTuples = repositoryTupleSwapperEngine.swapToRepositoryTuples(each);
74              if (!repositoryTuples.isEmpty()) {
75                  result.addAll(persistDataNodes(databaseName, Objects.requireNonNull(each.getClass().getAnnotation(RepositoryTupleEntity.class)).value(), repositoryTuples));
76              }
77          }
78          return result;
79      }
80      
81      private Collection<MetaDataVersion> persistDataNodes(final String databaseName, final String ruleName, final Collection<RepositoryTuple> repositoryTuples) {
82          Collection<MetaDataVersion> result = new LinkedList<>();
83          for (RepositoryTuple each : repositoryTuples) {
84              List<String> versions = repository.getChildrenKeys(DatabaseRuleMetaDataNode.getDatabaseRuleVersionsNode(databaseName, ruleName, each.getKey()));
85              String nextVersion = versions.isEmpty() ? MetaDataVersion.DEFAULT_VERSION : String.valueOf(Integer.parseInt(versions.get(0)) + 1);
86              repository.persist(DatabaseRuleMetaDataNode.getDatabaseRuleVersionNode(databaseName, ruleName, each.getKey(), nextVersion), each.getValue());
87              if (Strings.isNullOrEmpty(getActiveVersion(databaseName, ruleName, each.getKey()))) {
88                  repository.persist(DatabaseRuleMetaDataNode.getDatabaseRuleActiveVersionNode(databaseName, ruleName, each.getKey()), MetaDataVersion.DEFAULT_VERSION);
89              }
90              result.add(new MetaDataVersion(DatabaseRuleMetaDataNode.getDatabaseRuleNode(databaseName, ruleName, each.getKey()), getActiveVersion(databaseName, ruleName, each.getKey()), nextVersion));
91          }
92          return result;
93      }
94      
95      private String getActiveVersion(final String databaseName, final String ruleName, final String key) {
96          return repository.query(DatabaseRuleMetaDataNode.getDatabaseRuleActiveVersionNode(databaseName, ruleName, key));
97      }
98      
99      /**
100      * Delete configurations.
101      *
102      * @param databaseName database name
103      * @param ruleTypeName rule type name
104      */
105     public void delete(final String databaseName, final String ruleTypeName) {
106         repository.delete(DatabaseRuleMetaDataNode.getDatabaseRuleNode(databaseName, ruleTypeName));
107     }
108     
109     /**
110      * Delete configurations.
111      *
112      * @param databaseName database name
113      * @param configs to be deleted configurations
114      * @return meta data versions
115      */
116     public Collection<MetaDataVersion> delete(final String databaseName, final Collection<RuleConfiguration> configs) {
117         Collection<MetaDataVersion> result = new LinkedList<>();
118         RepositoryTupleSwapperEngine repositoryTupleSwapperEngine = new RepositoryTupleSwapperEngine();
119         for (YamlRuleConfiguration each : new YamlRuleConfigurationSwapperEngine().swapToYamlRuleConfigurations(configs)) {
120             Collection<RepositoryTuple> repositoryTuples = repositoryTupleSwapperEngine.swapToRepositoryTuples(each);
121             if (repositoryTuples.isEmpty()) {
122                 continue;
123             }
124             List<RepositoryTuple> newRepositoryTuples = new LinkedList<>(repositoryTuples);
125             Collections.reverse(newRepositoryTuples);
126             result.addAll(delete(databaseName, Objects.requireNonNull(each.getClass().getAnnotation(RepositoryTupleEntity.class)).value(), newRepositoryTuples));
127         }
128         return result;
129     }
130     
131     private Collection<MetaDataVersion> delete(final String databaseName, final String ruleName, final Collection<RepositoryTuple> repositoryTuples) {
132         Collection<MetaDataVersion> result = new LinkedList<>();
133         for (RepositoryTuple each : repositoryTuples) {
134             String toBeDeletedKey = DatabaseRuleMetaDataNode.getDatabaseRuleNode(databaseName, ruleName, each.getKey());
135             repository.delete(toBeDeletedKey);
136             result.add(new MetaDataVersion(toBeDeletedKey));
137         }
138         return result;
139     }
140 }
