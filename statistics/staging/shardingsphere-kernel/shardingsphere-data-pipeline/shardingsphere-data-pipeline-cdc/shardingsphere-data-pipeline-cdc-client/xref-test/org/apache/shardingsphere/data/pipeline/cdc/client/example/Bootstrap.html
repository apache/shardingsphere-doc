18  package org.apache.shardingsphere.data.pipeline.cdc.client.example;
19  
20  import lombok.extern.slf4j.Slf4j;
21  import org.apache.shardingsphere.data.pipeline.cdc.client.CDCClient;
22  import org.apache.shardingsphere.data.pipeline.cdc.client.parameter.StartCDCClientParameter;
23  import org.apache.shardingsphere.data.pipeline.cdc.protocol.request.StreamDataRequestBody.SchemaTable;
24  
25  import java.util.Collections;
26  
27  @Slf4j
28  public final class Bootstrap {
29      
30      /**
31       * Main entrance.
32       *
33       * @param args args
34       */
35      public static void main(final String[] args) {
36          // Pay attention to the time zone, to avoid the problem of incorrect time zone, it is best to ensure that the time zone of the program is consistent with the time zone of the database server
37          // and mysql-connector-java 5.x version will ignore serverTimezone jdbc parameter and use the default time zone in the program
38          // TimeZone.setDefault(TimeZone.getTimeZone("UTC"));
39          StartCDCClientParameter parameter = new StartCDCClientParameter(records -> log.info("records: {}", records));
40          parameter.setAddress("127.0.0.1");
41          parameter.setPort(33071);
42          parameter.setUsername("root");
43          parameter.setPassword("root");
44          parameter.setDatabase("sharding_db");
45          parameter.setFull(true);
46          parameter.setSchemaTables(Collections.singletonList(SchemaTable.newBuilder().setTable("t_order").build()));
47          CDCClient cdcClient = new CDCClient(parameter);
48          cdcClient.start();
49      }
50  }
