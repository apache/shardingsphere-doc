53  /**
54   * DML statement visitor for PostgreSQL.
55   */
56  public final class PostgreSQLDMLStatementVisitor extends PostgreSQLStatementVisitor implements DMLStatementVisitor {
57      
58      @Override
59      public ASTNode visitCall(final CallContext ctx) {
60          PostgreSQLCallStatement result = new PostgreSQLCallStatement();
61          result.setProcedureName(((IdentifierValue) visit(ctx.identifier())).getValue());
62          if (null != ctx.callArguments()) {
63              Collection<ExpressionSegment> params = new LinkedList<>();
64              for (CallArgumentContext each : ctx.callArguments().callArgument()) {
65                  params.add((ExpressionSegment) visit(each));
66              }
67              result.getParameters().addAll(params);
68          }
69          return result;
70      }
71      
72      @Override
73      public ASTNode visitCallArgument(final CallArgumentContext ctx) {
74          if (null == ctx.positionalNotation()) {
75              String text = ctx.namedNotation().start.getInputStream().getText(new Interval(ctx.namedNotation().start.getStartIndex(), ctx.namedNotation().stop.getStopIndex()));
76              return new CommonExpressionSegment(ctx.namedNotation().getStart().getStartIndex(), ctx.namedNotation().getStop().getStopIndex(), text);
77          }
78          return visit(ctx.positionalNotation().aExpr());
79      }
80      
81      @Override
82      public ASTNode visitDoStatement(final DoStatementContext ctx) {
83          return new PostgreSQLDoStatement();
84      }
85      
86      @Override
87      public ASTNode visitCopy(final CopyContext ctx) {
88          if (null != ctx.copyWithTableOrQuery()) {
89              return visit(ctx.copyWithTableOrQuery());
90          }
91          if (null != ctx.copyWithTableOrQueryBinaryCsv()) {
92              return visit(ctx.copyWithTableOrQueryBinaryCsv());
93          }
94          return visit(ctx.copyWithTableBinary());
95      }
96      
97      @Override
98      public ASTNode visitCopyWithTableOrQuery(final CopyWithTableOrQueryContext ctx) {
99          PostgreSQLCopyStatement result = new PostgreSQLCopyStatement();
100         if (null != ctx.qualifiedName()) {
101             result.setTableSegment((SimpleTableSegment) visit(ctx.qualifiedName()));
102             if (null != ctx.columnNames()) {
103                 result.getColumns().addAll(((CollectionValue<ColumnSegment>) visit(ctx.columnNames())).getValue());
104             }
105         }
106         if (null != ctx.preparableStmt()) {
107             result.setPrepareStatementQuery(extractPrepareStatementQuerySegmentFromPreparableStmt(ctx.preparableStmt()));
108         }
109         return result;
110     }
111     
112     private PrepareStatementQuerySegment extractPrepareStatementQuerySegmentFromPreparableStmt(final PreparableStmtContext ctx) {
113         PrepareStatementQuerySegment result = new PrepareStatementQuerySegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex());
114         if (null != ctx.select()) {
115             result.setSelect((SelectStatement) visit(ctx.select()));
116         } else if (null != ctx.insert()) {
117             result.setInsert((InsertStatement) visit(ctx.insert()));
118         } else if (null != ctx.update()) {
119             result.setUpdate((UpdateStatement) visit(ctx.update()));
120         } else {
121             result.setDelete((DeleteStatement) visit(ctx.delete()));
122         }
123         return result;
124     }
125     
126     @Override
127     public ASTNode visitCopyWithTableOrQueryBinaryCsv(final CopyWithTableOrQueryBinaryCsvContext ctx) {
128         PostgreSQLCopyStatement result = new PostgreSQLCopyStatement();
129         if (null != ctx.qualifiedName()) {
130             result.setTableSegment((SimpleTableSegment) visit(ctx.qualifiedName()));
131             if (null != ctx.columnNames()) {
132                 result.getColumns().addAll(((CollectionValue<ColumnSegment>) visit(ctx.columnNames())).getValue());
133             }
134         }
135         if (null != ctx.preparableStmt()) {
136             result.setPrepareStatementQuery(extractPrepareStatementQuerySegmentFromPreparableStmt(ctx.preparableStmt()));
137         }
138         return result;
139     }
140     
141     @Override
142     public ASTNode visitCopyWithTableBinary(final CopyWithTableBinaryContext ctx) {
143         PostgreSQLCopyStatement result = new PostgreSQLCopyStatement();
144         if (null != ctx.qualifiedName()) {
145             result.setTableSegment((SimpleTableSegment) visit(ctx.qualifiedName()));
146         }
147         return result;
148     }
149     
150     @Override
151     public ASTNode visitReturningClause(final ReturningClauseContext ctx) {
152         return new ReturningSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), (ProjectionsSegment) visit(ctx.targetList()));
153     }
154 }
