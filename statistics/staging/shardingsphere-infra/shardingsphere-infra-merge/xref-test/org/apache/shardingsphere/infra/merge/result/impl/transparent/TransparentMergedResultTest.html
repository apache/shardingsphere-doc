18  package org.apache.shardingsphere.infra.merge.result.impl.transparent;
19  
20  import org.apache.shardingsphere.infra.executor.sql.execute.result.query.QueryResult;
21  import org.junit.jupiter.api.Test;
22  
23  import java.io.InputStream;
24  import java.sql.Date;
25  import java.sql.SQLException;
26  
27  import static org.hamcrest.CoreMatchers.is;
28  import static org.junit.jupiter.api.Assertions.assertFalse;
29  import static org.hamcrest.MatcherAssert.assertThat;
30  import static org.junit.jupiter.api.Assertions.assertTrue;
31  import static org.mockito.Mockito.mock;
32  import static org.mockito.Mockito.when;
33  
34  class TransparentMergedResultTest {
35      
36      @Test
37      void assertNext() throws SQLException {
38          QueryResult queryResult = mock(QueryResult.class);
39          when(queryResult.next()).thenReturn(true, false);
40          TransparentMergedResult actual = new TransparentMergedResult(queryResult);
41          assertTrue(actual.next());
42          assertFalse(actual.next());
43      }
44      
45      @Test
46      void assertGetValue() throws SQLException {
47          QueryResult queryResult = mock(QueryResult.class);
48          when(queryResult.getValue(1, Object.class)).thenReturn("1");
49          TransparentMergedResult actual = new TransparentMergedResult(queryResult);
50          assertThat(actual.getValue(1, Object.class).toString(), is("1"));
51      }
52      
53      @Test
54      void assertGetCalendarValue() throws SQLException {
55          QueryResult queryResult = mock(QueryResult.class);
56          when(queryResult.getCalendarValue(1, Date.class, null)).thenReturn(new Date(0L));
57          TransparentMergedResult actual = new TransparentMergedResult(queryResult);
58          assertThat(actual.getCalendarValue(1, Date.class, null), is(new Date(0L)));
59      }
60      
61      @Test
62      void assertGetInputStream() throws SQLException {
63          QueryResult queryResult = mock(QueryResult.class);
64          InputStream value = mock(InputStream.class);
65          when(queryResult.getInputStream(1, "Ascii")).thenReturn(value);
66          TransparentMergedResult actual = new TransparentMergedResult(queryResult);
67          assertThat(actual.getInputStream(1, "Ascii"), is(value));
68      }
69      
70      @Test
71      void assertWasNull() throws SQLException {
72          TransparentMergedResult actual = new TransparentMergedResult(mock(QueryResult.class));
73          assertFalse(actual.wasNull());
74      }
75  }
