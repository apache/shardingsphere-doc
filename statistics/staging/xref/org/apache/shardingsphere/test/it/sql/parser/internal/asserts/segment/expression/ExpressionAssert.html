312      * Assert function.
313      *
314      * @param assertContext assert context
315      * @param actual actual function segment
316      * @param expected expected function segment
317      */
318     public static void assertFunction(final SQLCaseAssertContext assertContext, final FunctionSegment actual, final ExpectedFunction expected) {
319         SQLSegmentAssert.assertIs(assertContext, actual, expected);
320         assertThat(assertContext.getText("Function method name assertion error: "), actual.getFunctionName(), is(expected.getFunctionName()));
321         String expectedText = SQLCaseType.LITERAL == assertContext.getCaseType() && null != expected.getLiteralText()
322                 ? expected.getLiteralText()
323                 : expected.getText();
324         assertThat(assertContext.getText("Function text name assertion error: "), actual.getText(), is(expectedText));
325         assertThat(assertContext.getText("Function parameter size assertion error: "), actual.getParameters().size(), is(expected.getParameters().size()));
326         Iterator<ExpectedExpression> expectedIterator = expected.getParameters().iterator();
327         Iterator<ExpressionSegment> actualIterator = actual.getParameters().iterator();
328         while (expectedIterator.hasNext()) {
329             ExpressionAssert.assertExpression(assertContext, actualIterator.next(), expectedIterator.next());
330         }
331         if (expected.getOwner() != null) {
332             OwnerAssert.assertIs(assertContext, actual.getOwner(), expected.getOwner());
333         }
334     }

354      *
355      * @param assertContext assert context
356      * @param actual actual case when expression
357      * @param expected expected case when expression
358      */
359     public static void assertCaseWhenExpression(final SQLCaseAssertContext assertContext, final CaseWhenExpression actual, final ExpectedCaseWhenExpression expected) {
360         assertThat(assertContext.getText("When exprs size is not same!"), actual.getWhenExprs().size(), is(expected.getWhenExprs().size()));
361         assertThat(assertContext.getText("Then exprs size is not same!"), actual.getThenExprs().size(), is(expected.getThenExprs().size()));
362         Iterator<ExpectedExpression> whenExprsIterator = expected.getWhenExprs().iterator();
363         for (ExpressionSegment each : actual.getWhenExprs()) {
364             assertExpression(assertContext, each, whenExprsIterator.next());
365         }
366         Iterator<ExpectedExpression> thenExprsIterator = expected.getThenExprs().iterator();
367         for (ExpressionSegment each : actual.getThenExprs()) {
368             assertExpression(assertContext, each, thenExprsIterator.next());
369         }
370         assertExpression(assertContext, actual.getCaseExpr(), expected.getCaseExpr());
371         assertExpression(assertContext, actual.getElseExpr(), expected.getElseExpr());
372     }

491         } else {
492             assertNotNull(actual, assertContext.getText("Actual interval expression should exist"));
493             if (null != actual.getLeadingFieldPrecision()) {
494                 assertEquals(actual.getLeadingFieldPrecision(), expected.getLeadingFieldPrecision());
495             } else {
496                 assertNull(expected.getLeadingFieldPrecision(), assertContext.getText("Actual leading field precision should not exist."));
497             }
498         }
499     }
500     
501     /**
502      * Assert expression by actual expression segment class type.
503      *
504      * @param assertContext assert context
505      * @param actual actual expression segment
506      * @param expected expected expression
507      * @throws UnsupportedOperationException When expression segment class type is not supported.
508      */
509     public static void assertExpression(final SQLCaseAssertContext assertContext,
510                                         final ExpressionSegment actual, final ExpectedExpression expected) {
511         if (null == expected) {
512             assertNull(actual, assertContext.getText("Actual expression should not exist."));
513             return;
514         }
515         assertNotNull(actual, assertContext.getText("Actual expression should exist."));
516         if (actual instanceof BinaryOperationExpression) {
517             assertBinaryOperationExpression(assertContext, (BinaryOperationExpression) actual, expected.getBinaryOperationExpression());
518         } else if (actual instanceof SubqueryExpressionSegment) {
519             assertSubqueryExpression(assertContext, (SubqueryExpressionSegment) actual, expected.getSubquery());
520         } else if (actual instanceof ColumnSegment) {
521             ColumnAssert.assertIs(assertContext, (ColumnSegment) actual, expected.getColumn());
522         } else if (actual instanceof DataTypeSegment) {
523             DataTypeAssert.assertIs(assertContext, (DataTypeSegment) actual, expected.getDataType());
524         } else if (actual instanceof LiteralExpressionSegment) {
525             assertLiteralExpression(assertContext, (LiteralExpressionSegment) actual, expected.getLiteralExpression());
526         } else if (actual instanceof ParameterMarkerExpressionSegment) {
527             assertParameterMarkerExpression(assertContext, (ParameterMarkerExpressionSegment) actual, expected.getParameterMarkerExpression());
528         } else if (actual instanceof ExistsSubqueryExpression) {
529             assertExistsSubqueryExpression(assertContext, (ExistsSubqueryExpression) actual, expected.getExistsSubquery());
530         } else if (actual instanceof CommonExpressionSegment) {
531             assertCommonExpression(assertContext, (ComplexExpressionSegment) actual, expected.getCommonExpression());
532         } else if (actual instanceof InExpression) {
533             assertInExpression(assertContext, (InExpression) actual, expected.getInExpression());
534         } else if (actual instanceof NotExpression) {
535             assertNotExpression(assertContext, (NotExpression) actual, expected.getNotExpression());
536         } else if (actual instanceof ListExpression) {
537             assertListExpression(assertContext, (ListExpression) actual, expected.getListExpression());
538         } else if (actual instanceof BetweenExpression) {
539             assertBetweenExpression(assertContext, (BetweenExpression) actual, expected.getBetweenExpression());
540         } else if (actual instanceof ExpressionProjectionSegment) {
541             ProjectionAssert.assertProjection(assertContext, (ExpressionProjectionSegment) actual, expected.getExpressionProjection());
542         } else if (actual instanceof AggregationProjectionSegment) {
543             ProjectionAssert.assertProjection(assertContext, (AggregationProjectionSegment) actual, expected.getAggregationProjection());
544         } else if (actual instanceof FunctionSegment) {
545             assertFunction(assertContext, (FunctionSegment) actual, expected.getFunction());
546         } else if (actual instanceof CollateExpression) {
547             assertCollateExpression(assertContext, (CollateExpression) actual, expected.getCollateExpression());
548         } else if (actual instanceof CaseWhenExpression) {
549             assertCaseWhenExpression(assertContext, (CaseWhenExpression) actual, expected.getCaseWhenExpression());
550         } else if (actual instanceof TypeCastExpression) {
551             assertTypeCastExpression(assertContext, (TypeCastExpression) actual, expected.getTypeCastExpression());
552         } else if (actual instanceof VariableSegment) {
553             assertVariableSegment(assertContext, (VariableSegment) actual, expected.getVariableSegment());
554         } else if (actual instanceof ValuesExpression) {
555             assertValuesExpression(assertContext, (ValuesExpression) actual, expected.getValuesExpression());
556         } else if (actual instanceof ExtractArgExpression) {
557             assertExtractArgExpression(assertContext, (ExtractArgExpression) actual, expected.getExtractArgExpression());
558         } else if (actual instanceof MatchAgainstExpression) {
559             assertMatchSegment(assertContext, (MatchAgainstExpression) actual, expected.getMatchExpression());
560         } else if (actual instanceof ColumnWithJoinOperatorSegment) {
561             ColumnWithJoinOperatorAssert.assertIs(assertContext, (ColumnWithJoinOperatorSegment) actual, expected.getColumnWithJoinOperatorSegment());
562         } else if (actual instanceof IntervalExpressionProjection) {
563             assertIntervalExpression(assertContext, (IntervalExpressionProjection) actual, expected.getIntervalExpression());
564         } else {
565             throw new UnsupportedOperationException(String.format("Unsupported expression: %s", actual.getClass().getName()));
566         }
567     }
