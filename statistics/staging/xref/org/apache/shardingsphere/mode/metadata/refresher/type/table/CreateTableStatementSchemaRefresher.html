42  /**
43   * Schema refresher for create table statement.
44   */
45  public final class CreateTableStatementSchemaRefresher implements MetaDataRefresher<CreateTableStatement> {
46      
47      @Override
48      public void refresh(final MetaDataManagerPersistService metaDataManagerPersistService, final ShardingSphereDatabase database, final Collection<String> logicDataSourceNames,
49                          final String schemaName, final DatabaseType databaseType, final CreateTableStatement sqlStatement, final ConfigurationProperties props) throws SQLException {
50          String tableName = TableRefreshUtils.getTableName(databaseType, sqlStatement.getTable().getTableName().getIdentifier());
51          RuleMetaData ruleMetaData = new RuleMetaData(new LinkedList<>(database.getRuleMetaData().getRules()));
52          boolean isSingleTable = TableRefreshUtils.isSingleTable(tableName, database);
53          if (isSingleTable) {
54              ruleMetaData.getAttributes(MutableDataNodeRuleAttribute.class).forEach(each -> each.put(logicDataSourceNames.iterator().next(), schemaName, tableName));
55          }
56          GenericSchemaBuilderMaterial material = new GenericSchemaBuilderMaterial(
57                  database.getProtocolType(), database.getResourceMetaData().getStorageUnits(), ruleMetaData.getRules(), props, schemaName);
58          Map<String, ShardingSphereSchema> schemaMap = GenericSchemaBuilder.build(Collections.singletonList(tableName), material);
59          Optional<ShardingSphereTable> actualTableMetaData = Optional.ofNullable(schemaMap.get(schemaName)).map(optional -> optional.getTable(tableName));
60          if (actualTableMetaData.isPresent()) {
61              AlterSchemaMetaDataPOJO alterSchemaMetaDataPOJO = new AlterSchemaMetaDataPOJO(database.getName(), schemaName, logicDataSourceNames);
62              alterSchemaMetaDataPOJO.getAlteredTables().add(actualTableMetaData.get());
63              metaDataManagerPersistService.alterSchemaMetaData(alterSchemaMetaDataPOJO);
64              if (isSingleTable && TableRefreshUtils.isRuleRefreshRequired(ruleMetaData, schemaName, tableName)) {
65                  metaDataManagerPersistService.alterSingleRuleConfiguration(database.getName(), ruleMetaData.getConfigurations());
66              }
67          }
68      }
69      
70      @Override
71      public Class<CreateTableStatement> getType() {
72          return CreateTableStatement.class;
73      }
74  }
