135     private ShardingSpherePreparedStatement(final ShardingSphereConnection connection, final String sql, final int resultSetType,
136                                             final int resultSetConcurrency, final int resultSetHoldability, final boolean returnGeneratedKeys, final String[] columns) throws SQLException {
137         ShardingSpherePreconditions.checkNotEmpty(sql, () -> new EmptySQLException().toSQLException());
138         this.connection = connection;
139         metaData = connection.getContextManager().getMetaDataContexts().getMetaData();
140         this.sql = SQLHintUtils.removeHint(sql);
141         hintValueContext = SQLHintUtils.extractHint(sql);
142         SQLStatement sqlStatement = parseSQL(connection);
143         sqlStatementContext = new SQLBindEngine(metaData, connection.getDatabaseName(), hintValueContext).bind(sqlStatement, Collections.emptyList());
144         String databaseName = sqlStatementContext instanceof TableAvailable
145                 ? ((TableAvailable) sqlStatementContext).getTablesContext().getDatabaseName().orElse(connection.getDatabaseName())
146                 : connection.getDatabaseName();
147         connection.getDatabaseConnectionManager().getConnectionContext().setCurrentDatabase(databaseName);
148         database = metaData.getDatabase(databaseName);
149         statementOption = returnGeneratedKeys ? new StatementOption(true, columns) : new StatementOption(resultSetType, resultSetConcurrency, resultSetHoldability);
150         statementManager = new StatementManager();
151         connection.getStatementManagers().add(statementManager);
152         parameterMetaData = new ShardingSphereParameterMetaData(sqlStatement);
153         driverExecutorFacade = new DriverExecutorFacade(connection, statementOption, statementManager, JDBCDriverType.PREPARED_STATEMENT);
154         executeBatchExecutor = new DriverExecuteBatchExecutor(connection, metaData, statementOption, statementManager, database);
155         statementsCacheable = isStatementsCacheable();
156     }
157     
158     private SQLStatement parseSQL(final ShardingSphereConnection connection) {
159         SQLParserRule sqlParserRule = metaData.getGlobalRuleMetaData().getSingleRule(SQLParserRule.class);
160         return sqlParserRule.getSQLParserEngine(metaData.getDatabase(connection.getDatabaseName()).getProtocolType()).parse(sql, true);
161     }
162     
163     private boolean isStatementsCacheable() {
164         return database.getRuleMetaData().getAttributes(StorageConnectorReusableRuleAttribute.class).size() == database.getRuleMetaData().getRules().size() && !HintManager.isInstantiated();
165     }
166     
167     @Override
168     public ResultSet executeQuery() throws SQLException {
169         try {
170             if (statementsCacheable && !statements.isEmpty()) {
171                 resetParameters();
172                 return statements.iterator().next().executeQuery();
173             }
174             clearPrevious();
175             QueryContext queryContext = createQueryContext();
176             handleAutoCommit(queryContext.getSqlStatementContext().getSqlStatement());
177             findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
178             currentResultSet = driverExecutorFacade.executeQuery(database, queryContext, this, columnLabelAndIndexMap, (StatementAddCallback<PreparedStatement>) this::addStatements, this::replay);
179             if (currentResultSet instanceof ShardingSphereResultSet) {
180                 columnLabelAndIndexMap = ((ShardingSphereResultSet) currentResultSet).getColumnLabelAndIndexMap();
181             }
182             return currentResultSet;
183             // CHECKSTYLE:OFF
184         } catch (final RuntimeException ex) {
185             // CHECKSTYLE:ON
186             handleExceptionInTransaction(connection, metaData);
187             throw SQLExceptionTransformEngine.toSQLException(ex, database.getProtocolType());
188         } finally {
189             executeBatchExecutor.clear();
190             clearParameters();
191         }
192     }
193     
194     private void handleAutoCommit(final SQLStatement sqlStatement) throws SQLException {
195         if (AutoCommitUtils.needOpenTransaction(sqlStatement)) {
196             connection.handleAutoCommit();
197         }
198     }

200     private void addStatements(final Collection<PreparedStatement> statements, final Collection<List<Object>> parameterSets) {
201         this.statements.addAll(statements);
202         this.parameterSets.addAll(parameterSets);
203     }
204     
205     private void resetParameters() throws SQLException {
206         replaySetParameter(statements, Collections.singletonList(getParameters()));
207     }
208     
209     @Override
210     public int executeUpdate() throws SQLException {
211         try {
212             if (statementsCacheable && !statements.isEmpty()) {
213                 resetParameters();
214                 return statements.iterator().next().executeUpdate();
215             }
216             clearPrevious();
217             QueryContext queryContext = createQueryContext();
218             handleAutoCommit(queryContext.getSqlStatementContext().getSqlStatement());
219             int result = driverExecutorFacade.executeUpdate(database, queryContext,
220                     (sql, statement) -> ((PreparedStatement) statement).executeUpdate(), (StatementAddCallback<PreparedStatement>) this::addStatements, this::replay);
221             findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
222             return result;
223             // CHECKSTYLE:OFF
224         } catch (final RuntimeException ex) {
225             // CHECKSTYLE:ON
226             handleExceptionInTransaction(connection, metaData);
227             throw SQLExceptionTransformEngine.toSQLException(ex, database.getProtocolType());
228         } finally {
229             clearBatch();
230         }
231     }
232     
233     @Override
234     public boolean execute() throws SQLException {
235         try {
236             if (statementsCacheable && !statements.isEmpty()) {
237                 resetParameters();
238                 return statements.iterator().next().execute();
239             }
240             clearPrevious();
241             QueryContext queryContext = createQueryContext();
242             handleAutoCommit(queryContext.getSqlStatementContext().getSqlStatement());
243             boolean result = driverExecutorFacade.execute(database, queryContext, (sql, statement) -> ((PreparedStatement) statement).execute(),
244                     (StatementAddCallback<PreparedStatement>) this::addStatements, this::replay);
245             findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
246             return result;
247             // CHECKSTYLE:OFF
248         } catch (final RuntimeException ex) {
249             // CHECKSTYLE:ON
250             handleExceptionInTransaction(connection, metaData);
251             throw SQLExceptionTransformEngine.toSQLException(ex, database.getProtocolType());
252         } finally {
253             clearBatch();
254         }
255     }
256     
257     @Override
258     public ResultSet getResultSet() throws SQLException {
259         if (null != currentResultSet) {
260             return currentResultSet;
261         }
262         driverExecutorFacade.getResultSet(database, sqlStatementContext, this, statements).ifPresent(optional -> currentResultSet = optional);
263         if (null == columnLabelAndIndexMap && currentResultSet instanceof ShardingSphereResultSet) {
264             columnLabelAndIndexMap = ((ShardingSphereResultSet) currentResultSet).getColumnLabelAndIndexMap();
265         }
266         return currentResultSet;
267     }
268     
269     private QueryContext createQueryContext() {
270         List<Object> params = new ArrayList<>(getParameters());
271         if (sqlStatementContext instanceof ParameterAware) {
272             ((ParameterAware) sqlStatementContext).setUpParameters(params);
273         }
274         return new QueryContext(sqlStatementContext, sql, params, hintValueContext, true);
275     }
276     
277     private void replay() throws SQLException {
278         replaySetParameter(statements, parameterSets);
279         for (Statement each : statements) {
280             getMethodInvocationRecorder().replay(each);
281         }
282     }
283     
284     private void replaySetParameter(final List<PreparedStatement> statements, final List<List<Object>> parameterSets) throws SQLException {
285         for (int i = 0; i < statements.size(); i++) {
286             replaySetParameter(statements.get(i), parameterSets.get(i));
287         }
288     }
289     
290     private void clearPrevious() {
291         currentResultSet = null;
292         statements.clear();
293         parameterSets.clear();
294         generatedValues.clear();
295     }
296     
297     private Optional<GeneratedKeyContext> findGeneratedKey() {
298         return sqlStatementContext instanceof InsertStatementContext ? ((InsertStatementContext) sqlStatementContext).getGeneratedKeyContext() : Optional.empty();
299     }
300     
301     @Override
302     public ResultSet getGeneratedKeys() throws SQLException {
303         if (null != currentBatchGeneratedKeysResultSet) {
304             return currentBatchGeneratedKeysResultSet;
305         }
306         Optional<GeneratedKeyContext> generatedKey = findGeneratedKey();
307         if (generatedKey.isPresent() && statementOption.isReturnGeneratedKeys() && !generatedValues.isEmpty()) {
308             return new GeneratedKeysResultSet(getGeneratedKeysColumnName(generatedKey.get().getColumnName()), generatedValues.iterator(), this);
309         }
310         for (PreparedStatement each : statements) {
311             ResultSet resultSet = each.getGeneratedKeys();
312             while (resultSet.next()) {
313                 generatedValues.add((Comparable<?>) resultSet.getObject(1));
314             }
315         }
316         String columnName = generatedKey.map(GeneratedKeyContext::getColumnName).orElse(null);
317         return new GeneratedKeysResultSet(getGeneratedKeysColumnName(columnName), generatedValues.iterator(), this);
318     }
319     
320     private String getGeneratedKeysColumnName(final String columnName) {
321         return DatabaseTypedSPILoader.findService(GeneratedKeyColumnProvider.class, database.getProtocolType())
322                 .map(GeneratedKeyColumnProvider::getColumnName).orElse(columnName);
323     }
324     
325     @Override
326     public void addBatch() {
327         currentResultSet = null;
328         QueryContext queryContext = createQueryContext();
329         executeBatchExecutor.addBatch(queryContext, database);
330         findGeneratedKey().ifPresent(optional -> generatedValues.addAll(optional.getGeneratedValues()));
331         clearParameters();
332     }
333     
334     @Override
335     public int[] executeBatch() throws SQLException {
336         try {
337             return executeBatchExecutor.executeBatch(database, sqlStatementContext, generatedValues, statementOption,
338                     (StatementAddCallback<PreparedStatement>) (statements, parameterSets) -> this.statements.addAll(statements),
339                     this::replaySetParameter,
340                     () -> {
341                         currentBatchGeneratedKeysResultSet = getGeneratedKeys();
342                         statements.clear();
343                     });
344             // CHECKSTYLE:OFF
345         } catch (final RuntimeException ex) {
346             // CHECKSTYLE:ON
347             handleExceptionInTransaction(connection, metaData);
348             throw SQLExceptionTransformEngine.toSQLException(ex, database.getProtocolType());
349         } finally {
350             clearBatch();
351         }
352     }
353     
354     @Override
355     public void clearBatch() {
356         currentResultSet = null;
357         executeBatchExecutor.clear();
358         clearParameters();
359     }

378     @Override
379     public boolean isAccumulate() {
380         if (!(sqlStatementContext instanceof TableAvailable)) {
381             return false;
382         }
383         for (DataNodeRuleAttribute each : database.getRuleMetaData().getAttributes(DataNodeRuleAttribute.class)) {
384             if (each.isNeedAccumulate(((TableAvailable) sqlStatementContext).getTablesContext().getTableNames())) {
385                 return true;
386             }
387         }
388         return false;
389     }
