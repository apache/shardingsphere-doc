39  import java.util.Collection;

42  
43  /**
44   * Single SQL router.
45   */
46  @HighFrequencyInvocation
47  public final class SingleSQLRouter implements EntranceSQLRouter<SingleRule>, DecorateSQLRouter<SingleRule> {
48      
49      @Override
50      public RouteContext createRouteContext(final QueryContext queryContext, final RuleMetaData globalRuleMetaData, final ShardingSphereDatabase database,
51                                             final SingleRule rule, final ConfigurationProperties props) {
52          if (1 == database.getResourceMetaData().getStorageUnits().size()) {
53              return createSingleDataSourceRouteContext(rule, database, queryContext);
54          }
55          RouteContext result = new RouteContext();
56          SQLStatementContext sqlStatementContext = queryContext.getSqlStatementContext();
57          SingleMetaDataValidatorFactory.newInstance(sqlStatementContext.getSqlStatement()).ifPresent(optional -> optional.validate(rule, sqlStatementContext, database));
58          Collection<QualifiedTable> singleTables = getSingleTables(database, rule, result, sqlStatementContext);
59          SingleRouteEngineFactory.newInstance(singleTables, sqlStatementContext.getSqlStatement(), queryContext.getHintValueContext()).ifPresent(optional -> optional.route(result, rule));
60          return result;
61      }
62      
63      private Collection<QualifiedTable> getSingleTables(final ShardingSphereDatabase database, final SingleRule rule, final RouteContext routeContext, final SQLStatementContext sqlStatementContext) {
64          Collection<QualifiedTable> qualifiedTables = rule.getQualifiedTables(sqlStatementContext, database);
65          return routeContext.getRouteUnits().isEmpty() && sqlStatementContext.getSqlStatement() instanceof CreateTableStatement ? qualifiedTables : rule.getSingleTables(qualifiedTables);
66      }
67      
68      @Override
69      public void decorateRouteContext(final RouteContext routeContext, final QueryContext queryContext, final ShardingSphereDatabase database,
70                                       final SingleRule rule, final ConfigurationProperties props) {
71          SQLStatementContext sqlStatementContext = queryContext.getSqlStatementContext();
72          Collection<QualifiedTable> singleTables = getSingleTables(database, rule, routeContext, sqlStatementContext);
73          SingleRouteEngineFactory.newInstance(singleTables, sqlStatementContext.getSqlStatement(), queryContext.getHintValueContext()).ifPresent(optional -> optional.route(routeContext, rule));
74      }
75      
76      private RouteContext createSingleDataSourceRouteContext(final SingleRule rule, final ShardingSphereDatabase database, final QueryContext queryContext) {
77          String logicDataSource = rule.getDataSourceNames().iterator().next();
78          String actualDataSource = database.getResourceMetaData().getStorageUnits().keySet().iterator().next();
79          RouteContext result = new RouteContext();
80          Collection<String> tableNames = queryContext.getSqlStatementContext() instanceof TableAvailable
81                  ? ((TableAvailable) queryContext.getSqlStatementContext()).getTablesContext().getTableNames()
82                  : Collections.emptyList();
83          result.getRouteUnits().add(new RouteUnit(new RouteMapper(logicDataSource, actualDataSource), createTableMappers(tableNames)));
84          return result;
85      }
86      
87      private Collection<RouteMapper> createTableMappers(final Collection<String> tableNames) {
88          Collection<RouteMapper> result = new LinkedList<>();
89          for (String each : tableNames) {
90              result.add(new RouteMapper(each, each));
91          }
92          return result;
93      }
94      
95      @Override
96      public int getOrder() {
97          return SingleOrder.ORDER;
98      }
99      
100     @Override
101     public Class<SingleRule> getTypeClass() {
102         return SingleRule.class;
103     }
104 }
