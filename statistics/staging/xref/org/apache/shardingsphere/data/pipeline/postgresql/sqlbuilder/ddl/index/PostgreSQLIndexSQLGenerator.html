18  package org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.ddl.index;
19  
20  import org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.ddl.AbstractPostgreSQLDDLAdapter;
21  import org.apache.shardingsphere.data.pipeline.postgresql.util.PostgreSQLPipelineFreemarkerManager;
22  import org.postgresql.jdbc.PgArray;
23  
24  import java.sql.Connection;
25  import java.sql.SQLException;
26  import java.util.Collection;
27  import java.util.LinkedHashMap;
28  import java.util.LinkedList;
29  import java.util.Map;
30  
31  /**
32   * Index SQL generator for PostgreSQL.
33   */
34  public final class PostgreSQLIndexSQLGenerator extends AbstractPostgreSQLDDLAdapter {
35      
36      private static final Integer PG_INDEX_INCLUDE_VERSION = 11;
37      
38      public PostgreSQLIndexSQLGenerator(final Connection connection, final int majorVersion, final int minorVersion) {
39          super(connection, majorVersion, minorVersion);
40      }
41      
42      /**
43       * Generate create index SQL.
44       *
45       * @param context context
46       * @return generated SQL
47       * @throws SQLException SQL exception
48       */
49      public String generate(final Map<String, Object> context) throws SQLException {
50          StringBuilder result = new StringBuilder();
51          Collection<Map<String, Object>> indexNodes = getIndexNodes(context);
52          for (Map<String, Object> each : indexNodes) {
53              if (each.containsKey("is_inherited") && (Boolean) each.get("is_inherited")) {
54                  continue;
55              }
56              result.append(getIndexSql(context, each));
57          }
58          return result.toString().trim();
59      }
60      
61      private Collection<Map<String, Object>> getIndexNodes(final Map<String, Object> context) {
62          Map<String, Object> param = new LinkedHashMap<>();
63          param.put("tid", context.get("tid"));
64          return executeByTemplate(param, "component/indexes/%s/nodes.ftl");
65      }
66      
67      private String getIndexSql(final Map<String, Object> context, final Map<String, Object> indexNode) throws SQLException {
68          Map<String, Object> indexData = getIndexData(context, indexNode);
69          appendColumnDetails(indexData, (Long) indexNode.get("oid"));
70          if (getMajorVersion() >= PG_INDEX_INCLUDE_VERSION) {
71              appendIncludeDetails(indexData, (Long) indexNode.get("oid"));
72          }
73          return doGenerateIndexSql(indexData);
74      }
75      
76      private String doGenerateIndexSql(final Map<String, Object> indexData) {
77          String result = PostgreSQLPipelineFreemarkerManager.getSQLByVersion(indexData, "component/indexes/%s/create.ftl", getMajorVersion(), getMinorVersion());
78          result += System.lineSeparator();
79          result += PostgreSQLPipelineFreemarkerManager.getSQLByVersion(indexData, "component/indexes/%s/alter.ftl", getMajorVersion(), getMinorVersion());
80          return result;
81      }
82      
83      private Map<String, Object> getIndexData(final Map<String, Object> context, final Map<String, Object> indexNode) {
84          Collection<Map<String, Object>> indexProps = fetchIndexProperties(context, indexNode);
85          Map<String, Object> result = indexProps.iterator().next();
86          result.put("schema", context.get("schema"));
87          result.put("table", context.get("name"));
88          return result;
89      }
90      
91      private Collection<Map<String, Object>> fetchIndexProperties(final Map<String, Object> context, final Map<String, Object> indexNode) {
92          Map<String, Object> param = new LinkedHashMap<>();
93          param.put("did", context.get("did"));
94          param.put("tid", context.get("tid"));
95          param.put("idx", indexNode.get("oid"));
96          param.put("datlastsysoid", context.get("datlastsysoid"));
97          return executeByTemplate(param, "component/indexes/%s/properties.ftl");
98      }
99      
100     private void appendColumnDetails(final Map<String, Object> indexData, final Long indexId) throws SQLException {
101         Collection<Map<String, Object>> columnDetails = fetchColumnDetails(indexId);
102         Collection<Map<String, Object>> columns = new LinkedList<>();
103         Collection<String> columnDisplays = new LinkedList<>();
104         for (Map<String, Object> each : columnDetails) {
105             columns.add(getColumnData(indexData, each));
106             columnDisplays.add(getColumnPropertyDisplayData(each, indexData));
107         }
108         indexData.put("columns", columns);
109         indexData.put("columns_csv", String.join(", ", columnDisplays));
110     }
111     
112     private Map<String, Object> getColumnData(final Map<String, Object> indexData, final Map<String, Object> columnDetail) throws SQLException {
113         Map<String, Object> result = new LinkedHashMap<>();
114         result.put("colname", columnDetail.get("attdef"));
115         result.put("collspcname", columnDetail.get("collnspname"));
116         result.put("op_class", columnDetail.get("opcname"));
117         if ("btree".equals(indexData.get("amname"))) {
118             result.put("sort_order", isSortOrder(columnDetail));
119             result.put("nulls", isNulls(columnDetail));
120         }
121         return result;
122     }
123     
124     private boolean isSortOrder(final Map<String, Object> columnDetail) throws SQLException {
125         if (null == columnDetail.get("options")) {
126             return false;
127         }
128         String[] options = (String[]) ((PgArray) columnDetail.get("options")).getArray();
129         return options.length > 0 && "DESC".equals(options[0]);
130     }
131     
132     private Object isNulls(final Map<String, Object> columnDetail) throws SQLException {
133         if (null == columnDetail.get("options")) {
134             return false;
135         }
136         String[] options = (String[]) ((PgArray) columnDetail.get("options")).getArray();
137         return options.length > 1 && options[1].split(" ").length > 1 && "FIRST".equals(options[1].split(" ")[1]);
138     }
139     
140     private Collection<Map<String, Object>> fetchColumnDetails(final Long indexId) {
141         Map<String, Object> param = new LinkedHashMap<>();
142         param.put("idx", indexId);
143         return executeByTemplate(param, "component/indexes/%s/column_details.ftl");
144     }
145     
146     private String getColumnPropertyDisplayData(final Map<String, Object> columnDetail, final Map<String, Object> indexData) throws SQLException {
147         String result = (String) columnDetail.get("attdef");
148         if (null != columnDetail.get("collnspname")) {
149             result += " COLLATE " + columnDetail.get("collnspname");
150         }
151         if (null != columnDetail.get("opcname")) {
152             result += " " + columnDetail.get("opcname");
153         }
154         if ("btree".equals(indexData.get("amname"))) {
155             String[] options = (String[]) ((PgArray) columnDetail.get("options")).getArray();
156             if (options.length > 0) {
157                 result += " " + options[0];
158             }
159             if (options.length > 1) {
160                 result += " " + options[1];
161             }
162         }
163         return result;
164     }
165     
166     private void appendIncludeDetails(final Map<String, Object> indexData, final Long oid) {
167         Map<String, Object> param = new LinkedHashMap<>();
168         param.put("idx", oid);
169         Collection<Object> includes = new LinkedList<>();
170         for (Map<String, Object> each : executeByTemplate(param, "component/indexes/%s/include_details.ftl")) {
171             includes.add(each.get("colname"));
172         }
173         indexData.put("include", includes);
174     }
175 }
