39  import java.util.List;

42  /**
43   * Driver JDBC executor.
44   */
45  public final class DriverJDBCExecutor {
46      
47      private final String databaseName;
48      
49      private final MetaDataContexts metaDataContexts;
50      
51      private final ModeContextManager modeContextManager;
52      
53      private final JDBCExecutor jdbcExecutor;
54      
55      private final ProcessEngine processEngine = new ProcessEngine();
56      
57      public DriverJDBCExecutor(final String databaseName, final ContextManager contextManager, final JDBCExecutor jdbcExecutor) {
58          this.databaseName = databaseName;
59          this.jdbcExecutor = jdbcExecutor;
60          metaDataContexts = contextManager.getMetaDataContexts();
61          modeContextManager = contextManager.getComputeNodeInstanceContext().getModeContextManager();
62      }
63      
64      /**
65       * Execute query.
66       *
67       * @param executionGroupContext execution group context
68       * @param queryContext query context
69       * @param callback execute query callback
70       * @return query results
71       * @throws SQLException SQL exception
72       */
73      public List<QueryResult> executeQuery(final ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext,
74                                            final QueryContext queryContext, final ExecuteQueryCallback callback) throws SQLException {
75          try {
76              processEngine.executeSQL(executionGroupContext, queryContext);
77              return jdbcExecutor.execute(executionGroupContext, callback);
78          } finally {
79              processEngine.completeSQLExecution(executionGroupContext.getReportContext().getProcessId());
80          }
81      }
82      
83      /**
84       * Execute update.
85       *
86       * @param executionGroupContext execution group context
87       * @param queryContext query context
88       * @param routeUnits route units
89       * @param callback JDBC executor callback
90       * @return effected records count
91       * @throws SQLException SQL exception
92       */
93      public int executeUpdate(final ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext,
94                               final QueryContext queryContext, final Collection<RouteUnit> routeUnits, final JDBCExecutorCallback<Integer> callback) throws SQLException {
95          try {
96              processEngine.executeSQL(executionGroupContext, queryContext);
97              SQLStatementContext sqlStatementContext = queryContext.getSqlStatementContext();
98              List<Integer> results = doExecute(executionGroupContext, sqlStatementContext, routeUnits, callback);
99              return isNeedAccumulate(metaDataContexts.getMetaData().getDatabase(queryContext.getDatabaseNameFromSQLStatement().orElse(databaseName)).getRuleMetaData().getRules(), sqlStatementContext)
100                     ? accumulate(results)
101                     : results.get(0);
102         } finally {
103             processEngine.completeSQLExecution(executionGroupContext.getReportContext().getProcessId());
104         }
105     }
106     
107     private boolean isNeedAccumulate(final Collection<ShardingSphereRule> rules, final SQLStatementContext sqlStatementContext) {
108         for (ShardingSphereRule each : rules) {
109             Optional<DataNodeRuleAttribute> ruleAttribute = each.getAttributes().findAttribute(DataNodeRuleAttribute.class);
110             if (ruleAttribute.isPresent() && ruleAttribute.get().isNeedAccumulate(sqlStatementContext.getTablesContext().getTableNames())) {
111                 return true;
112             }
113         }
114         return false;
115     }
116     
117     private int accumulate(final List<Integer> updateResults) {
118         int result = 0;
119         for (Integer each : updateResults) {
120             result += null == each ? 0 : each;
121         }
122         return result;
123     }
124     
125     /**
126      * Execute SQL.
127      *
128      * @param executionGroupContext execution group context
129      * @param queryContext query context
130      * @param routeUnits route units
131      * @param callback JDBC executor callback
132      * @return return true if is DQL, false if is DML
133      * @throws SQLException SQL exception
134      */
135     public boolean execute(final ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext, final QueryContext queryContext,
136                            final Collection<RouteUnit> routeUnits, final JDBCExecutorCallback<Boolean> callback) throws SQLException {
137         try {
138             processEngine.executeSQL(executionGroupContext, queryContext);
139             List<Boolean> results = doExecute(executionGroupContext, queryContext.getSqlStatementContext(), routeUnits, callback);
140             return null != results && !results.isEmpty() && null != results.get(0) && results.get(0);
141         } finally {
142             processEngine.completeSQLExecution(executionGroupContext.getReportContext().getProcessId());
143         }
144     }
145     
146     private <T> List<T> doExecute(final ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext, final SQLStatementContext sqlStatementContext, final Collection<RouteUnit> routeUnits,
147                                   final JDBCExecutorCallback<T> callback) throws SQLException {
148         List<T> results = jdbcExecutor.execute(executionGroupContext, callback);
149         new MetaDataRefreshEngine(modeContextManager,
150                 metaDataContexts.getMetaData().getDatabase(sqlStatementContext.getTablesContext().getDatabaseName().orElse(databaseName)), metaDataContexts.getMetaData().getProps())
151                         .refresh(sqlStatementContext, routeUnits);
152         return results;
153     }
154 }
