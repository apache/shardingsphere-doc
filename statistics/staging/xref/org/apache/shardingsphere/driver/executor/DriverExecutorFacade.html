18  package org.apache.shardingsphere.driver.executor;
19  
20  import lombok.Getter;
21  import org.apache.shardingsphere.driver.jdbc.core.connection.ShardingSphereConnection;
22  import org.apache.shardingsphere.infra.database.core.type.DatabaseTypeRegistry;
23  import org.apache.shardingsphere.infra.executor.sql.execute.engine.driver.jdbc.JDBCExecutor;
24  import org.apache.shardingsphere.infra.executor.sql.execute.engine.raw.RawExecutor;
25  import org.apache.shardingsphere.infra.metadata.ShardingSphereMetaData;
26  import org.apache.shardingsphere.sqlfederation.engine.SQLFederationEngine;
27  import org.apache.shardingsphere.traffic.executor.TrafficExecutor;
28  
29  import java.sql.SQLException;
30  
31  /**
32   * Driver executor facade.
33   */
34  public final class DriverExecutorFacade implements AutoCloseable {
35      
36      private final TrafficExecutor trafficExecutor;
37      
38      private final SQLFederationEngine sqlFederationEngine;
39      
40      @Getter
41      private final DriverExecuteQueryExecutor queryExecutor;
42      
43      @Getter
44      private final DriverExecuteUpdateExecutor updateExecutor;
45      
46      @Getter
47      private final DriverExecuteExecutor executeExecutor;
48      
49      public DriverExecutorFacade(final ShardingSphereConnection connection) {
50          JDBCExecutor jdbcExecutor = new JDBCExecutor(connection.getContextManager().getExecutorEngine(), connection.getDatabaseConnectionManager().getConnectionContext());
51          DriverJDBCExecutor regularExecutor = new DriverJDBCExecutor(connection.getDatabaseName(), connection.getContextManager(), jdbcExecutor);
52          RawExecutor rawExecutor = new RawExecutor(connection.getContextManager().getExecutorEngine(), connection.getDatabaseConnectionManager().getConnectionContext());
53          trafficExecutor = new TrafficExecutor();
54          ShardingSphereMetaData metaData = connection.getContextManager().getMetaDataContexts().getMetaData();
55          String schemaName = new DatabaseTypeRegistry(metaData.getDatabase(connection.getDatabaseName()).getProtocolType()).getDefaultSchemaName(connection.getDatabaseName());
56          sqlFederationEngine = new SQLFederationEngine(connection.getDatabaseName(), schemaName, metaData, connection.getContextManager().getMetaDataContexts().getStatistics(), jdbcExecutor);
57          queryExecutor = new DriverExecuteQueryExecutor(connection, metaData, regularExecutor, rawExecutor, trafficExecutor, sqlFederationEngine);
58          updateExecutor = new DriverExecuteUpdateExecutor(connection, metaData, regularExecutor, rawExecutor, trafficExecutor);
59          executeExecutor = new DriverExecuteExecutor(connection, metaData, regularExecutor, rawExecutor, trafficExecutor, sqlFederationEngine);
60      }
61      
62      @Override
63      public void close() throws SQLException {
64          trafficExecutor.close();
65          sqlFederationEngine.close();
66      }
67  }
