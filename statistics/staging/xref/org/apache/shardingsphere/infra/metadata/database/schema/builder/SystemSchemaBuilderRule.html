209     private static final Map<String, SystemSchemaBuilderRule> SCHEMA_PATH_SYSTEM_SCHEMA_BUILDER_RULE_MAP = new HashMap<>(values().length, 1F);
210     
211     private final String databaseType;
212     
213     private final String schema;
214     
215     private final Collection<String> tables;
216     
217     static {
218         for (SystemSchemaBuilderRule each : values()) {
219             SCHEMA_PATH_SYSTEM_SCHEMA_BUILDER_RULE_MAP.put(each.getDatabaseType() + "." + each.getSchema(), each);
220         }
221     }
222     
223     /**
224      * Value of builder rule.
225      *
226      * @param databaseType database type
227      * @param schema schema
228      * @return builder rule
229      */
230     public static Optional<SystemSchemaBuilderRule> findBuilderRule(final String databaseType, final String schema) {
231         String schemaPath = databaseType + "." + schema;
232         return Optional.ofNullable(SCHEMA_PATH_SYSTEM_SCHEMA_BUILDER_RULE_MAP.get(schemaPath));
233     }
234     
235     /**
236      * Judge whether current table is system table or not.
237      *
238      * @param schema schema
239      * @param tableName table name
240      * @return whether current table is system table or not
241      */
242     public static boolean isSystemTable(final String schema, final String tableName) {
243         for (SystemSchemaBuilderRule each : values()) {
244             if (each.getSchema().equals(schema) && each.getTables().contains(tableName)) {
245                 return true;
246             }
247         }
248         return false;
249     }
