18  package org.apache.shardingsphere.agent.plugin.metrics.prometheus;
19  
20  import com.google.common.base.Strings;
21  import io.prometheus.client.CollectorRegistry;
22  import io.prometheus.client.exporter.HTTPServer;
23  import io.prometheus.client.hotspot.DefaultExports;
24  import lombok.extern.slf4j.Slf4j;
25  import org.apache.shardingsphere.agent.api.PluginConfiguration;
26  import org.apache.shardingsphere.agent.plugin.core.config.validator.PluginConfigurationValidator;
27  import org.apache.shardingsphere.agent.plugin.metrics.core.exporter.impl.BuildInfoExporter;
28  import org.apache.shardingsphere.agent.plugin.metrics.core.exporter.impl.jdbc.JDBCMetaDataInfoExporter;
29  import org.apache.shardingsphere.agent.plugin.metrics.core.exporter.impl.jdbc.JDBCStateExporter;
30  import org.apache.shardingsphere.agent.plugin.metrics.core.exporter.impl.proxy.ProxyMetaDataInfoExporter;
31  import org.apache.shardingsphere.agent.plugin.metrics.core.exporter.impl.proxy.ProxyStateExporter;
32  import org.apache.shardingsphere.agent.plugin.metrics.prometheus.exoprter.PrometheusMetricsExporter;
33  import org.apache.shardingsphere.agent.spi.PluginLifecycleService;
34  
35  import java.io.IOException;
36  import java.net.InetSocketAddress;
37  
38  /**
39   * Prometheus plugin lifecycle service.
40   */
41  @Slf4j
42  public final class PrometheusPluginLifecycleService implements PluginLifecycleService {
43      
44      private static final String KEY_JVM_INFORMATION_COLLECTOR_ENABLED = "jvm-information-collector-enabled";
45      
46      private HTTPServer httpServer;
47      
48      @Override
49      public void start(final PluginConfiguration pluginConfig, final boolean isEnhancedForProxy) {
50          PluginConfigurationValidator.validatePort(getType(), pluginConfig);
51          startServer(pluginConfig, isEnhancedForProxy);
52      }
53      
54      private void startServer(final PluginConfiguration pluginConfig, final boolean isEnhancedForProxy) {
55          registerCollector(Boolean.parseBoolean(pluginConfig.getProps().getProperty(KEY_JVM_INFORMATION_COLLECTOR_ENABLED)), isEnhancedForProxy);
56          InetSocketAddress socketAddress = getSocketAddress(pluginConfig);
57          try {
58              httpServer = new HTTPServer(socketAddress, CollectorRegistry.defaultRegistry, true);
59              log.info("Prometheus metrics HTTP server `{}:{}` start success.", socketAddress.getHostString(), socketAddress.getPort());
60          } catch (final IOException ex) {
61              log.error("Prometheus metrics HTTP server start fail.", ex);
62          }
63      }
64      
65      private void registerCollector(final boolean isCollectJVMInformation, final boolean isEnhancedForProxy) {
66          new PrometheusMetricsExporter(new BuildInfoExporter()).register();
67          if (isEnhancedForProxy) {
68              registerCollectorForProxy();
69          } else {
70              registerCollectorForJDBC();
71          }
72          if (isCollectJVMInformation) {
73              DefaultExports.initialize();
74          }
75      }
76      
77      private void registerCollectorForProxy() {
78          new PrometheusMetricsExporter(new ProxyStateExporter()).register();
79          new PrometheusMetricsExporter(new ProxyMetaDataInfoExporter()).register();
80      }
81      
82      private void registerCollectorForJDBC() {
83          new PrometheusMetricsExporter(new JDBCStateExporter()).register();
84          new PrometheusMetricsExporter(new JDBCMetaDataInfoExporter()).register();
85      }
86      
87      private InetSocketAddress getSocketAddress(final PluginConfiguration pluginConfig) {
88          return Strings.isNullOrEmpty(pluginConfig.getHost()) ? new InetSocketAddress(pluginConfig.getPort()) : new InetSocketAddress(pluginConfig.getHost(), pluginConfig.getPort());
89      }
90      
91      @Override
92      public void close() {
93          if (null != httpServer) {
94              httpServer.stop();
95          }
96      }
97      
98      @Override
99      public String getType() {
100         return "Prometheus";
101     }
102 }
