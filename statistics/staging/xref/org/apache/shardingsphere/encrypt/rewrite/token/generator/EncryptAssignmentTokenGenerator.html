47  import java.util.LinkedList;
48  import java.util.Optional;

50  /**
51   * Assignment generator for encrypt.
52   */
53  @Setter
54  public final class EncryptAssignmentTokenGenerator implements CollectionSQLTokenGenerator<SQLStatementContext>, EncryptRuleAware, DatabaseNameAware {
55      
56      private EncryptRule encryptRule;
57      
58      private String databaseName;
59      
60      @Override
61      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
62          return sqlStatementContext instanceof UpdateStatementContext || sqlStatementContext instanceof InsertStatementContext
63                  && (((InsertStatementContext) sqlStatementContext).getSqlStatement()).getSetAssignment().isPresent();
64      }
65      
66      @Override
67      public Collection<SQLToken> generateSQLTokens(final SQLStatementContext sqlStatementContext) {
68          String tableName = ((TableAvailable) sqlStatementContext).getTablesContext().getSimpleTables().iterator().next().getTableName().getIdentifier().getValue();
69          Optional<EncryptTable> encryptTable = encryptRule.findEncryptTable(tableName);
70          if (!encryptTable.isPresent()) {
71              return Collections.emptyList();
72          }
73          Collection<SQLToken> result = new LinkedList<>();
74          String schemaName = ((TableAvailable) sqlStatementContext).getTablesContext().getSchemaName()
75                  .orElseGet(() -> new DatabaseTypeRegistry(sqlStatementContext.getDatabaseType()).getDefaultSchemaName(databaseName));
76          for (ColumnAssignmentSegment each : getSetAssignmentSegment(sqlStatementContext.getSqlStatement()).getAssignments()) {
77              String columnName = each.getColumns().get(0).getIdentifier().getValue();
78              if (encryptTable.get().isEncryptColumn(columnName)) {
79                  generateSQLToken(schemaName, encryptTable.get().getTable(), encryptTable.get().getEncryptColumn(columnName), each).ifPresent(result::add);
80              }
81          }
82          return result;
83      }
84      
85      private SetAssignmentSegment getSetAssignmentSegment(final SQLStatement sqlStatement) {
86          if (sqlStatement instanceof InsertStatement) {
87              Optional<SetAssignmentSegment> result = ((InsertStatement) sqlStatement).getSetAssignment();
88              Preconditions.checkState(result.isPresent());
89              return result.get();
90          }
91          return ((UpdateStatement) sqlStatement).getSetAssignment();
92      }
93      
94      private Optional<EncryptAssignmentToken> generateSQLToken(final String schemaName, final String tableName, final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment) {
95          if (segment.getValue() instanceof ParameterMarkerExpressionSegment) {
96              return Optional.of(generateParameterSQLToken(encryptColumn, segment));
97          }
98          if (segment.getValue() instanceof LiteralExpressionSegment) {
99              return Optional.of(generateLiteralSQLToken(schemaName, tableName, encryptColumn, segment));
100         }
101         return Optional.empty();
102     }
103     
104     private EncryptAssignmentToken generateParameterSQLToken(final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment) {
105         EncryptParameterAssignmentToken result = new EncryptParameterAssignmentToken(segment.getColumns().get(0).getStartIndex(), segment.getStopIndex());
106         result.addColumnName(encryptColumn.getCipher().getName());
107         encryptColumn.getAssistedQuery().ifPresent(optional -> result.addColumnName(optional.getName()));
108         encryptColumn.getLikeQuery().ifPresent(optional -> result.addColumnName(optional.getName()));
109         return result;
110     }
111     
112     private EncryptAssignmentToken generateLiteralSQLToken(final String schemaName, final String tableName, final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment) {
113         EncryptLiteralAssignmentToken result = new EncryptLiteralAssignmentToken(segment.getColumns().get(0).getStartIndex(), segment.getStopIndex());
114         addCipherAssignment(schemaName, tableName, encryptColumn, segment, result);
115         addAssistedQueryAssignment(schemaName, tableName, encryptColumn, segment, result);
116         addLikeAssignment(schemaName, tableName, encryptColumn, segment, result);
117         return result;
118     }
119     
120     private void addCipherAssignment(final String schemaName, final String tableName,
121                                      final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment, final EncryptLiteralAssignmentToken token) {
122         Object originalValue = ((LiteralExpressionSegment) segment.getValue()).getLiterals();
123         Object cipherValue = encryptColumn.getCipher().encrypt(databaseName, schemaName, tableName, encryptColumn.getName(), Collections.singletonList(originalValue)).iterator().next();
124         token.addAssignment(encryptColumn.getCipher().getName(), cipherValue);
125     }
126     
127     private void addAssistedQueryAssignment(final String schemaName, final String tableName, final EncryptColumn encryptColumn,
128                                             final ColumnAssignmentSegment segment, final EncryptLiteralAssignmentToken token) {
129         Object originalValue = ((LiteralExpressionSegment) segment.getValue()).getLiterals();
130         if (encryptColumn.getAssistedQuery().isPresent()) {
131             Object assistedQueryValue = encryptColumn.getAssistedQuery().get().encrypt(
132                     databaseName, schemaName, tableName, encryptColumn.getName(), Collections.singletonList(originalValue)).iterator().next();
133             token.addAssignment(encryptColumn.getAssistedQuery().get().getName(), assistedQueryValue);
134         }
135     }
136     
137     private void addLikeAssignment(final String schemaName, final String tableName,
138                                    final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment, final EncryptLiteralAssignmentToken token) {
139         Object originalValue = ((LiteralExpressionSegment) segment.getValue()).getLiterals();
140         if (encryptColumn.getLikeQuery().isPresent()) {
141             Object assistedQueryValue = encryptColumn.getLikeQuery().get().encrypt(databaseName, schemaName,
142                     tableName, segment.getColumns().get(0).getIdentifier().getValue(), Collections.singletonList(originalValue)).iterator().next();
143             token.addAssignment(encryptColumn.getLikeQuery().get().getName(), assistedQueryValue);
144         }
145     }
146 }
