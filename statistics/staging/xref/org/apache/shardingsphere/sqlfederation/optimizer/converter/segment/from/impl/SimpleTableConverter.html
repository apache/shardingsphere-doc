18  package org.apache.shardingsphere.sqlfederation.optimizer.converter.segment.from.impl;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.calcite.sql.SqlBasicCall;
23  import org.apache.calcite.sql.SqlIdentifier;
24  import org.apache.calcite.sql.SqlNode;
25  import org.apache.calcite.sql.fun.SqlStdOperatorTable;
26  import org.apache.calcite.sql.parser.SqlParserPos;
27  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.OwnerSegment;
28  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.table.SimpleTableSegment;
29  import org.apache.shardingsphere.sql.parser.statement.core.segment.generic.table.TableNameSegment;
30  
31  import java.util.ArrayList;
32  import java.util.Arrays;
33  import java.util.List;
34  import java.util.Optional;
35  
36  /**
37   * Simple table converter.
38   */
39  @NoArgsConstructor(access = AccessLevel.PRIVATE)
40  public final class SimpleTableConverter {
41      
42      /**
43       * Convert simple table segment to sql node.
44       * 
45       * @param segment simple table segment
46       * @return sql node
47       */
48      public static Optional<SqlNode> convert(final SimpleTableSegment segment) {
49          if ("DUAL".equalsIgnoreCase(segment.getTableName().getIdentifier().getValue())) {
50              return Optional.empty();
51          }
52          TableNameSegment tableName = segment.getTableName();
53          List<String> names = new ArrayList<>();
54          if (segment.getOwner().isPresent()) {
55              addOwnerNames(names, segment.getOwner().get());
56          }
57          names.add(tableName.getIdentifier().getValue());
58          if (segment.getDbLink().isPresent() && segment.getAt().isPresent()) {
59              names.add(segment.getAt().get().getValue());
60              names.add(segment.getDbLink().get().getValue());
61          }
62          SqlNode tableNameSQLNode = new SqlIdentifier(names, SqlParserPos.ZERO);
63          if (segment.getAliasName().isPresent()) {
64              SqlNode aliasSQLNode = new SqlIdentifier(segment.getAliasName().get(), SqlParserPos.ZERO);
65              return Optional.of(new SqlBasicCall(SqlStdOperatorTable.AS, Arrays.asList(tableNameSQLNode, aliasSQLNode), SqlParserPos.ZERO));
66          }
67          return Optional.of(tableNameSQLNode);
68      }
69      
70      private static void addOwnerNames(final List<String> names, final OwnerSegment owner) {
71          if (null != owner) {
72              addOwnerNames(names, owner.getOwner().orElse(null));
73              names.add(owner.getIdentifier().getValue());
74          }
75      }
76  }
