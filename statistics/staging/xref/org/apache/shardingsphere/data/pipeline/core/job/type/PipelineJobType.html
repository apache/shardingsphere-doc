22  import org.apache.shardingsphere.data.pipeline.core.consistencycheck.PipelineDataConsistencyChecker;
23  import org.apache.shardingsphere.data.pipeline.core.context.TransmissionProcessContext;
24  import org.apache.shardingsphere.data.pipeline.core.job.PipelineJob;
25  import org.apache.shardingsphere.data.pipeline.core.job.config.PipelineJobConfiguration;
26  import org.apache.shardingsphere.data.pipeline.core.job.config.yaml.swapper.YamlPipelineJobConfigurationSwapper;
27  import org.apache.shardingsphere.data.pipeline.core.job.progress.PipelineJobItemProgress;
28  import org.apache.shardingsphere.data.pipeline.core.job.progress.yaml.config.YamlPipelineJobItemProgressConfiguration;
29  import org.apache.shardingsphere.data.pipeline.core.job.progress.yaml.swapper.YamlPipelineJobItemProgressSwapper;
30  import org.apache.shardingsphere.data.pipeline.core.pojo.PipelineJobInfo;
31  import org.apache.shardingsphere.infra.spi.annotation.SingletonSPI;
32  import org.apache.shardingsphere.infra.spi.type.typed.TypedSPI;
33  import org.apache.shardingsphere.infra.util.yaml.YamlConfiguration;
34  
35  import java.util.Optional;
36  
37  /**
38   * Pipeline job type.
39   */
40  @SingletonSPI
41  @JsonIgnoreType
42  public interface PipelineJobType extends TypedSPI {
43      
44      /**
45       * Get job type code.
46       *
47       * @return job type code
48       */
49      String getCode();
50      
51      /**
52       * Is transmission job.
53       *
54       * @return is transmission job or not
55       */
56      boolean isTransmissionJob();
57      
58      /**
59       * Get YAML pipeline job configuration swapper.
60       *
61       * @param <T> type of YAML configuration
62       * @param <Y> type of pipeline job configuration
63       * @return YAML pipeline job configuration swapper
64       */
65      <Y extends YamlConfiguration, T extends PipelineJobConfiguration> YamlPipelineJobConfigurationSwapper<Y, T> getYamlJobConfigurationSwapper();
66      
67      /**
68       * Get YAML pipeline job item progress swapper.
69       *
70       * @param <T> type of pipeline job item progress
71       * @return YAML pipeline job item progress swapper
72       */
73      <T extends PipelineJobItemProgress> YamlPipelineJobItemProgressSwapper<YamlPipelineJobItemProgressConfiguration, T> getYamlJobItemProgressSwapper();
74      
75      /**
76       * Get pipeline job class.
77       *
78       * @return pipeline job class
79       */
80      Class<? extends PipelineJob> getJobClass();
81      
82      /**
83       * Whether to ignore to start disabled job when job item progress is finished.
84       *
85       * @return ignore to start disabled job when job item progress is finished or not
86       */
87      default boolean isIgnoreToStartDisabledJobWhenJobItemProgressIsFinished() {
88          return false;
89      }
90      
91      /**
92       * Get to be start disabled next job type.
93       *
94       * @return to be start disabled next job type
95       */
96      default Optional<String> getToBeStartDisabledNextJobType() {
97          return Optional.empty();
98      }
99      
100     /**
101      * Get to be stopped previous job type.
102      *
103      * @return to be stopped previous job type
104      */
105     default Optional<String> getToBeStoppedPreviousJobType() {
106         return Optional.empty();
107     }
108     
109     /**
110      * Whether to force no sharding when convert to job configuration POJO.
111      *
112      * @return without sharding or not
113      */
114     default boolean isForceNoShardingWhenConvertToJobConfigurationPOJO() {
115         return false;
116     }
117     
118     /**
119      * Get pipeline job info.
120      *
121      * @param jobId job ID
122      * @return pipeline job info
123      */
124     PipelineJobInfo getJobInfo(String jobId);
125     
126     /**
127      * Build pipeline data consistency checker.
128      *
129      * @param jobConfig job configuration
130      * @param processContext process context
131      * @param progressContext consistency check job item progress context
132      * @return all logic tables check result
133      * @throws UnsupportedOperationException unsupported operation exception
134      */
135     default PipelineDataConsistencyChecker buildDataConsistencyChecker(PipelineJobConfiguration jobConfig,
136                                                                        TransmissionProcessContext processContext, ConsistencyCheckJobItemProgressContext progressContext) {
137         throw new UnsupportedOperationException("Build data consistency checker is not supported.");
138     }
139     
140     @Override
141     String getType();
142 }
