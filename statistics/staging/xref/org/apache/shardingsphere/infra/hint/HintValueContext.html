18  package org.apache.shardingsphere.infra.hint;
19  
20  import com.google.common.base.Joiner;
21  import com.google.common.collect.ArrayListMultimap;
22  import com.google.common.collect.Multimap;
23  import lombok.Getter;
24  import lombok.Setter;
25  
26  import java.util.Collection;
27  import java.util.LinkedHashSet;
28  import java.util.Optional;
29  
30  /**
31   * Hint value context.
32   */
33  @Getter
34  @Setter
35  public final class HintValueContext {
36      
37      private final Multimap<String, Comparable<?>> shardingDatabaseValues = ArrayListMultimap.create();
38      
39      private final Multimap<String, Comparable<?>> shardingTableValues = ArrayListMultimap.create();
40      
41      private final Collection<String> disableAuditNames = new LinkedHashSet<>();
42      
43      private String dataSourceName = "";
44      
45      private boolean databaseShardingOnly;
46      
47      private boolean writeRouteOnly;
48      
49      private boolean skipSQLRewrite;
50      
51      private boolean shadow;
52      
53      /**
54       * Find hint data source name.
55       *
56       * @return data source name
57       */
58      public Optional<String> findHintDataSourceName() {
59          return dataSourceName.isEmpty() ? Optional.empty() : Optional.of(dataSourceName);
60      }
61      
62      /**
63       * Judge contains hint sharding databases value or not.
64       *
65       * @param tableName table name
66       * @return contains hint sharding databases value or not
67       */
68      public boolean containsHintShardingDatabaseValue(final String tableName) {
69          String key = Joiner.on(".").join(tableName.toUpperCase(), SQLHintPropertiesKey.SHARDING_DATABASE_VALUE_KEY.getKey());
70          return shardingDatabaseValues.containsKey(key) || shardingDatabaseValues.containsKey(SQLHintPropertiesKey.SHARDING_DATABASE_VALUE_KEY.getKey());
71      }
72      
73      /**
74       * Judge contains hint sharding table value or not.
75       *
76       * @param tableName table name
77       * @return Contains hint sharding table value or not
78       */
79      public boolean containsHintShardingTableValue(final String tableName) {
80          String key = Joiner.on(".").join(tableName.toUpperCase(), SQLHintPropertiesKey.SHARDING_TABLE_VALUE_KEY.getKey());
81          return shardingTableValues.containsKey(key) || shardingTableValues.containsKey(SQLHintPropertiesKey.SHARDING_TABLE_VALUE_KEY.getKey());
82      }
83      
84      /**
85       * Judge contains hint sharding value or not.
86       *
87       * @param tableName table name
88       * @return Contains hint sharding value or not
89       */
90      public boolean containsHintShardingValue(final String tableName) {
91          return containsHintShardingDatabaseValue(tableName) || containsHintShardingTableValue(tableName);
92      }
93      
94      /**
95       * Get hint sharding table value.
96       *
97       * @param tableName table name
98       * @return sharding table value
99       */
100     public Collection<Comparable<?>> getHintShardingTableValue(final String tableName) {
101         String key = String.join(".", tableName.toUpperCase(), SQLHintPropertiesKey.SHARDING_TABLE_VALUE_KEY.getKey());
102         return shardingTableValues.containsKey(key)
103                 ? shardingTableValues.get(key)
104                 : shardingTableValues.get(SQLHintPropertiesKey.SHARDING_TABLE_VALUE_KEY.getKey());
105     }
106     
107     /**
108      * Get hint sharding database value.
109      *
110      * @param tableName table name
111      * @return sharding database value
112      */
113     public Collection<Comparable<?>> getHintShardingDatabaseValue(final String tableName) {
114         String key = String.join(".", tableName.toUpperCase(), SQLHintPropertiesKey.SHARDING_DATABASE_VALUE_KEY.getKey());
115         return shardingDatabaseValues.containsKey(key)
116                 ? shardingDatabaseValues.get(key)
117                 : shardingDatabaseValues.get(SQLHintPropertiesKey.SHARDING_DATABASE_VALUE_KEY.getKey());
118     }
119 }
