77  /**
78   * Proxy SQL Executor.
79   */
80  public final class ProxySQLExecutor {
81      
82      private final String type;
83      
84      private final ProxyDatabaseConnectionManager databaseConnectionManager;
85      
86      private final ProxyJDBCExecutor regularExecutor;
87      
88      private final RawExecutor rawExecutor;
89      
90      @Getter
91      private final SQLFederationEngine sqlFederationEngine;
92      
93      private final Collection<TransactionHook> transactionHooks = ShardingSphereServiceLoader.getServiceInstances(TransactionHook.class);
94      
95      public ProxySQLExecutor(final String type, final ProxyDatabaseConnectionManager databaseConnectionManager, final DatabaseConnector databaseConnector, final QueryContext queryContext) {
96          this.type = type;
97          this.databaseConnectionManager = databaseConnectionManager;
98          ExecutorEngine executorEngine = BackendExecutorContext.getInstance().getExecutorEngine();
99          ConnectionContext connectionContext = databaseConnectionManager.getConnectionSession().getConnectionContext();
100         JDBCExecutor jdbcExecutor = new JDBCExecutor(executorEngine, connectionContext);
101         regularExecutor = new ProxyJDBCExecutor(type, databaseConnectionManager.getConnectionSession(), databaseConnector, jdbcExecutor);
102         rawExecutor = new RawExecutor(executorEngine, connectionContext);
103         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
104         String databaseName = databaseConnectionManager.getConnectionSession().getDatabaseName();
105         String schemaName = queryContext.getSqlStatementContext().getTablesContext().getSchemaName()
106                 .orElseGet(() -> DatabaseTypeEngine.getDefaultSchemaName(queryContext.getSqlStatementContext().getDatabaseType(), databaseName));
107         sqlFederationEngine = new SQLFederationEngine(databaseName, schemaName, metaDataContexts.getMetaData(), metaDataContexts.getStatistics(), jdbcExecutor);
108     }
109     
110     /**
111      * Check execute prerequisites.
112      *
113      * @param executionContext execution context
114      */
115     public void checkExecutePrerequisites(final ExecutionContext executionContext) {
116         ShardingSpherePreconditions.checkState(isValidExecutePrerequisites(executionContext), () -> new TableModifyInTransactionException(getTableName(executionContext)));
117     }
118     
119     private boolean isValidExecutePrerequisites(final ExecutionContext executionContext) {
120         return !isExecuteDDLInXATransaction(executionContext.getSqlStatementContext().getSqlStatement())
121                 && !isExecuteDDLInPostgreSQLOpenGaussTransaction(executionContext.getSqlStatementContext().getSqlStatement());
122     }
123     
124     private boolean isExecuteDDLInXATransaction(final SQLStatement sqlStatement) {
125         TransactionStatus transactionStatus = databaseConnectionManager.getConnectionSession().getTransactionStatus();
126         return TransactionType.XA == transactionStatus.getTransactionType() && transactionStatus.isInTransaction() && isUnsupportedDDLStatement(sqlStatement);
127     }
128     
129     private boolean isExecuteDDLInPostgreSQLOpenGaussTransaction(final SQLStatement sqlStatement) {
130         // TODO implement DDL statement commit/rollback in PostgreSQL/openGauss transaction
131         boolean isPostgreSQLOpenGaussStatement = isPostgreSQLOrOpenGaussStatement(sqlStatement);
132         boolean isSupportedStatement = isSupportedSQLStatement(sqlStatement);
133         return sqlStatement instanceof DDLStatement
134                 && !isSupportedStatement && isPostgreSQLOpenGaussStatement && databaseConnectionManager.getConnectionSession().getTransactionStatus().isInTransaction();
135     }
136     
137     private boolean isSupportedSQLStatement(final SQLStatement sqlStatement) {
138         return isCursorStatement(sqlStatement) || sqlStatement instanceof TruncateStatement;
139     }
140     
141     private boolean isCursorStatement(final SQLStatement sqlStatement) {
142         return sqlStatement instanceof OpenGaussCursorStatement
143                 || sqlStatement instanceof CloseStatement || sqlStatement instanceof MoveStatement || sqlStatement instanceof FetchStatement;
144     }
145     
146     private boolean isUnsupportedDDLStatement(final SQLStatement sqlStatement) {
147         if (isPostgreSQLOrOpenGaussStatement(sqlStatement) && isSupportedSQLStatement(sqlStatement)) {
148             return false;
149         }
150         return sqlStatement instanceof DDLStatement;
151     }
152     
153     private boolean isPostgreSQLOrOpenGaussStatement(final SQLStatement sqlStatement) {
154         return sqlStatement instanceof PostgreSQLStatement || sqlStatement instanceof OpenGaussStatement;
155     }
156     
157     private String getTableName(final ExecutionContext executionContext) {
158         return executionContext.getSqlStatementContext() instanceof TableAvailable && !((TableAvailable) executionContext.getSqlStatementContext()).getAllTables().isEmpty()
159                 ? ((TableAvailable) executionContext.getSqlStatementContext()).getAllTables().iterator().next().getTableName().getIdentifier().getValue()
160                 : "unknown_table";
161     }
162     
163     /**
164      * Execute SQL.
165      *
166      * @param executionContext execution context
167      * @return execute results
168      * @throws SQLException SQL exception
169      */
170     public List<ExecuteResult> execute(final ExecutionContext executionContext) throws SQLException {
171         String databaseName = databaseConnectionManager.getConnectionSession().getDatabaseName();
172         Collection<ShardingSphereRule> rules = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getDatabase(databaseName).getRuleMetaData().getRules();
173         int maxConnectionsSizePerQuery = ProxyContext.getInstance()
174                 .getContextManager().getMetaDataContexts().getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
175         boolean isReturnGeneratedKeys = executionContext.getSqlStatementContext().getSqlStatement() instanceof MySQLInsertStatement;
176         return hasRawExecutionRule(rules) ? rawExecute(executionContext, rules, maxConnectionsSizePerQuery)
177                 : useDriverToExecute(executionContext, rules, maxConnectionsSizePerQuery, isReturnGeneratedKeys, SQLExecutorExceptionHandler.isExceptionThrown());
178     }
179     
180     private boolean hasRawExecutionRule(final Collection<ShardingSphereRule> rules) {
181         for (ShardingSphereRule each : rules) {
182             if (each instanceof RawExecutionRule) {
183                 return true;
184             }
185         }
186         return false;
187     }
188     
189     private List<ExecuteResult> rawExecute(final ExecutionContext executionContext, final Collection<ShardingSphereRule> rules, final int maxConnectionsSizePerQuery) throws SQLException {
190         RawExecutionPrepareEngine prepareEngine = new RawExecutionPrepareEngine(maxConnectionsSizePerQuery, rules);
191         ExecutionGroupContext<RawSQLExecutionUnit> executionGroupContext;
192         try {
193             executionGroupContext = prepareEngine.prepare(executionContext.getRouteContext(), executionContext.getExecutionUnits(),
194                     new ExecutionGroupReportContext(databaseConnectionManager.getConnectionSession().getProcessId(),
195                             databaseConnectionManager.getConnectionSession().getDatabaseName(), databaseConnectionManager.getConnectionSession().getGrantee()));
196         } catch (final SQLException ex) {
197             return getSaneExecuteResults(executionContext, ex);
198         }
199         // TODO handle query header
200         return rawExecutor.execute(executionGroupContext, executionContext.getQueryContext(), new RawSQLExecutorCallback());
201     }
202     
203     private List<ExecuteResult> useDriverToExecute(final ExecutionContext executionContext, final Collection<ShardingSphereRule> rules,
204                                                    final int maxConnectionsSizePerQuery, final boolean isReturnGeneratedKeys, final boolean isExceptionThrown) throws SQLException {
205         JDBCBackendStatement statementManager = (JDBCBackendStatement) databaseConnectionManager.getConnectionSession().getStatementManager();
206         DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine = new DriverExecutionPrepareEngine<>(
207                 type, maxConnectionsSizePerQuery, databaseConnectionManager, statementManager, new StatementOption(isReturnGeneratedKeys), rules,
208                 ProxyContext.getInstance().getDatabase(databaseConnectionManager.getConnectionSession().getDatabaseName()).getResourceMetaData().getStorageTypes());
209         ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext;
210         try {
211             executionGroupContext = prepareEngine.prepare(executionContext.getRouteContext(), executionContext.getExecutionUnits(),
212                     new ExecutionGroupReportContext(databaseConnectionManager.getConnectionSession().getProcessId(),
213                             databaseConnectionManager.getConnectionSession().getDatabaseName(), databaseConnectionManager.getConnectionSession().getGrantee()));
214         } catch (final SQLException ex) {
215             return getSaneExecuteResults(executionContext, ex);
216         }
217         executeTransactionHooksBeforeExecuteSQL(databaseConnectionManager.getConnectionSession());
218         return regularExecutor.execute(executionContext.getQueryContext(), executionGroupContext, isReturnGeneratedKeys, isExceptionThrown);
219     }
220     
221     private void executeTransactionHooksBeforeExecuteSQL(final ConnectionSession connectionSession) throws SQLException {
222         if (!getTransactionContext(connectionSession).isInTransaction()) {
223             return;
224         }
225         for (TransactionHook each : transactionHooks) {
226             each.beforeExecuteSQL(connectionSession.getDatabaseConnectionManager().getCachedConnections().values(), getTransactionContext(connectionSession), connectionSession.getIsolationLevel());
227         }
228     }
229     
230     private TransactionConnectionContext getTransactionContext(final ConnectionSession connectionSession) {
231         return connectionSession.getDatabaseConnectionManager().getConnectionSession().getConnectionContext().getTransactionContext();
232     }
233     
234     private List<ExecuteResult> getSaneExecuteResults(final ExecutionContext executionContext, final SQLException originalException) throws SQLException {
235         DatabaseType databaseType = ProxyContext.getInstance().getDatabase(databaseConnectionManager.getConnectionSession().getDatabaseName()).getProtocolType();
236         Optional<ExecuteResult> executeResult = DatabaseTypedSPILoader.getService(SaneQueryResultEngine.class, databaseType)
237                 .getSaneQueryResult(executionContext.getSqlStatementContext().getSqlStatement(), originalException);
238         if (executeResult.isPresent()) {
239             return Collections.singletonList(executeResult.get());
240         }
241         throw originalException;
242     }
