18  package org.apache.shardingsphere.data.pipeline.postgresql.ingest;
19  
20  import lombok.extern.slf4j.Slf4j;
21  import org.apache.shardingsphere.data.pipeline.core.exception.PipelineInternalException;
22  import org.apache.shardingsphere.data.pipeline.core.ingest.position.DialectIngestPositionManager;
23  import org.apache.shardingsphere.data.pipeline.postgresql.ingest.wal.WALPosition;
24  import org.apache.shardingsphere.data.pipeline.postgresql.ingest.wal.decode.PostgreSQLLogSequenceNumber;
25  import org.postgresql.replication.LogSequenceNumber;
26  
27  import javax.sql.DataSource;
28  import java.sql.Connection;
29  import java.sql.PreparedStatement;
30  import java.sql.ResultSet;
31  import java.sql.SQLException;
32  
33  /**
34   * Ingest position manager for PostgreSQL.
35   */
36  @Slf4j
37  public final class PostgreSQLIngestPositionManager implements DialectIngestPositionManager {
38      
39      private static final String DECODE_PLUGIN = "test_decoding";
40      
41      private static final String DUPLICATE_OBJECT_ERROR_CODE = "42710";
42      
43      @Override
44      public WALPosition init(final String data) {
45          return new WALPosition(new PostgreSQLLogSequenceNumber(LogSequenceNumber.valueOf(data)));
46      }
47      
48      @Override
49      public WALPosition init(final DataSource dataSource, final String slotNameSuffix) throws SQLException {
50          try (Connection connection = dataSource.getConnection()) {
51              createSlotIfNotExist(connection, PostgreSQLSlotNameGenerator.getUniqueSlotName(connection, slotNameSuffix));
52              return getWALPosition(connection);
53          }
54      }
55      
56      private void createSlotIfNotExist(final Connection connection, final String slotName) throws SQLException {
57          if (isSlotExisting(connection, slotName)) {
58              log.info("createSlotIfNotExist, slot exist, slotName={}", slotName);
59              return;
60          }
61          try (PreparedStatement preparedStatement = connection.prepareStatement("SELECT * FROM pg_create_logical_replication_slot(?, ?)")) {
62              preparedStatement.setString(1, slotName);
63              preparedStatement.setString(2, DECODE_PLUGIN);
64              preparedStatement.execute();
65          } catch (final SQLException ex) {
66              if (!DUPLICATE_OBJECT_ERROR_CODE.equals(ex.getSQLState())) {
67                  throw ex;
68              }
69          }
70      }
71      
72      private boolean isSlotExisting(final Connection connection, final String slotName) throws SQLException {
73          String checkSlotSQL = "SELECT slot_name FROM pg_replication_slots WHERE slot_name=? AND plugin=?";
74          try (PreparedStatement preparedStatement = connection.prepareStatement(checkSlotSQL)) {
75              preparedStatement.setString(1, slotName);
76              preparedStatement.setString(2, DECODE_PLUGIN);
77              try (ResultSet resultSet = preparedStatement.executeQuery()) {
78                  return resultSet.next();
79              }
80          }
81      }
82      
83      private WALPosition getWALPosition(final Connection connection) throws SQLException {
84          try (
85                  PreparedStatement preparedStatement = connection.prepareStatement(getLogSequenceNumberSQL(connection));
86                  ResultSet resultSet = preparedStatement.executeQuery()) {
87              resultSet.next();
88              return new WALPosition(new PostgreSQLLogSequenceNumber(LogSequenceNumber.valueOf(resultSet.getString(1))));
89          }
90      }
91      
92      private String getLogSequenceNumberSQL(final Connection connection) throws SQLException {
93          if (9 == connection.getMetaData().getDatabaseMajorVersion() && 6 <= connection.getMetaData().getDatabaseMinorVersion()) {
94              return "SELECT PG_CURRENT_XLOG_LOCATION()";
95          }
96          if (10 <= connection.getMetaData().getDatabaseMajorVersion()) {
97              return "SELECT PG_CURRENT_WAL_LSN()";
98          }
99          throw new PipelineInternalException("Unsupported PostgreSQL version: " + connection.getMetaData().getDatabaseProductVersion());
100     }
101     
102     @Override
103     public void destroy(final DataSource dataSource, final String slotNameSuffix) throws SQLException {
104         try (Connection connection = dataSource.getConnection()) {
105             dropSlotIfExist(connection, slotNameSuffix);
106         }
107     }
108     
109     private void dropSlotIfExist(final Connection connection, final String slotNameSuffix) throws SQLException {
110         String slotName = PostgreSQLSlotNameGenerator.getUniqueSlotName(connection, slotNameSuffix);
111         if (!isSlotExisting(connection, slotName)) {
112             log.info("dropSlotIfExist, slot not exist, slotName={}", slotName);
113             return;
114         }
115         log.info("dropSlotIfExist, slot exist, slotName={}", slotName);
116         String dropSlotSQL = "SELECT pg_drop_replication_slot(?)";
117         try (PreparedStatement preparedStatement = connection.prepareStatement(dropSlotSQL)) {
118             preparedStatement.setString(1, slotName);
119             preparedStatement.execute();
120         }
121     }
122     
123     @Override
124     public String getDatabaseType() {
125         return "PostgreSQL";
126     }
127 }
