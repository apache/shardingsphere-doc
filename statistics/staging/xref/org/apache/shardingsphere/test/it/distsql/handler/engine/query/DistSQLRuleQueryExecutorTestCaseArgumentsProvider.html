47  /**
48   * DistSQL rule query executor test case arguments provider.
49   */
50  public final class DistSQLRuleQueryExecutorTestCaseArgumentsProvider implements ArgumentsProvider {
51      
52      @SneakyThrows(JAXBException.class)
53      @Override
54      public Stream<Arguments> provideArguments(final ExtensionContext extensionContext) {
55          DistSQLRuleQueryExecutorSettings settings = extensionContext.getRequiredTestClass().getAnnotation(DistSQLRuleQueryExecutorSettings.class);
56          Preconditions.checkNotNull(settings, "Annotation DistSQLRuleQueryExecutorSettings is required.");
57          DistSQLRuleQueryExecutorTestCases testCases = (DistSQLRuleQueryExecutorTestCases) JAXBContext.newInstance(DistSQLRuleQueryExecutorTestCases.class)
58                  .createUnmarshaller().unmarshal(Thread.currentThread().getContextClassLoader().getResource(settings.value()));
59          return testCases.getTestCases().stream().map(each -> Arguments.arguments(
60                  each.getDistSQL(), getDistSQLStatement(each), getCurrentRuleConfiguration(each), getExpectedQueryResultRows(each))).collect(Collectors.toList()).stream();
61      }
62      
63      private DistSQLStatement getDistSQLStatement(final DistSQLRuleQueryExecutorTestCase testCase) {
64          return (DistSQLStatement) new DistSQLParserEngine().parse(testCase.getDistSQL());
65      }
66      
67      @SneakyThrows(IOException.class)
68      private RuleConfiguration getCurrentRuleConfiguration(final DistSQLRuleQueryExecutorTestCase testCase) {
69          URL url = Thread.currentThread().getContextClassLoader().getResource(testCase.getCurrentRuleConfigurationYAMLFile());
70          assertNotNull(url);
71          YamlRootConfiguration yamlRootConfig = YamlEngine.unmarshal(new File(url.getFile()), YamlRootConfiguration.class);
72          return new YamlRuleConfigurationSwapperEngine().swapToRuleConfiguration(yamlRootConfig.getRules().iterator().next());
73      }
74      
75      private Collection<LocalDataQueryResultRow> getExpectedQueryResultRows(final DistSQLRuleQueryExecutorTestCase testCase) {
76          return testCase.getExpectedQueryResultRows().stream().map(each -> new LocalDataQueryResultRow(Splitter.on('|').trimResults().splitToStream(each).toArray())).collect(Collectors.toList());
77      }
78  }
