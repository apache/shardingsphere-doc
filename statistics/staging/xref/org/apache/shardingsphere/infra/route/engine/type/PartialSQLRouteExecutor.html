45  /**
46   * Partial SQL route executor.
47   */
48  @HighFrequencyInvocation
49  public final class PartialSQLRouteExecutor implements SQLRouteExecutor {
50      
51      private final ConfigurationProperties props;
52      
53      @SuppressWarnings("rawtypes")
54      private final Map<ShardingSphereRule, SQLRouter> routers;
55      
56      public PartialSQLRouteExecutor(final Collection<ShardingSphereRule> rules, final ConfigurationProperties props) {
57          this.props = props;
58          routers = OrderedSPILoader.getServices(SQLRouter.class, rules);
59      }
60      
61      @Override
62      @SuppressWarnings({"unchecked", "rawtypes"})
63      public RouteContext route(final QueryContext queryContext, final RuleMetaData globalRuleMetaData, final ShardingSphereDatabase database) {
64          RouteContext result = new RouteContext();
65          Optional<String> dataSourceName = findDataSourceByHint(queryContext.getHintValueContext(), database.getResourceMetaData().getStorageUnits());
66          if (dataSourceName.isPresent()) {
67              result.getRouteUnits().add(new RouteUnit(new RouteMapper(dataSourceName.get(), dataSourceName.get()), Collections.emptyList()));
68              return result;
69          }
70          for (Entry<ShardingSphereRule, SQLRouter> entry : routers.entrySet()) {
71              if (result.getRouteUnits().isEmpty() && entry.getValue() instanceof EntranceSQLRouter) {
72                  result = ((EntranceSQLRouter) entry.getValue()).createRouteContext(queryContext, globalRuleMetaData, database, entry.getKey(), props);
73              } else if (entry.getValue() instanceof DecorateSQLRouter) {
74                  ((DecorateSQLRouter) entry.getValue()).decorateRouteContext(result, queryContext, database, entry.getKey(), props);
75              }
76          }
77          if (result.getRouteUnits().isEmpty() && 1 == database.getResourceMetaData().getStorageUnits().size()) {
78              String singleDataSourceName = database.getResourceMetaData().getStorageUnits().keySet().iterator().next();
79              result.getRouteUnits().add(new RouteUnit(new RouteMapper(singleDataSourceName, singleDataSourceName), Collections.emptyList()));
80          }
81          return result;
82      }
83      
84      private Optional<String> findDataSourceByHint(final HintValueContext hintValueContext, final Map<String, StorageUnit> storageUnits) {
85          Optional<String> result = HintManager.isInstantiated() && HintManager.getDataSourceName().isPresent() ? HintManager.getDataSourceName() : hintValueContext.findHintDataSourceName();
86          if (result.isPresent() && !storageUnits.containsKey(result.get())) {
87              throw new DataSourceHintNotExistsException(result.get());
88          }
89          return result;
90      }
91  }
