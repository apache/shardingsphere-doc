18  package org.apache.shardingsphere.infra.datasource.props;
19  
20  import com.zaxxer.hikari.HikariDataSource;
21  import org.apache.shardingsphere.infra.datasource.config.ConnectionConfiguration;
22  import org.apache.shardingsphere.infra.datasource.config.DataSourceConfiguration;
23  import org.apache.shardingsphere.infra.datasource.config.PoolConfiguration;
24  import org.apache.shardingsphere.test.fixture.jdbc.MockedDataSource;
25  import org.junit.jupiter.api.Test;
26  
27  import javax.sql.DataSource;
28  import java.util.HashMap;
29  import java.util.Map;
30  
31  import static org.hamcrest.CoreMatchers.is;
32  import static org.junit.jupiter.api.Assertions.assertNull;
33  import static org.hamcrest.MatcherAssert.assertThat;
34  
35  class DataSourcePropertiesCreatorTest {
36      
37      @Test
38      void assertCreateWithDataSourceConfiguration() {
39          assertParameter(DataSourcePropertiesCreator.create(HikariDataSource.class.getName(), createResourceConfiguration()));
40      }
41      
42      private DataSourceConfiguration createResourceConfiguration() {
43          ConnectionConfiguration connectionConfig = new ConnectionConfiguration("jdbc:mysql://localhost:3306/demo_ds", "root", "root");
44          PoolConfiguration poolConfig = new PoolConfiguration(null, null, null, null, null, null, null);
45          return new DataSourceConfiguration(connectionConfig, poolConfig);
46      }
47      
48      private void assertParameter(final DataSourceProperties actual) {
49          Map<String, Object> props = actual.getAllLocalProperties();
50          assertThat(props.size(), is(9));
51          assertThat(props.get("jdbcUrl"), is("jdbc:mysql://localhost:3306/demo_ds"));
52          assertThat(props.get("username"), is("root"));
53          assertThat(props.get("password"), is("root"));
54          assertNull(props.get("maximumPoolSize"));
55          assertNull(props.get("minimumIdle"));
56          assertNull(props.get("connectionTimeout"));
57          assertNull(props.get("idleTimeout"));
58          assertNull(props.get("maxLifetime"));
59      }
60      
61      @Test
62      void assertCreateWithDataSource() {
63          assertThat(DataSourcePropertiesCreator.create(createDataSource()), is(new DataSourceProperties(MockedDataSource.class.getName(), createProperties())));
64      }
65      
66      private DataSource createDataSource() {
67          MockedDataSource result = new MockedDataSource();
68          result.setDriverClassName(MockedDataSource.class.getName());
69          result.setUrl("jdbc:mock://127.0.0.1/foo_ds");
70          result.setUsername("root");
71          result.setPassword("root");
72          return result;
73      }
74      
75      private Map<String, Object> createProperties() {
76          Map<String, Object> result = new HashMap<>();
77          result.put("driverClassName", MockedDataSource.class.getName());
78          result.put("url", "jdbc:mock://127.0.0.1/foo_ds");
79          result.put("username", "root");
80          result.put("password", "root");
81          result.put("maximumPoolSize", "-1");
82          return result;
83      }
84  }
