73  @ExtendWith(AutoMockExtension.class)
74  @StaticMockSettings(ProxyContext.class)
75  @MockitoSettings(strictness = Strictness.LENIENT)
76  class MySQLComQueryPacketExecutorTest {
77      
78      @Mock
79      private ProxyBackendHandler proxyBackendHandler;
80      
81      @Mock
82      private MySQLComQueryPacket packet;
83      
84      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
85      private ConnectionSession connectionSession;
86      
87      @BeforeEach
88      void setUp() {
89          when(packet.getSql()).thenReturn("");
90          when(connectionSession.getAttributeMap().attr(MySQLConstants.MYSQL_CHARACTER_SET_ATTRIBUTE_KEY).get()).thenReturn(MySQLCharacterSet.UTF8MB4_GENERAL_CI);
91      }
92      
93      @Test
94      void assertIsQueryResponse() throws SQLException, NoSuchFieldException, IllegalAccessException {
95          MySQLComQueryPacketExecutor mysqlComQueryPacketExecutor = new MySQLComQueryPacketExecutor(packet, connectionSession);
96          MemberAccessor accessor = Plugins.getMemberAccessor();
97          accessor.set(MySQLComQueryPacketExecutor.class.getDeclaredField("proxyBackendHandler"), mysqlComQueryPacketExecutor, proxyBackendHandler);
98          QueryHeader queryHeader = mock(QueryHeader.class);
99          when(queryHeader.getColumnTypeName()).thenReturn("VARCHAR");
100         when(proxyBackendHandler.execute()).thenReturn(new QueryResponseHeader(Collections.singletonList(queryHeader)));
101         mysqlComQueryPacketExecutor.execute();
102         assertThat(mysqlComQueryPacketExecutor.getResponseType(), is(ResponseType.QUERY));
103     }
104     
105     @Test
106     void assertIsUpdateResponse() throws SQLException, NoSuchFieldException, IllegalAccessException {
107         MySQLComQueryPacketExecutor mysqlComQueryPacketExecutor = new MySQLComQueryPacketExecutor(packet, connectionSession);
108         MemberAccessor accessor = Plugins.getMemberAccessor();
109         accessor.set(MySQLComQueryPacketExecutor.class.getDeclaredField("proxyBackendHandler"), mysqlComQueryPacketExecutor, proxyBackendHandler);
110         when(proxyBackendHandler.execute()).thenReturn(new UpdateResponseHeader(mock(SQLStatement.class)));
111         mysqlComQueryPacketExecutor.execute();
112         assertThat(mysqlComQueryPacketExecutor.getResponseType(), is(ResponseType.UPDATE));
113     }
114     
115     @Test
116     void assertExecuteMultiUpdateStatements() throws SQLException, NoSuchFieldException, IllegalAccessException {
117         when(connectionSession.getAttributeMap().hasAttr(MySQLConstants.MYSQL_OPTION_MULTI_STATEMENTS)).thenReturn(true);
118         when(connectionSession.getAttributeMap().attr(MySQLConstants.MYSQL_OPTION_MULTI_STATEMENTS).get()).thenReturn(0);
119         when(connectionSession.getDatabaseName()).thenReturn("foo_db");
120         when(packet.getSql()).thenReturn("update t set v=v+1 where id=1;update t set v=v+1 where id=2;update t set v=v+1 where id=3");
121         ContextManager contextManager = mock(ContextManager.class);
122         MetaDataContexts metaDataContexts = mockMetaDataContexts();
123         when(contextManager.getMetaDataContexts()).thenReturn(metaDataContexts);
124         when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
125         MySQLComQueryPacketExecutor actual = new MySQLComQueryPacketExecutor(packet, connectionSession);
126         MemberAccessor accessor = Plugins.getMemberAccessor();
127         accessor.set(MySQLComQueryPacketExecutor.class.getDeclaredField("proxyBackendHandler"), actual, proxyBackendHandler);
128         when(proxyBackendHandler.execute()).thenReturn(new UpdateResponseHeader(mock(SQLStatement.class)));
129         Collection<DatabasePacket<?>> actualPackets = actual.execute();
130         assertThat(actualPackets.size(), is(1));
131         assertThat(actualPackets.iterator().next(), instanceOf(MySQLOKPacket.class));
132     }
133     
134     private MetaDataContexts mockMetaDataContexts() {
135         MetaDataContexts result = mock(MetaDataContexts.class, RETURNS_DEEP_STUBS);
136         when(result.getMetaData().getDatabase("foo_db").getResourceMetaData().getStorageTypes()).thenReturn(Collections.singletonMap("foo_ds", new MySQLDatabaseType()));
137         when(result.getMetaData().getDatabase("foo_db").getProtocolType()).thenReturn(new MySQLDatabaseType());
138         ShardingSphereRuleMetaData globalRuleMetaData = new ShardingSphereRuleMetaData(
139                 Arrays.asList(new SQLParserRule(new DefaultSQLParserRuleConfigurationBuilder().build()), new SQLTranslatorRule(new DefaultSQLTranslatorRuleConfigurationBuilder().build()),
140                         new LoggingRule(new DefaultLoggingRuleConfigurationBuilder().build())));
141         when(result.getMetaData().getGlobalRuleMetaData()).thenReturn(globalRuleMetaData);
142         when(result.getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.KERNEL_EXECUTOR_SIZE)).thenReturn(1);
143         when(result.getMetaData().getProps().<Boolean>getValue(ConfigurationPropertyKey.SQL_SHOW)).thenReturn(false);
144         ShardingSphereDatabase database = mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS);
145         when(database.getRuleMetaData()).thenReturn(new ShardingSphereRuleMetaData(Collections.emptyList()));
146         when(result.getMetaData().getDatabase("foo_db")).thenReturn(database);
147         return result;
148         
149     }
150     
151     @Test
152     void assertNext() throws SQLException, NoSuchFieldException, IllegalAccessException {
153         MySQLComQueryPacketExecutor actual = new MySQLComQueryPacketExecutor(packet, connectionSession);
154         MemberAccessor accessor = Plugins.getMemberAccessor();
155         accessor.set(MySQLComQueryPacketExecutor.class.getDeclaredField("proxyBackendHandler"), actual, proxyBackendHandler);
156         when(proxyBackendHandler.next()).thenReturn(true, false);
157         assertTrue(actual.next());
158         assertFalse(actual.next());
159     }

164     }

173     }
