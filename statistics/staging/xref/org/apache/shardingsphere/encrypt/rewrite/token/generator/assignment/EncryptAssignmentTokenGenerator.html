40  import java.util.Optional;

42  /**
43   * Assignment generator for encrypt.
44   */
45  @HighFrequencyInvocation
46  @RequiredArgsConstructor
47  public final class EncryptAssignmentTokenGenerator {
48      
49      private final EncryptRule encryptRule;
50      
51      private final String databaseName;
52      
53      private final DatabaseType databaseType;
54      
55      /**
56       * Generate SQL tokens.
57       *
58       * @param tablesContext SQL statement context
59       * @param setAssignmentSegment set assignment segment
60       * @return generated SQL tokens
61       */
62      public Collection<SQLToken> generateSQLTokens(final TablesContext tablesContext, final SetAssignmentSegment setAssignmentSegment) {
63          String tableName = tablesContext.getSimpleTables().iterator().next().getTableName().getIdentifier().getValue();
64          EncryptTable encryptTable = encryptRule.getEncryptTable(tableName);
65          Collection<SQLToken> result = new LinkedList<>();
66          String schemaName = tablesContext.getSchemaName().orElseGet(() -> new DatabaseTypeRegistry(databaseType).getDefaultSchemaName(databaseName));
67          for (ColumnAssignmentSegment each : setAssignmentSegment.getAssignments()) {
68              String columnName = each.getColumns().get(0).getIdentifier().getValue();
69              if (encryptTable.isEncryptColumn(columnName)) {
70                  generateSQLToken(schemaName, encryptTable.getTable(), encryptTable.getEncryptColumn(columnName), each).ifPresent(result::add);
71              }
72          }
73          return result;
74      }
75      
76      private Optional<EncryptAssignmentToken> generateSQLToken(final String schemaName, final String tableName, final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment) {
77          if (segment.getValue() instanceof ParameterMarkerExpressionSegment) {
78              return Optional.of(generateParameterSQLToken(encryptColumn, segment));
79          }
80          if (segment.getValue() instanceof LiteralExpressionSegment) {
81              return Optional.of(generateLiteralSQLToken(schemaName, tableName, encryptColumn, segment));
82          }
83          return Optional.empty();
84      }
85      
86      private EncryptAssignmentToken generateParameterSQLToken(final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment) {
87          EncryptParameterAssignmentToken result = new EncryptParameterAssignmentToken(segment.getColumns().get(0).getStartIndex(), segment.getStopIndex());
88          result.addColumnName(encryptColumn.getCipher().getName());
89          encryptColumn.getAssistedQuery().ifPresent(optional -> result.addColumnName(optional.getName()));
90          encryptColumn.getLikeQuery().ifPresent(optional -> result.addColumnName(optional.getName()));
91          return result;
92      }
93      
94      private EncryptAssignmentToken generateLiteralSQLToken(final String schemaName, final String tableName, final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment) {
95          EncryptLiteralAssignmentToken result = new EncryptLiteralAssignmentToken(segment.getColumns().get(0).getStartIndex(), segment.getStopIndex());
96          addCipherAssignment(schemaName, tableName, encryptColumn, segment, result);
97          addAssistedQueryAssignment(schemaName, tableName, encryptColumn, segment, result);
98          addLikeAssignment(schemaName, tableName, encryptColumn, segment, result);
99          return result;
100     }
101     
102     private void addCipherAssignment(final String schemaName, final String tableName,
103                                      final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment, final EncryptLiteralAssignmentToken token) {
104         Object originalValue = ((LiteralExpressionSegment) segment.getValue()).getLiterals();
105         Object cipherValue = encryptColumn.getCipher().encrypt(databaseName, schemaName, tableName, encryptColumn.getName(), Collections.singletonList(originalValue)).iterator().next();
106         token.addAssignment(encryptColumn.getCipher().getName(), cipherValue);
107     }
108     
109     private void addAssistedQueryAssignment(final String schemaName, final String tableName, final EncryptColumn encryptColumn,
110                                             final ColumnAssignmentSegment segment, final EncryptLiteralAssignmentToken token) {
111         Object originalValue = ((LiteralExpressionSegment) segment.getValue()).getLiterals();
112         if (encryptColumn.getAssistedQuery().isPresent()) {
113             Object assistedQueryValue = encryptColumn.getAssistedQuery().get().encrypt(
114                     databaseName, schemaName, tableName, encryptColumn.getName(), Collections.singletonList(originalValue)).iterator().next();
115             token.addAssignment(encryptColumn.getAssistedQuery().get().getName(), assistedQueryValue);
116         }
117     }
118     
119     private void addLikeAssignment(final String schemaName, final String tableName,
120                                    final EncryptColumn encryptColumn, final ColumnAssignmentSegment segment, final EncryptLiteralAssignmentToken token) {
121         Object originalValue = ((LiteralExpressionSegment) segment.getValue()).getLiterals();
122         if (encryptColumn.getLikeQuery().isPresent()) {
123             Object assistedQueryValue = encryptColumn.getLikeQuery().get().encrypt(databaseName, schemaName,
124                     tableName, segment.getColumns().get(0).getIdentifier().getValue(), Collections.singletonList(originalValue)).iterator().next();
125             token.addAssignment(encryptColumn.getLikeQuery().get().getName(), assistedQueryValue);
126         }
127     }
128 }
