50  @E2ETestCaseSettings(SQLCommandType.DAL)
51  class DALE2EIT {
52      
53      @ParameterizedTest(name = "{0}")
54      @EnabledIf("isEnabled")
55      @ArgumentsSource(E2ETestCaseArgumentsProvider.class)
56      void assertExecute(final AssertionTestParameter testParam) throws SQLException, ParseException {
57          // TODO make sure test case can not be null
58          if (null == testParam.getTestCaseContext()) {
59              return;
60          }
61          try (SingleE2EContainerComposer containerComposer = new SingleE2EContainerComposer(testParam)) {
62              assertExecute(containerComposer);
63          }
64      }
65      
66      private void assertExecute(final SingleE2EContainerComposer containerComposer) throws SQLException, ParseException {
67          try (Connection connection = containerComposer.getTargetDataSource().getConnection()) {
68              try (Statement statement = connection.createStatement()) {
69                  statement.execute(containerComposer.getSQL());
70                  assertExecuteResult(containerComposer, statement);
71              }
72          }
73      }
74      
75      private void assertExecuteResult(final SingleE2EContainerComposer containerComposer, final Statement statement) throws SQLException {
76          try (ResultSet resultSet = statement.getResultSet()) {
77              if (null == containerComposer.getAssertion().getAssertionSQL()) {
78                  assertResultSet(containerComposer, resultSet);
79              } else {
80                  statement.execute(containerComposer.getAssertion().getAssertionSQL().getSql());
81                  try (ResultSet assertionSQLResultSet = statement.getResultSet()) {
82                      assertResultSet(containerComposer, assertionSQLResultSet);
83                  }
84              }
85          }
86      }
87      
88      private void assertResultSet(final SingleE2EContainerComposer containerComposer, final ResultSet resultSet) throws SQLException {
89          assertMetaData(resultSet.getMetaData(), getExpectedColumns(containerComposer));
90          assertRows(resultSet, containerComposer.getDataSet().getRows());
91      }
92      
93      private Collection<DataSetColumn> getExpectedColumns(final SingleE2EContainerComposer containerComposer) {
94          Collection<DataSetColumn> result = new LinkedList<>();
95          for (DataSetMetaData each : containerComposer.getDataSet().getMetaDataList()) {
96              result.addAll(each.getColumns());
97          }
98          return result;
99      }
100     
101     private void assertMetaData(final ResultSetMetaData actual, final Collection<DataSetColumn> expected) throws SQLException {
102         assertThat(actual.getColumnCount(), is(expected.size()));
103         int index = 1;
104         for (DataSetColumn each : expected) {
105             assertThat(actual.getColumnLabel(index++).toLowerCase(), is(each.getName().toLowerCase()));
106         }
107     }
108     
109     private void assertRows(final ResultSet actual, final List<DataSetRow> expected) throws SQLException {
110         int rowCount = 0;
111         ResultSetMetaData actualMetaData = actual.getMetaData();
112         while (actual.next()) {
113             assertTrue(rowCount < expected.size(), "Size of actual result set is different with size of expected dat set rows.");
114             assertRow(actual, actualMetaData, expected.get(rowCount));
115             rowCount++;
116         }
117         assertThat("Size of actual result set is different with size of expected dat set rows.", rowCount, is(expected.size()));
118     }
119     
120     private void assertRow(final ResultSet actual, final ResultSetMetaData actualMetaData, final DataSetRow expected) throws SQLException {
121         int columnIndex = 1;
122         for (String each : expected.splitValues(",")) {
123             String columnLabel = actualMetaData.getColumnLabel(columnIndex);
124             if (Types.DATE == actual.getMetaData().getColumnType(columnIndex)) {
125                 assertDateValue(actual, columnIndex, columnLabel, each);
126             } else {
127                 assertObjectValue(actual, columnIndex, columnLabel, each);
128             }
129             columnIndex++;
130         }
131     }
132     
133     private void assertDateValue(final ResultSet actual, final int columnIndex, final String columnLabel, final String expected) throws SQLException {
134         if (E2EContainerComposer.NOT_VERIFY_FLAG.equals(expected)) {
135             return;
136         }
137         SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
138         assertThat(dateFormat.format(actual.getDate(columnIndex)), is(expected));
139         assertThat(dateFormat.format(actual.getDate(columnLabel)), is(expected));
140     }
141     
142     private void assertObjectValue(final ResultSet actual, final int columnIndex, final String columnLabel, final String expected) throws SQLException {
143         assertThat(String.valueOf(actual.getObject(columnIndex)), is(expected));
144         assertThat(String.valueOf(actual.getObject(columnLabel)), is(expected));
145     }
146     
147     private static boolean isEnabled() {
148         return E2ETestParameterFactory.containsTestParameter();
149     }
