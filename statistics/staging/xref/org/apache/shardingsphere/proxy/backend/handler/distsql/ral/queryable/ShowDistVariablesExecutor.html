45  /**
46   * Show dist variables executor.
47   */
48  @Setter
49  public final class ShowDistVariablesExecutor implements DistSQLQueryExecutor<ShowDistVariablesStatement>, DistSQLExecutorConnectionContextAware {
50      
51      private DistSQLConnectionContext connectionContext;
52      
53      @Override
54      public Collection<String> getColumnNames(final ShowDistVariablesStatement sqlStatement) {
55          return Arrays.asList("variable_name", "variable_value");
56      }
57      
58      @Override
59      public Collection<LocalDataQueryResultRow> getRows(final ShowDistVariablesStatement sqlStatement, final ContextManager contextManager) {
60          ShardingSphereMetaData metaData = contextManager.getMetaDataContexts().getMetaData();
61          Collection<LocalDataQueryResultRow> result = ConfigurationPropertyKey.getKeyNames().stream()
62                  .filter(each -> !ConfigurationPropertyKey.SQL_SHOW.name().equals(each) && !ConfigurationPropertyKey.SQL_SIMPLE.name().equals(each))
63                  .map(each -> new LocalDataQueryResultRow(each.toLowerCase(), getStringResult(metaData.getProps().getValue(ConfigurationPropertyKey.valueOf(each))))).collect(Collectors.toList());
64          result.addAll(TemporaryConfigurationPropertyKey.getKeyNames().stream()
65                  .map(each -> new LocalDataQueryResultRow(each.toLowerCase(), getStringResult(metaData.getTemporaryProps().getValue(TemporaryConfigurationPropertyKey.valueOf(each)))))
66                  .collect(Collectors.toList()));
67          result.add(new LocalDataQueryResultRow(DistSQLVariable.CACHED_CONNECTIONS.name().toLowerCase(), connectionContext.getConnectionSize()));
68          addLoggingPropsRows(metaData, result);
69          if (sqlStatement.getLikePattern().isPresent()) {
70              String pattern = RegexUtils.convertLikePatternToRegex(sqlStatement.getLikePattern().get());
71              result = result.stream().filter(each -> Pattern.compile(pattern, Pattern.CASE_INSENSITIVE).matcher((String) each.getCell(1)).matches()).collect(Collectors.toList());
72          }
73          return result.stream().sorted(Comparator.comparing(each -> each.getCell(1).toString())).collect(Collectors.toList());
74      }
75      
76      private String getStringResult(final Object value) {
77          if (null == value) {
78              return "";
79          }
80          return value instanceof TypedSPI ? ((TypedSPI) value).getType().toString() : value.toString();
81      }
82      
83      private void addLoggingPropsRows(final ShardingSphereMetaData metaData, final Collection<LocalDataQueryResultRow> rows) {
84          Optional<ShardingSphereLogger> sqlLogger = metaData.getGlobalRuleMetaData().findSingleRule(LoggingRule.class).flatMap(LoggingRule::getSQLLogger);
85          if (sqlLogger.isPresent()) {
86              Properties sqlLoggerProps = sqlLogger.get().getProps();
87              rows.add(new LocalDataQueryResultRow(LoggingConstants.SQL_SHOW_VARIABLE_NAME, sqlLoggerProps.getOrDefault(LoggingConstants.SQL_LOG_ENABLE, Boolean.FALSE).toString()));
88              rows.add(new LocalDataQueryResultRow(LoggingConstants.SQL_SIMPLE_VARIABLE_NAME, sqlLoggerProps.getOrDefault(LoggingConstants.SQL_LOG_SIMPLE, Boolean.FALSE).toString()));
89          } else {
90              rows.add(new LocalDataQueryResultRow(LoggingConstants.SQL_SHOW_VARIABLE_NAME,
91                      metaData.getProps().getValue(ConfigurationPropertyKey.valueOf(LoggingConstants.SQL_SHOW_VARIABLE_NAME.toUpperCase())).toString()));
92              rows.add(new LocalDataQueryResultRow(LoggingConstants.SQL_SIMPLE_VARIABLE_NAME,
93                      metaData.getProps().getValue(ConfigurationPropertyKey.valueOf(LoggingConstants.SQL_SIMPLE_VARIABLE_NAME.toUpperCase())).toString()));
94          }
95      }
96      
97      @Override
98      public Class<ShowDistVariablesStatement> getType() {
99          return ShowDistVariablesStatement.class;
100     }
101 }
