62  @ExtendWith(AutoMockExtension.class)
63  @StaticMockSettings({ProxyContext.class, SystemSchemaUtils.class})
64  class DefaultDatabaseMetaDataExecutorTest {
65      
66      private final Grantee grantee = new Grantee("root", "127.0.0.1");
67      
68      @Mock
69      private ConnectionSession connectionSession;
70      
71      @BeforeEach
72      void setUp() {
73          when(connectionSession.getGrantee()).thenReturn(grantee);
74      }
75      
76      @Test
77      void assertExecuteWithAlias() throws SQLException {
78          Map<String, String> expectedResultSetMap = new HashMap<>(2, 1);
79          expectedResultSetMap.put("sn", "foo_ds");
80          expectedResultSetMap.put("DEFAULT_CHARACTER_SET_NAME", "utf8mb4");
81          String sql = "SELECT SCHEMA_NAME AS sn, DEFAULT_CHARACTER_SET_NAME FROM information_schema.SCHEMATA";
82          ShardingSphereDatabase database = createDatabase(expectedResultSetMap);
83          ContextManager contextManager = mockContextManager(database);
84          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
85          DefaultDatabaseMetaDataExecutor executor = new DefaultDatabaseMetaDataExecutor(sql);
86          executor.execute(connectionSession);
87          assertThat(executor.getRows().get(0).get("sn"), is("foo_ds"));
88          assertThat(executor.getRows().get(0).get("DEFAULT_CHARACTER_SET_NAME"), is("utf8mb4"));
89      }
90      
91      @Test
92      void assertExecuteWithDefaultValue() throws SQLException {
93          String sql = "SELECT COUNT(*) AS support_ndb FROM information_schema.ENGINES WHERE Engine = 'ndbcluster'";
94          ShardingSphereDatabase database = createDatabase(Collections.singletonMap("support_ndb", "0"));
95          ContextManager contextManager = mockContextManager(database);
96          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
97          DefaultDatabaseMetaDataExecutor executor = new DefaultDatabaseMetaDataExecutor(sql);
98          executor.execute(connectionSession);
99          assertThat(executor.getQueryResultMetaData().getColumnCount(), is(1));
100         while (executor.getMergedResult().next()) {
101             assertThat(executor.getMergedResult().getValue(1, String.class), is("0"));
102         }
103     }
104     
105     private ShardingSphereDatabase createDatabase(final Map<String, String> expectedResultSetMap) throws SQLException {
106         return new ShardingSphereDatabase("auth_db", new MySQLDatabaseType(),
107                 new ShardingSphereResourceMetaData("sharding_db", Collections.singletonMap("foo_ds", new MockedDataSource(mockConnection(expectedResultSetMap)))),
108                 mock(ShardingSphereRuleMetaData.class), Collections.emptyMap());
109     }
110     
111     private Connection mockConnection(final Map<String, String> expectedResultSetMap) throws SQLException {
112         Connection result = mock(Connection.class, RETURNS_DEEP_STUBS);
113         when(result.getMetaData().getURL()).thenReturn("jdbc:mysql://localhost:3306/foo_ds");
114         ResultSet resultSet = mockResultSet(expectedResultSetMap);
115         when(result.prepareStatement(any(String.class)).executeQuery()).thenReturn(resultSet);
116         return result;
117     }
118     
119     private ResultSet mockResultSet(final Map<String, String> expectedResultSetMap) throws SQLException {
120         ResultSet result = mock(ResultSet.class, RETURNS_DEEP_STUBS);
121         List<String> keys = new ArrayList<>(expectedResultSetMap.keySet());
122         for (int i = 0; i < keys.size(); i++) {
123             when(result.getMetaData().getColumnName(i + 1)).thenReturn(keys.get(i));
124             when(result.getMetaData().getColumnLabel(i + 1)).thenReturn(keys.get(i));
125             when(result.getString(i + 1)).thenReturn(expectedResultSetMap.get(keys.get(i)));
126         }
127         when(result.next()).thenReturn(true, false);
128         when(result.getMetaData().getColumnCount()).thenReturn(expectedResultSetMap.size());
129         return result;
130     }
131     
132     private ContextManager mockContextManager(final ShardingSphereDatabase database) {
133         AuthorityRule authorityRule = mock(AuthorityRule.class);
134         when(authorityRule.findPrivileges(grantee)).thenReturn(Optional.of(new DatabasePermittedPrivileges(Collections.singleton("auth_db"))));
135         ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
136         MetaDataContexts metaDataContexts = new MetaDataContexts(mock(MetaDataPersistService.class), new ShardingSphereMetaData(Collections.singletonMap("auth_db", database),
137                 new ShardingSphereRuleMetaData(Collections.singleton(authorityRule)), new ConfigurationProperties(new Properties())));
138         when(result.getMetaDataContexts()).thenReturn(metaDataContexts);
139         when(ProxyContext.getInstance().getDatabase("auth_db")).thenReturn(database);
140         when(ProxyContext.getInstance().getAllDatabaseNames()).thenReturn(Collections.singleton("auth_db"));
141         return result;
142     }
