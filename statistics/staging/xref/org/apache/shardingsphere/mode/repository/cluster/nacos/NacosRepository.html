71      @Override
72      public void init(final ClusterPersistRepositoryConfiguration config) {
73          nacosProps = new NacosProperties(config.getProps());
74          client = createClient(config);
75          initServiceMetaData();
76      }

78      private NamingService createClient(final ClusterPersistRepositoryConfiguration config) {
79          Properties props = new Properties();
80          props.setProperty("serverAddr", config.getServerLists());
81          props.setProperty("namespace", config.getNamespace());
82          props.setProperty("username", nacosProps.getValue(NacosPropertyKey.USERNAME));
83          props.setProperty("password", nacosProps.getValue(NacosPropertyKey.PASSWORD));
84          try {
85              return NamingFactory.createNamingService(props);
86          } catch (final NacosException ex) {
87              throw new ClusterPersistRepositoryException(ex);
88          }
89      }
90      
91      private void initServiceMetaData() {
92          try {
93              String clusterIp = nacosProps.getValue(NacosPropertyKey.CLUSTER_IP);
94              String ip = Strings.isNullOrEmpty(clusterIp) ? IpUtils.getIp() : clusterIp;
95              serviceController = new ServiceController();
96              for (ServiceMetaData each : serviceController.getAllServices()) {
97                  Integer port = client.getAllInstances(each.getServiceName(), false).stream()
98                          .filter(instance -> ip.equals(instance.getIp())).map(Instance::getPort).max(Comparator.naturalOrder()).orElse(Integer.MIN_VALUE);
99                  each.setIp(ip);
100                 each.setPort(new AtomicInteger(port));
101             }
102         } catch (final NacosException ex) {
103             throw new ClusterPersistRepositoryException(ex);
104         }
105     }

107     @Override
108     public void persistEphemeral(final String key, final String value) {
109         try {
110             Preconditions.checkNotNull(value, "Value can not be null");
111             if (!findExistedInstance(key, true).isEmpty()) {
112                 delete(key);
113             }
114             put(key, value, true);
115         } catch (final NacosException ex) {
116             throw new ClusterPersistRepositoryException(ex);
117         }
118     }

120     @Override
121     public void persistExclusiveEphemeral(final String key, final String value) {
122         try {
123             Preconditions.checkState(findExistedInstance(key, true).isEmpty(), "Key `%s` already exists", key);
124             put(key, value, true);
125         } catch (final NacosException ex) {
126             throw new ClusterPersistRepositoryException(ex);
127         }
128     }

135     @Override
136     public void watch(final String key, final DataChangedEventListener listener) {
137         try {
138             for (ServiceMetaData each : serviceController.getAllServices()) {
139                 NamingEventListener eventListener = each.getListener();
140                 if (null != eventListener) {
141                     eventListener.put(key, listener);
142                     return;
143                 }
144                 eventListener = new NamingEventListener();
145                 eventListener.put(key, listener);
146                 each.setListener(eventListener);
147                 client.subscribe(each.getServiceName(), eventListener);
148             }
149         } catch (final NacosException ex) {
150             throw new ClusterPersistRepositoryException(ex);
151         }
152     }

154     @Override
155     public String getDirectly(final String key) {
156         try {
157             for (ServiceMetaData each : serviceController.getAllServices()) {
158                 Optional<Instance> instance = findExistedInstance(key, each.isEphemeral()).stream().max(Comparator.comparing(NacosMetaDataUtils::getTimestamp));
159                 if (instance.isPresent()) {
160                     return NacosMetaDataUtils.getValue(instance.get());
161                 }
162             }
163             return null;
164         } catch (final NacosException ex) {
165             throw new ClusterPersistRepositoryException(ex);
166         }
167     }

169     @Override
170     public List<String> getChildrenKeys(final String key) {
171         try {
172             Stream<String> concatKeys = Stream.empty();
173             for (ServiceMetaData each : serviceController.getAllServices()) {
174                 Stream<String> keys = findExistedInstance(each.isEphemeral()).stream()
175                         .map(instance -> {
176                             String fullPath = NacosMetaDataUtils.getKey(instance);
177                             if (fullPath.startsWith(key + PATH_SEPARATOR)) {
178                                 String pathWithoutPrefix = fullPath.substring((key + PATH_SEPARATOR).length());
179                                 return pathWithoutPrefix.contains(PATH_SEPARATOR) ? pathWithoutPrefix.substring(0, pathWithoutPrefix.indexOf(PATH_SEPARATOR)) : pathWithoutPrefix;
180                             }
181                             return null;
182                         }).filter(Objects::nonNull);
183                 concatKeys = Stream.concat(concatKeys, keys);
184             }
185             return concatKeys.distinct().sorted(Comparator.reverseOrder()).collect(Collectors.toList());
186         } catch (final NacosException ex) {
187             throw new ClusterPersistRepositoryException(ex);
188         }
189     }

196     @Override
197     public void persist(final String key, final String value) {
198         try {
199             Preconditions.checkNotNull(value, "Value can not be null");
200             Optional<Instance> instance = findExistedInstance(key, false).stream().max(Comparator.comparing(NacosMetaDataUtils::getTimestamp));
201             if (instance.isPresent()) {
202                 update(instance.get(), value);
203             } else {
204                 put(key, value, false);
205             }
206         } catch (final NacosException ex) {
207             throw new ClusterPersistRepositoryException(ex);
208         }
209     }

216     private void update(final Instance instance, final String value) throws NacosException {
217         Map<String, String> metaDataMap = instance.getMetadata();
218         String key = NacosMetaDataUtils.getKey(instance);
219         metaDataMap.put(key, value);
220         metaDataMap.put(NacosMetaDataUtils.UTC_ZONE_OFFSET.toString(), String.valueOf(NacosMetaDataUtils.getTimestamp()));
221         instance.setMetadata(metaDataMap);
222         ServiceMetaData persistentService = serviceController.getPersistentService();
223         client.registerInstance(persistentService.getServiceName(), instance);
224         Collection<KeyValue> keyValues = new LinkedList<>();
225         keyValues.add(new KeyValue(key, value, instance.isEphemeral()));
226         waitValue(keyValues);
227     }
228     
229     private void put(final String key, final String value, final boolean ephemeral) throws NacosException {
230         final Collection<KeyValue> keyValues = buildParentPath(key);
231         ServiceMetaData serviceMetaData = serviceController.getService(ephemeral);
232         Instance instance = new Instance();
233         instance.setIp(serviceMetaData.getIp());
234         instance.setPort(serviceMetaData.getPort());
235         instance.setEphemeral(ephemeral);
236         Map<String, String> metadataMap = new HashMap<>(5, 1F);
237         if (ephemeral) {
238             fillEphemeralMetaData(metadataMap);
239         }
240         metadataMap.put(key, value);
241         metadataMap.put(NacosMetaDataUtils.UTC_ZONE_OFFSET.toString(), String.valueOf(NacosMetaDataUtils.getTimestamp()));
242         instance.setMetadata(metadataMap);
243         client.registerInstance(serviceMetaData.getServiceName(), instance);
244         keyValues.add(new KeyValue(key, value, ephemeral));
245         waitValue(keyValues);
246     }
247     
248     private Collection<KeyValue> buildParentPath(final String key) throws NacosException {
249         Collection<KeyValue> result = new LinkedList<>();
250         StringBuilder parentPath = new StringBuilder();
251         String[] partPath = key.split(PATH_SEPARATOR);
252         for (int index = 1; index < partPath.length - 1; index++) {
253             String path = parentPath.append(PATH_SEPARATOR).append(partPath[index]).toString();
254             if (findExistedInstance(path, false).isEmpty()) {
255                 result.addAll(build(path));
256             }
257         }
258         return result;
259     }
260     
261     private Collection<KeyValue> build(final String key) throws NacosException {
262         Collection<KeyValue> result = new LinkedList<>();
263         if (findExistedInstance(key, false).isEmpty()) {
264             Instance instance = new Instance();
265             ServiceMetaData persistentService = serviceController.getPersistentService();
266             instance.setIp(persistentService.getIp());
267             instance.setPort(persistentService.getPort());
268             instance.setEphemeral(false);
269             Map<String, String> metaDataMap = new HashMap<>(2, 1F);
270             metaDataMap.put(key, "");
271             metaDataMap.put(NacosMetaDataUtils.UTC_ZONE_OFFSET.toString(), String.valueOf(NacosMetaDataUtils.getTimestamp()));
272             instance.setMetadata(metaDataMap);
273             client.registerInstance(persistentService.getServiceName(), instance);
274             result.add(new KeyValue(key, "", false));
275         }
276         return result;
277     }
278     
279     private void fillEphemeralMetaData(final Map<String, String> metaDataMap) {
280         int timeToLiveSeconds = nacosProps.getValue(NacosPropertyKey.TIME_TO_LIVE_SECONDS);
281         metaDataMap.put(PreservedMetadataKeys.HEART_BEAT_INTERVAL, String.valueOf(timeToLiveSeconds * 1000 / 3));
282         metaDataMap.put(PreservedMetadataKeys.HEART_BEAT_TIMEOUT, String.valueOf(timeToLiveSeconds * 1000 * 2 / 3));
283         metaDataMap.put(PreservedMetadataKeys.IP_DELETE_TIMEOUT, String.valueOf(timeToLiveSeconds * 1000));
284     }

286     @Override
287     public void delete(final String key) {
288         try {
289             for (ServiceMetaData each : serviceController.getAllServices()) {
290                 Collection<Instance> instances = findExistedInstance(each.isEphemeral()).stream()
291                         .filter(instance -> {
292                             String fullPath = NacosMetaDataUtils.getKey(instance);
293                             return fullPath.startsWith(key + PATH_SEPARATOR) || key.equals(fullPath);
294                         })
295                         .sorted(Comparator.comparing(NacosMetaDataUtils::getKey).reversed()).collect(Collectors.toList());
296                 Collection<KeyValue> keyValues = new LinkedList<>();
297                 for (Instance instance : instances) {
298                     client.deregisterInstance(each.getServiceName(), instance);
299                     keyValues.add(new KeyValue(NacosMetaDataUtils.getKey(instance), null, each.isEphemeral()));
300                 }
301                 waitValue(keyValues);
302             }
303         } catch (final NacosException ex) {
304             throw new ClusterPersistRepositoryException(ex);
305         }
306     }
307     
308     private Collection<Instance> findExistedInstance(final String key, final boolean ephemeral) throws NacosException {
309         return client.getAllInstances(serviceController.getService(ephemeral).getServiceName(), false).stream()
310                 .filter(each -> Objects.equals(key, NacosMetaDataUtils.getKey(each))).collect(Collectors.toList());
311     }
312     
313     private Collection<Instance> findExistedInstance(final boolean ephemeral) throws NacosException {
314         return client.getAllInstances(serviceController.getService(ephemeral).getServiceName(), false);
315     }
316     
317     @SneakyThrows(InterruptedException.class)
318     private void waitValue(final Collection<KeyValue> keyValues) throws NacosException {
319         if (!isAvailable(keyValues)) {
320             long retryIntervalMilliseconds = nacosProps.getValue(NacosPropertyKey.RETRY_INTERVAL_MILLISECONDS);
321             int maxRetries = nacosProps.getValue(NacosPropertyKey.MAX_RETRIES);
322             for (int retry = 0; retry < maxRetries; retry++) {
323                 Thread.sleep(getSleepTimeMs(retry, retryIntervalMilliseconds));
324                 if (isAvailable(keyValues)) {
325                     return;
326                 }
327             }
328             throw new NacosException(NacosException.RESOURCE_NOT_FOUND, "Wait value availability timeout exceeded");
329         }
330     }
331     
332     private boolean isAvailable(final Collection<KeyValue> keyValues) throws NacosException {
333         Map<Boolean, List<KeyValue>> keyValueMap = keyValues.stream().collect(Collectors.groupingBy(KeyValue::isEphemeral));
334         for (Entry<Boolean, List<KeyValue>> entry : keyValueMap.entrySet()) {
335             ServiceMetaData service = serviceController.getService(entry.getKey());
336             Map<String, List<Instance>> instanceMap = client.getAllInstances(service.getServiceName(), false).stream().collect(Collectors.groupingBy(NacosMetaDataUtils::getKey));
337             keyValues.removeIf(keyValue -> {
338                 String key = keyValue.getKey();
339                 String value = keyValue.getValue();
340                 return instanceMap.containsKey(key) ? instanceMap.get(key).stream().anyMatch(each -> Objects.equals(NacosMetaDataUtils.getValue(each), value)) : null == value;
341             });
342         }
343         return keyValues.isEmpty();
344     }
345     
346     private long getSleepTimeMs(final int retryCount, final long baseSleepTimeMs) {
347         return baseSleepTimeMs * Math.max(1, random.nextInt(1 << (retryCount + 1)));
348     }
