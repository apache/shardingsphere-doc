95      @Test
96      void assertUpdate() {
97          CreateShardingTableRuleStatement statement = new CreateShardingTableRuleStatement(false, Arrays.asList(createCompleteAutoTableRule(), createCompleteTableRule()));
98          updater.checkSQLStatement(database, statement, currentRuleConfig);
99          ShardingRuleConfiguration toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, statement);
100         updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
101         assertThat(currentRuleConfig.getTables().size(), is(2));
102         Iterator<ShardingTableRuleConfiguration> tableRuleIterator = currentRuleConfig.getTables().iterator();
103         ShardingTableRuleConfiguration tableRule = tableRuleIterator.next();
104         assertThat(tableRule.getTableShardingStrategy(), instanceOf(StandardShardingStrategyConfiguration.class));
105         assertThat(tableRule.getLogicTable(), is("t_order"));
106         assertThat(tableRule.getActualDataNodes(), is("ds_${0..1}.t_order_${0..1}"));
107         assertThat(((StandardShardingStrategyConfiguration) tableRule.getTableShardingStrategy()).getShardingColumn(), is("order_id"));
108         assertThat(tableRule.getTableShardingStrategy().getShardingAlgorithmName(), is("t_order_algorithm"));
109         tableRule = tableRuleIterator.next();
110         assertThat(tableRule.getTableShardingStrategy(), instanceOf(StandardShardingStrategyConfiguration.class));
111         assertThat(tableRule.getLogicTable(), is("t_order_input"));
112         assertThat(tableRule.getActualDataNodes(), is("ds_${0..1}.t_order_${0..1}"));
113         assertThat(((StandardShardingStrategyConfiguration) tableRule.getTableShardingStrategy()).getShardingColumn(), is("product_id"));
114         assertThat(tableRule.getTableShardingStrategy().getShardingAlgorithmName(), is("t_order_input_table_core.standard.fixture"));
115         assertThat(tableRule.getDatabaseShardingStrategy(), instanceOf(StandardShardingStrategyConfiguration.class));
116         assertThat(tableRule.getDatabaseShardingStrategy().getShardingAlgorithmName(), is("t_order_input_database_inline"));
117         assertThat(currentRuleConfig.getTables().size(), is(2));
118         Iterator<ShardingAutoTableRuleConfiguration> autoTableIterator = currentRuleConfig.getAutoTables().iterator();
119         ShardingAutoTableRuleConfiguration autoTableRule = autoTableIterator.next();
120         assertThat(autoTableRule.getLogicTable(), is("t_order_item"));
121         assertThat(autoTableRule.getActualDataSources(), is("ds_0"));
122         assertThat(autoTableRule.getShardingStrategy().getShardingAlgorithmName(), is("t_order_mod_test"));
123         assertThat(((StandardShardingStrategyConfiguration) autoTableRule.getShardingStrategy()).getShardingColumn(), is("order_id"));
124         assertThat(autoTableRule.getKeyGenerateStrategy().getColumn(), is("product_id"));
125         assertThat(autoTableRule.getKeyGenerateStrategy().getKeyGeneratorName(), is("product_id_DISTSQL.FIXTURE"));
126         autoTableRule = autoTableIterator.next();
127         assertThat(autoTableRule.getLogicTable(), is("t_order_item_input"));
128         assertThat(autoTableRule.getActualDataSources(), is("logic_ds"));
129         assertThat(autoTableRule.getShardingStrategy().getShardingAlgorithmName(), is("t_order_item_input_foo.distsql.fixture"));
130         assertThat(((StandardShardingStrategyConfiguration) autoTableRule.getShardingStrategy()).getShardingColumn(), is("order_id"));
131         assertThat(autoTableRule.getKeyGenerateStrategy().getColumn(), is("product_id"));
132         assertThat(autoTableRule.getKeyGenerateStrategy().getKeyGeneratorName(), is("t_order_item_input_distsql.fixture"));
133     }

199     private AutoTableRuleSegment createCompleteAutoTableRule() {
200         AutoTableRuleSegment result = new AutoTableRuleSegment("t_order_item_input", Collections.singletonList("logic_ds"));
201         result.setKeyGenerateStrategySegment(new KeyGenerateStrategySegment("product_id", new AlgorithmSegment("DISTSQL.FIXTURE", new Properties())));
202         result.setShardingColumn("order_id");
203         result.setShardingAlgorithmSegment(new AlgorithmSegment("FOO.DISTSQL.FIXTURE", PropertiesBuilder.build(new Property("", ""))));
204         return result;
205     }
206     
207     private TableRuleSegment createCompleteTableRule() {
208         KeyGenerateStrategySegment keyGenerator = new KeyGenerateStrategySegment("product_id", new AlgorithmSegment("DISTSQL.FIXTURE", new Properties()));
209         TableRuleSegment result = new TableRuleSegment("t_order_input", Collections.singletonList("ds_${0..1}.t_order_${0..1}"), keyGenerator, null);
210         result.setTableStrategySegment(new ShardingStrategySegment("standard", "product_id", new AlgorithmSegment("CORE.STANDARD.FIXTURE", new Properties())));
211         AlgorithmSegment databaseAlgorithmSegment = new AlgorithmSegment("inline", PropertiesBuilder.build(new Property("algorithm-expression", "ds_${user_id % 2}")));
212         result.setDatabaseStrategySegment(new ShardingStrategySegment("standard", "product_id", databaseAlgorithmSegment));
213         return result;
214     }

216     private ShardingRuleConfiguration createCurrentShardingRuleConfiguration() {
217         ShardingRuleConfiguration result = new ShardingRuleConfiguration();
218         result.getTables().add(createTableRuleConfiguration());
219         result.getAutoTables().add(createAutoTableRuleConfiguration());
220         result.getShardingAlgorithms().put("t_order_algorithm", new AlgorithmConfiguration("hash_mod", PropertiesBuilder.build(new Property("sharding-count", "4"))));
221         result.getKeyGenerators().put("t_order_item_snowflake", new AlgorithmConfiguration("snowflake", new Properties()));
222         return result;
223     }

245     @SneakyThrows(ReflectiveOperationException.class)
246     @SuppressWarnings("rawtypes")
247     private DistSQLStatement getDistSQLStatement(final String sql) {
248         ShardingDistSQLStatementParserFacade facade = new ShardingDistSQLStatementParserFacade();
249         ParseASTNode parseASTNode = (ParseASTNode) SQLParserFactory.newInstance(sql, facade.getLexerClass(), facade.getParserClass()).parse();
250         SQLVisitor visitor = TypedSPILoader.getService(FeaturedDistSQLStatementParserFacade.class, facade.getType()).getVisitorClass().getDeclaredConstructor().newInstance();
251         return (DistSQLStatement) ((ParseTreeVisitor) visitor).visit(parseASTNode.getRootNode());
252     }
