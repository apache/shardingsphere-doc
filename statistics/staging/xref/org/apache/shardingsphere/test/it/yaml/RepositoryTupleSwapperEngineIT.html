27  import org.apache.shardingsphere.mode.tuple.annotation.RepositoryTupleEntity;

30  import java.io.File;
31  import java.io.IOException;
32  import java.net.URL;
33  import java.nio.file.Files;

35  import java.util.Arrays;
36  import java.util.Collection;
37  import java.util.Collections;
38  import java.util.List;

42  import java.util.stream.Collectors;

44  import static org.hamcrest.CoreMatchers.is;
45  import static org.hamcrest.MatcherAssert.assertThat;
46  import static org.junit.jupiter.api.Assertions.assertNotNull;
47  import static org.junit.jupiter.api.Assertions.assertTrue;
48  
49  @Slf4j
50  public abstract class RepositoryTupleSwapperEngineIT {
51      
52      private final File yamlFile;
53      
54      private final RepositoryTupleSwapperEngine engine;
55      
56      protected RepositoryTupleSwapperEngineIT(final String yamlFileName) {
57          URL url = Thread.currentThread().getContextClassLoader().getResource(yamlFileName);
58          assertNotNull(url);
59          yamlFile = new File(url.getFile());
60          engine = new RepositoryTupleSwapperEngine();
61      }
62      
63      @Test
64      void assertSwapToRepositoryTuples() throws IOException {
65          YamlRuleConfiguration yamlRuleConfig = loadYamlRuleConfiguration();
66          assertRepositoryTuples(new ArrayList<>(engine.swapToRepositoryTuples(yamlRuleConfig)), yamlRuleConfig);
67      }
68      
69      private YamlRuleConfiguration loadYamlRuleConfiguration() throws IOException {
70          YamlRootConfiguration yamlRootConfig = YamlEngine.unmarshal(yamlFile, YamlRootConfiguration.class);
71          assertThat(yamlRootConfig.getRules().size(), is(1));
72          return yamlRootConfig.getRules().iterator().next();
73      }
74      
75      protected abstract void assertRepositoryTuples(List<RepositoryTuple> actualRepositoryTuples, YamlRuleConfiguration expectedYamlRuleConfig);
76      
77      protected void assertRepositoryTuple(final RepositoryTuple actual, final String expectedKey, final Object expectedValue) {
78          assertThat(actual.getKey(), is(expectedKey));
79          assertThat(actual.getValue(), is(isSimpleObject(expectedValue) ? expectedValue : YamlEngine.marshal(expectedValue)));
80      }
81      
82      private boolean isSimpleObject(final Object expectedValue) {
83          return expectedValue instanceof Boolean || expectedValue instanceof Integer || expectedValue instanceof Long || expectedValue instanceof String;
84      }
85      
86      @Test
87      void assertSwapToYamlRuleConfiguration() throws IOException {
88          String actualYamlContent = getActualYamlContent();
89          String expectedYamlContent = getExpectedYamlContent();
90          if (!sameYamlContext(actualYamlContent, expectedYamlContent)) {
91              assertThat(actualYamlContent, is(expectedYamlContent));
92          }
93      }
94      
95      private String getActualYamlContent() throws IOException {
96          YamlRuleConfiguration yamlRuleConfig = loadYamlRuleConfiguration();
97          String ruleTypeName = Objects.requireNonNull(yamlRuleConfig.getClass().getAnnotation(RepositoryTupleEntity.class)).value();
98          Collection<RepositoryTuple> repositoryTuples = engine.swapToRepositoryTuples(yamlRuleConfig).stream()
99                  .map(each -> new RepositoryTuple(getRepositoryTupleKey(yamlRuleConfig instanceof YamlGlobalRuleConfiguration, ruleTypeName, each), each.getValue())).collect(Collectors.toList());
100         Optional<YamlRuleConfiguration> actualYamlRuleConfig = engine.swapToYamlRuleConfiguration(repositoryTuples, yamlRuleConfig.getClass());
101         assertTrue(actualYamlRuleConfig.isPresent());
102         YamlRootConfiguration yamlRootConfig = new YamlRootConfiguration();
103         yamlRootConfig.setRules(Collections.singletonList(actualYamlRuleConfig.get()));
104         return YamlEngine.marshal(yamlRootConfig);
105     }
106     
107     private String getRepositoryTupleKey(final boolean isGlobalRule, final String ruleTypeName, final RepositoryTuple tuple) {
108         return isGlobalRule
109                 ? String.format("/metadata/rules/%s/versions/0", ruleTypeName)
110                 : String.format("/metadata/foo_db/rules/%s/%s/versions/0", ruleTypeName, tuple.getKey());
111     }
112     
113     private String getExpectedYamlContent() throws IOException {
114         String content = Files.readAllLines(yamlFile.toPath()).stream()
115                 .filter(each -> !each.contains("#") && !each.isEmpty()).collect(Collectors.joining(System.lineSeparator())) + System.lineSeparator();
116         return YamlEngine.marshal(YamlEngine.unmarshal(content, Map.class));
117     }
118     
119     private boolean sameYamlContext(final String actualYamlContext, final String expectedYamlContext) {
120         char[] actualArray = actualYamlContext.toCharArray();
121         char[] expectedArray = expectedYamlContext.toCharArray();
122         Arrays.sort(actualArray);
123         Arrays.sort(expectedArray);
124         return Arrays.equals(actualArray, expectedArray);
125     }
126 }
