18  package org.apache.shardingsphere.sql.parser.sql.dialect.statement.opengauss.dml;
19  
20  import lombok.Setter;
21  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.ReturningSegment;
22  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.column.OnDuplicateKeyColumnsSegment;
23  import org.apache.shardingsphere.sql.parser.sql.common.segment.generic.WithSegment;
24  import org.apache.shardingsphere.sql.parser.sql.common.statement.dml.InsertStatement;
25  import org.apache.shardingsphere.sql.parser.sql.dialect.statement.opengauss.OpenGaussStatement;
26  
27  import java.util.Optional;
28  
29  /**
30   * OpenGauss insert statement.
31   */
32  @Setter
33  public final class OpenGaussInsertStatement extends InsertStatement implements OpenGaussStatement {
34      
35      private WithSegment withSegment;
36      
37      private OnDuplicateKeyColumnsSegment onDuplicateKeyColumnsSegment;
38      
39      private ReturningSegment returningSegment;
40      
41      /**
42       * Get with segment.
43       *
44       * @return with segment.
45       */
46      public Optional<WithSegment> getWithSegment() {
47          return Optional.ofNullable(withSegment);
48      }
49      
50      /**
51       * Get on duplicate key columns segment.
52       *
53       * @return on duplicate key columns segment
54       */
55      public Optional<OnDuplicateKeyColumnsSegment> getOnDuplicateKeyColumns() {
56          return Optional.ofNullable(onDuplicateKeyColumnsSegment);
57      }
58      
59      /**
60       * Get returning segment.
61       *
62       * @return returning segment
63       */
64      public Optional<ReturningSegment> getReturningSegment() {
65          return Optional.ofNullable(returningSegment);
66      }
67  }
