101     @Override
102     public Map<Integer, InventoryIncrementalJobItemProgress> getJobProgress(final PipelineJobConfiguration jobConfig) {
103         String jobId = jobConfig.getJobId();
104         JobConfigurationPOJO jobConfigPOJO = getElasticJobConfigPOJO(jobId);
105         return IntStream.range(0, jobConfig.getJobShardingCount()).boxed().collect(LinkedHashMap::new, (map, each) -> {
106             Optional<InventoryIncrementalJobItemProgress> jobItemProgress = getJobItemProgress(jobId, each);
107             jobItemProgress.ifPresent(optional -> optional.setActive(!jobConfigPOJO.isDisabled()));
108             map.put(each, jobItemProgress.orElse(null));
109         }, LinkedHashMap::putAll);
110     }
111     
112     @Override
113     public List<InventoryIncrementalJobItemInfo> getJobItemInfos(final String jobId) {
114         JobConfigurationPOJO jobConfigPOJO = getElasticJobConfigPOJO(jobId);
115         PipelineJobConfiguration jobConfig = getJobConfiguration(jobConfigPOJO);
116         long startTimeMillis = Long.parseLong(Optional.ofNullable(jobConfigPOJO.getProps().getProperty("start_time_millis")).orElse("0"));
117         Map<Integer, InventoryIncrementalJobItemProgress> jobProgress = getJobProgress(jobConfig);
118         List<InventoryIncrementalJobItemInfo> result = new LinkedList<>();
119         for (Entry<Integer, InventoryIncrementalJobItemProgress> entry : jobProgress.entrySet()) {
120             int shardingItem = entry.getKey();
121             TableBasedPipelineJobInfo jobInfo = getJobInfo(jobId);
122             InventoryIncrementalJobItemProgress jobItemProgress = entry.getValue();
123             String errorMessage = getJobItemErrorMessage(jobId, shardingItem);
124             if (null == jobItemProgress) {
125                 result.add(new InventoryIncrementalJobItemInfo(shardingItem, jobInfo.getTable(), null, startTimeMillis, 0, errorMessage));
126                 continue;
127             }
128             int inventoryFinishedPercentage = 0;
129             if (JobStatus.EXECUTE_INCREMENTAL_TASK == jobItemProgress.getStatus()) {
130                 inventoryFinishedPercentage = 100;
131             } else if (0 != jobItemProgress.getProcessedRecordsCount() && 0 != jobItemProgress.getInventoryRecordsCount()) {
132                 inventoryFinishedPercentage = (int) Math.min(100, jobItemProgress.getProcessedRecordsCount() * 100 / jobItemProgress.getInventoryRecordsCount());
133             }
134             result.add(new InventoryIncrementalJobItemInfo(shardingItem, jobInfo.getTable(), jobItemProgress, startTimeMillis, inventoryFinishedPercentage, errorMessage));
135         }
136         return result;
137     }

139     @Override
140     public void persistJobItemProgress(final PipelineJobItemContext jobItemContext) {
141         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobItemContext.getJobId()))
142                 .persistJobItemProgress(jobItemContext.getJobId(), jobItemContext.getShardingItem(), convertJobItemProgress(jobItemContext));
143     }
144     
145     private String convertJobItemProgress(final PipelineJobItemContext jobItemContext) {
146         InventoryIncrementalJobItemContext context = (InventoryIncrementalJobItemContext) jobItemContext;
147         InventoryIncrementalJobItemProgress jobItemProgress = new InventoryIncrementalJobItemProgress();
148         jobItemProgress.setStatus(context.getStatus());
149         jobItemProgress.setSourceDatabaseType(context.getJobConfig().getSourceDatabaseType());
150         jobItemProgress.setDataSourceName(context.getDataSourceName());
151         jobItemProgress.setIncremental(getIncrementalTasksProgress(context.getIncrementalTasks()));
152         jobItemProgress.setInventory(getInventoryTasksProgress(context.getInventoryTasks()));
153         jobItemProgress.setProcessedRecordsCount(context.getProcessedRecordsCount());
154         jobItemProgress.setInventoryRecordsCount(context.getInventoryRecordsCount());
155         return YamlEngine.marshal(jobItemProgressSwapper.swapToYamlConfiguration(jobItemProgress));
156     }
157     
158     @Override
159     public void updateJobItemProgress(final PipelineJobItemContext jobItemContext) {
160         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobItemContext.getJobId()))
161                 .updateJobItemProgress(jobItemContext.getJobId(), jobItemContext.getShardingItem(), convertJobItemProgress(jobItemContext));
162     }

164     private JobItemIncrementalTasksProgress getIncrementalTasksProgress(final Collection<PipelineTask> incrementalTasks) {
165         return new JobItemIncrementalTasksProgress(incrementalTasks.isEmpty() ? null : (IncrementalTaskProgress) incrementalTasks.iterator().next().getTaskProgress());
166     }
167     
168     private JobItemInventoryTasksProgress getInventoryTasksProgress(final Collection<PipelineTask> inventoryTasks) {
169         Map<String, InventoryTaskProgress> inventoryTaskProgressMap = new HashMap<>();
170         for (PipelineTask each : inventoryTasks) {
171             inventoryTaskProgressMap.put(each.getTaskId(), (InventoryTaskProgress) each.getTaskProgress());
172         }
173         return new JobItemInventoryTasksProgress(inventoryTaskProgressMap);
174     }

176     @Override
177     public void persistJobOffsetInfo(final String jobId, final JobOffsetInfo jobOffsetInfo) {
178         String value = YamlEngine.marshal(jobOffsetInfoSwapper.swapToYamlConfiguration(jobOffsetInfo));
179         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).persistJobOffsetInfo(jobId, value);
180     }
181     
182     @Override
183     public JobOffsetInfo getJobOffsetInfo(final String jobId) {
184         Optional<String> offsetInfo = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).getJobOffsetInfo(jobId);
185         if (offsetInfo.isPresent()) {
186             YamlJobOffsetInfo info = YamlEngine.unmarshal(offsetInfo.get(), YamlJobOffsetInfo.class);
187             return jobOffsetInfoSwapper.swapToObject(info);
188         }
189         return jobOffsetInfoSwapper.swapToObject(new YamlJobOffsetInfo());
190     }
191     
192     @Override
193     public Optional<InventoryIncrementalJobItemProgress> getJobItemProgress(final String jobId, final int shardingItem) {
194         Optional<String> progress = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).getJobItemProgress(jobId, shardingItem);
195         return progress.map(optional -> jobItemProgressSwapper.swapToObject(YamlEngine.unmarshal(optional, YamlInventoryIncrementalJobItemProgress.class)));
196     }
197     
198     @Override
199     public void updateJobItemStatus(final String jobId, final int shardingItem, final JobStatus status) {
200         Optional<InventoryIncrementalJobItemProgress> jobItemProgress = getJobItemProgress(jobId, shardingItem);
201         if (!jobItemProgress.isPresent()) {
202             return;
203         }
204         jobItemProgress.get().setStatus(status);
205         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).updateJobItemProgress(jobId, shardingItem,
206                 YamlEngine.marshal(jobItemProgressSwapper.swapToYamlConfiguration(jobItemProgress.get())));
207     }

209     @Override
210     public Collection<DataConsistencyCheckAlgorithmInfo> listDataConsistencyCheckAlgorithms() {
211         Collection<DataConsistencyCheckAlgorithmInfo> result = new LinkedList<>();
212         for (DataConsistencyCalculateAlgorithm each : ShardingSphereServiceLoader.getServiceInstances(DataConsistencyCalculateAlgorithm.class)) {
213             SPIDescription description = each.getClass().getAnnotation(SPIDescription.class);
214             result.add(new DataConsistencyCheckAlgorithmInfo(each.getType(), getSupportedDatabaseTypes(each.getSupportedDatabaseTypes()), null == description ? "" : description.value()));
215         }
216         return result;
217     }
218     
219     private Collection<DatabaseType> getSupportedDatabaseTypes(final Collection<DatabaseType> supportedDatabaseTypes) {
220         return supportedDatabaseTypes.isEmpty() ? ShardingSphereServiceLoader.getServiceInstances(DatabaseType.class) : supportedDatabaseTypes;
221     }

223     @Override
224     public DataConsistencyCalculateAlgorithm buildDataConsistencyCalculateAlgorithm(final String algorithmType, final Properties algorithmProps) {
225         return TypedSPILoader.getService(DataConsistencyCalculateAlgorithm.class, null == algorithmType ? "DATA_MATCH" : algorithmType, algorithmProps);
226     }
227     
228     @Override
229     public Map<String, DataConsistencyCheckResult> dataConsistencyCheck(final PipelineJobConfiguration jobConfig, final DataConsistencyCalculateAlgorithm calculateAlgorithm,
230                                                                         final ConsistencyCheckJobItemProgressContext progressContext) {
231         String jobId = jobConfig.getJobId();
232         PipelineDataConsistencyChecker dataConsistencyChecker = buildPipelineDataConsistencyChecker(jobConfig, buildPipelineProcessContext(jobConfig), progressContext);
233         Map<String, DataConsistencyCheckResult> result = dataConsistencyChecker.check(calculateAlgorithm);
234         log.info("job {} with check algorithm '{}' data consistency checker result {}", jobId, calculateAlgorithm.getType(), result);
235         return result;
236     }
237     
238     protected abstract PipelineDataConsistencyChecker buildPipelineDataConsistencyChecker(PipelineJobConfiguration pipelineJobConfig, InventoryIncrementalProcessContext processContext,
239                                                                                           ConsistencyCheckJobItemProgressContext progressContext);
240     
241     @Override
242     public boolean aggregateDataConsistencyCheckResults(final String jobId, final Map<String, DataConsistencyCheckResult> checkResults) {
243         if (checkResults.isEmpty()) {
244             throw new IllegalArgumentException("checkResults empty, jobId:" + jobId);
245         }
246         for (Entry<String, DataConsistencyCheckResult> entry : checkResults.entrySet()) {
247             DataConsistencyCheckResult checkResult = entry.getValue();
248             if (!checkResult.isMatched()) {
249                 return false;
250             }
251         }
252         return true;
253     }
