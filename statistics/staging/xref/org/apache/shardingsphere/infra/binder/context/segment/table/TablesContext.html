49  /**
50   * Tables context.
51   */
52  @Getter
53  @ToString
54  public final class TablesContext {
55      
56      @Getter(AccessLevel.NONE)
57      private final Collection<TableSegment> tables = new LinkedList<>();
58      
59      private final Collection<SimpleTableSegment> simpleTables = new LinkedList<>();
60      
61      private final Collection<String> tableNames = new CaseInsensitiveSet<>();
62      
63      private final Collection<String> schemaNames = new CaseInsensitiveSet<>();
64      
65      private final Collection<String> databaseNames = new CaseInsensitiveSet<>();
66      
67      @Getter(AccessLevel.NONE)
68      private final Map<String, Collection<SubqueryTableContext>> subqueryTables = new HashMap<>();
69      
70      private final Map<String, IdentifierValue> tableNameAliasMap = new HashMap<>();
71      
72      public TablesContext(final SimpleTableSegment table, final DatabaseType databaseType, final String currentDatabaseName) {
73          this(null == table ? Collections.emptyList() : Collections.singletonList(table), databaseType, currentDatabaseName);
74      }
75      
76      public TablesContext(final Collection<SimpleTableSegment> tables, final DatabaseType databaseType, final String currentDatabaseName) {
77          this(tables, Collections.emptyMap(), databaseType, currentDatabaseName);
78      }
79      
80      public TablesContext(final Collection<? extends TableSegment> tables, final Map<Integer, SelectStatementContext> subqueryContexts,
81                           final DatabaseType databaseType, final String currentDatabaseName) {
82          if (tables.isEmpty()) {
83              return;
84          }
85          this.tables.addAll(tables);
86          for (TableSegment each : tables) {
87              if (each instanceof SimpleTableSegment && !"DUAL".equalsIgnoreCase(((SimpleTableSegment) each).getTableName().getIdentifier().getValue())) {
88                  SimpleTableSegment simpleTableSegment = (SimpleTableSegment) each;
89                  simpleTables.add(simpleTableSegment);
90                  tableNames.add(simpleTableSegment.getTableName().getIdentifier().getValue());
91                  // TODO use sql binder result when statement which contains tables support bind logic
92                  simpleTableSegment.getOwner().ifPresent(optional -> schemaNames.add(optional.getIdentifier().getValue()));
93                  databaseNames.add(findDatabaseName(simpleTableSegment, databaseType).orElse(currentDatabaseName));
94                  tableNameAliasMap.put(simpleTableSegment.getTableName().getIdentifier().getValue().toLowerCase(), each.getAlias().orElse(simpleTableSegment.getTableName().getIdentifier()));
95              }
96              if (each instanceof SubqueryTableSegment) {
97                  subqueryTables.putAll(createSubqueryTables(subqueryContexts, (SubqueryTableSegment) each));
98              }
99          }
100     }
101     
102     private Optional<String> findDatabaseName(final SimpleTableSegment tableSegment, final DatabaseType databaseType) {
103         DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(databaseType).getDialectDatabaseMetaData();
104         Optional<OwnerSegment> owner = dialectDatabaseMetaData.getDefaultSchema().isPresent() ? tableSegment.getOwner().flatMap(OwnerSegment::getOwner) : tableSegment.getOwner();
105         return owner.map(optional -> optional.getIdentifier().getValue());
106     }
107     
108     private Map<String, Collection<SubqueryTableContext>> createSubqueryTables(final Map<Integer, SelectStatementContext> subqueryContexts, final SubqueryTableSegment subqueryTable) {
109         if (!subqueryContexts.containsKey(subqueryTable.getSubquery().getStartIndex())) {
110             return Collections.emptyMap();
111         }
112         SelectStatementContext subqueryContext = subqueryContexts.get(subqueryTable.getSubquery().getStartIndex());
113         Map<String, SubqueryTableContext> subqueryTableContexts = new SubqueryTableContextEngine().createSubqueryTableContexts(subqueryContext, subqueryTable.getAliasName().orElse(null));
114         Map<String, Collection<SubqueryTableContext>> result = new HashMap<>(subqueryTableContexts.size(), 1F);
115         for (SubqueryTableContext each : subqueryTableContexts.values()) {
116             if (null != each.getAliasName()) {
117                 result.computeIfAbsent(each.getAliasName(), unused -> new LinkedList<>()).add(each);
118             }
119         }
120         return result;
121     }
122     
123     /**
124      * Find expression table name map.
125      *
126      * @param columns column segments
127      * @param schema schema
128      * @return expression table name map
129      */
130     public Map<String, String> findTableNames(final Collection<ColumnSegment> columns, final ShardingSphereSchema schema) {
131         if (1 == simpleTables.size()) {
132             return findTableNameFromSingleTable(columns);
133         }
134         Map<String, String> result = new CaseInsensitiveMap<>();
135         Map<String, Collection<String>> ownerColumnNames = getOwnerColumnNames(columns);
136         result.putAll(findTableNameFromSQL(ownerColumnNames));
137         Collection<String> noOwnerColumnNames = getNoOwnerColumnNames(columns);
138         result.putAll(findTableNameFromMetaData(noOwnerColumnNames, schema));
139         result.putAll(findTableNameFromSubquery(columns, result));
140         return result;
141     }
142     
143     private Map<String, String> findTableNameFromSubquery(final Collection<ColumnSegment> columns, final Map<String, String> ownerTableNames) {
144         if (ownerTableNames.size() == columns.size() || subqueryTables.isEmpty()) {
145             return Collections.emptyMap();
146         }
147         Map<String, String> result = new LinkedHashMap<>(columns.size(), 1F);
148         for (ColumnSegment each : columns) {
149             if (ownerTableNames.containsKey(each.getExpression())) {
150                 continue;
151             }
152             String owner = each.getOwner().map(optional -> optional.getIdentifier().getValue()).orElse("");
153             Collection<SubqueryTableContext> subqueryTableContexts = subqueryTables.getOrDefault(owner, Collections.emptyList());
154             for (SubqueryTableContext subqueryTableContext : subqueryTableContexts) {
155                 if (subqueryTableContext.getColumnNames().contains(each.getIdentifier().getValue())) {
156                     result.put(each.getExpression(), subqueryTableContext.getTableName());
157                 }
158             }
159         }
160         return result;
161     }
162     
163     private Map<String, String> findTableNameFromSingleTable(final Collection<ColumnSegment> columns) {
164         String tableName = simpleTables.iterator().next().getTableName().getIdentifier().getValue();
165         Map<String, String> result = new CaseInsensitiveMap<>();
166         for (ColumnSegment each : columns) {
167             result.putIfAbsent(each.getExpression(), tableName);
168         }
169         return result;
170     }
171     
172     private Map<String, Collection<String>> getOwnerColumnNames(final Collection<ColumnSegment> columns) {
173         Map<String, Collection<String>> result = new CaseInsensitiveMap<>();
174         for (ColumnSegment each : columns) {
175             if (!each.getOwner().isPresent()) {
176                 continue;
177             }
178             result.computeIfAbsent(each.getOwner().get().getIdentifier().getValue(), unused -> new LinkedList<>()).add(each.getExpression());
179         }
180         return result;
181     }
182     
183     private Map<String, String> findTableNameFromSQL(final Map<String, Collection<String>> ownerColumnNames) {
184         if (ownerColumnNames.isEmpty()) {
185             return Collections.emptyMap();
186         }
187         Map<String, String> result = new LinkedHashMap<>(simpleTables.size(), 1F);
188         for (SimpleTableSegment each : simpleTables) {
189             String tableName = each.getTableName().getIdentifier().getValue();
190             if (ownerColumnNames.containsKey(tableName)) {
191                 ownerColumnNames.get(tableName).forEach(column -> result.put(column, tableName));
192             }
193             Optional<String> alias = each.getAliasName();
194             if (alias.isPresent() && ownerColumnNames.containsKey(alias.get())) {
195                 ownerColumnNames.get(alias.get()).forEach(column -> result.put(column, tableName));
196             }
197         }
198         return result;
199     }
200     
201     private Map<String, String> findTableNameFromMetaData(final Collection<String> noOwnerColumnNames, final ShardingSphereSchema schema) {
202         if (noOwnerColumnNames.isEmpty()) {
203             return Collections.emptyMap();
204         }
205         Map<String, String> result = new LinkedHashMap<>(noOwnerColumnNames.size(), 1F);
206         for (SimpleTableSegment each : simpleTables) {
207             String tableName = each.getTableName().getIdentifier().getValue();
208             for (String columnName : schema.getAllColumnNames(tableName)) {
209                 if (noOwnerColumnNames.contains(columnName)) {
210                     result.put(columnName, tableName);
211                 }
212             }
213         }
214         return result;
215     }
216     
217     private Collection<String> getNoOwnerColumnNames(final Collection<ColumnSegment> columns) {
218         Collection<String> result = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
219         for (ColumnSegment each : columns) {
220             if (!each.getOwner().isPresent()) {
221                 result.add(each.getIdentifier().getValue());
222             }
223         }
224         return result;
225     }
226     
227     /**
228      * Get database name.
229      *
230      * @return database name
231      */
232     public Optional<String> getDatabaseName() {
233         Preconditions.checkState(databaseNames.size() <= 1, "Can not support multiple different database.");
234         return databaseNames.isEmpty() ? Optional.empty() : Optional.of(databaseNames.iterator().next());
235     }
236     
237     /**
238      * Get schema name.
239      *
240      * @return schema name
241      */
242     public Optional<String> getSchemaName() {
243         return schemaNames.isEmpty() ? Optional.empty() : Optional.of(schemaNames.iterator().next());
244     }
