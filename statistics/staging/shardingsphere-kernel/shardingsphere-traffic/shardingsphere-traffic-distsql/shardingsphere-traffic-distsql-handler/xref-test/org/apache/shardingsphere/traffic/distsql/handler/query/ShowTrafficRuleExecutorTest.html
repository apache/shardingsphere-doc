45  class ShowTrafficRuleExecutorTest {
46      
47      @Test
48      void assertExecute() {
49          ShardingSphereMetaData metaData = mockMetaData();
50          ShowTrafficRuleExecutor executor = new ShowTrafficRuleExecutor();
51          Collection<LocalDataQueryResultRow> actual = executor.getRows(metaData, mock(ShowTrafficRulesStatement.class));
52          assertThat(actual.size(), is(2));
53          Iterator<LocalDataQueryResultRow> iterator = actual.iterator();
54          LocalDataQueryResultRow row = iterator.next();
55          assertThat(row.getCell(1), is("rule_name_1"));
56          assertThat(row.getCell(2), is("olap,order_by"));
57          assertThat(row.getCell(3), is("SQL_MATCH"));
58          assertThat(row.getCell(4), is("sql=select * from t_order"));
59          assertThat(row.getCell(5), is("RANDOM"));
60          assertThat(row.getCell(6), is(""));
61          row = iterator.next();
62          assertThat(row.getCell(1), is("rule_name_2"));
63          assertThat(row.getCell(2), is("oltp"));
64          assertThat(row.getCell(3), is("SQL_HINT"));
65          assertThat(row.getCell(4), is(""));
66          assertThat(row.getCell(5), is("ROBIN"));
67          assertThat(row.getCell(6), is(""));
68      }
69      
70      @Test
71      void assertGetColumnNames() {
72          ShowTrafficRuleExecutor executor = new ShowTrafficRuleExecutor();
73          Collection<String> columns = executor.getColumnNames();
74          assertThat(columns.size(), is(6));
75          Iterator<String> iterator = columns.iterator();
76          assertThat(iterator.next(), is("name"));
77          assertThat(iterator.next(), is("labels"));
78          assertThat(iterator.next(), is("algorithm_type"));
79          assertThat(iterator.next(), is("algorithm_props"));
80          assertThat(iterator.next(), is("load_balancer_type"));
81          assertThat(iterator.next(), is("load_balancer_props"));
82      }
83      
84      private ShardingSphereMetaData mockMetaData() {
85          TrafficRule trafficRule = mock(TrafficRule.class);
86          when(trafficRule.getConfiguration()).thenReturn(createTrafficRuleConfiguration());
87          return new ShardingSphereMetaData(new LinkedHashMap<>(), new ShardingSphereRuleMetaData(Collections.singleton(trafficRule)), new ConfigurationProperties(new Properties()));
88      }
89      
90      private TrafficRuleConfiguration createTrafficRuleConfiguration() {
91          TrafficRuleConfiguration result = new TrafficRuleConfiguration();
92          result.getTrafficStrategies().add(new TrafficStrategyConfiguration("rule_name_1", Arrays.asList("olap", "order_by"), "algorithm_1", "load_balancer_1"));
93          result.getTrafficStrategies().add(new TrafficStrategyConfiguration("rule_name_2", Collections.singletonList("oltp"), "algorithm_2", "load_balancer_2"));
94          result.getLoadBalancers().put("load_balancer_1", new AlgorithmConfiguration("RANDOM", new Properties()));
95          result.getLoadBalancers().put("load_balancer_2", new AlgorithmConfiguration("ROBIN", new Properties()));
96          result.getTrafficAlgorithms().put("algorithm_1", new AlgorithmConfiguration("SQL_MATCH", PropertiesBuilder.build(new Property("sql", "select * from t_order"))));
97          result.getTrafficAlgorithms().put("algorithm_2", new AlgorithmConfiguration("SQL_HINT", new Properties()));
98          return result;
99      }
