38  /**
39   * Insert statement binder.
40   */
41  public final class InsertStatementBinder implements SQLStatementBinder<InsertStatement> {
42      
43      @Override
44      public InsertStatement bind(final InsertStatement sqlStatement, final SQLStatementBinderContext binderContext) {
45          InsertStatement result = copy(sqlStatement);
46          Map<String, TableSegmentBinderContext> tableBinderContexts = new LinkedHashMap<>();
47          Optional.ofNullable(sqlStatement.getTable()).ifPresent(optional -> result.setTable(SimpleTableSegmentBinder.bind(optional, binderContext, tableBinderContexts)));
48          if (sqlStatement.getInsertColumns().isPresent() && !sqlStatement.getInsertColumns().get().getColumns().isEmpty()) {
49              result.setInsertColumns(InsertColumnsSegmentBinder.bind(sqlStatement.getInsertColumns().get(), binderContext, tableBinderContexts));
50          } else {
51              sqlStatement.getInsertColumns().ifPresent(result::setInsertColumns);
52              tableBinderContexts.values().forEach(each -> result.getDerivedInsertColumns().addAll(getVisibleColumns(each.getProjectionSegments())));
53          }
54          sqlStatement.getInsertSelect().ifPresent(optional -> result.setInsertSelect(SubquerySegmentBinder.bind(optional, binderContext, tableBinderContexts)));
55          return result;
56      }
57      
58      @SneakyThrows(ReflectiveOperationException.class)
59      private InsertStatement copy(final InsertStatement sqlStatement) {
60          InsertStatement result = sqlStatement.getClass().getDeclaredConstructor().newInstance();
61          result.getValues().addAll(sqlStatement.getValues());
62          sqlStatement.getSetAssignment().ifPresent(result::setSetAssignment);
63          sqlStatement.getOnDuplicateKeyColumns().ifPresent(result::setOnDuplicateKeyColumns);
64          sqlStatement.getWithSegment().ifPresent(result::setWithSegment);
65          sqlStatement.getOutputSegment().ifPresent(result::setOutputSegment);
66          sqlStatement.getMultiTableInsertType().ifPresent(result::setMultiTableInsertType);
67          sqlStatement.getMultiTableInsertIntoSegment().ifPresent(result::setMultiTableInsertIntoSegment);
68          sqlStatement.getMultiTableConditionalIntoSegment().ifPresent(result::setMultiTableConditionalIntoSegment);
69          sqlStatement.getReturningSegment().ifPresent(result::setReturningSegment);
70          result.addParameterMarkerSegments(sqlStatement.getParameterMarkerSegments());
71          result.getCommentSegments().addAll(sqlStatement.getCommentSegments());
72          return result;
73      }
74      
75      private Collection<ColumnSegment> getVisibleColumns(final Collection<ProjectionSegment> projectionSegments) {
76          return projectionSegments.stream()
77                  .filter(each -> each instanceof ColumnProjectionSegment && each.isVisible()).map(each -> ((ColumnProjectionSegment) each).getColumn()).collect(Collectors.toList());
78      }
79  }
