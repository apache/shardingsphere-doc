44  /**
45   * Transaction backend handler factory.
46   */
47  @NoArgsConstructor(access = AccessLevel.PRIVATE)
48  public final class TransactionBackendHandlerFactory {
49      
50      /**
51       * New instance of backend handler.
52       * 
53       * @param sqlStatementContext SQL statement context
54       * @param sql SQL
55       * @param connectionSession connection session
56       * @return backend handler
57       */
58      public static ProxyBackendHandler newInstance(final SQLStatementContext sqlStatementContext, final String sql, final ConnectionSession connectionSession) {
59          TCLStatement tclStatement = (TCLStatement) sqlStatementContext.getSqlStatement();
60          if (tclStatement instanceof BeginTransactionStatement || tclStatement instanceof StartTransactionStatement) {
61              return new TransactionBackendHandler(tclStatement, TransactionOperationType.BEGIN, connectionSession);
62          }
63          if (tclStatement instanceof SetAutoCommitStatement) {
64              return new TransactionBackendHandler(tclStatement, TransactionOperationType.SET_AUTOCOMMIT, connectionSession);
65          }
66          if (tclStatement instanceof SavepointStatement) {
67              return new TransactionBackendHandler(tclStatement, TransactionOperationType.SAVEPOINT, connectionSession);
68          }
69          if (tclStatement instanceof ReleaseSavepointStatement) {
70              return new TransactionBackendHandler(tclStatement, TransactionOperationType.RELEASE_SAVEPOINT, connectionSession);
71          }
72          if (tclStatement instanceof CommitStatement) {
73              return new TransactionBackendHandler(tclStatement, TransactionOperationType.COMMIT, connectionSession);
74          }
75          if (tclStatement instanceof RollbackStatement) {
76              return ((RollbackStatement) tclStatement).getSavepointName().isPresent()
77                      ? new TransactionBackendHandler(tclStatement, TransactionOperationType.ROLLBACK_TO_SAVEPOINT, connectionSession)
78                      : new TransactionBackendHandler(tclStatement, TransactionOperationType.ROLLBACK, connectionSession);
79          }
80          if (tclStatement instanceof SetTransactionStatement && OperationScope.GLOBAL != ((SetTransactionStatement) tclStatement).getScope()) {
81              return new TransactionSetHandler((SetTransactionStatement) tclStatement, connectionSession);
82          }
83          if (tclStatement instanceof XAStatement) {
84              return new TransactionXAHandler(sqlStatementContext, sql, connectionSession);
85          }
86          QueryContext queryContext = new QueryContext(sqlStatementContext, sql, Collections.emptyList(), new HintValueContext(), connectionSession.getConnectionContext(),
87                  ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData());
88          return DatabaseConnectorFactory.getInstance().newInstance(queryContext, connectionSession.getDatabaseConnectionManager(), false);
89      }
90  }
