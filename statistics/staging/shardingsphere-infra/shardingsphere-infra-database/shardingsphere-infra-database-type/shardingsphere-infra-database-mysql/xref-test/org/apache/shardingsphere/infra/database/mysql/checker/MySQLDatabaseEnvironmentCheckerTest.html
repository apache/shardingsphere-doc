18  package org.apache.shardingsphere.infra.database.mysql.checker;
19  
20  import org.apache.shardingsphere.infra.database.core.checker.PrivilegeCheckType;
21  import org.apache.shardingsphere.infra.database.core.exception.CheckDatabaseEnvironmentFailedException;
22  import org.apache.shardingsphere.infra.database.core.exception.MissingRequiredPrivilegeException;
23  import org.apache.shardingsphere.infra.database.core.exception.UnexpectedVariableValueException;
24  import org.junit.jupiter.api.BeforeEach;
25  import org.junit.jupiter.api.Test;
26  import org.junit.jupiter.api.extension.ExtendWith;
27  import org.mockito.Answers;
28  import org.mockito.Mock;
29  import org.mockito.junit.jupiter.MockitoExtension;
30  
31  import javax.sql.DataSource;
32  import java.sql.PreparedStatement;
33  import java.sql.ResultSet;
34  import java.sql.SQLException;
35  
36  import static org.junit.jupiter.api.Assertions.assertDoesNotThrow;
37  import static org.junit.jupiter.api.Assertions.assertThrows;
38  import static org.mockito.ArgumentMatchers.anyString;
39  import static org.mockito.Mockito.times;
40  import static org.mockito.Mockito.verify;
41  import static org.mockito.Mockito.when;
42  
43  @ExtendWith(MockitoExtension.class)
44  class MySQLDatabaseEnvironmentCheckerTest {
45      
46      @Mock
47      private PreparedStatement preparedStatement;
48      
49      @Mock
50      private ResultSet resultSet;
51      
52      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
53      private DataSource dataSource;
54      
55      @BeforeEach
56      void setUp() throws SQLException {
57          when(dataSource.getConnection().prepareStatement(anyString())).thenReturn(preparedStatement);
58      }
59      
60      @Test
61      void assertCheckPrivilegeWithParticularSuccess() throws SQLException {
62          when(preparedStatement.executeQuery()).thenReturn(resultSet);
63          when(resultSet.next()).thenReturn(true);
64          when(resultSet.getString(1)).thenReturn("GRANT REPLICATION SLAVE, REPLICATION CLIENT ON *.* TO '%'@'%'");
65          new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.PIPELINE);
66          verify(preparedStatement).executeQuery();
67      }
68      
69      @Test
70      void assertCheckPrivilegeWithAllSuccess() throws SQLException {
71          when(preparedStatement.executeQuery()).thenReturn(resultSet);
72          when(resultSet.next()).thenReturn(true);
73          when(resultSet.getString(1)).thenReturn("GRANT ALL PRIVILEGES CLIENT ON *.* TO '%'@'%'");
74          new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.PIPELINE);
75          verify(preparedStatement).executeQuery();
76      }
77      
78      @Test
79      void assertCheckPrivilegeLackPrivileges() throws SQLException {
80          when(preparedStatement.executeQuery()).thenReturn(resultSet);
81          assertThrows(MissingRequiredPrivilegeException.class, () -> new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.PIPELINE));
82      }
83      
84      @Test
85      void assertCheckPrivilegeFailure() throws SQLException {
86          when(preparedStatement.executeQuery()).thenReturn(resultSet);
87          when(resultSet.next()).thenThrow(new SQLException(""));
88          assertThrows(CheckDatabaseEnvironmentFailedException.class, () -> new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.PIPELINE));
89      }
90      
91      @Test
92      void assertCheckVariableSuccess() throws SQLException {
93          when(preparedStatement.executeQuery()).thenReturn(resultSet);
94          when(resultSet.next()).thenReturn(true, true, true, false);
95          when(resultSet.getString(1)).thenReturn("LOG_BIN", "BINLOG_FORMAT", "BINLOG_ROW_IMAGE");
96          when(resultSet.getString(2)).thenReturn("ON", "ROW", "FULL");
97          assertDoesNotThrow(() -> new MySQLDatabaseEnvironmentChecker().checkVariable(dataSource));
98          verify(preparedStatement, times(1)).executeQuery();
99      }
100     
101     @Test
102     void assertCheckVariableWithWrongVariable() throws SQLException {
103         when(preparedStatement.executeQuery()).thenReturn(resultSet);
104         when(resultSet.next()).thenReturn(true, true, false);
105         when(resultSet.getString(1)).thenReturn("BINLOG_FORMAT", "LOG_BIN");
106         when(resultSet.getString(2)).thenReturn("ROW", "OFF");
107         assertThrows(UnexpectedVariableValueException.class, () -> new MySQLDatabaseEnvironmentChecker().checkVariable(dataSource));
108     }
109     
110     @Test
111     void assertCheckVariableFailure() throws SQLException {
112         when(preparedStatement.executeQuery()).thenReturn(resultSet);
113         when(resultSet.next()).thenThrow(new SQLException(""));
114         assertThrows(CheckDatabaseEnvironmentFailedException.class, () -> new MySQLDatabaseEnvironmentChecker().checkVariable(dataSource));
115     }
116     
117     @Test
118     void assertCheckXAPrivilegeWithParticularSuccessInMySQL8() throws SQLException {
119         when(preparedStatement.executeQuery()).thenReturn(resultSet);
120         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
121         when(resultSet.next()).thenReturn(true);
122         when(resultSet.getString(1)).thenReturn("GRANT XA_RECOVER_ADMIN ON *.* TO '%'@'%'");
123         new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.XA);
124         verify(preparedStatement).executeQuery();
125     }
126     
127     @Test
128     void assertUnCheckXAPrivilegeInMySQL5() throws SQLException {
129         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(5);
130         new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.XA);
131         verify(preparedStatement, times(0)).executeQuery();
132     }
133     
134     @Test
135     void assertCheckXAPrivilegeWithAllSuccessInMySQL8() throws SQLException {
136         when(preparedStatement.executeQuery()).thenReturn(resultSet);
137         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
138         when(resultSet.next()).thenReturn(true);
139         when(resultSet.getString(1)).thenReturn("GRANT ALL PRIVILEGES ON *.* TO '%'@'%'");
140         new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.XA);
141         verify(preparedStatement).executeQuery();
142     }
143     
144     @Test
145     void assertCheckXAPrivilegeLackPrivilegesInMySQL8() throws SQLException {
146         when(preparedStatement.executeQuery()).thenReturn(resultSet);
147         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
148         assertThrows(MissingRequiredPrivilegeException.class, () -> new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.XA));
149     }
150     
151     @Test
152     void assertCheckXAPrivilegeFailureInMySQL8() throws SQLException {
153         when(preparedStatement.executeQuery()).thenReturn(resultSet);
154         when(dataSource.getConnection().getMetaData().getDatabaseMajorVersion()).thenReturn(8);
155         when(resultSet.next()).thenThrow(new SQLException(""));
156         assertThrows(CheckDatabaseEnvironmentFailedException.class, () -> new MySQLDatabaseEnvironmentChecker().checkPrivilege(dataSource, PrivilegeCheckType.XA));
157     }
158 }
