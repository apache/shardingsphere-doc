18  package org.apache.shardingsphere.agent.plugin.metrics.prometheus.collector.type;
19  
20  import io.prometheus.client.Histogram;
21  import io.prometheus.client.Histogram.Builder;
22  import org.apache.shardingsphere.agent.plugin.metrics.core.config.MetricConfiguration;
23  import org.apache.shardingsphere.agent.plugin.metrics.core.collector.type.HistogramMetricsCollector;
24  
25  import java.util.Map;
26  
27  /**
28   * Metrics histogram collector of Prometheus.
29   */
30  public final class PrometheusMetricsHistogramCollector implements HistogramMetricsCollector {
31      
32      private static final String BUCKETS_KEY = "buckets";
33      
34      private static final String TYPE_KEY = "type";
35      
36      private static final String EXP_TYPE_KEY = "exp";
37      
38      private static final String LINEAR_TYPE_KEY = "linear";
39      
40      private static final String START_KEY = "start";
41      
42      private static final String COUNT_KEY = "count";
43      
44      private static final String FACTOR_KEY = "factor";
45      
46      private static final String WIDTH_KEY = "width";
47      
48      private final Histogram histogram;
49      
50      public PrometheusMetricsHistogramCollector(final MetricConfiguration config) {
51          Builder builder = Histogram.build().name(config.getId()).help(config.getHelp()).labelNames(config.getLabels().toArray(new String[0]));
52          appendProperties(builder, config.getProps());
53          histogram = builder.register();
54      }
55      
56      @SuppressWarnings("unchecked")
57      private void appendProperties(final Builder builder, final Map<String, Object> props) {
58          Map<String, Object> buckets = (Map<String, Object>) props.get(BUCKETS_KEY);
59          if (null == buckets) {
60              return;
61          }
62          if (EXP_TYPE_KEY.equals(buckets.get(TYPE_KEY))) {
63              double start = null == buckets.get(START_KEY) ? 1D : Double.parseDouble(buckets.get(START_KEY).toString());
64              double factor = null == buckets.get(FACTOR_KEY) ? 1D : Double.parseDouble(buckets.get(FACTOR_KEY).toString());
65              int count = null == buckets.get(COUNT_KEY) ? 1 : (int) buckets.get(COUNT_KEY);
66              builder.exponentialBuckets(start, factor, count);
67          } else if (LINEAR_TYPE_KEY.equals(buckets.get(TYPE_KEY))) {
68              double start = null == buckets.get(START_KEY) ? 1D : Double.parseDouble(buckets.get(START_KEY).toString());
69              double width = null == buckets.get(WIDTH_KEY) ? 1D : Double.parseDouble(buckets.get(WIDTH_KEY).toString());
70              int count = null == buckets.get(COUNT_KEY) ? 1 : (int) buckets.get(COUNT_KEY);
71              builder.linearBuckets(start, width, count);
72          }
73      }
74      
75      @Override
76      public void observe(final double value) {
77          histogram.observe(value);
78      }
79  }
