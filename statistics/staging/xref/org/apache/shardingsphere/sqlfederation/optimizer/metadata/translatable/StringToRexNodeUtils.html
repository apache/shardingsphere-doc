20  import org.antlr.v4.runtime.CharStream;
21  import org.antlr.v4.runtime.CharStreams;
22  import org.antlr.v4.runtime.CommonTokenStream;
23  import org.antlr.v4.runtime.tree.ParseTree;
24  import org.apache.calcite.jdbc.JavaTypeFactoryImpl;
25  import org.apache.calcite.rel.type.RelDataTypeSystem;
26  import org.apache.calcite.rex.RexBuilder;
27  import org.apache.calcite.rex.RexNode;
28  import org.apache.shardingsphere.sqlfederation.optimizer.parser.rexnode.ParseRexNodeLexer;
29  import org.apache.shardingsphere.sqlfederation.optimizer.parser.rexnode.ParseRexNodeParser;
30  
31  import java.util.Map;
32  
33  /**
34   * Parsing string and generate rex node utility class.
35   */
36  public final class StringToRexNodeUtils {
37      
38      /**
39       * Parse string and generate rex node.
40       * @param filterValue filter condition
41       * @param rexBuilder used to build rex node
42       * @param parameters parameters for SQL placeholder
43       * @param columnMap mapping of column id and column type
44       * @return rex node
45       */
46      public static RexNode buildRexNode(final String filterValue, final RexBuilder rexBuilder, final Map<String, Object> parameters, final Map<Integer, Integer> columnMap) {
47          CharStream input = CharStreams.fromString(filterValue);
48          ParseRexNodeLexer lexer = new ParseRexNodeLexer(input);
49          CommonTokenStream tokens = new CommonTokenStream(lexer);
50          ParseRexNodeParser parser = new ParseRexNodeParser(tokens);
51          ParseTree tree = parser.expression();
52          ParseRexNodeVisitorImpl visitor = new ParseRexNodeVisitorImpl(rexBuilder, new JavaTypeFactoryImpl(RelDataTypeSystem.DEFAULT), parameters, columnMap);
53          return visitor.visit(tree);
54      }
55  }
