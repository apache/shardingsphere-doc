77  @ExtendWith(AutoMockExtension.class)
78  @StaticMockSettings(ProxyContext.class)
79  @MockitoSettings(strictness = Strictness.LENIENT)
80  class PostgreSQLAggregatedBatchedStatementsCommandExecutorTest {
81      
82      private static final int CONNECTION_ID = 1;
83      
84      private static final String STATEMENT_ID = "S_1";
85      
86      private static final String SQL = "INSERT INTO t_order (id) VALUES (?)";
87      
88      private static final int BATCH_SIZE = 10;
89      
90      private final ShardingSphereSQLParserEngine parserEngine = new ShardingSphereSQLParserEngine("PostgreSQL", new CacheOption(2000, 65535L), new CacheOption(128, 1024L), false);
91      
92      @Test
93      void assertExecute() throws SQLException {
94          ConnectionSession connectionSession = mockConnectionSession();
95          PostgreSQLAggregatedBatchedStatementsCommandExecutor executor = new PostgreSQLAggregatedBatchedStatementsCommandExecutor(connectionSession, createPackets());
96          ContextManager contextManager = mockContextManager();
97          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
98          List<DatabasePacket<?>> actualPackets = new ArrayList<>(executor.execute());
99          assertThat(actualPackets.size(), is(BATCH_SIZE * 3));
100         for (int i = 0; i < BATCH_SIZE; i++) {
101             assertThat(actualPackets.get(i * 3), is(PostgreSQLBindCompletePacket.getInstance()));
102             assertThat(actualPackets.get(i * 3 + 1), is(PostgreSQLNoDataPacket.getInstance()));
103             assertThat(actualPackets.get(i * 3 + 2), instanceOf(PostgreSQLCommandCompletePacket.class));
104         }
105     }
106     
107     private ConnectionSession mockConnectionSession() throws SQLException {
108         ConnectionSession result = mock(ConnectionSession.class);
109         @SuppressWarnings("rawtypes")
110         SQLStatementContext sqlStatementContext = mock(InsertStatementContext.class);
111         when(sqlStatementContext.getSqlStatement()).thenReturn(parserEngine.parse(SQL, false));
112         when(result.getDatabaseName()).thenReturn("foo_db");
113         when(result.getConnectionContext()).thenReturn(new ConnectionContext());
114         when(result.getServerPreparedStatementRegistry()).thenReturn(new ServerPreparedStatementRegistry());
115         result.getServerPreparedStatementRegistry().addPreparedStatement(STATEMENT_ID,
116                 new PostgreSQLServerPreparedStatement(SQL, sqlStatementContext, Collections.singletonList(PostgreSQLColumnType.POSTGRESQL_TYPE_INT4)));
117         when(result.getConnectionId()).thenReturn(CONNECTION_ID);
118         BackendConnection backendConnection = mock(BackendConnection.class);
119         Connection connection = mock(Connection.class, RETURNS_DEEP_STUBS);
120         when(connection.getMetaData().getURL()).thenReturn("jdbc:postgresql://127.0.0.1/db");
121         when(backendConnection.getConnections(nullable(String.class), anyInt(), any(ConnectionMode.class))).thenReturn(Collections.singletonList(connection));
122         PreparedStatement preparedStatement = mock(PreparedStatement.class);
123         when(preparedStatement.getConnection()).thenReturn(connection);
124         JDBCBackendStatement backendStatement = mock(JDBCBackendStatement.class);
125         when(backendStatement.createStorageResource(any(ExecutionUnit.class), any(Connection.class), any(ConnectionMode.class), any(StatementOption.class), nullable(DatabaseType.class)))
126                 .thenReturn(preparedStatement);
127         when(result.getStatementManager()).thenReturn(backendStatement);
128         when(result.getBackendConnection()).thenReturn(backendConnection);
129         return result;
130     }
131     
132     private List<PostgreSQLCommandPacket> createPackets() {
133         List<PostgreSQLCommandPacket> result = new ArrayList<>(BATCH_SIZE);
134         for (int i = 0; i < BATCH_SIZE; i++) {
135             PostgreSQLComBindPacket bindPacket = mock(PostgreSQLComBindPacket.class);
136             when(bindPacket.getStatementId()).thenReturn(STATEMENT_ID);
137             when(bindPacket.readParameters(Collections.singletonList(PostgreSQLColumnType.POSTGRESQL_TYPE_INT4))).thenReturn(Collections.singletonList(i));
138             PostgreSQLComDescribePacket describePacket = mock(PostgreSQLComDescribePacket.class);
139             PostgreSQLComExecutePacket executePacket = mock(PostgreSQLComExecutePacket.class);
140             result.add(bindPacket);
141             result.add(describePacket);
142             result.add(executePacket);
143         }
144         return result;
145     }
146     
147     private ContextManager mockContextManager() {
148         ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
149         when(result.getMetaDataContexts().getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.KERNEL_EXECUTOR_SIZE)).thenReturn(0);
150         when(result.getMetaDataContexts().getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY)).thenReturn(1);
151         when(result.getMetaDataContexts().getMetaData().getProps().<Boolean>getValue(ConfigurationPropertyKey.SQL_SHOW)).thenReturn(false);
152         ShardingSphereRuleMetaData globalRuleMetaData = new ShardingSphereRuleMetaData(Arrays.asList(new SQLTranslatorRule(new DefaultSQLTranslatorRuleConfigurationBuilder().build()),
153                 new LoggingRule(new DefaultLoggingRuleConfigurationBuilder().build())));
154         when(result.getMetaDataContexts().getMetaData().getGlobalRuleMetaData()).thenReturn(globalRuleMetaData);
155         ShardingSphereDatabase database = mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS);
156         when(database.getResourceMetaData().getAllInstanceDataSourceNames()).thenReturn(Collections.singletonList("foo_ds"));
157         when(database.getResourceMetaData().getStorageTypes()).thenReturn(Collections.singletonMap("foo_ds", new PostgreSQLDatabaseType()));
158         when(database.getRuleMetaData()).thenReturn(new ShardingSphereRuleMetaData(Collections.emptyList()));
159         when(result.getMetaDataContexts().getMetaData().getDatabase("foo_db")).thenReturn(database);
160         return result;
161     }
