18  package org.apache.shardingsphere.data.pipeline.common.ingest.channel.memory;
19  
20  import org.apache.shardingsphere.data.pipeline.api.ingest.channel.AckCallback;
21  import org.apache.shardingsphere.data.pipeline.api.ingest.channel.PipelineChannel;
22  import org.apache.shardingsphere.data.pipeline.common.ingest.channel.PipelineChannelCreator;
23  
24  import java.util.Properties;
25  
26  /**
27   * Memory implementation of pipeline channel creator.
28   */
29  public final class MemoryPipelineChannelCreator implements PipelineChannelCreator {
30      
31      private static final String BLOCK_QUEUE_SIZE_KEY = "block-queue-size";
32      
33      private static final String BLOCK_QUEUE_SIZE_DEFAULT_VALUE = "2000";
34      
35      private int blockQueueSize;
36      
37      @Override
38      public void init(final Properties props) {
39          blockQueueSize = Integer.parseInt(props.getProperty(BLOCK_QUEUE_SIZE_KEY, BLOCK_QUEUE_SIZE_DEFAULT_VALUE));
40      }
41      
42      @Override
43      public PipelineChannel createPipelineChannel(final int outputConcurrency, final int averageElementSize, final AckCallback ackCallback) {
44          return 1 == outputConcurrency ? new SimpleMemoryPipelineChannel((int) Math.ceil((double) blockQueueSize / averageElementSize), ackCallback)
45                  : new MultiplexMemoryPipelineChannel(outputConcurrency, blockQueueSize, ackCallback);
46      }
47      
48      @Override
49      public String getType() {
50          return "MEMORY";
51      }
52  }
