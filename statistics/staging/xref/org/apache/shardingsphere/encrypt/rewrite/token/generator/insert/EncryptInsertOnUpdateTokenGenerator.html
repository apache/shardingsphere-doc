53  /**
54   * Insert on update values token generator for encrypt.
55   */
56  @HighFrequencyInvocation
57  @RequiredArgsConstructor
58  @Setter
59  public final class EncryptInsertOnUpdateTokenGenerator implements CollectionSQLTokenGenerator<InsertStatementContext>, DatabaseNameAware {
60      
61      private final EncryptRule encryptRule;
62      
63      private String databaseName;
64      
65      @Override
66      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
67          return sqlStatementContext instanceof InsertStatementContext && (((InsertStatementContext) sqlStatementContext).getSqlStatement()).getOnDuplicateKeyColumns().isPresent();
68      }
69      
70      @Override
71      public Collection<SQLToken> generateSQLTokens(final InsertStatementContext insertStatementContext) {
72          InsertStatement insertStatement = insertStatementContext.getSqlStatement();
73          Preconditions.checkState(insertStatement.getOnDuplicateKeyColumns().isPresent());
74          Collection<ColumnAssignmentSegment> onDuplicateKeyColumnsSegments = insertStatement.getOnDuplicateKeyColumns().get().getColumns();
75          if (onDuplicateKeyColumnsSegments.isEmpty()) {
76              return Collections.emptyList();
77          }
78          String schemaName = insertStatementContext.getTablesContext().getSchemaName()
79                  .orElseGet(() -> new DatabaseTypeRegistry(insertStatementContext.getDatabaseType()).getDefaultSchemaName(databaseName));
80          String tableName = insertStatement.getTable().getTableName().getIdentifier().getValue();
81          EncryptTable encryptTable = encryptRule.getEncryptTable(tableName);
82          Collection<SQLToken> result = new LinkedList<>();
83          for (ColumnAssignmentSegment each : onDuplicateKeyColumnsSegments) {
84              boolean leftColumnIsEncrypt = encryptTable.isEncryptColumn(each.getColumns().get(0).getIdentifier().getValue());
85              if (each.getValue() instanceof FunctionSegment && "VALUES".equalsIgnoreCase(((FunctionSegment) each.getValue()).getFunctionName())) {
86                  Optional<ExpressionSegment> rightColumnSegment = ((FunctionSegment) each.getValue()).getParameters().stream().findFirst();
87                  Preconditions.checkState(rightColumnSegment.isPresent());
88                  boolean rightColumnIsEncrypt = encryptTable.isEncryptColumn(((ColumnSegment) rightColumnSegment.get()).getIdentifier().getValue());
89                  if (!leftColumnIsEncrypt && !rightColumnIsEncrypt) {
90                      continue;
91                  }
92              }
93              if (!leftColumnIsEncrypt) {
94                  continue;
95              }
96              EncryptColumn encryptColumn = encryptTable.getEncryptColumn(each.getColumns().get(0).getIdentifier().getValue());
97              generateSQLToken(schemaName, encryptTable, encryptColumn, each).ifPresent(result::add);
98          }
99          return result;
100     }
101     
102     private Optional<EncryptAssignmentToken> generateSQLToken(final String schemaName, final EncryptTable encryptTable,
103                                                               final EncryptColumn encryptColumn, final ColumnAssignmentSegment assignmentSegment) {
104         if (assignmentSegment.getValue() instanceof ParameterMarkerExpressionSegment) {
105             return Optional.of(generateParameterSQLToken(encryptTable, assignmentSegment));
106         }
107         if (assignmentSegment.getValue() instanceof FunctionSegment && "VALUES".equalsIgnoreCase(((FunctionSegment) assignmentSegment.getValue()).getFunctionName())) {
108             return Optional.of(generateValuesSQLToken(encryptTable, assignmentSegment, (FunctionSegment) assignmentSegment.getValue()));
109         }
110         if (assignmentSegment.getValue() instanceof LiteralExpressionSegment) {
111             return Optional.of(generateLiteralSQLToken(schemaName, encryptTable.getTable(), encryptColumn, assignmentSegment));
112         }
113         return Optional.empty();
114     }
115     
116     private EncryptAssignmentToken generateParameterSQLToken(final EncryptTable encryptTable, final ColumnAssignmentSegment assignmentSegment) {
117         EncryptParameterAssignmentToken result = new EncryptParameterAssignmentToken(assignmentSegment.getColumns().get(0).getStartIndex(), assignmentSegment.getStopIndex());
118         String columnName = assignmentSegment.getColumns().get(0).getIdentifier().getValue();
119         EncryptColumn encryptColumn = encryptTable.getEncryptColumn(columnName);
120         result.addColumnName(encryptColumn.getCipher().getName());
121         encryptColumn.getAssistedQuery().ifPresent(optional -> result.addColumnName(optional.getName()));
122         encryptColumn.getLikeQuery().ifPresent(optional -> result.addColumnName(optional.getName()));
123         return result;
124     }
125     
126     private EncryptAssignmentToken generateLiteralSQLToken(final String schemaName, final String tableName,
127                                                            final EncryptColumn encryptColumn, final ColumnAssignmentSegment assignmentSegment) {
128         EncryptLiteralAssignmentToken result = new EncryptLiteralAssignmentToken(assignmentSegment.getColumns().get(0).getStartIndex(), assignmentSegment.getStopIndex());
129         addCipherAssignment(schemaName, tableName, encryptColumn, assignmentSegment, result);
130         addAssistedQueryAssignment(schemaName, tableName, encryptColumn, assignmentSegment, result);
131         addLikeAssignment(schemaName, tableName, encryptColumn, assignmentSegment, result);
132         return result;
133     }
134     
135     private EncryptAssignmentToken generateValuesSQLToken(final EncryptTable encryptTable, final ColumnAssignmentSegment assignmentSegment, final FunctionSegment functionSegment) {
136         ColumnSegment columnSegment = assignmentSegment.getColumns().get(0);
137         String column = columnSegment.getIdentifier().getValue();
138         Optional<ExpressionSegment> valueColumnSegment = functionSegment.getParameters().stream().findFirst();
139         Preconditions.checkState(valueColumnSegment.isPresent());
140         String valueColumn = ((ColumnSegment) valueColumnSegment.get()).getIdentifier().getValue();
141         EncryptFunctionAssignmentToken result = new EncryptFunctionAssignmentToken(columnSegment.getStartIndex(), assignmentSegment.getStopIndex());
142         boolean isEncryptColumn = encryptTable.isEncryptColumn(column);
143         boolean isEncryptValueColumn = encryptTable.isEncryptColumn(valueColumn);
144         EncryptColumn encryptColumn = encryptTable.getEncryptColumn(column);
145         EncryptColumn encryptValueColumn = encryptTable.getEncryptColumn(column);
146         if (isEncryptColumn && isEncryptValueColumn) {
147             String cipherColumn = encryptColumn.getCipher().getName();
148             String cipherValueColumn = encryptValueColumn.getCipher().getName();
149             result.addAssignment(cipherColumn, "VALUES(" + cipherValueColumn + ")");
150         } else if (isEncryptColumn != isEncryptValueColumn) {
151             throw new UnsupportedEncryptSQLException(String.format("%s=VALUES(%s)", column, valueColumn));
152         }
153         Optional<AssistedQueryColumnItem> assistedQueryColumn = encryptColumn.getAssistedQuery();
154         Optional<AssistedQueryColumnItem> valueAssistedQueryColumn = encryptValueColumn.getAssistedQuery();
155         if (assistedQueryColumn.isPresent() && valueAssistedQueryColumn.isPresent()) {
156             result.addAssignment(assistedQueryColumn.get().getName(), "VALUES(" + valueAssistedQueryColumn.get().getName() + ")");
157         } else if (assistedQueryColumn.isPresent() != valueAssistedQueryColumn.isPresent()) {
158             throw new UnsupportedEncryptSQLException(String.format("%s=VALUES(%s)", column, valueColumn));
159         }
160         Optional<LikeQueryColumnItem> likeQueryColumn = encryptColumn.getLikeQuery();
161         Optional<LikeQueryColumnItem> valueLikeQueryColumn = encryptValueColumn.getLikeQuery();
162         if (likeQueryColumn.isPresent() && valueLikeQueryColumn.isPresent()) {
163             result.addAssignment(likeQueryColumn.get().getName(), "VALUES(" + valueLikeQueryColumn.get().getName() + ")");
164         } else if (likeQueryColumn.isPresent() != valueLikeQueryColumn.isPresent()) {
165             throw new UnsupportedEncryptSQLException(String.format("%s=VALUES(%s)", column, valueColumn));
166         }
167         if (result.isAssignmentsEmpty()) {
168             throw new UnsupportedEncryptSQLException(String.format("%s=VALUES(%s)", column, valueColumn));
169         }
170         return result;
171     }
172     
173     private void addCipherAssignment(final String schemaName, final String tableName, final EncryptColumn encryptColumn,
174                                      final ColumnAssignmentSegment assignmentSegment, final EncryptLiteralAssignmentToken token) {
175         Object originalValue = ((LiteralExpressionSegment) assignmentSegment.getValue()).getLiterals();
176         Object cipherValue = encryptColumn.getCipher().encrypt(databaseName, schemaName, tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue(),
177                 Collections.singletonList(originalValue)).iterator().next();
178         token.addAssignment(encryptColumn.getCipher().getName(), cipherValue);
179     }
180     
181     private void addAssistedQueryAssignment(final String schemaName, final String tableName, final EncryptColumn encryptColumn,
182                                             final ColumnAssignmentSegment assignmentSegment, final EncryptLiteralAssignmentToken token) {
183         encryptColumn.getAssistedQuery().ifPresent(optional -> {
184             Object originalValue = ((LiteralExpressionSegment) assignmentSegment.getValue()).getLiterals();
185             Object assistedQueryValue = optional.encrypt(
186                     databaseName, schemaName, tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue(), Collections.singletonList(originalValue)).iterator().next();
187             token.addAssignment(optional.getName(), assistedQueryValue);
188         });
189     }
190     
191     private void addLikeAssignment(final String schemaName, final String tableName, final EncryptColumn encryptColumn,
192                                    final ColumnAssignmentSegment assignmentSegment, final EncryptLiteralAssignmentToken token) {
193         encryptColumn.getLikeQuery().ifPresent(optional -> {
194             Object originalValue = ((LiteralExpressionSegment) assignmentSegment.getValue()).getLiterals();
195             Object likeValue = optional.encrypt(
196                     databaseName, schemaName, tableName, assignmentSegment.getColumns().get(0).getIdentifier().getValue(), Collections.singletonList(originalValue)).iterator().next();
197             token.addAssignment(optional.getName(), likeValue);
198         });
199     }
