18  package org.apache.shardingsphere.schedule.core.job.statistics.collect;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import lombok.extern.slf4j.Slf4j;
23  import org.apache.shardingsphere.elasticjob.api.JobConfiguration;
24  import org.apache.shardingsphere.elasticjob.lite.api.bootstrap.impl.ScheduleJobBootstrap;
25  import org.apache.shardingsphere.elasticjob.reg.base.CoordinatorRegistryCenter;
26  import org.apache.shardingsphere.elasticjob.reg.zookeeper.ZookeeperConfiguration;
27  import org.apache.shardingsphere.elasticjob.reg.zookeeper.ZookeeperRegistryCenter;
28  import org.apache.shardingsphere.infra.config.mode.ModeConfiguration;
29  import org.apache.shardingsphere.metadata.persist.node.ShardingSphereDataNode;
30  import org.apache.shardingsphere.mode.manager.ContextManager;
31  import org.apache.shardingsphere.mode.repository.cluster.ClusterPersistRepositoryConfiguration;
32  
33  import java.util.concurrent.atomic.AtomicBoolean;
34  
35  /**
36   * Statistics collect job worker.
37   */
38  @NoArgsConstructor(access = AccessLevel.PRIVATE)
39  @Slf4j
40  public final class StatisticsCollectJobWorker {
41      
42      private static final String JOB_NAME = "statistics-collect";
43      
44      private static final String CRON_EXPRESSION = "*/30 * * * * ?";
45      
46      private static final AtomicBoolean WORKER_INITIALIZED = new AtomicBoolean(false);
47      
48      /**
49       * Initialize job worker.
50       *
51       * @param contextManager context manager
52       */
53      public static void initialize(final ContextManager contextManager) {
54          if (WORKER_INITIALIZED.compareAndSet(false, true)) {
55              start(contextManager);
56          }
57      }
58      
59      private static void start(final ContextManager contextManager) {
60          ModeConfiguration modeConfig = contextManager.getComputeNodeInstanceContext().getModeConfiguration();
61          if ("ZooKeeper".equals(modeConfig.getRepository().getType())) {
62              ScheduleJobBootstrap bootstrap = new ScheduleJobBootstrap(createRegistryCenter(modeConfig), new StatisticsCollectJob(contextManager), createJobConfiguration());
63              bootstrap.schedule();
64              return;
65          }
66          log.warn("Can not collect statistics because of unsupported cluster type: {}", modeConfig.getRepository().getType());
67      }
68      
69      private static CoordinatorRegistryCenter createRegistryCenter(final ModeConfiguration modeConfig) {
70          ClusterPersistRepositoryConfiguration repositoryConfig = (ClusterPersistRepositoryConfiguration) modeConfig.getRepository();
71          String namespace = String.join("/", repositoryConfig.getNamespace(), ShardingSphereDataNode.getJobPath());
72          CoordinatorRegistryCenter result = new ZookeeperRegistryCenter(new ZookeeperConfiguration(repositoryConfig.getServerLists(), namespace));
73          result.init();
74          return result;
75      }
76      
77      private static JobConfiguration createJobConfiguration() {
78          return JobConfiguration.newBuilder(JOB_NAME, 1).cron(CRON_EXPRESSION).overwrite(true).build();
79      }
80  }
