39  /**
40   * Driver push down execute update executor.
41   */
42  public final class DriverPushDownExecuteUpdateExecutor {
43      
44      private final DriverJDBCPushDownExecuteUpdateExecutor jdbcPushDownExecutor;
45      
46      private final DriverRawPushDownExecuteUpdateExecutor rawPushDownExecutor;
47      
48      public DriverPushDownExecuteUpdateExecutor(final ShardingSphereConnection connection, final ShardingSphereMetaData metaData, final Grantee grantee,
49                                                 final JDBCExecutor jdbcExecutor, final RawExecutor rawExecutor) {
50          jdbcPushDownExecutor = new DriverJDBCPushDownExecuteUpdateExecutor(connection, metaData, grantee, jdbcExecutor);
51          rawPushDownExecutor = new DriverRawPushDownExecuteUpdateExecutor(connection, metaData, grantee, rawExecutor);
52      }
53      
54      /**
55       * Execute update.
56       *
57       * @param database database
58       * @param executionContext execution context
59       * @param prepareEngine prepare engine
60       * @param updateCallback statement execute update callback
61       * @param replayCallback statement replay callback
62       * @param addCallback statement add callback
63       * @return updated row count
64       * @throws SQLException SQL exception
65       */
66      @SuppressWarnings("rawtypes")
67      public int executeUpdate(final ShardingSphereDatabase database, final ExecutionContext executionContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
68                               final StatementExecuteUpdateCallback updateCallback, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
69          return database.getRuleMetaData().getAttributes(RawExecutionRuleAttribute.class).isEmpty()
70                  ? jdbcPushDownExecutor.executeUpdate(database, executionContext, prepareEngine, updateCallback, addCallback, replayCallback)
71                  : rawPushDownExecutor.executeUpdate(database, executionContext);
72      }
73  }
