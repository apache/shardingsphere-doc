18  package org.apache.shardingsphere.infra.rewrite.sql.token.pojo.generic;
19  
20  import org.apache.shardingsphere.infra.binder.segment.select.projection.Projection;
21  import org.apache.shardingsphere.infra.binder.segment.select.projection.impl.ColumnProjection;
22  import org.apache.shardingsphere.infra.binder.segment.select.projection.impl.SubqueryProjection;
23  import org.apache.shardingsphere.infra.database.type.dialect.OracleDatabaseType;
24  import org.apache.shardingsphere.infra.route.context.RouteUnit;
25  import org.apache.shardingsphere.sql.parser.sql.common.enums.QuoteCharacter;
26  import org.apache.shardingsphere.sql.parser.sql.common.value.identifier.IdentifierValue;
27  import org.junit.jupiter.api.Test;
28  
29  import java.util.Arrays;
30  import java.util.Collection;
31  import java.util.Collections;
32  
33  import static org.hamcrest.CoreMatchers.is;
34  import static org.hamcrest.MatcherAssert.assertThat;
35  import static org.mockito.Mockito.mock;
36  
37  class SubstitutableColumnNameTokenTest {
38      
39      @Test
40      void assertToString() {
41          Collection<Projection> projections = Collections.singletonList(new ColumnProjection(null, "id", null));
42          assertThat(new SubstitutableColumnNameToken(0, 1, projections).toString(mock(RouteUnit.class)), is("id"));
43      }
44      
45      @Test
46      void assertToStringWithQuote() {
47          Collection<Projection> projections = Collections.singletonList(new ColumnProjection(null,
48                  new IdentifierValue("id", QuoteCharacter.BACK_QUOTE), new IdentifierValue("id", QuoteCharacter.BACK_QUOTE)));
49          assertThat(new SubstitutableColumnNameToken(0, 1, projections, QuoteCharacter.BACK_QUOTE).toString(mock(RouteUnit.class)), is("`id` AS `id`"));
50      }
51      
52      @Test
53      void assertToStringWithOwnerQuote() {
54          Collection<Projection> projectionsWithOwnerQuote = Collections.singletonList(new ColumnProjection(new IdentifierValue("temp", QuoteCharacter.BACK_QUOTE),
55                  new IdentifierValue("id", QuoteCharacter.BACK_QUOTE), new IdentifierValue("id", QuoteCharacter.BACK_QUOTE)));
56          assertThat(new SubstitutableColumnNameToken(0, 1, projectionsWithOwnerQuote, QuoteCharacter.BACK_QUOTE).toString(mock(RouteUnit.class)), is("`temp`.`id` AS `id`"));
57          Collection<Projection> projectionsWithoutOwnerQuote = Collections.singletonList(new ColumnProjection(new IdentifierValue("temp", QuoteCharacter.NONE),
58                  new IdentifierValue("id", QuoteCharacter.BACK_QUOTE), new IdentifierValue("id", QuoteCharacter.BACK_QUOTE)));
59          assertThat(new SubstitutableColumnNameToken(0, 1, projectionsWithoutOwnerQuote, QuoteCharacter.BACK_QUOTE).toString(mock(RouteUnit.class)), is("temp.`id` AS `id`"));
60      }
61      
62      @Test
63      void assertToStringWithSubqueryProjection() {
64          Collection<Projection> projections = Arrays.asList(new ColumnProjection(new IdentifierValue("temp", QuoteCharacter.BACK_QUOTE),
65                  new IdentifierValue("id", QuoteCharacter.BACK_QUOTE), new IdentifierValue("id", QuoteCharacter.BACK_QUOTE)),
66                  new SubqueryProjection("(SELECT name FROM t_order)", new ColumnProjection(null, "name", null), "name", new OracleDatabaseType()));
67          assertThat(new SubstitutableColumnNameToken(0, 1, projections, QuoteCharacter.BACK_QUOTE).toString(mock(RouteUnit.class)),
68                  is("`temp`.`id` AS `id`, `name`"));
69      }
70  }
