18  package org.apache.shardingsphere.mode.persist;
19  
20  import lombok.Getter;
21  import org.apache.shardingsphere.infra.config.mode.ModeConfiguration;
22  import org.apache.shardingsphere.infra.spi.type.typed.TypedSPILoader;
23  import org.apache.shardingsphere.metadata.persist.MetaDataPersistService;
24  import org.apache.shardingsphere.mode.metadata.MetaDataContextManager;
25  import org.apache.shardingsphere.mode.persist.service.ComputeNodePersistService;
26  import org.apache.shardingsphere.mode.persist.service.ListenerAssistedPersistService;
27  import org.apache.shardingsphere.mode.persist.service.MetaDataManagerPersistService;
28  import org.apache.shardingsphere.mode.persist.service.PersistServiceBuilder;
29  import org.apache.shardingsphere.mode.persist.service.ProcessPersistService;
30  import org.apache.shardingsphere.mode.persist.service.QualifiedDataSourceStatePersistService;
31  import org.apache.shardingsphere.mode.persist.service.StatePersistService;
32  import org.apache.shardingsphere.mode.spi.PersistRepository;
33  
34  /**
35   * Persist service facade.
36   */
37  @Getter
38  public final class PersistServiceFacade {
39      
40      private final PersistRepository repository;
41      
42      private final MetaDataPersistService metaDataPersistService;
43      
44      private final ComputeNodePersistService computeNodePersistService;
45      
46      private final StatePersistService statePersistService;
47      
48      private final MetaDataManagerPersistService metaDataManagerPersistService;
49      
50      private final ProcessPersistService processPersistService;
51      
52      private final ListenerAssistedPersistService listenerAssistedPersistService;
53      
54      private final QualifiedDataSourceStatePersistService qualifiedDataSourceStatePersistService;
55      
56      public PersistServiceFacade(final PersistRepository repository, final ModeConfiguration modeConfiguration, final MetaDataContextManager metaDataContextManager) {
57          this.repository = repository;
58          metaDataPersistService = new MetaDataPersistService(repository);
59          computeNodePersistService = new ComputeNodePersistService(repository);
60          statePersistService = new StatePersistService(repository);
61          qualifiedDataSourceStatePersistService = new QualifiedDataSourceStatePersistService(repository);
62          PersistServiceBuilder persistServiceBuilder = TypedSPILoader.getService(PersistServiceBuilder.class, modeConfiguration.getType());
63          metaDataManagerPersistService = persistServiceBuilder.buildMetaDataManagerPersistService(repository, metaDataContextManager);
64          processPersistService = persistServiceBuilder.buildProcessPersistService(repository);
65          listenerAssistedPersistService = new ListenerAssistedPersistService(repository);
66      }
67  }
