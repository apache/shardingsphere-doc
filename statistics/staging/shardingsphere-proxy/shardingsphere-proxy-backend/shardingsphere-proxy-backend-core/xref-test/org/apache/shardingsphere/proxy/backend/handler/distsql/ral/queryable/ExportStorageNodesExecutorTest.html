75      @Test
76      void assertGetColumns() {
77          Collection<String> columns = new ExportStorageNodesExecutor().getColumnNames();
78          assertThat(columns.size(), is(3));
79          Iterator<String> columnIterator = columns.iterator();
80          assertThat(columnIterator.next(), is("id"));
81          assertThat(columnIterator.next(), is("create_time"));
82          assertThat(columnIterator.next(), is("storage_nodes"));
83      }
84      
85      @Test
86      void assertExecuteWithWrongDatabaseName() {
87          ContextManager contextManager = mockEmptyContextManager();
88          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
89          when(ProxyContext.getInstance().getAllDatabaseNames()).thenReturn(Collections.singleton("empty_metadata"));
90          ExportStorageNodesStatement sqlStatement = new ExportStorageNodesStatement("foo", null);
91          assertThrows(IllegalArgumentException.class, () -> new ExportStorageNodesExecutor().getRows(contextManager.getMetaDataContexts().getMetaData(), sqlStatement));
92      }
93      
94      @Test
95      void assertExecuteWithEmptyMetaData() {
96          ContextManager contextManager = mockEmptyContextManager();
97          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
98          when(ProxyContext.getInstance().getAllDatabaseNames()).thenReturn(Collections.singleton("empty_metadata"));
99          ExportStorageNodesStatement sqlStatement = new ExportStorageNodesStatement(null, null);
100         Collection<LocalDataQueryResultRow> actual = new ExportStorageNodesExecutor().getRows(contextManager.getMetaDataContexts().getMetaData(), sqlStatement);
101         assertThat(actual.size(), is(1));
102         LocalDataQueryResultRow row = actual.iterator().next();
103         assertThat(row.getCell(3), is("{\"storage_nodes\":{}}"));
104     }

106     private ContextManager mockEmptyContextManager() {
107         ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
108         MetaDataContexts metaDataContexts = new MetaDataContexts(mock(MetaDataPersistService.class), new ShardingSphereMetaData(new HashMap<>(),
109                 new ShardingSphereRuleMetaData(Collections.emptyList()), new ConfigurationProperties(new Properties())));
110         when(result.getMetaDataContexts()).thenReturn(metaDataContexts);
111         return result;
112     }
113     
114     @Test
115     void assertExecute() throws SQLException {
116         when(database.getName()).thenReturn("normal_db");
117         when(database.getResourceMetaData().getDataSources()).thenReturn(createDataSourceMap());
118         when(database.getRuleMetaData().getConfigurations()).thenReturn(Collections.singleton(createShardingRuleConfiguration()));
119         ContextManager contextManager = mockContextManager();
120         when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
121         Collection<LocalDataQueryResultRow> actual = new ExportStorageNodesExecutor().getRows(contextManager.getMetaDataContexts().getMetaData(), new ExportStorageNodesStatement(null, null));
122         assertThat(actual.size(), is(1));
123         LocalDataQueryResultRow row = actual.iterator().next();
124         assertThat(row.getCell(3), is(loadExpectedRow()));
125     }
126     
127     @Test
128     void assertExecuteWithDatabaseName() {
129         when(database.getName()).thenReturn("normal_db");
130         when(database.getResourceMetaData().getDataSources()).thenReturn(createDataSourceMap());
131         when(database.getRuleMetaData().getConfigurations()).thenReturn(Collections.singleton(createShardingRuleConfiguration()));
132         ContextManager contextManager = mockContextManager();
133         when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
134         Collection<LocalDataQueryResultRow> actual = new ExportStorageNodesExecutor().getRows(contextManager.getMetaDataContexts().getMetaData(), new ExportStorageNodesStatement("normal_db", null));
135         assertThat(actual.size(), is(1));
136         LocalDataQueryResultRow row = actual.iterator().next();
137         assertThat(row.getCell(3), is(loadExpectedRow()));
138     }

140     private ContextManager mockContextManager() {
141         ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
142         MetaDataContexts metaDataContexts = new MetaDataContexts(mock(MetaDataPersistService.class), new ShardingSphereMetaData(Collections.singletonMap(database.getName(), database),
143                 new ShardingSphereRuleMetaData(Collections.singleton(new AuthorityRule(new DefaultAuthorityRuleConfigurationBuilder().build(), Collections.emptyMap()))),
144                 new ConfigurationProperties(PropertiesBuilder.build(new Property(ConfigurationPropertyKey.SQL_SHOW.getKey(), "true")))));
145         when(result.getMetaDataContexts()).thenReturn(metaDataContexts);
146         return result;
147     }

149     private Map<String, DataSource> createDataSourceMap() {
150         Map<String, DataSource> result = new LinkedHashMap<>(2, 1);
151         result.put("ds_0", createDataSource("demo_ds_0"));
152         result.put("ds_1", createDataSource("demo_ds_1"));
153         return result;
154     }
155     
156     private DataSource createDataSource(final String name) {
157         MockedDataSource result = new MockedDataSource();
158         result.setUrl(String.format("jdbc:mock://127.0.0.1/%s", name));
159         result.setUsername("root");
160         result.setPassword("test");
161         result.setMaxPoolSize(50);
162         result.setMinPoolSize(1);
163         return result;
164     }
165     
166     private ShardingRuleConfiguration createShardingRuleConfiguration() {
167         ShardingRuleConfiguration result = new ShardingRuleConfiguration();
168         result.getTables().add(createTableRuleConfiguration());
169         result.setDefaultDatabaseShardingStrategy(new StandardShardingStrategyConfiguration("order_id", "ds_inline"));
170         result.setDefaultTableShardingStrategy(new NoneShardingStrategyConfiguration());
171         result.getKeyGenerators().put("snowflake", new AlgorithmConfiguration("SNOWFLAKE", new Properties()));
172         result.getShardingAlgorithms().put("ds_inline", new AlgorithmConfiguration("INLINE", PropertiesBuilder.build(new Property("algorithm-expression", "ds_${order_id % 2}"))));
173         return result;
174     }
175     
176     private ShardingTableRuleConfiguration createTableRuleConfiguration() {
177         ShardingTableRuleConfiguration result = new ShardingTableRuleConfiguration("t_order", "ds_${0..1}.t_order_${0..1}");
178         result.setKeyGenerateStrategy(new KeyGenerateStrategyConfiguration("order_id", "snowflake"));
179         return result;
180     }
181     
182     @SneakyThrows(IOException.class)
183     private String loadExpectedRow() {
184         StringBuilder result = new StringBuilder();
185         String fileName = Objects.requireNonNull(ExportStorageNodesExecutorTest.class.getResource("/expected/export-storage-nodes.json")).getFile();
186         try (
187                 FileReader fileReader = new FileReader(fileName);
188                 BufferedReader reader = new BufferedReader(fileReader)) {
189             String line;
190             while (null != (line = reader.readLine())) {
191                 result.append(line);
192             }
193         }
194         return result.toString();
195     }
