39      @Test
40      void assertSQLHintWriteRouteOnlyWithCommentString() {

43      }

54          assertTrue(actual.isSkipMetadataValidate());

57      @Test
58      void assertSQLHintDisableAuditNames() {
59          HintValueContext actual = SQLHintUtils.extractHint("/* SHARDINGSPHERE_HINT: DISABLE_AUDIT_NAMES=sharding_audit1 sharding_audit2 */");
60          assertThat(actual.getDisableAuditNames().size(), is(2));
61          assertTrue(actual.getDisableAuditNames().containsAll(Arrays.asList("sharding_audit1", "sharding_audit2")));
62      }

64      @Test
65      void assertSQLHintShardingDatabaseValue() {
66          HintValueContext actual = SQLHintUtils.extractHint("/* SHARDINGSPHERE_HINT: SHARDING_DATABASE_VALUE=10 */");
67          assertThat(actual.getHintShardingDatabaseValue("t_order"), is(Collections.singletonList(new BigInteger("10"))));
68      }

106     @ParameterizedTest(name = "extractHintFormat:{0}")
107     @ArgumentsSource(ExtractHintTestCaseArgumentsProvider.class)
108     void assertExtractHintFormat(@SuppressWarnings("unused") final String name, final String actualSQL, final boolean found) {
109         HintValueContext actual = SQLHintUtils.extractHint(actualSQL);
110         if (found) {
111             assertTrue(actual.findHintDataSourceName().isPresent());
112             assertThat(actual.findHintDataSourceName().get(), is("foo_ds"));
113         } else {
114             assertFalse(actual.findHintDataSourceName().isPresent());
115         }
116     }
117     
118     @ParameterizedTest(name = "extractHintFormat:{0}")
119     @ArgumentsSource(RemoveHintTestCaseArgumentsProvider.class)
120     void assertRemoveHint(@SuppressWarnings("unused") final String name, final String actualSQL, final String expectedSQL) {
121         assertThat(SQLHintUtils.removeHint(actualSQL), is(expectedSQL));
122     }

134         }
