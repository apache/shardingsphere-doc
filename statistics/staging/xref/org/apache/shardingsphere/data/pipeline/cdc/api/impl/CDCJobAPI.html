112     /**
113      * Create CDC job config.
114      *
115      * @param param create CDC job param
116      * @param sinkType sink type
117      * @param sinkProps sink properties
118      * @return job id
119      */
120     public String createJob(final StreamDataParameter param, final CDCSinkType sinkType, final Properties sinkProps) {
121         PipelineContextKey contextKey = PipelineContextKey.buildForProxy(param.getDatabaseName());
122         YamlCDCJobConfiguration yamlJobConfig = getYamlCDCJobConfiguration(param, sinkType, sinkProps, contextKey);
123         extendYamlJobConfiguration(contextKey, yamlJobConfig);
124         CDCJobConfiguration jobConfig = new YamlCDCJobConfigurationSwapper().swapToObject(yamlJobConfig);
125         ShardingSpherePreconditions.checkState(0 != jobConfig.getJobShardingCount(), () -> new PipelineJobCreationWithInvalidShardingCountException(jobConfig.getJobId()));
126         GovernanceRepositoryAPI repositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobConfig.getJobId()));
127         String jobConfigKey = PipelineMetaDataNode.getJobConfigPath(jobConfig.getJobId());
128         if (repositoryAPI.isExisted(jobConfigKey)) {
129             log.warn("CDC job already exists in registry center, ignore, jobConfigKey={}", jobConfigKey);
130         } else {
131             repositoryAPI.persist(PipelineMetaDataNode.getJobRootPath(jobConfig.getJobId()), getJobClassName());
132             JobConfigurationPOJO jobConfigPOJO = convertJobConfiguration(jobConfig);
133             jobConfigPOJO.setDisabled(true);
134             repositoryAPI.persist(jobConfigKey, YamlEngine.marshal(jobConfigPOJO));
135             if (!param.isFull()) {
136                 initIncrementalPosition(jobConfig);
137             }
138         }
139         return jobConfig.getJobId();
140     }

142     private YamlCDCJobConfiguration getYamlCDCJobConfiguration(final StreamDataParameter param, final CDCSinkType sinkType, final Properties sinkProps, final PipelineContextKey contextKey) {
143         YamlCDCJobConfiguration result = new YamlCDCJobConfiguration();
144         result.setDatabaseName(param.getDatabaseName());
145         result.setSchemaTableNames(param.getSchemaTableNames());
146         result.setFull(param.isFull());
147         result.setDecodeWithTX(param.isDecodeWithTX());
148         YamlSinkConfiguration sinkConfig = new YamlSinkConfiguration();
149         sinkConfig.setSinkType(sinkType.name());
150         sinkConfig.setProps(sinkProps);
151         result.setSinkConfig(sinkConfig);
152         ShardingSphereDatabase database = PipelineContextManager.getContext(contextKey).getContextManager().getMetaDataContexts().getMetaData().getDatabase(param.getDatabaseName());
153         result.setDataSourceConfiguration(pipelineDataSourceConfigSwapper.swapToYamlConfiguration(getDataSourceConfiguration(database)));
154         List<JobDataNodeLine> jobDataNodeLines = JobDataNodeLineConvertUtils.convertDataNodesToLines(param.getDataNodesMap());
155         result.setJobShardingDataNodes(jobDataNodeLines.stream().map(JobDataNodeLine::marshal).collect(Collectors.toList()));
156         JobDataNodeLine tableFirstDataNodes = new JobDataNodeLine(param.getDataNodesMap().entrySet().stream()
157                 .map(each -> new JobDataNodeEntry(each.getKey(), each.getValue().subList(0, 1))).collect(Collectors.toList()));
158         result.setTablesFirstDataNodes(tableFirstDataNodes.marshal());
159         return result;
160     }

162     private ShardingSpherePipelineDataSourceConfiguration getDataSourceConfiguration(final ShardingSphereDatabase database) {
163         Map<String, Map<String, Object>> dataSourceProps = new HashMap<>();
164         for (Entry<String, DataSourceProperties> entry : database.getResourceMetaData().getDataSourcePropsMap().entrySet()) {
165             dataSourceProps.put(entry.getKey(), dataSourceConfigSwapper.swapToMap(entry.getValue()));
166         }
167         YamlRootConfiguration targetRootConfig = new YamlRootConfiguration();
168         targetRootConfig.setDatabaseName(database.getName());
169         targetRootConfig.setDataSources(dataSourceProps);
170         Collection<YamlRuleConfiguration> yamlRuleConfigurations = ruleConfigSwapperEngine.swapToYamlRuleConfigurations(database.getRuleMetaData().getConfigurations());
171         targetRootConfig.setRules(yamlRuleConfigurations);
172         return new ShardingSpherePipelineDataSourceConfiguration(targetRootConfig);
173     }

175     private void initIncrementalPosition(final CDCJobConfiguration jobConfig) {
176         String jobId = jobConfig.getJobId();
177         try (PipelineDataSourceManager pipelineDataSourceManager = new DefaultPipelineDataSourceManager()) {
178             for (int i = 0; i < jobConfig.getJobShardingCount(); i++) {
179                 if (getJobItemProgress(jobId, i).isPresent()) {
180                     continue;
181                 }
182                 DumperConfiguration dumperConfig = buildDumperConfiguration(jobConfig, i, getTableNameSchemaNameMapping(jobConfig.getSchemaTableNames()));
183                 InventoryIncrementalJobItemProgress jobItemProgress = getInventoryIncrementalJobItemProgress(jobConfig, pipelineDataSourceManager, dumperConfig);
184                 PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).persistJobItemProgress(
185                         jobId, i, YamlEngine.marshal(getJobItemProgressSwapper().swapToYamlConfiguration(jobItemProgress)));
186             }
187         } catch (final SQLException ex) {
188             throw new PrepareJobWithGetBinlogPositionException(jobId, ex);
189         }
190     }

192     private static InventoryIncrementalJobItemProgress getInventoryIncrementalJobItemProgress(final CDCJobConfiguration jobConfig,
193                                                                                               final PipelineDataSourceManager dataSourceManager,
194                                                                                               final DumperConfiguration dumperConfig) throws SQLException {
195         InventoryIncrementalJobItemProgress result = new InventoryIncrementalJobItemProgress();
196         result.setSourceDatabaseType(jobConfig.getSourceDatabaseType());
197         result.setDataSourceName(dumperConfig.getDataSourceName());
198         IncrementalTaskProgress incrementalTaskProgress = new IncrementalTaskProgress(PipelineJobPreparerUtils.getIncrementalPosition(null, dumperConfig, dataSourceManager));
199         result.setIncremental(new JobItemIncrementalTasksProgress(incrementalTaskProgress));
200         return result;
201     }

210     /**
211      * Start job.
212      *
213      * @param jobId job id
214      * @param sink sink
215      */
216     public void startJob(final String jobId, final PipelineSink sink) {
217         CDCJob job = new CDCJob(jobId, sink);
218         PipelineJobCenter.addJob(jobId, job);
219         updateJobConfigurationDisabled(jobId, false);
220         JobConfigurationPOJO jobConfigPOJO = getElasticJobConfigPOJO(jobId);
221         OneOffJobBootstrap oneOffJobBootstrap = new OneOffJobBootstrap(PipelineAPIFactory.getRegistryCenter(PipelineJobIdUtils.parseContextKey(jobId)), job, jobConfigPOJO.toJobConfiguration());
222         job.setJobBootstrap(oneOffJobBootstrap);
223         oneOffJobBootstrap.execute();
224     }

226     /**
227      * Update job configuration disabled.
228      *
229      * @param jobId job id
230      * @param disabled disabled
231      */
232     public void updateJobConfigurationDisabled(final String jobId, final boolean disabled) {
233         JobConfigurationPOJO jobConfigPOJO = getElasticJobConfigPOJO(jobId);
234         jobConfigPOJO.setDisabled(disabled);
235         if (disabled) {
236             jobConfigPOJO.getProps().setProperty("stop_time_millis", String.valueOf(System.currentTimeMillis()));
237         } else {
238             jobConfigPOJO.getProps().setProperty("start_time_millis", String.valueOf(System.currentTimeMillis()));
239             jobConfigPOJO.getProps().remove("stop_time_millis");
240         }
241         PipelineAPIFactory.getJobConfigurationAPI(PipelineJobIdUtils.parseContextKey(jobConfigPOJO.getJobName())).updateJobConfiguration(jobConfigPOJO);
242     }

244     @Override
245     public void extendYamlJobConfiguration(final PipelineContextKey contextKey, final YamlPipelineJobConfiguration yamlJobConfig) {
246         YamlCDCJobConfiguration config = (YamlCDCJobConfiguration) yamlJobConfig;
247         if (null == yamlJobConfig.getJobId()) {
248             config.setJobId(generateJobId(contextKey, config));
249         }
250         if (Strings.isNullOrEmpty(config.getSourceDatabaseType())) {
251             PipelineDataSourceConfiguration sourceDataSourceConfig = PipelineDataSourceConfigurationFactory.newInstance(config.getDataSourceConfiguration().getType(),
252                     config.getDataSourceConfiguration().getParameter());
253             config.setSourceDatabaseType(sourceDataSourceConfig.getDatabaseType().getType());
254         }
255     }

257     private String generateJobId(final PipelineContextKey contextKey, final YamlCDCJobConfiguration config) {
258         CDCJobId jobId = new CDCJobId(contextKey, config.getSchemaTableNames(), config.isFull(), config.getSinkConfig().getSinkType());
259         return marshalJobId(jobId);
260     }

262     @Override
263     protected String marshalJobIdLeftPart(final PipelineJobId pipelineJobId) {
264         CDCJobId jobId = (CDCJobId) pipelineJobId;
265         String text = Joiner.on('|').join(jobId.getContextKey().getDatabaseName(), jobId.getSchemaTableNames(), jobId.isFull());
266         return DigestUtils.md5Hex(text.getBytes(StandardCharsets.UTF_8));
267     }

269     @Override
270     public CDCTaskConfiguration buildTaskConfiguration(final PipelineJobConfiguration pipelineJobConfig, final int jobShardingItem, final PipelineProcessConfiguration pipelineProcessConfig) {
271         CDCJobConfiguration jobConfig = (CDCJobConfiguration) pipelineJobConfig;
272         TableNameSchemaNameMapping tableNameSchemaNameMapping = getTableNameSchemaNameMapping(jobConfig.getSchemaTableNames());
273         DumperConfiguration dumperConfig = buildDumperConfiguration(jobConfig, jobShardingItem, tableNameSchemaNameMapping);
274         ImporterConfiguration importerConfig = buildImporterConfiguration(jobConfig, pipelineProcessConfig, jobConfig.getSchemaTableNames(), tableNameSchemaNameMapping);
275         CDCTaskConfiguration result = new CDCTaskConfiguration(dumperConfig, importerConfig);
276         log.debug("buildTaskConfiguration, result={}", result);
277         return result;
278     }

280     private TableNameSchemaNameMapping getTableNameSchemaNameMapping(final Collection<String> tableNames) {
281         Map<String, String> tableNameSchemaMap = new LinkedHashMap<>();
282         for (String each : tableNames) {
283             String[] split = each.split("\\.");
284             if (split.length > 1) {
285                 tableNameSchemaMap.put(split[1], split[0]);
286             }
287         }
288         return new TableNameSchemaNameMapping(tableNameSchemaMap);
289     }
290     
291     private DumperConfiguration buildDumperConfiguration(final CDCJobConfiguration jobConfig, final int jobShardingItem, final TableNameSchemaNameMapping tableNameSchemaNameMapping) {
292         JobDataNodeLine dataNodeLine = jobConfig.getJobShardingDataNodes().get(jobShardingItem);
293         Map<ActualTableName, LogicTableName> tableNameMap = new LinkedHashMap<>();
294         dataNodeLine.getEntries().forEach(each -> each.getDataNodes().forEach(node -> tableNameMap.put(new ActualTableName(node.getTableName()), new LogicTableName(each.getLogicTableName()))));
295         String dataSourceName = dataNodeLine.getEntries().iterator().next().getDataNodes().iterator().next().getDataSourceName();
296         StandardPipelineDataSourceConfiguration actualDataSourceConfig = jobConfig.getDataSourceConfig().getActualDataSourceConfiguration(dataSourceName);
297         DumperConfiguration result = new DumperConfiguration();
298         result.setJobId(jobConfig.getJobId());
299         result.setDataSourceName(dataSourceName);
300         result.setDataSourceConfig(actualDataSourceConfig);
301         result.setTableNameMap(tableNameMap);
302         result.setTableNameSchemaNameMapping(tableNameSchemaNameMapping);
303         result.setDecodeWithTX(jobConfig.isDecodeWithTX());
304         return result;
305     }

307     private ImporterConfiguration buildImporterConfiguration(final CDCJobConfiguration jobConfig, final PipelineProcessConfiguration pipelineProcessConfig, final Collection<String> schemaTableNames,
308                                                              final TableNameSchemaNameMapping tableNameSchemaNameMapping) {
309         PipelineDataSourceConfiguration dataSourceConfig = PipelineDataSourceConfigurationFactory.newInstance(jobConfig.getDataSourceConfig().getType(),
310                 jobConfig.getDataSourceConfig().getParameter());
311         CDCProcessContext processContext = new CDCProcessContext(jobConfig.getJobId(), pipelineProcessConfig);
312         JobRateLimitAlgorithm writeRateLimitAlgorithm = processContext.getWriteRateLimitAlgorithm();
313         int batchSize = pipelineProcessConfig.getWrite().getBatchSize();
314         Map<LogicTableName, Set<String>> shardingColumnsMap = new ShardingColumnsExtractor()
315                 .getShardingColumnsMap(jobConfig.getDataSourceConfig().getRootConfig().getRules(), schemaTableNames.stream().map(LogicTableName::new).collect(Collectors.toSet()));
316         return new ImporterConfiguration(dataSourceConfig, shardingColumnsMap, tableNameSchemaNameMapping, batchSize, writeRateLimitAlgorithm, 0, 1);
317     }

353      *
354      * @param jobId job id
355      */
356     public void stopAndDrop(final String jobId) {
357         CDCJobConfiguration jobConfig = getJobConfiguration(jobId);
358         if (CDCSinkType.SOCKET == jobConfig.getSinkConfig().getSinkType()) {
359             PipelineJobCenter.stop(jobId);
360         } else {
361             stop(jobId);
362         }
363         dropJob(jobId);
364     }
