138     private YamlMigrationJobConfiguration buildYamlJobConfiguration(final PipelineContextKey contextKey, final MigrateTableStatement param) {
139         YamlMigrationJobConfiguration result = new YamlMigrationJobConfiguration();
140         result.setTargetDatabaseName(param.getTargetDatabaseName());
141         Map<String, DataSourceProperties> metaDataDataSource = dataSourcePersistService.load(contextKey, new MigrationJobType());
142         Map<String, List<DataNode>> sourceDataNodes = new LinkedHashMap<>();
143         Map<String, YamlPipelineDataSourceConfiguration> configSources = new LinkedHashMap<>();
144         List<SourceTargetEntry> sourceTargetEntries = new ArrayList<>(new HashSet<>(param.getSourceTargetEntries())).stream().sorted(Comparator.comparing(SourceTargetEntry::getTargetTableName)
145                 .thenComparing(each -> DataNodeUtils.formatWithSchema(each.getSource()))).collect(Collectors.toList());
146         YamlDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlDataSourceConfigurationSwapper();
147         for (SourceTargetEntry each : sourceTargetEntries) {
148             sourceDataNodes.computeIfAbsent(each.getTargetTableName(), key -> new LinkedList<>()).add(each.getSource());
149             ShardingSpherePreconditions.checkState(1 == sourceDataNodes.get(each.getTargetTableName()).size(),
150                     () -> new PipelineInvalidParameterException("more than one source table for " + each.getTargetTableName()));
151             String dataSourceName = each.getSource().getDataSourceName();
152             if (configSources.containsKey(dataSourceName)) {
153                 continue;
154             }
155             ShardingSpherePreconditions.checkState(metaDataDataSource.containsKey(dataSourceName),
156                     () -> new PipelineInvalidParameterException(dataSourceName + " doesn't exist. Run `SHOW MIGRATION SOURCE STORAGE UNITS;` to verify it."));
157             Map<String, Object> sourceDataSourceProps = dataSourceConfigSwapper.swapToMap(metaDataDataSource.get(dataSourceName));
158             StandardPipelineDataSourceConfiguration sourceDataSourceConfig = new StandardPipelineDataSourceConfiguration(sourceDataSourceProps);
159             configSources.put(dataSourceName, buildYamlPipelineDataSourceConfiguration(sourceDataSourceConfig.getType(), sourceDataSourceConfig.getParameter()));
160             DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, sourceDataSourceConfig.getDatabaseType());
161             if (null == each.getSource().getSchemaName() && dialectDatabaseMetaData.isSchemaAvailable()) {
162                 each.getSource().setSchemaName(PipelineSchemaUtils.getDefaultSchema(sourceDataSourceConfig));
163             }
164             DatabaseType sourceDatabaseType = sourceDataSourceConfig.getDatabaseType();
165             if (null == result.getSourceDatabaseType()) {
166                 result.setSourceDatabaseType(sourceDatabaseType.getType());
167             } else if (!result.getSourceDatabaseType().equals(sourceDatabaseType.getType())) {
168                 throw new PipelineInvalidParameterException("Source storage units have different database types");
169             }
170         }
171         result.setSources(configSources);
172         ShardingSphereDatabase targetDatabase = PipelineContextManager.getProxyContext().getContextManager().getMetaDataContexts().getMetaData().getDatabase(param.getTargetDatabaseName());
173         PipelineDataSourceConfiguration targetPipelineDataSourceConfig = buildTargetPipelineDataSourceConfiguration(targetDatabase);
174         result.setTarget(buildYamlPipelineDataSourceConfiguration(targetPipelineDataSourceConfig.getType(), targetPipelineDataSourceConfig.getParameter()));
175         result.setTargetDatabaseType(targetPipelineDataSourceConfig.getDatabaseType().getType());
176         List<JobDataNodeEntry> tablesFirstDataNodes = sourceDataNodes.entrySet().stream()
177                 .map(entry -> new JobDataNodeEntry(entry.getKey(), entry.getValue().subList(0, 1))).collect(Collectors.toList());
178         result.setTargetTableNames(new ArrayList<>(sourceDataNodes.keySet()).stream().sorted().collect(Collectors.toList()));
179         result.setTargetTableSchemaMap(buildTargetTableSchemaMap(sourceDataNodes));
180         result.setTablesFirstDataNodes(new JobDataNodeLine(tablesFirstDataNodes).marshal());
181         result.setJobShardingDataNodes(JobDataNodeLineConvertUtils.convertDataNodesToLines(sourceDataNodes).stream().map(JobDataNodeLine::marshal).collect(Collectors.toList()));
182         extendYamlJobConfiguration(contextKey, result);
183         return result;
184     }

193     private PipelineDataSourceConfiguration buildTargetPipelineDataSourceConfiguration(final ShardingSphereDatabase targetDatabase) {
194         Map<String, Map<String, Object>> targetDataSourceProps = new HashMap<>();
195         YamlDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlDataSourceConfigurationSwapper();
196         for (Entry<String, DataSourceProperties> entry : targetDatabase.getResourceMetaData().getDataSourcePropsMap().entrySet()) {
197             Map<String, Object> dataSourceProps = dataSourceConfigSwapper.swapToMap(entry.getValue());
198             targetDataSourceProps.put(entry.getKey(), dataSourceProps);
199         }
200         YamlRootConfiguration targetRootConfig = buildYamlRootConfiguration(targetDatabase.getName(), targetDataSourceProps, targetDatabase.getRuleMetaData().getConfigurations());
201         return new ShardingSpherePipelineDataSourceConfiguration(targetRootConfig);
202     }

204     private YamlRootConfiguration buildYamlRootConfiguration(final String databaseName, final Map<String, Map<String, Object>> yamlDataSources, final Collection<RuleConfiguration> rules) {
205         if (rules.isEmpty()) {
206             throw new NoAnyRuleExistsException(databaseName);
207         }
208         YamlRootConfiguration result = new YamlRootConfiguration();
209         result.setDatabaseName(databaseName);
210         result.setDataSources(yamlDataSources);
211         Collection<YamlRuleConfiguration> yamlRuleConfigurations = new YamlRuleConfigurationSwapperEngine().swapToYamlRuleConfigurations(rules);
212         result.setRules(yamlRuleConfigurations);
213         return result;
214     }

217         Map<String, String> result = new LinkedHashMap<>();

219         return result;

266     @Override
267     public MigrationTaskConfiguration buildTaskConfiguration(final PipelineJobConfiguration pipelineJobConfig, final int jobShardingItem, final PipelineProcessConfiguration pipelineProcessConfig) {
268         MigrationJobConfiguration jobConfig = (MigrationJobConfiguration) pipelineJobConfig;
269         JobDataNodeLine dataNodeLine = jobConfig.getJobShardingDataNodes().get(jobShardingItem);
270         Map<ActualTableName, LogicTableName> tableNameMap = buildTableNameMap(dataNodeLine);
271         TableNameSchemaNameMapping tableNameSchemaNameMapping = new TableNameSchemaNameMapping(jobConfig.getTargetTableSchemaMap());
272         CreateTableConfiguration createTableConfig = buildCreateTableConfiguration(jobConfig, tableNameSchemaNameMapping);
273         String dataSourceName = dataNodeLine.getEntries().get(0).getDataNodes().get(0).getDataSourceName();
274         DumperConfiguration dumperConfig = buildDumperConfiguration(jobConfig.getJobId(), dataSourceName, jobConfig.getSources().get(dataSourceName), tableNameMap, tableNameSchemaNameMapping);
275         Set<LogicTableName> targetTableNames = jobConfig.getTargetTableNames().stream().map(LogicTableName::new).collect(Collectors.toSet());
276         Map<LogicTableName, Set<String>> shardingColumnsMap = new ShardingColumnsExtractor().getShardingColumnsMap(
277                 ((ShardingSpherePipelineDataSourceConfiguration) jobConfig.getTarget()).getRootConfig().getRules(), targetTableNames);
278         ImporterConfiguration importerConfig = buildImporterConfiguration(jobConfig, pipelineProcessConfig, shardingColumnsMap, tableNameSchemaNameMapping);
279         MigrationTaskConfiguration result = new MigrationTaskConfiguration(dataSourceName, createTableConfig, dumperConfig, importerConfig);
280         log.info("buildTaskConfiguration, result={}", result);
281         return result;
282     }

284     private Map<ActualTableName, LogicTableName> buildTableNameMap(final JobDataNodeLine dataNodeLine) {
285         Map<ActualTableName, LogicTableName> result = new LinkedHashMap<>();
286         for (JobDataNodeEntry each : dataNodeLine.getEntries()) {
287             for (DataNode dataNode : each.getDataNodes()) {
288                 result.put(new ActualTableName(dataNode.getTableName()), new LogicTableName(each.getLogicTableName()));
289             }
290         }
291         return result;
292     }

294     private CreateTableConfiguration buildCreateTableConfiguration(final MigrationJobConfiguration jobConfig,
295                                                                    final TableNameSchemaNameMapping tableNameSchemaNameMapping) {
296         Collection<CreateTableEntry> createTableEntries = new LinkedList<>();
297         for (JobDataNodeEntry each : jobConfig.getTablesFirstDataNodes().getEntries()) {
298             String sourceSchemaName = tableNameSchemaNameMapping.getSchemaName(each.getLogicTableName());
299             DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, jobConfig.getTargetDatabaseType());
300             String targetSchemaName = dialectDatabaseMetaData.isSchemaAvailable() ? sourceSchemaName : null;
301             DataNode dataNode = each.getDataNodes().get(0);
302             PipelineDataSourceConfiguration sourceDataSourceConfig = jobConfig.getSources().get(dataNode.getDataSourceName());
303             CreateTableEntry createTableEntry = new CreateTableEntry(
304                     sourceDataSourceConfig, new SchemaTableName(new SchemaName(sourceSchemaName), new TableName(dataNode.getTableName())),
305                     jobConfig.getTarget(), new SchemaTableName(new SchemaName(targetSchemaName), new TableName(each.getLogicTableName())));
306             createTableEntries.add(createTableEntry);
307         }
308         CreateTableConfiguration result = new CreateTableConfiguration(createTableEntries);
309         log.info("getCreateTableConfiguration, result={}", result);
310         return result;
311     }

374     @Override
375     public void rollback(final String jobId) throws SQLException {
376         final long startTimeMillis = System.currentTimeMillis();
377         stop(jobId);
378         dropCheckJobs(jobId);
379         cleanTempTableOnRollback(jobId);
380         dropJob(jobId);
381         log.info("Rollback job {} cost {} ms", jobId, System.currentTimeMillis() - startTimeMillis);
382     }
383     
384     private void dropCheckJobs(final String jobId) {
385         Collection<String> checkJobIds = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).listCheckJobIds(jobId);
386         if (checkJobIds.isEmpty()) {
387             return;
388         }
389         for (String each : checkJobIds) {
390             try {
391                 dropJob(each);
392                 // CHECKSTYLE:OFF
393             } catch (final RuntimeException ex) {
394                 // CHECKSTYLE:ON
395                 log.info("drop check job failed, check job id: {}, error: {}", each, ex.getMessage());
396             }
397         }
398     }
399     
400     private void cleanTempTableOnRollback(final String jobId) throws SQLException {
401         MigrationJobConfiguration jobConfig = getJobConfiguration(jobId);
402         PipelineCommonSQLBuilder pipelineSQLBuilder = new PipelineCommonSQLBuilder(jobConfig.getTargetDatabaseType());
403         TableNameSchemaNameMapping mapping = new TableNameSchemaNameMapping(jobConfig.getTargetTableSchemaMap());
404         try (
405                 PipelineDataSourceWrapper dataSource = PipelineDataSourceFactory.newInstance(jobConfig.getTarget());
406                 Connection connection = dataSource.getConnection()) {
407             for (String each : jobConfig.getTargetTableNames()) {
408                 String targetSchemaName = mapping.getSchemaName(each);
409                 String sql = pipelineSQLBuilder.buildDropSQL(targetSchemaName, each);
410                 log.info("cleanTempTableOnRollback, targetSchemaName={}, targetTableName={}, sql={}", targetSchemaName, each, sql);
411                 try (Statement statement = connection.createStatement()) {
412                     statement.execute(sql);
413                 }
414             }
415         }
416     }

418     @Override
419     public void commit(final String jobId) {
420         log.info("Commit job {}", jobId);
421         final long startTimeMillis = System.currentTimeMillis();
422         stop(jobId);
423         dropCheckJobs(jobId);
424         MigrationJobConfiguration jobConfig = getJobConfiguration(jobId);
425         refreshTableMetadata(jobId, jobConfig.getTargetDatabaseName());
426         dropJob(jobId);
427         log.info("Commit cost {} ms", System.currentTimeMillis() - startTimeMillis);
428     }

432      *
433      * @param contextKey context key
434      * @param dataSourcePropsMap data source properties map
435      */
436     public void addMigrationSourceResources(final PipelineContextKey contextKey, final Map<String, DataSourceProperties> dataSourcePropsMap) {
437         Map<String, DataSourceProperties> existDataSources = dataSourcePersistService.load(contextKey, getJobType());
438         Collection<String> duplicateDataSourceNames = new HashSet<>(dataSourcePropsMap.size(), 1F);
439         for (Entry<String, DataSourceProperties> entry : dataSourcePropsMap.entrySet()) {
440             if (existDataSources.containsKey(entry.getKey())) {
441                 duplicateDataSourceNames.add(entry.getKey());
442             }
443         }
444         ShardingSpherePreconditions.checkState(duplicateDataSourceNames.isEmpty(), () -> new RegisterMigrationSourceStorageUnitException(duplicateDataSourceNames));
445         Map<String, DataSourceProperties> result = new LinkedHashMap<>(existDataSources);
446         result.putAll(dataSourcePropsMap);
447         dataSourcePersistService.persist(contextKey, getJobType(), result);
448     }

450     /**
451      * Drop migration source resources.
452      *
453      * @param contextKey context key
454      * @param resourceNames resource names
455      */
456     public void dropMigrationSourceResources(final PipelineContextKey contextKey, final Collection<String> resourceNames) {
457         Map<String, DataSourceProperties> metaDataDataSource = dataSourcePersistService.load(contextKey, getJobType());
458         List<String> noExistResources = resourceNames.stream().filter(each -> !metaDataDataSource.containsKey(each)).collect(Collectors.toList());
459         ShardingSpherePreconditions.checkState(noExistResources.isEmpty(), () -> new UnregisterMigrationSourceStorageUnitException(noExistResources));
460         for (String each : resourceNames) {
461             metaDataDataSource.remove(each);
462         }
463         dataSourcePersistService.persist(contextKey, getJobType(), metaDataDataSource);
464     }

466     /**
467      * Query migration source resources list.
468      *
469      * @param contextKey context key
470      * @return migration source resources
471      */
472     public Collection<Collection<Object>> listMigrationSourceResources(final PipelineContextKey contextKey) {
473         Map<String, DataSourceProperties> dataSourcePropertiesMap = dataSourcePersistService.load(contextKey, getJobType());
474         Collection<Collection<Object>> result = new ArrayList<>(dataSourcePropertiesMap.size());
475         for (Entry<String, DataSourceProperties> entry : dataSourcePropertiesMap.entrySet()) {
476             String dataSourceName = entry.getKey();
477             DataSourceProperties value = entry.getValue();
478             Collection<Object> props = new LinkedList<>();
479             props.add(dataSourceName);
480             String url = String.valueOf(value.getConnectionPropertySynonyms().getStandardProperties().get("url"));
481             DatabaseType databaseType = DatabaseTypeFactory.get(url);
482             props.add(databaseType.getType());
483             ConnectionProperties connectionProps = DatabaseTypedSPILoader.getService(ConnectionPropertiesParser.class, databaseType).parse(url, "", null);
484             props.add(connectionProps.getHostname());
485             props.add(connectionProps.getPort());
486             props.add(connectionProps.getCatalog());
487             Map<String, Object> standardProps = value.getPoolPropertySynonyms().getStandardProperties();
488             props.add(getStandardProperty(standardProps, "connectionTimeoutMilliseconds"));
489             props.add(getStandardProperty(standardProps, "idleTimeoutMilliseconds"));
490             props.add(getStandardProperty(standardProps, "maxLifetimeMilliseconds"));
491             props.add(getStandardProperty(standardProps, "maxPoolSize"));
492             props.add(getStandardProperty(standardProps, "minPoolSize"));
493             props.add(getStandardProperty(standardProps, "readOnly"));
494             Map<String, Object> otherProps = value.getCustomDataSourceProperties().getProperties();
495             props.add(otherProps.isEmpty() ? "" : new Gson().toJson(otherProps));
496             result.add(props);
497         }
498         return result;
499     }
500     
501     private String getStandardProperty(final Map<String, Object> standardProps, final String key) {
502         if (standardProps.containsKey(key) && null != standardProps.get(key)) {
503             return standardProps.get(key).toString();
504         }
505         return "";
506     }
