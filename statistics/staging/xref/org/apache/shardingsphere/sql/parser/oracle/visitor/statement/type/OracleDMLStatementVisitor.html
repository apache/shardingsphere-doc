177     @Override
178     public ASTNode visitUpdate(final UpdateContext ctx) {
179         OracleUpdateStatement result = new OracleUpdateStatement();
180         result.setTable((TableSegment) visit(ctx.updateSpecification()));
181         if (null != ctx.alias()) {
182             result.getTable().setAlias((AliasSegment) visit(ctx.alias()));
183         }
184         result.setSetAssignment((SetAssignmentSegment) visit(ctx.updateSetClause()));
185         if (null != ctx.whereClause()) {
186             result.setWhere((WhereSegment) visit(ctx.whereClause()));
187         }
188         result.addParameterMarkerSegments(getParameterMarkerSegments());
189         return result;
190     }
191     
192     @Override
193     public ASTNode visitUpdateSpecification(final UpdateSpecificationContext ctx) {
194         if (null != ctx.dmlTableExprClause().dmlTableClause()) {
195             return visit(ctx.dmlTableExprClause().dmlTableClause());
196         }
197         if (null != ctx.dmlTableExprClause().dmlSubqueryClause()) {
198             SubquerySegment subquerySegment = (SubquerySegment) visit(ctx.dmlTableExprClause().dmlSubqueryClause());
199             return new SubqueryTableSegment(subquerySegment);
200         }
201         SubquerySegment subquerySegment = (SubquerySegment) visit(ctx.dmlTableExprClause().tableCollectionExpr());
202         return new SubqueryTableSegment(subquerySegment);
203     }
204     
205     @Override
206     public ASTNode visitUpdateSetClause(final UpdateSetClauseContext ctx) {
207         Collection<AssignmentSegment> assignments = new LinkedList<>();
208         if (null != ctx.updateSetColumnList()) {
209             for (UpdateSetColumnClauseContext each : ctx.updateSetColumnList().updateSetColumnClause()) {
210                 assignments.add((AssignmentSegment) visit(each));
211             }
212         } else {
213             assignments.add((AssignmentSegment) visit(ctx.updateSetValueClause()));
214         }
215         return new SetAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), assignments);
216     }
217     
218     @Override
219     public ASTNode visitUpdateSetColumnClause(final UpdateSetColumnClauseContext ctx) {
220         return 1 == ctx.columnName().size() ? createAssignmentSegmentFromSingleColumnAssignment(ctx) : createAssignmentSegmentFromMultiColumnAssignment(ctx);
221     }
222     
223     private AssignmentSegment createAssignmentSegmentFromSingleColumnAssignment(final UpdateSetColumnClauseContext ctx) {
224         ColumnSegment column = (ColumnSegment) visitColumnName(ctx.columnName(0));
225         List<ColumnSegment> columnSegments = new LinkedList<>();
226         columnSegments.add(column);
227         if (null != ctx.expr()) {
228             ExpressionSegment value = (ExpressionSegment) visit(ctx.expr());
229             return new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
230         }
231         if (null != ctx.selectSubquery()) {
232             SubquerySegment subquerySegment = new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(),
233                     (OracleSelectStatement) visit(ctx.selectSubquery()));
234             SubqueryExpressionSegment value = new SubqueryExpressionSegment(subquerySegment);
235             AssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
236             result.getColumns().add(column);
237             return result;
238         }
239         CommonExpressionSegment value = new CommonExpressionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), ctx.DEFAULT().getText());
240         AssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
241         result.getColumns().add(column);
242         return result;
243     }
244     
245     private AssignmentSegment createAssignmentSegmentFromMultiColumnAssignment(final UpdateSetColumnClauseContext ctx) {
246         List<ColumnSegment> columnSegments = new LinkedList<>();
247         for (ColumnNameContext each : ctx.columnName()) {
248             columnSegments.add((ColumnSegment) visit(each));
249         }
250         SubquerySegment subquerySegment =
251                 new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(), (OracleSelectStatement) visit(ctx.selectSubquery()));
252         SubqueryExpressionSegment value = new SubqueryExpressionSegment(subquerySegment);
253         return new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
254     }
255     
256     @Override
257     public ASTNode visitUpdateSetValueClause(final UpdateSetValueClauseContext ctx) {
258         ColumnSegment column = new ColumnSegment(ctx.alias().start.getStartIndex(), ctx.alias().stop.getStopIndex(), (IdentifierValue) visit(ctx.alias().identifier()));
259         List<ColumnSegment> columnSegments = new LinkedList<>();
260         columnSegments.add(column);
261         if (null != ctx.expr()) {
262             ExpressionSegment value = (ExpressionSegment) visit(ctx.expr());
263             AssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
264             result.getColumns().add(column);
265             return result;
266         } else {
267             SubquerySegment subquerySegment =
268                     new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(), (OracleSelectStatement) visit(ctx.selectSubquery()));
269             SubqueryExpressionSegment value = new SubqueryExpressionSegment(subquerySegment);
270             AssignmentSegment result = new ColumnAssignmentSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnSegments, value);
271             result.getColumns().add(column);
272             return result;
273         }
274     }

278         // TODO :FIXME, since there is no segment for insertValuesClause, InsertStatement is created by sub rule.

282     @Override
283     public ASTNode visitInsertSingleTable(final InsertSingleTableContext ctx) {
284         OracleInsertStatement result = (OracleInsertStatement) visit(ctx.insertIntoClause());
285         if (null != ctx.insertValuesClause()) {
286             result.getValues().addAll(createInsertValuesSegments(ctx.insertValuesClause().assignmentValues()));
287         }
288         if (null != ctx.selectSubquery()) {
289             OracleSelectStatement subquery = (OracleSelectStatement) visit(ctx.selectSubquery());
290             SubquerySegment subquerySegment = new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(), subquery);
291             result.setInsertSelect(subquerySegment);
292         }
293         result.addParameterMarkerSegments(getParameterMarkerSegments());
294         return result;
295     }
296     
297     private Collection<InsertValuesSegment> createInsertValuesSegments(final AssignmentValuesContext ctx) {
298         Collection<InsertValuesSegment> result = new LinkedList<>();
299         result.add((InsertValuesSegment) visit(ctx));
300         return result;
301     }

303     @Override
304     public ASTNode visitInsertMultiTable(final InsertMultiTableContext ctx) {
305         OracleInsertStatement result = new OracleInsertStatement();
306         result.setInsertMultiTableElementSegment(null == ctx.conditionalInsertClause()
307                 ? createInsertMultiTableElementSegment(ctx.multiTableElement())
308                 : (InsertMultiTableElementSegment) visit(ctx.conditionalInsertClause()));
309         OracleSelectStatement subquery = (OracleSelectStatement) visit(ctx.selectSubquery());
310         SubquerySegment subquerySegment = new SubquerySegment(ctx.selectSubquery().start.getStartIndex(), ctx.selectSubquery().stop.getStopIndex(), subquery);
311         result.setInsertSelect(subquerySegment);
312         result.addParameterMarkerSegments(getParameterMarkerSegments());
313         return result;
314     }
315     
316     private InsertMultiTableElementSegment createInsertMultiTableElementSegment(final List<MultiTableElementContext> ctx) {
317         Collection<OracleInsertStatement> insertStatements = new LinkedList<>();
318         for (MultiTableElementContext each : ctx) {
319             insertStatements.add((OracleInsertStatement) visit(each));
320         }
321         InsertMultiTableElementSegment result = new InsertMultiTableElementSegment(ctx.get(0).getStart().getStartIndex(), ctx.get(ctx.size() - 1).getStop().getStopIndex());
322         result.getInsertStatements().addAll(insertStatements);
323         return result;
324     }

326     @Override
327     public ASTNode visitInsertValuesClause(final InsertValuesClauseContext ctx) {
328         OracleInsertStatement result = new OracleInsertStatement();
329         result.getValues().addAll(createInsertValuesSegments(ctx.assignmentValues()));
330         return result;
331     }

333     @SuppressWarnings("unchecked")
334     @Override
335     public ASTNode visitInsertIntoClause(final InsertIntoClauseContext ctx) {
336         OracleInsertStatement result = new OracleInsertStatement();
337         if (null != ctx.dmlTableExprClause().dmlTableClause()) {
338             result.setTable((SimpleTableSegment) visit(ctx.dmlTableExprClause().dmlTableClause()));
339         } else if (null != ctx.dmlTableExprClause().dmlSubqueryClause()) {
340             result.setInsertSelect((SubquerySegment) visit(ctx.dmlTableExprClause().dmlSubqueryClause()));
341         } else {
342             result.setInsertSelect((SubquerySegment) visit(ctx.dmlTableExprClause().tableCollectionExpr()));
343         }
344         if (null != ctx.columnNames()) {
345             ColumnNamesContext columnNames = ctx.columnNames();
346             CollectionValue<ColumnSegment> columnSegments = (CollectionValue<ColumnSegment>) visit(columnNames);
347             result.setInsertColumns(new InsertColumnsSegment(columnNames.start.getStartIndex(), columnNames.stop.getStopIndex(), columnSegments.getValue()));
348         } else {
349             result.setInsertColumns(new InsertColumnsSegment(ctx.stop.getStopIndex() + 1, ctx.stop.getStopIndex() + 1, Collections.emptyList()));
350         }
351         return result;
352     }

361         if (null != ctx.whereClause()) {
362             result.setWhere((WhereSegment) visit(ctx.whereClause()));
363         }
364         result.addParameterMarkerSegments(getParameterMarkerSegments());
365         return result;

688     private ASTNode createProjection(final SelectProjectionExprClauseContext ctx) {
689         AliasSegment alias = null == ctx.alias() ? null : (AliasSegment) visit(ctx.alias());
690         ASTNode projection = visit(ctx.expr());
691         if (projection instanceof FunctionSegment) {
692             FunctionSegment segment = (FunctionSegment) projection;
693             ExpressionProjectionSegment result = new ExpressionProjectionSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getText(), segment);
694             result.setAlias(alias);
695             return result;
696         }
697         if (projection instanceof CommonExpressionSegment) {
698             CommonExpressionSegment segment = (CommonExpressionSegment) projection;
699             ExpressionProjectionSegment result = new ExpressionProjectionSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getText(), segment);
700             result.setAlias(alias);
701             return result;
702         }
703         // FIXME :For DISTINCT()
704         if (projection instanceof ColumnSegment) {
705             ColumnProjectionSegment result = new ColumnProjectionSegment((ColumnSegment) projection);
706             result.setAlias(alias);
707             return result;
708         }
709         if (projection instanceof SubqueryExpressionSegment) {
710             SubqueryExpressionSegment subqueryExpressionSegment = (SubqueryExpressionSegment) projection;
711             String text = ctx.start.getInputStream().getText(new Interval(subqueryExpressionSegment.getStartIndex(), subqueryExpressionSegment.getStopIndex()));
712             SubqueryProjectionSegment result = new SubqueryProjectionSegment(((SubqueryExpressionSegment) projection).getSubquery(), text);
713             result.setAlias(alias);
714             return result;
715         }
716         if (projection instanceof BinaryOperationExpression) {
717             BinaryOperationExpression binaryExpression = (BinaryOperationExpression) projection;
718             int startIndex = binaryExpression.getStartIndex();
719             int stopIndex = null == alias ? binaryExpression.getStopIndex() : alias.getStopIndex();
720             ExpressionProjectionSegment result = new ExpressionProjectionSegment(startIndex, stopIndex, binaryExpression.getText(), binaryExpression);
721             result.setAlias(alias);
722             return result;
723         }
724         if (projection instanceof DatetimeExpression) {
725             DatetimeExpression datetimeExpression = (DatetimeExpression) projection;
726             return null == datetimeExpression.getRight()
727                     ? new DatetimeProjectionSegment(datetimeExpression.getStartIndex(), datetimeExpression.getStopIndex(), datetimeExpression.getLeft(), datetimeExpression.getText())
728                     : new DatetimeProjectionSegment(datetimeExpression.getStartIndex(), datetimeExpression.getStopIndex(),
729                             datetimeExpression.getLeft(), datetimeExpression.getRight(), datetimeExpression.getText());
730         }
731         if (projection instanceof XmlQueryAndExistsFunctionSegment || projection instanceof XmlPiFunctionSegment || projection instanceof XmlSerializeFunctionSegment) {
732             return projection;
733         }
734         if (projection instanceof AliasAvailable) {
735             ((AliasAvailable) projection).setAlias(alias);
736             return projection;
737         }
738         if (projection instanceof IntervalExpressionProjection) {
739             IntervalExpressionProjection intervalExpressionProjection = (IntervalExpressionProjection) projection;
740             IntervalExpressionProjection result = new IntervalExpressionProjection(intervalExpressionProjection.getStartIndex(), intervalExpressionProjection.getStopIndex(),
741                     intervalExpressionProjection.getLeft(), intervalExpressionProjection.getMinus(), intervalExpressionProjection.getRight());
742             if (null != intervalExpressionProjection.getDayToSecondExpression()) {
743                 result.setDayToSecondExpression(intervalExpressionProjection.getDayToSecondExpression());
744             } else {
745                 result.setYearToMonthExpression(intervalExpressionProjection.getYearToMonthExpression());
746             }
747             return result;
748         }
749         LiteralExpressionSegment column = (LiteralExpressionSegment) projection;
750         ExpressionProjectionSegment result = null == alias
751                 ? new ExpressionProjectionSegment(column.getStartIndex(), column.getStopIndex(), String.valueOf(column.getLiterals()), column)
752                 : new ExpressionProjectionSegment(column.getStartIndex(), ctx.alias().stop.getStopIndex(), String.valueOf(column.getLiterals()), column);
753         result.setAlias(alias);
754         return result;
755     }

942         } else {

965         Collection<OrderByItemSegment> result = new LinkedList<>();
966         if (null != ctx.expr()) {

968             result.add(item);
969         } else if (null != ctx.rollupCubeClause()) {
970             result.addAll(generateOrderByItemSegmentsFromRollupCubeClause(ctx.rollupCubeClause()));
971         } else {

977     private ASTNode extractValueFromGroupByItemExpression(final ExprContext ctx) {
978         ASTNode expression = visit(ctx);
979         if (expression instanceof ColumnSegment) {
980             ColumnSegment column = (ColumnSegment) expression;
981             return new ColumnOrderByItemSegment(column, OrderDirection.ASC, null);
982         }
983         if (expression instanceof LiteralExpressionSegment) {
984             LiteralExpressionSegment literalExpression = (LiteralExpressionSegment) expression;
985             return new IndexOrderByItemSegment(literalExpression.getStartIndex(), literalExpression.getStopIndex(),
986                     SQLUtils.getExactlyNumber(literalExpression.getLiterals().toString(), 10).intValue(), OrderDirection.ASC, null);
987         }
988         return new ExpressionOrderByItemSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), getOriginalText(ctx), OrderDirection.ASC, null, (ExpressionSegment) expression);
989     }
