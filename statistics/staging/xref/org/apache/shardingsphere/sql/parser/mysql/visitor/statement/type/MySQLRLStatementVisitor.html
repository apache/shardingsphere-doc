32  /**
33   * RL statement visitor for MySQL.
34   */
35  public final class MySQLRLStatementVisitor extends MySQLStatementVisitor implements RLStatementVisitor {
36      
37      @Override
38      public ASTNode visitChangeMasterTo(final ChangeMasterToContext ctx) {
39          return new MySQLChangeMasterStatement();
40      }
41      
42      @Override
43      public ASTNode visitStartSlave(final StartSlaveContext ctx) {
44          return new MySQLStartSlaveStatement();
45      }
46      
47      @Override
48      public ASTNode visitStopSlave(final StopSlaveContext ctx) {
49          return new MySQLStopSlaveStatement();
50      }
51      
52      @Override
53      public ASTNode visitChangeReplicationSourceTo(final ChangeReplicationSourceToContext ctx) {
54          return new MySQLChangeReplicationSourceToStatement();
55      }
56  }
