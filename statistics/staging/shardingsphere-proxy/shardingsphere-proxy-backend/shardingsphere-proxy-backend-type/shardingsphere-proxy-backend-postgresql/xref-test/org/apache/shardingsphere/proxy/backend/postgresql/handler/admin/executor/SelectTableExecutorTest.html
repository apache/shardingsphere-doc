57  @ExtendWith(AutoMockExtension.class)
58  @StaticMockSettings(ProxyContext.class)
59  class SelectTableExecutorTest {
60      
61      @Test
62      void assertExecute() throws SQLException {
63          String sql = "SELECT c.oid, n.nspname AS schemaname, c.relname AS tablename from pg_tablespace";
64          ShardingSphereDatabase database = createDatabase();
65          ContextManager contextManager = mockContextManager(database);
66          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
67          when(ProxyContext.getInstance().getAllDatabaseNames()).thenReturn(Collections.singleton("public"));
68          when(ProxyContext.getInstance().getDatabase("public")).thenReturn(database);
69          SelectTableExecutor executor = new SelectTableExecutor(sql);
70          executor.execute(mock(ConnectionSession.class));
71          assertThat(executor.getQueryResultMetaData().getColumnCount(), is(3));
72          int count = 0;
73          while (executor.getMergedResult().next()) {
74              count++;
75              if ("t_order".equals(executor.getMergedResult().getValue(1, String.class))) {
76                  assertThat(executor.getMergedResult().getValue(2, String.class), is("0000"));
77                  assertThat(executor.getMergedResult().getValue(3, String.class), is("public"));
78              } else {
79                  fail("expected : `t_order`");
80              }
81          }
82          assertThat(count, is(1));
83      }
84      
85      private ShardingSphereDatabase createDatabase() throws SQLException {
86          return new ShardingSphereDatabase("public", new PostgreSQLDatabaseType(),
87                  new ShardingSphereResourceMetaData("sharding_db", Collections.singletonMap("foo_ds", new MockedDataSource(mockConnection()))),
88                  new ShardingSphereRuleMetaData(Collections.emptyList()), Collections.singletonMap("public",
89                          new ShardingSphereSchema(Collections.singletonMap("t_order", mock(ShardingSphereTable.class)), Collections.emptyMap())));
90      }
91      
92      private ContextManager mockContextManager(final ShardingSphereDatabase database) {
93          ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
94          MetaDataContexts metaDataContexts = new MetaDataContexts(mock(MetaDataPersistService.class),
95                  new ShardingSphereMetaData(Collections.singletonMap("public", database), mock(ShardingSphereRuleMetaData.class), new ConfigurationProperties(new Properties())));
96          when(result.getMetaDataContexts()).thenReturn(metaDataContexts);
97          return result;
98      }
99      
100     private Connection mockConnection() throws SQLException {
101         Connection result = mock(Connection.class, RETURNS_DEEP_STUBS);
102         when(result.getMetaData().getURL()).thenReturn("jdbc:mysql://localhost:3306/foo_ds");
103         ResultSet resultSet = mockResultSet();
104         when(result.prepareStatement(any(String.class)).executeQuery()).thenReturn(resultSet);
105         return result;
106     }
107     
108     private ResultSet mockResultSet() throws SQLException {
109         ResultSet result = mock(ResultSet.class, RETURNS_DEEP_STUBS);
110         Map<String, String> expectedResultSetMap = createExpectedResultSetMap();
111         List<String> keys = new ArrayList<>(expectedResultSetMap.keySet());
112         for (int i = 0; i < keys.size(); i++) {
113             when(result.getMetaData().getColumnName(i + 1)).thenReturn(keys.get(i));
114             when(result.getMetaData().getColumnLabel(i + 1)).thenReturn(keys.get(i));
115             when(result.getString(i + 1)).thenReturn(expectedResultSetMap.get(keys.get(i)));
116         }
117         when(result.next()).thenReturn(true, false);
118         when(result.getMetaData().getColumnCount()).thenReturn(expectedResultSetMap.size());
119         return result;
120     }
121     
122     private Map<String, String> createExpectedResultSetMap() {
123         Map<String, String> result = new LinkedHashMap<>(3, 1);
124         result.put("tablename", "t_order_1");
125         result.put("c.oid", "0000");
126         result.put("schemaname", "public");
127         return result;
128     }
