97      public InventoryDumper(final InventoryDumperContext dumperContext, final PipelineChannel channel, final DataSource dataSource, final PipelineTableMetaDataLoader metaDataLoader) {
98          this.dumperContext = dumperContext;
99          this.channel = channel;
100         this.dataSource = dataSource;
101         this.metaDataLoader = metaDataLoader;
102         DatabaseType databaseType = dumperContext.getCommonContext().getDataSourceConfig().getDatabaseType();
103         inventoryDumpSQLBuilder = new PipelineInventoryDumpSQLBuilder(databaseType);
104         columnValueReaderEngine = new InventoryColumnValueReaderEngine(databaseType);
105     }
106     
107     @Override
108     protected void runBlocking() {
109         IngestPosition position = dumperContext.getCommonContext().getPosition();
110         if (position instanceof IngestFinishedPosition) {
111             log.info("Ignored because of already finished.");
112             return;
113         }
114         init();
115         try (Connection connection = dataSource.getConnection()) {
116             if (!Strings.isNullOrEmpty(dumperContext.getQuerySQL()) || !dumperContext.hasUniqueKey() || isPrimaryKeyWithoutRanged(position)) {
117                 dumpWithStreamingQuery(connection);
118             } else {
119                 dumpPageByPage(connection);
120             }
121             // CHECKSTYLE:OFF
122         } catch (final SQLException | RuntimeException ex) {
123             // CHECKSTYLE:ON
124             log.error("Inventory dump failed on {}", dumperContext.getActualTableName(), ex);
125             throw new IngestException("Inventory dump failed on " + dumperContext.getActualTableName(), ex);
126         }
127     }
128     
129     private boolean isPrimaryKeyWithoutRanged(final IngestPosition position) {
130         return position instanceof PrimaryKeyIngestPosition && null == ((PrimaryKeyIngestPosition<?>) position).getBeginValue() && null == ((PrimaryKeyIngestPosition<?>) position).getEndValue();
131     }
132     
133     /**
134      * Initialize.
135      */
136     public void init() {
137         if (null == uniqueKeysNames) {
138             uniqueKeysNames = getUniqueKeysNames();
139         }
140         if (null == tableMetaData) {
141             tableMetaData = metaDataLoader.getTableMetaData(
142                     dumperContext.getCommonContext().getTableAndSchemaNameMapper().getSchemaName(dumperContext.getLogicTableName()), dumperContext.getActualTableName());
143         }
144     }
145     
146     private List<CaseInsensitiveIdentifier> getUniqueKeysNames() {
147         if (dumperContext.hasUniqueKey()) {
148             return dumperContext.getUniqueKeyColumns().stream().map(each -> new CaseInsensitiveIdentifier(each.getName())).collect(Collectors.toList());
149         }
150         return Collections.emptyList();
151     }

153     @SuppressWarnings("MagicConstant")
154     private void dumpWithStreamingQuery(final Connection connection) throws SQLException {
155         int batchSize = dumperContext.getBatchSize();
156         DatabaseType databaseType = dumperContext.getCommonContext().getDataSourceConfig().getDatabaseType();
157         if (null != dumperContext.getTransactionIsolation()) {
158             connection.setTransactionIsolation(dumperContext.getTransactionIsolation());
159         }
160         try (PreparedStatement preparedStatement = JDBCStreamQueryBuilder.build(databaseType, connection, buildInventoryDumpSQLWithStreamingQuery())) {
161             runningStatement.set(preparedStatement);
162             if (!DatabaseTypeUtils.isMySQL(databaseType)) {
163                 preparedStatement.setFetchSize(batchSize);
164             }
165             PrimaryKeyIngestPosition<?> primaryPosition = (PrimaryKeyIngestPosition<?>) dumperContext.getCommonContext().getPosition();
166             InventoryQueryParameter queryParam = InventoryQueryParameter.buildForRangeQuery(new QueryRange(primaryPosition.getBeginValue(), true, primaryPosition.getEndValue()));
167             setParameters(preparedStatement, queryParam, true);
168             try (ResultSet resultSet = preparedStatement.executeQuery()) {
169                 int rowCount = 0;
170                 JobRateLimitAlgorithm rateLimitAlgorithm = dumperContext.getRateLimitAlgorithm();
171                 ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
172                 List<Record> dataRecords = new LinkedList<>();
173                 while (resultSet.next()) {
174                     if (dataRecords.size() >= batchSize) {
175                         channel.push(dataRecords);
176                         dataRecords = new LinkedList<>();
177                     }
178                     dataRecords.add(loadDataRecord(resultSet, resultSetMetaData));
179                     ++rowCount;
180                     if (!isRunning()) {
181                         log.info("Broke because of inventory dump is not running.");
182                         break;
183                     }
184                     if (null != rateLimitAlgorithm && 0 == rowCount % batchSize) {
185                         rateLimitAlgorithm.intercept(PipelineSQLOperationType.SELECT, 1);
186                     }
187                 }
188                 dataRecords.add(new FinishedRecord(new IngestFinishedPosition()));
189                 channel.push(dataRecords);
190                 log.info("Inventory dump with streaming query done, rowCount={}, dataSource={}, actualTable={}",
191                         rowCount, dumperContext.getCommonContext().getDataSourceName(), dumperContext.getActualTableName());
192             } finally {
193                 runningStatement.set(null);
194             }
195         }
196     }
197     
198     private String buildInventoryDumpSQLWithStreamingQuery() {
199         if (!Strings.isNullOrEmpty(dumperContext.getQuerySQL())) {
200             return dumperContext.getQuerySQL();
201         }
202         String schemaName = dumperContext.getCommonContext().getTableAndSchemaNameMapper().getSchemaName(dumperContext.getLogicTableName());
203         List<String> columnNames = getQueryColumnNames();
204         return inventoryDumpSQLBuilder.buildFetchAllSQL(schemaName, dumperContext.getActualTableName(), columnNames);
205     }
206     
207     @SuppressWarnings("MagicConstant")
208     private void dumpPageByPage(final Connection connection) throws SQLException {
209         if (null != dumperContext.getTransactionIsolation()) {
210             connection.setTransactionIsolation(dumperContext.getTransactionIsolation());
211         }
212         boolean firstQuery = true;
213         AtomicLong rowCount = new AtomicLong();
214         IngestPosition position = dumperContext.getCommonContext().getPosition();
215         while (true) {
216             PrimaryKeyIngestPosition<?> primaryPosition = (PrimaryKeyIngestPosition<?>) position;
217             InventoryQueryParameter queryParam = InventoryQueryParameter.buildForRangeQuery(new QueryRange(primaryPosition.getBeginValue(), firstQuery, primaryPosition.getEndValue()));
218             List<Record> dataRecords = dumpPageByPage0(connection, queryParam, rowCount);
219             if (dataRecords.size() > 1 && Objects.deepEquals(getFirstUniqueKeyValue(dataRecords, 0), getFirstUniqueKeyValue(dataRecords, dataRecords.size() - 1))) {
220                 queryParam = InventoryQueryParameter.buildForPointQuery(getFirstUniqueKeyValue(dataRecords, 0));
221                 dataRecords = dumpPageByPage0(connection, queryParam, rowCount);
222             }
223             firstQuery = false;
224             if (dataRecords.isEmpty()) {
225                 position = new IngestFinishedPosition();
226                 dataRecords.add(new FinishedRecord(position));
227                 log.info("Inventory dump done, rowCount={}, dataSource={}, actualTable={}", rowCount, dumperContext.getCommonContext().getDataSourceName(), dumperContext.getActualTableName());
228             } else {
229                 position = PrimaryKeyIngestPositionFactory.newInstance(getFirstUniqueKeyValue(dataRecords, dataRecords.size() - 1), primaryPosition.getEndValue());
230             }
231             channel.push(dataRecords);
232             dumperContext.getCommonContext().setPosition(position);
233             if (position instanceof IngestFinishedPosition) {
234                 break;
235             }
236         }
237     }
238     
239     private Object getFirstUniqueKeyValue(final List<Record> dataRecords, final int index) {
240         return ((DataRecord) dataRecords.get(index)).getUniqueKeyValue().iterator().next();
241     }
242     
243     private List<Record> dumpPageByPage0(final Connection connection, final InventoryQueryParameter queryParam,
244                                          final AtomicLong rowCount) throws SQLException {
245         DatabaseType databaseType = dumperContext.getCommonContext().getDataSourceConfig().getDatabaseType();
246         int batchSize = dumperContext.getBatchSize();
247         try (PreparedStatement preparedStatement = JDBCStreamQueryBuilder.build(databaseType, connection, buildInventoryDumpPageByPageSQL(queryParam))) {
248             runningStatement.set(preparedStatement);
249             if (!DatabaseTypeUtils.isMySQL(databaseType)) {
250                 preparedStatement.setFetchSize(batchSize);
251             }
252             setParameters(preparedStatement, queryParam, false);
253             try (ResultSet resultSet = preparedStatement.executeQuery()) {
254                 JobRateLimitAlgorithm rateLimitAlgorithm = dumperContext.getRateLimitAlgorithm();
255                 ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
256                 List<Record> result = new LinkedList<>();
257                 while (resultSet.next()) {
258                     if (result.size() >= batchSize) {
259                         if (!dumperContext.hasUniqueKey()) {
260                             channel.push(result);
261                         }
262                         result = new LinkedList<>();
263                     }
264                     result.add(loadDataRecord(resultSet, resultSetMetaData));
265                     rowCount.incrementAndGet();
266                     if (!isRunning()) {
267                         log.info("Broke because of inventory dump is not running.");
268                         break;
269                     }
270                     if (null != rateLimitAlgorithm && 0 == rowCount.get() % batchSize) {
271                         rateLimitAlgorithm.intercept(PipelineSQLOperationType.SELECT, 1);
272                     }
273                 }
274                 return result;
275             } finally {
276                 runningStatement.set(null);
277             }
278         }
279     }
280     
281     private String buildInventoryDumpPageByPageSQL(final InventoryQueryParameter queryParam) {
282         String schemaName = dumperContext.getCommonContext().getTableAndSchemaNameMapper().getSchemaName(dumperContext.getLogicTableName());
283         PipelineColumnMetaData firstColumn = dumperContext.getUniqueKeyColumns().get(0);
284         List<String> columnNames = getQueryColumnNames();
285         if (QueryType.POINT_QUERY == queryParam.getQueryType()) {
286             return inventoryDumpSQLBuilder.buildPointQuerySQL(schemaName, dumperContext.getActualTableName(), columnNames, firstColumn.getName());
287         }
288         QueryRange queryRange = queryParam.getUniqueKeyValueRange();
289         boolean lowerInclusive = queryRange.isLowerInclusive();
290         if (null != queryRange.getLower() && null != queryRange.getUpper()) {
291             return inventoryDumpSQLBuilder.buildDivisibleSQL(schemaName, dumperContext.getActualTableName(), columnNames, firstColumn.getName(), lowerInclusive);
292         }
293         if (null != queryRange.getLower()) {
294             return inventoryDumpSQLBuilder.buildUnlimitedDivisibleSQL(schemaName, dumperContext.getActualTableName(), columnNames, firstColumn.getName(), lowerInclusive);
295         }
296         throw new PipelineInternalException("Primary key position is invalid.");
297     }
298     
299     private List<String> getQueryColumnNames() {
300         return Optional.ofNullable(dumperContext.getInsertColumnNames()).orElse(Collections.singletonList("*"));
301     }
302     
303     private void setParameters(final PreparedStatement preparedStatement, final InventoryQueryParameter queryParam, final boolean streamingQuery) throws SQLException {
304         if (!Strings.isNullOrEmpty(dumperContext.getQuerySQL())) {
305             for (int i = 0; i < dumperContext.getQueryParams().size(); i++) {
306                 preparedStatement.setObject(i + 1, dumperContext.getQueryParams().get(i));
307             }
308             return;
309         }
310         if (!dumperContext.hasUniqueKey()) {
311             return;
312         }
313         int parameterIndex = 1;
314         if (QueryType.RANGE_QUERY == queryParam.getQueryType()) {
315             Object lower = queryParam.getUniqueKeyValueRange().getLower();
316             if (null != lower) {
317                 preparedStatement.setObject(parameterIndex++, lower);
318             }
319             Object upper = queryParam.getUniqueKeyValueRange().getUpper();
320             if (null != upper) {
321                 preparedStatement.setObject(parameterIndex++, upper);
322             }
323             if (!streamingQuery) {
324                 preparedStatement.setInt(parameterIndex, dumperContext.getBatchSize());
325             }
326         } else if (QueryType.POINT_QUERY == queryParam.getQueryType()) {
327             preparedStatement.setObject(parameterIndex, queryParam.getUniqueKeyValue());
328         } else {
329             throw new UnsupportedOperationException("Query type: " + queryParam.getQueryType());
330         }
331     }
332     
333     private DataRecord loadDataRecord(final ResultSet resultSet, final ResultSetMetaData resultSetMetaData) throws SQLException {
334         int columnCount = resultSetMetaData.getColumnCount();
335         DataRecord result = new DataRecord(PipelineSQLOperationType.INSERT, dumperContext.getLogicTableName(), newDataRecordPosition(resultSet), columnCount);
336         List<String> insertColumnNames = Optional.ofNullable(dumperContext.getInsertColumnNames()).orElse(Collections.emptyList());
337         ShardingSpherePreconditions.checkState(insertColumnNames.isEmpty() || insertColumnNames.size() == resultSetMetaData.getColumnCount(),
338                 () -> new PipelineInvalidParameterException("Insert column names count not equals ResultSet column count"));
339         for (int i = 1; i <= columnCount; i++) {
340             String columnName = insertColumnNames.isEmpty() ? resultSetMetaData.getColumnName(i) : insertColumnNames.get(i - 1);
341             ShardingSpherePreconditions.checkNotNull(tableMetaData.getColumnMetaData(columnName), () -> new PipelineInvalidParameterException(String.format("Column name is %s", columnName)));
342             boolean isUniqueKey = tableMetaData.getColumnMetaData(columnName).isUniqueKey();
343             result.addColumn(new Column(columnName, columnValueReaderEngine.read(resultSet, resultSetMetaData, i), true, isUniqueKey));
344         }
345         result.setActualTableName(dumperContext.getActualTableName());
346         return result;
347     }
348     
349     protected IngestPosition newDataRecordPosition(final ResultSet resultSet) throws SQLException {
350         return dumperContext.hasUniqueKey()
351                 ? PrimaryKeyIngestPositionFactory.newInstance(
352                         resultSet.getObject(dumperContext.getUniqueKeyColumns().get(0).getName()), ((PrimaryKeyIngestPosition<?>) dumperContext.getCommonContext().getPosition()).getEndValue())
353                 : new IngestPlaceholderPosition();
354     }
355     
356     @Override
357     protected void doStop() {
358         Optional.ofNullable(runningStatement.get()).ifPresent(PipelineJdbcUtils::cancelStatement);
359     }
