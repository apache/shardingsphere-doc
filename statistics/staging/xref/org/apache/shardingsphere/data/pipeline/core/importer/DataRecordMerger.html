24  import org.apache.shardingsphere.data.pipeline.core.ingest.IngestDataChangeType;

27  import org.apache.shardingsphere.infra.util.exception.external.sql.type.generic.UnsupportedSQLOperationException;
28  
29  import java.util.ArrayList;
30  import java.util.HashMap;
31  import java.util.List;
32  import java.util.Map;
33  import java.util.Map.Entry;
34  import java.util.stream.Collectors;
35  
36  /**
37   * Data Record merger.
38   */
39  public final class DataRecordMerger {
40      
41      /**
42       * Merge data record.
43       * <pre>
44       * insert + insert -&gt; exception
45       * update + insert -&gt; exception
46       * delete + insert -&gt; insert
47       * insert + update -&gt; insert
48       * update + update -&gt; update
49       * delete + update -&gt; exception
50       * insert + delete -&gt; delete
51       * update + delete -&gt; delete
52       * delete + delete -&gt; exception
53       * </pre>
54       *
55       * @param dataRecords data records
56       * @return merged data records
57       */
58      public List<DataRecord> merge(final List<DataRecord> dataRecords) {
59          Map<DataRecord.Key, DataRecord> result = new HashMap<>();
60          dataRecords.forEach(each -> {
61              if (IngestDataChangeType.INSERT.equals(each.getType())) {
62                  mergeInsert(each, result);
63              } else if (IngestDataChangeType.UPDATE.equals(each.getType())) {
64                  mergeUpdate(each, result);
65              } else if (IngestDataChangeType.DELETE.equals(each.getType())) {
66                  mergeDelete(each, result);
67              }
68          });
69          return new ArrayList<>(result.values());
70      }
71      
72      /**
73       * Group by table and type.
74       *
75       * @param dataRecords data records
76       * @return grouped data records
77       */
78      public List<GroupedDataRecord> group(final List<DataRecord> dataRecords) {
79          List<GroupedDataRecord> result = new ArrayList<>(100);
80          List<DataRecord> mergedDataRecords = dataRecords.get(0).getUniqueKeyValue().isEmpty() ? dataRecords : merge(dataRecords);
81          Map<String, List<DataRecord>> tableGroup = mergedDataRecords.stream().collect(Collectors.groupingBy(DataRecord::getTableName));
82          for (Entry<String, List<DataRecord>> entry : tableGroup.entrySet()) {
83              Map<String, List<DataRecord>> typeGroup = entry.getValue().stream().collect(Collectors.groupingBy(DataRecord::getType));
84              result.add(new GroupedDataRecord(entry.getKey(), typeGroup.get(IngestDataChangeType.INSERT), typeGroup.get(IngestDataChangeType.UPDATE), typeGroup.get(IngestDataChangeType.DELETE)));
85          }
86          return result;
87      }
88      
89      private void mergeInsert(final DataRecord dataRecord, final Map<DataRecord.Key, DataRecord> dataRecords) {
90          DataRecord beforeDataRecord = dataRecords.get(dataRecord.getKey());
91          ShardingSpherePreconditions.checkState(null == beforeDataRecord
92                  || IngestDataChangeType.DELETE.equals(beforeDataRecord.getType()), () -> new PipelineUnexpectedDataRecordOrderException(beforeDataRecord, dataRecord));
93          dataRecords.put(dataRecord.getKey(), dataRecord);
94      }
95      
96      private void mergeUpdate(final DataRecord dataRecord, final Map<DataRecord.Key, DataRecord> dataRecords) {
97          DataRecord beforeDataRecord = checkUpdatedPrimaryKey(dataRecord) ? dataRecords.get(dataRecord.getOldKey()) : dataRecords.get(dataRecord.getKey());
98          if (null == beforeDataRecord) {
99              dataRecords.put(dataRecord.getKey(), dataRecord);
100             return;
101         }
102         ShardingSpherePreconditions.checkState(!IngestDataChangeType.DELETE.equals(beforeDataRecord.getType()), () -> new UnsupportedSQLOperationException("Not Delete"));
103         if (checkUpdatedPrimaryKey(dataRecord)) {
104             dataRecords.remove(dataRecord.getOldKey());
105         }
106         if (IngestDataChangeType.INSERT.equals(beforeDataRecord.getType())) {
107             DataRecord mergedDataRecord = mergeColumn(beforeDataRecord, dataRecord);
108             mergedDataRecord.setTableName(dataRecord.getTableName());
109             mergedDataRecord.setType(IngestDataChangeType.INSERT);
110             dataRecords.put(mergedDataRecord.getKey(), mergedDataRecord);
111             return;
112         }
113         if (IngestDataChangeType.UPDATE.equals(beforeDataRecord.getType())) {
114             DataRecord mergedDataRecord = mergeColumn(beforeDataRecord, dataRecord);
115             mergedDataRecord.setTableName(dataRecord.getTableName());
116             mergedDataRecord.setType(IngestDataChangeType.UPDATE);
117             dataRecords.put(mergedDataRecord.getKey(), mergedDataRecord);
118         }
119     }
120     
121     private void mergeDelete(final DataRecord dataRecord, final Map<DataRecord.Key, DataRecord> dataRecords) {
122         DataRecord beforeDataRecord = dataRecords.get(dataRecord.getKey());
123         ShardingSpherePreconditions.checkState(null == beforeDataRecord
124                 || !IngestDataChangeType.DELETE.equals(beforeDataRecord.getType()), () -> new PipelineUnexpectedDataRecordOrderException(beforeDataRecord, dataRecord));
125         if (null != beforeDataRecord && IngestDataChangeType.UPDATE.equals(beforeDataRecord.getType()) && checkUpdatedPrimaryKey(beforeDataRecord)) {
126             DataRecord mergedDataRecord = new DataRecord(dataRecord.getPosition(), dataRecord.getColumnCount());
127             for (int i = 0; i < dataRecord.getColumnCount(); i++) {
128                 mergedDataRecord.addColumn(new Column(dataRecord.getColumn(i).getName(),
129                         dataRecord.getColumn(i).isUniqueKey() ? beforeDataRecord.getColumn(i).getOldValue() : beforeDataRecord.getColumn(i).getValue(), true, dataRecord.getColumn(i).isUniqueKey()));
130             }
131             mergedDataRecord.setTableName(dataRecord.getTableName());
132             mergedDataRecord.setType(IngestDataChangeType.DELETE);
133             dataRecords.remove(beforeDataRecord.getKey());
134             dataRecords.put(mergedDataRecord.getKey(), mergedDataRecord);
135         } else {
136             dataRecords.put(dataRecord.getKey(), dataRecord);
137         }
138     }
139     
140     private boolean checkUpdatedPrimaryKey(final DataRecord dataRecord) {
141         return RecordUtils.extractPrimaryColumns(dataRecord).stream().anyMatch(Column::isUpdated);
142     }
143     
144     private DataRecord mergeColumn(final DataRecord preDataRecord, final DataRecord curDataRecord) {
145         DataRecord result = new DataRecord(curDataRecord.getPosition(), curDataRecord.getColumnCount());
146         for (int i = 0; i < curDataRecord.getColumnCount(); i++) {
147             result.addColumn(new Column(
148                     curDataRecord.getColumn(i).getName(),
149                     preDataRecord.getColumn(i).isUniqueKey()
150                             ? mergePrimaryKeyOldValue(preDataRecord.getColumn(i), curDataRecord.getColumn(i))
151                             : null,
152                     curDataRecord.getColumn(i).getValue(),
153                     preDataRecord.getColumn(i).isUpdated() || curDataRecord.getColumn(i).isUpdated(),
154                     curDataRecord.getColumn(i).isUniqueKey()));
155         }
156         return result;
157     }
158     
159     private Object mergePrimaryKeyOldValue(final Column beforeColumn, final Column column) {
160         return beforeColumn.isUpdated() ? beforeColumn.getOldValue() : (column.isUpdated() ? column.getOldValue() : null);
161     }
