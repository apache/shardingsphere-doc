55  /**
56   * Encrypt condition engine.
57   */
58  @HighFrequencyInvocation
59  @RequiredArgsConstructor
60  public final class EncryptConditionEngine {
61      
62      private static final Set<String> LOGICAL_OPERATOR = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
63      
64      private static final Set<String> SUPPORTED_COMPARE_OPERATOR = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
65      
66      private final EncryptRule encryptRule;
67      
68      private final Map<String, ShardingSphereSchema> schemas;
69      
70      static {
71          LOGICAL_OPERATOR.add("AND");
72          LOGICAL_OPERATOR.add("&&");
73          LOGICAL_OPERATOR.add("OR");
74          LOGICAL_OPERATOR.add("||");
75          SUPPORTED_COMPARE_OPERATOR.add("=");
76          SUPPORTED_COMPARE_OPERATOR.add("<>");
77          SUPPORTED_COMPARE_OPERATOR.add("!=");
78          SUPPORTED_COMPARE_OPERATOR.add(">");
79          SUPPORTED_COMPARE_OPERATOR.add("<");
80          SUPPORTED_COMPARE_OPERATOR.add(">=");
81          SUPPORTED_COMPARE_OPERATOR.add("<=");
82          SUPPORTED_COMPARE_OPERATOR.add("IS");
83          SUPPORTED_COMPARE_OPERATOR.add("LIKE");
84      }
85      
86      /**
87       * Create encrypt conditions.
88       *
89       * @param whereSegments where segments
90       * @param columnSegments column segments
91       * @param sqlStatementContext SQL statement context
92       * @param databaseName database name
93       * @return encrypt conditions
94       */
95      public Collection<EncryptCondition> createEncryptConditions(final Collection<WhereSegment> whereSegments, final Collection<ColumnSegment> columnSegments,
96                                                                  final SQLStatementContext sqlStatementContext, final String databaseName) {
97          Collection<EncryptCondition> result = new LinkedList<>();
98          String defaultSchema = new DatabaseTypeRegistry(sqlStatementContext.getDatabaseType()).getDefaultSchemaName(databaseName);
99          ShardingSphereSchema schema = ((TableAvailable) sqlStatementContext).getTablesContext().getSchemaName().map(schemas::get).orElseGet(() -> schemas.get(defaultSchema));
100         Map<String, String> expressionTableNames = ((TableAvailable) sqlStatementContext).getTablesContext().findTableNames(columnSegments, schema);
101         for (WhereSegment each : whereSegments) {
102             Collection<AndPredicate> andPredicates = ExpressionExtractUtils.getAndPredicates(each.getExpr());
103             for (AndPredicate predicate : andPredicates) {
104                 addEncryptConditions(result, predicate.getPredicates(), expressionTableNames);
105             }
106         }
107         return result;
108     }
109     
110     private void addEncryptConditions(final Collection<EncryptCondition> encryptConditions, final Collection<ExpressionSegment> predicates, final Map<String, String> expressionTableNames) {
111         Collection<Integer> stopIndexes = new HashSet<>(predicates.size(), 1F);
112         for (ExpressionSegment each : predicates) {
113             if (stopIndexes.add(each.getStopIndex())) {
114                 addEncryptConditions(encryptConditions, each, expressionTableNames);
115             }
116         }
117     }
118     
119     private void addEncryptConditions(final Collection<EncryptCondition> encryptConditions, final ExpressionSegment expression, final Map<String, String> expressionTableNames) {
120         if (!findNotContainsNullLiteralsExpression(expression).isPresent()) {
121             return;
122         }
123         for (ColumnSegment each : ColumnExtractUtils.extract(expression)) {
124             String tableName = expressionTableNames.getOrDefault(each.getExpression(), "");
125             Optional<EncryptTable> encryptTable = encryptRule.findEncryptTable(tableName);
126             if (encryptTable.isPresent() && encryptTable.get().isEncryptColumn(each.getIdentifier().getValue())) {
127                 createEncryptCondition(expression, tableName).ifPresent(encryptConditions::add);
128             }
129         }
130     }
131     
132     private Optional<ExpressionSegment> findNotContainsNullLiteralsExpression(final ExpressionSegment expression) {
133         if (isContainsNullLiterals(expression)) {
134             return Optional.empty();
135         }
136         if (expression instanceof BinaryOperationExpression && isContainsNullLiterals(((BinaryOperationExpression) expression).getRight())) {
137             return Optional.empty();
138         }
139         return Optional.ofNullable(expression);
140     }
141     
142     private boolean isContainsNullLiterals(final ExpressionSegment expression) {
143         if (!(expression instanceof LiteralExpressionSegment)) {
144             return false;
145         }
146         String literals = String.valueOf(((LiteralExpressionSegment) expression).getLiterals());
147         return "NULL".equalsIgnoreCase(literals) || "NOT NULL".equalsIgnoreCase(literals);
148     }
149     
150     private Optional<EncryptCondition> createEncryptCondition(final ExpressionSegment expression, final String tableName) {
151         if (expression instanceof BinaryOperationExpression) {
152             return createBinaryEncryptCondition((BinaryOperationExpression) expression, tableName);
153         }
154         if (expression instanceof InExpression) {
155             return createInEncryptCondition(tableName, (InExpression) expression, ((InExpression) expression).getRight());
156         }
157         if (expression instanceof BetweenExpression) {
158             throw new UnsupportedEncryptSQLException("BETWEEN...AND...");
159         }
160         return Optional.empty();
161     }
162     
163     private Optional<EncryptCondition> createBinaryEncryptCondition(final BinaryOperationExpression expression, final String tableName) {
164         String operator = expression.getOperator();
165         if (LOGICAL_OPERATOR.contains(operator)) {
166             return Optional.empty();
167         }
168         ShardingSpherePreconditions.checkContains(SUPPORTED_COMPARE_OPERATOR, operator, () -> new UnsupportedEncryptSQLException(operator));
169         return createCompareEncryptCondition(tableName, expression, expression.getRight());
170     }
171     
172     private Optional<EncryptCondition> createCompareEncryptCondition(final String tableName, final BinaryOperationExpression expression, final ExpressionSegment compareRightValue) {
173         if (!(expression.getLeft() instanceof ColumnSegment) || compareRightValue instanceof SubqueryExpressionSegment) {
174             return Optional.empty();
175         }
176         if (compareRightValue instanceof SimpleExpressionSegment) {
177             return Optional.of(createEncryptBinaryOperationCondition(tableName, expression, compareRightValue));
178         }
179         if (compareRightValue instanceof ListExpression) {
180             return Optional.of(createEncryptBinaryOperationCondition(tableName, expression, ((ListExpression) compareRightValue).getItems().get(0)));
181         }
182         return Optional.empty();
183     }
184     
185     private EncryptBinaryCondition createEncryptBinaryOperationCondition(final String tableName, final BinaryOperationExpression expression, final ExpressionSegment compareRightValue) {
186         String columnName = ((ColumnSegment) expression.getLeft()).getIdentifier().getValue();
187         return new EncryptBinaryCondition(columnName, tableName, expression.getOperator(), compareRightValue.getStartIndex(), expression.getStopIndex(), compareRightValue);
188     }
189     
190     private static Optional<EncryptCondition> createInEncryptCondition(final String tableName, final InExpression inExpression, final ExpressionSegment inRightValue) {
191         if (!(inExpression.getLeft() instanceof ColumnSegment)) {
192             return Optional.empty();
193         }
194         List<ExpressionSegment> expressionSegments = new LinkedList<>();
195         for (ExpressionSegment each : inExpression.getExpressionList()) {
196             if (each instanceof SimpleExpressionSegment) {
197                 expressionSegments.add(each);
198             }
199         }
200         if (expressionSegments.isEmpty()) {
201             return Optional.empty();
202         }
203         String columnName = ((ColumnSegment) inExpression.getLeft()).getIdentifier().getValue();
204         return Optional.of(new EncryptInCondition(columnName, tableName, inRightValue.getStartIndex(), inRightValue.getStopIndex(), expressionSegments));
205     }
