18  package org.apache.shardingsphere.infra.exception.core;
19  
20  import com.google.common.base.Strings;
21  import lombok.AccessLevel;
22  import lombok.NoArgsConstructor;
23  
24  import java.util.function.Supplier;
25  
26  /**
27   * ShardingSphere preconditions.
28   */
29  @NoArgsConstructor(access = AccessLevel.PRIVATE)
30  public final class ShardingSpherePreconditions {
31      
32      /**
33       * Ensures the truth of an expression involving the state of the calling instance.
34       *
35       * @param <T> type of exception
36       * @param expectedExpression expected expression
37       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
38       * @throws T exception to be thrown
39       */
40      public static <T extends Throwable> void checkState(final boolean expectedExpression, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
41          if (!expectedExpression) {
42              throw exceptionSupplierIfUnexpected.get();
43          }
44      }
45      
46      /**
47       * Ensures that an object reference passed as a parameter to the calling method is not null.
48       *
49       * @param <T> type of exception
50       * @param reference object reference to be checked
51       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
52       * @throws T exception to be thrown
53       */
54      public static <T extends Throwable> void checkNotNull(final Object reference, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
55          if (null == reference) {
56              throw exceptionSupplierIfUnexpected.get();
57          }
58      }
59      
60      /**
61       * Ensures that a string passed as a parameter to the calling method is not empty.
62       *
63       * @param <T> type of exception
64       * @param value string to be checked
65       * @param exceptionSupplierIfUnexpected exception from this supplier will be thrown if expression is unexpected
66       * @throws T exception to be thrown
67       */
68      public static <T extends Throwable> void checkNotEmpty(final String value, final Supplier<T> exceptionSupplierIfUnexpected) throws T {
69          if (Strings.isNullOrEmpty(value)) {
70              throw exceptionSupplierIfUnexpected.get();
71          }
72      }
73  }
