37  /**
38   * COM_INIT_DB command executor for MySQL.
39   */
40  @RequiredArgsConstructor
41  public final class MySQLComInitDbExecutor implements CommandExecutor {
42      
43      private final MySQLComInitDbPacket packet;
44      
45      private final ConnectionSession connectionSession;
46      
47      @Override
48      public Collection<DatabasePacket> execute() {
49          String databaseName = SQLUtils.getExactlyValue(packet.getSchema());
50          AuthorityRule authorityRule = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getGlobalRuleMetaData().getSingleRule(AuthorityRule.class);
51          AuthorityChecker authorityChecker = new AuthorityChecker(authorityRule, connectionSession.getConnectionContext().getGrantee());
52          ShardingSpherePreconditions.checkState(ProxyContext.getInstance().databaseExists(databaseName) && authorityChecker.isAuthorized(databaseName),
53                  () -> new UnknownDatabaseException(packet.getSchema()));
54          connectionSession.setCurrentDatabase(packet.getSchema());
55          return Collections.singleton(new MySQLOKPacket(ServerStatusFlagCalculator.calculateFor(connectionSession)));
56      }
57  }
