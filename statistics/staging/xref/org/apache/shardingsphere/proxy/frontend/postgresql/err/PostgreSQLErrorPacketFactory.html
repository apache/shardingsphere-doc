18  package org.apache.shardingsphere.proxy.frontend.postgresql.err;
19  
20  import com.google.common.base.Strings;
21  import lombok.AccessLevel;
22  import lombok.NoArgsConstructor;
23  import org.apache.shardingsphere.db.protocol.postgresql.constant.PostgreSQLMessageSeverityLevel;
24  import org.apache.shardingsphere.db.protocol.postgresql.packet.generic.PostgreSQLErrorResponsePacket;
25  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
26  import org.apache.shardingsphere.infra.exception.core.external.sql.sqlstate.XOpenSQLState;
27  import org.apache.shardingsphere.infra.exception.dialect.SQLExceptionTransformEngine;
28  import org.apache.shardingsphere.infra.exception.postgresql.exception.PostgreSQLException;
29  import org.apache.shardingsphere.infra.exception.postgresql.vendor.PostgreSQLVendorError;
30  import org.apache.shardingsphere.infra.spi.type.typed.TypedSPILoader;
31  import org.postgresql.util.PSQLException;
32  import org.postgresql.util.ServerErrorMessage;
33  
34  import java.sql.SQLException;
35  import java.util.Optional;
36  
37  /**
38   * Error packet factory for PostgreSQL.
39   */
40  @NoArgsConstructor(access = AccessLevel.PRIVATE)
41  public final class PostgreSQLErrorPacketFactory {
42      
43      private static final DatabaseType DATABASE_TYPE = TypedSPILoader.getService(DatabaseType.class, "PostgreSQL");
44      
45      /**
46       * Create new instance of PostgreSQL error packet.
47       *
48       * @param cause cause
49       * @return created instance
50       */
51      public static PostgreSQLErrorResponsePacket newInstance(final Exception cause) {
52          Optional<ServerErrorMessage> serverErrorMessage = findServerErrorMessage(cause);
53          return serverErrorMessage.map(PostgreSQLErrorPacketFactory::createErrorResponsePacket)
54                  .orElseGet(() -> createErrorResponsePacket(SQLExceptionTransformEngine.toSQLException(cause, DATABASE_TYPE)));
55      }
56      
57      private static Optional<ServerErrorMessage> findServerErrorMessage(final Exception cause) {
58          return cause instanceof PSQLException ? Optional.ofNullable(((PSQLException) cause).getServerErrorMessage()) : Optional.empty();
59      }
60      
61      private static PostgreSQLErrorResponsePacket createErrorResponsePacket(final ServerErrorMessage serverErrorMessage) {
62          return PostgreSQLErrorResponsePacket.newBuilder(serverErrorMessage.getSeverity(), serverErrorMessage.getSQLState(), serverErrorMessage.getMessage())
63                  .detail(serverErrorMessage.getDetail()).hint(serverErrorMessage.getHint()).position(serverErrorMessage.getPosition())
64                  .internalQueryAndInternalPosition(serverErrorMessage.getInternalQuery(), serverErrorMessage.getInternalPosition()).where(serverErrorMessage.getWhere())
65                  .schemaName(serverErrorMessage.getSchema()).tableName(serverErrorMessage.getTable()).columnName(serverErrorMessage.getColumn()).dataTypeName(serverErrorMessage.getDatatype())
66                  .constraintName(serverErrorMessage.getConstraint()).file(serverErrorMessage.getFile()).line(serverErrorMessage.getLine()).routine(serverErrorMessage.getRoutine()).build();
67      }
68      
69      private static PostgreSQLErrorResponsePacket createErrorResponsePacket(final SQLException cause) {
70          if (cause instanceof PostgreSQLException && null != ((PostgreSQLException) cause).getServerErrorMessage()) {
71              return createErrorResponsePacket(((PostgreSQLException) cause).getServerErrorMessage());
72          }
73          if (cause instanceof PSQLException && null != ((PSQLException) cause).getServerErrorMessage()) {
74              return createErrorResponsePacket(((PSQLException) cause).getServerErrorMessage());
75          }
76          String sqlState = Strings.isNullOrEmpty(cause.getSQLState()) || XOpenSQLState.GENERAL_ERROR.getValue().equals(cause.getSQLState())
77                  ? PostgreSQLVendorError.SYSTEM_ERROR.getSqlState().getValue()
78                  : cause.getSQLState();
79          String message = Strings.isNullOrEmpty(cause.getMessage()) ? cause.toString() : cause.getMessage();
80          return PostgreSQLErrorResponsePacket.newBuilder(PostgreSQLMessageSeverityLevel.ERROR, sqlState, message).build();
81      }
82      
83      private static PostgreSQLErrorResponsePacket createErrorResponsePacket(final PostgreSQLException.ServerErrorMessage serverErrorMessage) {
84          return PostgreSQLErrorResponsePacket.newBuilder(serverErrorMessage.getSeverity(), serverErrorMessage.getSqlState(), serverErrorMessage.getMessage()).build();
85      }
86  }
