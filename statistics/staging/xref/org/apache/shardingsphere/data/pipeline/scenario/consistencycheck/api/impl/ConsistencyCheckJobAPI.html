93      /**
94       * Create consistency check configuration and start job.
95       *
96       * @param param create consistency check job parameter
97       * @return job id
98       * @throws UncompletedConsistencyCheckJobExistsException uncompleted consistency check job exists exception
99       */
100     public String createJobAndStart(final CreateConsistencyCheckJobParameter param) {
101         String parentJobId = param.getParentJobId();
102         GovernanceRepositoryAPI repositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId));
103         Optional<String> latestCheckJobId = repositoryAPI.getLatestCheckJobId(parentJobId);
104         if (latestCheckJobId.isPresent()) {
105             Optional<ConsistencyCheckJobItemProgress> progress = getJobItemProgress(latestCheckJobId.get(), 0);
106             if (!progress.isPresent() || JobStatus.FINISHED != progress.get().getStatus()) {
107                 log.info("check job already exists and status is not FINISHED, progress={}", progress);
108                 throw new UncompletedConsistencyCheckJobExistsException(latestCheckJobId.get());
109             }
110         }
111         PipelineContextKey contextKey = PipelineJobIdUtils.parseContextKey(parentJobId);
112         String result = marshalJobId(latestCheckJobId.map(s -> new ConsistencyCheckJobId(contextKey, parentJobId, s)).orElseGet(() -> new ConsistencyCheckJobId(contextKey, parentJobId)));
113         repositoryAPI.persistLatestCheckJobId(parentJobId, result);
114         repositoryAPI.deleteCheckJobResult(parentJobId, result);
115         dropJob(result);
116         YamlConsistencyCheckJobConfiguration yamlConfig = new YamlConsistencyCheckJobConfiguration();
117         yamlConfig.setJobId(result);
118         yamlConfig.setParentJobId(parentJobId);
119         yamlConfig.setAlgorithmTypeName(param.getAlgorithmTypeName());
120         yamlConfig.setAlgorithmProps(param.getAlgorithmProps());
121         start(new YamlConsistencyCheckJobConfigurationSwapper().swapToObject(yamlConfig));
122         return result;
123     }

129      * @return latest data consistency check result

131     public Map<String, DataConsistencyCheckResult> getLatestDataConsistencyCheckResult(final String parentJobId) {
132         GovernanceRepositoryAPI governanceRepositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId));
133         Optional<String> latestCheckJobId = governanceRepositoryAPI.getLatestCheckJobId(parentJobId);
134         if (!latestCheckJobId.isPresent()) {
135             return Collections.emptyMap();
136         }
137         return governanceRepositoryAPI.getCheckJobResult(parentJobId, latestCheckJobId.get());
138     }

142         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobItemContext.getJobId()))
143                 .persistJobItemProgress(jobItemContext.getJobId(), jobItemContext.getShardingItem(), convertJobItemProgress(jobItemContext));

146     private String convertJobItemProgress(final PipelineJobItemContext jobItemContext) {
147         ConsistencyCheckJobItemContext context = (ConsistencyCheckJobItemContext) jobItemContext;
148         ConsistencyCheckJobItemProgressContext progressContext = context.getProgressContext();
149         String tableNames = String.join(",", progressContext.getTableNames());
150         String ignoredTableNames = String.join(",", progressContext.getIgnoredTableNames());
151         ConsistencyCheckJobItemProgress jobItemProgress = new ConsistencyCheckJobItemProgress(tableNames, ignoredTableNames, progressContext.getCheckedRecordsCount().get(),
152                 progressContext.getRecordsCount(), progressContext.getCheckBeginTimeMillis(), progressContext.getCheckEndTimeMillis(), progressContext.getTableCheckPositions());
153         jobItemProgress.setStatus(context.getStatus());
154         return YamlEngine.marshal(swapper.swapToYamlConfiguration(jobItemProgress));
155     }

169     @Override
170     public void updateJobItemStatus(final String jobId, final int shardingItem, final JobStatus status) {
171         Optional<ConsistencyCheckJobItemProgress> jobItemProgress = getJobItemProgress(jobId, shardingItem);
172         if (!jobItemProgress.isPresent()) {
173             log.warn("updateJobItemStatus, jobProgress is null, jobId={}, shardingItem={}", jobId, shardingItem);
174             return;
175         }
176         jobItemProgress.get().setStatus(status);
177         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).updateJobItemProgress(jobId, shardingItem,
178                 YamlEngine.marshal(swapper.swapToYamlConfiguration(jobItemProgress.get())));
179     }

181     @Override
182     public void startDisabledJob(final String jobId) {
183         Optional<ConsistencyCheckJobItemProgress> jobItemProgress = getJobItemProgress(jobId, 0);
184         if (jobItemProgress.isPresent() && JobStatus.FINISHED == jobItemProgress.get().getStatus()) {
185             log.info("job status is FINISHED, ignore, jobId={}", jobId);
186             return;
187         }
188         super.startDisabledJob(jobId);
189     }

200     private String getLatestCheckJobId(final String parentJobId) {
201         Optional<String> result = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId)).getLatestCheckJobId(parentJobId);
202         ShardingSpherePreconditions.checkState(result.isPresent(), () -> new ConsistencyCheckJobNotFoundException(parentJobId));
203         return result.get();

211     public void stopByParentJobId(final String parentJobId) {
212         stop(getLatestCheckJobId(parentJobId));
213     }

215     /**
216      * Drop by parent job id.
217      *
218      * @param parentJobId parent job id
219      */
220     public void dropByParentJobId(final String parentJobId) {
221         String latestCheckJobId = getLatestCheckJobId(parentJobId);
222         stop(latestCheckJobId);
223         PipelineContextKey contextKey = PipelineJobIdUtils.parseContextKey(parentJobId);
224         GovernanceRepositoryAPI repositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(contextKey);
225         Collection<String> checkJobIds = repositoryAPI.listCheckJobIds(parentJobId);
226         Optional<Integer> previousSequence = ConsistencyCheckSequence.getPreviousSequence(
227                 checkJobIds.stream().map(ConsistencyCheckJobId::parseSequence).collect(Collectors.toList()), ConsistencyCheckJobId.parseSequence(latestCheckJobId));
228         if (previousSequence.isPresent()) {
229             String checkJobId = marshalJobId(new ConsistencyCheckJobId(contextKey, parentJobId, previousSequence.get()));
230             repositoryAPI.persistLatestCheckJobId(parentJobId, checkJobId);
231         } else {
232             repositoryAPI.deleteLatestCheckJobId(parentJobId);
233         }
234         repositoryAPI.deleteCheckJobResult(parentJobId, latestCheckJobId);
235         dropJob(latestCheckJobId);
236     }

238     /**
239      * Get consistency job item infos.
240      *
241      * @param parentJobId parent job id
242      * @return consistency job item infos
243      */
244     public List<ConsistencyCheckJobItemInfo> getJobItemInfos(final String parentJobId) {
245         GovernanceRepositoryAPI governanceRepositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId));
246         Optional<String> latestCheckJobId = governanceRepositoryAPI.getLatestCheckJobId(parentJobId);
247         ShardingSpherePreconditions.checkState(latestCheckJobId.isPresent(), () -> new ConsistencyCheckJobNotFoundException(parentJobId));
248         String checkJobId = latestCheckJobId.get();
249         Optional<ConsistencyCheckJobItemProgress> progressOptional = getJobItemProgress(checkJobId, 0);
250         if (!progressOptional.isPresent()) {
251             return Collections.emptyList();
252         }
253         List<ConsistencyCheckJobItemInfo> result = new LinkedList<>();
254         ConsistencyCheckJobItemProgress jobItemProgress = progressOptional.get();
255         if (!Strings.isNullOrEmpty(jobItemProgress.getIgnoredTableNames())) {
256             Map<String, DataConsistencyCheckResult> checkJobResult = governanceRepositoryAPI.getCheckJobResult(parentJobId, latestCheckJobId.get());
257             result.addAll(buildIgnoredTableInfo(jobItemProgress.getIgnoredTableNames().split(","), checkJobResult));
258         }
259         if (Objects.equals(jobItemProgress.getIgnoredTableNames(), jobItemProgress.getTableNames())) {
260             return result;
261         }
262         result.add(getJobItemInfo(parentJobId));
263         return result;
264     }
265     
266     private List<ConsistencyCheckJobItemInfo> buildIgnoredTableInfo(final String[] ignoredTables, final Map<String, DataConsistencyCheckResult> checkJobResult) {
267         if (null == ignoredTables) {
268             return Collections.emptyList();
269         }
270         List<ConsistencyCheckJobItemInfo> result = new LinkedList<>();
271         for (String each : ignoredTables) {
272             ConsistencyCheckJobItemInfo info = new ConsistencyCheckJobItemInfo();
273             info.setTableNames(each);
274             info.setCheckSuccess(null);
275             DataConsistencyCheckResult checkResult = checkJobResult.get(each);
276             if (null != checkResult && checkResult.isIgnored()) {
277                 info.setErrorMessage(checkResult.getIgnoredType().getMessage());
278             }
279             result.add(info);
280         }
281         return result;
282     }

284     private ConsistencyCheckJobItemInfo getJobItemInfo(final String parentJobId) {
285         GovernanceRepositoryAPI governanceRepositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId));
286         Optional<String> latestCheckJobId = governanceRepositoryAPI.getLatestCheckJobId(parentJobId);
287         ShardingSpherePreconditions.checkState(latestCheckJobId.isPresent(), () -> new ConsistencyCheckJobNotFoundException(parentJobId));
288         String checkJobId = latestCheckJobId.get();
289         Optional<ConsistencyCheckJobItemProgress> progressOptional = getJobItemProgress(checkJobId, 0);
290         ConsistencyCheckJobItemInfo result = new ConsistencyCheckJobItemInfo();
291         if (!progressOptional.isPresent()) {
292             return result;
293         }
294         ConsistencyCheckJobItemProgress jobItemProgress = progressOptional.get();
295         if (null == jobItemProgress.getRecordsCount() || null == jobItemProgress.getCheckedRecordsCount()) {
296             result.setFinishedPercentage(0);
297             result.setCheckSuccess(null);
298             return result;
299         }
300         LocalDateTime checkBeginTime = new Timestamp(jobItemProgress.getCheckBeginTimeMillis()).toLocalDateTime();
301         long recordsCount = jobItemProgress.getRecordsCount();
302         long checkedRecordsCount = Math.min(jobItemProgress.getCheckedRecordsCount(), recordsCount);
303         if (JobStatus.FINISHED == jobItemProgress.getStatus()) {
304             result.setFinishedPercentage(100);
305             LocalDateTime checkEndTime = new Timestamp(jobItemProgress.getCheckEndTimeMillis()).toLocalDateTime();
306             Duration duration = Duration.between(checkBeginTime, checkEndTime);
307             result.setDurationSeconds(duration.getSeconds());
308             result.setCheckEndTime(DATE_TIME_FORMATTER.format(checkEndTime));
309             result.setRemainingSeconds(0L);
310         } else if (0 != recordsCount && 0 != checkedRecordsCount) {
311             result.setFinishedPercentage((int) (checkedRecordsCount * 100 / recordsCount));
312             JobConfigurationPOJO jobConfigPOJO = getElasticJobConfigPOJO(checkJobId);
313             Long stopTimeMillis = jobConfigPOJO.isDisabled() ? Long.parseLong(jobConfigPOJO.getProps().getProperty("stop_time_millis")) : null;
314             long durationMillis = (null != stopTimeMillis ? stopTimeMillis : System.currentTimeMillis()) - jobItemProgress.getCheckBeginTimeMillis();
315             result.setDurationSeconds(TimeUnit.MILLISECONDS.toSeconds(durationMillis));
316             if (null != stopTimeMillis) {
317                 result.setCheckEndTime(DATE_TIME_FORMATTER.format(new Timestamp(stopTimeMillis).toLocalDateTime()));
318             }
319             long remainingMills = Math.max(0, (long) ((recordsCount - checkedRecordsCount) * 1.0D / checkedRecordsCount * durationMillis));
320             result.setRemainingSeconds(remainingMills / 1000);
321         }
322         String tableNames = jobItemProgress.getTableNames();
323         result.setTableNames(Optional.ofNullable(tableNames).orElse(""));
324         result.setCheckBeginTime(DATE_TIME_FORMATTER.format(checkBeginTime));
325         result.setErrorMessage(getJobItemErrorMessage(checkJobId, 0));
326         Map<String, DataConsistencyCheckResult> checkJobResult = governanceRepositoryAPI.getCheckJobResult(parentJobId, checkJobId);
327         if (checkJobResult.isEmpty()) {
328             result.setCheckSuccess(null);
329         } else {
330             InventoryIncrementalJobAPI inventoryIncrementalJobAPI = (InventoryIncrementalJobAPI) TypedSPILoader.getService(
331                     PipelineJobAPI.class, PipelineJobIdUtils.parseJobType(parentJobId).getType());
332             result.setCheckSuccess(inventoryIncrementalJobAPI.aggregateDataConsistencyCheckResults(parentJobId, checkJobResult));
333         }
334         result.setCheckFailedTableNames(checkJobResult.entrySet().stream().filter(each -> !each.getValue().isIgnored() && !each.getValue().isMatched())
335                 .map(Entry::getKey).collect(Collectors.joining(",")));
336         return result;
337     }
