48  /**
49   * Auto repository tuple swapper engine.
50   */
51  public final class AutoRepositoryTupleSwapperEngine {
52      
53      /**
54       * Swap to repository tuples.
55       *
56       * @param yamlRuleConfig YAML rule configuration to be swapped
57       * @return repository tuples
58       */
59      public Collection<RepositoryTuple> swapToRepositoryTuples(final YamlRuleConfiguration yamlRuleConfig) {
60          if (null == yamlRuleConfig.getClass().getAnnotation(RepositoryTupleEntity.class)) {
61              return Collections.emptyList();
62          }
63          RepositoryTupleType tupleType = yamlRuleConfig.getClass().getAnnotation(RepositoryTupleType.class);
64          if (null != tupleType) {
65              return Collections.singleton(new RepositoryTuple(tupleType.value(), YamlEngine.marshal(yamlRuleConfig)));
66          }
67          Collection<RepositoryTuple> result = new LinkedList<>();
68          RuleNodePath ruleNodePath = TypedSPILoader.getService(RuleNodePathProvider.class, yamlRuleConfig.getRuleConfigurationType()).getRuleNodePath();
69          for (Field each : getFields(yamlRuleConfig.getClass())) {
70              @SuppressWarnings("deprecation")
71              boolean isAccessible = each.isAccessible();
72              each.setAccessible(true);
73              result.addAll(swapToRepositoryTuples(yamlRuleConfig, ruleNodePath, each));
74              each.setAccessible(isAccessible);
75          }
76          return result;
77      }
78      
79      @SneakyThrows(ReflectiveOperationException.class)
80      @SuppressWarnings("rawtypes")
81      private Collection<RepositoryTuple> swapToRepositoryTuples(final YamlRuleConfiguration yamlRuleConfig, final RuleNodePath ruleNodePath, final Field field) {
82          Object fieldValue = field.get(yamlRuleConfig);
83          if (null == fieldValue) {
84              return Collections.emptyList();
85          }
86          RepositoryTupleField tupleField = field.getAnnotation(RepositoryTupleField.class);
87          RepositoryTupleKeyNameGenerator tupleKeyNameGenerator = field.getAnnotation(RepositoryTupleKeyNameGenerator.class);
88          if (null != tupleKeyNameGenerator && fieldValue instanceof Collection) {
89              Collection<RepositoryTuple> result = new LinkedList<>();
90              for (Object value : (Collection) fieldValue) {
91                  String tupleKeyName = tupleKeyNameGenerator.value().getConstructor().newInstance().generate(value);
92                  result.add(new RepositoryTuple(ruleNodePath.getNamedItem(tupleField.value()).getPath(tupleKeyName), value.toString()));
93              }
94              return result;
95          }
96          if (fieldValue instanceof Map) {
97              Collection<RepositoryTuple> result = new LinkedList<>();
98              for (Object entry : ((Map) fieldValue).entrySet()) {
99                  result.add(new RepositoryTuple(ruleNodePath.getNamedItem(tupleField.value()).getPath(((Entry) entry).getKey().toString()), YamlEngine.marshal(((Entry) entry).getValue())));
100             }
101             return result;
102         }
103         if (fieldValue instanceof Collection && !((Collection) fieldValue).isEmpty()) {
104             return Collections.singleton(new RepositoryTuple(ruleNodePath.getUniqueItem(tupleField.value()).getPath(), YamlEngine.marshal(fieldValue)));
105         }
106         if (fieldValue instanceof String && !((String) fieldValue).isEmpty()) {
107             return Collections.singleton(new RepositoryTuple(ruleNodePath.getUniqueItem(tupleField.value()).getPath(), fieldValue.toString()));
108         }
109         if (fieldValue instanceof Boolean || fieldValue instanceof Integer || fieldValue instanceof Long) {
110             return Collections.singleton(new RepositoryTuple(ruleNodePath.getUniqueItem(tupleField.value()).getPath(), fieldValue.toString()));
111         }
112         return Collections.singleton(new RepositoryTuple(ruleNodePath.getUniqueItem(tupleField.value()).getPath(), YamlEngine.marshal(fieldValue)));
113     }
114     
115     private Collection<Field> getFields(final Class<? extends YamlRuleConfiguration> yamlRuleConfigurationClass) {
116         return Arrays.stream(yamlRuleConfigurationClass.getDeclaredFields())
117                 .filter(each -> null != each.getAnnotation(RepositoryTupleField.class))
118                 .sorted(Comparator.comparingInt(o -> o.getAnnotation(RepositoryTupleField.class).order())).collect(Collectors.toList());
119     }

121     /**
122      * Swap from repository tuple to YAML rule configurations.
123      *
124      * @param repositoryTuples repository tuples
125      * @param yamlRuleConfigurationClass YAML rule configuration class
126      * @return swapped YAML rule configurations
127      */
128     public Optional<YamlRuleConfiguration> swapToObject(final Collection<RepositoryTuple> repositoryTuples, final Class<? extends YamlRuleConfiguration> yamlRuleConfigurationClass) {
129         if (null == yamlRuleConfigurationClass.getAnnotation(RepositoryTupleEntity.class)) {
130             return Optional.empty();
131         }
132         RepositoryTupleType tupleType = yamlRuleConfigurationClass.getAnnotation(RepositoryTupleType.class);
133         return null == tupleType
134                 ? swapToObject(repositoryTuples, yamlRuleConfigurationClass, getFields(yamlRuleConfigurationClass))
135                 : swapToObject(repositoryTuples, yamlRuleConfigurationClass, tupleType);
136     }
137     
138     @SneakyThrows(ReflectiveOperationException.class)
139     private Optional<YamlRuleConfiguration> swapToObject(final Collection<RepositoryTuple> repositoryTuples,
140                                                          final Class<? extends YamlRuleConfiguration> yamlRuleConfigurationClass, final RepositoryTupleType tupleType) {
141         if (YamlGlobalRuleConfiguration.class.isAssignableFrom(yamlRuleConfigurationClass)) {
142             for (RepositoryTuple each : repositoryTuples) {
143                 if (GlobalNodePath.getVersion(tupleType.value(), each.getKey()).isPresent()) {
144                     return Optional.of(YamlEngine.unmarshal(each.getValue(), yamlRuleConfigurationClass));
145                 }
146             }
147             return Optional.empty();
148         }
149         YamlRuleConfiguration yamlRuleConfig = yamlRuleConfigurationClass.getConstructor().newInstance();
150         RuleNodePath ruleNodePath = TypedSPILoader.getService(RuleNodePathProvider.class, yamlRuleConfig.getRuleConfigurationType()).getRuleNodePath();
151         for (RepositoryTuple each : repositoryTuples.stream().filter(each -> ruleNodePath.getRoot().isValidatedPath(each.getKey())).collect(Collectors.toList())) {
152             if (ruleNodePath.getUniqueItem(tupleType.value()).isValidatedPath(each.getKey())) {
153                 return Optional.of(YamlEngine.unmarshal(each.getValue(), yamlRuleConfigurationClass));
154             }
155         }
156         return Optional.empty();
157     }
158     
159     @SneakyThrows(ReflectiveOperationException.class)
160     private Optional<YamlRuleConfiguration> swapToObject(final Collection<RepositoryTuple> repositoryTuples,
161                                                          final Class<? extends YamlRuleConfiguration> yamlRuleConfigurationClass, final Collection<Field> fields) {
162         YamlRuleConfiguration yamlRuleConfig = yamlRuleConfigurationClass.getConstructor().newInstance();
163         RuleNodePath ruleNodePath = TypedSPILoader.getService(RuleNodePathProvider.class, yamlRuleConfig.getRuleConfigurationType()).getRuleNodePath();
164         List<RepositoryTuple> validTuples = repositoryTuples.stream().filter(each -> ruleNodePath.getRoot().isValidatedPath(each.getKey())).collect(Collectors.toList());
165         if (validTuples.isEmpty()) {
166             return Optional.empty();
167         }
168         for (RepositoryTuple each : validTuples) {
169             if (!Strings.isNullOrEmpty(each.getValue())) {
170                 setFieldValue(yamlRuleConfig, fields, ruleNodePath, each);
171             }
172         }
173         return Optional.of(yamlRuleConfig);
174     }
175     
176     @SneakyThrows(ReflectiveOperationException.class)
177     private void setFieldValue(final YamlRuleConfiguration yamlRuleConfig, final Collection<Field> fields, final RuleNodePath ruleNodePath, final RepositoryTuple repositoryTuple) {
178         for (Field each : fields) {
179             @SuppressWarnings("deprecation")
180             boolean isAccessible = each.isAccessible();
181             each.setAccessible(true);
182             setFieldValue(yamlRuleConfig, each, ruleNodePath, repositoryTuple);
183             each.setAccessible(isAccessible);
184         }
185     }
186     
187     @SuppressWarnings({"unchecked", "rawtypes"})
188     private void setFieldValue(final YamlRuleConfiguration yamlRuleConfig, final Field field, final RuleNodePath ruleNodePath, final RepositoryTuple repositoryTuple) throws IllegalAccessException {
189         Object fieldValue = field.get(yamlRuleConfig);
190         RepositoryTupleField tupleField = field.getAnnotation(RepositoryTupleField.class);
191         RepositoryTupleKeyNameGenerator tupleKeyNameGenerator = field.getAnnotation(RepositoryTupleKeyNameGenerator.class);
192         if (null != tupleKeyNameGenerator && fieldValue instanceof Collection) {
193             ruleNodePath.getNamedItem(tupleField.value()).getName(repositoryTuple.getKey()).ifPresent(optional -> ((Collection) fieldValue).add(repositoryTuple.getValue()));
194             return;
195         }
196         if (fieldValue instanceof Map) {
197             Class<?> valueClass = (Class) ((ParameterizedType) field.getGenericType()).getActualTypeArguments()[1];
198             ruleNodePath.getNamedItem(tupleField.value()).getName(repositoryTuple.getKey())
199                     .ifPresent(optional -> ((Map) fieldValue).put(optional, YamlEngine.unmarshal(repositoryTuple.getValue(), valueClass)));
200             return;
201         }
202         if (fieldValue instanceof Collection) {
203             if (ruleNodePath.getUniqueItem(tupleField.value()).isValidatedPath(repositoryTuple.getKey())) {
204                 field.set(yamlRuleConfig, YamlEngine.unmarshal(repositoryTuple.getValue(), List.class));
205             }
206             return;
207         }
208         if (field.getType().equals(String.class)) {
209             if (ruleNodePath.getUniqueItem(tupleField.value()).isValidatedPath(repositoryTuple.getKey())) {
210                 field.set(yamlRuleConfig, repositoryTuple.getValue());
211             }
212             return;
213         }
214         if (field.getType().equals(boolean.class) || field.getType().equals(Boolean.class)) {
215             if (ruleNodePath.getUniqueItem(tupleField.value()).isValidatedPath(repositoryTuple.getKey())) {
216                 field.set(yamlRuleConfig, Boolean.parseBoolean(repositoryTuple.getValue()));
217             }
218             return;
219         }
220         if (field.getType().equals(int.class) || field.getType().equals(Integer.class)) {
221             if (ruleNodePath.getUniqueItem(tupleField.value()).isValidatedPath(repositoryTuple.getKey())) {
222                 field.set(yamlRuleConfig, Integer.parseInt(repositoryTuple.getValue()));
223             }
224             return;
225         }
226         if (field.getType().equals(long.class) || field.getType().equals(Long.class)) {
227             if (ruleNodePath.getUniqueItem(tupleField.value()).isValidatedPath(repositoryTuple.getKey())) {
228                 field.set(yamlRuleConfig, Long.parseLong(repositoryTuple.getValue()));
229             }
230             return;
231         }
232         if (ruleNodePath.getUniqueItem(tupleField.value()).isValidatedPath(repositoryTuple.getKey())) {
233             field.set(yamlRuleConfig, YamlEngine.unmarshal(repositoryTuple.getValue(), field.getType()));
234         }
235     }
