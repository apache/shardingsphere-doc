27  import javax.sql.DataSource;
28  import java.sql.Connection;
29  import java.sql.DatabaseMetaData;
30  import java.sql.PreparedStatement;
31  import java.sql.ResultSet;
32  import java.sql.SQLException;
33  import java.util.Collection;
34  import java.util.Collections;
35  import java.util.HashMap;

38  import java.util.Map.Entry;
39  import java.util.stream.Collectors;

41  /**
42   * Schema meta data loader for SQLServer.
43   */
44  public final class SQLServerSchemaMetaDataLoader implements DialectSchemaMetaDataLoader {
45      
46      private static final String TABLE_META_DATA_SQL_NO_ORDER = "SELECT obj.name AS TABLE_NAME, col.name AS COLUMN_NAME, t.name AS DATA_TYPE,"
47              + " col.collation_name AS COLLATION_NAME, col.column_id, is_identity AS IS_IDENTITY, %s"
48              + " (SELECT TOP 1 ind.is_primary_key FROM sys.index_columns ic LEFT JOIN sys.indexes ind ON ic.object_id = ind.object_id"
49              + " AND ic.index_id = ind.index_id AND ind.name LIKE 'PK_%%' WHERE ic.object_id = obj.object_id AND ic.column_id = col.column_id) AS IS_PRIMARY_KEY"
50              + " FROM sys.objects obj INNER JOIN sys.columns col ON obj.object_id = col.object_id LEFT JOIN sys.types t ON t.user_type_id = col.user_type_id";
51      
52      private static final String ORDER_BY_COLUMN_ID = " ORDER BY col.column_id";
53      
54      private static final String TABLE_META_DATA_SQL = TABLE_META_DATA_SQL_NO_ORDER + ORDER_BY_COLUMN_ID;
55      
56      private static final String TABLE_META_DATA_SQL_IN_TABLES = TABLE_META_DATA_SQL_NO_ORDER + " WHERE obj.name IN (%s)" + ORDER_BY_COLUMN_ID;
57      
58      private static final String INDEX_META_DATA_SQL = "SELECT a.name AS INDEX_NAME, c.name AS TABLE_NAME FROM sys.indexes a"
59              + " JOIN sys.objects c ON a.object_id = c.object_id WHERE a.index_id NOT IN (0, 255) AND c.name IN (%s)";
60      
61      private static final int HIDDEN_COLUMN_START_MAJOR_VERSION = 15;
62      
63      @Override
64      public Collection<SchemaMetaData> load(final DataSource dataSource, final Collection<String> tables, final String defaultSchemaName) throws SQLException {
65          Collection<TableMetaData> tableMetaDataList = new LinkedList<>();
66          Map<String, Collection<ColumnMetaData>> columnMetaDataMap = loadColumnMetaDataMap(dataSource, tables);
67          if (!columnMetaDataMap.isEmpty()) {
68              Map<String, Collection<IndexMetaData>> indexMetaDataMap = loadIndexMetaData(dataSource, columnMetaDataMap.keySet());
69              for (Entry<String, Collection<ColumnMetaData>> entry : columnMetaDataMap.entrySet()) {
70                  Collection<IndexMetaData> indexMetaDataList = indexMetaDataMap.getOrDefault(entry.getKey(), Collections.emptyList());
71                  tableMetaDataList.add(new TableMetaData(entry.getKey(), entry.getValue(), indexMetaDataList, Collections.emptyList()));
72              }
73          }
74          return Collections.singleton(new SchemaMetaData(defaultSchemaName, tableMetaDataList));
75      }
76      
77      private Map<String, Collection<ColumnMetaData>> loadColumnMetaDataMap(final DataSource dataSource, final Collection<String> tables) throws SQLException {
78          Map<String, Collection<ColumnMetaData>> result = new HashMap<>();
79          try (
80                  Connection connection = dataSource.getConnection();
81                  PreparedStatement preparedStatement = connection.prepareStatement(getTableMetaDataSQL(tables, connection.getMetaData()))) {
82              Map<String, Integer> dataTypes = new DataTypeLoader().load(connection.getMetaData(), getType());
83              try (ResultSet resultSet = preparedStatement.executeQuery()) {
84                  while (resultSet.next()) {
85                      String tableName = resultSet.getString("TABLE_NAME");
86                      ColumnMetaData columnMetaData = loadColumnMetaData(dataTypes, resultSet, connection.getMetaData());
87                      if (!result.containsKey(tableName)) {
88                          result.put(tableName, new LinkedList<>());
89                      }
90                      result.get(tableName).add(columnMetaData);
91                  }
92              }
93          }
94          return result;
95      }
96      
97      private ColumnMetaData loadColumnMetaData(final Map<String, Integer> dataTypeMap, final ResultSet resultSet, final DatabaseMetaData databaseMetaData) throws SQLException {
98          String columnName = resultSet.getString("COLUMN_NAME");
99          String dataType = resultSet.getString("DATA_TYPE");
100         String collationName = resultSet.getString("COLLATION_NAME");
101         boolean primaryKey = "1".equals(resultSet.getString("IS_PRIMARY_KEY"));
102         boolean generated = "1".equals(resultSet.getString("IS_IDENTITY"));
103         boolean caseSensitive = null != collationName && collationName.contains("_CS");
104         boolean isVisible = !(versionContainsHiddenColumn(databaseMetaData) && "1".equals(resultSet.getString("IS_HIDDEN")));
105         return new ColumnMetaData(columnName, dataTypeMap.get(dataType), primaryKey, generated, caseSensitive, isVisible, false);
106     }
107     
108     private String getTableMetaDataSQL(final Collection<String> tables, final DatabaseMetaData databaseMetaData) throws SQLException {
109         StringBuilder stringBuilder = new StringBuilder(24);
110         if (versionContainsHiddenColumn(databaseMetaData)) {
111             stringBuilder.append("is_hidden AS IS_HIDDEN,");
112         }
113         String hiddenFlag = stringBuilder.toString();
114         return tables.isEmpty() ? String.format(TABLE_META_DATA_SQL, hiddenFlag)
115                 : String.format(TABLE_META_DATA_SQL_IN_TABLES, hiddenFlag, tables.stream().map(each -> String.format("'%s'", each)).collect(Collectors.joining(",")));
116     }
117     
118     private boolean versionContainsHiddenColumn(final DatabaseMetaData databaseMetaData) throws SQLException {
119         return databaseMetaData.getDatabaseMajorVersion() >= HIDDEN_COLUMN_START_MAJOR_VERSION;
120     }
121     
122     private Map<String, Collection<IndexMetaData>> loadIndexMetaData(final DataSource dataSource, final Collection<String> tableNames) throws SQLException {
123         Map<String, Collection<IndexMetaData>> result = new HashMap<>();
124         try (
125                 Connection connection = dataSource.getConnection();
126                 PreparedStatement preparedStatement = connection.prepareStatement(getIndexMetaDataSQL(tableNames))) {
127             try (ResultSet resultSet = preparedStatement.executeQuery()) {
128                 while (resultSet.next()) {
129                     String indexName = resultSet.getString("INDEX_NAME");
130                     String tableName = resultSet.getString("TABLE_NAME");
131                     if (!result.containsKey(tableName)) {
132                         result.put(tableName, new LinkedList<>());
133                     }
134                     result.get(tableName).add(new IndexMetaData(indexName));
135                 }
136             }
137         }
138         return result;
139     }
140     
141     private String getIndexMetaDataSQL(final Collection<String> tableNames) {
142         return String.format(INDEX_META_DATA_SQL, tableNames.stream().map(each -> String.format("'%s'", each)).collect(Collectors.joining(",")));
143     }
144     
145     @Override
146     public String getDatabaseType() {
147         return "SQLServer";
148     }
149 }
