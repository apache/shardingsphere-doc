30  /**
31   * Kernel supported system tables.
32   */
33  @RequiredArgsConstructor
34  @Getter
35  public enum KernelSupportedSystemTables {
36      
37      MYSQL_SYS("MySQL", "sys", new HashSet<>(Collections.singleton("sys_config"))),
38      
39      MYSQL_SHARDING_SPHERE("MySQL", "shardingsphere", new HashSet<>(Arrays.asList("sharding_table_statistics", "cluster_information"))),
40      
41      POSTGRESQL_INFORMATION_SCHEMA("PostgreSQL", "information_schema", new HashSet<>(Arrays.asList("columns", "tables", "views"))),
42      
43      POSTGRESQL_PG_CATALOG("PostgreSQL", "pg_catalog", new HashSet<>(Arrays.asList("pg_aggregate", "pg_class", "pg_database", "pg_tables", "pg_inherits",
44              "pg_tablespace", "pg_trigger", "pg_namespace", "pg_roles"))),
45      
46      POSTGRESQL_SHARDING_SPHERE("PostgreSQL", "shardingsphere", new HashSet<>(Arrays.asList("sharding_table_statistics", "cluster_information"))),
47      
48      OPEN_GAUSS_INFORMATION_SCHEMA("openGauss", "information_schema", Collections.emptySet()),
49      
50      OPEN_GAUSS_PG_CATALOG("openGauss", "pg_catalog", new HashSet<>(Arrays.asList("pg_class", "pg_namespace", "pg_database", "pg_roles", "pg_tables", "pg_tablespace"))),
51      
52      OPEN_GAUSS_BLOCKCHAIN("openGauss", "blockchain", Collections.emptySet()),
53      
54      OPEN_GAUSS_CSTORE("openGauss", "cstore", Collections.emptySet()),
55      
56      OPEN_GAUSS_DB4AI("openGauss", "db4ai", Collections.emptySet()),
57      
58      OPEN_GAUSS_DBE_PERF("openGauss", "dbe_perf", Collections.emptySet()),
59      
60      OPEN_GAUSS_DBE_PLDEBUGGER("openGauss", "dbe_pldebugger", Collections.emptySet()),
61      
62      OPEN_GAUSS_GAUSSDB("openGauss", "gaussdb", Collections.emptySet()),
63      
64      OPEN_GAUSS_ORACLE("openGauss", "oracle", Collections.emptySet()),
65      
66      OPEN_GAUSS_PKG_SERVICE("openGauss", "pkg_service", Collections.emptySet()),
67      
68      OPEN_GAUSS_SNAPSHOT("openGauss", "snapshot", Collections.emptySet()),
69      
70      OPEN_GAUSS_PLDEVELOPER("openGauss", "dbe_pldeveloper", Collections.emptySet()),
71      
72      OPEN_GAUSS_PG_TOAST("openGauss", "pg_toast", Collections.emptySet()),
73      
74      OPEN_GAUSS_PKG_UTIL("openGauss", "pkg_util", Collections.emptySet()),
75      
76      OPEN_GAUSS_SQLADVISOR("openGauss", "sqladvisor", Collections.emptySet()),
77      
78      OPEN_GAUSS_SHARDING_SPHERE("openGauss", "shardingsphere", new HashSet<>(Arrays.asList("sharding_table_statistics", "cluster_information")));
79      
80      private static final Map<String, KernelSupportedSystemTables> SCHEMA_NAME_TO_TABLES = new HashMap<>(values().length, 1F);
81      
82      private final String databaseType;
83      
84      private final String schema;
85      
86      private final Collection<String> tables;
87      
88      static {
89          for (KernelSupportedSystemTables each : values()) {
90              SCHEMA_NAME_TO_TABLES.put(each.getDatabaseType() + "." + each.getSchema(), each);
91          }
92      }
93      
94      /**
95       * Judge whether current table is kernel supported system table or not.
96       *
97       * @param schema schema
98       * @param tableName table name
99       * @return whether current table is kernel supported system table or not
100      */
101     public static boolean isSupportedSystemTable(final String schema, final String tableName) {
102         for (KernelSupportedSystemTables each : values()) {
103             if (each.getSchema().equals(schema) && each.getTables().contains(tableName)) {
104                 return true;
105             }
106         }
107         return false;
108     }
109 }
