59  /**
60   * Data consistency checker for migration job.
61   */
62  @Slf4j
63  public final class MigrationDataConsistencyChecker implements PipelineDataConsistencyChecker {
64      
65      private final MigrationJobConfiguration jobConfig;
66      
67      private final JobRateLimitAlgorithm readRateLimitAlgorithm;
68      
69      private final ConsistencyCheckJobItemProgressContext progressContext;
70      
71      public MigrationDataConsistencyChecker(final MigrationJobConfiguration jobConfig, final InventoryIncrementalProcessContext processContext,
72                                             final ConsistencyCheckJobItemProgressContext progressContext) {
73          this.jobConfig = jobConfig;
74          readRateLimitAlgorithm = null == processContext ? null : processContext.getReadRateLimitAlgorithm();
75          this.progressContext = progressContext;
76      }
77      
78      @Override
79      public Map<String, DataConsistencyCheckResult> check(final DataConsistencyCalculateAlgorithm calculateAlgorithm) {
80          verifyPipelineDatabaseType(calculateAlgorithm, jobConfig.getSources().values().iterator().next());
81          verifyPipelineDatabaseType(calculateAlgorithm, jobConfig.getTarget());
82          List<String> sourceTableNames = new LinkedList<>();
83          jobConfig.getJobShardingDataNodes().forEach(each -> each.getEntries().forEach(entry -> entry.getDataNodes()
84                  .forEach(dataNode -> sourceTableNames.add(DataNodeUtils.formatWithSchema(dataNode)))));
85          progressContext.setRecordsCount(getRecordsCount());
86          progressContext.getTableNames().addAll(sourceTableNames);
87          progressContext.onProgressUpdated(new PipelineJobProgressUpdatedParameter(0));
88          Map<String, DataConsistencyCheckResult> result = new LinkedHashMap<>();
89          PipelineDataSourceManager dataSourceManager = new DefaultPipelineDataSourceManager();
90          try {
91              AtomicBoolean checkFailed = new AtomicBoolean(false);
92              for (JobDataNodeLine each : jobConfig.getJobShardingDataNodes()) {
93                  each.getEntries().forEach(entry -> entry.getDataNodes().forEach(dataNode -> check(calculateAlgorithm, result, dataSourceManager, checkFailed, each, entry, dataNode)));
94              }
95          } finally {
96              dataSourceManager.close();
97          }
98          return result;
99      }
100     
101     private void check(final DataConsistencyCalculateAlgorithm calculateAlgorithm, final Map<String, DataConsistencyCheckResult> checkResults, final PipelineDataSourceManager dataSourceManager,
102                        final AtomicBoolean checkFailed, final JobDataNodeLine jobDataNodeLine, final JobDataNodeEntry entry, final DataNode dataNode) {
103         if (checkFailed.get()) {
104             return;
105         }
106         DataConsistencyCheckResult checkResult = checkSingleTable(entry.getLogicTableName(), dataNode, calculateAlgorithm, dataSourceManager);
107         checkResults.put(DataNodeUtils.formatWithSchema(dataNode), checkResult);
108         if (!checkResult.isMatched()) {
109             log.info("unmatched on table '{}', ignore left tables", jobDataNodeLine);
110             checkFailed.set(true);
111         }
112     }
113     
114     private DataConsistencyCheckResult checkSingleTable(final String targetTableName, final DataNode dataNode,
115                                                         final DataConsistencyCalculateAlgorithm calculateAlgorithm, final PipelineDataSourceManager dataSourceManager) {
116         SchemaTableName sourceTable = new SchemaTableName(new SchemaName(dataNode.getSchemaName()), new TableName(dataNode.getTableName()));
117         SchemaTableName targetTable = new SchemaTableName(new SchemaName(dataNode.getSchemaName()), new TableName(targetTableName));
118         PipelineDataSourceWrapper sourceDataSource = dataSourceManager.getDataSource(jobConfig.getSources().get(dataNode.getDataSourceName()));
119         PipelineDataSourceWrapper targetDataSource = dataSourceManager.getDataSource(jobConfig.getTarget());
120         PipelineTableMetaDataLoader metaDataLoader = new StandardPipelineTableMetaDataLoader(sourceDataSource);
121         PipelineTableMetaData tableMetaData = metaDataLoader.getTableMetaData(dataNode.getSchemaName(), dataNode.getTableName());
122         ShardingSpherePreconditions.checkNotNull(tableMetaData, () -> new PipelineTableDataConsistencyCheckLoadingFailedException(dataNode.getSchemaName(), dataNode.getTableName()));
123         List<String> columnNames = tableMetaData.getColumnNames();
124         List<PipelineColumnMetaData> uniqueKeyColumns = PipelineTableMetaDataUtils.getUniqueKeyColumns(
125                 sourceTable.getSchemaName().getOriginal(), sourceTable.getTableName().getOriginal(), metaDataLoader);
126         PipelineColumnMetaData uniqueKey = uniqueKeyColumns.isEmpty() ? null : uniqueKeyColumns.get(0);
127         SingleTableInventoryDataConsistencyChecker singleTableInventoryChecker = new SingleTableInventoryDataConsistencyChecker(
128                 jobConfig.getJobId(), sourceDataSource, targetDataSource, sourceTable, targetTable, columnNames, uniqueKey, readRateLimitAlgorithm, progressContext);
129         return singleTableInventoryChecker.check(calculateAlgorithm);
130     }
131     
132     private void verifyPipelineDatabaseType(final DataConsistencyCalculateAlgorithm calculateAlgorithm, final PipelineDataSourceConfiguration dataSourceConfig) {
133         ShardingSpherePreconditions.checkState(calculateAlgorithm.getSupportedDatabaseTypes().contains(dataSourceConfig.getDatabaseType()),
134                 () -> new UnsupportedPipelineDatabaseTypeException(dataSourceConfig.getDatabaseType()));
135     }
136     
137     private long getRecordsCount() {
138         Map<Integer, InventoryIncrementalJobItemProgress> jobProgress = new MigrationJobAPI().getJobProgress(jobConfig);
139         return jobProgress.values().stream().filter(Objects::nonNull).mapToLong(InventoryIncrementalJobItemProgress::getProcessedRecordsCount).sum();
140     }
