51  /**
52   * Predicate right value token generator for encrypt.
53   */
54  @HighFrequencyInvocation
55  @Setter
56  public final class EncryptPredicateRightValueTokenGenerator implements CollectionSQLTokenGenerator<SQLStatementContext>, ParametersAware, EncryptConditionsAware, EncryptRuleAware, DatabaseNameAware {
57      
58      private List<Object> parameters;
59      
60      private Collection<EncryptCondition> encryptConditions;
61      
62      private EncryptRule encryptRule;
63      
64      private String databaseName;
65      
66      @Override
67      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
68          return sqlStatementContext instanceof WhereAvailable && !((WhereAvailable) sqlStatementContext).getWhereSegments().isEmpty();
69      }
70      
71      @Override
72      public Collection<SQLToken> generateSQLTokens(final SQLStatementContext sqlStatementContext) {
73          Collection<SQLToken> result = new LinkedHashSet<>(encryptConditions.size(), 1F);
74          String schemaName = ((TableAvailable) sqlStatementContext).getTablesContext().getSchemaName()
75                  .orElseGet(() -> new DatabaseTypeRegistry(sqlStatementContext.getDatabaseType()).getDefaultSchemaName(databaseName));
76          for (EncryptCondition each : encryptConditions) {
77              encryptRule.findEncryptTable(each.getTableName()).ifPresent(optional -> result.add(generateSQLToken(schemaName, optional, each)));
78          }
79          return result;
80      }
81      
82      private SQLToken generateSQLToken(final String schemaName, final EncryptTable encryptTable, final EncryptCondition encryptCondition) {
83          List<Object> originalValues = encryptCondition.getValues(parameters);
84          int startIndex = encryptCondition.getStartIndex();
85          return generateSQLToken(schemaName, encryptTable, encryptCondition, originalValues, startIndex);
86      }
87      
88      private SQLToken generateSQLToken(final String schemaName, final EncryptTable encryptTable, final EncryptCondition encryptCondition, final List<Object> originalValues, final int startIndex) {
89          int stopIndex = encryptCondition.getStopIndex();
90          Map<Integer, Object> indexValues = getPositionValues(encryptCondition.getPositionValueMap().keySet(), getEncryptedValues(schemaName, encryptTable, encryptCondition, originalValues));
91          Collection<Integer> parameterMarkerIndexes = encryptCondition.getPositionIndexMap().keySet();
92          if (encryptCondition instanceof EncryptBinaryCondition && ((EncryptBinaryCondition) encryptCondition).getExpressionSegment() instanceof FunctionSegment) {
93              return new EncryptPredicateFunctionRightValueToken(startIndex, stopIndex,
94                      ((FunctionSegment) ((EncryptBinaryCondition) encryptCondition).getExpressionSegment()).getFunctionName(), indexValues, parameterMarkerIndexes);
95          }
96          return encryptCondition instanceof EncryptInCondition
97                  ? new EncryptPredicateInRightValueToken(startIndex, stopIndex, indexValues, parameterMarkerIndexes)
98                  : new EncryptPredicateEqualRightValueToken(startIndex, stopIndex, indexValues, parameterMarkerIndexes);
99      }
100     
101     private List<Object> getEncryptedValues(final String schemaName, final EncryptTable encryptTable, final EncryptCondition encryptCondition, final List<Object> originalValues) {
102         EncryptColumn encryptColumn = encryptTable.getEncryptColumn(encryptCondition.getColumnName());
103         if (encryptCondition instanceof EncryptBinaryCondition && "LIKE".equalsIgnoreCase(((EncryptBinaryCondition) encryptCondition).getOperator())) {
104             LikeQueryColumnItem likeQueryColumnItem = encryptColumn.getLikeQuery().orElseThrow(() -> new UnsupportedEncryptSQLException("LIKE"));
105             return likeQueryColumnItem.encrypt(databaseName, schemaName, encryptCondition.getTableName(), encryptCondition.getColumnName(), originalValues);
106         }
107         return encryptColumn.getAssistedQuery()
108                 .map(optional -> optional.encrypt(databaseName, schemaName, encryptCondition.getTableName(), encryptCondition.getColumnName(), originalValues))
109                 .orElseGet(() -> encryptColumn.getCipher().encrypt(databaseName, schemaName, encryptCondition.getTableName(), encryptCondition.getColumnName(), originalValues));
110     }
111     
112     private Map<Integer, Object> getPositionValues(final Collection<Integer> valuePositions, final List<Object> encryptValues) {
113         Map<Integer, Object> result = new LinkedHashMap<>(valuePositions.size(), 1F);
114         for (int each : valuePositions) {
115             result.put(each, encryptValues.get(each));
116         }
117         return result;
118     }
119 }
