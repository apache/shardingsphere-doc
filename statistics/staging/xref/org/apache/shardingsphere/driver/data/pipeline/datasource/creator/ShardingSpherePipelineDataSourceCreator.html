43  /**
44   * ShardingSphere pipeline data source creator.
45   */
46  public final class ShardingSpherePipelineDataSourceCreator implements PipelineDataSourceCreator {
47      
48      private static final AtomicInteger STANDALONE_DATABASE_ID = new AtomicInteger(1);
49      
50      @Override
51      public DataSource createPipelineDataSource(final Object dataSourceConfig) throws SQLException {
52          YamlRootConfiguration rootConfig = YamlEngine.unmarshal(YamlEngine.marshal(dataSourceConfig), YamlRootConfiguration.class);
53          disableSystemSchemaMetadata(rootConfig);
54          enableStreamingQuery(rootConfig);
55          updateSingleRuleConfiguration(rootConfig);
56          Optional<YamlShardingRuleConfiguration> yamlShardingRuleConfig = ShardingRuleConfigurationConverter.findYamlShardingRuleConfiguration(rootConfig.getRules());
57          if (yamlShardingRuleConfig.isPresent()) {
58              enableRangeQueryForInline(yamlShardingRuleConfig.get());
59              removeAuditStrategy(yamlShardingRuleConfig.get());
60          }
61          rootConfig.setDatabaseName(rootConfig.getDatabaseName());
62          rootConfig.setSchemaName(rootConfig.getSchemaName());
63          rootConfig.setMode(createStandaloneModeConfiguration());
64          return YamlShardingSphereDataSourceFactory.createDataSourceWithoutCache(rootConfig);
65      }
66      
67      private void updateSingleRuleConfiguration(final YamlRootConfiguration rootConfig) {
68          rootConfig.getRules().removeIf(YamlSingleRuleConfiguration.class::isInstance);
69          YamlSingleRuleConfiguration singleRuleConfig = new YamlSingleRuleConfiguration();
70          singleRuleConfig.setTables(Collections.singletonList(SingleTableConstants.ALL_TABLES));
71          rootConfig.getRules().add(singleRuleConfig);
72      }
73      
74      private void disableSystemSchemaMetadata(final YamlRootConfiguration rootConfig) {
75          rootConfig.getProps().put(TemporaryConfigurationPropertyKey.SYSTEM_SCHEMA_METADATA_ENABLED.getKey(), String.valueOf(Boolean.FALSE));
76      }
77      
78      // TODO Another way is improving ExecuteQueryCallback.executeSQL to enable streaming query, then remove it
79      private void enableStreamingQuery(final YamlRootConfiguration rootConfig) {
80          // Set a large enough value to enable ConnectionMode.MEMORY_STRICTLY, make sure streaming query work.
81          rootConfig.getProps().put(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY.getKey(), 100000);
82      }
83      
84      private void enableRangeQueryForInline(final YamlShardingRuleConfiguration yamlShardingRuleConfig) {
85          for (YamlAlgorithmConfiguration each : yamlShardingRuleConfig.getShardingAlgorithms().values()) {
86              if ("INLINE".equalsIgnoreCase(each.getType())) {
87                  each.getProps().put("allow-range-query-with-inline-sharding", Boolean.TRUE.toString());
88              }
89          }
90      }
91      
92      private void removeAuditStrategy(final YamlShardingRuleConfiguration yamlShardingRuleConfig) {
93          yamlShardingRuleConfig.setDefaultAuditStrategy(null);
94          yamlShardingRuleConfig.setAuditors(null);
95          if (null != yamlShardingRuleConfig.getTables()) {
96              yamlShardingRuleConfig.getTables().forEach((key, value) -> value.setAuditStrategy(null));
97          }
98          if (null != yamlShardingRuleConfig.getAutoTables()) {
99              yamlShardingRuleConfig.getAutoTables().forEach((key, value) -> value.setAuditStrategy(null));
100         }
101     }
102     
103     private YamlModeConfiguration createStandaloneModeConfiguration() {
104         YamlModeConfiguration result = new YamlModeConfiguration();
105         result.setType("Standalone");
106         YamlPersistRepositoryConfiguration repository = new YamlPersistRepositoryConfiguration();
107         result.setRepository(repository);
108         repository.setType("JDBC");
109         Properties props = new Properties();
110         repository.setProps(props);
111         props.setProperty(JDBCRepositoryPropertyKey.JDBC_URL.getKey(),
112                 String.format("jdbc:h2:mem:pipeline_db_%d;DB_CLOSE_DELAY=0;DATABASE_TO_UPPER=false;MODE=MYSQL", STANDALONE_DATABASE_ID.getAndIncrement()));
113         return result;
114     }
115     
116     @Override
117     public String getType() {
118         return ShardingSpherePipelineDataSourceConfiguration.TYPE;
119     }
120 }
