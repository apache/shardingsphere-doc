77      private final YamlProxyDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlProxyDataSourceConfigurationSwapper();
78      
79      @Override
80      public Collection<String> getColumnNames() {
81          return Collections.singleton("dist_sql");
82      }
83      
84      @Override
85      public Collection<LocalDataQueryResultRow> getRows(final ConvertYamlConfigurationStatement sqlStatement) {
86          File file = new File(sqlStatement.getFilePath());
87          YamlProxyDatabaseConfiguration yamlConfig;
88          try {
89              yamlConfig = YamlEngine.unmarshal(file, YamlProxyDatabaseConfiguration.class);
90          } catch (final IOException ex) {
91              throw new FileIOException(ex);
92          }
93          Preconditions.checkNotNull(yamlConfig, "Invalid yaml file `%s`", file.getName());
94          Preconditions.checkNotNull(yamlConfig.getDatabaseName(), "`databaseName` in file `%s` is required.", file.getName());
95          return Collections.singleton(new LocalDataQueryResultRow(generateDistSQL(yamlConfig)));
96      }
97      
98      private String generateDistSQL(final YamlProxyDatabaseConfiguration yamlConfig) {
99          StringBuilder result = new StringBuilder();
100         appendResourceDistSQL(yamlConfig, result);
101         for (RuleConfiguration each : swapToRuleConfigs(yamlConfig).values()) {
102             if (each instanceof CompatibleEncryptRuleConfiguration) {
103                 ConvertRuleConfigurationProvider convertRuleConfigProvider = TypedSPILoader.getService(ConvertRuleConfigurationProvider.class,
104                         ((CompatibleEncryptRuleConfiguration) each).convertToEncryptRuleConfiguration().getClass().getName());
105                 result.append(convertRuleConfigProvider.convert(each));
106             } else if (each instanceof ShadowRuleConfiguration) {
107                 appendShadowDistSQL((ShadowRuleConfiguration) each, result);
108             } else if (each instanceof MaskRuleConfiguration) {
109                 appendMaskDistSQL((MaskRuleConfiguration) each, result);
110             } else {
111                 ConvertRuleConfigurationProvider convertRuleConfigProvider = TypedSPILoader.getService(ConvertRuleConfigurationProvider.class, each.getClass().getName());
112                 result.append(convertRuleConfigProvider.convert(each));
113             }
114         }
115         return result.toString();
116     }
117     
118     private Map<Integer, RuleConfiguration> swapToRuleConfigs(final YamlProxyDatabaseConfiguration yamlConfig) {
119         Map<Integer, RuleConfiguration> result = new TreeMap<>(Comparator.reverseOrder());
120         for (YamlRuleConfiguration each : yamlConfig.getRules()) {
121             if (each instanceof YamlShardingRuleConfiguration) {
122                 YamlShardingRuleConfigurationSwapper swapper = new YamlShardingRuleConfigurationSwapper();
123                 result.put(swapper.getOrder(), swapper.swapToObject((YamlShardingRuleConfiguration) each));
124             } else if (each instanceof YamlReadwriteSplittingRuleConfiguration) {
125                 YamlReadwriteSplittingRuleConfigurationSwapper swapper = new YamlReadwriteSplittingRuleConfigurationSwapper();
126                 result.put(swapper.getOrder(), swapper.swapToObject((YamlReadwriteSplittingRuleConfiguration) each));
127             } else if (each instanceof YamlEncryptRuleConfiguration) {
128                 YamlEncryptRuleConfigurationSwapper swapper = new YamlEncryptRuleConfigurationSwapper();
129                 result.put(swapper.getOrder(), swapper.swapToObject((YamlEncryptRuleConfiguration) each));
130             } else if (each instanceof YamlCompatibleEncryptRuleConfiguration) {
131                 YamlCompatibleEncryptRuleConfigurationSwapper swapper = new YamlCompatibleEncryptRuleConfigurationSwapper();
132                 result.put(swapper.getOrder(), swapper.swapToObject((YamlCompatibleEncryptRuleConfiguration) each));
133             } else if (each instanceof YamlShadowRuleConfiguration) {
134                 YamlShadowRuleConfigurationSwapper swapper = new YamlShadowRuleConfigurationSwapper();
135                 result.put(swapper.getOrder(), swapper.swapToObject((YamlShadowRuleConfiguration) each));
136             } else if (each instanceof YamlMaskRuleConfiguration) {
137                 YamlMaskRuleConfigurationSwapper swapper = new YamlMaskRuleConfigurationSwapper();
138                 result.put(swapper.getOrder(), swapper.swapToObject((YamlMaskRuleConfiguration) each));
139             }
140         }
141         return result;
142     }
143     
144     private void appendResourceDistSQL(final YamlProxyDatabaseConfiguration yamlConfig, final StringBuilder stringBuilder) {
145         appendDatabase(yamlConfig.getDatabaseName(), stringBuilder);
146         appendResources(yamlConfig.getDataSources(), stringBuilder);
147     }
148     
149     private void appendDatabase(final String databaseName, final StringBuilder stringBuilder) {
150         stringBuilder.append(String.format(DistSQLScriptConstants.CREATE_DATABASE, databaseName)).append(System.lineSeparator())
151                 .append(String.format(DistSQLScriptConstants.USE_DATABASE, databaseName)).append(System.lineSeparator()).append(System.lineSeparator());
152     }
153     
154     private void appendResources(final Map<String, YamlProxyDataSourceConfiguration> dataSources, final StringBuilder stringBuilder) {
155         if (dataSources.isEmpty()) {
156             return;
157         }
158         stringBuilder.append(DistSQLScriptConstants.REGISTER_STORAGE_UNIT);
159         Iterator<Entry<String, YamlProxyDataSourceConfiguration>> iterator = dataSources.entrySet().iterator();
160         while (iterator.hasNext()) {
161             Entry<String, YamlProxyDataSourceConfiguration> entry = iterator.next();
162             DataSourceProperties dataSourceProps = DataSourcePropertiesCreator.create(HikariDataSource.class.getName(), dataSourceConfigSwapper.swap(entry.getValue()));
163             appendResource(entry.getKey(), dataSourceProps, stringBuilder);
164             if (iterator.hasNext()) {
165                 stringBuilder.append(DistSQLScriptConstants.COMMA);
166             }
167         }
168         stringBuilder.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
169     }
170     
171     private void appendResource(final String resourceName, final DataSourceProperties dataSourceProps, final StringBuilder stringBuilder) {
172         Map<String, Object> connectionProps = dataSourceProps.getConnectionPropertySynonyms().getStandardProperties();
173         String url = (String) connectionProps.get(DistSQLScriptConstants.KEY_URL);
174         String username = (String) connectionProps.get(DistSQLScriptConstants.KEY_USERNAME);
175         String password = (String) connectionProps.get(DistSQLScriptConstants.KEY_PASSWORD);
176         String props = getResourceProperties(dataSourceProps.getPoolPropertySynonyms(), dataSourceProps.getCustomDataSourceProperties());
177         if (Strings.isNullOrEmpty(password)) {
178             stringBuilder.append(String.format(DistSQLScriptConstants.RESOURCE_DEFINITION_WITHOUT_PASSWORD, resourceName, url, username, props));
179         } else {
180             stringBuilder.append(String.format(DistSQLScriptConstants.RESOURCE_DEFINITION, resourceName, url, username, password, props));
181         }
182     }
183     
184     private String getResourceProperties(final PoolPropertySynonyms poolPropertySynonyms, final CustomDataSourceProperties customDataSourceProps) {
185         StringBuilder result = new StringBuilder();
186         appendProperties(poolPropertySynonyms.getStandardProperties(), result);
187         if (!customDataSourceProps.getProperties().isEmpty()) {
188             result.append(DistSQLScriptConstants.COMMA);
189             appendProperties(customDataSourceProps.getProperties(), result);
190         }
191         return result.toString();
192     }
193     
194     private void appendProperties(final Map<String, Object> props, final StringBuilder stringBuilder) {
195         Iterator<Entry<String, Object>> iterator = props.entrySet().iterator();
196         while (iterator.hasNext()) {
197             Entry<String, Object> entry = iterator.next();
198             if (null == entry.getValue()) {
199                 continue;
200             }
201             stringBuilder.append(String.format(DistSQLScriptConstants.PROPERTY, entry.getKey(), entry.getValue()));
202             if (iterator.hasNext()) {
203                 stringBuilder.append(DistSQLScriptConstants.COMMA).append(' ');
204             }
205         }
206     }
207     
208     private void appendShadowDistSQL(final ShadowRuleConfiguration ruleConfig, final StringBuilder stringBuilder) {
209         if (ruleConfig.getDataSources().isEmpty()) {
210             return;
211         }
212         stringBuilder.append(DistSQLScriptConstants.CREATE_SHADOW);
213         Iterator<ShadowDataSourceConfiguration> iterator = ruleConfig.getDataSources().iterator();
214         while (iterator.hasNext()) {
215             ShadowDataSourceConfiguration dataSourceConfig = iterator.next();
216             String shadowRuleName = dataSourceConfig.getName();
217             String shadowTables = getShadowTables(shadowRuleName, ruleConfig.getTables(), ruleConfig.getShadowAlgorithms());
218             stringBuilder.append(
219                     String.format(DistSQLScriptConstants.SHADOW, shadowRuleName, dataSourceConfig.getProductionDataSourceName(), dataSourceConfig.getShadowDataSourceName(), shadowTables));
220             if (iterator.hasNext()) {
221                 stringBuilder.append(DistSQLScriptConstants.COMMA);
222             }
223         }
224         stringBuilder.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
225     }
226     
227     private String getShadowTables(final String shadowRuleName, final Map<String, ShadowTableConfiguration> ruleConfig, final Map<String, AlgorithmConfiguration> algorithmConfigs) {
228         StringBuilder result = new StringBuilder();
229         Iterator<Entry<String, ShadowTableConfiguration>> iterator = ruleConfig.entrySet().iterator();
230         while (iterator.hasNext()) {
231             Entry<String, ShadowTableConfiguration> shadowTableConfig = iterator.next();
232             if (shadowTableConfig.getValue().getDataSourceNames().contains(shadowRuleName)) {
233                 String shadowTableTypes = getShadowTableTypes(shadowTableConfig.getValue().getShadowAlgorithmNames(), algorithmConfigs);
234                 result.append(String.format(DistSQLScriptConstants.SHADOW_TABLE, shadowTableConfig.getKey(), shadowTableTypes));
235             }
236             if (iterator.hasNext()) {
237                 result.append(DistSQLScriptConstants.COMMA).append(System.lineSeparator());
238             }
239         }
240         return result.toString();
241     }
242     
243     private String getShadowTableTypes(final Collection<String> shadowAlgorithmNames, final Map<String, AlgorithmConfiguration> algorithmConfigs) {
244         StringBuilder result = new StringBuilder();
245         Iterator<String> iterator = shadowAlgorithmNames.iterator();
246         while (iterator.hasNext()) {
247             result.append(getAlgorithmType(algorithmConfigs.get(iterator.next())));
248             if (iterator.hasNext()) {
249                 result.append(DistSQLScriptConstants.COMMA).append(' ');
250             }
251         }
252         return result.toString();
253     }
254     
255     private void appendMaskDistSQL(final MaskRuleConfiguration ruleConfig, final StringBuilder stringBuilder) {
256         if (ruleConfig.getTables().isEmpty()) {
257             return;
258         }
259         stringBuilder.append(DistSQLScriptConstants.CREATE_MASK);
260         Iterator<MaskTableRuleConfiguration> iterator = ruleConfig.getTables().iterator();
261         while (iterator.hasNext()) {
262             MaskTableRuleConfiguration tableRuleConfig = iterator.next();
263             stringBuilder.append(String.format(DistSQLScriptConstants.MASK, tableRuleConfig.getName(), getMaskColumns(tableRuleConfig.getColumns(), ruleConfig.getMaskAlgorithms())));
264             if (iterator.hasNext()) {
265                 stringBuilder.append(DistSQLScriptConstants.COMMA).append(System.lineSeparator());
266             }
267         }
268         stringBuilder.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
269     }
270     
271     private String getMaskColumns(final Collection<MaskColumnRuleConfiguration> columnRuleConfig, final Map<String, AlgorithmConfiguration> maskAlgorithms) {
272         StringBuilder result = new StringBuilder();
273         Iterator<MaskColumnRuleConfiguration> iterator = columnRuleConfig.iterator();
274         if (iterator.hasNext()) {
275             MaskColumnRuleConfiguration column = iterator.next();
276             result.append(String.format(DistSQLScriptConstants.MASK_COLUMN, column.getLogicColumn(), getMaskAlgorithms(column, maskAlgorithms)));
277         }
278         return result.toString();
279     }
280     
281     private String getMaskAlgorithms(final MaskColumnRuleConfiguration columnRuleConfig, final Map<String, AlgorithmConfiguration> maskAlgorithms) {
282         return getAlgorithmType(maskAlgorithms.get(columnRuleConfig.getMaskAlgorithm()));
283     }
284     
285     private String getAlgorithmType(final AlgorithmConfiguration algorithmConfig) {
286         StringBuilder result = new StringBuilder();
287         if (null == algorithmConfig) {
288             return result.toString();
289         }
290         String type = algorithmConfig.getType().toLowerCase();
291         if (algorithmConfig.getProps().isEmpty()) {
292             result.append(String.format(DistSQLScriptConstants.ALGORITHM_TYPE_WITHOUT_PROPS, type));
293         } else {
294             result.append(String.format(DistSQLScriptConstants.ALGORITHM_TYPE, type, getAlgorithmProperties(algorithmConfig.getProps())));
295         }
296         return result.toString();
297     }
298     
299     @SuppressWarnings({"unchecked", "rawtypes"})
300     private String getAlgorithmProperties(final Properties props) {
301         StringBuilder result = new StringBuilder();
302         Iterator<String> iterator = new TreeMap(props).keySet().iterator();
303         while (iterator.hasNext()) {
304             String key = iterator.next();
305             Object value = props.get(key);
306             if (null == value) {
307                 continue;
308             }
309             result.append(String.format(DistSQLScriptConstants.PROPERTY, key, value));
310             if (iterator.hasNext()) {
311                 result.append(DistSQLScriptConstants.COMMA).append(' ');
312             }
313         }
314         return result.toString();
315     }
