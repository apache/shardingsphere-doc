18  package org.apache.shardingsphere.infra.yaml.schema.swapper;
19  
20  import lombok.SneakyThrows;
21  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereColumn;
22  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereConstraint;
23  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereIndex;
24  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereTable;
25  import org.apache.shardingsphere.infra.util.yaml.YamlEngine;
26  import org.apache.shardingsphere.infra.yaml.schema.pojo.YamlShardingSphereTable;
27  import org.junit.jupiter.api.Test;
28  
29  import java.io.IOException;
30  import java.net.URISyntaxException;
31  import java.nio.file.Files;
32  import java.nio.file.Paths;
33  import java.util.Collections;
34  import java.util.stream.Collectors;
35  
36  import static org.apache.shardingsphere.test.matcher.ShardingSphereAssertionMatchers.deepEqual;
37  import static org.hamcrest.MatcherAssert.assertThat;
38  
39  class YamlTableSwapperTest {
40      
41      private static final String YAML_FILE = "yaml/schema/table.yaml";
42      
43      private static final String EMPTY_YAML_FILE = "yaml/schema/empty-table.yaml";
44      
45      private final YamlTableSwapper swapper = new YamlTableSwapper();
46      
47      @Test
48      void assertSwapToYamlConfiguration() {
49          ShardingSphereTable table = createShardingSphereTable();
50          YamlShardingSphereTable actual = swapper.swapToYamlConfiguration(table);
51          YamlShardingSphereTable expected = unmarshal(YAML_FILE);
52          assertThat(actual, deepEqual(expected));
53      }
54      
55      @Test
56      void assertSwapToObject() {
57          ShardingSphereTable actual = swapper.swapToObject(unmarshal(YAML_FILE));
58          ShardingSphereTable expected = createShardingSphereTable();
59          assertThat(actual, deepEqual(expected));
60      }
61      
62      @Test
63      void assertSwapToObjectWithEmptySchema() {
64          ShardingSphereTable actual = swapper.swapToObject(unmarshal(EMPTY_YAML_FILE));
65          ShardingSphereTable expected = new ShardingSphereTable();
66          assertThat(actual, deepEqual(expected));
67      }
68      
69      @SneakyThrows({URISyntaxException.class, IOException.class})
70      private YamlShardingSphereTable unmarshal(final String yamlFile) {
71          String yamlContent = Files.readAllLines(Paths.get(ClassLoader.getSystemResource(yamlFile).toURI())).stream().collect(Collectors.joining(System.lineSeparator()));
72          return YamlEngine.unmarshal(yamlContent, YamlShardingSphereTable.class);
73      }
74      
75      private ShardingSphereTable createShardingSphereTable() {
76          return new ShardingSphereTable("foo_tbl",
77                  Collections.singleton(new ShardingSphereColumn("foo_col", 0, true, false, false, true, false, false)),
78                  Collections.singleton(new ShardingSphereIndex("PRIMARY")),
79                  Collections.singleton(new ShardingSphereConstraint("foo_constraint", "foo_tbl")), null);
80      }
81  }
