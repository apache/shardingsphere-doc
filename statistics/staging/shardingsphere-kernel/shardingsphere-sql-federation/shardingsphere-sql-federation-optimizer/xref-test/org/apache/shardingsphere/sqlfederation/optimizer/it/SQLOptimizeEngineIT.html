235     private SqlToRelConverter createSqlToRelConverter(final ShardingSphereSchema schema) {
236         CalciteConnectionConfig connectionConfig = new CalciteConnectionConfigImpl(new Properties());
237         RelDataTypeFactory relDataTypeFactory = new JavaTypeFactoryImpl();
238         DatabaseType databaseType = DatabaseTypeEngine.getDatabaseType("H2");
239         TranslatableSchema federationSchema = new TranslatableSchema(SCHEMA_NAME, schema, databaseType, new JavaTypeFactoryImpl(), null);
240         CalciteCatalogReader catalogReader = SQLFederationPlannerUtils.createCatalogReader(SCHEMA_NAME, federationSchema, relDataTypeFactory, connectionConfig);
241         SqlValidator validator = SQLFederationPlannerUtils.createSqlValidator(catalogReader, relDataTypeFactory, databaseType, connectionConfig);
242         RelOptCluster cluster = RelOptCluster.create(SQLFederationPlannerUtils.createVolcanoPlanner(), new RexBuilder(relDataTypeFactory));
243         return SQLFederationPlannerUtils.createSqlToRelConverter(catalogReader, validator, cluster, mock(SQLParserRule.class), databaseType, false);
244     }

246     @ParameterizedTest(name = "{0}")
247     @ArgumentsSource(TestCaseArgumentsProvider.class)
248     void assertOptimize(final TestCase testcase) {
249         SQLStatement sqlStatement = sqlParserRule.getSQLParserEngine(DatabaseTypeEngine.getTrunkDatabaseTypeName(new H2DatabaseType())).parse(testcase.getSql(), false);
250         String actual = optimizeEngine.optimize(sqlStatement).getBestPlan().explain().replaceAll("[\r\n]", "");
251         assertThat(actual, is(testcase.getAssertion().getExpectedResult()));
252     }
