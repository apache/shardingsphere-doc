18  package org.apache.shardingsphere.infra.database.core.metadata.data.loader;
19  
20  import com.google.common.util.concurrent.ThreadFactoryBuilder;
21  import lombok.AccessLevel;
22  import lombok.NoArgsConstructor;
23  import lombok.extern.slf4j.Slf4j;
24  import org.apache.shardingsphere.infra.database.core.metadata.data.loader.type.TableMetaDataLoader;
25  import org.apache.shardingsphere.infra.database.core.metadata.data.model.SchemaMetaData;
26  import org.apache.shardingsphere.infra.database.core.metadata.data.model.TableMetaData;
27  import org.apache.shardingsphere.infra.database.core.spi.DatabaseTypedSPILoader;
28  
29  import java.sql.SQLException;
30  import java.util.Collection;
31  import java.util.Collections;
32  import java.util.LinkedHashMap;
33  import java.util.LinkedList;
34  import java.util.Map;
35  import java.util.Optional;
36  import java.util.concurrent.ExecutionException;
37  import java.util.concurrent.ExecutorService;
38  import java.util.concurrent.Future;
39  import java.util.concurrent.LinkedBlockingQueue;
40  import java.util.concurrent.ThreadPoolExecutor;
41  import java.util.concurrent.TimeUnit;
42  
43  /**
44   * Meta data loader.
45   */
46  @NoArgsConstructor(access = AccessLevel.PRIVATE)
47  @Slf4j
48  public final class MetaDataLoader {
49      
50      private static final ExecutorService EXECUTOR_SERVICE = new ThreadPoolExecutor(Runtime.getRuntime().availableProcessors() * 2, Runtime.getRuntime().availableProcessors() * 2,
51              0L, TimeUnit.MILLISECONDS, new LinkedBlockingQueue<>(), new ThreadFactoryBuilder().setDaemon(true).setNameFormat("ShardingSphere-SchemaMetaDataLoaderEngine-%d").build());
52      
53      /**
54       * Load meta data.
55       *
56       * @param materials meta data loader materials
57       * @return meta data map
58       * @throws SQLException SQL exception
59       */
60      public static Map<String, SchemaMetaData> load(final Collection<MetaDataLoaderMaterial> materials) throws SQLException {
61          Map<String, SchemaMetaData> result = new LinkedHashMap<>(materials.size(), 1F);
62          Collection<Future<Collection<SchemaMetaData>>> futures = new LinkedList<>();
63          for (MetaDataLoaderMaterial each : materials) {
64              futures.add(EXECUTOR_SERVICE.submit(() -> load(each)));
65          }
66          try {
67              for (Future<Collection<SchemaMetaData>> each : futures) {
68                  merge(result, each.get());
69              }
70          } catch (final InterruptedException ex) {
71              Thread.currentThread().interrupt();
72          } catch (final ExecutionException ex) {
73              if (ex.getCause() instanceof SQLException) {
74                  throw (SQLException) ex.getCause();
75              }
76              throw new SQLException(ex);
77          }
78          return result;
79      }
80      
81      private static Collection<SchemaMetaData> load(final MetaDataLoaderMaterial material) throws SQLException {
82          Optional<DialectMetaDataLoader> dialectLoader = DatabaseTypedSPILoader.findService(DialectMetaDataLoader.class, material.getStorageType());
83          if (dialectLoader.isPresent()) {
84              try {
85                  return dialectLoader.get().load(material);
86              } catch (final SQLException ex) {
87                  log.debug("Dialect load schema meta data error.", ex);
88              }
89          }
90          return loadByDefault(material);
91      }
92      
93      private static Collection<SchemaMetaData> loadByDefault(final MetaDataLoaderMaterial material) throws SQLException {
94          Collection<TableMetaData> tableMetaData = new LinkedList<>();
95          for (String each : material.getActualTableNames()) {
96              TableMetaDataLoader.load(material.getDataSource(), each, material.getStorageType()).ifPresent(tableMetaData::add);
97          }
98          return Collections.singleton(new SchemaMetaData(material.getDefaultSchemaName(), tableMetaData));
99      }
100     
101     private static void merge(final Map<String, SchemaMetaData> schemaMetaDataMap, final Collection<SchemaMetaData> addedSchemaMetaDataList) {
102         for (SchemaMetaData each : addedSchemaMetaDataList) {
103             SchemaMetaData schemaMetaData = schemaMetaDataMap.computeIfAbsent(each.getName(), key -> new SchemaMetaData(each.getName(), new LinkedList<>()));
104             schemaMetaData.getTables().addAll(each.getTables());
105         }
106     }
107 }
