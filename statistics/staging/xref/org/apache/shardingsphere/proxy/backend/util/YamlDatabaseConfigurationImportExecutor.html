113     /**
114      * Import proxy database from yaml configuration.
115      *
116      * @param yamlConfig yaml proxy database configuration
117      */
118     public void importDatabaseConfiguration(final YamlProxyDatabaseConfiguration yamlConfig) {
119         String databaseName = yamlConfig.getDatabaseName();
120         checkDatabase(databaseName);
121         checkDataSource(yamlConfig.getDataSources());
122         addDatabase(databaseName);
123         addResources(databaseName, yamlConfig.getDataSources());
124         try {
125             addRules(databaseName, yamlConfig.getRules());
126         } catch (final DistSQLException ex) {
127             dropDatabase(databaseName);
128             throw ex;
129         }
130     }
131     
132     private void checkDatabase(final String databaseName) {
133         ShardingSpherePreconditions.checkNotNull(databaseName, () -> new UnsupportedSQLOperationException("Property `databaseName` in imported config is required"));
134         if (ProxyContext.getInstance().databaseExists(databaseName)) {
135             ShardingSpherePreconditions.checkState(ProxyContext.getInstance().getDatabase(databaseName).getResourceMetaData().getStorageUnits().isEmpty(),
136                     () -> new UnsupportedSQLOperationException(String.format("Database `%s` exists and is not emptyï¼Œoverwrite is not supported", databaseName)));
137         }
138     }
139     
140     private void checkDataSource(final Map<String, YamlProxyDataSourceConfiguration> dataSources) {
141         ShardingSpherePreconditions.checkState(!dataSources.isEmpty(), () -> new MissingRequiredDataSourcesException("Data source configurations in imported config is required"));
142     }
143     
144     private void addDatabase(final String databaseName) {
145         ContextManager contextManager = ProxyContext.getInstance().getContextManager();
146         contextManager.getInstanceContext().getModeContextManager().createDatabase(databaseName);
147         DatabaseType protocolType = DatabaseTypeEngine.getProtocolType(Collections.emptyMap(), contextManager.getMetaDataContexts().getMetaData().getProps());
148         contextManager.getMetaDataContexts().getMetaData().addDatabase(databaseName, protocolType, contextManager.getMetaDataContexts().getMetaData().getProps());
149     }
150     
151     private void addResources(final String databaseName, final Map<String, YamlProxyDataSourceConfiguration> yamlDataSourceMap) {
152         Map<String, DataSourcePoolProperties> propsMap = new LinkedHashMap<>(yamlDataSourceMap.size(), 1F);
153         for (Entry<String, YamlProxyDataSourceConfiguration> entry : yamlDataSourceMap.entrySet()) {
154             DataSourceConfiguration dataSourceConfig = dataSourceConfigSwapper.swap(entry.getValue());
155             propsMap.put(entry.getKey(), DataSourcePoolPropertiesCreator.create(dataSourceConfig));
156         }
157         validateHandler.validate(propsMap);
158         try {
159             ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().registerStorageUnits(databaseName, propsMap);
160         } catch (final SQLException ex) {
161             throw new InvalidStorageUnitsException(Collections.singleton(ex.getMessage()));
162         }
163         Map<String, StorageUnit> storageUnits = ProxyContext.getInstance().getContextManager()
164                 .getMetaDataContexts().getMetaData().getDatabase(databaseName).getResourceMetaData().getStorageUnits();
165         Map<String, StorageNode> toBeAddedStorageNode = StorageUnitNodeMapUtils.fromDataSourcePoolProperties(propsMap);
166         for (Entry<String, DataSourcePoolProperties> entry : propsMap.entrySet()) {
167             storageUnits.put(entry.getKey(), new StorageUnit(toBeAddedStorageNode.get(entry.getKey()), entry.getValue(), DataSourcePoolCreator.create(entry.getValue())));
168         }
169     }

171     private void addRules(final String databaseName, final Collection<YamlRuleConfiguration> yamlRuleConfigs) {
172         if (null == yamlRuleConfigs || yamlRuleConfigs.isEmpty()) {
173             return;
174         }
175         Collection<RuleConfiguration> allRuleConfigs = new LinkedList<>();
176         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
177         ShardingSphereDatabase database = metaDataContexts.getMetaData().getDatabase(databaseName);
178         Map<Integer, Collection<RuleConfiguration>> ruleConfigsMap = new HashMap<>();
179         for (YamlRuleConfiguration each : yamlRuleConfigs) {
180             if (each instanceof YamlShardingRuleConfiguration) {
181                 YamlShardingRuleConfigurationSwapper swapper = new YamlShardingRuleConfigurationSwapper();
182                 ShardingRuleConfiguration shardingRuleConfig = swapper.swapToObject((YamlShardingRuleConfiguration) each);
183                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
184                 ruleConfigsMap.get(swapper.getOrder()).add(shardingRuleConfig);
185             } else if (each instanceof YamlReadwriteSplittingRuleConfiguration) {
186                 YamlReadwriteSplittingRuleConfigurationSwapper swapper = new YamlReadwriteSplittingRuleConfigurationSwapper();
187                 ReadwriteSplittingRuleConfiguration readwriteSplittingRuleConfig = swapper.swapToObject((YamlReadwriteSplittingRuleConfiguration) each);
188                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
189                 ruleConfigsMap.get(swapper.getOrder()).add(readwriteSplittingRuleConfig);
190             } else if (each instanceof YamlCompatibleEncryptRuleConfiguration) {
191                 YamlCompatibleEncryptRuleConfigurationSwapper swapper = new YamlCompatibleEncryptRuleConfigurationSwapper();
192                 CompatibleEncryptRuleConfiguration encryptRuleConfig = swapper.swapToObject((YamlCompatibleEncryptRuleConfiguration) each);
193                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
194                 ruleConfigsMap.get(swapper.getOrder()).add(encryptRuleConfig);
195             } else if (each instanceof YamlEncryptRuleConfiguration) {
196                 YamlEncryptRuleConfigurationSwapper swapper = new YamlEncryptRuleConfigurationSwapper();
197                 EncryptRuleConfiguration encryptRuleConfig = swapper.swapToObject((YamlEncryptRuleConfiguration) each);
198                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
199                 ruleConfigsMap.get(swapper.getOrder()).add(encryptRuleConfig);
200             } else if (each instanceof YamlShadowRuleConfiguration) {
201                 YamlShadowRuleConfigurationSwapper swapper = new YamlShadowRuleConfigurationSwapper();
202                 ShadowRuleConfiguration shadowRuleConfig = swapper.swapToObject((YamlShadowRuleConfiguration) each);
203                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
204                 ruleConfigsMap.get(swapper.getOrder()).add(shadowRuleConfig);
205             } else if (each instanceof YamlMaskRuleConfiguration) {
206                 YamlMaskRuleConfigurationSwapper swapper = new YamlMaskRuleConfigurationSwapper();
207                 MaskRuleConfiguration maskRuleConfig = swapper.swapToObject((YamlMaskRuleConfiguration) each);
208                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
209                 ruleConfigsMap.get(swapper.getOrder()).add(maskRuleConfig);
210             } else if (each instanceof YamlBroadcastRuleConfiguration) {
211                 YamlBroadcastRuleConfigurationSwapper swapper = new YamlBroadcastRuleConfigurationSwapper();
212                 BroadcastRuleConfiguration maskRuleConfig = swapper.swapToObject((YamlBroadcastRuleConfiguration) each);
213                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
214                 ruleConfigsMap.get(swapper.getOrder()).add(maskRuleConfig);
215             } else if (each instanceof YamlSingleRuleConfiguration) {
216                 YamlSingleRuleConfigurationSwapper swapper = new YamlSingleRuleConfigurationSwapper();
217                 SingleRuleConfiguration maskRuleConfig = swapper.swapToObject((YamlSingleRuleConfiguration) each);
218                 ruleConfigsMap.computeIfAbsent(swapper.getOrder(), key -> new LinkedList<>());
219                 ruleConfigsMap.get(swapper.getOrder()).add(maskRuleConfig);
220             }
221         }
222         ruleConfigsMap.keySet().stream().sorted(Comparator.reverseOrder()).collect(Collectors.toList())
223                 .forEach(each -> addRules(allRuleConfigs, ruleConfigsMap.get(each), database));
224         metaDataContexts.getPersistService().getDatabaseRulePersistService().persist(metaDataContexts.getMetaData().getDatabase(databaseName).getName(), allRuleConfigs);
225     }
226     
227     private void addRules(final Collection<RuleConfiguration> allRuleConfigs, final Collection<RuleConfiguration> ruleConfigs, final ShardingSphereDatabase database) {
228         RuleConfiguration ruleConfig = ruleConfigs.stream().findFirst().orElse(null);
229         if (null == ruleConfig) {
230             return;
231         }
232         if (ruleConfig instanceof ShardingRuleConfiguration) {
233             ruleConfigs.forEach(each -> addShardingRuleConfiguration((ShardingRuleConfiguration) each, allRuleConfigs, database));
234         } else if (ruleConfig instanceof ReadwriteSplittingRuleConfiguration) {
235             ruleConfigs.forEach(each -> addReadwriteSplittingRuleConfiguration((ReadwriteSplittingRuleConfiguration) each, allRuleConfigs, database));
236         } else if (ruleConfig instanceof EncryptRuleConfiguration) {
237             ruleConfigs.forEach(each -> addEncryptRuleConfiguration((EncryptRuleConfiguration) each, allRuleConfigs, database));
238         } else if (ruleConfig instanceof CompatibleEncryptRuleConfiguration) {
239             ruleConfigs.forEach(each -> addEncryptRuleConfiguration(((CompatibleEncryptRuleConfiguration) each).convertToEncryptRuleConfiguration(), allRuleConfigs, database));
240         } else if (ruleConfig instanceof ShadowRuleConfiguration) {
241             ruleConfigs.forEach(each -> addShadowRuleConfiguration((ShadowRuleConfiguration) each, allRuleConfigs, database));
242         } else if (ruleConfig instanceof MaskRuleConfiguration) {
243             ruleConfigs.forEach(each -> addMaskRuleConfiguration((MaskRuleConfiguration) each, allRuleConfigs, database));
244         } else if (ruleConfig instanceof BroadcastRuleConfiguration) {
245             ruleConfigs.forEach(each -> addBroadcastRuleConfiguration((BroadcastRuleConfiguration) each, allRuleConfigs, database));
246         } else if (ruleConfig instanceof SingleRuleConfiguration) {
247             ruleConfigs.forEach(each -> addSingleRuleConfiguration((SingleRuleConfiguration) each, allRuleConfigs, database));
248         }
249     }
250     
251     private void addShardingRuleConfiguration(final ShardingRuleConfiguration shardingRuleConfig, final Collection<RuleConfiguration> allRuleConfigs, final ShardingSphereDatabase database) {
252         InstanceContext instanceContext = ProxyContext.getInstance().getContextManager().getInstanceContext();
253         shardingRuleConfigImportChecker.check(database, shardingRuleConfig);
254         allRuleConfigs.add(shardingRuleConfig);
255         database.getRuleMetaData().getRules().add(new ShardingRule(shardingRuleConfig, database.getResourceMetaData().getStorageUnits().keySet(), instanceContext));
256     }
257     
258     private void addReadwriteSplittingRuleConfiguration(final ReadwriteSplittingRuleConfiguration readwriteSplittingRuleConfig,
259                                                         final Collection<RuleConfiguration> allRuleConfigs, final ShardingSphereDatabase database) {
260         InstanceContext instanceContext = ProxyContext.getInstance().getContextManager().getInstanceContext();
261         Collection<ShardingSphereRule> rules = database.getRuleMetaData().getRules();
262         readwriteSplittingRuleConfigImportChecker.check(database, readwriteSplittingRuleConfig);
263         allRuleConfigs.add(readwriteSplittingRuleConfig);
264         rules.add(new ReadwriteSplittingRule(database.getName(), readwriteSplittingRuleConfig, instanceContext));
265     }
266     
267     private void addEncryptRuleConfiguration(final EncryptRuleConfiguration encryptRuleConfig, final Collection<RuleConfiguration> allRuleConfigs, final ShardingSphereDatabase database) {
268         encryptRuleConfigImportChecker.check(database, encryptRuleConfig);
269         allRuleConfigs.add(encryptRuleConfig);
270         database.getRuleMetaData().getRules().add(new EncryptRule(database.getName(), encryptRuleConfig));
271     }

273     private void addShadowRuleConfiguration(final ShadowRuleConfiguration shadowRuleConfig, final Collection<RuleConfiguration> allRuleConfigs, final ShardingSphereDatabase database) {
274         shadowRuleConfigImportChecker.check(database, shadowRuleConfig);
275         allRuleConfigs.add(shadowRuleConfig);
276         database.getRuleMetaData().getRules().add(new ShadowRule(shadowRuleConfig));
277     }
278     
279     private void addMaskRuleConfiguration(final MaskRuleConfiguration maskRuleConfig, final Collection<RuleConfiguration> allRuleConfigs, final ShardingSphereDatabase database) {
280         maskRuleConfigImportChecker.check(database, maskRuleConfig);
281         allRuleConfigs.add(maskRuleConfig);
282         database.getRuleMetaData().getRules().add(new MaskRule(maskRuleConfig));
283     }
284     
285     private void addBroadcastRuleConfiguration(final BroadcastRuleConfiguration broadcastRuleConfig, final Collection<RuleConfiguration> allRuleConfigs, final ShardingSphereDatabase database) {
286         allRuleConfigs.add(broadcastRuleConfig);
287         database.getRuleMetaData().getRules().add(new BroadcastRule(broadcastRuleConfig, database.getName(),
288                 database.getResourceMetaData().getStorageUnits().entrySet().stream()
289                         .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new))));
290     }
291     
292     private void addSingleRuleConfiguration(final SingleRuleConfiguration singleRuleConfig, final Collection<RuleConfiguration> allRuleConfigs, final ShardingSphereDatabase database) {
293         allRuleConfigs.add(singleRuleConfig);
294         database.getRuleMetaData().getRules().add(
295                 new SingleRule(singleRuleConfig, database.getName(),
296                         database.getResourceMetaData().getStorageUnits().entrySet().stream()
297                                 .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
298                         database.getRuleMetaData().getRules()));
299     }
300     
301     private void dropDatabase(final String databaseName) {
302         ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().dropDatabase(databaseName);
303     }
