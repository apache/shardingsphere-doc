73  @ExtendWith(AutoMockExtension.class)
74  @StaticMockSettings(ProxyContext.class)
75  class OpenGaussComBatchBindExecutorTest {
76      
77      private final ShardingSphereSQLParserEngine parserEngine = new ShardingSphereSQLParserEngine("openGauss", new CacheOption(2000, 65535L), new CacheOption(128, 1024L), false);
78      
79      @SuppressWarnings("rawtypes")
80      @Test
81      void assertExecute() throws SQLException {
82          String statement = "S_1";
83          String sql = "insert into bmsql (id) values (?)";
84          SQLStatement sqlStatement = parserEngine.parse(sql, false);
85          SQLStatementContext sqlStatementContext = mock(InsertStatementContext.class);
86          when(sqlStatementContext.getSqlStatement()).thenReturn(sqlStatement);
87          ConnectionSession connectionSession = mockConnectionSession();
88          connectionSession.getServerPreparedStatementRegistry().addPreparedStatement(statement, new PostgreSQLServerPreparedStatement(sql, sqlStatementContext, Collections.emptyList()));
89          ContextManager contextManager = mockContextManager();
90          when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
91          Iterator<DatabasePacket<?>> actualPacketsIterator = new OpenGaussComBatchBindExecutor(mockComBatchBindPacket(), connectionSession).execute().iterator();
92          assertThat(actualPacketsIterator.next(), is(PostgreSQLBindCompletePacket.getInstance()));
93          assertThat(actualPacketsIterator.next(), instanceOf(PostgreSQLCommandCompletePacket.class));
94          assertFalse(actualPacketsIterator.hasNext());
95      }
96      
97      private ConnectionSession mockConnectionSession() throws SQLException {
98          ConnectionSession result = mock(ConnectionSession.class);
99          when(result.getConnectionContext()).thenReturn(new ConnectionContext());
100         when(result.getDatabaseName()).thenReturn("foo_db");
101         BackendConnection backendConnection = mock(BackendConnection.class);
102         Connection connection = mock(Connection.class, RETURNS_DEEP_STUBS);
103         when(connection.getMetaData().getURL()).thenReturn("jdbc:opengauss://127.0.0.1/foo_db");
104         when(backendConnection.getConnections(nullable(String.class), anyInt(), any(ConnectionMode.class))).thenReturn(Collections.singletonList(connection));
105         PreparedStatement preparedStatement = mock(PreparedStatement.class);
106         when(preparedStatement.getConnection()).thenReturn(connection);
107         JDBCBackendStatement backendStatement = mock(JDBCBackendStatement.class);
108         when(backendStatement.createStorageResource(any(ExecutionUnit.class), any(Connection.class), any(ConnectionMode.class), any(StatementOption.class), nullable(DatabaseType.class)))
109                 .thenReturn(preparedStatement);
110         when(result.getStatementManager()).thenReturn(backendStatement);
111         when(result.getBackendConnection()).thenReturn(backendConnection);
112         when(result.getServerPreparedStatementRegistry()).thenReturn(new ServerPreparedStatementRegistry());
113         return result;
114     }
115     
116     private OpenGaussComBatchBindPacket mockComBatchBindPacket() {
117         OpenGaussComBatchBindPacket result = mock(OpenGaussComBatchBindPacket.class);
118         when(result.getStatementId()).thenReturn("S_1");
119         when(result.readParameterSets(anyList())).thenReturn(Collections.singletonList(Collections.singletonList(0)));
120         return result;
121     }
122     
123     private ContextManager mockContextManager() {
124         ContextManager result = mock(ContextManager.class, RETURNS_DEEP_STUBS);
125         when(result.getMetaDataContexts().getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.KERNEL_EXECUTOR_SIZE)).thenReturn(0);
126         when(result.getMetaDataContexts().getMetaData().getProps().<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY)).thenReturn(1);
127         when(result.getMetaDataContexts().getMetaData().getProps().<Boolean>getValue(ConfigurationPropertyKey.SQL_SHOW)).thenReturn(false);
128         when(result.getMetaDataContexts().getMetaData().getGlobalRuleMetaData()).thenReturn(new ShardingSphereRuleMetaData(Arrays.asList(
129                 new SQLTranslatorRule(new DefaultSQLTranslatorRuleConfigurationBuilder().build()), new LoggingRule(new DefaultLoggingRuleConfigurationBuilder().build()))));
130         ShardingSphereDatabase database = mockDatabase();
131         when(result.getMetaDataContexts().getMetaData().getDatabase("foo_db")).thenReturn(database);
132         return result;
133     }
134     
135     private static ShardingSphereDatabase mockDatabase() {
136         ShardingSphereDatabase result = mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS);
137         when(result.getResourceMetaData().getAllInstanceDataSourceNames()).thenReturn(Collections.singletonList("foo_ds"));
138         when(result.getResourceMetaData().getStorageTypes()).thenReturn(Collections.singletonMap("foo_ds", new OpenGaussDatabaseType()));
139         when(result.getRuleMetaData()).thenReturn(new ShardingSphereRuleMetaData(Collections.emptyList()));
140         return result;
141     }
