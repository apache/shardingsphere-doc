86      private static final String CREATE_SHARDING_RULE_SQL = String.format("CREATE SHARDING TABLE RULE t_order("
87              + "STORAGE_UNITS(%s,%s),"
88              + "SHARDING_COLUMN=user_id,"
89              + "TYPE(NAME='hash_mod',PROPERTIES('sharding-count'='4')),"
90              + "KEY_GENERATE_STRATEGY(COLUMN=order_id,TYPE(NAME='snowflake'))"
91              + ")", PipelineContainerComposer.DS_0, PipelineContainerComposer.DS_1);
92      
93      private static final String SOURCE_TABLE_NAME = "t_order";
94      
95      private final ExecutorService executor = Executors.newSingleThreadExecutor();
96      
97      @ParameterizedTest(name = "{0}")
98      @EnabledIf("isEnabled")
99      @ArgumentsSource(PipelineE2ETestCaseArgumentsProvider.class)
100     void assertCDCDataImportSuccess(final PipelineTestParameter testParam) throws SQLException, InterruptedException {
101         if (TimeZone.getDefault() != TimeZone.getTimeZone("UTC") && PipelineEnvTypeEnum.DOCKER == PipelineE2EEnvironment.getInstance().getItEnvType()) {
102             // make sure the time zone of locally running program same with the database server at CI.
103             TimeZone.setDefault(TimeZone.getTimeZone("UTC"));
104         }
105         try (PipelineContainerComposer containerComposer = new PipelineContainerComposer(testParam, new CDCJobType())) {
106             for (String each : Arrays.asList(PipelineContainerComposer.DS_0, PipelineContainerComposer.DS_1)) {
107                 containerComposer.registerStorageUnit(each);
108             }
109             createOrderTableRule(containerComposer);
110             try (Connection connection = containerComposer.getProxyDataSource().getConnection()) {
111                 initSchemaAndTable(containerComposer, connection, 2);
112             }
113             DataSource jdbcDataSource = containerComposer.generateShardingSphereDataSourceFromProxy();
114             Pair<List<Object[]>, List<Object[]>> dataPair = PipelineCaseHelper.generateFullInsertData(containerComposer.getDatabaseType(), PipelineContainerComposer.TABLE_INIT_ROW_COUNT);
115             log.info("init data begin: {}", LocalDateTime.now());
116             DataSourceExecuteUtils.execute(jdbcDataSource, containerComposer.getExtraSQLCommand().getFullInsertOrder(SOURCE_TABLE_NAME), dataPair.getLeft());
117             log.info("init data end: {}", LocalDateTime.now());
118             try (
119                     Connection connection = DriverManager.getConnection(containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_4, false),
120                             containerComposer.getUsername(), containerComposer.getPassword())) {
121                 initSchemaAndTable(containerComposer, connection, 0);
122             }
123             startCDCClient(containerComposer);
124             Awaitility.await().atMost(10, TimeUnit.SECONDS).pollInterval(1, TimeUnit.SECONDS).until(() -> !containerComposer.queryForListWithLog("SHOW STREAMING LIST").isEmpty());
125             String jobId = containerComposer.queryForListWithLog("SHOW STREAMING LIST").get(0).get("id").toString();
126             containerComposer.waitIncrementTaskFinished(String.format("SHOW STREAMING STATUS '%s'", jobId));
127             String tableName = containerComposer.getDatabaseType().isSchemaAvailable() ? String.join(".", "test", SOURCE_TABLE_NAME) : SOURCE_TABLE_NAME;
128             containerComposer.startIncrementTask(new E2EIncrementalTask(jdbcDataSource, tableName, new SnowflakeKeyGenerateAlgorithm(), containerComposer.getDatabaseType(), 20));
129             containerComposer.getIncreaseTaskThread().join(10000L);
130             List<Map<String, Object>> actualProxyList;
131             try (Connection connection = jdbcDataSource.getConnection()) {
132                 ResultSet resultSet = connection.createStatement().executeQuery(String.format("SELECT * FROM %s ORDER BY order_id ASC", getOrderTableNameWithSchema(containerComposer)));
133                 actualProxyList = containerComposer.transformResultSetToList(resultSet);
134             }
135             Awaitility.await().atMost(20, TimeUnit.SECONDS).pollInterval(2, TimeUnit.SECONDS)
136                     .until(() -> listOrderRecords(containerComposer, getOrderTableNameWithSchema(containerComposer)).size() == actualProxyList.size());
137             SchemaTableName schemaTableName = containerComposer.getDatabaseType().isSchemaAvailable()
138                     ? new SchemaTableName(new SchemaName(PipelineContainerComposer.SCHEMA_NAME), new TableName(SOURCE_TABLE_NAME))
139                     : new SchemaTableName(new SchemaName(null), new TableName(SOURCE_TABLE_NAME));
140             PipelineDataSourceWrapper targetDataSource = new PipelineDataSourceWrapper(StorageContainerUtils.generateDataSource(
141                     containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_4, false),
142                     containerComposer.getUsername(), containerComposer.getPassword()), containerComposer.getDatabaseType());
143             PipelineDataSourceWrapper sourceDataSource = new PipelineDataSourceWrapper(jdbcDataSource, containerComposer.getDatabaseType());
144             StandardPipelineTableMetaDataLoader metaDataLoader = new StandardPipelineTableMetaDataLoader(targetDataSource);
145             PipelineTableMetaData tableMetaData = metaDataLoader.getTableMetaData(PipelineContainerComposer.SCHEMA_NAME, "t_order");
146             PipelineColumnMetaData primaryKeyMetaData = tableMetaData.getColumnMetaData(tableMetaData.getPrimaryKeyColumns().get(0));
147             ConsistencyCheckJobItemProgressContext progressContext = new ConsistencyCheckJobItemProgressContext("", 0);
148             SingleTableInventoryDataConsistencyChecker checker = new SingleTableInventoryDataConsistencyChecker("", sourceDataSource, targetDataSource, schemaTableName, schemaTableName,
149                     tableMetaData.getColumnNames(), primaryKeyMetaData, null, progressContext);
150             DataConsistencyCheckResult checkResult = checker.check(new DataMatchDataConsistencyCalculateAlgorithm());
151             assertTrue(checkResult.isMatched());
152         }
153     }
154     
155     private void createOrderTableRule(final PipelineContainerComposer containerComposer) throws SQLException {
156         containerComposer.proxyExecuteWithLog(CREATE_SHARDING_RULE_SQL, 2);
157     }
158     
159     @SneakyThrows(InterruptedException.class)
160     private void initSchemaAndTable(final PipelineContainerComposer containerComposer, final Connection connection, final int sleepSeconds) throws SQLException {
161         containerComposer.createSchema(connection, sleepSeconds);
162         String sql = containerComposer.getExtraSQLCommand().getCreateTableOrder(SOURCE_TABLE_NAME);
163         log.info("create table sql: {}", sql);
164         connection.createStatement().execute(sql);
165         if (sleepSeconds > 0) {
166             Thread.sleep(sleepSeconds * 1000L);
167         }
168     }
169     
170     private void startCDCClient(final PipelineContainerComposer containerComposer) {
171         DataSource dataSource = StorageContainerUtils.generateDataSource(containerComposer.appendExtraParameter(containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_4, false)),
172                 containerComposer.getUsername(), containerComposer.getPassword());
173         DataSourceRecordConsumer dataSourceRecordConsumer = new DataSourceRecordConsumer(dataSource, containerComposer.getDatabaseType());
174         StartCDCClientParameter parameter = new StartCDCClientParameter(dataSourceRecordConsumer);
175         parameter.setAddress("localhost");
176         parameter.setPort(containerComposer.getContainerComposer().getProxyCDCPort());
177         parameter.setUsername(ProxyContainerConstants.USERNAME);
178         parameter.setPassword(ProxyContainerConstants.PASSWORD);
179         parameter.setDatabase("sharding_db");
180         // TODO add full=false test case later
181         parameter.setFull(true);
182         String schema = containerComposer.getDatabaseType().isSchemaAvailable() ? "test" : "";
183         parameter.setSchemaTables(Collections.singletonList(SchemaTable.newBuilder().setTable(SOURCE_TABLE_NAME).setSchema(schema).build()));
184         CompletableFuture.runAsync(() -> new CDCClient(parameter).start(), executor).whenComplete((unused, throwable) -> {
185             if (null != throwable) {
186                 log.error("cdc client sync failed, ", throwable);
187             }
188         });
189     }
190     
191     private List<Map<String, Object>> listOrderRecords(final PipelineContainerComposer containerComposer, final String tableNameWithSchema) throws SQLException {
192         try (
193                 Connection connection = DriverManager.getConnection(
194                         containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_4, false), containerComposer.getUsername(), containerComposer.getPassword())) {
195             ResultSet resultSet = connection.createStatement().executeQuery(String.format("SELECT * FROM %s ORDER BY order_id ASC", tableNameWithSchema));
196             return containerComposer.transformResultSetToList(resultSet);
197         }
198     }
199     
200     private String getOrderTableNameWithSchema(final PipelineContainerComposer containerComposer) {
201         return containerComposer.getDatabaseType().isSchemaAvailable() ? String.join(".", PipelineContainerComposer.SCHEMA_NAME, SOURCE_TABLE_NAME) : SOURCE_TABLE_NAME;
202     }

204     private static boolean isEnabled() {
205         return PipelineE2ECondition.isEnabled(new MySQLDatabaseType(), new OpenGaussDatabaseType());
