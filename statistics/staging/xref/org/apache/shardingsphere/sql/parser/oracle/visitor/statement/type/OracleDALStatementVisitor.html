34  /**
35   * DAL statement visitor for Oracle.
36   */
37  public final class OracleDALStatementVisitor extends OracleStatementVisitor implements DALStatementVisitor {
38      
39      @Override
40      public ASTNode visitAlterResourceCost(final AlterResourceCostContext ctx) {
41          return new OracleAlterResourceCostStatement();
42      }
43      
44      @Override
45      public ASTNode visitExplain(final ExplainContext ctx) {
46          OracleExplainStatement result = new OracleExplainStatement();
47          OracleDMLStatementVisitor visitor = new OracleDMLStatementVisitor();
48          getGlobalParameterMarkerSegments().addAll(visitor.getGlobalParameterMarkerSegments());
49          getStatementParameterMarkerSegments().addAll(visitor.getStatementParameterMarkerSegments());
50          if (null != ctx.insert()) {
51              result.setSqlStatement((SQLStatement) visitor.visit(ctx.insert()));
52          } else if (null != ctx.delete()) {
53              result.setSqlStatement((SQLStatement) visitor.visit(ctx.delete()));
54          } else if (null != ctx.update()) {
55              result.setSqlStatement((SQLStatement) visitor.visit(ctx.update()));
56          } else if (null != ctx.select()) {
57              result.setSqlStatement((SQLStatement) visitor.visit(ctx.select()));
58          }
59          result.addParameterMarkerSegments(ctx.getParent() instanceof ExecuteContext ? getGlobalParameterMarkerSegments() : popAllStatementParameterMarkerSegments());
60          result.getVariableNames().addAll(getVariableNames());
61          return result;
62      }
63      
64      @Override
65      public ASTNode visitShow(final ShowContext ctx) {
66          return new OracleShowStatement();
67      }
68      
69      @Override
70      public ASTNode visitSpool(final SpoolContext ctx) {
71          OracleSpoolStatement result = new OracleSpoolStatement();
72          if (null != ctx.spoolFileName()) {
73              result.setFileName(ctx.spoolFileName().getText());
74          }
75          return result;
76      }
77  }
