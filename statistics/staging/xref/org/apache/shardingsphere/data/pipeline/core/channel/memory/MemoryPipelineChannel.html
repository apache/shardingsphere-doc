18  package org.apache.shardingsphere.data.pipeline.core.channel.memory;
19  
20  import lombok.SneakyThrows;
21  import org.apache.shardingsphere.data.pipeline.core.channel.PipelineChannelAckCallback;
22  import org.apache.shardingsphere.data.pipeline.core.channel.PipelineChannel;
23  import org.apache.shardingsphere.data.pipeline.core.ingest.record.Record;
24  
25  import java.util.Collections;
26  import java.util.LinkedList;
27  import java.util.List;
28  import java.util.concurrent.ArrayBlockingQueue;
29  import java.util.concurrent.BlockingQueue;
30  import java.util.concurrent.SynchronousQueue;
31  import java.util.concurrent.TimeUnit;
32  
33  /**
34   * Memory pipeline channel.
35   */
36  public final class MemoryPipelineChannel implements PipelineChannel {
37      
38      private final BlockingQueue<List<Record>> queue;
39      
40      private final PipelineChannelAckCallback ackCallback;
41      
42      public MemoryPipelineChannel(final int blockQueueSize, final PipelineChannelAckCallback ackCallback) {
43          queue = blockQueueSize < 1 ? new SynchronousQueue<>(true) : new ArrayBlockingQueue<>(blockQueueSize, true);
44          this.ackCallback = ackCallback;
45      }
46      
47      @SneakyThrows(InterruptedException.class)
48      @Override
49      public void push(final List<Record> records) {
50          queue.put(records);
51      }
52      
53      @SneakyThrows(InterruptedException.class)
54      @Override
55      public List<Record> fetch(final int batchSize, final long timeoutMillis) {
56          List<Record> result = new LinkedList<>();
57          long startMillis = System.currentTimeMillis();
58          int recordsCount = 0;
59          do {
60              List<Record> records = queue.poll(Math.max(0L, timeoutMillis - (System.currentTimeMillis() - startMillis)), TimeUnit.MILLISECONDS);
61              if (null == records || records.isEmpty()) {
62                  continue;
63              }
64              recordsCount += records.size();
65              result.addAll(records);
66          } while (recordsCount < batchSize && System.currentTimeMillis() - startMillis < timeoutMillis);
67          return result;
68      }
69      
70      @Override
71      public List<Record> peek() {
72          List<Record> result = queue.peek();
73          return null == result ? Collections.emptyList() : result;
74      }
75      
76      @Override
77      public List<Record> poll() {
78          List<Record> result = queue.poll();
79          return null == result ? Collections.emptyList() : result;
80      }
81      
82      @Override
83      public void ack(final List<Record> records) {
84          ackCallback.onAck(records);
85      }
86  }
