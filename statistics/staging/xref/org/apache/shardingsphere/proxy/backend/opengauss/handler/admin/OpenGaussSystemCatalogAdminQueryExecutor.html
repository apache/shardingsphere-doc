54  /**
55   * Select database executor for openGauss.
56   */
57  @SuppressWarnings("unused")
58  public final class OpenGaussSystemCatalogAdminQueryExecutor implements DatabaseAdminQueryExecutor {
59      
60      private static final String PG_CATALOG = "pg_catalog";
61      
62      private static final String DAT_COMPATIBILITY = "PG";
63      
64      private final String sql;
65      
66      @Getter
67      private QueryResultMetaData queryResultMetaData;
68      
69      @Getter
70      private MergedResult mergedResult;
71      
72      public OpenGaussSystemCatalogAdminQueryExecutor(final String sql) {
73          this.sql = SQLUtils.trimSemicolon(sql);
74      }
75      
76      @Override
77      public void execute(final ConnectionSession connectionSession) throws SQLException {
78          try (Connection connection = DriverManager.getConnection("jdbc:calcite:caseSensitive=false")) {
79              prepareCalciteConnection(connection);
80              connection.setSchema(PG_CATALOG);
81              try (PreparedStatement statement = connection.prepareStatement(sql); ResultSet resultSet = statement.executeQuery()) {
82                  queryResultMetaData = new JDBCQueryResultMetaData(resultSet.getMetaData());
83                  mergedResult = new IteratorStreamMergedResult(Collections.singletonList(new JDBCMemoryQueryResult(resultSet, connectionSession.getProtocolType())));
84              }
85          }
86      }
87      
88      private void prepareCalciteConnection(final Connection connection) throws SQLException {
89          CalciteConnection calciteConnection = connection.unwrap(CalciteConnection.class);
90          calciteConnection.getRootSchema().add(PG_CATALOG, new ReflectiveSchema(constructOgCatalog()));
91          calciteConnection.getRootSchema().add("version", ScalarFunctionImpl.create(getClass(), "version"));
92          calciteConnection.getRootSchema().add("gs_password_deadline", ScalarFunctionImpl.create(getClass(), "gsPasswordDeadline"));
93          calciteConnection.getRootSchema().add("intervaltonum", ScalarFunctionImpl.create(getClass(), "intervalToNum"));
94          calciteConnection.getRootSchema().add("gs_password_notifyTime", ScalarFunctionImpl.create(getClass(), "gsPasswordNotifyTime"));
95      }
96      
97      private OpenGaussSystemCatalog constructOgCatalog() {
98          Collection<String> allDatabaseNames = ProxyContext.getInstance().getAllDatabaseNames();
99          OpenGaussDatabase[] openGaussDatabases = new OpenGaussDatabase[allDatabaseNames.size()];
100         List<OpenGaussTables> openGaussTables = new LinkedList<>();
101         List<OpenGaussRoles> openGaussRoles = new LinkedList<>();
102         ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getGlobalRuleMetaData().getSingleRule(AuthorityRule.class)
103                 .getConfiguration().getUsers().stream().map(user -> user.getGrantee().getUsername())
104                 .forEach(userName -> openGaussRoles.add(new OpenGaussRoles(userName)));
105         int index = 0;
106         for (String each : allDatabaseNames) {
107             for (Entry<String, ShardingSphereSchema> entry : ProxyContext.getInstance().getDatabase(each).getSchemas().entrySet()) {
108                 for (String tableName : entry.getValue().getAllTableNames()) {
109                     openGaussTables.add(new OpenGaussTables(entry.getKey(), tableName));
110                 }
111             }
112             openGaussDatabases[index++] = new OpenGaussDatabase(each, DAT_COMPATIBILITY);
113         }
114         openGaussTables.addAll(SystemSchemaBuilderRule.OPEN_GAUSS_PG_CATALOG.getTables().stream().map(tableName -> new OpenGaussTables(PG_CATALOG, tableName)).collect(Collectors.toSet()));
115         openGaussTables.addAll(SystemSchemaBuilderRule.POSTGRESQL_PG_CATALOG.getTables().stream().map(tableName -> new OpenGaussTables(PG_CATALOG, tableName)).collect(Collectors.toSet()));
116         return new OpenGaussSystemCatalog(openGaussDatabases, openGaussTables.toArray(new OpenGaussTables[0]), openGaussRoles.toArray(new OpenGaussRoles[0]));
117     }
118     
119     /**
120      * Get version of ShardingSphere-Proxy.
121      *
122      * @return version message
123      */
124     public static String version() {
125         return "ShardingSphere-Proxy " + ShardingSphereVersion.VERSION + ("-" + ShardingSphereVersion.BUILD_GIT_COMMIT_ID_ABBREV) + (ShardingSphereVersion.BUILD_GIT_DIRTY ? "-dirty" : "");
126     }
127     
128     /**
129      * The type interval is not supported in standard JDBC.
130      * Indicates the number of remaining days before the password of the current user expires.
131      *
132      * @return 90 days
133      */
134     public static int gsPasswordDeadline() {
135         return 90;
136     }
137     
138     /**
139      * The type interval is not supported in standard JDBC.
140      * Convert interval to num.
141      *
142      * @param result result
143      * @return result
144      */
145     public static int intervalToNum(final int result) {
146         return result;
147     }
148     
149     /**
150      * Specifies the number of days prior to password expiration that a user will receive a reminder.
151      *
152      * @return 7 days
153      */
154     public static int gsPasswordNotifyTime() {
155         return 7;
156     }
