61  /**
62   * CDC job preparer.
63   */
64  @Slf4j
65  public final class CDCJobPreparer {
66      
67      private final CDCJobAPI jobAPI = new CDCJobAPI();
68      
69      /**
70       * Do prepare work.
71       *
72       * @param jobItemContexts job item contexts
73       */
74      public void initTasks(final Collection<CDCJobItemContext> jobItemContexts) {
75          // TODO Use pipeline tree to build it
76          AtomicBoolean inventoryImporterUsed = new AtomicBoolean();
77          List<CDCChannelProgressPair> inventoryChannelProgressPairs = new LinkedList<>();
78          AtomicBoolean incrementalImporterUsed = new AtomicBoolean();
79          List<CDCChannelProgressPair> incrementalChannelProgressPairs = new LinkedList<>();
80          for (CDCJobItemContext each : jobItemContexts) {
81              initTasks0(each, inventoryImporterUsed, inventoryChannelProgressPairs, incrementalImporterUsed, incrementalChannelProgressPairs);
82          }
83      }
84      
85      private void initTasks0(final CDCJobItemContext jobItemContext, final AtomicBoolean inventoryImporterUsed, final List<CDCChannelProgressPair> inventoryChannelProgressPairs,
86                              final AtomicBoolean incrementalImporterUsed, final List<CDCChannelProgressPair> incrementalChannelProgressPairs) {
87          Optional<InventoryIncrementalJobItemProgress> jobItemProgress = jobAPI.getJobItemProgress(jobItemContext.getJobId(), jobItemContext.getShardingItem());
88          if (!jobItemProgress.isPresent()) {
89              jobAPI.persistJobItemProgress(jobItemContext);
90          }
91          if (jobItemContext.isStopping()) {
92              PipelineJobCenter.stop(jobItemContext.getJobId());
93              return;
94          }
95          initIncrementalPosition(jobItemContext);
96          if (jobItemContext.getJobConfig().isFull()) {
97              initInventoryTasks(jobItemContext, inventoryImporterUsed, inventoryChannelProgressPairs);
98          }
99          initIncrementalTask(jobItemContext, incrementalImporterUsed, incrementalChannelProgressPairs);
100     }
101     
102     private void initIncrementalPosition(final CDCJobItemContext jobItemContext) {
103         CDCTaskConfiguration taskConfig = jobItemContext.getTaskConfig();
104         JobItemIncrementalTasksProgress initIncremental = null == jobItemContext.getInitProgress() ? null : jobItemContext.getInitProgress().getIncremental();
105         try {
106             taskConfig.getDumperConfig().setPosition(PipelineJobPreparerUtils.getIncrementalPosition(initIncremental, taskConfig.getDumperConfig(), jobItemContext.getDataSourceManager()));
107         } catch (final SQLException ex) {
108             throw new PrepareJobWithGetBinlogPositionException(jobItemContext.getJobId(), ex);
109         }
110     }
111     
112     private void initInventoryTasks(final CDCJobItemContext jobItemContext, final AtomicBoolean importerUsed, final List<CDCChannelProgressPair> channelProgressPairs) {
113         long startTimeMillis = System.currentTimeMillis();
114         CDCTaskConfiguration taskConfig = jobItemContext.getTaskConfig();
115         ImporterConfiguration importerConfig = taskConfig.getImporterConfig();
116         CDCProcessContext processContext = jobItemContext.getJobProcessContext();
117         for (InventoryDumperConfiguration each : new InventoryTaskSplitter(jobItemContext.getSourceDataSource(), new InventoryDumperConfiguration(taskConfig.getDumperConfig()), importerConfig)
118                 .splitInventoryDumperConfig(jobItemContext)) {
119             AtomicReference<IngestPosition> position = new AtomicReference<>(each.getPosition());
120             PipelineChannel channel = PipelineTaskUtils.createInventoryChannel(processContext.getPipelineChannelCreator(), importerConfig.getBatchSize(), position);
121             channelProgressPairs.add(new CDCChannelProgressPair(channel, jobItemContext));
122             Dumper dumper = new InventoryDumper(each, channel, jobItemContext.getSourceDataSource(), jobItemContext.getSourceMetaDataLoader());
123             Importer importer = importerUsed.get() ? null
124                     : new CDCImporter(channelProgressPairs, importerConfig.getBatchSize(), 3, TimeUnit.SECONDS, jobItemContext.getSink(),
125                             needSorting(ImporterType.INVENTORY, hasGlobalCSN(taskConfig.getDumperConfig().getDataSourceConfig().getDatabaseType())),
126                             importerConfig.getRateLimitAlgorithm());
127             jobItemContext.getInventoryTasks().add(new CDCInventoryTask(PipelineTaskUtils.generateInventoryTaskId(each), processContext.getInventoryDumperExecuteEngine(),
128                     processContext.getInventoryImporterExecuteEngine(), dumper, importer, position));
129             importerUsed.set(true);
130         }
131         log.info("initInventoryTasks cost {} ms", System.currentTimeMillis() - startTimeMillis);
132     }
133     
134     private boolean needSorting(final ImporterType importerType, final boolean hasGlobalCSN) {
135         return ImporterType.INCREMENTAL == importerType && hasGlobalCSN;
136     }
137     
138     private boolean hasGlobalCSN(final DatabaseType databaseType) {
139         return databaseType instanceof OpenGaussDatabaseType;
140     }
141     
142     private void initIncrementalTask(final CDCJobItemContext jobItemContext, final AtomicBoolean importerUsed, final List<CDCChannelProgressPair> channelProgressPairs) {
143         CDCTaskConfiguration taskConfig = jobItemContext.getTaskConfig();
144         DumperConfiguration dumperConfig = taskConfig.getDumperConfig();
145         ImporterConfiguration importerConfig = taskConfig.getImporterConfig();
146         IncrementalTaskProgress taskProgress = PipelineTaskUtils.createIncrementalTaskProgress(dumperConfig.getPosition(), jobItemContext.getInitProgress());
147         PipelineChannel channel = PipelineTaskUtils.createIncrementalChannel(importerConfig.getConcurrency(), jobItemContext.getJobProcessContext().getPipelineChannelCreator(), taskProgress);
148         channelProgressPairs.add(new CDCChannelProgressPair(channel, jobItemContext));
149         Dumper dumper = DatabaseTypedSPILoader.getService(IncrementalDumperCreator.class, dumperConfig.getDataSourceConfig().getDatabaseType())
150                 .createIncrementalDumper(dumperConfig, dumperConfig.getPosition(), channel, jobItemContext.getSourceMetaDataLoader());
151         boolean needSorting = needSorting(ImporterType.INCREMENTAL, hasGlobalCSN(importerConfig.getDataSourceConfig().getDatabaseType()));
152         Importer importer = importerUsed.get() ? null
153                 : new CDCImporter(channelProgressPairs, importerConfig.getBatchSize(), 300, TimeUnit.MILLISECONDS,
154                         jobItemContext.getSink(), needSorting, importerConfig.getRateLimitAlgorithm());
155         PipelineTask incrementalTask = new CDCIncrementalTask(dumperConfig.getDataSourceName(), jobItemContext.getJobProcessContext().getIncrementalExecuteEngine(), dumper, importer, taskProgress);
156         jobItemContext.getIncrementalTasks().add(incrementalTask);
157         importerUsed.set(true);
158     }
