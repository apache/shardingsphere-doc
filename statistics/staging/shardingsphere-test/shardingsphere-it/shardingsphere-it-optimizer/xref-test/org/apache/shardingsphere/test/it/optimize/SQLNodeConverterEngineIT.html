63      private static final SQLCases SQL_CASES = SQLCasesRegistry.getInstance().getCases();
64      
65      private static final SQLParserTestCases SQL_PARSER_TEST_CASES = SQLParserTestCasesRegistry.getInstance().getCases();
66      
67      private static final String SELECT_STATEMENT_PREFIX = "SELECT";
68      
69      @ParameterizedTest(name = "{0}")
70      @ArgumentsSource(TestCaseArgumentsProvider.class)
71      void assertConvert(final String sqlCaseId, final SQLCaseType sqlCaseType, final String databaseType) {
72          String sql = SQL_CASES.getSQL(sqlCaseId, sqlCaseType, SQL_PARSER_TEST_CASES.get(sqlCaseId).getParameters());
73          SqlNode actual = SQLNodeConverterEngine.convert(parseSQLStatement(databaseType, sql));
74          SqlNode expected = parseSQLNode(databaseType, sql);
75          assertTrue(actual.equalsDeep(expected, Litmus.THROW));
76      }
77      
78      private SQLStatement parseSQLStatement(final String databaseType, final String sql) {
79          return new SQLVisitorEngine(databaseType, "STATEMENT", true, new Properties()).visit(new SQLParserEngine(databaseType, new CacheOption(128, 1024L)).parse(sql, false));
80      }
81      
82      @SneakyThrows(SqlParseException.class)
83      private SqlNode parseSQLNode(final String databaseType, final String sql) {
84          return SqlParser.create(sql, createConfiguration(databaseType)).parseQuery();
85      }
86      
87      private Config createConfiguration(final String databaseType) {
88          CalciteConnectionConfig connectionConfig = new CalciteConnectionConfigImpl(createSQLDialectProperties(databaseType));
89          return SqlParser.config().withLex(connectionConfig.lex()).withUnquotedCasing(Casing.UNCHANGED)
90                  .withIdentifierMaxLength(SqlParser.DEFAULT_IDENTIFIER_MAX_LENGTH).withConformance(connectionConfig.conformance()).withParserFactory(SqlParserImpl.FACTORY);
91      }
92      
93      private Properties createSQLDialectProperties(final String databaseType) {
94          Properties result = new Properties();
95          result.setProperty(CalciteConnectionProperty.TIME_ZONE.camelName(), "UTC");
96          result.putAll(TypedSPILoader.getService(OptimizerSQLDialectBuilder.class, databaseType).build());
97          return result;
98      }

199         public Stream<? extends Arguments> provideArguments(final ExtensionContext extensionContext) {
200             return getTestParameters("MySQL", "PostgreSQL", "openGauss").stream();
201         }

203         private Collection<Arguments> getTestParameters(final String... databaseTypes) {
204             Collection<Arguments> result = new LinkedList<>();
205             for (InternalSQLParserTestParameter each : SQL_CASES.generateTestParameters(Arrays.stream(databaseTypes).collect(Collectors.toSet()))) {
206                 if (!isPlaceholderWithoutParameter(each) && isSupportedSQLCase(each)) {
207                     result.add(Arguments.of(each.getSqlCaseId(), each.getSqlCaseType(), "H2".equals(each.getDatabaseType()) ? "MySQL" : each.getDatabaseType()));
208                 }
209             }
210             return result;
211         }
212         
213         private boolean isPlaceholderWithoutParameter(final InternalSQLParserTestParameter testParam) {
214             return SQLCaseType.Placeholder == testParam.getSqlCaseType() && SQL_PARSER_TEST_CASES.get(testParam.getSqlCaseId()).getParameters().isEmpty();
215         }
216         
217         private boolean isSupportedSQLCase(final InternalSQLParserTestParameter testParam) {
218             return testParam.getSqlCaseId().toUpperCase().startsWith(SELECT_STATEMENT_PREFIX) && supportedSQLCaseIDs.contains(testParam.getSqlCaseId());
219         }
