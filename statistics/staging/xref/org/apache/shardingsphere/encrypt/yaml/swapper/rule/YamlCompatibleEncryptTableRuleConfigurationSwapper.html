18  package org.apache.shardingsphere.encrypt.yaml.swapper.rule;
19  
20  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptColumnRuleConfiguration;
21  import org.apache.shardingsphere.encrypt.api.config.rule.EncryptTableRuleConfiguration;
22  import org.apache.shardingsphere.encrypt.yaml.config.rule.YamlCompatibleEncryptColumnRuleConfiguration;
23  import org.apache.shardingsphere.encrypt.yaml.config.rule.YamlCompatibleEncryptTableRuleConfiguration;
24  import org.apache.shardingsphere.infra.util.yaml.swapper.YamlConfigurationSwapper;
25  
26  import java.util.Collection;
27  import java.util.LinkedList;
28  import java.util.Map.Entry;
29  
30  /**
31   * YAML encrypt table configuration swapper.
32   *
33   * @deprecated Should use new api, compatible api will remove in next version.
34   */
35  @Deprecated
36  public final class YamlCompatibleEncryptTableRuleConfigurationSwapper implements YamlConfigurationSwapper<YamlCompatibleEncryptTableRuleConfiguration, EncryptTableRuleConfiguration> {
37      
38      private final YamlCompatibleEncryptColumnRuleConfigurationSwapper columnSwapper = new YamlCompatibleEncryptColumnRuleConfigurationSwapper();
39      
40      @Override
41      public YamlCompatibleEncryptTableRuleConfiguration swapToYamlConfiguration(final EncryptTableRuleConfiguration data) {
42          YamlCompatibleEncryptTableRuleConfiguration result = new YamlCompatibleEncryptTableRuleConfiguration();
43          for (EncryptColumnRuleConfiguration each : data.getColumns()) {
44              result.getColumns().put(each.getName(), columnSwapper.swapToYamlConfiguration(each));
45          }
46          result.setName(data.getName());
47          return result;
48      }
49      
50      @Override
51      public EncryptTableRuleConfiguration swapToObject(final YamlCompatibleEncryptTableRuleConfiguration yamlConfig) {
52          Collection<EncryptColumnRuleConfiguration> columns = new LinkedList<>();
53          for (Entry<String, YamlCompatibleEncryptColumnRuleConfiguration> entry : yamlConfig.getColumns().entrySet()) {
54              YamlCompatibleEncryptColumnRuleConfiguration yamlEncryptColumnRuleConfig = entry.getValue();
55              yamlEncryptColumnRuleConfig.setLogicColumn(entry.getKey());
56              columns.add(columnSwapper.swapToObject(yamlEncryptColumnRuleConfig));
57          }
58          return new EncryptTableRuleConfiguration(yamlConfig.getName(), columns);
59      }
60  }
