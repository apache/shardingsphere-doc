18  package org.apache.shardingsphere.infra.config.props;
19  
20  import lombok.Getter;
21  import lombok.RequiredArgsConstructor;
22  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
23  import org.apache.shardingsphere.infra.props.TypedPropertyKey;
24  import org.slf4j.event.Level;
25  
26  import java.util.Arrays;
27  import java.util.Collection;
28  import java.util.stream.Collectors;
29  
30  /**
31   * Typed property key of configuration.
32   */
33  @RequiredArgsConstructor
34  @Getter
35  public enum ConfigurationPropertyKey implements TypedPropertyKey {
36      
37      /**
38       * The system log level.
39       */
40      SYSTEM_LOG_LEVEL("system-log-level", Level.INFO.toString(), Level.class, false),
41      
42      /**
43       * Whether show SQL in log.
44       */
45      SQL_SHOW("sql-show", String.valueOf(Boolean.FALSE), boolean.class, false),
46      
47      /**
48       * Whether show SQL details in simple style.
49       */
50      SQL_SIMPLE("sql-simple", String.valueOf(Boolean.FALSE), boolean.class, false),
51      
52      /**
53       * The max thread size of worker group to execute SQL.
54       */
55      KERNEL_EXECUTOR_SIZE("kernel-executor-size", String.valueOf(0), int.class, true),
56      
57      /**
58       * Max opened connection size for each query.
59       */
60      MAX_CONNECTIONS_SIZE_PER_QUERY("max-connections-size-per-query", String.valueOf(1), int.class, false),
61      
62      /**
63       * Whether validate table metadata consistency when application startup or updated.
64       */
65      CHECK_TABLE_METADATA_ENABLED("check-table-metadata-enabled", String.valueOf(Boolean.FALSE), boolean.class, false),
66      
67      /**
68       * Frontend database protocol for ShardingSphere-Proxy.
69       */
70      PROXY_FRONTEND_DATABASE_PROTOCOL_TYPE("proxy-frontend-database-protocol-type", null, DatabaseType.class, false),
71      
72      /**
73       * Flush threshold for every record from databases for ShardingSphere-Proxy.
74       */
75      PROXY_FRONTEND_FLUSH_THRESHOLD("proxy-frontend-flush-threshold", String.valueOf(128), int.class, false),
76      
77      /**
78       * Proxy backend query fetch size. A larger value may increase the memory usage of ShardingSphere Proxy.
79       * The default value is -1, which means set the minimum value for different JDBC drivers.
80       */
81      PROXY_BACKEND_QUERY_FETCH_SIZE("proxy-backend-query-fetch-size", String.valueOf(-1), int.class, false),
82      
83      /**
84       * Proxy frontend executor size. The default value is 0, which means let Netty decide.
85       */
86      PROXY_FRONTEND_EXECUTOR_SIZE("proxy-frontend-executor-size", String.valueOf(0), int.class, true),
87      
88      /**
89       * Less than or equal to 0 means no limitation.
90       */
91      PROXY_FRONTEND_MAX_CONNECTIONS("proxy-frontend-max-connections", "0", int.class, false),
92      
93      /**
94       * Proxy default start port.
95       */
96      PROXY_DEFAULT_PORT("proxy-default-port", "3307", int.class, true),
97      
98      /**
99       * Proxy Netty backlog size.
100      */
101     PROXY_NETTY_BACKLOG("proxy-netty-backlog", "1024", int.class, false),
102     
103     /**
104      * CDC server port.
105      */
106     CDC_SERVER_PORT("cdc-server-port", "33071", int.class, true),
107     
108     /**
109      * Proxy frontend SSL enabled.
110      */
111     PROXY_FRONTEND_SSL_ENABLED("proxy-frontend-ssl-enabled", String.valueOf(Boolean.FALSE), boolean.class, true),
112     
113     /**
114      * Proxy frontend SSL protocol version.
115      */
116     PROXY_FRONTEND_SSL_VERSION("proxy-frontend-ssl-version", "TLSv1.2,TLSv1.3", String.class, true),
117     
118     /**
119      * Proxy frontend SSL cipher.
120      */
121     PROXY_FRONTEND_SSL_CIPHER("proxy-frontend-ssl-cipher", "", String.class, true),
122     
123     /**
124      * Agent plugins enabled.
125      */
126     AGENT_PLUGINS_ENABLED("agent-plugins-enabled", String.valueOf(Boolean.TRUE), boolean.class, false);
127     
128     private final String key;
129     
130     private final String defaultValue;
131     
132     private final Class<?> type;
133     
134     private final boolean rebootRequired;
135     
136     /**
137      * Get property key names.
138      *
139      * @return collection of key names
140      */
141     public static Collection<String> getKeyNames() {
142         return Arrays.stream(values()).map(ConfigurationPropertyKey::name).collect(Collectors.toList());
143     }
144 }
