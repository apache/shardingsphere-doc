61  /**
62   * Projection assert.
63   */
64  @NoArgsConstructor(access = AccessLevel.PRIVATE)
65  public final class ProjectionAssert {
66      
67      /**
68       * Assert actual projections segment is correct with expected projections.
69       *
70       * @param assertContext assert context
71       * @param actual actual projection
72       * @param expected expected projections
73       */
74      public static void assertIs(final SQLCaseAssertContext assertContext, final ProjectionsSegment actual, final ExpectedProjections expected) {
75          assertProjections(assertContext, actual, expected);
76          List<ExpectedProjection> expectedProjections = expected.getExpectedProjections();
77          int count = 0;
78          for (ProjectionSegment each : actual.getProjections()) {
79              assertProjection(assertContext, each, expectedProjections.get(count));
80              count++;
81          }
82      }
83      
84      private static void assertProjections(final SQLCaseAssertContext assertContext, final ProjectionsSegment actual, final ExpectedProjections expected) {
85          assertThat(assertContext.getText("Projections size assertion error: "), actual.getProjections().size(), is(expected.getSize()));
86          assertThat(assertContext.getText("Projections distinct row assertion error: "), actual.isDistinctRow(), is(expected.isDistinctRow()));
87          SQLSegmentAssert.assertIs(assertContext, actual, expected);
88      }
89      
90      /**
91       * Assert actual projection segment is correct with expected projection.
92       * @param assertContext assert context
93       * @param actual actual projection
94       * @param expected expected projection
95       */
96      public static void assertProjection(final SQLCaseAssertContext assertContext, final ProjectionSegment actual, final ExpectedProjection expected) {
97          if (actual instanceof ShorthandProjectionSegment) {
98              assertThat(assertContext.getText("Projection type assertion error: "), expected, instanceOf(ExpectedShorthandProjection.class));
99              assertShorthandProjection(assertContext, (ShorthandProjectionSegment) actual, (ExpectedShorthandProjection) expected);
100         } else if (actual instanceof ColumnProjectionSegment) {
101             assertThat(assertContext.getText("Projection type assertion error: "), expected, instanceOf(ExpectedColumnProjection.class));
102             assertColumnProjection(assertContext, (ColumnProjectionSegment) actual, (ExpectedColumnProjection) expected);
103         } else if (actual instanceof AggregationProjectionSegment) {
104             assertThat(assertContext.getText("Projection type assertion error: "), expected, instanceOf(ExpectedAggregationProjection.class));
105             assertAggregationProjection(assertContext, (AggregationProjectionSegment) actual, (ExpectedAggregationProjection) expected);
106         } else if (actual instanceof ExpressionProjectionSegment) {
107             assertThat(assertContext.getText("Projection type assertion error: "), expected, instanceOf(ExpectedExpressionProjection.class));
108             assertExpressionProjection(assertContext, (ExpressionProjectionSegment) actual, (ExpectedExpressionProjection) expected);
109         } else if (actual instanceof TopProjectionSegment) {
110             assertThat(assertContext.getText("Projection type assertion error: "), expected, instanceOf(ExpectedTopProjection.class));
111             assertTopProjection(assertContext, (TopProjectionSegment) actual, (ExpectedTopProjection) expected);
112         } else if (actual instanceof SubqueryProjectionSegment) {
113             assertThat(assertContext.getText("Projection type assertion error: "), expected, instanceOf(ExpectedSubqueryProjection.class));
114             assertSubqueryProjection(assertContext, (SubqueryProjectionSegment) actual, (ExpectedSubqueryProjection) expected);
115         }
116         SQLSegmentAssert.assertIs(assertContext, actual, expected);
117     }
118     
119     private static void assertSubqueryProjection(final SQLCaseAssertContext assertContext, final SubqueryProjectionSegment actual, final ExpectedSubqueryProjection expected) {
120         assertThat(assertContext.getText("Subquery projection alias assertion error: "), actual.getAliasName().orElse(null), is(expected.getAlias()));
121         String expectedText = SQLCaseType.LITERAL == assertContext.getCaseType() && null != expected.getLiteralText() ? expected.getLiteralText() : expected.getText();
122         assertThat(assertContext.getText("Subquery projection text assertion error: "), actual.getText(), is(expectedText));
123         SelectStatementAssert.assertIs(assertContext, actual.getSubquery().getSelect(), expected.getSubquery().getSelectTestCases());
124     }
125     
126     private static void assertShorthandProjection(final SQLCaseAssertContext assertContext, final ShorthandProjectionSegment actual, final ExpectedShorthandProjection expected) {
127         if (null == expected.getOwner()) {
128             assertFalse(actual.getOwner().isPresent(), assertContext.getText("Actual owner should not exist."));
129         } else {
130             assertTrue(actual.getOwner().isPresent(), assertContext.getText("Actual owner should exist."));
131             OwnerAssert.assertIs(assertContext, actual.getOwner().get(), expected.getOwner());
132         }
133     }
134     
135     private static void assertColumnProjection(final SQLCaseAssertContext assertContext, final ColumnProjectionSegment actual, final ExpectedColumnProjection expected) {
136         assertThat(assertContext.getText("Column projection alias assertion error: "), actual.getAliasName().orElse(null), is(expected.getAlias()));
137         if (null != actual.getColumn().getNestedObjectAttributes()) {
138             assertThat(assertContext.getText("Nested Object attributes assertion error: "), actual.getColumn().getExpression(), is(expected.getName()));
139         } else {
140             IdentifierValueAssert.assertIs(assertContext, actual.getColumn().getIdentifier(), expected, "Column projection");
141         }
142         if (!expected.getParentheses().isEmpty()) {
143             assertThat(expected.getParentheses().size(), is(actual.getColumn().getParentheses().size()));
144             Iterator<ParenthesesSegment> iterator = actual.getColumn().getParentheses().iterator();
145             expected.getParentheses().forEach(each -> ParenthesesAssert.assertIs(assertContext, iterator.next(), each));
146         }
147         if (null == expected.getOwner()) {
148             assertFalse(actual.getColumn().getOwner().isPresent(), assertContext.getText("Actual owner should not exist."));
149         } else {
150             assertTrue(actual.getColumn().getOwner().isPresent(), assertContext.getText("Actual owner should exist."));
151             OwnerAssert.assertIs(assertContext, actual.getColumn().getOwner().get(), expected.getOwner());
152         }
153     }
154     
155     private static void assertAggregationProjection(final SQLCaseAssertContext assertContext, final AggregationProjectionSegment actual, final ExpectedAggregationProjection expected) {
156         assertThat(assertContext.getText("Aggregation projection type assertion error: "), actual.getType().name(), is(expected.getType()));
157         assertThat(assertContext.getText("Aggregation projection inner expression assertion error: "), actual.getExpression(), is(expected.getExpression()));
158         assertThat(assertContext.getText("Aggregation projection alias assertion error: "), actual.getAliasName().orElse(null), is(expected.getAlias()));
159         if (actual instanceof AggregationDistinctProjectionSegment) {
160             assertThat(assertContext.getText("Projection type assertion error: "), expected, instanceOf(ExpectedAggregationDistinctProjection.class));
161             assertThat(assertContext.getText("Aggregation projection distinct inner expression assertion error: "),
162                     ((AggregationDistinctProjectionSegment) actual).getDistinctInnerExpression(), is(((ExpectedAggregationDistinctProjection) expected).getDistinctInnerExpression()));
163         }
164     }
165     
166     private static void assertExpressionProjection(final SQLCaseAssertContext assertContext, final ExpressionProjectionSegment actual, final ExpectedExpressionProjection expected) {
167         assertThat(assertContext.getText("Expression projection alias assertion error: "),
168                 actual.getAliasName().orElse(null), is(expected.getAlias()));
169         String expectedText = SQLCaseType.LITERAL == assertContext.getCaseType() && null != expected.getLiteralText()
170                 ? expected.getLiteralText()
171                 : expected.getText();
172         assertThat(assertContext.getText("Expression projection text assertion error: "), actual.getText(), is(expectedText));
173         if (null != expected.getExpr()) {
174             ExpressionAssert.assertExpression(assertContext, actual.getExpr(), expected.getExpr());
175         }
176     }
177     
178     private static void assertTopProjection(final SQLCaseAssertContext assertContext, final TopProjectionSegment actual, final ExpectedTopProjection expected) {
179         if (actual.getTop() instanceof NumberLiteralRowNumberValueSegment) {
180             assertThat(assertContext.getText("Expression projection top value assertion error: "),
181                     ((NumberLiteralRowNumberValueSegment) actual.getTop()).getValue(), is(expected.getTopValue().getValue()));
182         } else {
183             assertThat(assertContext.getText("Expression projection top parameter index assertion error: "),
184                     ((ParameterMarkerRowNumberValueSegment) actual.getTop()).getParameterIndex(), is(expected.getTopValue().getParameterIndex()));
185         }
186         assertThat(assertContext.getText("Expression projection alias assertion error: "), actual.getAlias(), is(expected.getAlias()));
187         SQLSegmentAssert.assertIs(assertContext, actual, expected);
188     }
