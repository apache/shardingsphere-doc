156     @Test
157     void assertPersistForDirectory() throws Exception {
158         final String key = "/parent/child/test1";
159         final String value = "test1_content";
160         when(mockJdbcConnection.prepareStatement(fixture.selectByKeySQL())).thenReturn(mockPreparedStatement);
161         when(mockJdbcConnection.prepareStatement(fixture.insertSQL())).thenReturn(mockPreparedStatementForPersist);
162         when(mockPreparedStatement.executeQuery()).thenReturn(mockResultSet);
163         repository.persist(key, value);
164         int depthOfDirectory = (int) key.chars().filter(ch -> ch == '/').count();
165         int beginIndex = 0;
166         String parentDirectory = "/";
167         for (int i = 0; i < depthOfDirectory; i++) {
168             int separatorIndex = key.indexOf('/', beginIndex);
169             int nextSeparatorIndex = key.indexOf('/', separatorIndex + 1);
170             if (nextSeparatorIndex == -1) {
171                 nextSeparatorIndex = key.length();
172             }
173             String directoryPath = key.substring(0, nextSeparatorIndex);
174             // Verifying if get operation is called for every directory level
175             verify(mockPreparedStatement).setString(1, directoryPath);
176             // Verifying that during insert operation, setString at index 2 is called for every directory level
177             verify(mockPreparedStatementForPersist).setString(2, directoryPath);
178             // Verifying that during insert operation, setString at index 4 is called for every parent directory
179             verify(mockPreparedStatementForPersist).setString(4, parentDirectory);
180             beginIndex = nextSeparatorIndex;
181             parentDirectory = directoryPath;
182         }
183         // Verifying that during insert operation, setString at index 3 is called with "" for all the parent directories
184         verify(mockPreparedStatementForPersist, times(depthOfDirectory - 1)).setString(3, "");
185         // Verifying that during insert operation, setString at index 3 is called with the leaf node once
186         verify(mockPreparedStatementForPersist).setString(3, "test1_content");
187         // Verifying that during insert operation, setString at index 1 is called with a UUID
188         verify(mockPreparedStatementForPersist, times(depthOfDirectory)).setString(eq(1), anyString());
189         // Verifying that executeOperation in insert is called for all the directory levels
190         verify(mockPreparedStatementForPersist, times(depthOfDirectory)).executeUpdate();
191     }
