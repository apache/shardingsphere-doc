40  /**
41   * CDC schema table utility class.
42   */
43  @NoArgsConstructor(access = AccessLevel.PRIVATE)
44  public final class CDCSchemaTableUtils {
45      
46      /**
47       * Parse table expression with schema.
48       *
49       * @param database database
50       * @param schemaTables schema tables
51       * @return map key is schema, value is table names
52       */
53      public static Map<String, Set<String>> parseTableExpressionWithSchema(final ShardingSphereDatabase database, final Collection<SchemaTable> schemaTables) {
54          Collection<String> systemSchemas = DatabaseTypedSPILoader.getService(DialectSystemDatabase.class, database.getProtocolType()).getSystemSchemas();
55          if (schemaTables.stream().anyMatch(each -> "*".equals(each.getTable()) && ("*".equals(each.getSchema()) || each.getSchema().isEmpty()))) {
56              return parseTableExpressionWithAllTables(database, systemSchemas);
57          }
58          Map<String, Set<String>> result = new HashMap<>();
59          for (SchemaTable each : schemaTables) {
60              if ("*".equals(each.getSchema())) {
61                  result.putAll(parseTableExpressionWithAllSchema(database, systemSchemas, each));
62              } else if ("*".equals(each.getTable())) {
63                  result.putAll(parseTableExpressionWithAllTable(database, each));
64              } else {
65                  result.computeIfAbsent(each.getSchema(), ignored -> new HashSet<>()).add(each.getTable());
66              }
67          }
68          return result;
69      }
70      
71      private static Map<String, Set<String>> parseTableExpressionWithAllTables(final ShardingSphereDatabase database, final Collection<String> systemSchemas) {
72          Map<String, Set<String>> result = new HashMap<>(database.getSchemas().size(), 1);
73          for (Entry<String, ShardingSphereSchema> entry : database.getSchemas().entrySet()) {
74              if (!systemSchemas.contains(entry.getKey())) {
75                  entry.getValue().getAllTableNames().forEach(tableName -> result.computeIfAbsent(entry.getKey(), ignored -> new HashSet<>()).add(tableName));
76              }
77              
78          }
79          return result;
80      }
81      
82      private static Map<String, Set<String>> parseTableExpressionWithAllSchema(final ShardingSphereDatabase database, final Collection<String> systemSchemas, final SchemaTable table) {
83          Map<String, Set<String>> result = new HashMap<>();
84          for (Entry<String, ShardingSphereSchema> entry : database.getSchemas().entrySet()) {
85              if (!systemSchemas.contains(entry.getKey())) {
86                  entry.getValue().getAllTableNames().stream().filter(tableName -> tableName.equals(table.getTable())).findFirst()
87                          .ifPresent(optional -> result.computeIfAbsent(entry.getKey(), ignored -> new HashSet<>()).add(optional));
88              }
89          }
90          return result;
91      }
92      
93      private static Map<String, Set<String>> parseTableExpressionWithAllTable(final ShardingSphereDatabase database, final SchemaTable each) {
94          Map<String, Set<String>> result = new HashMap<>();
95          DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, database.getProtocolType());
96          String schemaName = each.getSchema().isEmpty() ? dialectDatabaseMetaData.getDefaultSchema().orElseThrow(() -> new IllegalStateException("Default schema should exist.")) : each.getSchema();
97          ShardingSphereSchema schema = database.getSchema(schemaName);
98          ShardingSpherePreconditions.checkNotNull(schema, () -> new SchemaNotFoundException(each.getSchema()));
99          schema.getAllTableNames().forEach(tableName -> result.computeIfAbsent(schemaName, ignored -> new HashSet<>()).add(tableName));
100         return result;
101     }
102     
103     /**
104      * Parse table expression without schema.
105      *
106      * @param database database
107      * @param tableNames table names
108      * @return parsed table names
109      */
110     public static Collection<String> parseTableExpressionWithoutSchema(final ShardingSphereDatabase database, final List<String> tableNames) {
111         ShardingSphereSchema schema = database.getSchema(database.getName());
112         Set<String> allTableNames = null == schema ? Collections.emptySet() : new HashSet<>(schema.getAllTableNames());
113         return tableNames.stream().anyMatch("*"::equals) ? allTableNames : new HashSet<>(tableNames);
114     }
115 }
