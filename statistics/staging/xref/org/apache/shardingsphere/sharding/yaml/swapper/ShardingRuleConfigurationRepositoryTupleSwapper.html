45  /**
46   * Sharding rule configuration repository tuple swapper.
47   */
48  public final class ShardingRuleConfigurationRepositoryTupleSwapper implements RepositoryTupleSwapper<ShardingRuleConfiguration, YamlShardingRuleConfiguration> {
49      
50      private final YamlShardingRuleConfigurationSwapper ruleConfigSwapper = new YamlShardingRuleConfigurationSwapper();
51      
52      private final RuleNodePath shardingRuleNodePath = new ShardingRuleNodePathProvider().getRuleNodePath();
53      
54      @Override
55      public Collection<RepositoryTuple> swapToRepositoryTuples(final YamlShardingRuleConfiguration yamlRuleConfig) {
56          Collection<RepositoryTuple> result = new LinkedList<>();
57          swapAlgorithms(yamlRuleConfig, result);
58          swapStrategies(yamlRuleConfig, result);
59          if (null != yamlRuleConfig.getDefaultShardingColumn()) {
60              result.add(new RepositoryTuple(shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_SHARDING_COLUMN).getPath(), yamlRuleConfig.getDefaultShardingColumn()));
61          }
62          if (null != yamlRuleConfig.getShardingCache()) {
63              result.add(new RepositoryTuple(shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.SHARDING_CACHE).getPath(), YamlEngine.marshal(yamlRuleConfig.getShardingCache())));
64          }
65          swapTableRules(yamlRuleConfig, result);
66          return result;
67      }
68      
69      private void swapAlgorithms(final YamlShardingRuleConfiguration yamlRuleConfig, final Collection<RepositoryTuple> repositoryTuples) {
70          for (Entry<String, YamlAlgorithmConfiguration> each : yamlRuleConfig.getShardingAlgorithms().entrySet()) {
71              repositoryTuples.add(new RepositoryTuple(shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.ALGORITHMS).getPath(each.getKey()), YamlEngine.marshal(each.getValue())));
72          }
73          for (Entry<String, YamlAlgorithmConfiguration> each : yamlRuleConfig.getKeyGenerators().entrySet()) {
74              repositoryTuples.add(new RepositoryTuple(shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.KEY_GENERATORS).getPath(each.getKey()), YamlEngine.marshal(each.getValue())));
75          }
76          for (Entry<String, YamlAlgorithmConfiguration> each : yamlRuleConfig.getAuditors().entrySet()) {
77              repositoryTuples.add(new RepositoryTuple(shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.AUDITORS).getPath(each.getKey()), YamlEngine.marshal(each.getValue())));
78          }
79      }
80      
81      private void swapStrategies(final YamlShardingRuleConfiguration yamlRuleConfig, final Collection<RepositoryTuple> repositoryTuples) {
82          if (null != yamlRuleConfig.getDefaultDatabaseStrategy()) {
83              repositoryTuples.add(new RepositoryTuple(
84                      shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_DATABASE_STRATEGY).getPath(), YamlEngine.marshal(yamlRuleConfig.getDefaultDatabaseStrategy())));
85          }
86          if (null != yamlRuleConfig.getDefaultTableStrategy()) {
87              repositoryTuples.add(new RepositoryTuple(
88                      shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_TABLE_STRATEGY).getPath(), YamlEngine.marshal(yamlRuleConfig.getDefaultTableStrategy())));
89          }
90          if (null != yamlRuleConfig.getDefaultKeyGenerateStrategy()) {
91              repositoryTuples.add(new RepositoryTuple(
92                      shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_KEY_GENERATE_STRATEGY).getPath(), YamlEngine.marshal(yamlRuleConfig.getDefaultKeyGenerateStrategy())));
93          }
94          if (null != yamlRuleConfig.getDefaultAuditStrategy()) {
95              repositoryTuples.add(new RepositoryTuple(
96                      shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_AUDIT_STRATEGY).getPath(), YamlEngine.marshal(yamlRuleConfig.getDefaultAuditStrategy())));
97          }
98      }
99      
100     private void swapTableRules(final YamlShardingRuleConfiguration yamlRuleConfig, final Collection<RepositoryTuple> repositoryTuples) {
101         for (YamlTableRuleConfiguration each : yamlRuleConfig.getTables().values()) {
102             repositoryTuples.add(new RepositoryTuple(shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.TABLES).getPath(each.getLogicTable()), YamlEngine.marshal(each)));
103         }
104         for (YamlShardingAutoTableRuleConfiguration each : yamlRuleConfig.getAutoTables().values()) {
105             repositoryTuples.add(new RepositoryTuple(shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.AUTO_TABLES).getPath(each.getLogicTable()), YamlEngine.marshal(each)));
106         }
107         for (String each : yamlRuleConfig.getBindingTables()) {
108             repositoryTuples.add(new RepositoryTuple(shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.BINDING_TABLES).getPath(getBindingGroupName(each)), each));
109         }
110     }
111     
112     private String getBindingGroupName(final String bindingGroup) {
113         return bindingGroup.contains(":") ? bindingGroup.substring(0, bindingGroup.indexOf(":")) : bindingGroup;
114     }
115     
116     @Override
117     public Optional<ShardingRuleConfiguration> swapToObject(final Collection<RepositoryTuple> repositoryTuples) {
118         List<RepositoryTuple> validRepositoryTuples = repositoryTuples.stream().filter(each -> shardingRuleNodePath.getRoot().isValidatedPath(each.getKey())).collect(Collectors.toList());
119         if (validRepositoryTuples.isEmpty()) {
120             return Optional.empty();
121         }
122         YamlShardingRuleConfiguration yamlRuleConfig = new YamlShardingRuleConfiguration();
123         Map<String, YamlTableRuleConfiguration> tables = new LinkedHashMap<>();
124         Map<String, YamlShardingAutoTableRuleConfiguration> autoTables = new LinkedHashMap<>();
125         Collection<String> bindingTables = new LinkedList<>();
126         Map<String, YamlAlgorithmConfiguration> shardingAlgorithms = new LinkedHashMap<>();
127         Map<String, YamlAlgorithmConfiguration> keyGenerators = new LinkedHashMap<>();
128         Map<String, YamlAlgorithmConfiguration> auditors = new LinkedHashMap<>();
129         for (RepositoryTuple each : validRepositoryTuples) {
130             shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.TABLES).getName(each.getKey())
131                     .ifPresent(optional -> tables.put(optional, YamlEngine.unmarshal(each.getValue(), YamlTableRuleConfiguration.class)));
132             shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.AUTO_TABLES).getName(each.getKey())
133                     .ifPresent(optional -> autoTables.put(optional, YamlEngine.unmarshal(each.getValue(), YamlShardingAutoTableRuleConfiguration.class)));
134             shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.BINDING_TABLES).getName(each.getKey()).ifPresent(optional -> bindingTables.add(each.getValue()));
135             shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.ALGORITHMS).getName(each.getKey())
136                     .ifPresent(optional -> shardingAlgorithms.put(optional, YamlEngine.unmarshal(each.getValue(), YamlAlgorithmConfiguration.class)));
137             shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.KEY_GENERATORS).getName(each.getKey())
138                     .ifPresent(optional -> keyGenerators.put(optional, YamlEngine.unmarshal(each.getValue(), YamlAlgorithmConfiguration.class)));
139             shardingRuleNodePath.getNamedItem(ShardingRuleNodePathProvider.AUDITORS).getName(each.getKey())
140                     .ifPresent(optional -> auditors.put(optional, YamlEngine.unmarshal(each.getValue(), YamlAlgorithmConfiguration.class)));
141             if (shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_DATABASE_STRATEGY).isValidatedPath(each.getKey())) {
142                 yamlRuleConfig.setDefaultDatabaseStrategy(YamlEngine.unmarshal(each.getValue(), YamlShardingStrategyConfiguration.class));
143             } else if (shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_TABLE_STRATEGY).isValidatedPath(each.getKey())) {
144                 yamlRuleConfig.setDefaultTableStrategy(YamlEngine.unmarshal(each.getValue(), YamlShardingStrategyConfiguration.class));
145             } else if (shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_KEY_GENERATE_STRATEGY).isValidatedPath(each.getKey())) {
146                 yamlRuleConfig.setDefaultKeyGenerateStrategy(YamlEngine.unmarshal(each.getValue(), YamlKeyGenerateStrategyConfiguration.class));
147             } else if (shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_AUDIT_STRATEGY).isValidatedPath(each.getKey())) {
148                 yamlRuleConfig.setDefaultAuditStrategy(YamlEngine.unmarshal(each.getValue(), YamlShardingAuditStrategyConfiguration.class));
149             } else if (shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.DEFAULT_SHARDING_COLUMN).isValidatedPath(each.getKey())) {
150                 yamlRuleConfig.setDefaultShardingColumn(each.getValue());
151             } else if (shardingRuleNodePath.getUniqueItem(ShardingRuleNodePathProvider.SHARDING_CACHE).isValidatedPath(each.getKey())) {
152                 yamlRuleConfig.setShardingCache(YamlEngine.unmarshal(each.getValue(), YamlShardingCacheConfiguration.class));
153             }
154         }
155         yamlRuleConfig.setTables(tables);
156         yamlRuleConfig.setAutoTables(autoTables);
157         yamlRuleConfig.setBindingTables(bindingTables);
158         yamlRuleConfig.setShardingAlgorithms(shardingAlgorithms);
159         yamlRuleConfig.setKeyGenerators(keyGenerators);
160         yamlRuleConfig.setAuditors(auditors);
161         return Optional.of(ruleConfigSwapper.swapToObject(yamlRuleConfig));
162     }
163     
164     @Override
165     public Class<ShardingRuleConfiguration> getTypeClass() {
166         return ShardingRuleConfiguration.class;
167     }
168     
169     @Override
170     public String getRuleTagName() {
171         return "SHARDING";
172     }
173     
174     @Override
175     public int getOrder() {
176         return ShardingOrder.ORDER;
177     }
