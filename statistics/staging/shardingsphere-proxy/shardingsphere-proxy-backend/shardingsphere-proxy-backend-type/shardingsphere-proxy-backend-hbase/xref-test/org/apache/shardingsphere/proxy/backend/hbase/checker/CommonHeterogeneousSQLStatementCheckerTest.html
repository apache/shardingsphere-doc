18  package org.apache.shardingsphere.proxy.backend.hbase.checker;
19  
20  import org.apache.shardingsphere.proxy.backend.hbase.result.HBaseSupportedSQLStatement;
21  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.expr.BetweenExpression;
22  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.predicate.WhereSegment;
23  import org.apache.shardingsphere.sql.parser.sql.common.statement.SQLStatement;
24  import org.apache.shardingsphere.sql.parser.sql.common.statement.dml.SelectStatement;
25  import org.junit.jupiter.api.Test;
26  
27  import java.util.Optional;
28  
29  import static org.hamcrest.CoreMatchers.is;
30  import static org.hamcrest.MatcherAssert.assertThat;
31  import static org.junit.jupiter.api.Assertions.assertFalse;
32  import static org.junit.jupiter.api.Assertions.assertThrows;
33  import static org.junit.jupiter.api.Assertions.assertTrue;
34  
35  class CommonHeterogeneousSQLStatementCheckerTest {
36      
37      @Test
38      void assertIsSinglePoint() {
39          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where rowKey = '1'");
40          new CommonHeterogeneousSQLStatementChecker<>(sqlStatement).checkIsSinglePointQuery(sqlStatement.getWhere());
41      }
42      
43      @Test
44      void assertIsSinglePointWithErrorKey() {
45          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where a = '1'");
46          Exception ex = assertThrows(IllegalArgumentException.class, () -> new CommonHeterogeneousSQLStatementChecker<>(sqlStatement).checkIsSinglePointQuery(sqlStatement.getWhere()));
47          assertThat(ex.getMessage(), is("a is not a allowed key"));
48      }
49      
50      @Test
51      void assertIsSinglePointWithErrorOperation() {
52          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where rowKey between '1' and '2' ");
53          Exception ex = assertThrows(IllegalArgumentException.class, () -> new CommonHeterogeneousSQLStatementChecker<>(sqlStatement).checkIsSinglePointQuery(sqlStatement.getWhere()));
54          assertThat(ex.getMessage(), is("Only Support BinaryOperationExpression"));
55      }
56      
57      @Test
58      void assertInExpression() {
59          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where rowKey in ('1', '2') ");
60          Optional<WhereSegment> whereSegment = sqlStatement.getWhere();
61          assertTrue(whereSegment.isPresent());
62          new CommonHeterogeneousSQLStatementChecker<>(sqlStatement).checkInExpressionIsExpected(whereSegment.get().getExpr());
63      }
64      
65      @Test
66      void assertInExpressionWithNotIn() {
67          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where rowKey not in ('1', '2') ");
68          Optional<WhereSegment> whereSegment = sqlStatement.getWhere();
69          assertTrue(whereSegment.isPresent());
70          Exception ex = assertThrows(IllegalArgumentException.class, () -> new CommonHeterogeneousSQLStatementChecker<>(sqlStatement).checkInExpressionIsExpected(whereSegment.get().getExpr()));
71          assertThat(ex.getMessage(), is("Do not supported `not in`"));
72      }
73      
74      @Test
75      void assertInExpressionWithErrorKey() {
76          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where a in ('1', '2') ");
77          Optional<WhereSegment> whereSegment = sqlStatement.getWhere();
78          assertTrue(whereSegment.isPresent());
79          Exception ex = assertThrows(IllegalArgumentException.class, () -> new CommonHeterogeneousSQLStatementChecker<>(sqlStatement).checkInExpressionIsExpected(whereSegment.get().getExpr()));
80          assertThat(ex.getMessage(), is("a is not a allowed key"));
81      }
82      
83      @Test
84      void assertIsAllowExpressionSegment() {
85          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where rowKey BETWEEN 'v1' AND 'v2' ");
86          Optional<WhereSegment> whereSegment = sqlStatement.getWhere();
87          assertTrue(whereSegment.isPresent());
88          BetweenExpression betweenExpression = (BetweenExpression) whereSegment.get().getExpr();
89          CommonHeterogeneousSQLStatementChecker<SQLStatement> commonHeterogeneousSQLStatementChecker = new CommonHeterogeneousSQLStatementChecker<>(sqlStatement);
90          assertTrue(commonHeterogeneousSQLStatementChecker.isAllowExpressionSegment(betweenExpression.getBetweenExpr()));
91          assertTrue(commonHeterogeneousSQLStatementChecker.isAllowExpressionSegment(betweenExpression.getAndExpr()));
92      }
93      
94      @Test
95      void assertIsAllowExpressionSegmentError() {
96          SelectStatement sqlStatement = (SelectStatement) HBaseSupportedSQLStatement.parseSQLStatement("select /*+ hbase */ * from t_order where rowKey = '1'");
97          Optional<WhereSegment> whereSegment = sqlStatement.getWhere();
98          assertTrue(whereSegment.isPresent());
99          assertFalse(new CommonHeterogeneousSQLStatementChecker<>(sqlStatement).isAllowExpressionSegment(whereSegment.get().getExpr()));
100     }
101 }
