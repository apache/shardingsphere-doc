40  import static org.mockito.ArgumentMatchers.any;

43  import static org.mockito.Mockito.when;

45  @ExtendWith(AutoMockExtension.class)
46  @StaticMockSettings(ProcessRegistry.class)
47  class ProcessEngineTest {
48      
49      @Mock
50      private ProcessRegistry processRegistry;
51      
52      @BeforeEach
53      void setUp() {
54          when(ProcessRegistry.getInstance()).thenReturn(processRegistry);
55      }
56      
57      @Test
58      void assertExecuteSQL() {
59          ExecutionGroupContext<? extends SQLExecutionUnit> executionGroupContext = mockExecutionGroupContext();
60          new ProcessEngine().executeSQL(executionGroupContext, new QueryContext(new UpdateStatementContext(getSQLStatement()), null, null));
61          verify(processRegistry).add(any());
62      }
63      
64      @SuppressWarnings("unchecked")
65      private ExecutionGroupContext<? extends SQLExecutionUnit> mockExecutionGroupContext() {
66          ExecutionGroupContext<? extends SQLExecutionUnit> result = mock(ExecutionGroupContext.class);
67          ExecutionGroupReportContext reportContext = mock(ExecutionGroupReportContext.class);
68          when(reportContext.getProcessId()).thenReturn(UUID.fromString("00000000-000-0000-0000-000000000001").toString());
69          when(result.getReportContext()).thenReturn(reportContext);
70          return result;
71      }
72      
73      private MySQLUpdateStatement getSQLStatement() {
74          MySQLUpdateStatement result = new MySQLUpdateStatement();
75          result.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("foo_tbl"))));
76          result.setSetAssignment(new SetAssignmentSegment(0, 0, Collections.emptyList()));
77          return result;
78      }
79      
80      @Test
81      void assertCompleteSQLUnitExecution() {
82          ProcessIdContext.set("foo_id");
83          when(processRegistry.get("foo_id")).thenReturn(mock(Process.class));
84          new ProcessEngine().completeSQLUnitExecution();
85          verify(processRegistry).get("foo_id");
86          ProcessIdContext.remove();
87      }
88  }
