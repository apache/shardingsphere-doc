52  @NoArgsConstructor(access = AccessLevel.PRIVATE)

55      private static final Collection<String> EXCLUDE_BIND_COLUMNS = new LinkedHashSet<>(Arrays.asList("ROWNUM", "ROW_NUMBER", "ROWNUM_", "SYSDATE", "SYSTIMESTAMP", "CURRENT_TIMESTAMP",
56              "LOCALTIMESTAMP", "UID", "USER", "NEXTVAL", "ROWID", "LEVEL"));
57      
58      private static final Map<SegmentType, String> SEGMENT_TYPE_MESSAGES = Maps.of(SegmentType.PROJECTION, "field list", SegmentType.JOIN_ON, "on clause", SegmentType.JOIN_USING, "from clause",
59              SegmentType.PREDICATE, "where clause", SegmentType.ORDER_BY, "order clause", SegmentType.GROUP_BY, "group statement", SegmentType.INSERT_COLUMNS, "field list");
60      
61      private static final String UNKNOWN_SEGMENT_TYPE_MESSAGE = "unknown clause";
62      
63      /**
64       * Bind column segment with metadata.
65       *
66       * @param segment table segment
67       * @param parentSegmentType parent segment type
68       * @param statementBinderContext statement binder context
69       * @param tableBinderContexts table binder contexts
70       * @param outerTableBinderContexts outer table binder contexts
71       * @return bounded column segment
72       */
73      public static ColumnSegment bind(final ColumnSegment segment, final SegmentType parentSegmentType, final SQLStatementBinderContext statementBinderContext,
74                                       final Map<String, TableSegmentBinderContext> tableBinderContexts, final Map<String, TableSegmentBinderContext> outerTableBinderContexts) {
75          if (EXCLUDE_BIND_COLUMNS.contains(segment.getIdentifier().getValue().toUpperCase())) {
76              return segment;
77          }
78          ColumnSegment result = new ColumnSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getIdentifier());
79          segment.getOwner().ifPresent(result::setOwner);
80          Collection<TableSegmentBinderContext> tableBinderContextValues =
81                  getTableSegmentBinderContexts(segment, parentSegmentType, statementBinderContext, tableBinderContexts, outerTableBinderContexts);
82          Optional<ColumnSegment> inputColumnSegment = findInputColumnSegment(segment, parentSegmentType, tableBinderContextValues, outerTableBinderContexts, statementBinderContext);
83          inputColumnSegment.ifPresent(optional -> result.setVariable(optional.isVariable()));
84          result.setColumnBoundedInfo(createColumnSegmentBoundedInfo(segment, inputColumnSegment.orElse(null)));
85          segment.getParentheses().forEach(each -> result.getParentheses().add(each));
86          return result;
87      }
88      
89      private static Collection<TableSegmentBinderContext> getTableSegmentBinderContexts(final ColumnSegment segment, final SegmentType parentSegmentType,
90                                                                                         final SQLStatementBinderContext statementBinderContext,
91                                                                                         final Map<String, TableSegmentBinderContext> tableBinderContexts,
92                                                                                         final Map<String, TableSegmentBinderContext> outerTableBinderContexts) {
93          if (segment.getOwner().isPresent()) {
94              return getTableBinderContextByOwner(segment.getOwner().get().getIdentifier().getValue().toLowerCase(), tableBinderContexts, outerTableBinderContexts,
95                      statementBinderContext.getExternalTableBinderContexts());
96          }
97          if (!statementBinderContext.getJoinTableProjectionSegments().isEmpty() && isNeedUseJoinTableProjectionBind(segment, parentSegmentType, statementBinderContext)) {
98              return Collections.singleton(new SimpleTableSegmentBinderContext(statementBinderContext.getJoinTableProjectionSegments()));
99          }
100         return tableBinderContexts.values();
101     }
102     
103     private static boolean isNeedUseJoinTableProjectionBind(final ColumnSegment segment, final SegmentType parentSegmentType, final SQLStatementBinderContext statementBinderContext) {
104         return SegmentType.PROJECTION == parentSegmentType
105                 || SegmentType.PREDICATE == parentSegmentType && statementBinderContext.getUsingColumnNames().contains(segment.getIdentifier().getValue().toLowerCase());
106     }
107     
108     private static Collection<TableSegmentBinderContext> getTableBinderContextByOwner(final String owner, final Map<String, TableSegmentBinderContext> tableBinderContexts,
109                                                                                       final Map<String, TableSegmentBinderContext> outerTableBinderContexts,
110                                                                                       final Map<String, TableSegmentBinderContext> externalTableBinderContexts) {
111         if (tableBinderContexts.containsKey(owner)) {
112             return Collections.singleton(tableBinderContexts.get(owner));
113         }
114         if (outerTableBinderContexts.containsKey(owner)) {
115             return Collections.singleton(outerTableBinderContexts.get(owner));
116         }
117         if (externalTableBinderContexts.containsKey(owner)) {
118             return Collections.singleton(externalTableBinderContexts.get(owner));
119         }
120         return Collections.emptyList();
121     }
122     
123     private static Optional<ColumnSegment> findInputColumnSegment(final ColumnSegment segment, final SegmentType parentSegmentType, final Collection<TableSegmentBinderContext> tableBinderContexts,
124                                                                   final Map<String, TableSegmentBinderContext> outerTableBinderContexts, final SQLStatementBinderContext statementBinderContext) {
125         ColumnSegment result = null;
126         boolean isFindInputColumn = false;
127         for (TableSegmentBinderContext each : tableBinderContexts) {
128             Optional<ProjectionSegment> projectionSegment = each.findProjectionSegmentByColumnLabel(segment.getIdentifier().getValue());
129             if (projectionSegment.isPresent() && projectionSegment.get() instanceof ColumnProjectionSegment) {
130                 ShardingSpherePreconditions.checkState(null == result,
131                         () -> new AmbiguousColumnException(segment.getExpression(), SEGMENT_TYPE_MESSAGES.getOrDefault(parentSegmentType, UNKNOWN_SEGMENT_TYPE_MESSAGE)));
132                 result = ((ColumnProjectionSegment) projectionSegment.get()).getColumn();
133             }
134             if (!isFindInputColumn && projectionSegment.isPresent()) {
135                 isFindInputColumn = true;
136             }
137         }
138         if (!isFindInputColumn) {
139             Optional<ProjectionSegment> projectionSegment = findInputColumnSegmentFromOuterTable(segment, outerTableBinderContexts);
140             isFindInputColumn = projectionSegment.isPresent();
141             if (projectionSegment.isPresent() && projectionSegment.get() instanceof ColumnProjectionSegment) {
142                 result = ((ColumnProjectionSegment) projectionSegment.get()).getColumn();
143             }
144         }
145         if (!isFindInputColumn) {
146             Optional<ProjectionSegment> projectionSegment = findInputColumnSegmentFromExternalTables(segment, statementBinderContext.getExternalTableBinderContexts());
147             isFindInputColumn = projectionSegment.isPresent();
148             if (projectionSegment.isPresent() && projectionSegment.get() instanceof ColumnProjectionSegment) {
149                 result = ((ColumnProjectionSegment) projectionSegment.get()).getColumn();
150             }
151         }
152         if (!isFindInputColumn) {
153             result = findInputColumnSegmentByVariables(segment, statementBinderContext.getVariableNames()).orElse(null);
154             isFindInputColumn = null != result;
155         }
156         if (!isFindInputColumn) {
157             result = findInputColumnSegmentByPivotColumns(segment, statementBinderContext.getPivotColumnNames()).orElse(null);
158             isFindInputColumn = null != result;
159         }
160         ShardingSpherePreconditions.checkState(isFindInputColumn || containsFunctionTable(tableBinderContexts, outerTableBinderContexts.values()),
161                 () -> new ColumnNotFoundException(segment.getExpression(), SEGMENT_TYPE_MESSAGES.getOrDefault(parentSegmentType, UNKNOWN_SEGMENT_TYPE_MESSAGE)));
162         return Optional.ofNullable(result);
163     }
164     
165     private static Optional<ColumnSegment> findInputColumnSegmentByPivotColumns(final ColumnSegment segment, final Collection<String> pivotColumnNames) {
166         if (pivotColumnNames.isEmpty()) {
167             return Optional.empty();
168         }
169         if (pivotColumnNames.contains(segment.getIdentifier().getValue().toLowerCase())) {
170             return Optional.of(new ColumnSegment(0, 0, segment.getIdentifier()));
171         }
172         return Optional.empty();
173     }
174     
175     private static Optional<ProjectionSegment> findInputColumnSegmentFromOuterTable(final ColumnSegment segment, final Map<String, TableSegmentBinderContext> outerTableBinderContexts) {
176         ListIterator<TableSegmentBinderContext> listIterator = new ArrayList<>(outerTableBinderContexts.values()).listIterator(outerTableBinderContexts.size());
177         while (listIterator.hasPrevious()) {
178             TableSegmentBinderContext each = listIterator.previous();
179             Optional<ProjectionSegment> result = each.findProjectionSegmentByColumnLabel(segment.getIdentifier().getValue());
180             if (result.isPresent()) {
181                 return result;
182             }
183         }
184         return Optional.empty();
185     }
186     
187     private static Optional<ProjectionSegment> findInputColumnSegmentFromExternalTables(final ColumnSegment segment, final Map<String, TableSegmentBinderContext> externalTableBinderContexts) {
188         for (TableSegmentBinderContext each : externalTableBinderContexts.values()) {
189             Optional<ProjectionSegment> result = each.findProjectionSegmentByColumnLabel(segment.getIdentifier().getValue());
190             if (result.isPresent()) {
191                 return result;
192             }
193         }
194         return Optional.empty();
195     }
196     
197     private static Optional<ColumnSegment> findInputColumnSegmentByVariables(final ColumnSegment segment, final Collection<String> variableNames) {
198         if (variableNames.isEmpty()) {
199             return Optional.empty();
200         }
201         if (variableNames.contains(segment.getIdentifier().getValue().toLowerCase())) {
202             ColumnSegment result = new ColumnSegment(0, 0, segment.getIdentifier());
203             result.setVariable(true);
204             return Optional.of(result);
205         }
206         return Optional.empty();
207     }
208     
209     private static boolean containsFunctionTable(final Collection<TableSegmentBinderContext> tableBinderContexts, final Collection<TableSegmentBinderContext> outerBinderContexts) {
210         for (TableSegmentBinderContext each : tableBinderContexts) {
211             if (each instanceof FunctionTableSegmentBinderContext) {
212                 return true;
213             }
214         }
215         for (TableSegmentBinderContext each : outerBinderContexts) {
216             if (each instanceof FunctionTableSegmentBinderContext) {
217                 return true;
218             }
219         }
220         return false;
221     }

223     private static ColumnSegmentBoundedInfo createColumnSegmentBoundedInfo(final ColumnSegment segment, final ColumnSegment inputColumnSegment) {
224         IdentifierValue originalDatabase = null == inputColumnSegment ? null : inputColumnSegment.getColumnBoundedInfo().getOriginalDatabase();
225         IdentifierValue originalSchema = null == inputColumnSegment ? null : inputColumnSegment.getColumnBoundedInfo().getOriginalSchema();
226         IdentifierValue segmentOriginalTable = segment.getColumnBoundedInfo().getOriginalTable();
227         IdentifierValue originalTable = Strings.isNullOrEmpty(segmentOriginalTable.getValue())
228                 ? Optional.ofNullable(inputColumnSegment).map(optional -> optional.getColumnBoundedInfo().getOriginalTable()).orElse(segmentOriginalTable)
229                 : segmentOriginalTable;
230         IdentifierValue segmentOriginalColumn = segment.getColumnBoundedInfo().getOriginalColumn();
231         IdentifierValue originalColumn = Optional.ofNullable(inputColumnSegment).map(optional -> optional.getColumnBoundedInfo().getOriginalColumn()).orElse(segmentOriginalColumn);
232         return new ColumnSegmentBoundedInfo(originalDatabase, originalSchema, originalTable, originalColumn);
233     }

235     /**
236      * Bind using column segment with metadata.
237      *
238      * @param segment using column segment
239      * @param parentSegmentType parent segment type
240      * @param tableBinderContexts table binder contexts
241      * @return bounded using column segment
242      */
243     public static ColumnSegment bindUsingColumn(final ColumnSegment segment, final SegmentType parentSegmentType, final Map<String, TableSegmentBinderContext> tableBinderContexts) {
244         ColumnSegment result = new ColumnSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getIdentifier());
245         segment.getOwner().ifPresent(result::setOwner);
246         Collection<TableSegmentBinderContext> tableBinderContextValues = tableBinderContexts.values();
247         Collection<ColumnSegment> usingInputColumnSegments = findUsingInputColumnSegments(segment.getIdentifier().getValue(), tableBinderContextValues);
248         ShardingSpherePreconditions.checkState(usingInputColumnSegments.size() >= 2,
249                 () -> new ColumnNotFoundException(segment.getExpression(), SEGMENT_TYPE_MESSAGES.getOrDefault(parentSegmentType, UNKNOWN_SEGMENT_TYPE_MESSAGE)));
250         Iterator<ColumnSegment> iterator = usingInputColumnSegments.iterator();
251         result.setColumnBoundedInfo(createColumnSegmentBoundedInfo(segment, iterator.next()));
252         result.setOtherUsingColumnBoundedInfo(createColumnSegmentBoundedInfo(segment, iterator.next()));
253         return result;
254     }
255     
256     private static Collection<ColumnSegment> findUsingInputColumnSegments(final String columnName, final Collection<TableSegmentBinderContext> tableBinderContexts) {
257         Collection<ColumnSegment> result = new LinkedList<>();
258         for (TableSegmentBinderContext each : tableBinderContexts) {
259             Optional<ProjectionSegment> projectionSegment = each.findProjectionSegmentByColumnLabel(columnName);
260             if (projectionSegment.isPresent() && projectionSegment.get() instanceof ColumnProjectionSegment) {
261                 result.add(((ColumnProjectionSegment) projectionSegment.get()).getColumn());
262             }
263         }
264         return result;
265     }
