49  /**
50   * Database connection manager of ShardingSphere-Proxy.
51   */
52  @RequiredArgsConstructor
53  @Getter
54  public final class ProxyDatabaseConnectionManager implements DatabaseConnectionManager<Connection> {
55      
56      private final ConnectionSession connectionSession;
57      
58      private final Multimap<String, Connection> cachedConnections = LinkedHashMultimap.create();
59      
60      private final Collection<ProxyBackendHandler> backendHandlers = Collections.newSetFromMap(new ConcurrentHashMap<>(64));
61      
62      private final Collection<ProxyBackendHandler> inUseBackendHandlers = Collections.newSetFromMap(new ConcurrentHashMap<>(64));
63      
64      private final Collection<ConnectionPostProcessor> connectionPostProcessors = new LinkedList<>();
65      
66      private final ResourceLock resourceLock = new ResourceLock();
67      
68      private final AtomicBoolean closed = new AtomicBoolean(false);
69      
70      private final Collection<TransactionHook> transactionHooks = ShardingSphereServiceLoader.getServiceInstances(TransactionHook.class);
71      
72      @Override
73      public List<Connection> getConnections(final String databaseName, final String dataSourceName, final int connectionOffset, final int connectionSize,
74                                             final ConnectionMode connectionMode) throws SQLException {
75          Preconditions.checkNotNull(databaseName, "Current database name is null.");
76          Collection<Connection> connections;
77          String cacheKey = getKey(databaseName, dataSourceName);
78          synchronized (cachedConnections) {
79              connections = cachedConnections.get(cacheKey);
80          }
81          List<Connection> result;
82          int maxConnectionSize = connectionOffset + connectionSize;
83          if (connections.size() >= maxConnectionSize) {
84              result = new ArrayList<>(connections).subList(connectionOffset, maxConnectionSize);
85          } else if (connections.isEmpty()) {
86              Collection<Connection> newConnections = createNewConnections(databaseName, dataSourceName, maxConnectionSize, connectionMode);
87              result = new ArrayList<>(newConnections).subList(connectionOffset, maxConnectionSize);
88              synchronized (cachedConnections) {
89                  cachedConnections.putAll(cacheKey, newConnections);
90              }
91              executeTransactionHooksAfterCreateConnections(result);
92          } else {
93              List<Connection> allConnections = new ArrayList<>(maxConnectionSize);
94              allConnections.addAll(connections);
95              List<Connection> newConnections = createNewConnections(databaseName, dataSourceName, maxConnectionSize - connections.size(), connectionMode);
96              allConnections.addAll(newConnections);
97              result = allConnections.subList(connectionOffset, maxConnectionSize);
98              synchronized (cachedConnections) {
99                  cachedConnections.putAll(cacheKey, newConnections);
100             }
101         }
102         return result;
103     }
104     
105     private String getKey(final String databaseName, final String dataSourceName) {
106         return databaseName.toLowerCase() + "." + dataSourceName;
107     }
108     
109     private void executeTransactionHooksAfterCreateConnections(final List<Connection> connections) throws SQLException {
110         if (connectionSession.getTransactionStatus().isInTransaction()) {
111             for (TransactionHook each : transactionHooks) {
112                 each.afterCreateConnections(connections, connectionSession.getConnectionContext().getTransactionContext());
113             }
114         }
115     }
116     
117     private List<Connection> createNewConnections(final String databaseName, final String dataSourceName, final int connectionSize, final ConnectionMode connectionMode) throws SQLException {
118         List<Connection> result = ProxyContext.getInstance().getBackendDataSource().getConnections(databaseName.toLowerCase(), dataSourceName, connectionSize, connectionMode);
119         setSessionVariablesIfNecessary(result);
120         for (Connection each : result) {
121             replayTransactionOption(each);
122         }
123         if (connectionSession.getTransactionStatus().isInTransaction()) {
124             for (Connection each : result) {
125                 replayMethodsInvocation(each);
126             }
127         }
128         return result;
129     }
130     
131     private void setSessionVariablesIfNecessary(final List<Connection> connections) throws SQLException {
132         if (connectionSession.getRequiredSessionVariableRecorder().isEmpty() || connections.isEmpty()) {
133             return;
134         }
135         String databaseType = connections.iterator().next().getMetaData().getDatabaseProductName();
136         List<String> setSQLs = connectionSession.getRequiredSessionVariableRecorder().toSetSQLs(databaseType);
137         try {
138             executeSetSessionVariables(connections, setSQLs);
139         } catch (final SQLException ex) {
140             releaseConnection(connections, ex);
141             throw ex;
142         }
143     }
144     
145     private void executeSetSessionVariables(final List<Connection> connections, final List<String> setSQLs) throws SQLException {
146         for (Connection each : connections) {
147             try (Statement statement = each.createStatement()) {
148                 for (String eachSetSQL : setSQLs) {
149                     statement.execute(eachSetSQL);
150                 }
151             }
152         }
153     }
154     
155     private void releaseConnection(final List<Connection> connections, final SQLException sqlException) {
156         for (Connection each : connections) {
157             try {
158                 each.close();
159             } catch (final SQLException ex) {
160                 sqlException.setNextException(ex);
161             }
162         }
163     }
164     
165     private void replayMethodsInvocation(final Connection target) throws SQLException {
166         for (ConnectionPostProcessor each : connectionPostProcessors) {
167             each.process(target);
168         }
169     }
170     
171     private void replayTransactionOption(final Connection connection) throws SQLException {
172         if (null == connection) {
173             return;
174         }
175         if (connectionSession.isReadOnly()) {
176             connection.setReadOnly(true);
177         }
178         if (connectionSession.getIsolationLevel().isPresent()) {
179             connection.setTransactionIsolation(TransactionUtils.getTransactionIsolationLevel(connectionSession.getIsolationLevel().get()));
180         }
181     }
182     
183     /**
184      * Get used data source names.
185      *
186      * @return used data source names
187      */
188     public Collection<String> getUsedDataSourceNames() {
189         Collection<String> result = new ArrayList<>(cachedConnections.size());
190         String databaseName = connectionSession.getUsedDatabaseName().toLowerCase();
191         for (String each : cachedConnections.keySet()) {
192             String[] split = each.split("\\.", 2);
193             String cachedDatabaseName = split[0];
194             String cachedDataSourceName = split[1];
195             if (databaseName.equals(cachedDatabaseName)) {
196                 result.add(cachedDataSourceName);
197             }
198         }
199         return result;
200     }

235         inUseBackendHandlers.remove(handler);
236     }
237     
238     /**
239      * Handle auto commit.
240      */
241     public void handleAutoCommit() {
242         if (!connectionSession.isAutoCommit() && !connectionSession.getTransactionStatus().isInTransaction()) {
243             BackendTransactionManager transactionManager = new BackendTransactionManager(this);
244             transactionManager.begin();
245         }
246     }
247     
248     /**
249      * Close resources used in execution.
250      *
251      * @throws BackendConnectionException backend connection exception
252      */
253     public void closeExecutionResources() throws BackendConnectionException {
254         synchronized (this) {
255             Collection<Exception> result = new LinkedList<>(closeHandlers(false));
256             if (!connectionSession.getTransactionStatus().isInConnectionHeldTransaction(TransactionUtils.getTransactionType(connectionSession.getConnectionContext().getTransactionContext()))) {
257                 result.addAll(closeHandlers(true));
258                 result.addAll(closeConnections(false));
259             } else if (closed.get()) {
260                 result.addAll(closeHandlers(true));
261                 result.addAll(closeConnections(true));
262             }
263             if (result.isEmpty()) {
264                 return;
265             }
266             throw new BackendConnectionException(result);
267         }
268     }
269     
270     /**
271      * Close all resources.
272      */
273     public void closeAllResources() {
274         synchronized (this) {
275             closed.set(true);
276             closeHandlers(true);
277             closeConnections(true);
278         }
279     }
280     
281     /**
282      * Close handlers.
283      *
284      * @param includeInUse include handlers in use
285      * @return SQL exception when handler close
286      */
287     public Collection<SQLException> closeHandlers(final boolean includeInUse) {
288         Collection<SQLException> result = new LinkedList<>();
289         for (ProxyBackendHandler each : backendHandlers) {
290             if (!includeInUse && inUseBackendHandlers.contains(each)) {
291                 continue;
292             }
293             try {
294                 each.close();
295             } catch (final SQLException ex) {
296                 result.add(ex);
297             }
298         }
299         if (includeInUse) {
300             inUseBackendHandlers.clear();
301         }
302         backendHandlers.retainAll(inUseBackendHandlers);
303         return result;
304     }
305     
306     /**
307      * Close connections.
308      *
309      * @param forceRollback is force rollback
310      * @return SQL exception when connections close
311      */
312     public Collection<SQLException> closeConnections(final boolean forceRollback) {
313         Collection<SQLException> result = new LinkedList<>();
314         synchronized (cachedConnections) {
315             resetSessionVariablesIfNecessary(cachedConnections.values(), result);
316             for (Connection each : cachedConnections.values()) {
317                 try {
318                     if (forceRollback && connectionSession.getTransactionStatus().isInTransaction()) {
319                         each.rollback();
320                     }
321                     each.close();
322                 } catch (final SQLException ex) {
323                     result.add(ex);
324                 }
325             }
326             cachedConnections.clear();
327         }
328         if (!forceRollback) {
329             connectionPostProcessors.clear();
330         }
331         return result;
332     }
333     
334     private void resetSessionVariablesIfNecessary(final Collection<Connection> values, final Collection<SQLException> exceptions) {
335         if (connectionSession.getRequiredSessionVariableRecorder().isEmpty() || values.isEmpty()) {
336             return;
337         }
338         String databaseType;
339         try {
340             databaseType = values.iterator().next().getMetaData().getDatabaseProductName();
341         } catch (final SQLException ex) {
342             exceptions.add(ex);
343             return;
344         }
345         List<String> resetSQLs = connectionSession.getRequiredSessionVariableRecorder().toResetSQLs(databaseType);
346         for (Connection each : values) {
347             try (Statement statement = each.createStatement()) {
348                 for (String eachResetSQL : resetSQLs) {
349                     statement.execute(eachResetSQL);
350                 }
351             } catch (final SQLException ex) {
352                 exceptions.add(ex);
353             }
354         }
355         connectionSession.getRequiredSessionVariableRecorder().removeVariablesWithDefaultValue();
356     }
