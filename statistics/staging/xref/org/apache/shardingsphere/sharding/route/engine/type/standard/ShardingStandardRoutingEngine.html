57  /**
58   * Sharding standard routing engine.
59   */
60  public final class ShardingStandardRoutingEngine implements ShardingRouteEngine {
61      
62      private final String logicTableName;
63      
64      private final ShardingConditions shardingConditions;
65      
66      private final SQLStatementContext sqlStatementContext;
67      
68      private final ConfigurationProperties props;
69      
70      private final Collection<Collection<DataNode>> originalDataNodes = new LinkedList<>();
71      
72      private final HintValueContext hintValueContext;
73      
74      public ShardingStandardRoutingEngine(final String logicTableName, final ShardingConditions shardingConditions, final SQLStatementContext sqlStatementContext,
75                                           final HintValueContext hintValueContext, final ConfigurationProperties props) {
76          this.logicTableName = logicTableName;
77          this.shardingConditions = shardingConditions;
78          this.sqlStatementContext = sqlStatementContext;
79          this.props = props;
80          this.hintValueContext = hintValueContext;
81      }
82      
83      @Override
84      public RouteContext route(final ShardingRule shardingRule) {
85          RouteContext result = new RouteContext();
86          Collection<DataNode> dataNodes = getDataNodes(shardingRule, shardingRule.getShardingTable(logicTableName));
87          result.getOriginalDataNodes().addAll(originalDataNodes);
88          for (DataNode each : dataNodes) {
89              result.getRouteUnits().add(
90                      new RouteUnit(new RouteMapper(each.getDataSourceName(), each.getDataSourceName()), Collections.singleton(new RouteMapper(logicTableName, each.getTableName()))));
91          }
92          return result;
93      }
94      
95      private Collection<DataNode> getDataNodes(final ShardingRule shardingRule, final ShardingTable shardingTable) {
96          ShardingStrategy databaseShardingStrategy = createShardingStrategy(shardingRule.getDatabaseShardingStrategyConfiguration(shardingTable),
97                  shardingRule.getShardingAlgorithms(), shardingRule.getDefaultShardingColumn());
98          ShardingStrategy tableShardingStrategy = createShardingStrategy(shardingRule.getTableShardingStrategyConfiguration(shardingTable),
99                  shardingRule.getShardingAlgorithms(), shardingRule.getDefaultShardingColumn());
100         if (isRoutingByHint(shardingRule, shardingTable)) {
101             return routeByHint(shardingTable, databaseShardingStrategy, tableShardingStrategy);
102         }
103         if (isRoutingByShardingConditions(shardingRule, shardingTable)) {
104             return routeByShardingConditions(shardingRule, shardingTable, databaseShardingStrategy, tableShardingStrategy);
105         }
106         return routeByMixedConditions(shardingRule, shardingTable, databaseShardingStrategy, tableShardingStrategy);
107     }
108     
109     private boolean isRoutingByHint(final ShardingRule shardingRule, final ShardingTable shardingTable) {
110         return shardingRule.getDatabaseShardingStrategyConfiguration(shardingTable) instanceof HintShardingStrategyConfiguration
111                 && shardingRule.getTableShardingStrategyConfiguration(shardingTable) instanceof HintShardingStrategyConfiguration;
112     }
113     
114     private boolean isRoutingBySQLHint() {
115         Collection<String> tableNames = ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames();
116         for (String each : tableNames) {
117             if (hintValueContext.containsHintShardingValue(each)) {
118                 return true;
119             }
120         }
121         return false;
122     }
123     
124     private Collection<DataNode> routeByHint(final ShardingTable shardingTable, final ShardingStrategy databaseShardingStrategy, final ShardingStrategy tableShardingStrategy) {
125         return route0(shardingTable, databaseShardingStrategy, getDatabaseShardingValuesFromHint(), tableShardingStrategy, getTableShardingValuesFromHint());
126     }
127     
128     private boolean isRoutingByShardingConditions(final ShardingRule shardingRule, final ShardingTable shardingTable) {
129         return !(shardingRule.getDatabaseShardingStrategyConfiguration(shardingTable) instanceof HintShardingStrategyConfiguration
130                 || shardingRule.getTableShardingStrategyConfiguration(shardingTable) instanceof HintShardingStrategyConfiguration);
131     }
132     
133     private Collection<DataNode> routeByShardingConditions(final ShardingRule shardingRule, final ShardingTable shardingTable,
134                                                            final ShardingStrategy databaseShardingStrategy, final ShardingStrategy tableShardingStrategy) {
135         return shardingConditions.getConditions().isEmpty()
136                 ? route0(shardingTable, databaseShardingStrategy, Collections.emptyList(), tableShardingStrategy, Collections.emptyList())
137                 : routeByShardingConditionsWithCondition(shardingRule, shardingTable, databaseShardingStrategy, tableShardingStrategy);
138     }
139     
140     private Collection<DataNode> routeByShardingConditionsWithCondition(final ShardingRule shardingRule, final ShardingTable shardingTable,
141                                                                         final ShardingStrategy databaseShardingStrategy, final ShardingStrategy tableShardingStrategy) {
142         Collection<DataNode> result = new LinkedList<>();
143         for (ShardingCondition each : shardingConditions.getConditions()) {
144             Collection<DataNode> dataNodes = route0(shardingTable,
145                     databaseShardingStrategy, getShardingValuesFromShardingConditions(shardingRule, databaseShardingStrategy.getShardingColumns(), each),
146                     tableShardingStrategy, getShardingValuesFromShardingConditions(shardingRule, tableShardingStrategy.getShardingColumns(), each));
147             result.addAll(dataNodes);
148             originalDataNodes.add(dataNodes);
149         }
150         return result;
151     }
152     
153     private Collection<DataNode> routeByMixedConditions(final ShardingRule shardingRule, final ShardingTable shardingTable,
154                                                         final ShardingStrategy databaseShardingStrategy, final ShardingStrategy tableShardingStrategy) {
155         return shardingConditions.getConditions().isEmpty()
156                 ? routeByMixedConditionsWithHint(shardingRule, shardingTable, databaseShardingStrategy, tableShardingStrategy)
157                 : routeByMixedConditionsWithCondition(shardingRule, shardingTable, databaseShardingStrategy, tableShardingStrategy);
158     }
159     
160     private Collection<DataNode> routeByMixedConditionsWithCondition(final ShardingRule shardingRule, final ShardingTable shardingTable,
161                                                                      final ShardingStrategy databaseShardingStrategy, final ShardingStrategy tableShardingStrategy) {
162         Collection<DataNode> result = new LinkedList<>();
163         for (ShardingCondition each : shardingConditions.getConditions()) {
164             Collection<DataNode> dataNodes = route0(shardingTable, databaseShardingStrategy,
165                     getDatabaseShardingValues(shardingRule, databaseShardingStrategy, each), tableShardingStrategy, getTableShardingValues(shardingRule, tableShardingStrategy, each));
166             result.addAll(dataNodes);
167             originalDataNodes.add(dataNodes);
168         }
169         return result;
170     }
171     
172     private Collection<DataNode> routeByMixedConditionsWithHint(final ShardingRule shardingRule, final ShardingTable shardingTable,
173                                                                 final ShardingStrategy databaseShardingStrategy, final ShardingStrategy tableShardingStrategy) {
174         if (shardingRule.getDatabaseShardingStrategyConfiguration(shardingTable) instanceof HintShardingStrategyConfiguration) {
175             return route0(shardingTable, databaseShardingStrategy, getDatabaseShardingValuesFromHint(), tableShardingStrategy, Collections.emptyList());
176         }
177         return route0(shardingTable, databaseShardingStrategy, Collections.emptyList(), tableShardingStrategy, getTableShardingValuesFromHint());
178     }
179     
180     private List<ShardingConditionValue> getDatabaseShardingValues(final ShardingRule shardingRule, final ShardingStrategy databaseShardingStrategy, final ShardingCondition shardingCondition) {
181         return isGettingShardingValuesFromHint(databaseShardingStrategy)
182                 ? getDatabaseShardingValuesFromHint()
183                 : getShardingValuesFromShardingConditions(shardingRule, databaseShardingStrategy.getShardingColumns(), shardingCondition);
184     }
185     
186     private List<ShardingConditionValue> getTableShardingValues(final ShardingRule shardingRule, final ShardingStrategy tableShardingStrategy, final ShardingCondition shardingCondition) {
187         return isGettingShardingValuesFromHint(tableShardingStrategy)
188                 ? getTableShardingValuesFromHint()
189                 : getShardingValuesFromShardingConditions(shardingRule, tableShardingStrategy.getShardingColumns(), shardingCondition);
190     }
191     
192     private boolean isGettingShardingValuesFromHint(final ShardingStrategy shardingStrategy) {
193         return shardingStrategy instanceof HintShardingStrategy;
194     }
195     
196     private List<ShardingConditionValue> getDatabaseShardingValuesFromHint() {
197         if (isRoutingBySQLHint()) {
198             return getDatabaseShardingValuesFromSQLHint();
199         }
200         return getShardingConditions(HintManager.isDatabaseShardingOnly() ? HintManager.getDatabaseShardingValues() : HintManager.getDatabaseShardingValues(logicTableName));
201     }
202     
203     private List<ShardingConditionValue> getDatabaseShardingValuesFromSQLHint() {
204         Collection<Comparable<?>> shardingValues = new LinkedList<>();
205         Collection<String> tableNames = ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames();
206         for (String each : tableNames) {
207             if (each.equals(logicTableName) && hintValueContext.containsHintShardingDatabaseValue(each)) {
208                 shardingValues.addAll(hintValueContext.getHintShardingDatabaseValue(each));
209             }
210         }
211         return getShardingConditions(shardingValues);
212     }
213     
214     private List<ShardingConditionValue> getTableShardingValuesFromHint() {
215         if (isRoutingBySQLHint()) {
216             return getTableShardingValuesFromSQLHint();
217         }
218         return getShardingConditions(HintManager.getTableShardingValues(logicTableName));
219     }
220     
221     private List<ShardingConditionValue> getTableShardingValuesFromSQLHint() {
222         Collection<Comparable<?>> shardingValues = new LinkedList<>();
223         Collection<String> tableNames = ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames();
224         for (String each : tableNames) {
225             if (each.equals(logicTableName) && hintValueContext.containsHintShardingTableValue(each)) {
226                 shardingValues.addAll(hintValueContext.getHintShardingTableValue(each));
227             }
228         }
229         return getShardingConditions(shardingValues);
230     }
231     
232     private List<ShardingConditionValue> getShardingConditions(final Collection<Comparable<?>> shardingValue) {
233         return shardingValue.isEmpty() ? Collections.emptyList() : Collections.singletonList(new ListShardingConditionValue<>("", logicTableName, shardingValue));
234     }
235     
236     private List<ShardingConditionValue> getShardingValuesFromShardingConditions(final ShardingRule shardingRule, final Collection<String> shardingColumns, final ShardingCondition shardingCondition) {
237         List<ShardingConditionValue> result = new ArrayList<>(shardingColumns.size());
238         for (ShardingConditionValue each : shardingCondition.getValues()) {
239             Optional<BindingTableRule> bindingTableRule = shardingRule.findBindingTableRule(each.getTableName());
240             if ((logicTableName.equalsIgnoreCase(each.getTableName()) || bindingTableRule.isPresent() && bindingTableRule.get().hasLogicTable(logicTableName))
241                     && new CaseInsensitiveSet<>(shardingColumns).contains(each.getColumnName())) {
242                 result.add(each);
243             }
244         }
245         return result;
246     }
247     
248     private Collection<DataNode> route0(final ShardingTable shardingTable,
249                                         final ShardingStrategy databaseShardingStrategy, final List<ShardingConditionValue> databaseShardingValues,
250                                         final ShardingStrategy tableShardingStrategy, final List<ShardingConditionValue> tableShardingValues) {
251         Collection<String> routedDataSources = routeDataSources(shardingTable, databaseShardingStrategy, databaseShardingValues);
252         Collection<DataNode> result = new LinkedList<>();
253         for (String each : routedDataSources) {
254             result.addAll(routeTables(shardingTable, each, tableShardingStrategy, tableShardingValues));
255         }
256         return result;
257     }
258     
259     private Collection<String> routeDataSources(final ShardingTable shardingTable, final ShardingStrategy databaseShardingStrategy, final List<ShardingConditionValue> databaseShardingValues) {
260         if (databaseShardingValues.isEmpty()) {
261             return shardingTable.getActualDataSourceNames();
262         }
263         Collection<String> result = databaseShardingStrategy.doSharding(shardingTable.getActualDataSourceNames(), databaseShardingValues, shardingTable.getDataSourceDataNode(), props);
264         ShardingSpherePreconditions.checkNotEmpty(result, NoShardingDatabaseRouteInfoException::new);
265         ShardingSpherePreconditions.checkState(shardingTable.getActualDataSourceNames().containsAll(result),
266                 () -> new MismatchedShardingDataSourceRouteInfoException(result, shardingTable.getActualDataSourceNames()));
267         return result;
268     }
269     
270     private Collection<DataNode> routeTables(final ShardingTable shardingTable, final String routedDataSource,
271                                              final ShardingStrategy tableShardingStrategy, final List<ShardingConditionValue> tableShardingValues) {
272         Collection<String> availableTargetTables = shardingTable.getActualTableNames(routedDataSource);
273         Collection<String> routedTables = tableShardingValues.isEmpty()
274                 ? availableTargetTables
275                 : tableShardingStrategy.doSharding(availableTargetTables, tableShardingValues, shardingTable.getTableDataNode(), props);
276         Collection<DataNode> result = new LinkedList<>();
277         for (String each : routedTables) {
278             result.add(new DataNode(routedDataSource, each));
279         }
280         return result;
281     }
282     
283     private ShardingStrategy createShardingStrategy(final ShardingStrategyConfiguration shardingStrategyConfig, final Map<String, ShardingAlgorithm> shardingAlgorithms,
284                                                     final String defaultShardingColumn) {
285         return null == shardingStrategyConfig ? new NoneShardingStrategy()
286                 : ShardingStrategyFactory.newInstance(shardingStrategyConfig, shardingAlgorithms.get(shardingStrategyConfig.getShardingAlgorithmName()), defaultShardingColumn);
287     }
