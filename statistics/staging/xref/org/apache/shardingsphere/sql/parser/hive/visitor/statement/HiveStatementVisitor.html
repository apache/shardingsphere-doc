106     @Override
107     public final ASTNode visitLiterals(final LiteralsContext ctx) {
108         if (null != ctx.stringLiterals()) {
109             return visit(ctx.stringLiterals());
110         }
111         if (null != ctx.numberLiterals()) {
112             return visit(ctx.numberLiterals());
113         }
114         if (null != ctx.temporalLiterals()) {
115             return visit(ctx.temporalLiterals());
116         }
117         if (null != ctx.hexadecimalLiterals()) {
118             return visit(ctx.hexadecimalLiterals());
119         }
120         if (null != ctx.bitValueLiterals()) {
121             return visit(ctx.bitValueLiterals());
122         }
123         if (null != ctx.booleanLiterals()) {
124             return visit(ctx.booleanLiterals());
125         }
126         if (null != ctx.nullValueLiterals()) {
127             return visit(ctx.nullValueLiterals());
128         }
129         throw new IllegalStateException("Literals must have string, number, dateTime, hex, bit, boolean or null.");
130     }

259     @Override
260     public final ASTNode visitExpr(final ExprContext ctx) {
261         if (null != ctx.booleanPrimary()) {
262             return visit(ctx.booleanPrimary());
263         }
264         if (null != ctx.XOR()) {
265             return createBinaryOperationExpression(ctx, "XOR");
266         }
267         if (null != ctx.andOperator()) {
268             return createBinaryOperationExpression(ctx, ctx.andOperator().getText());
269         }
270         if (null != ctx.orOperator()) {
271             return createBinaryOperationExpression(ctx, ctx.orOperator().getText());
272         }
273         return new NotExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), (ExpressionSegment) visit(ctx.expr(0)), false);
274     }
275     
276     private BinaryOperationExpression createBinaryOperationExpression(final ExprContext ctx, final String operator) {
277         ExpressionSegment left = (ExpressionSegment) visit(ctx.expr(0));
278         ExpressionSegment right = (ExpressionSegment) visit(ctx.expr(1));
279         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
280         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
281     }
282     
283     @Override
284     public final ASTNode visitBooleanPrimary(final BooleanPrimaryContext ctx) {
285         if (null != ctx.IS()) {
286             // TODO optimize operatorToken
287             String rightText = "";
288             if (null != ctx.NOT()) {
289                 rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(ctx.NOT().getSymbol().getStartIndex(),
290                         ctx.NOT().getSymbol().getStopIndex()))).concat(" ");
291             }
292             Token operatorToken = null;
293             if (null != ctx.NULL()) {
294                 operatorToken = ctx.NULL().getSymbol();
295             }
296             if (null != ctx.TRUE()) {
297                 operatorToken = ctx.TRUE().getSymbol();
298             }
299             if (null != ctx.FALSE()) {
300                 operatorToken = ctx.FALSE().getSymbol();
301             }
302             int startIndex = null == operatorToken ? ctx.IS().getSymbol().getStopIndex() + 2 : operatorToken.getStartIndex();
303             rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(startIndex, ctx.stop.getStopIndex())));
304             ExpressionSegment right = new LiteralExpressionSegment(ctx.IS().getSymbol().getStopIndex() + 2, ctx.stop.getStopIndex(), rightText);
305             String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
306             ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
307             String operator = "IS";
308             return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
309         }
310         if (null != ctx.comparisonOperator() || null != ctx.SAFE_EQ_()) {
311             return createCompareSegment(ctx);
312         }
313         if (null != ctx.MEMBER()) {
314             int startIndex = ctx.MEMBER().getSymbol().getStopIndex() + 5;
315             int endIndex = ctx.stop.getStopIndex() - 1;
316             String rightText = ctx.start.getInputStream().getText(new Interval(startIndex, endIndex));
317             ExpressionSegment right = new ExpressionProjectionSegment(startIndex, endIndex, rightText);
318             String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
319             ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
320             String operator = "MEMBER OF";
321             return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
322         }
323         if (null != ctx.assignmentOperator()) {
324             return createAssignmentSegment(ctx);
325         }
326         return visit(ctx.predicate());
327     }
328     
329     private ASTNode createAssignmentSegment(final BooleanPrimaryContext ctx) {
330         ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
331         ExpressionSegment right = (ExpressionSegment) visit(ctx.predicate());
332         String operator = ctx.assignmentOperator().getText();
333         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
334         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
335     }
336     
337     private ASTNode createCompareSegment(final BooleanPrimaryContext ctx) {
338         ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
339         ExpressionSegment right;
340         String operator;
341         if (null != ctx.ALL()) {
342             operator = null == ctx.SAFE_EQ_() ? ctx.comparisonOperator().getText() + " ALL" : ctx.SAFE_EQ_().getText();
343         } else {
344             operator = null == ctx.SAFE_EQ_() ? ctx.comparisonOperator().getText() : ctx.SAFE_EQ_().getText();
345         }
346         if (null != ctx.predicate()) {
347             right = (ExpressionSegment) visit(ctx.predicate());
348         } else {
349             right = new SubqueryExpressionSegment(new SubquerySegment(ctx.subquery().start.getStartIndex(), ctx.subquery().stop.getStopIndex(), (MySQLSelectStatement) visit(ctx.subquery()),
350                     getOriginalText(ctx.subquery())));
351         }
352         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
353         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
354     }
355     
356     @Override
357     public final ASTNode visitPredicate(final PredicateContext ctx) {
358         if (null != ctx.IN()) {
359             return createInSegment(ctx);
360         }
361         if (null != ctx.BETWEEN()) {
362             return createBetweenSegment(ctx);
363         }
364         if (null != ctx.LIKE()) {
365             return createBinaryOperationExpressionFromLike(ctx);
366         }
367         if (null != ctx.REGEXP()) {
368             return createBinaryOperationExpressionFromRegexp(ctx);
369         }
370         if (null != ctx.RLIKE()) {
371             return createBinaryOperationExpressionFromRlike(ctx);
372         }
373         return visit(ctx.bitExpr(0));
374     }
375     
376     private InExpression createInSegment(final PredicateContext ctx) {
377         boolean not = null != ctx.NOT();
378         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
379         ExpressionSegment right;
380         if (null != ctx.subquery()) {
381             right = new SubqueryExpressionSegment(new SubquerySegment(ctx.subquery().start.getStartIndex(), ctx.subquery().stop.getStopIndex(), (MySQLSelectStatement) visit(ctx.subquery()),
382                     getOriginalText(ctx.subquery())));
383         } else {
384             right = new ListExpression(ctx.LP_().getSymbol().getStartIndex(), ctx.RP_().getSymbol().getStopIndex());
385             for (ExprContext each : ctx.expr()) {
386                 ((ListExpression) right).getItems().add((ExpressionSegment) visit(each));
387             }
388         }
389         return new InExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, not);
390     }
391     
392     private BinaryOperationExpression createBinaryOperationExpressionFromLike(final PredicateContext ctx) {
393         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
394         String operator;
395         ExpressionSegment right;
396         if (null != ctx.SOUNDS()) {
397             right = (ExpressionSegment) visit(ctx.bitExpr(1));
398             operator = "SOUNDS LIKE";
399         } else {
400             ListExpression listExpression = new ListExpression(ctx.simpleExpr(0).start.getStartIndex(), ctx.simpleExpr().get(ctx.simpleExpr().size() - 1).stop.getStopIndex());
401             for (SimpleExprContext each : ctx.simpleExpr()) {
402                 listExpression.getItems().add((ExpressionSegment) visit(each));
403             }
404             right = listExpression;
405             operator = null == ctx.NOT() ? "LIKE" : "NOT LIKE";
406         }
407         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
408         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
409     }
410     
411     private BinaryOperationExpression createBinaryOperationExpressionFromRegexp(final PredicateContext ctx) {
412         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
413         ExpressionSegment right = (ExpressionSegment) visit(ctx.bitExpr(1));
414         String operator = null == ctx.NOT() ? "REGEXP" : "NOT REGEXP";
415         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
416         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
417     }
418     
419     private BinaryOperationExpression createBinaryOperationExpressionFromRlike(final PredicateContext ctx) {
420         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
421         ExpressionSegment right = (ExpressionSegment) visit(ctx.bitExpr(1));
422         String operator = null == ctx.NOT() ? "RLIKE" : "NOT RLIKE";
423         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
424         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
425     }
426     
427     private BetweenExpression createBetweenSegment(final PredicateContext ctx) {
428         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
429         ExpressionSegment between = (ExpressionSegment) visit(ctx.bitExpr(1));
430         ExpressionSegment and = (ExpressionSegment) visit(ctx.predicate());
431         boolean not = null != ctx.NOT();
432         return new BetweenExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, between, and, not);
433     }

435     @Override
436     public final ASTNode visitBitExpr(final BitExprContext ctx) {
437         if (null != ctx.simpleExpr()) {
438             return visit(ctx.simpleExpr());
439         }
440         ExpressionSegment left = (ExpressionSegment) visit(ctx.getChild(0));
441         ExpressionSegment right = (ExpressionSegment) visit(ctx.getChild(2));
442         String operator = ctx.getChild(1).getText();
443         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
444         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
445     }

447     @Override
448     public final ASTNode visitSimpleExpr(final SimpleExprContext ctx) {
449         int startIndex = ctx.start.getStartIndex();
450         int stopIndex = ctx.stop.getStopIndex();
451         if (null != ctx.subquery()) {
452             SubquerySegment subquerySegment = new SubquerySegment(ctx.subquery().getStart().getStartIndex(), ctx.subquery().getStop().getStopIndex(),
453                     (MySQLSelectStatement) visit(ctx.subquery()), getOriginalText(ctx.subquery()));
454             return null == ctx.EXISTS() ? new SubqueryExpressionSegment(subquerySegment) : new ExistsSubqueryExpression(startIndex, stopIndex, subquerySegment);
455         }
456         if (null != ctx.parameterMarker()) {
457             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) visit(ctx.parameterMarker());
458             ParameterMarkerExpressionSegment result = new ParameterMarkerExpressionSegment(startIndex, stopIndex, parameterMarker.getValue(), parameterMarker.getType());
459             parameterMarkerSegments.add(result);
460             return result;
461         }
462         if (null != ctx.literals()) {
463             return SQLUtils.createLiteralExpression(visit(ctx.literals()), startIndex, stopIndex, ctx.literals().start.getInputStream().getText(new Interval(startIndex, stopIndex)));
464         }
465         if (null != ctx.intervalExpression()) {
466             return visit(ctx.intervalExpression());
467         }
468         if (null != ctx.functionCall()) {
469             return visit(ctx.functionCall());
470         }
471         if (null != ctx.collateClause()) {
472             if (null != ctx.simpleExpr()) {
473                 ExpressionSegment expr = (ExpressionSegment) visit(ctx.simpleExpr(0));
474                 return new CollateExpression(startIndex, stopIndex, (SimpleExpressionSegment) visit(ctx.collateClause()), expr);
475             }
476             return new CollateExpression(startIndex, stopIndex, (SimpleExpressionSegment) visit(ctx.collateClause()), null);
477         }
478         if (null != ctx.columnRef()) {
479             return visit(ctx.columnRef());
480         }
481         if (null != ctx.matchExpression()) {
482             return visit(ctx.matchExpression());
483         }
484         if (null != ctx.notOperator()) {
485             ASTNode expression = visit(ctx.simpleExpr(0));
486             if (expression instanceof ExistsSubqueryExpression) {
487                 ((ExistsSubqueryExpression) expression).setNot(true);
488                 return expression;
489             }
490             return new NotExpression(startIndex, stopIndex, (ExpressionSegment) expression, "!".equalsIgnoreCase(ctx.notOperator().getText()));
491         }
492         if (null != ctx.LP_() && 1 == ctx.expr().size()) {
493             return visit(ctx.expr(0));
494         }
495         if (null != ctx.OR_()) {
496             ExpressionSegment left = (ExpressionSegment) visit(ctx.simpleExpr(0));
497             ExpressionSegment right = (ExpressionSegment) visit(ctx.simpleExpr(1));
498             String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
499             return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, ctx.OR_().getText(), text);
500         }
501         return visitRemainSimpleExpr(ctx);
502     }
503     
504     private ASTNode visitRemainSimpleExpr(final SimpleExprContext ctx) {
505         if (null != ctx.caseExpression()) {
506             return visit(ctx.caseExpression());
507         }
508         if (null != ctx.BINARY()) {
509             return visit(ctx.simpleExpr(0));
510         }
511         if (null != ctx.variable()) {
512             return visit(ctx.variable());
513         }
514         for (ExprContext each : ctx.expr()) {
515             visit(each);
516         }
517         for (SimpleExprContext each : ctx.simpleExpr()) {
518             visit(each);
519         }
520         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
521         return new CommonExpressionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), text);
522     }

524     @Override
525     public ASTNode visitColumnRef(final ColumnRefContext ctx) {
526         int identifierCount = ctx.identifier().size();
527         ColumnSegment result;
528         if (1 == identifierCount) {
529             result = new ColumnSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), (IdentifierValue) visit(ctx.identifier(0)));
530         } else if (2 == identifierCount) {
531             result = new ColumnSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), (IdentifierValue) visit(ctx.identifier(1)));
532             result.setOwner(new OwnerSegment(ctx.identifier(0).start.getStartIndex(), ctx.identifier(0).stop.getStopIndex(), (IdentifierValue) visit(ctx.identifier(0))));
533         } else {
534             result = new ColumnSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), (IdentifierValue) visit(ctx.identifier(2)));
535             OwnerSegment owner = new OwnerSegment(ctx.identifier(1).start.getStartIndex(), ctx.identifier(1).stop.getStopIndex(), (IdentifierValue) visit(ctx.identifier(1)));
536             owner.setOwner(new OwnerSegment(ctx.identifier(0).start.getStartIndex(), ctx.identifier(0).stop.getStopIndex(), (IdentifierValue) visit(ctx.identifier(0))));
537             result.setOwner(owner);
538         }
539         return result;
540     }
