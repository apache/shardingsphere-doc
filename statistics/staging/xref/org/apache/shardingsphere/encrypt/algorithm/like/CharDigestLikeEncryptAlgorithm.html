18  package org.apache.shardingsphere.encrypt.algorithm.like;
19  
20  import com.google.common.base.Strings;
21  import lombok.SneakyThrows;
22  import org.apache.shardingsphere.encrypt.api.encrypt.like.LikeEncryptAlgorithm;
23  import org.apache.shardingsphere.encrypt.exception.algorithm.EncryptAlgorithmInitializationException;
24  import org.apache.shardingsphere.encrypt.api.context.EncryptContext;
25  
26  import java.io.IOException;
27  import java.io.InputStream;
28  import java.util.Map;
29  import java.util.Objects;
30  import java.util.Properties;
31  import java.util.Scanner;
32  import java.util.stream.Collectors;
33  import java.util.stream.IntStream;
34  
35  /**
36   * Char digest like encrypt algorithm.
37   */
38  public final class CharDigestLikeEncryptAlgorithm implements LikeEncryptAlgorithm<Object, String> {
39      
40      private static final String DELTA_KEY = "delta";
41      
42      private static final String MASK_KEY = "mask";
43      
44      private static final String START_KEY = "start";
45      
46      private static final String DICT_KEY = "dict";
47      
48      private static final int DEFAULT_DELTA = 1;
49      
50      private static final int DEFAULT_MASK = 0b1111_0111_1101;
51      
52      private static final int DEFAULT_START = 0x4e00;
53      
54      private static final int MAX_NUMERIC_LETTER_CHAR = 255;
55      
56      private int delta;
57      
58      private int mask;
59      
60      private int start;
61      
62      private Map<Character, Integer> charIndexes;
63      
64      @Override
65      public void init(final Properties props) {
66          delta = createDelta(props);
67          mask = createMask(props);
68          start = createStart(props);
69          charIndexes = createCharIndexes(props);
70      }
71      
72      private int createDelta(final Properties props) {
73          if (props.containsKey(DELTA_KEY)) {
74              try {
75                  return Integer.parseInt(props.getProperty(DELTA_KEY));
76              } catch (final NumberFormatException ignored) {
77                  throw new EncryptAlgorithmInitializationException(getType(), "delta can only be a decimal number");
78              }
79          }
80          return DEFAULT_DELTA;
81      }
82      
83      private int createMask(final Properties props) {
84          if (props.containsKey(MASK_KEY)) {
85              try {
86                  return Integer.parseInt(props.getProperty(MASK_KEY));
87              } catch (final NumberFormatException ignored) {
88                  throw new EncryptAlgorithmInitializationException(getType(), "mask can only be a decimal number");
89              }
90          }
91          return DEFAULT_MASK;
92      }
93      
94      private int createStart(final Properties props) {
95          if (props.containsKey(START_KEY)) {
96              try {
97                  return Integer.parseInt(props.getProperty(START_KEY));
98              } catch (final NumberFormatException ignored) {
99                  throw new EncryptAlgorithmInitializationException(getType(), "start can only be a decimal number");
100             }
101         }
102         return DEFAULT_START;
103     }
104     
105     private Map<Character, Integer> createCharIndexes(final Properties props) {
106         String dictContent = props.containsKey(DICT_KEY) && !Strings.isNullOrEmpty(props.getProperty(DICT_KEY)) ? props.getProperty(DICT_KEY) : initDefaultDict();
107         return IntStream.range(0, dictContent.length()).boxed().collect(Collectors.toMap(dictContent::charAt, index -> index, (a, b) -> b));
108     }
109     
110     @SneakyThrows(IOException.class)
111     private String initDefaultDict() {
112         StringBuilder result = new StringBuilder();
113         try (
114                 InputStream inputStream = Objects.requireNonNull(Thread.currentThread().getContextClassLoader().getResourceAsStream("algorithm/like/common_chinese_character.dict"));
115                 Scanner scanner = new Scanner(inputStream)) {
116             while (scanner.hasNextLine()) {
117                 String line = scanner.nextLine();
118                 if (!line.isEmpty() && !line.startsWith("#")) {
119                     result.append(line);
120                 }
121             }
122         }
123         return result.toString();
124     }
125     
126     @Override
127     public String encrypt(final Object plainValue, final EncryptContext encryptContext) {
128         return null == plainValue ? null : digest(String.valueOf(plainValue));
129     }
130     
131     private String digest(final String plainValue) {
132         StringBuilder result = new StringBuilder(plainValue.length());
133         for (char each : plainValue.toCharArray()) {
134             char maskedChar = getMaskedChar(each);
135             if ('%' == maskedChar || '_' == maskedChar) {
136                 result.append(each);
137             } else {
138                 result.append(maskedChar);
139             }
140         }
141         return result.toString();
142     }
143     
144     private char getMaskedChar(final char originalChar) {
145         if ('%' == originalChar || '_' == originalChar) {
146             return originalChar;
147         }
148         if (originalChar <= MAX_NUMERIC_LETTER_CHAR) {
149             return (char) ((originalChar + delta) & mask);
150         }
151         if (charIndexes.containsKey(originalChar)) {
152             return (char) (((charIndexes.get(originalChar) + delta) & mask) + start);
153         }
154         return (char) (((originalChar + delta) & mask) + start);
155     }
156     
157     @Override
158     public String getType() {
159         return "CHAR_DIGEST_LIKE";
160     }
161 }
