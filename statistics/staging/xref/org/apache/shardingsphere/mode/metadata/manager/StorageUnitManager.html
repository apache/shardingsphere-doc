18  package org.apache.shardingsphere.mode.metadata.manager;
19  
20  import lombok.SneakyThrows;
21  import lombok.extern.slf4j.Slf4j;
22  import org.apache.shardingsphere.infra.datasource.pool.props.domain.DataSourcePoolProperties;
23  import org.apache.shardingsphere.infra.instance.ComputeNodeInstanceContext;
24  import org.apache.shardingsphere.infra.metadata.database.ShardingSphereDatabase;
25  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereSchema;
26  import org.apache.shardingsphere.infra.rule.ShardingSphereRule;
27  import org.apache.shardingsphere.metadata.persist.MetaDataPersistService;
28  import org.apache.shardingsphere.mode.metadata.MetaDataContexts;
29  import org.apache.shardingsphere.mode.metadata.MetaDataContextsFactory;
30  import org.apache.shardingsphere.mode.spi.PersistRepository;
31  
32  import java.sql.SQLException;
33  import java.util.Collections;
34  import java.util.LinkedHashMap;
35  import java.util.Map;
36  import java.util.concurrent.atomic.AtomicReference;
37  
38  /**
39   * Storage unit manager.
40   */
41  @Slf4j
42  public final class StorageUnitManager {
43      
44      private final AtomicReference<MetaDataContexts> metaDataContexts;
45      
46      private final ComputeNodeInstanceContext computeNodeInstanceContext;
47      
48      private final ResourceSwitchManager resourceSwitchManager;
49      
50      private final MetaDataPersistService metaDataPersistService;
51      
52      public StorageUnitManager(final AtomicReference<MetaDataContexts> metaDataContexts, final ComputeNodeInstanceContext computeNodeInstanceContext,
53                                final PersistRepository repository, final ResourceSwitchManager resourceSwitchManager) {
54          this.metaDataContexts = metaDataContexts;
55          this.computeNodeInstanceContext = computeNodeInstanceContext;
56          this.resourceSwitchManager = resourceSwitchManager;
57          metaDataPersistService = new MetaDataPersistService(repository);
58      }
59      
60      /**
61       * Register storage unit.
62       *
63       * @param databaseName database name
64       * @param propsMap data source pool properties map
65       */
66      public synchronized void registerStorageUnit(final String databaseName, final Map<String, DataSourcePoolProperties> propsMap) {
67          try {
68              closeStaleRules(databaseName);
69              SwitchingResource switchingResource = resourceSwitchManager.switchByRegisterStorageUnit(metaDataContexts.get().getMetaData().getDatabase(databaseName).getResourceMetaData(), propsMap);
70              buildNewMetaDataContext(databaseName, switchingResource);
71          } catch (final SQLException ex) {
72              log.error("Alter database: {} register storage unit failed", databaseName, ex);
73          }
74      }
75      
76      /**
77       * Alter storage unit.
78       *
79       * @param databaseName database name
80       * @param propsMap data source pool properties map
81       */
82      public synchronized void alterStorageUnit(final String databaseName, final Map<String, DataSourcePoolProperties> propsMap) {
83          try {
84              closeStaleRules(databaseName);
85              SwitchingResource switchingResource = resourceSwitchManager.switchByAlterStorageUnit(metaDataContexts.get().getMetaData().getDatabase(databaseName).getResourceMetaData(), propsMap);
86              buildNewMetaDataContext(databaseName, switchingResource);
87          } catch (final SQLException ex) {
88              log.error("Alter database: {} register storage unit failed", databaseName, ex);
89          }
90      }
91      
92      /**
93       * UnRegister storage unit.
94       *
95       * @param databaseName database name
96       * @param storageUnitName storage unit name
97       */
98      public synchronized void unregisterStorageUnit(final String databaseName, final String storageUnitName) {
99          try {
100             closeStaleRules(databaseName);
101             SwitchingResource switchingResource = resourceSwitchManager.switchByUnregisterStorageUnit(metaDataContexts.get().getMetaData().getDatabase(databaseName).getResourceMetaData(),
102                     Collections.singletonList(storageUnitName));
103             buildNewMetaDataContext(databaseName, switchingResource);
104         } catch (final SQLException ex) {
105             log.error("Alter database: {} register storage unit failed", databaseName, ex);
106         }
107     }
108     
109     private void buildNewMetaDataContext(final String databaseName, final SwitchingResource switchingResource) throws SQLException {
110         MetaDataContexts reloadMetaDataContexts = MetaDataContextsFactory.createBySwitchResource(databaseName, true,
111                 switchingResource, metaDataContexts.get(), metaDataPersistService, computeNodeInstanceContext);
112         metaDataContexts.set(reloadMetaDataContexts);
113         metaDataContexts.get().getMetaData().getDatabases().putAll(buildShardingSphereDatabase(reloadMetaDataContexts.getMetaData().getDatabase(databaseName)));
114         switchingResource.closeStaleDataSources();
115     }
116     
117     private Map<String, ShardingSphereDatabase> buildShardingSphereDatabase(final ShardingSphereDatabase originalDatabase) {
118         return Collections.singletonMap(originalDatabase.getName().toLowerCase(), new ShardingSphereDatabase(originalDatabase.getName(),
119                 originalDatabase.getProtocolType(), originalDatabase.getResourceMetaData(), originalDatabase.getRuleMetaData(), buildSchemas(originalDatabase)));
120     }
121     
122     private Map<String, ShardingSphereSchema> buildSchemas(final ShardingSphereDatabase originalDatabase) {
123         Map<String, ShardingSphereSchema> result = new LinkedHashMap<>(originalDatabase.getSchemas().size(), 1F);
124         originalDatabase.getSchemas().keySet().forEach(schemaName -> result.put(schemaName.toLowerCase(), new ShardingSphereSchema(
125                 schemaName, originalDatabase.getSchema(schemaName).getTables(), metaDataPersistService.getDatabaseMetaDataFacade().getView().load(originalDatabase.getName(), schemaName))));
126         return result;
127     }
128     
129     @SneakyThrows(Exception.class)
130     private void closeStaleRules(final String databaseName) {
131         for (ShardingSphereRule each : metaDataContexts.get().getMetaData().getDatabase(databaseName).getRuleMetaData().getRules()) {
132             if (each instanceof AutoCloseable) {
133                 ((AutoCloseable) each).close();
134             }
135         }
136     }
137 }
