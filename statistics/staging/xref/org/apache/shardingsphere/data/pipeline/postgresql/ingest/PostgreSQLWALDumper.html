46  /**
47   * PostgreSQL WAL dumper.
48   */
49  public final class PostgreSQLWALDumper extends AbstractLifecycleExecutor implements IncrementalDumper {
50      
51      private final DumperConfiguration dumperConfig;
52      
53      private final WALPosition walPosition;
54      
55      private final PipelineChannel channel;
56      
57      private final WALEventConverter walEventConverter;
58      
59      private final PostgreSQLLogicalReplication logicalReplication;
60      
61      public PostgreSQLWALDumper(final DumperConfiguration dumperConfig, final IngestPosition<WALPosition> position,
62                                 final PipelineChannel channel, final PipelineTableMetaDataLoader metaDataLoader) {
63          ShardingSpherePreconditions.checkState(StandardPipelineDataSourceConfiguration.class.equals(dumperConfig.getDataSourceConfig().getClass()),
64                  () -> new UnsupportedSQLOperationException("PostgreSQLWALDumper only support PipelineDataSourceConfiguration"));
65          this.dumperConfig = dumperConfig;
66          walPosition = (WALPosition) position;
67          this.channel = channel;
68          walEventConverter = new WALEventConverter(dumperConfig, metaDataLoader);
69          logicalReplication = new PostgreSQLLogicalReplication();
70      }
71      
72      @SneakyThrows(InterruptedException.class)
73      @Override
74      protected void runBlocking() {
75          // TODO use unified PgConnection
76          try (
77                  Connection connection = logicalReplication.createConnection((StandardPipelineDataSourceConfiguration) dumperConfig.getDataSourceConfig());
78                  PGReplicationStream stream = logicalReplication.createReplicationStream(connection, PostgreSQLPositionInitializer.getUniqueSlotName(connection, dumperConfig.getJobId()),
79                          walPosition.getLogSequenceNumber())) {
80              PostgreSQLTimestampUtils utils = new PostgreSQLTimestampUtils(connection.unwrap(PgConnection.class).getTimestampUtils());
81              DecodingPlugin decodingPlugin = new TestDecodingPlugin(utils);
82              while (isRunning()) {
83                  ByteBuffer message = stream.readPending();
84                  if (null == message) {
85                      Thread.sleep(10L);
86                      continue;
87                  }
88                  AbstractWALEvent event = decodingPlugin.decode(message, new PostgreSQLLogSequenceNumber(stream.getLastReceiveLSN()));
89                  channel.pushRecord(walEventConverter.convert(event));
90              }
91          } catch (final SQLException ex) {
92              throw new IngestException(ex);
93          }
94      }
95      
96      @Override
97      protected void doStop() {
98      }
99  }
