81  @ExtendWith(MockitoExtension.class)
82  @MockitoSettings(strictness = Strictness.LENIENT)
83  class StateChangedSubscriberTest {
84      
85      private StateChangedSubscriber subscriber;
86      
87      private ContextManager contextManager;
88      
89      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
90      private ShardingSphereDatabase database;
91      
92      @BeforeEach
93      void setUp() throws SQLException {
94          contextManager = new ClusterContextManagerBuilder().build(createContextManagerBuilderParameter());
95          contextManager.renewMetaDataContexts(new MetaDataContexts(contextManager.getMetaDataContexts().getPersistService(), new ShardingSphereMetaData(createDatabases(),
96                  contextManager.getMetaDataContexts().getMetaData().getGlobalRuleMetaData(), new ConfigurationProperties(new Properties()))));
97          subscriber = new StateChangedSubscriber(new RegistryCenter(mock(ClusterPersistRepository.class), new EventBusContext(), mock(ProxyInstanceMetaData.class), null), contextManager);
98      }
99      
100     private ContextManagerBuilderParameter createContextManagerBuilderParameter() {
101         ModeConfiguration modeConfig = new ModeConfiguration("Cluster", new ClusterPersistRepositoryConfiguration("FIXTURE", "", "", new Properties()));
102         InstanceMetaData instanceMetaData = new ProxyInstanceMetaData("foo_instance_id", 3307);
103         return new ContextManagerBuilderParameter(modeConfig, Collections.emptyMap(), Collections.emptyList(), new Properties(), Collections.emptyList(), instanceMetaData, false);
104     }
105     
106     private Map<String, ShardingSphereDatabase> createDatabases() {
107         when(database.getResourceMetaData().getDataSources()).thenReturn(new LinkedHashMap<>());
108         when(database.getResourceMetaData().getStorageTypes()).thenReturn(Collections.singletonMap("ds_0", new MySQLDatabaseType()));
109         when(database.getSchemas()).thenReturn(Collections.singletonMap("foo_schema", new ShardingSphereSchema()));
110         when(database.getProtocolType()).thenReturn(new MySQLDatabaseType());
111         when(database.getSchema("foo_schema")).thenReturn(mock(ShardingSphereSchema.class));
112         when(database.getRuleMetaData().getRules()).thenReturn(new LinkedList<>());
113         when(database.getRuleMetaData().getConfigurations()).thenReturn(Collections.emptyList());
114         when(database.getRuleMetaData().findRules(ResourceHeldRule.class)).thenReturn(Collections.emptyList());
115         return Collections.singletonMap("db", database);
116     }
117     
118     @Test
119     void assertRenewForDisableStateChanged() {
120         StaticDataSourceContainedRule staticDataSourceRule = mock(StaticDataSourceContainedRule.class);
121         when(database.getRuleMetaData().findSingleRule(StaticDataSourceContainedRule.class)).thenReturn(Optional.of(staticDataSourceRule));
122         StorageNodeChangedEvent event = new StorageNodeChangedEvent(new QualifiedDatabase("db.readwrite_ds.ds_0"), new StorageNodeDataSource(StorageNodeRole.MEMBER, DataSourceState.DISABLED));
123         subscriber.renew(event);
124         verify(staticDataSourceRule).updateStatus(argThat(
125                 (ArgumentMatcher<StorageNodeDataSourceChangedEvent>) argumentEvent -> Objects.equals(event.getQualifiedDatabase(), argumentEvent.getQualifiedDatabase())
126                         && Objects.equals(event.getDataSource(), argumentEvent.getDataSource())));
127     }
128     
129     @Test
130     void assertResetClusterStatus() {
131         ClusterLockDeletedEvent mockLockDeletedEvent = new ClusterLockDeletedEvent(ClusterState.OK);
132         subscriber.renew(mockLockDeletedEvent);
133         assertThat(contextManager.getClusterStateContext().getCurrentState(), is(ClusterState.OK));
134     }
135     
136     @Test
137     void assertRenewClusterStatus() {
138         ClusterStateEvent mockClusterStateEvent = new ClusterStateEvent("READ_ONLY");
139         subscriber.renew(mockClusterStateEvent);
140         assertThat(contextManager.getClusterStateContext().getCurrentState(), is(ClusterState.READ_ONLY));
141     }
142     
143     @Test
144     void assertRenewInstanceStatus() {
145         StateEvent mockStateEvent = new StateEvent(contextManager.getInstanceContext().getInstance().getMetaData().getId(), InstanceState.OK.name());
146         subscriber.renew(mockStateEvent);
147         assertThat(contextManager.getInstanceContext().getInstance().getState().getCurrentState(), is(InstanceState.OK));
148     }
149     
150     @Test
151     void assertRenewInstanceWorkerIdEvent() {
152         subscriber.renew(new WorkerIdEvent(contextManager.getInstanceContext().getInstance().getMetaData().getId(), 0));
153         assertThat(contextManager.getInstanceContext().getInstance().getWorkerId(), is(0));
154     }
155     
156     @Test
157     void assertRenewInstanceLabels() {
158         Collection<String> labels = Collections.singleton("test");
159         subscriber.renew(new LabelsEvent(contextManager.getInstanceContext().getInstance().getMetaData().getId(), labels));
160         assertThat(contextManager.getInstanceContext().getInstance().getLabels(), is(labels));
161     }
162     
163     @Test
164     void assertRenewInstanceOfflineEvent() {
165         subscriber.renew(new InstanceOfflineEvent(contextManager.getInstanceContext().getInstance().getMetaData()));
166         assertThat(((ProxyInstanceMetaData) contextManager.getInstanceContext().getInstance().getMetaData()).getPort(), is(3307));
167     }
168     
169     @Test
170     void assertRenewInstanceOnlineEvent() {
171         InstanceMetaData instanceMetaData1 = new ProxyInstanceMetaData("foo_instance_3307", 3307);
172         InstanceOnlineEvent instanceOnlineEvent1 = new InstanceOnlineEvent(instanceMetaData1);
173         subscriber.renew(instanceOnlineEvent1);
174         assertThat(contextManager.getInstanceContext().getAllClusterInstances().size(), is(1));
175         assertThat(((LinkedList<ComputeNodeInstance>) contextManager.getInstanceContext().getAllClusterInstances()).get(0).getMetaData(), is(instanceMetaData1));
176         InstanceMetaData instanceMetaData2 = new ProxyInstanceMetaData("foo_instance_3308", 3308);
177         InstanceOnlineEvent instanceOnlineEvent2 = new InstanceOnlineEvent(instanceMetaData2);
178         subscriber.renew(instanceOnlineEvent2);
179         assertThat(contextManager.getInstanceContext().getAllClusterInstances().size(), is(2));
180         assertThat(((LinkedList<ComputeNodeInstance>) contextManager.getInstanceContext().getAllClusterInstances()).get(1).getMetaData(), is(instanceMetaData2));
181         subscriber.renew(instanceOnlineEvent1);
182         assertThat(contextManager.getInstanceContext().getAllClusterInstances().size(), is(2));
183         assertThat(((LinkedList<ComputeNodeInstance>) contextManager.getInstanceContext().getAllClusterInstances()).get(1).getMetaData(), is(instanceMetaData1));
184     }
