68   * Abstract inventory incremental job API implementation.
69   */
70  @Slf4j

73      private final PipelineProcessConfigurationPersistService processConfigPersistService = new PipelineProcessConfigurationPersistService();
74      
75      @Getter(AccessLevel.PROTECTED)
76      private final YamlInventoryIncrementalJobItemProgressSwapper jobItemProgressSwapper = new YamlInventoryIncrementalJobItemProgressSwapper();
77      
78      private final YamlJobOffsetInfoSwapper jobOffsetInfoSwapper = new YamlJobOffsetInfoSwapper();
79      
80      @Override
81      public void alterProcessConfiguration(final PipelineContextKey contextKey, final PipelineProcessConfiguration processConfig) {
82          // TODO check rateLimiter type match or not
83          processConfigPersistService.persist(contextKey, getJobType(), processConfig);
84      }
85      
86      @Override
87      public PipelineProcessConfiguration showProcessConfiguration(final PipelineContextKey contextKey) {
88          return PipelineProcessConfigurationUtils.convertWithDefaultValue(processConfigPersistService.load(contextKey, getJobType()));
89      }
90      
91      @Override
92      protected abstract TableBasedPipelineJobInfo getJobInfo(String jobId);
93      
94      @Override
95      public Map<Integer, InventoryIncrementalJobItemProgress> getJobProgress(final PipelineJobConfiguration jobConfig) {
96          String jobId = jobConfig.getJobId();
97          JobConfigurationPOJO jobConfigPOJO = getElasticJobConfigPOJO(jobId);
98          return IntStream.range(0, jobConfig.getJobShardingCount()).boxed().collect(LinkedHashMap::new, (map, each) -> {
99              Optional<InventoryIncrementalJobItemProgress> jobItemProgress = getJobItemProgress(jobId, each);
100             jobItemProgress.ifPresent(optional -> optional.setActive(!jobConfigPOJO.isDisabled()));
101             map.put(each, jobItemProgress.orElse(null));
102         }, LinkedHashMap::putAll);
103     }
104     
105     @Override
106     public List<InventoryIncrementalJobItemInfo> getJobItemInfos(final String jobId) {
107         JobConfigurationPOJO jobConfigPOJO = getElasticJobConfigPOJO(jobId);
108         PipelineJobConfiguration jobConfig = getJobConfiguration(jobConfigPOJO);
109         long startTimeMillis = Long.parseLong(Optional.ofNullable(jobConfigPOJO.getProps().getProperty("start_time_millis")).orElse("0"));
110         Map<Integer, InventoryIncrementalJobItemProgress> jobProgress = getJobProgress(jobConfig);
111         List<InventoryIncrementalJobItemInfo> result = new LinkedList<>();
112         for (Entry<Integer, InventoryIncrementalJobItemProgress> entry : jobProgress.entrySet()) {
113             int shardingItem = entry.getKey();
114             TableBasedPipelineJobInfo jobInfo = getJobInfo(jobId);
115             InventoryIncrementalJobItemProgress jobItemProgress = entry.getValue();
116             String errorMessage = getJobItemErrorMessage(jobId, shardingItem);
117             if (null == jobItemProgress) {
118                 result.add(new InventoryIncrementalJobItemInfo(shardingItem, jobInfo.getTable(), null, startTimeMillis, 0, errorMessage));
119                 continue;
120             }
121             int inventoryFinishedPercentage = 0;
122             if (JobStatus.EXECUTE_INCREMENTAL_TASK == jobItemProgress.getStatus() || JobStatus.FINISHED == jobItemProgress.getStatus()) {
123                 inventoryFinishedPercentage = 100;
124             } else if (0 != jobItemProgress.getProcessedRecordsCount() && 0 != jobItemProgress.getInventoryRecordsCount()) {
125                 inventoryFinishedPercentage = (int) Math.min(100, jobItemProgress.getProcessedRecordsCount() * 100 / jobItemProgress.getInventoryRecordsCount());
126             }
127             result.add(new InventoryIncrementalJobItemInfo(shardingItem, jobInfo.getTable(), jobItemProgress, startTimeMillis, inventoryFinishedPercentage, errorMessage));
128         }
129         return result;
130     }
131     
132     @Override
133     public void persistJobItemProgress(final PipelineJobItemContext jobItemContext) {
134         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobItemContext.getJobId()))
135                 .persistJobItemProgress(jobItemContext.getJobId(), jobItemContext.getShardingItem(), convertJobItemProgress(jobItemContext));
136     }
137     
138     private String convertJobItemProgress(final PipelineJobItemContext jobItemContext) {
139         InventoryIncrementalJobItemContext context = (InventoryIncrementalJobItemContext) jobItemContext;
140         InventoryIncrementalJobItemProgress jobItemProgress = new InventoryIncrementalJobItemProgress();
141         jobItemProgress.setStatus(context.getStatus());
142         jobItemProgress.setSourceDatabaseType(context.getJobConfig().getSourceDatabaseType());
143         jobItemProgress.setDataSourceName(context.getDataSourceName());
144         jobItemProgress.setIncremental(getIncrementalTasksProgress(context.getIncrementalTasks()));
145         jobItemProgress.setInventory(getInventoryTasksProgress(context.getInventoryTasks()));
146         jobItemProgress.setProcessedRecordsCount(context.getProcessedRecordsCount());
147         jobItemProgress.setInventoryRecordsCount(context.getInventoryRecordsCount());
148         return YamlEngine.marshal(jobItemProgressSwapper.swapToYamlConfiguration(jobItemProgress));
149     }
150     
151     @Override
152     public void updateJobItemProgress(final PipelineJobItemContext jobItemContext) {
153         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobItemContext.getJobId()))
154                 .updateJobItemProgress(jobItemContext.getJobId(), jobItemContext.getShardingItem(), convertJobItemProgress(jobItemContext));
155     }
156     
157     private JobItemIncrementalTasksProgress getIncrementalTasksProgress(final Collection<PipelineTask> incrementalTasks) {
158         return new JobItemIncrementalTasksProgress(incrementalTasks.isEmpty() ? null : (IncrementalTaskProgress) incrementalTasks.iterator().next().getTaskProgress());
159     }
160     
161     private JobItemInventoryTasksProgress getInventoryTasksProgress(final Collection<PipelineTask> inventoryTasks) {
162         Map<String, InventoryTaskProgress> inventoryTaskProgressMap = new HashMap<>();
163         for (PipelineTask each : inventoryTasks) {
164             inventoryTaskProgressMap.put(each.getTaskId(), (InventoryTaskProgress) each.getTaskProgress());
165         }
166         return new JobItemInventoryTasksProgress(inventoryTaskProgressMap);
167     }
168     
169     @Override
170     public void persistJobOffsetInfo(final String jobId, final JobOffsetInfo jobOffsetInfo) {
171         String value = YamlEngine.marshal(jobOffsetInfoSwapper.swapToYamlConfiguration(jobOffsetInfo));
172         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).persistJobOffsetInfo(jobId, value);
173     }
174     
175     @Override
176     public JobOffsetInfo getJobOffsetInfo(final String jobId) {
177         Optional<String> offsetInfo = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).getJobOffsetInfo(jobId);
178         if (offsetInfo.isPresent()) {
179             YamlJobOffsetInfo info = YamlEngine.unmarshal(offsetInfo.get(), YamlJobOffsetInfo.class);
180             return jobOffsetInfoSwapper.swapToObject(info);
181         }
182         return jobOffsetInfoSwapper.swapToObject(new YamlJobOffsetInfo());
183     }
184     
185     @Override
186     public Optional<InventoryIncrementalJobItemProgress> getJobItemProgress(final String jobId, final int shardingItem) {
187         Optional<String> progress = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).getJobItemProgress(jobId, shardingItem);
188         return progress.map(optional -> jobItemProgressSwapper.swapToObject(YamlEngine.unmarshal(optional, YamlInventoryIncrementalJobItemProgress.class)));
189     }
190     
191     @Override
192     public void updateJobItemStatus(final String jobId, final int shardingItem, final JobStatus status) {
193         Optional<InventoryIncrementalJobItemProgress> jobItemProgress = getJobItemProgress(jobId, shardingItem);
194         if (!jobItemProgress.isPresent()) {
195             return;
196         }
197         jobItemProgress.get().setStatus(status);
198         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).updateJobItemProgress(jobId, shardingItem,
199                 YamlEngine.marshal(jobItemProgressSwapper.swapToYamlConfiguration(jobItemProgress.get())));
200     }
201     
202     @Override
203     public Collection<DataConsistencyCheckAlgorithmInfo> listDataConsistencyCheckAlgorithms() {
204         Collection<DataConsistencyCheckAlgorithmInfo> result = new LinkedList<>();
205         for (TableDataConsistencyChecker each : ShardingSphereServiceLoader.getServiceInstances(TableDataConsistencyChecker.class)) {
206             SPIDescription description = each.getClass().getAnnotation(SPIDescription.class);
207             String typeAliases = each.getTypeAliases().stream().map(Object::toString).collect(Collectors.joining(","));
208             result.add(new DataConsistencyCheckAlgorithmInfo(each.getType(), typeAliases, getSupportedDatabaseTypes(each.getSupportedDatabaseTypes()), null == description ? "" : description.value()));
209         }
210         return result;
211     }
212     
213     private Collection<DatabaseType> getSupportedDatabaseTypes(final Collection<DatabaseType> supportedDatabaseTypes) {
214         return supportedDatabaseTypes.isEmpty() ? ShardingSphereServiceLoader.getServiceInstances(DatabaseType.class) : supportedDatabaseTypes;
215     }
216     
217     @Override
218     public boolean aggregateDataConsistencyCheckResults(final String jobId, final Map<String, TableDataConsistencyCheckResult> checkResults) {
219         if (checkResults.isEmpty()) {
220             throw new IllegalArgumentException("checkResults empty, jobId:" + jobId);
221         }
222         for (Entry<String, TableDataConsistencyCheckResult> entry : checkResults.entrySet()) {
223             TableDataConsistencyCheckResult checkResult = entry.getValue();
224             if (!checkResult.isMatched()) {
225                 return false;
226             }
227         }
228         return true;
229     }
