18  package org.apache.shardingsphere.db.protocol.mysql.packet.command.query.binary.execute.protocol;
19  
20  import org.apache.shardingsphere.db.protocol.mysql.payload.MySQLPacketPayload;
21  
22  import java.sql.SQLException;
23  import java.sql.SQLFeatureNotSupportedException;
24  import java.sql.Timestamp;
25  import java.time.LocalDate;
26  import java.time.LocalDateTime;
27  import java.util.Date;
28  
29  /**
30   * Binary protocol value for date for MySQL.
31   */
32  public final class MySQLDateBinaryProtocolValue implements MySQLBinaryProtocolValue {
33      
34      @Override
35      public Object read(final MySQLPacketPayload payload, final boolean unsigned) throws SQLException {
36          int length = payload.readInt1();
37          switch (length) {
38              case 0:
39                  throw new SQLFeatureNotSupportedException("Can not support date format if year, month, day is absent.");
40              case 4:
41                  return getTimestampForDate(payload);
42              case 7:
43                  return getTimestampForDatetime(payload);
44              case 11:
45                  Timestamp result = getTimestampForDatetime(payload);
46                  result.setNanos(payload.readInt4() * 1000);
47                  return result;
48              default:
49                  throw new SQLFeatureNotSupportedException(String.format("Wrong length `%d` of MYSQL_TYPE_TIME", length));
50          }
51      }
52      
53      private Timestamp getTimestampForDate(final MySQLPacketPayload payload) {
54          return Timestamp.valueOf(LocalDate.of(payload.readInt2(), payload.readInt1(), payload.readInt1()).atStartOfDay());
55      }
56      
57      private Timestamp getTimestampForDatetime(final MySQLPacketPayload payload) {
58          return Timestamp.valueOf(LocalDateTime.of(payload.readInt2(), payload.readInt1(), payload.readInt1(), payload.readInt1(), payload.readInt1(), payload.readInt1()));
59      }
60      
61      @Override
62      public void write(final MySQLPacketPayload payload, final Object value) {
63          LocalDateTime dateTime = value instanceof LocalDateTime ? (LocalDateTime) value : new Timestamp(((Date) value).getTime()).toLocalDateTime();
64          int year = dateTime.getYear();
65          int month = dateTime.getMonthValue();
66          int dayOfMonth = dateTime.getDayOfMonth();
67          int hours = dateTime.getHour();
68          int minutes = dateTime.getMinute();
69          int seconds = dateTime.getSecond();
70          int nanos = dateTime.getNano();
71          boolean isTimeAbsent = 0 == hours && 0 == minutes && 0 == seconds;
72          boolean isNanosAbsent = 0 == nanos;
73          if (isTimeAbsent && isNanosAbsent) {
74              payload.writeInt1(4);
75              writeDate(payload, year, month, dayOfMonth);
76              return;
77          }
78          if (isNanosAbsent) {
79              payload.writeInt1(7);
80              writeDate(payload, year, month, dayOfMonth);
81              writeTime(payload, hours, minutes, seconds);
82              return;
83          }
84          payload.writeInt1(11);
85          writeDate(payload, year, month, dayOfMonth);
86          writeTime(payload, hours, minutes, seconds);
87          writeNanos(payload, nanos);
88      }
89      
90      private void writeDate(final MySQLPacketPayload payload, final int year, final int month, final int dayOfMonth) {
91          payload.writeInt2(year);
92          payload.writeInt1(month);
93          payload.writeInt1(dayOfMonth);
94      }
95      
96      private void writeTime(final MySQLPacketPayload payload, final int hourOfDay, final int minutes, final int seconds) {
97          payload.writeInt1(hourOfDay);
98          payload.writeInt1(minutes);
99          payload.writeInt1(seconds);
100     }
101     
102     private void writeNanos(final MySQLPacketPayload payload, final int nanos) {
103         payload.writeInt4(nanos);
104     }
105 }
