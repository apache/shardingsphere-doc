54  /**
55   * Generic schema builder.
56   */
57  @NoArgsConstructor(access = AccessLevel.PRIVATE)
58  public final class GenericSchemaBuilder {
59      
60      /**
61       * Build generic schema.
62       *
63       * @param material generic schema builder material
64       * @return generic schema map
65       * @throws SQLException SQL exception
66       */
67      public static Map<String, ShardingSphereSchema> build(final GenericSchemaBuilderMaterial material) throws SQLException {
68          return build(getAllTableNames(material.getRules()), material);
69      }
70      
71      /**
72       * Build generic schema.
73       *
74       * @param tableNames table names
75       * @param material generic schema builder material
76       * @return generic schema map
77       * @throws SQLException SQL exception
78       */
79      public static Map<String, ShardingSphereSchema> build(final Collection<String> tableNames, final GenericSchemaBuilderMaterial material) throws SQLException {
80          Map<String, SchemaMetaData> result = loadSchemas(tableNames, material);
81          if (!material.isSameProtocolAndStorageTypes()) {
82              result = translate(result, material);
83          }
84          return revise(result, material);
85      }
86      
87      private static Collection<String> getAllTableNames(final Collection<ShardingSphereRule> rules) {
88          Collection<String> result = new HashSet<>();
89          for (ShardingSphereRule each : rules) {
90              each.getAttributes().findAttribute(TableMapperRuleAttribute.class).ifPresent(mapperRule -> result.addAll(mapperRule.getLogicTableNames()));
91          }
92          return result;
93      }
94      
95      private static Map<String, SchemaMetaData> loadSchemas(final Collection<String> tableNames, final GenericSchemaBuilderMaterial material) throws SQLException {
96          boolean checkMetaDataEnable = material.getProps().getValue(ConfigurationPropertyKey.CHECK_TABLE_METADATA_ENABLED);
97          Collection<MetaDataLoaderMaterial> materials = SchemaMetaDataUtils.getMetaDataLoaderMaterials(tableNames, material, checkMetaDataEnable);
98          return materials.isEmpty() ? Collections.emptyMap() : MetaDataLoader.load(materials);
99      }
100     
101     private static Map<String, SchemaMetaData> translate(final Map<String, SchemaMetaData> schemaMetaDataMap, final GenericSchemaBuilderMaterial material) {
102         Collection<TableMetaData> tableMetaDataList = new LinkedList<>();
103         for (DatabaseType each : material.getStorageTypes().values()) {
104             String defaultSchemaName = new DatabaseTypeRegistry(each).getDefaultSchemaName(material.getDefaultSchemaName());
105             tableMetaDataList.addAll(Optional.ofNullable(schemaMetaDataMap.get(defaultSchemaName)).map(SchemaMetaData::getTables).orElseGet(Collections::emptyList));
106         }
107         String frontendSchemaName = new DatabaseTypeRegistry(material.getProtocolType()).getDefaultSchemaName(material.getDefaultSchemaName());
108         Map<String, SchemaMetaData> result = new LinkedHashMap<>();
109         result.put(frontendSchemaName, new SchemaMetaData(frontendSchemaName, tableMetaDataList));
110         return result;
111     }
112     
113     private static Map<String, ShardingSphereSchema> revise(final Map<String, SchemaMetaData> schemaMetaDataMap, final GenericSchemaBuilderMaterial material) {
114         Map<String, SchemaMetaData> result = new LinkedHashMap<>(schemaMetaDataMap);
115         result.putAll(new MetaDataReviseEngine(material.getRules().stream()
116                 .filter(each -> each.getAttributes().findAttribute(TableMapperRuleAttribute.class).isPresent()).collect(Collectors.toList())).revise(result, material));
117         return convertToSchemaMap(result, material);
118     }
119     
120     private static Map<String, ShardingSphereSchema> convertToSchemaMap(final Map<String, SchemaMetaData> schemaMetaDataMap, final GenericSchemaBuilderMaterial material) {
121         if (schemaMetaDataMap.isEmpty()) {
122             return Collections.singletonMap(material.getDefaultSchemaName(), new ShardingSphereSchema(material.getDefaultSchemaName()));
123         }
124         Map<String, ShardingSphereSchema> result = new ConcurrentHashMap<>(schemaMetaDataMap.size(), 1F);
125         for (Entry<String, SchemaMetaData> entry : schemaMetaDataMap.entrySet()) {
126             Map<String, ShardingSphereTable> tables = convertToTableMap(entry.getValue().getTables());
127             result.put(entry.getKey().toLowerCase(), new ShardingSphereSchema(entry.getKey(), tables, new LinkedHashMap<>()));
128         }
129         return result;
130     }
131     
132     private static Map<String, ShardingSphereTable> convertToTableMap(final Collection<TableMetaData> tableMetaDataList) {
133         Map<String, ShardingSphereTable> result = new LinkedHashMap<>(tableMetaDataList.size(), 1F);
134         for (TableMetaData each : tableMetaDataList) {
135             Collection<ShardingSphereColumn> columns = convertToColumns(each.getColumns());
136             Collection<ShardingSphereIndex> indexes = convertToIndexes(each.getIndexes());
137             Collection<ShardingSphereConstraint> constraints = convertToConstraints(each.getConstraints());
138             result.put(each.getName(), new ShardingSphereTable(each.getName(), columns, indexes, constraints, each.getType()));
139         }
140         return result;
141     }
142     
143     private static Collection<ShardingSphereColumn> convertToColumns(final Collection<ColumnMetaData> columnMetaDataList) {
144         Collection<ShardingSphereColumn> result = new LinkedList<>();
145         for (ColumnMetaData each : columnMetaDataList) {
146             result.add(new ShardingSphereColumn(each.getName(), each.getDataType(), each.isPrimaryKey(), each.isGenerated(), each.isCaseSensitive(), each.isVisible(), each.isUnsigned(),
147                     each.isNullable()));
148         }
149         return result;
150     }
151     
152     private static Collection<ShardingSphereIndex> convertToIndexes(final Collection<IndexMetaData> indexMetaDataList) {
153         Collection<ShardingSphereIndex> result = new LinkedList<>();
154         for (IndexMetaData each : indexMetaDataList) {
155             ShardingSphereIndex index = new ShardingSphereIndex(each.getName());
156             index.getColumns().addAll(each.getColumns());
157             index.setUnique(each.isUnique());
158             result.add(index);
159         }
160         return result;
161     }
162     
163     private static Collection<ShardingSphereConstraint> convertToConstraints(final Collection<ConstraintMetaData> constraintMetaDataList) {
164         Collection<ShardingSphereConstraint> result = new LinkedList<>();
165         for (ConstraintMetaData each : constraintMetaDataList) {
166             result.add(new ShardingSphereConstraint(each.getName(), each.getReferencedTableName()));
167         }
168         return result;
169     }
