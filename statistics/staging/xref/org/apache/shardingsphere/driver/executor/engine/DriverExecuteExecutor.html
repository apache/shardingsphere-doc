50  /**
51   * Driver execute executor.
52   */
53  @RequiredArgsConstructor
54  public final class DriverExecuteExecutor {
55      
56      private final ShardingSphereConnection connection;
57      
58      private final ShardingSphereMetaData metaData;
59      
60      private final DriverPushDownExecuteExecutor pushDownExecuteExecutor;
61      
62      private final TrafficExecutor trafficExecutor;
63      
64      private final SQLFederationEngine sqlFederationEngine;
65      
66      private ExecuteType executeType = ExecuteType.PUSH_DOWN;
67      
68      public DriverExecuteExecutor(final ShardingSphereConnection connection, final ShardingSphereMetaData metaData, final Grantee grantee,
69                                   final JDBCExecutor jdbcExecutor, final RawExecutor rawExecutor,
70                                   final TrafficExecutor trafficExecutor, final SQLFederationEngine sqlFederationEngine) {
71          this.connection = connection;
72          this.metaData = metaData;
73          pushDownExecuteExecutor = new DriverPushDownExecuteExecutor(connection, metaData, grantee, jdbcExecutor, rawExecutor);
74          this.trafficExecutor = trafficExecutor;
75          this.sqlFederationEngine = sqlFederationEngine;
76      }
77      
78      /**
79       * Execute.
80       *
81       * @param database database
82       * @param queryContext query context
83       * @param prepareEngine prepare engine
84       * @param executeCallback statement execute callback
85       * @param addCallback statement add callback
86       * @param replayCallback statement replay callback
87       * @return execute result
88       * @throws SQLException SQL exception
89       */
90      @SuppressWarnings("rawtypes")
91      public boolean execute(final ShardingSphereDatabase database, final QueryContext queryContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
92                             final StatementExecuteCallback executeCallback, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
93          Optional<String> trafficInstanceId = connection.getTrafficInstanceId(metaData.getGlobalRuleMetaData().getSingleRule(TrafficRule.class), queryContext);
94          if (trafficInstanceId.isPresent()) {
95              executeType = ExecuteType.TRAFFIC;
96              return trafficExecutor.execute(connection.getProcessId(), database.getName(), trafficInstanceId.get(), queryContext, prepareEngine, executeCallback::execute);
97          }
98          if (sqlFederationEngine.decide(queryContext.getSqlStatementContext(), queryContext.getParameters(), database, metaData.getGlobalRuleMetaData())) {
99              executeType = ExecuteType.FEDERATION;
100             ResultSet resultSet = sqlFederationEngine.executeQuery(prepareEngine,
101                     new ExecuteQueryCallbackFactory(prepareEngine.getType()).newInstance(database, queryContext), new SQLFederationContext(false, queryContext, metaData, connection.getProcessId()));
102             return null != resultSet;
103         }
104         ExecutionContext executionContext = new KernelProcessor().generateExecutionContext(
105                 queryContext, database, metaData.getGlobalRuleMetaData(), metaData.getProps(), connection.getDatabaseConnectionManager().getConnectionContext());
106         return pushDownExecuteExecutor.execute(database, executionContext, prepareEngine, executeCallback, addCallback, replayCallback);
107     }
108     
109     /**
110      * Get result set.
111      *
112      * @param database database
113      * @param sqlStatementContext SQL statement context
114      * @param statement statement
115      * @param statements statements
116      * @return result set
117      * @throws SQLException SQL exception
118      */
119     public Optional<ResultSet> getResultSet(final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext,
120                                             final Statement statement, final List<? extends Statement> statements) throws SQLException {
121         switch (executeType) {
122             case TRAFFIC:
123                 return Optional.of(trafficExecutor.getResultSet());
124             case FEDERATION:
125                 return Optional.of(sqlFederationEngine.getResultSet());
126             case PUSH_DOWN:
127                 return pushDownExecuteExecutor.getResultSet(database, sqlStatementContext, statement, statements);
128             default:
129                 return Optional.empty();
130         }
131     }
132     
133     public enum ExecuteType {
134         
135         TRAFFIC,
136         
137         FEDERATION,
138         
139         PUSH_DOWN
140     }
141 }
