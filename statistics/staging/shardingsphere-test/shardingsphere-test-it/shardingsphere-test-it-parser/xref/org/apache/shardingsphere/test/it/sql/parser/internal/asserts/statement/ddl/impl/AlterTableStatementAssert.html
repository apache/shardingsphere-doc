67  /**
68   * Alter table statement assert.
69   */
70  @NoArgsConstructor(access = AccessLevel.PRIVATE)
71  public final class AlterTableStatementAssert {
72      
73      /**
74       * Assert alter table statement is correct with expected parser result.
75       *
76       * @param assertContext assert context
77       * @param actual actual alter table statement
78       * @param expected expected alter table statement test case
79       */
80      public static void assertIs(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
81          assertTable(assertContext, actual, expected);
82          assertRenameTable(assertContext, actual, expected);
83          assertAddColumnDefinitions(assertContext, actual, expected);
84          assertAddConstraintDefinitions(assertContext, actual, expected);
85          assertModifyConstraintDefinitions(assertContext, actual, expected);
86          assertModifyColumnDefinitions(assertContext, actual, expected);
87          assertChangeColumnDefinitions(assertContext, actual, expected);
88          assertDropColumns(assertContext, actual, expected);
89          assertRenameIndexDefinitions(assertContext, actual, expected);
90          assertRenameColumnDefinitions(assertContext, actual, expected);
91          assertConvertTable(assertContext, actual, expected);
92          assertModifyCollectionRetrievalDefinitions(assertContext, actual, expected);
93      }
94      
95      private static void assertConvertTable(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
96          Optional<ConvertTableDefinitionSegment> convertTable = actual.getConvertTableDefinition();
97          if (null == expected.getConvertTable()) {
98              assertFalse(convertTable.isPresent(), assertContext.getText("Actual convert table segment should not exist."));
99          } else {
100             assertTrue(convertTable.isPresent(), assertContext.getText("Actual convert table segment should exist."));
101             CharsetAssert.assertIs(assertContext, convertTable.get().getCharsetName(), expected.getConvertTable().getCharsetName());
102             if (null == expected.getConvertTable().getCollateExpression()) {
103                 assertNull(convertTable.get().getCollateValue(), assertContext.getText("Actual collate expression should not exist."));
104             } else {
105                 ExpressionAssert.assertExpression(assertContext, convertTable.get().getCollateValue(), expected.getConvertTable().getCollateExpression().getCollateName());
106             }
107             SQLSegmentAssert.assertIs(assertContext, convertTable.get(), expected.getConvertTable());
108         }
109     }
110     
111     private static void assertRenameTable(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
112         Optional<SimpleTableSegment> tableSegment = actual.getRenameTable();
113         if (null == expected.getRenameTable()) {
114             assertFalse(tableSegment.isPresent(), assertContext.getText("Actual table segment should not exist."));
115         } else {
116             assertTrue(tableSegment.isPresent(), assertContext.getText("Actual table segment should exist."));
117             TableAssert.assertIs(assertContext, tableSegment.get(), expected.getRenameTable());
118         }
119     }
120     
121     private static void assertTable(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
122         TableAssert.assertIs(assertContext, actual.getTable(), expected.getTable());
123     }
124     
125     private static void assertAddColumnDefinitions(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
126         assertThat(assertContext.getText("Add column definitions size assertion error: "), actual.getAddColumnDefinitions().size(), is(expected.getAddColumns().size()));
127         int count = 0;
128         for (AddColumnDefinitionSegment each : actual.getAddColumnDefinitions()) {
129             ExpectedAddColumnDefinition expectedAddColumnDefinition = expected.getAddColumns().get(count);
130             assertColumnDefinitions(assertContext, each.getColumnDefinitions(), expectedAddColumnDefinition.getColumnDefinitions());
131             if (each.getColumnPosition().isPresent()) {
132                 assertNotNull(expectedAddColumnDefinition.getColumnPosition(), assertContext.getText("Column position should exist."));
133                 ColumnPositionAssert.assertIs(assertContext, each.getColumnPosition().get(), expectedAddColumnDefinition.getColumnPosition());
134             } else {
135                 assertNull(expectedAddColumnDefinition.getColumnPosition(), assertContext.getText("Column position should not exist."));
136             }
137             count++;
138         }
139     }

141     private static void assertColumnDefinitions(final SQLCaseAssertContext assertContext, final Collection<ColumnDefinitionSegment> actual, final List<ExpectedColumnDefinition> expected) {
142         int count = 0;
143         for (ColumnDefinitionSegment each : actual) {
144             ColumnDefinitionAssert.assertIs(assertContext, each, expected.get(count));
145             count++;
146         }
147     }

150         int count = 0;
151         for (AddConstraintDefinitionSegment each : actual.getAddConstraintDefinitions()) {
152             ConstraintDefinitionAssert.assertIs(assertContext, each.getConstraintDefinition(), expected.getAddConstraints().get(count));
153             count++;
154         }
155     }

157     private static void assertModifyConstraintDefinitions(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
158         assertThat(assertContext.getText("Modify constraint definitions size assertion error: "), actual.getModifyConstraintDefinitions().size(), is(expected.getModifyConstraints().size()));
159         int count = 0;
160         for (ModifyConstraintDefinitionSegment each : actual.getModifyConstraintDefinitions()) {
161             ExpectedConstraintDefinition expectedConstraintDefinition = expected.getModifyConstraints().get(count);
162             if (null == expectedConstraintDefinition.getConstraintName()) {
163                 assertNull(each.getConstraintName(), "Actual modify constraint name should not exist.");
164             } else {
165                 assertNotNull(each.getConstraintName(), "Actual modify constraint name should exist.");
166                 assertThat(assertContext.getText("Actual modify constraint name assertion error."),
167                         each.getConstraintName().getIdentifier().getValue(), is(expectedConstraintDefinition.getConstraintName()));
168             }
169             count++;
170         }
171     }

173     private static void assertModifyColumnDefinitions(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
174         assertThat(assertContext.getText("Modify column definitions size assertion error: "), actual.getModifyColumnDefinitions().size(), is(expected.getModifyColumns().size()));
175         int count = 0;
176         for (ModifyColumnDefinitionSegment each : actual.getModifyColumnDefinitions()) {
177             ExpectedModifyColumnDefinition expectedModifyColumnDefinition = expected.getModifyColumns().get(count);
178             ColumnDefinitionAssert.assertIs(assertContext, each.getColumnDefinition(), expectedModifyColumnDefinition.getColumnDefinition());
179             if (each.getColumnPosition().isPresent()) {
180                 assertNotNull(expectedModifyColumnDefinition.getColumnPosition(), assertContext.getText("Column position should exist."));
181                 ColumnPositionAssert.assertIs(assertContext, each.getColumnPosition().get(), expectedModifyColumnDefinition.getColumnPosition());
182             } else {
183                 assertNull(expectedModifyColumnDefinition.getColumnPosition(), assertContext.getText("Column position should not exist."));
184             }
185             count++;
186         }
187     }

189     private static void assertChangeColumnDefinitions(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
190         assertThat(assertContext.getText("Change column definitions size assertion error: "), actual.getChangeColumnDefinitions().size(), is(expected.getChangeColumns().size()));
191         int count = 0;
192         for (ChangeColumnDefinitionSegment each : actual.getChangeColumnDefinitions()) {
193             ExpectedChangeColumnDefinition expectedChangeColumnDefinition = expected.getChangeColumns().get(count);
194             ColumnDefinitionAssert.assertIs(assertContext, each.getColumnDefinition(), expectedChangeColumnDefinition.getColumnDefinition());
195             if (each.getColumnPosition().isPresent()) {
196                 assertNotNull(expectedChangeColumnDefinition.getColumnPosition(), assertContext.getText("Column position should exist."));
197                 ColumnPositionAssert.assertIs(assertContext, each.getColumnPosition().get(), expectedChangeColumnDefinition.getColumnPosition());
198             } else {
199                 assertNull(expectedChangeColumnDefinition.getColumnPosition(), assertContext.getText("Column position should not exist."));
200             }
201             if (null != each.getPreviousColumn()) {
202                 ColumnAssert.assertIs(assertContext, each.getPreviousColumn(), expectedChangeColumnDefinition.getPreviousColumn());
203             }
204             count++;
205         }
206     }

208     private static void assertDropColumns(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
209         Collection<ColumnSegment> actualDropColumns = getDropColumns(actual);
210         assertThat(assertContext.getText("Drop columns size assertion error: "), actualDropColumns.size(), is(expected.getDropColumns().size()));
211         int count = 0;
212         for (ColumnSegment each : actualDropColumns) {
213             ColumnAssert.assertIs(assertContext, each, expected.getDropColumns().get(count));
214             count++;
215         }
216     }
217     
218     private static Collection<ColumnSegment> getDropColumns(final AlterTableStatement actual) {
219         Collection<ColumnSegment> result = new LinkedList<>();
220         for (DropColumnDefinitionSegment each : actual.getDropColumnDefinitions()) {
221             result.addAll(each.getColumns());
222         }
223         return result;
224     }

237     private static void assertRenameColumnDefinitions(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
238         assertThat(assertContext.getText("Rename columns definitions size assertion error:"), actual.getRenameColumnDefinitions().size(), is(expected.getRenameColumns().size()));
239         int count = 0;
240         for (RenameColumnSegment each : actual.getRenameColumnDefinitions()) {
241             ExpectedRenameColumnDefinition expectedRenameColumnDefinition = expected.getRenameColumns().get(count);
242             ColumnAssert.assertIs(assertContext, each.getOldColumnName(), expectedRenameColumnDefinition.getOldColumnName());
243             ColumnAssert.assertIs(assertContext, each.getColumnName(), expectedRenameColumnDefinition.getColumnName());
244             count++;
245         }
246     }
247     
248     private static void assertModifyCollectionRetrievalDefinitions(final SQLCaseAssertContext assertContext, final AlterTableStatement actual, final AlterTableStatementTestCase expected) {
249         Optional<ModifyCollectionRetrievalSegment> modifyCollectionRetrieval = actual.getModifyCollectionRetrieval();
250         if (null == expected.getModifyCollectionRetrievalDefinition()) {
251             assertFalse(modifyCollectionRetrieval.isPresent(), assertContext.getText("Actual modify collection retrieval definitions should not exist."));
252         } else {
253             assertTrue(modifyCollectionRetrieval.isPresent(), assertContext.getText("Actual modify collection retrieval definitions should exist."));
254             if (null == expected.getModifyCollectionRetrievalDefinition().getTable()) {
255                 assertNull(modifyCollectionRetrieval.get().getNestedTable(), "Actual nested table should not exist.");
256             } else {
257                 assertNotNull(modifyCollectionRetrieval.get().getNestedTable(), "Actual nested table should exist.");
258                 TableAssert.assertIs(assertContext, modifyCollectionRetrieval.get().getNestedTable(), expected.getModifyCollectionRetrievalDefinition().getTable());
259             }
260         }
