18  package org.apache.shardingsphere.db.protocol.mysql.packet.command.query;
19  
20  import lombok.Getter;
21  import lombok.RequiredArgsConstructor;
22  
23  /**
24   * MySQL Column Field Detail Flag.
25   * 
26   * @see <a href="https://dev.mysql.com/doc/dev/mysql-server/latest/group__group__cs__column__definition__flags.html">Column Definition Flags</a>
27   * @see <a href="https://mariadb.com/kb/en/library/resultset/#field-detail-flag">Field detail flag</a>
28   */
29  @RequiredArgsConstructor
30  @Getter
31  public enum MySQLColumnDefinitionFlag {
32      
33      NOT_NULL(0x00000001),
34      
35      PRIMARY_KEY(0x00000002),
36      
37      UNIQUE_KEY(0x00000004),
38      
39      MULTIPLE_KEY(0x00000008),
40      
41      BLOB(0x00000010),
42      
43      UNSIGNED(0x00000020),
44      
45      ZEROFILL_FLAG(0x00000040),
46      
47      BINARY_COLLATION(0x00000080),
48      
49      ENUM(0x00000100),
50      
51      AUTO_INCREMENT(0x00000200),
52      
53      TIMESTAMP(0x00000400),
54      
55      SET(0x00000800),
56      
57      NO_DEFAULT_VALUE_FLAG(0x00001000),
58      
59      ON_UPDATE_NOW_FLAG(0x00002000),
60      
61      NUM_FLAG(0x00008000);
62      
63      private final int value;
64  }
