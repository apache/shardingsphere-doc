62  /**
63   * TCL statement visitor for PostgreSQL.
64   */
65  public final class PostgreSQLTCLStatementVisitor extends PostgreSQLStatementVisitor implements TCLStatementVisitor {
66      
67      @Override
68      public ASTNode visitSetTransaction(final SetTransactionContext ctx) {
69          PostgreSQLSetTransactionStatement result = new PostgreSQLSetTransactionStatement();
70          if (null != ctx.transactionModeList()) {
71              ctx.transactionModeList().transactionModeItem().forEach(each -> {
72                  result.setAccessMode(getTransactionAccessType(each));
73                  result.setIsolationLevel(getTransactionIsolationLevel(each));
74              });
75          }
76          return result;
77      }
78      
79      private TransactionAccessType getTransactionAccessType(final TransactionModeItemContext modeItemContext) {
80          if (null != modeItemContext.ONLY()) {
81              return TransactionAccessType.READ_ONLY;
82          }
83          if (null != modeItemContext.WRITE()) {
84              return TransactionAccessType.READ_WRITE;
85          }
86          return null;
87      }
88      
89      private TransactionIsolationLevel getTransactionIsolationLevel(final TransactionModeItemContext modeItemContext) {
90          if (null == modeItemContext.isoLevel()) {
91              return null;
92          }
93          if (null != modeItemContext.isoLevel().UNCOMMITTED()) {
94              return TransactionIsolationLevel.READ_UNCOMMITTED;
95          }
96          if (null != modeItemContext.isoLevel().COMMITTED()) {
97              return TransactionIsolationLevel.READ_COMMITTED;
98          }
99          if (null != modeItemContext.isoLevel().REPEATABLE()) {
100             return TransactionIsolationLevel.REPEATABLE_READ;
101         }
102         if (null != modeItemContext.isoLevel().SERIALIZABLE()) {
103             return TransactionIsolationLevel.SERIALIZABLE;
104         }
105         return null;
106     }

126     }
127     
128     @Override
129     public ASTNode visitSavepoint(final SavepointContext ctx) {
130         String savepointName = ctx.colId().getText();
131         PostgreSQLSavepointStatement result = new PostgreSQLSavepointStatement();
132         result.setSavepointName(savepointName);
133         return result;
134     }
135     
136     @Override

139         result.setSavepointName(ctx.colId().getText());
140         return result;
141     }
142     
143     @Override
144     public ASTNode visitReleaseSavepoint(final ReleaseSavepointContext ctx) {
145         String savepointName = ctx.colId().getText();
146         PostgreSQLReleaseSavepointStatement result = new PostgreSQLReleaseSavepointStatement();
147         result.setSavepointName(savepointName);
148         return result;
149     }
150     
151     @Override
152     public ASTNode visitStartTransaction(final StartTransactionContext ctx) {
153         return new PostgreSQLStartTransactionStatement();
154     }
155

158         return new PostgreSQLCommitStatement();
159     }
160

164     }
165     
166     @Override
167     public ASTNode visitCommitPrepared(final CommitPreparedContext ctx) {
168         return new PostgreSQLCommitPreparedStatement();
169     }
170     
171     @Override
172     public ASTNode visitRollbackPrepared(final RollbackPreparedContext ctx) {
173         return new PostgreSQLRollbackPreparedStatement();
174     }
175     
176     @Override
177     public ASTNode visitLock(final LockContext ctx) {
178         PostgreSQLLockStatement result = new PostgreSQLLockStatement();
179         if (null != ctx.relationExprList()) {
180             result.getTables().addAll(getLockTables(ctx.relationExprList().relationExpr()));
181         }
182         return result;
183     }
184     
185     private Collection<SimpleTableSegment> getLockTables(final List<RelationExprContext> relationExprContexts) {
186         Collection<SimpleTableSegment> result = new LinkedList<>();
187         for (RelationExprContext each : relationExprContexts) {
188             SimpleTableSegment tableSegment = (SimpleTableSegment) visit(each.qualifiedName());
189             result.add(tableSegment);
190         }
191         return result;
192     }
193     
194     @Override
195     public ASTNode visitPrepareTransaction(final PrepareTransactionContext ctx) {
196         return new PostgreSQLPrepareTransactionStatement();
197     }
198     
199     @Override
200     public ASTNode visitCheckpoint(final CheckpointContext ctx) {
201         return new PostgreSQLCheckpointStatement();
202     }
