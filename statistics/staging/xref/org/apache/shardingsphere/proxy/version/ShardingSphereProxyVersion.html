22  import lombok.NoArgsConstructor;
23  import lombok.extern.slf4j.Slf4j;
24  import org.apache.shardingsphere.db.protocol.constant.CommonConstants;

31  import javax.sql.DataSource;
32  import java.util.LinkedHashMap;
33  import java.util.Map;
34  import java.util.Map.Entry;
35  import java.util.Optional;
36  import java.util.stream.Collectors;
37  
38  /**
39   * ShardingSphere-Proxy version.
40   */
41  @NoArgsConstructor(access = AccessLevel.PRIVATE)
42  @Slf4j
43  public final class ShardingSphereProxyVersion {
44      
45      /**
46       * Set version.
47       * 
48       * @param contextManager context manager
49       */
50      public static void setVersion(final ContextManager contextManager) {
51          CommonConstants.PROXY_VERSION.set(getProxyVersion());
52          contextManager.getMetaDataContexts().getMetaData().getDatabases().values().forEach(ShardingSphereProxyVersion::setDatabaseVersion);
53      }
54      
55      private static String getProxyVersion() {
56          String result = ShardingSphereVersion.VERSION;
57          if (!ShardingSphereVersion.IS_SNAPSHOT || Strings.isNullOrEmpty(ShardingSphereVersion.BUILD_GIT_COMMIT_ID_ABBREV)) {
58              return result;
59          }
60          result += ShardingSphereVersion.BUILD_GIT_DIRTY ? "-dirty" : "";
61          result += "-" + ShardingSphereVersion.BUILD_GIT_COMMIT_ID_ABBREV;
62          return result;
63      }
64      
65      private static void setDatabaseVersion(final ShardingSphereDatabase database) {
66          Optional<DataSource> dataSource = findDataSourceByProtocolType(database);
67          if (!dataSource.isPresent()) {
68              return;
69          }
70          DatabaseServerInfo databaseServerInfo = new DatabaseServerInfo(dataSource.get());
71          log.info("{}, database name is `{}`", databaseServerInfo, database.getName());
72          DatabaseProtocolServerInfo.setProtocolVersion(database.getName(), databaseServerInfo.getDatabaseVersion());
73      }
74      
75      private static Optional<DataSource> findDataSourceByProtocolType(final ShardingSphereDatabase database) {
76          Optional<String> dataSourceName = database.getResourceMetaData().getStorageUnits().entrySet()
77                  .stream().filter(entry -> entry.getValue().getStorageType().equals(database.getProtocolType())).map(Entry::getKey).findFirst();
78          Map<String, DataSource> dataSources = database.getResourceMetaData().getStorageUnits().entrySet().stream()
79                  .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new));
80          return dataSourceName.flatMap(optional -> Optional.ofNullable(dataSources.get(optional)));
81      }
82  }
