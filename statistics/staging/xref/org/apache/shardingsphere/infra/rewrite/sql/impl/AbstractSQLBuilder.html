18  package org.apache.shardingsphere.infra.rewrite.sql.impl;
19  
20  import lombok.RequiredArgsConstructor;
21  import org.apache.shardingsphere.infra.rewrite.sql.SQLBuilder;
22  import org.apache.shardingsphere.infra.rewrite.sql.token.pojo.SQLToken;
23  import org.apache.shardingsphere.infra.rewrite.sql.token.pojo.Substitutable;
24  import org.apache.shardingsphere.infra.rewrite.sql.token.pojo.generic.ComposableSQLToken;
25  import org.apache.shardingsphere.infra.rewrite.sql.token.pojo.generic.SubstitutableColumnNameToken;
26  import org.apache.shardingsphere.infra.route.context.RouteUnit;
27  
28  import java.util.Collections;
29  import java.util.List;
30  
31  /**
32   * Abstract SQL builder.
33   */
34  @RequiredArgsConstructor
35  public abstract class AbstractSQLBuilder implements SQLBuilder {
36      
37      private final String sql;
38      
39      private final List<SQLToken> sqlTokens;
40      
41      private final RouteUnit routeUnit;
42      
43      @Override
44      public final String toSQL() {
45          if (sqlTokens.isEmpty()) {
46              return sql;
47          }
48          Collections.sort(sqlTokens);
49          StringBuilder result = new StringBuilder(sql.length());
50          result.append(sql, 0, sqlTokens.get(0).getStartIndex());
51          for (SQLToken each : sqlTokens) {
52              if (each instanceof ComposableSQLToken) {
53                  result.append(getComposableSQLTokenText((ComposableSQLToken) each));
54              } else if (each instanceof SubstitutableColumnNameToken) {
55                  result.append(((SubstitutableColumnNameToken) each).toString(routeUnit));
56              } else {
57                  result.append(getSQLTokenText(each));
58              }
59              result.append(getConjunctionText(each));
60          }
61          return result.toString();
62      }
63      
64      protected abstract String getSQLTokenText(SQLToken sqlToken);
65      
66      private String getComposableSQLTokenText(final ComposableSQLToken composableSQLToken) {
67          StringBuilder result = new StringBuilder();
68          for (SQLToken each : composableSQLToken.getSqlTokens()) {
69              result.append(getSQLTokenText(each));
70              result.append(getConjunctionText(each));
71          }
72          return result.toString();
73      }
74      
75      private String getConjunctionText(final SQLToken sqlToken) {
76          return sql.substring(getStartIndex(sqlToken), getStopIndex(sqlToken));
77      }
78      
79      private int getStartIndex(final SQLToken sqlToken) {
80          int startIndex = sqlToken instanceof Substitutable ? ((Substitutable) sqlToken).getStopIndex() + 1 : sqlToken.getStartIndex();
81          return Math.min(startIndex, sql.length());
82      }
83      
84      private int getStopIndex(final SQLToken sqlToken) {
85          int currentSQLTokenIndex = sqlTokens.indexOf(sqlToken);
86          return sqlTokens.size() - 1 == currentSQLTokenIndex ? sql.length() : sqlTokens.get(currentSQLTokenIndex + 1).getStartIndex();
87      }
88  }
