313     UTF8MB4_0900_BIN(309, () -> StandardCharsets.UTF_8);
314     
315     private static final Map<Integer, MySQLCharacterSet> CHARACTER_SET_MAP = Collections.unmodifiableMap(Arrays.stream(values()).collect(Collectors.toMap(each -> each.id, Function.identity())));
316     
317     private final int id;
318     
319     private final Charset charset;
320     
321     MySQLCharacterSet(final int id, final Supplier<Charset> charsetSupplier) {
322         this.id = id;
323         Charset result = null;
324         try {
325             result = charsetSupplier.get();
326         } catch (final UnsupportedCharsetException ignored) {
327         }
328         charset = result;
329     }
330     
331     /**
332      * Get character set by id.
333      *
334      * @param id id
335      * @return MySQL character set
336      */
337     public static MySQLCharacterSet findById(final int id) {
338         MySQLCharacterSet result = CHARACTER_SET_MAP.get(id);
339         ShardingSpherePreconditions.checkNotNull(result, () -> new UnknownCollationException(id));
340         ShardingSpherePreconditions.checkNotNull(result.getCharset(), () -> new UnknownCollationException(id));
341         return result;
342     }
