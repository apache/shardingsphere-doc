18  package org.apache.shardingsphere.db.protocol.mysql.packet.handshake;
19  
20  import com.google.common.base.Preconditions;
21  import lombok.Getter;
22  import lombok.RequiredArgsConstructor;
23  import org.apache.shardingsphere.db.protocol.mysql.packet.MySQLPacket;
24  import org.apache.shardingsphere.db.protocol.mysql.payload.MySQLPacketPayload;
25  
26  import java.util.Arrays;
27  
28  /**
29   * MySQL authentication switch request packet.
30   *
31   * @see <a href="https://dev.mysql.com/doc/dev/mysql-server/latest/page_protocol_connection_phase_packets_protocol_auth_switch_request.html">AuthSwitchRequest</a>
32   */
33  @RequiredArgsConstructor
34  @Getter
35  public final class MySQLAuthSwitchRequestPacket extends MySQLPacket {
36      
37      /**
38       * Header of MySQL auth switch request packet.
39       */
40      public static final int HEADER = 0xfe;
41      
42      private final String authPluginName;
43      
44      private final MySQLAuthenticationPluginData authPluginData;
45      
46      public MySQLAuthSwitchRequestPacket(final MySQLPacketPayload payload) {
47          Preconditions.checkArgument(HEADER == payload.readInt1(), "Header of MySQL auth switch request packet must be `0xfe`.");
48          authPluginName = payload.readStringNul();
49          String strAuthPluginData = payload.readStringNul();
50          authPluginData = new MySQLAuthenticationPluginData(Arrays.copyOfRange(strAuthPluginData.getBytes(), 0, 8), Arrays.copyOfRange(strAuthPluginData.getBytes(), 8, 20));
51      }
52      
53      @Override
54      protected void write(final MySQLPacketPayload payload) {
55          payload.writeInt1(HEADER);
56          payload.writeStringNul(authPluginName);
57          payload.writeStringNul(new String(authPluginData.getAuthenticationPluginData()));
58      }
59  }
