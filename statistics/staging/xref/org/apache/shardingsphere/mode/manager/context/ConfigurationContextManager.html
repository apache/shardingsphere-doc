78      /**
79       * Register storage unit.
80       *
81       * @param databaseName database name
82       * @param propsMap data source pool properties map
83       */
84      public synchronized void registerStorageUnit(final String databaseName, final Map<String, DataSourcePoolProperties> propsMap) {
85          try {
86              closeStaleRules(databaseName);
87              SwitchingResource switchingResource =
88                      new ResourceSwitchManager().registerStorageUnit(metaDataContexts.get().getMetaData().getDatabase(databaseName).getResourceMetaData(), propsMap);
89              buildNewMetaDataContext(databaseName, switchingResource, false);
90          } catch (final SQLException ex) {
91              log.error("Alter database: {} register storage unit failed", databaseName, ex);
92          }
93      }

129     private void buildNewMetaDataContext(final String databaseName, final SwitchingResource switchingResource, final boolean isDropConfig) throws SQLException {
130         metaDataContexts.get().getMetaData().getDatabases().putAll(renewDatabase(metaDataContexts.get().getMetaData().getDatabase(databaseName), switchingResource));
131         MetaDataContexts reloadMetaDataContexts = createMetaDataContexts(databaseName, false, switchingResource, null);
132         persistSchemaMetaData(databaseName, reloadMetaDataContexts, isDropConfig);
133         Optional.ofNullable(reloadMetaDataContexts.getStatistics().getDatabaseData().get(databaseName))
134                 .ifPresent(optional -> optional.getSchemaData().forEach((schemaName, schemaData) -> reloadMetaDataContexts.getPersistService().getShardingSphereDataPersistService()
135                         .persist(databaseName, schemaName, schemaData, metaDataContexts.get().getMetaData().getDatabases())));
136         alterSchemaMetaData(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName), metaDataContexts.get().getMetaData().getDatabase(databaseName), isDropConfig);
137         metaDataContexts.set(reloadMetaDataContexts);
138         metaDataContexts.get().getMetaData().getDatabases().putAll(newShardingSphereDatabase(metaDataContexts.get().getMetaData().getDatabase(databaseName)));
139         switchingResource.closeStaleDataSources();
140     }

142     private void persistSchemaMetaData(final String databaseName, final MetaDataContexts reloadMetaDataContexts, final boolean isDropConfig) {
143         if (isDropConfig) {
144             reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getSchemas().forEach((schemaName, schema) -> reloadMetaDataContexts.getPersistService().getDatabaseMetaDataService()
145                     .persistByDropConfiguration(reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getName(), schemaName, schema));
146         } else {
147             reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getSchemas().forEach((schemaName, schema) -> reloadMetaDataContexts.getPersistService().getDatabaseMetaDataService()
148                     .persistByAlterConfiguration(reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getName(), schemaName, schema));
149         }
150     }

152     /**
153      * Alter rule configuration.
154      *
155      * @param databaseName database name
156      * @param ruleConfig rule configurations
157      */
158     @SuppressWarnings({"unchecked", "rawtypes"})
159     public synchronized void alterRuleConfiguration(final String databaseName, final RuleConfiguration ruleConfig) {
160         ShardingSphereDatabase database = metaDataContexts.get().getMetaData().getDatabase(databaseName);
161         Collection<ShardingSphereRule> rules = new LinkedList<>(database.getRuleMetaData().getRules());
162         Optional<ShardingSphereRule> toBeChangedRule = rules.stream().filter(each -> each.getConfiguration().getClass().equals(ruleConfig.getClass())).findFirst();
163         if (toBeChangedRule.isPresent() && toBeChangedRule.get() instanceof PartialRuleUpdateSupported && ((PartialRuleUpdateSupported) toBeChangedRule.get()).partialUpdate(ruleConfig)) {
164             ((PartialRuleUpdateSupported) toBeChangedRule.get()).updateConfiguration(ruleConfig);
165             return;
166         }
167         try {
168             rules.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
169             rules.addAll(DatabaseRulesBuilder.build(databaseName, database.getProtocolType(),
170                     database.getResourceMetaData().getStorageUnits().entrySet().stream()
171                             .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
172                     database.getRuleMetaData().getRules(), ruleConfig, instanceContext));
173             refreshMetadata(databaseName, database, rules, false);
174         } catch (final SQLException ex) {
175             log.error("Alter database: {} rule configurations failed", databaseName, ex);
176         }
177     }

179     /**
180      * Drop rule configuration.
181      *
182      * @param databaseName database name
183      * @param ruleConfig rule configurations
184      */
185     @SuppressWarnings({"unchecked", "rawtypes"})
186     public synchronized void dropRuleConfiguration(final String databaseName, final RuleConfiguration ruleConfig) {
187         ShardingSphereDatabase database = metaDataContexts.get().getMetaData().getDatabase(databaseName);
188         Collection<ShardingSphereRule> rules = new LinkedList<>(database.getRuleMetaData().getRules());
189         Optional<ShardingSphereRule> toBeChangedRule = rules.stream().filter(each -> each.getConfiguration().getClass().equals(ruleConfig.getClass())).findFirst();
190         if (toBeChangedRule.isPresent() && toBeChangedRule.get() instanceof PartialRuleUpdateSupported && ((PartialRuleUpdateSupported) toBeChangedRule.get()).partialUpdate(ruleConfig)) {
191             ((PartialRuleUpdateSupported) toBeChangedRule.get()).updateConfiguration(ruleConfig);
192             return;
193         }
194         try {
195             rules.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
196             if (!((DatabaseRuleConfiguration) ruleConfig).isEmpty()) {
197                 rules.addAll(DatabaseRulesBuilder.build(databaseName, database.getProtocolType(),
198                         database.getResourceMetaData().getStorageUnits().entrySet().stream()
199                                 .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
200                         database.getRuleMetaData().getRules(), ruleConfig, instanceContext));
201             }
202             refreshMetadata(databaseName, database, rules, true);
203         } catch (final SQLException ex) {
204             log.error("Drop database: {} rule configurations failed", databaseName, ex);
205         }
206     }

208     private void refreshMetadata(final String databaseName, final ShardingSphereDatabase database, final Collection<ShardingSphereRule> rules, final boolean isDropConfig) throws SQLException {
209         database.getRuleMetaData().getRules().clear();
210         database.getRuleMetaData().getRules().addAll(rules);
211         MetaDataContexts reloadMetaDataContexts = createMetaDataContextsByAlterRule(databaseName, database.getRuleMetaData().getConfigurations());
212         alterSchemaMetaData(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName), metaDataContexts.get().getMetaData().getDatabase(databaseName), isDropConfig);
213         metaDataContexts.set(reloadMetaDataContexts);
214         metaDataContexts.get().getMetaData().getDatabase(databaseName).getSchemas().putAll(newShardingSphereSchemas(metaDataContexts.get().getMetaData().getDatabase(databaseName)));
215     }

217     private MetaDataContexts createMetaDataContextsByAlterRule(final String databaseName, final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
218         Map<String, ShardingSphereDatabase> changedDatabases = createChangedDatabases(databaseName, false, null, ruleConfigs);
219         return newMetaDataContexts(new ShardingSphereMetaData(changedDatabases, metaDataContexts.get().getMetaData().getGlobalResourceMetaData(),
220                 metaDataContexts.get().getMetaData().getGlobalRuleMetaData(), metaDataContexts.get().getMetaData().getProps()));
221     }

223     /**
224      * Alter schema meta data.
225      *
226      * @param databaseName database name
227      * @param reloadDatabase reload database
228      * @param currentDatabase current database
229      * @param isDropConfig is drop configuration
230      */
231     public void alterSchemaMetaData(final String databaseName, final ShardingSphereDatabase reloadDatabase, final ShardingSphereDatabase currentDatabase, final boolean isDropConfig) {
232         Map<String, ShardingSphereSchema> toBeAlterSchemas = GenericSchemaManager.getToBeDeletedTablesBySchemas(reloadDatabase.getSchemas(), currentDatabase.getSchemas());
233         Map<String, ShardingSphereSchema> toBeAddedSchemas = GenericSchemaManager.getToBeAddedTablesBySchemas(reloadDatabase.getSchemas(), currentDatabase.getSchemas());
234         if (isDropConfig) {
235             toBeAddedSchemas.forEach((key, value) -> metaDataContexts.get().getPersistService().getDatabaseMetaDataService().persistByDropConfiguration(databaseName, key, value));
236         } else {
237             toBeAddedSchemas.forEach((key, value) -> metaDataContexts.get().getPersistService().getDatabaseMetaDataService().persistByAlterConfiguration(databaseName, key, value));
238         }
239         toBeAlterSchemas.forEach((key, value) -> metaDataContexts.get().getPersistService().getDatabaseMetaDataService().delete(databaseName, key, value));
240     }

242     /**
243      * Renew ShardingSphere databases.
244      *
245      * @param database database
246      * @param resource resource
247      * @return ShardingSphere databases
248      */
249     public Map<String, ShardingSphereDatabase> renewDatabase(final ShardingSphereDatabase database, final SwitchingResource resource) {
250         Map<StorageNode, DataSource> storageNodes = getStorageNodes(database.getResourceMetaData().getDataSources(), resource);
251         Map<String, StorageUnit> storageUnits = getStorageUnits(database.getResourceMetaData().getStorageUnits(), resource);
252         return Collections.singletonMap(database.getName().toLowerCase(), new ShardingSphereDatabase(database.getName(), database.getProtocolType(),
253                 new ResourceMetaData(storageNodes, storageUnits), database.getRuleMetaData(), database.getSchemas()));
254     }
255     
256     private Map<StorageNode, DataSource> getStorageNodes(final Map<StorageNode, DataSource> currentStorageNodes, final SwitchingResource resource) {
257         Map<StorageNode, DataSource> result = new LinkedHashMap<>(currentStorageNodes.size(), 1F);
258         for (Entry<StorageNode, DataSource> entry : currentStorageNodes.entrySet()) {
259             if (!resource.getStaleDataSources().containsKey(entry.getKey())) {
260                 result.put(entry.getKey(), entry.getValue());
261             }
262         }
263         return result;
264     }
265     
266     private Map<String, StorageUnit> getStorageUnits(final Map<String, StorageUnit> currentStorageUnits, final SwitchingResource resource) {
267         Map<String, StorageUnit> result = new LinkedHashMap<>(currentStorageUnits.size(), 1F);
268         for (Entry<String, StorageUnit> entry : currentStorageUnits.entrySet()) {
269             if (!resource.getStaleStorageUnitNames().contains(entry.getKey())) {
270                 result.put(entry.getKey(), entry.getValue());
271             }
272         }
273         return result;
274     }

276     @SneakyThrows(Exception.class)
277     private void closeStaleRules(final String databaseName) {
278         for (ShardingSphereRule each : metaDataContexts.get().getMetaData().getDatabase(databaseName).getRuleMetaData().getRules()) {
279             if (each instanceof AutoCloseable) {
280                 ((AutoCloseable) each).close();
281             }
282         }
283     }

285     /**
286      * Create meta data contexts.
287      *
288      * @param databaseName database name
289      * @param internalLoadMetaData internal load meta data
290      * @param switchingResource switching resource
291      * @param ruleConfigs rule configs
292      * @return MetaDataContexts meta data contexts
293      * @throws SQLException SQL exception
294      */
295     public MetaDataContexts createMetaDataContexts(final String databaseName, final boolean internalLoadMetaData, final SwitchingResource switchingResource,
296                                                    final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
297         Map<String, ShardingSphereDatabase> changedDatabases = createChangedDatabases(databaseName, internalLoadMetaData, switchingResource, ruleConfigs);
298         ConfigurationProperties props = metaDataContexts.get().getMetaData().getProps();
299         RuleMetaData changedGlobalMetaData = new RuleMetaData(
300                 GlobalRulesBuilder.buildRules(metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getConfigurations(), changedDatabases, props));
301         return newMetaDataContexts(new ShardingSphereMetaData(changedDatabases, metaDataContexts.get().getMetaData().getGlobalResourceMetaData(), changedGlobalMetaData, props));
302     }

304     /**
305      * Create changed databases.
306      *
307      * @param databaseName database name
308      * @param internalLoadMetaData internal load meta data
309      * @param switchingResource switching resource
310      * @param ruleConfigs rule configs
311      * @return ShardingSphere databases
312      * @throws SQLException SQL exception
313      */
314     public synchronized Map<String, ShardingSphereDatabase> createChangedDatabases(final String databaseName, final boolean internalLoadMetaData,
315                                                                                    final SwitchingResource switchingResource, final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
316         Collection<RuleConfiguration> toBeCreatedRuleConfigs = null == ruleConfigs
317                 ? metaDataContexts.get().getMetaData().getDatabase(databaseName).getRuleMetaData().getConfigurations()
318                 : ruleConfigs;
319         DatabaseConfiguration toBeCreatedDatabaseConfig = getDatabaseConfiguration(
320                 metaDataContexts.get().getMetaData().getDatabase(databaseName).getResourceMetaData(), switchingResource, toBeCreatedRuleConfigs);
321         ShardingSphereDatabase changedDatabase = createChangedDatabase(metaDataContexts.get().getMetaData().getDatabase(databaseName).getName(), internalLoadMetaData,
322                 metaDataContexts.get().getPersistService(), toBeCreatedDatabaseConfig, metaDataContexts.get().getMetaData().getProps(), instanceContext);
323         Map<String, ShardingSphereDatabase> result = new LinkedHashMap<>(metaDataContexts.get().getMetaData().getDatabases());
324         result.put(databaseName.toLowerCase(), changedDatabase);
325         return result;
326     }

328     private DatabaseConfiguration getDatabaseConfiguration(final ResourceMetaData resourceMetaData, final SwitchingResource switchingResource,
329                                                            final Collection<RuleConfiguration> toBeCreatedRuleConfigs) {
330         Map<String, DataSourcePoolProperties> propsMap = null == switchingResource
331                 ? resourceMetaData.getStorageUnits().entrySet().stream()
332                         .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSourcePoolProperties(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new))
333                 : switchingResource.getMergedDataSourcePoolPropertiesMap();
334         return new DataSourceProvidedDatabaseConfiguration(getMergedStorageNodeDataSources(resourceMetaData, switchingResource), toBeCreatedRuleConfigs, propsMap);
335     }
336     
337     private Map<StorageNode, DataSource> getMergedStorageNodeDataSources(final ResourceMetaData currentResourceMetaData, final SwitchingResource switchingResource) {
338         Map<StorageNode, DataSource> result = currentResourceMetaData.getDataSources();
339         if (null != switchingResource && !switchingResource.getNewDataSources().isEmpty()) {
340             result.putAll(switchingResource.getNewDataSources());
341         }
342         return result;
343     }

345     private ShardingSphereDatabase createChangedDatabase(final String databaseName, final boolean internalLoadMetaData, final MetaDataPersistService persistService,
346                                                          final DatabaseConfiguration databaseConfig, final ConfigurationProperties props, final InstanceContext instanceContext) throws SQLException {
347         return internalLoadMetaData
348                 ? InternalMetaDataFactory.create(databaseName, persistService, databaseConfig, props, instanceContext)
349                 : ExternalMetaDataFactory.create(databaseName, databaseConfig, props, instanceContext);
350     }

352     private Map<String, ShardingSphereSchema> newShardingSphereSchemas(final ShardingSphereDatabase database) {
353         Map<String, ShardingSphereSchema> result = new LinkedHashMap<>(database.getSchemas().size(), 1F);
354         database.getSchemas().forEach((key, value) -> result.put(key, new ShardingSphereSchema(value.getTables(), value.getViews())));
355         return result;
356     }

370     /**
371      * Alter global rule configuration.
372      *
373      * @param ruleConfig global rule configuration
374      */
375     public synchronized void alterGlobalRuleConfiguration(final RuleConfiguration ruleConfig) {
376         if (null == ruleConfig) {
377             return;
378         }
379         closeStaleTransactionRule(ruleConfig);
380         Collection<ShardingSphereRule> rules = new LinkedList<>(metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getRules());
381         rules.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
382         rules.addAll(GlobalRulesBuilder.buildSingleRules(ruleConfig, metaDataContexts.get().getMetaData().getDatabases(), metaDataContexts.get().getMetaData().getProps()));
383         metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getRules().clear();
384         metaDataContexts.get().getMetaData().getGlobalRuleMetaData().getRules().addAll(rules);
385         ShardingSphereMetaData toBeChangedMetaData = new ShardingSphereMetaData(metaDataContexts.get().getMetaData().getDatabases(), metaDataContexts.get().getMetaData().getGlobalResourceMetaData(),
386                 metaDataContexts.get().getMetaData().getGlobalRuleMetaData(), metaDataContexts.get().getMetaData().getProps());
387         metaDataContexts.set(newMetaDataContexts(toBeChangedMetaData));
388     }

390     // Optimize string comparison rule type.
391     @SneakyThrows(Exception.class)
392     private void closeStaleTransactionRule(final RuleConfiguration ruleConfig) {
393         YamlRuleConfiguration yamlRuleConfig = new YamlRuleConfigurationSwapperEngine().swapToYamlRuleConfiguration(ruleConfig);
394         if (!"transaction".equals(Objects.requireNonNull(yamlRuleConfig.getClass().getAnnotation(RepositoryTupleEntity.class)).value())) {
395             return;
396         }
397         Optional<TransactionRule> transactionRule = metaDataContexts.get().getMetaData().getGlobalRuleMetaData().findSingleRule(TransactionRule.class);
398         if (transactionRule.isPresent()) {
399             ((AutoCloseable) transactionRule.get()).close();
400         }
401     }

410                 metaDataContexts.get().getMetaData().getGlobalRuleMetaData(), new ConfigurationProperties(props));
411         metaDataContexts.set(newMetaDataContexts(toBeChangedMetaData));
412     }
