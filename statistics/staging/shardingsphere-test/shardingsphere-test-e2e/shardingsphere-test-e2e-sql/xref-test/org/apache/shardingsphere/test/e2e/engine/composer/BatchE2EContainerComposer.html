56  /**
57   * Batch E2E container composer.
58   */
59  public final class BatchE2EContainerComposer extends E2EContainerComposer {
60      
61      private final DatabaseType databaseType;
62      
63      private final Collection<DataSet> dataSets = new LinkedList<>();
64      
65      private final DataSetEnvironmentManager dataSetEnvironmentManager;
66      
67      public BatchE2EContainerComposer(final CaseTestParameter testParam) throws JAXBException, IOException, SQLException, ParseException {
68          super(testParam);
69          databaseType = testParam.getDatabaseType();
70          for (IntegrationTestCaseAssertion each : testParam.getTestCaseContext().getTestCase().getAssertions()) {
71              dataSets.add(DataSetLoader.load(testParam.getTestCaseContext().getParentPath(), testParam.getScenario(), testParam.getDatabaseType(), testParam.getMode(), each.getExpectedDataFile()));
72          }
73          dataSetEnvironmentManager = new DataSetEnvironmentManager(new ScenarioDataPath(testParam.getScenario()).getDataSetFile(Type.ACTUAL), getActualDataSourceMap());
74          dataSetEnvironmentManager.fillData();
75      }
76      
77      /**
78       * Assert data sets.
79       * 
80       * @param actualUpdateCounts actual update counts
81       * @throws SQLException SQL exception
82       */
83      public void assertDataSets(final int[] actualUpdateCounts) throws SQLException {
84          DataSet expected = getDataSet(actualUpdateCounts);
85          assertThat("Only support single table for DML.", expected.getMetaDataList().size(), is(1));
86          DataSetMetaData expectedDataSetMetaData = expected.getMetaDataList().get(0);
87          for (String each : new InlineExpressionParser(expectedDataSetMetaData.getDataNodes()).splitAndEvaluate()) {
88              DataNode dataNode = new DataNode(each);
89              DataSource dataSource = getActualDataSourceMap().get(dataNode.getDataSourceName());
90              try (
91                      Connection connection = dataSource.getConnection();
92                      PreparedStatement preparedStatement = connection.prepareStatement(String.format("SELECT * FROM %s ORDER BY 1", dataNode.getTableName()))) {
93                  assertDataSet(preparedStatement, expected.findRows(dataNode), expectedDataSetMetaData);
94              }
95          }
96      }
97      
98      private DataSet getDataSet(final int[] actualUpdateCounts) {
99          Collection<DataSet> dataSets = new LinkedList<>();
100         assertThat(actualUpdateCounts.length, is(this.dataSets.size()));
101         int count = 0;
102         for (DataSet each : this.dataSets) {
103             assertThat(actualUpdateCounts[count], is(each.getUpdateCount()));
104             dataSets.add(each);
105             count++;
106         }
107         return mergeDataSets(dataSets);
108     }
109     
110     private DataSet mergeDataSets(final Collection<DataSet> dataSets) {
111         DataSet result = new DataSet();
112         Set<DataSetRow> existedRows = new HashSet<>();
113         for (DataSet each : dataSets) {
114             mergeMetaData(each, result);
115             mergeRow(each, result, existedRows);
116         }
117         sortRow(result);
118         return result;
119     }
120     
121     private void mergeMetaData(final DataSet original, final DataSet dist) {
122         if (dist.getMetaDataList().isEmpty()) {
123             dist.getMetaDataList().addAll(original.getMetaDataList());
124         }
125     }
126     
127     private void mergeRow(final DataSet original, final DataSet dist, final Set<DataSetRow> existedRows) {
128         for (DataSetRow each : original.getRows()) {
129             if (existedRows.add(each)) {
130                 dist.getRows().add(each);
131             }
132         }
133     }
134     
135     private void sortRow(final DataSet dataSet) {
136         dataSet.getRows().sort(Comparator.comparingLong(o -> Long.parseLong(o.splitValues(",").get(0))));
137     }
138     
139     private void assertDataSet(final PreparedStatement actualPreparedStatement, final List<DataSetRow> expectedDataSetRows, final DataSetMetaData expectedDataSetMetaData) throws SQLException {
140         try (ResultSet actualResultSet = actualPreparedStatement.executeQuery()) {
141             assertMetaData(actualResultSet.getMetaData(), expectedDataSetMetaData.getColumns());
142             assertRows(actualResultSet, expectedDataSetRows);
143         }
144     }
145     
146     private void assertMetaData(final ResultSetMetaData actualMetaData, final Collection<DataSetColumn> columnMetaDataList) throws SQLException {
147         assertThat(actualMetaData.getColumnCount(), is(columnMetaDataList.size()));
148         int index = 1;
149         for (DataSetColumn each : columnMetaDataList) {
150             assertThat(actualMetaData.getColumnLabel(index++), is(each.getName()));
151         }
152     }
153     
154     private void assertRows(final ResultSet actual, final List<DataSetRow> expectedDatSetRows) throws SQLException {
155         int count = 0;
156         while (actual.next()) {
157             int columnIndex = 1;
158             for (String expected : expectedDatSetRows.get(count).splitValues(", ")) {
159                 if (Types.DATE == actual.getMetaData().getColumnType(columnIndex)) {
160                     if (!E2EContainerComposer.NOT_VERIFY_FLAG.equals(expected)) {
161                         assertThat(new SimpleDateFormat("yyyy-MM-dd").format(actual.getDate(columnIndex)), is(expected));
162                     }
163                 } else if (Types.CHAR == actual.getMetaData().getColumnType(columnIndex)
164                         && ("PostgreSQL".equals(databaseType.getType()) || "openGauss".equals(databaseType.getType()))) {
165                     assertThat(String.valueOf(actual.getObject(columnIndex)).trim(), is(expected));
166                 } else if (isPostgreSQLOrOpenGaussMoney(actual.getMetaData().getColumnTypeName(columnIndex))) {
167                     assertThat(actual.getString(columnIndex), is(expected));
168                 } else if (Types.BINARY == actual.getMetaData().getColumnType(columnIndex)) {
169                     assertThat(actual.getObject(columnIndex), is(expected.getBytes(StandardCharsets.UTF_8)));
170                 } else {
171                     assertThat(String.valueOf(actual.getObject(columnIndex)), is(expected));
172                 }
173                 columnIndex++;
174             }
175             count++;
176         }
177         assertThat("Size of actual result set is different with size of expected dat set rows.", count, is(expectedDatSetRows.size()));
178     }
179     
180     private boolean isPostgreSQLOrOpenGaussMoney(final String columnTypeName) {
181         return "money".equalsIgnoreCase(columnTypeName) && ("PostgreSQL".equals(databaseType.getType()) || "openGauss".equals(databaseType.getType()));
182     }
183     
184     @Override
185     public void close() {
186         dataSetEnvironmentManager.cleanData();
187         super.close();
188     }
189 }
