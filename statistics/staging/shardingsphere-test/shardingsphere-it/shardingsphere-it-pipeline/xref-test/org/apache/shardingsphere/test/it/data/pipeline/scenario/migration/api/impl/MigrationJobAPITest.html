241     @SneakyThrows(SQLException.class)
242     private void initTableData(final MigrationJobConfiguration jobConfig) {
243         PipelineDataSourceConfiguration source = jobConfig.getSources().values().iterator().next();
244         PipelineDataSourceConfiguration sourceDataSourceConfig = PipelineDataSourceConfigurationFactory.newInstance(source.getType(), source.getParameter());
245         initTableData(TypedSPILoader.getService(
246                 PipelineDataSourceCreator.class, sourceDataSourceConfig.getType()).createPipelineDataSource(sourceDataSourceConfig.getDataSourceConfiguration()));
247         PipelineDataSourceConfiguration targetDataSourceConfig = PipelineDataSourceConfigurationFactory.newInstance(jobConfig.getTarget().getType(), jobConfig.getTarget().getParameter());
248         initTableData(TypedSPILoader.getService(
249                 PipelineDataSourceCreator.class, targetDataSourceConfig.getType()).createPipelineDataSource(targetDataSourceConfig.getDataSourceConfiguration()));
250     }
251     
252     private void initTableData(final DataSource pipelineDataSource) throws SQLException {
253         try (
254                 Connection connection = pipelineDataSource.getConnection();
255                 Statement statement = connection.createStatement()) {
256             statement.execute("DROP TABLE IF EXISTS t_order");
257             statement.execute("CREATE TABLE t_order (order_id INT PRIMARY KEY, user_id INT(11))");
258             statement.execute("INSERT INTO t_order (order_id, user_id) VALUES (1, 0), (999, 15)");
259         }
260     }

293     @Test
294     void assertCreateJobConfig() throws SQLException {
295         initIntPrimaryEnvironment();
296         SourceTargetEntry sourceTargetEntry = new SourceTargetEntry("logic_db", new DataNode("ds_0", "t_order"), "t_order");
297         String jobId = jobAPI.createJobAndStart(PipelineContextUtils.getContextKey(), new MigrateTableStatement(Collections.singletonList(sourceTargetEntry), "logic_db"));
298         MigrationJobConfiguration actual = jobAPI.getJobConfiguration(jobId);
299         assertThat(actual.getTargetDatabaseName(), is("logic_db"));
300         List<JobDataNodeLine> dataNodeLines = actual.getJobShardingDataNodes();
301         assertThat(dataNodeLines.size(), is(1));
302         assertThat(dataNodeLines.get(0).getEntries().size(), is(1));
303         JobDataNodeEntry entry = dataNodeLines.get(0).getEntries().get(0);
304         assertThat(entry.getDataNodes().size(), is(1));
305         DataNode dataNode = entry.getDataNodes().get(0);
306         assertThat(dataNode.getDataSourceName(), is("ds_0"));
307         assertThat(dataNode.getTableName(), is("t_order"));
308         assertThat(entry.getLogicTableName(), is("t_order"));
309     }

312         Map<String, DataSourceProperties> metaDataDataSource = new PipelineDataSourcePersistService().load(PipelineContextUtils.getContextKey(), new MigrationJobType());

315                 PipelineDataSourceWrapper dataSource = new PipelineDataSourceWrapper(DataSourcePoolCreator.create(dataSourceProps), databaseType);
316                 Connection connection = dataSource.getConnection();
317                 Statement statement = connection.createStatement()) {
318             statement.execute("DROP TABLE IF EXISTS t_order");
319             statement.execute("CREATE TABLE t_order (order_id INT PRIMARY KEY, user_id INT)");
320         }
321     }
