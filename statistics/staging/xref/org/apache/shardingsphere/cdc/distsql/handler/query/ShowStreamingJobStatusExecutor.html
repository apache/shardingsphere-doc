36  /**
37   * Show streaming job status executor.
38   */
39  public final class ShowStreamingJobStatusExecutor implements QueryableRALExecutor<ShowStreamingStatusStatement> {
40      
41      @Override
42      public Collection<LocalDataQueryResultRow> getRows(final ShowStreamingStatusStatement sqlStatement) {
43          InventoryIncrementalJobAPI jobAPI = (InventoryIncrementalJobAPI) TypedSPILoader.getService(PipelineJobAPI.class, new CDCJobType().getTypeName());
44          List<InventoryIncrementalJobItemInfo> jobItemInfos = jobAPI.getJobItemInfos(sqlStatement.getJobId());
45          long currentTimeMillis = System.currentTimeMillis();
46          return jobItemInfos.stream().map(each -> generateResultRow(each, currentTimeMillis)).collect(Collectors.toList());
47      }
48      
49      private LocalDataQueryResultRow generateResultRow(final InventoryIncrementalJobItemInfo jobItemInfo, final long currentTimeMillis) {
50          InventoryIncrementalJobItemProgress jobItemProgress = jobItemInfo.getJobItemProgress();
51          if (null == jobItemProgress) {
52              return new LocalDataQueryResultRow(jobItemInfo.getShardingItem(), "", "", "", "", "", "", jobItemInfo.getErrorMessage());
53          }
54          String incrementalIdleSeconds = "";
55          if (jobItemProgress.getIncremental().getIncrementalLatestActiveTimeMillis() > 0) {
56              long latestActiveTimeMillis = Math.max(jobItemInfo.getStartTimeMillis(), jobItemProgress.getIncremental().getIncrementalLatestActiveTimeMillis());
57              incrementalIdleSeconds = String.valueOf(TimeUnit.MILLISECONDS.toSeconds(currentTimeMillis - latestActiveTimeMillis));
58          }
59          return new LocalDataQueryResultRow(jobItemInfo.getShardingItem(), jobItemProgress.getDataSourceName(), jobItemProgress.getStatus(),
60                  jobItemProgress.isActive() ? Boolean.TRUE.toString() : Boolean.FALSE.toString(), jobItemProgress.getProcessedRecordsCount(), jobItemInfo.getInventoryFinishedPercentage(),
61                  incrementalIdleSeconds, jobItemInfo.getErrorMessage());
62      }
63      
64      @Override
65      public Collection<String> getColumnNames() {
66          return Arrays.asList("item", "data_source", "status", "active", "processed_records_count", "inventory_finished_percentage", "incremental_idle_seconds", "error_message");
67      }
68      
69      @Override
70      public String getType() {
71          return ShowStreamingStatusStatement.class.getName();
72      }
73  }
