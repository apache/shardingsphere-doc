44  /**
45   * Databases shadow rule.
46   */
47  @Getter
48  public final class ShadowRule implements DatabaseRule, DataSourceContainedRule {
49      
50      private final RuleConfiguration configuration;
51      
52      private final Collection<String> hintShadowAlgorithmNames = new LinkedList<>();
53      
54      private final Map<String, ShadowDataSourceRule> shadowDataSourceMappings = new LinkedHashMap<>();
55      
56      private final Map<String, ShadowAlgorithm> shadowAlgorithms = new LinkedHashMap<>();
57      
58      private final Map<String, ShadowTableRule> shadowTableRules = new LinkedHashMap<>();
59      
60      private final ShadowAlgorithm defaultShadowAlgorithm;
61      
62      public ShadowRule(final ShadowRuleConfiguration ruleConfig) {
63          configuration = ruleConfig;
64          initShadowDataSourceMappings(ruleConfig.getDataSources());
65          initShadowAlgorithmConfigurations(ruleConfig.getShadowAlgorithms());
66          defaultShadowAlgorithm = shadowAlgorithms.get(ruleConfig.getDefaultShadowAlgorithmName());
67          if (defaultShadowAlgorithm instanceof HintShadowAlgorithm<?>) {
68              hintShadowAlgorithmNames.add(ruleConfig.getDefaultShadowAlgorithmName());
69          }
70          initShadowTableRules(ruleConfig.getTables());
71      }
72      
73      private void initShadowDataSourceMappings(final Collection<ShadowDataSourceConfiguration> dataSources) {
74          dataSources.forEach(each -> shadowDataSourceMappings.put(each.getName(), new ShadowDataSourceRule(each.getProductionDataSourceName(), each.getShadowDataSourceName())));
75      }
76      
77      private void initShadowAlgorithmConfigurations(final Map<String, AlgorithmConfiguration> shadowAlgorithmConfigs) {
78          shadowAlgorithmConfigs.forEach((key, value) -> {
79              ShadowAlgorithm algorithm = TypedSPILoader.getService(ShadowAlgorithm.class, value.getType(), value.getProps());
80              if (algorithm instanceof HintShadowAlgorithm<?>) {
81                  hintShadowAlgorithmNames.add(key);
82              }
83              shadowAlgorithms.put(key, algorithm);
84          });
85      }
86      
87      private void initShadowTableRules(final Map<String, ShadowTableConfiguration> tables) {
88          tables.forEach((key, value) -> shadowTableRules.put(key, new ShadowTableRule(key, value.getDataSourceNames(), value.getShadowAlgorithmNames(), shadowAlgorithms)));
89      }
90      
91      /**
92       * Get default shadow algorithm.
93       *
94       * @return shadow algorithm
95       */
96      public Optional<ShadowAlgorithm> getDefaultShadowAlgorithm() {
97          return null == defaultShadowAlgorithm ? Optional.empty() : Optional.of(defaultShadowAlgorithm);
98      }
99      
100     /**
101      * Get related shadow tables.
102      *
103      * @param tableNames table names
104      * @return related shadow tables
105      */
106     public Collection<String> getRelatedShadowTables(final Collection<String> tableNames) {
107         Collection<String> result = new LinkedList<>();
108         for (String each : tableNames) {
109             if (shadowTableRules.containsKey(each)) {
110                 result.add(each);
111             }
112         }
113         return result;
114     }
115     
116     /**
117      * Get all shadow table names.
118      *
119      * @return shadow table names
120      */
121     public Collection<String> getAllShadowTableNames() {
122         return shadowTableRules.keySet();
123     }
124     
125     /**
126      * Get related hint shadow algorithms.
127      *
128      * @return related hint shadow algorithms
129      */
130     @SuppressWarnings("unchecked")
131     public Collection<HintShadowAlgorithm<Comparable<?>>> getAllHintShadowAlgorithms() {
132         Collection<HintShadowAlgorithm<Comparable<?>>> result = new LinkedList<>();
133         for (String each : hintShadowAlgorithmNames) {
134             result.add((HintShadowAlgorithm<Comparable<?>>) shadowAlgorithms.get(each));
135         }
136         return result;
137     }
138     
139     /**
140      * Get related hint shadow algorithms by table name.
141      *
142      * @param tableName table name
143      * @return hint shadow algorithms
144      */
145     @SuppressWarnings("unchecked")
146     public Collection<HintShadowAlgorithm<Comparable<?>>> getRelatedHintShadowAlgorithms(final String tableName) {
147         Collection<HintShadowAlgorithm<Comparable<?>>> result = new LinkedList<>();
148         Collection<String> hintShadowAlgorithmNames = shadowTableRules.get(tableName).getHintShadowAlgorithmNames();
149         for (String each : hintShadowAlgorithmNames) {
150             result.add((HintShadowAlgorithm<Comparable<?>>) shadowAlgorithms.get(each));
151         }
152         return result;
153     }
154     
155     /**
156      * Get related column shadow algorithms by table name.
157      *
158      * @param shadowOperationType shadow operation type
159      * @param tableName table name
160      * @param shadowColumn shadow column
161      * @return column shadow algorithms
162      */
163     @SuppressWarnings("unchecked")
164     public Collection<ColumnShadowAlgorithm<Comparable<?>>> getRelatedColumnShadowAlgorithms(final ShadowOperationType shadowOperationType, final String tableName, final String shadowColumn) {
165         Collection<ColumnShadowAlgorithm<Comparable<?>>> result = new LinkedList<>();
166         Map<ShadowOperationType, Collection<ShadowAlgorithmNameRule>> columnShadowAlgorithmNames = shadowTableRules.get(tableName).getColumnShadowAlgorithmNames();
167         Collection<ShadowAlgorithmNameRule> names = columnShadowAlgorithmNames.get(shadowOperationType);
168         if (Objects.isNull(names)) {
169             return result;
170         }
171         for (ShadowAlgorithmNameRule each : names) {
172             if (shadowColumn.equals(each.getShadowColumnName())) {
173                 result.add((ColumnShadowAlgorithm<Comparable<?>>) shadowAlgorithms.get(each.getShadowAlgorithmName()));
174             }
175         }
176         return result;
177     }
178     
179     /**
180      * Get related shadow column names.
181      *
182      * @param shadowOperationType shadow operation type
183      * @param tableName table name
184      * @return related shadow column names
185      */
186     public Collection<String> getRelatedShadowColumnNames(final ShadowOperationType shadowOperationType, final String tableName) {
187         Collection<String> result = new LinkedList<>();
188         Map<ShadowOperationType, Collection<ShadowAlgorithmNameRule>> columnShadowAlgorithmNames = shadowTableRules.get(tableName).getColumnShadowAlgorithmNames();
189         Collection<ShadowAlgorithmNameRule> names = columnShadowAlgorithmNames.get(shadowOperationType);
190         if (Objects.isNull(names)) {
191             return result;
192         }
193         for (ShadowAlgorithmNameRule each : names) {
194             result.add(each.getShadowColumnName());
195         }
196         return result;
197     }
198     
199     /**
200      * Get shadow data source mappings.
201      *
202      * @param tableName table name
203      * @return shadow data source rules
204      */
205     public Map<String, String> getRelatedShadowDataSourceMappings(final String tableName) {
206         Map<String, String> result = new LinkedHashMap<>();
207         Collection<String> shadowDataSources = shadowTableRules.get(tableName).getShadowDataSources();
208         for (String each : shadowDataSources) {
209             ShadowDataSourceRule shadowDataSourceRule = shadowDataSourceMappings.get(each);
210             result.put(shadowDataSourceRule.getProductionDataSource(), shadowDataSourceRule.getShadowDataSource());
211         }
212         return result;
213     }
214     
215     /**
216      * Get all shadow data source mappings.
217      *
218      * @return all shadow data source mappings
219      */
220     public Map<String, String> getAllShadowDataSourceMappings() {
221         Map<String, String> result = new LinkedHashMap<>();
222         for (Entry<String, ShadowDataSourceRule> entry : shadowDataSourceMappings.entrySet()) {
223             ShadowDataSourceRule rule = entry.getValue();
224             result.put(rule.getProductionDataSource(), rule.getShadowDataSource());
225         }
226         return result;
227     }
228     
229     /**
230      * Get source data source name.
231      *
232      * @param actualDataSourceName actual data source name
233      * @return source data source name
234      */
235     public Optional<String> getSourceDataSourceName(final String actualDataSourceName) {
236         ShadowDataSourceRule shadowDataSourceRule = shadowDataSourceMappings.get(actualDataSourceName);
237         return null == shadowDataSourceRule ? Optional.empty() : Optional.of(shadowDataSourceRule.getProductionDataSource());
238     }
239     
240     @Override
241     public Map<String, Collection<DataSourceRoleInfo>> getDataSourceMapper() {
242         Map<String, Collection<DataSourceRoleInfo>> result = new LinkedHashMap<>();
243         shadowDataSourceMappings.forEach((key, value) -> result.put(key, createShadowDataSources(value)));
244         return result;
245     }
246     
247     private Collection<DataSourceRoleInfo> createShadowDataSources(final ShadowDataSourceRule shadowDataSourceRule) {
248         Collection<DataSourceRoleInfo> result = new LinkedList<>();
249         result.add(new DataSourceRoleInfo(shadowDataSourceRule.getProductionDataSource(), DataSourceRole.PRODUCTION));
250         result.add(new DataSourceRoleInfo(shadowDataSourceRule.getShadowDataSource(), DataSourceRole.SHADOW));
251         return result;
252     }
253     
254     @Override
255     public String getType() {
256         return ShadowRule.class.getSimpleName();
257     }
