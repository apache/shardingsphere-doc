44  /**
45   * Show storage unit executor.
46   */
47  public final class ShowStorageUnitExecutor implements RQLExecutor<ShowStorageUnitsStatement> {
48      
49      private static final String CONNECTION_TIMEOUT_MILLISECONDS = "connectionTimeoutMilliseconds";
50      
51      private static final String IDLE_TIMEOUT_MILLISECONDS = "idleTimeoutMilliseconds";
52      
53      private static final String MAX_LIFETIME_MILLISECONDS = "maxLifetimeMilliseconds";
54      
55      private static final String MAX_POOL_SIZE = "maxPoolSize";
56      
57      private static final String MIN_POOL_SIZE = "minPoolSize";
58      
59      private static final String READ_ONLY = "readOnly";
60      
61      @Override
62      public Collection<String> getColumnNames() {
63          return Arrays.asList("name", "type", "host", "port", "db", "connection_timeout_milliseconds", "idle_timeout_milliseconds",
64                  "max_lifetime_milliseconds", "max_pool_size", "min_pool_size", "read_only", "other_attributes");
65      }
66      
67      @Override
68      public Collection<LocalDataQueryResultRow> getRows(final ShardingSphereDatabase database, final ShowStorageUnitsStatement sqlStatement) {
69          ShardingSphereResourceMetaData resourceMetaData = database.getResourceMetaData();
70          Map<String, DataSourceProperties> dataSourcePropsMap = getDataSourcePropsMap(database, sqlStatement);
71          Collection<LocalDataQueryResultRow> result = new LinkedList<>();
72          for (Entry<String, DataSourceProperties> entry : dataSourcePropsMap.entrySet()) {
73              String key = entry.getKey();
74              DataSourceProperties dataSourceProps = entry.getValue();
75              ConnectionProperties connectionProps = resourceMetaData.getConnectionProperties(key);
76              Map<String, Object> standardProps = dataSourceProps.getPoolPropertySynonyms().getStandardProperties();
77              Map<String, Object> otherProps = dataSourceProps.getCustomDataSourceProperties().getProperties();
78              result.add(new LocalDataQueryResultRow(key,
79                      resourceMetaData.getStorageType(key).getType(),
80                      connectionProps.getHostname(),
81                      connectionProps.getPort(),
82                      connectionProps.getCatalog(),
83                      getStandardProperty(standardProps, CONNECTION_TIMEOUT_MILLISECONDS),
84                      getStandardProperty(standardProps, IDLE_TIMEOUT_MILLISECONDS),
85                      getStandardProperty(standardProps, MAX_LIFETIME_MILLISECONDS),
86                      getStandardProperty(standardProps, MAX_POOL_SIZE),
87                      getStandardProperty(standardProps, MIN_POOL_SIZE),
88                      getStandardProperty(standardProps, READ_ONLY),
89                      otherProps.isEmpty() ? "" : new Gson().toJson(otherProps)));
90          }
91          return result;
92      }
93      
94      private Map<String, DataSourceProperties> getDataSourcePropsMap(final ShardingSphereDatabase database, final ShowStorageUnitsStatement sqlStatement) {
95          Map<String, DataSourceProperties> result = new LinkedHashMap<>(database.getResourceMetaData().getDataSources().size(), 1F);
96          Map<String, DataSourceProperties> dataSourcePropsMap = database.getResourceMetaData().getDataSourcePropsMap();
97          Map<String, DatabaseType> storageTypes = database.getResourceMetaData().getStorageTypes();
98          Optional<Integer> usageCountOptional = sqlStatement.getUsageCount();
99          if (usageCountOptional.isPresent()) {
100             Map<String, Collection<String>> inUsedStorageUnits = StorageUnitUtils.getInUsedStorageUnits(database.getRuleMetaData(), database.getResourceMetaData().getDataSources().size());
101             for (Entry<String, DataSource> entry : database.getResourceMetaData().getDataSources().entrySet()) {
102                 Integer currentUsageCount = inUsedStorageUnits.containsKey(entry.getKey()) ? inUsedStorageUnits.get(entry.getKey()).size() : 0;
103                 if (usageCountOptional.get().equals(currentUsageCount)) {
104                     result.put(entry.getKey(), getDataSourceProperties(dataSourcePropsMap, entry.getKey(), storageTypes.get(entry.getKey()), entry.getValue()));
105                 }
106             }
107         } else {
108             for (Entry<String, DataSource> entry : database.getResourceMetaData().getDataSources().entrySet()) {
109                 result.put(entry.getKey(), getDataSourceProperties(dataSourcePropsMap, entry.getKey(), storageTypes.get(entry.getKey()), entry.getValue()));
110             }
111         }
112         return result;
113     }
114     
115     private DataSourceProperties getDataSourceProperties(final Map<String, DataSourceProperties> dataSourcePropsMap, final String storageUnitName,
116                                                          final DatabaseType databaseType, final DataSource dataSource) {
117         DataSourceProperties result = getDataSourceProperties(dataSource);
118         DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, databaseType);
119         if (dialectDatabaseMetaData.isInstanceConnectionAvailable() && dataSourcePropsMap.containsKey(storageUnitName)) {
120             DataSourceProperties unitDataSourceProperties = dataSourcePropsMap.get(storageUnitName);
121             for (Entry<String, Object> entry : unitDataSourceProperties.getPoolPropertySynonyms().getStandardProperties().entrySet()) {
122                 if (null != entry.getValue()) {
123                     result.getPoolPropertySynonyms().getStandardProperties().put(entry.getKey(), entry.getValue());
124                 }
125             }
126         }
127         return result;
128     }
129     
130     private DataSourceProperties getDataSourceProperties(final DataSource dataSource) {
131         return dataSource instanceof ShardingSphereStorageDataSourceWrapper
132                 ? DataSourcePropertiesCreator.create(((ShardingSphereStorageDataSourceWrapper) dataSource).getDataSource())
133                 : DataSourcePropertiesCreator.create(dataSource);
134     }
135     
136     private String getStandardProperty(final Map<String, Object> standardProps, final String key) {
137         if (standardProps.containsKey(key) && null != standardProps.get(key)) {
138             return standardProps.get(key).toString();
139         }
140         return "";
141     }
142     
143     @Override
144     public String getType() {
145         return ShowStorageUnitsStatement.class.getName();
146     }
147 }
