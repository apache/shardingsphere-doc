43  class ProxyConfigurationLoaderTest {
44      
45      @Test
46      void assertLoadEmptyConfiguration() throws IOException {
47          YamlProxyConfiguration actual = ProxyConfigurationLoader.load("/conf/empty/");
48          YamlProxyServerConfiguration serverConfig = actual.getServerConfiguration();
49          assertNull(serverConfig.getMode());
50          assertNull(serverConfig.getAuthority());
51          assertNull(serverConfig.getLabels());
52          assertTrue(serverConfig.getProps().isEmpty());
53          assertTrue(serverConfig.getRules().isEmpty());
54          assertTrue(actual.getDatabaseConfigurations().isEmpty());
55      }
56      
57      @Test
58      void assertLoad() throws IOException {
59          YamlProxyConfiguration actual = ProxyConfigurationLoader.load("/conf/config_loader/");
60          Iterator<YamlRuleConfiguration> actualGlobalRules = actual.getServerConfiguration().getRules().iterator();
61          // TODO assert mode
62          // TODO assert authority rule
63          actualGlobalRules.next();
64          assertThat(actual.getDatabaseConfigurations().size(), is(3));
65          assertShardingRuleConfiguration(actual.getDatabaseConfigurations().get("sharding_db"));
66          assertReadwriteSplittingRuleConfiguration(actual.getDatabaseConfigurations().get("readwrite_splitting_db"));
67          assertEncryptRuleConfiguration(actual.getDatabaseConfigurations().get("encrypt_db"));
68      }
69      
70      private void assertShardingRuleConfiguration(final YamlProxyDatabaseConfiguration actual) {
71          assertThat(actual.getDatabaseName(), is("sharding_db"));
72          assertThat(actual.getDataSources().size(), is(2));
73          assertDataSourceConfiguration(actual.getDataSources().get("ds_0"), "jdbc:mysql://127.0.0.1:3306/ds_0");
74          assertDataSourceConfiguration(actual.getDataSources().get("ds_1"), "jdbc:mysql://127.0.0.1:3306/ds_1");
75          Optional<YamlShardingRuleConfiguration> shardingRuleConfig = actual.getRules().stream()
76                  .filter(each -> each instanceof YamlShardingRuleConfiguration).findFirst().map(each -> (YamlShardingRuleConfiguration) each);
77          assertTrue(shardingRuleConfig.isPresent());
78          assertShardingRuleConfiguration(shardingRuleConfig.get());
79          assertFalse(actual.getRules().stream().filter(each -> each instanceof YamlEncryptRuleConfiguration).findFirst().map(each -> (YamlEncryptRuleConfiguration) each).isPresent());
80      }
81      
82      private void assertShardingRuleConfiguration(final YamlShardingRuleConfiguration actual) {
83          assertThat(actual.getTables().size(), is(1));
84          assertThat(actual.getTables().get("t_order").getActualDataNodes(), is("ds_${0..1}.t_order_${0..1}"));
85          assertThat(actual.getTables().get("t_order").getDatabaseStrategy().getStandard().getShardingColumn(), is("user_id"));
86          assertThat(actual.getTables().get("t_order").getDatabaseStrategy().getStandard().getShardingAlgorithmName(), is("database_inline"));
87          assertThat(actual.getTables().get("t_order").getTableStrategy().getStandard().getShardingColumn(), is("order_id"));
88          assertThat(actual.getTables().get("t_order").getTableStrategy().getStandard().getShardingAlgorithmName(), is("table_inline"));
89          assertNotNull(actual.getDefaultDatabaseStrategy().getNone());
90      }
91      
92      private void assertReadwriteSplittingRuleConfiguration(final YamlProxyDatabaseConfiguration actual) {
93          assertThat(actual.getDatabaseName(), is("readwrite_splitting_db"));
94          assertThat(actual.getDataSources().size(), is(3));
95          assertDataSourceConfiguration(actual.getDataSources().get("write_ds"), "jdbc:mysql://127.0.0.1:3306/write_ds");
96          assertDataSourceConfiguration(actual.getDataSources().get("read_ds_0"), "jdbc:mysql://127.0.0.1:3306/read_ds_0");
97          assertDataSourceConfiguration(actual.getDataSources().get("read_ds_1"), "jdbc:mysql://127.0.0.1:3306/read_ds_1");
98          assertFalse(actual.getRules().stream().filter(each -> each instanceof YamlShardingRuleConfiguration).findFirst().map(each -> (YamlShardingRuleConfiguration) each).isPresent());
99          assertFalse(actual.getRules().stream().filter(each -> each instanceof YamlEncryptRuleConfiguration).findFirst().map(each -> (YamlEncryptRuleConfiguration) each).isPresent());
100         Optional<YamlReadwriteSplittingRuleConfiguration> ruleConfig = actual.getRules().stream()
101                 .filter(each -> each instanceof YamlReadwriteSplittingRuleConfiguration).findFirst().map(each -> (YamlReadwriteSplittingRuleConfiguration) each);
102         assertTrue(ruleConfig.isPresent());
103         for (YamlReadwriteSplittingDataSourceRuleConfiguration each : ruleConfig.get().getDataSources().values()) {
104             assertReadwriteSplittingRuleConfiguration(each);
105         }
106     }
107     
108     private void assertReadwriteSplittingRuleConfiguration(final YamlReadwriteSplittingDataSourceRuleConfiguration actual) {
109         assertNotNull(actual.getStaticStrategy());
110         assertThat(actual.getStaticStrategy().getWriteDataSourceName(), is("write_ds"));
111         assertThat(actual.getStaticStrategy().getReadDataSourceNames(), is(Arrays.asList("read_ds_0", "read_ds_1")));
112     }
113     
114     private void assertEncryptRuleConfiguration(final YamlProxyDatabaseConfiguration actual) {
115         assertThat(actual.getDatabaseName(), is("encrypt_db"));
116         assertThat(actual.getDataSources().size(), is(1));
117         assertDataSourceConfiguration(actual.getDataSources().get("ds_0"), "jdbc:mysql://127.0.0.1:3306/encrypt_ds");
118         assertFalse(actual.getRules().stream()
119                 .filter(each -> each instanceof YamlShardingRuleConfiguration).findFirst().map(each -> (YamlShardingRuleConfiguration) each).isPresent());
120         Optional<YamlEncryptRuleConfiguration> encryptRuleConfig = actual.getRules().stream()
121                 .filter(each -> each instanceof YamlEncryptRuleConfiguration).findFirst().map(each -> (YamlEncryptRuleConfiguration) each);
122         assertTrue(encryptRuleConfig.isPresent());
123         assertEncryptRuleConfiguration(encryptRuleConfig.get());
124     }
125     
126     private void assertEncryptRuleConfiguration(final YamlEncryptRuleConfiguration actual) {
127         assertThat(actual.getEncryptors().size(), is(2));
128         assertTrue(actual.getEncryptors().containsKey("aes_encryptor"));
129         assertTrue(actual.getEncryptors().containsKey("md5_encryptor"));
130         YamlAlgorithmConfiguration aesEncryptAlgorithmConfig = actual.getEncryptors().get("aes_encryptor");
131         assertThat(aesEncryptAlgorithmConfig.getType(), is("AES"));
132         assertThat(aesEncryptAlgorithmConfig.getProps().getProperty("aes-key-value"), is("123456abc"));
133         YamlAlgorithmConfiguration md5EncryptAlgorithmConfig = actual.getEncryptors().get("md5_encryptor");
134         assertThat(md5EncryptAlgorithmConfig.getType(), is("MD5"));
135     }
136     
137     private void assertDataSourceConfiguration(final YamlProxyDataSourceConfiguration actual, final String expectedURL) {
138         assertThat(actual.getUrl(), is(expectedURL));
139         assertThat(actual.getUsername(), is("root"));
140         assertNull(actual.getPassword());
141         assertThat(actual.getConnectionTimeoutMilliseconds(), is(30000L));
142         assertThat(actual.getIdleTimeoutMilliseconds(), is(60000L));
143         assertThat(actual.getMaxLifetimeMilliseconds(), is(1800000L));
144         assertThat(actual.getMaxPoolSize(), is(50));
145     }
