18  package org.apache.shardingsphere.encrypt.algorithm.standard;
19  
20  import lombok.EqualsAndHashCode;
21  import lombok.Getter;
22  import lombok.SneakyThrows;
23  import org.apache.commons.codec.digest.DigestUtils;
24  import org.apache.commons.codec.digest.MessageDigestAlgorithms;
25  import org.apache.shardingsphere.encrypt.spi.EncryptAlgorithm;
26  import org.apache.shardingsphere.encrypt.spi.EncryptAlgorithmMetaData;
27  import org.apache.shardingsphere.infra.algorithm.core.context.AlgorithmSQLContext;
28  import org.apache.shardingsphere.infra.algorithm.core.exception.AlgorithmInitializationException;
29  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
30  
31  import javax.crypto.Cipher;
32  import javax.crypto.spec.SecretKeySpec;
33  import java.nio.charset.StandardCharsets;
34  import java.security.GeneralSecurityException;
35  import java.util.Arrays;
36  import java.util.Base64;
37  import java.util.Properties;
38  
39  /**
40   * AES encrypt algorithm.
41   */
42  @EqualsAndHashCode
43  public final class AESEncryptAlgorithm implements EncryptAlgorithm {
44      
45      private static final String AES_KEY = "aes-key-value";
46      
47      private static final String DIGEST_ALGORITHM_NAME = "digest-algorithm-name";
48      
49      @Getter
50      private final EncryptAlgorithmMetaData metaData = new EncryptAlgorithmMetaData(true, true, false);
51      
52      private byte[] secretKey;
53      
54      @Override
55      public void init(final Properties props) {
56          secretKey = getSecretKey(props);
57      }
58      
59      private byte[] getSecretKey(final Properties props) {
60          String aesKey = props.getProperty(AES_KEY);
61          ShardingSpherePreconditions.checkNotEmpty(aesKey, () -> new AlgorithmInitializationException(this, "%s can not be null or empty", AES_KEY));
62          String digestAlgorithm = props.getProperty(DIGEST_ALGORITHM_NAME, MessageDigestAlgorithms.SHA_1);
63          return Arrays.copyOf(DigestUtils.getDigest(digestAlgorithm.toUpperCase()).digest(aesKey.getBytes(StandardCharsets.UTF_8)), 16);
64      }
65      
66      @SneakyThrows(GeneralSecurityException.class)
67      @Override
68      public String encrypt(final Object plainValue, final AlgorithmSQLContext algorithmSQLContext) {
69          if (null == plainValue) {
70              return null;
71          }
72          byte[] result = getCipher(Cipher.ENCRYPT_MODE).doFinal(String.valueOf(plainValue).getBytes(StandardCharsets.UTF_8));
73          return Base64.getEncoder().encodeToString(result);
74      }
75      
76      @SneakyThrows(GeneralSecurityException.class)
77      @Override
78      public Object decrypt(final Object cipherValue, final AlgorithmSQLContext algorithmSQLContext) {
79          if (null == cipherValue) {
80              return null;
81          }
82          byte[] result = getCipher(Cipher.DECRYPT_MODE).doFinal(Base64.getDecoder().decode(cipherValue.toString().trim()));
83          return new String(result, StandardCharsets.UTF_8);
84      }
85      
86      private Cipher getCipher(final int decryptMode) throws GeneralSecurityException {
87          Cipher result = Cipher.getInstance(getType());
88          result.init(decryptMode, new SecretKeySpec(secretKey, getType()));
89          return result;
90      }
91      
92      @Override
93      public String getType() {
94          return "AES";
95      }
96  }
