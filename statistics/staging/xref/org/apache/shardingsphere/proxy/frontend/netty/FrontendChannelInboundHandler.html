18  package org.apache.shardingsphere.proxy.frontend.netty;
19  
20  import io.netty.buffer.ByteBuf;
21  import io.netty.channel.Channel;
22  import io.netty.channel.ChannelHandlerContext;
23  import io.netty.channel.ChannelInboundHandlerAdapter;
24  import lombok.extern.slf4j.Slf4j;
25  import org.apache.shardingsphere.db.protocol.constant.CommonConstants;
26  import org.apache.shardingsphere.infra.executor.sql.process.ProcessEngine;
27  import org.apache.shardingsphere.infra.metadata.user.Grantee;
28  import org.apache.shardingsphere.proxy.backend.session.ConnectionSession;
29  import org.apache.shardingsphere.authentication.result.AuthenticationResult;
30  import org.apache.shardingsphere.proxy.frontend.exception.ExpectedExceptions;
31  import org.apache.shardingsphere.proxy.frontend.executor.ConnectionThreadExecutorGroup;
32  import org.apache.shardingsphere.proxy.frontend.executor.UserExecutorGroup;
33  import org.apache.shardingsphere.proxy.frontend.spi.DatabaseProtocolFrontendEngine;
34  import org.apache.shardingsphere.proxy.frontend.state.ProxyStateContext;
35  
36  import java.util.Optional;
37  
38  /**
39   * Frontend channel inbound handler.
40   */
41  @Slf4j
42  public final class FrontendChannelInboundHandler extends ChannelInboundHandlerAdapter {
43      
44      private final DatabaseProtocolFrontendEngine databaseProtocolFrontendEngine;
45      
46      private final ConnectionSession connectionSession;
47      
48      private final ProcessEngine processEngine = new ProcessEngine();
49      
50      private volatile boolean authenticated;
51      
52      public FrontendChannelInboundHandler(final DatabaseProtocolFrontendEngine databaseProtocolFrontendEngine, final Channel channel) {
53          this.databaseProtocolFrontendEngine = databaseProtocolFrontendEngine;
54          connectionSession = new ConnectionSession(databaseProtocolFrontendEngine.getType(), channel);
55      }
56      
57      @Override
58      public void channelActive(final ChannelHandlerContext context) {
59          int connectionId = databaseProtocolFrontendEngine.getAuthenticationEngine().handshake(context);
60          ConnectionThreadExecutorGroup.getInstance().register(connectionId);
61          connectionSession.setConnectionId(connectionId);
62      }
63      
64      @Override
65      public void channelRead(final ChannelHandlerContext context, final Object message) {
66          if (!authenticated) {
67              authenticated = authenticate(context, (ByteBuf) message);
68              return;
69          }
70          ProxyStateContext.execute(context, message, databaseProtocolFrontendEngine, connectionSession);
71      }
72      
73      private boolean authenticate(final ChannelHandlerContext context, final ByteBuf message) {
74          try {
75              AuthenticationResult authResult = databaseProtocolFrontendEngine.getAuthenticationEngine().authenticate(context,
76                      databaseProtocolFrontendEngine.getCodecEngine().createPacketPayload(message, context.channel().attr(CommonConstants.CHARSET_ATTRIBUTE_KEY).get()));
77              if (authResult.isFinished()) {
78                  connectionSession.setGrantee(new Grantee(authResult.getUsername(), authResult.getHostname()));
79                  connectionSession.setCurrentDatabase(authResult.getDatabase());
80                  connectionSession.setProcessId(processEngine.connect(connectionSession.getGrantee(), connectionSession.getDatabaseName()));
81              }
82              return authResult.isFinished();
83              // CHECKSTYLE:OFF
84          } catch (final Exception ex) {
85              // CHECKSTYLE:ON
86              if (ExpectedExceptions.isExpected(ex.getClass())) {
87                  log.debug("Exception occur: ", ex);
88              } else {
89                  log.error("Exception occur: ", ex);
90              }
91              context.writeAndFlush(databaseProtocolFrontendEngine.getCommandExecuteEngine().getErrorPacket(ex));
92              context.close();
93          } finally {
94              message.release();
95          }
96          return false;
97      }
98      
99      @Override
100     public void channelInactive(final ChannelHandlerContext context) {
101         context.fireChannelInactive();
102         UserExecutorGroup.getInstance().getExecutorService().execute(this::closeAllResources);
103     }
104     
105     private void closeAllResources() {
106         ConnectionThreadExecutorGroup.getInstance().unregisterAndAwaitTermination(connectionSession.getConnectionId());
107         connectionSession.getDatabaseConnectionManager().closeAllResources();
108         Optional.ofNullable(connectionSession.getProcessId()).ifPresent(processEngine::disconnect);
109         databaseProtocolFrontendEngine.release(connectionSession);
110     }
111     
112     @Override
113     public void channelWritabilityChanged(final ChannelHandlerContext context) {
114         if (context.channel().isWritable()) {
115             connectionSession.getDatabaseConnectionManager().getResourceLock().doNotify();
116         }
117     }
118 }
