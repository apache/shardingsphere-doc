65      @Override
66      public QuoteCharacter getQuoteCharacter() {
67          return QuoteCharacter.BACK_QUOTE;
68      }
69      
70      @Override
71      public boolean isReservedWord(final String item) {
72          return RESERVED_WORDS.contains(item.toUpperCase());
73      }
74      
75      @Override
76      public Collection<String> getJdbcUrlPrefixes() {
77          return Arrays.asList("jdbc:mysql:", "jdbc:mysqlx:");
78      }
79      
80      @Override
81      public MySQLDataSourceMetaData getDataSourceMetaData(final String url, final String username) {

83      }
84      
85      @Override
86      public Map<String, Collection<String>> getSystemDatabaseSchemaMap() {

89      
90      @Override

95      @Override
