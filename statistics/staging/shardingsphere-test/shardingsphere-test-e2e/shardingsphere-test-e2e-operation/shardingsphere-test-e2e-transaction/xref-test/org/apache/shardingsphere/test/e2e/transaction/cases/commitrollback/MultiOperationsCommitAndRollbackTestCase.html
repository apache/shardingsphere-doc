36  /**
37   * Integration test of multiple operations in one transaction.
38   */
39  @TransactionTestCase
40  public final class MultiOperationsCommitAndRollbackTestCase extends BaseTransactionTestCase {
41      
42      public MultiOperationsCommitAndRollbackTestCase(final TransactionBaseE2EIT baseTransactionITCase, final DataSource dataSource) {
43          super(baseTransactionITCase, dataSource);
44      }
45      
46      @Override
47      public void executeTest(final TransactionContainerComposer containerComposer) throws SQLException {
48          assertRollback();
49          assertCommit();
50      }
51      
52      private void assertRollback() throws SQLException {
53          try (Connection connection = getDataSource().getConnection()) {
54              connection.setAutoCommit(false);
55              assertTableRowCount(connection, TransactionTestConstants.ACCOUNT, 0);
56              executeWithLog(connection, "insert into account(id, balance, transaction_id) values(1, 1, 1);");
57              executeWithLog(connection, "insert into account(id, balance, transaction_id) values(2, 2, 2);");
58              executeUpdateWithLog(connection, "update account set balance = 3, transaction_id = 3 where id = 2;");
59              assertQueryAccount(connection, 1, 3);
60              connection.rollback();
61          }
62          try (Connection connection = getDataSource().getConnection()) {
63              assertTableRowCount(connection, TransactionTestConstants.ACCOUNT, 0);
64              assertQueryAccount(connection, 1, 2);
65          }
66      }
67      
68      private void assertCommit() throws SQLException {
69          try (Connection connection = getDataSource().getConnection()) {
70              connection.setAutoCommit(false);
71              assertTableRowCount(connection, TransactionTestConstants.ACCOUNT, 0);
72              executeWithLog(connection, "insert into account(id, balance, transaction_id) values(1, 1, 1);");
73              executeWithLog(connection, "insert into account(id, balance, transaction_id) values(2, 2, 2);");
74              executeUpdateWithLog(connection, "update account set balance = 3, transaction_id = 3 where id = 2;");
75              assertQueryAccount(connection, 1, 3);
76              connection.commit();
77          }
78          try (Connection connection = getDataSource().getConnection()) {
79              assertTableRowCount(connection, TransactionTestConstants.ACCOUNT, 2);
80              assertQueryAccount(connection, 1, 3);
81          }
82      }
83      
84      private void assertQueryAccount(final Connection connection, final int... expectedBalances) throws SQLException {
85          Preconditions.checkArgument(2 == expectedBalances.length);
86          Statement queryStatement = connection.createStatement();
87          ResultSet resultSet = queryStatement.executeQuery("select * from account;");
88          while (resultSet.next()) {
89              int id = resultSet.getInt("id");
90              int actualBalance = resultSet.getInt("balance");
91              if (1 == id) {
92                  assertBalance(actualBalance, expectedBalances[0]);
93              }
94              if (2 == id) {
95                  assertBalance(actualBalance, expectedBalances[1]);
96              }
97          }
98      }
99      
100     private void assertBalance(final int actual, final int expected) {
101         assertThat(String.format("Balance is %s, should be %s.", actual, expected), actual, is(expected));
102     }
103 }
