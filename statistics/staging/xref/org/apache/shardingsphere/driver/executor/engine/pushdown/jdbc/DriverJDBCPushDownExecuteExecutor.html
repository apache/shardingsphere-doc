59  /**
60   * Driver JDBC push down execute executor.
61   */
62  @RequiredArgsConstructor
63  public final class DriverJDBCPushDownExecuteExecutor {
64      
65      private final ShardingSphereConnection connection;
66      
67      private final ShardingSphereMetaData metaData;
68      
69      private final JDBCExecutor jdbcExecutor;
70      
71      /**
72       * Execute.
73       * 
74       * @param database database
75       * @param executionContext execution context
76       * @param prepareEngine prepare engine
77       * @param executeCallback execute callback
78       * @param addCallback add callback
79       * @param replayCallback replay callback
80       * @return execute result
81       * @throws SQLException SQL exception
82       */
83      @SuppressWarnings("rawtypes")
84      public boolean execute(final ShardingSphereDatabase database, final ExecutionContext executionContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
85                             final StatementExecuteCallback executeCallback, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
86          return new DriverTransactionalExecutor(connection).execute(
87                  database, executionContext, () -> doExecute(database, executionContext, prepareEngine, executeCallback, addCallback, replayCallback));
88      }
89      
90      @SuppressWarnings({"rawtypes", "unchecked"})
91      private boolean doExecute(final ShardingSphereDatabase database, final ExecutionContext executionContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
92                                final StatementExecuteCallback executeCallback, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
93          ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext = prepareEngine.prepare(database.getName(), executionContext.getRouteContext(), executionContext.getExecutionUnits(),
94                  new ExecutionGroupReportContext(connection.getProcessId(), database.getName(), connection.getDatabaseConnectionManager().getConnectionContext().getGrantee()));
95          for (ExecutionGroup<JDBCExecutionUnit> each : executionGroupContext.getInputGroups()) {
96              addCallback.add(getStatements(each), JDBCDriverType.PREPARED_STATEMENT.equals(prepareEngine.getType()) ? getParameterSets(each) : Collections.emptyList());
97          }
98          replayCallback.replay();
99          ProcessEngine processEngine = new ProcessEngine();
100         try {
101             processEngine.executeSQL(executionGroupContext, executionContext.getQueryContext());
102             List<Boolean> results = jdbcExecutor.execute(executionGroupContext,
103                     new ExecuteCallbackFactory(prepareEngine.getType()).newInstance(database, executeCallback, executionContext.getSqlStatementContext().getSqlStatement()));
104             new MetaDataRefreshEngine(connection.getContextManager().getPersistServiceFacade().getMetaDataManagerPersistService(), database, metaData.getProps())
105                     .refresh(executionContext.getQueryContext().getSqlStatementContext(), executionContext.getRouteContext().getRouteUnits());
106             return null != results && !results.isEmpty() && null != results.get(0) && results.get(0);
107         } finally {
108             processEngine.completeSQLExecution(executionGroupContext.getReportContext().getProcessId());
109         }
110     }
111     
112     private Collection<Statement> getStatements(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
113         Collection<Statement> result = new LinkedList<>();
114         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
115             result.add(each.getStorageResource());
116         }
117         return result;
118     }
119     
120     private Collection<List<Object>> getParameterSets(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
121         Collection<List<Object>> result = new LinkedList<>();
122         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
123             result.add(each.getExecutionUnit().getSqlUnit().getParameters());
124         }
125         return result;
126     }
127     
128     /**
129      * Get result set.
130      *
131      * @param database database
132      * @param sqlStatementContext SQL statement context
133      * @param statement statement
134      * @param statements statements
135      * @return result set
136      * @throws SQLException SQL exception
137      */
138     public Optional<ResultSet> getResultSet(final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext,
139                                             final Statement statement, final List<? extends Statement> statements) throws SQLException {
140         if (sqlStatementContext instanceof SelectStatementContext || sqlStatementContext.getSqlStatement() instanceof DALStatement) {
141             List<ResultSet> resultSets = getResultSets(statements);
142             if (resultSets.isEmpty()) {
143                 return Optional.empty();
144             }
145             List<QueryResult> queryResults = getQueryResults(resultSets);
146             MergedResult mergedResult = new MergeEngine(metaData.getGlobalRuleMetaData(), database, metaData.getProps(), connection.getDatabaseConnectionManager().getConnectionContext())
147                     .merge(queryResults, sqlStatementContext);
148             return Optional.of(new ShardingSphereResultSet(resultSets, mergedResult, statement, sqlStatementContext));
149         }
150         return Optional.empty();
151     }
152     
153     @SuppressWarnings("JDBCResourceOpenedButNotSafelyClosed")
154     private List<ResultSet> getResultSets(final List<? extends Statement> statements) throws SQLException {
155         List<ResultSet> result = new ArrayList<>(statements.size());
156         for (Statement each : statements) {
157             if (null != each.getResultSet()) {
158                 result.add(each.getResultSet());
159             }
160         }
161         return result;
162     }
163     
164     private List<QueryResult> getQueryResults(final List<ResultSet> resultSets) throws SQLException {
165         List<QueryResult> result = new ArrayList<>(resultSets.size());
166         for (ResultSet each : resultSets) {
167             if (null != each) {
168                 result.add(new JDBCStreamQueryResult(each));
169             }
170         }
171         return result;
172     }
