21  import org.apache.shardingsphere.data.pipeline.api.datasource.config.PipelineDataSourceConfiguration;

23  import org.apache.shardingsphere.data.pipeline.common.config.CreateTableConfiguration.CreateTableEntry;
24  import org.apache.shardingsphere.data.pipeline.common.datasource.PipelineDataSourceManager;
25  import org.apache.shardingsphere.data.pipeline.common.datasource.PipelineDataSourceWrapper;
26  import org.apache.shardingsphere.data.pipeline.common.metadata.generator.PipelineDDLGenerator;
27  import org.apache.shardingsphere.data.pipeline.common.sqlbuilder.PipelineCommonSQLBuilder;
28  import org.apache.shardingsphere.infra.database.type.DatabaseType;

31  import javax.sql.DataSource;
32  import java.sql.Connection;
33  import java.sql.SQLException;
34  import java.sql.Statement;
35  import java.util.Collection;
36  import java.util.HashSet;
37  import java.util.Optional;
38  import java.util.regex.Pattern;
39  
40  /**
41   * Abstract data source preparer.
42   */
43  @Slf4j
44  public abstract class AbstractDataSourcePreparer implements DataSourcePreparer {
45      
46      private static final Pattern PATTERN_CREATE_TABLE_IF_NOT_EXISTS = Pattern.compile("CREATE\\s+TABLE\\s+IF\\s+NOT\\s+EXISTS\\s+", Pattern.CASE_INSENSITIVE);
47      
48      private static final Pattern PATTERN_CREATE_TABLE = Pattern.compile("CREATE\\s+TABLE\\s+", Pattern.CASE_INSENSITIVE);
49      
50      @Override
51      public void prepareTargetSchemas(final PrepareTargetSchemasParameter param) throws SQLException {
52          DatabaseType targetDatabaseType = param.getTargetDatabaseType();
53          if (!targetDatabaseType.isSchemaAvailable()) {
54              return;
55          }
56          CreateTableConfiguration createTableConfig = param.getCreateTableConfig();
57          String defaultSchema = targetDatabaseType.getDefaultSchema().orElse(null);
58          PipelineCommonSQLBuilder pipelineSQLBuilder = new PipelineCommonSQLBuilder(targetDatabaseType);
59          Collection<String> createdSchemaNames = new HashSet<>();
60          for (CreateTableEntry each : createTableConfig.getCreateTableEntries()) {
61              String targetSchemaName = each.getTargetName().getSchemaName().getOriginal();
62              if (null == targetSchemaName || targetSchemaName.equalsIgnoreCase(defaultSchema) || createdSchemaNames.contains(targetSchemaName)) {
63                  continue;
64              }
65              Optional<String> sql = pipelineSQLBuilder.buildCreateSchemaSQL(targetSchemaName);
66              if (sql.isPresent()) {
67                  executeCreateSchema(param.getDataSourceManager(), each.getTargetDataSourceConfig(), sql.get());
68                  createdSchemaNames.add(targetSchemaName);
69              }
70          }
71      }
72      
73      private void executeCreateSchema(final PipelineDataSourceManager dataSourceManager, final PipelineDataSourceConfiguration targetDataSourceConfig, final String sql) throws SQLException {
74          log.info("prepareTargetSchemas, sql={}", sql);
75          try (Connection connection = getCachedDataSource(dataSourceManager, targetDataSourceConfig).getConnection()) {
76              try (Statement statement = connection.createStatement()) {
77                  statement.execute(sql);
78              }
79          }
80      }
81      
82      protected final PipelineDataSourceWrapper getCachedDataSource(final PipelineDataSourceManager dataSourceManager, final PipelineDataSourceConfiguration dataSourceConfig) {
83          return dataSourceManager.getDataSource(dataSourceConfig);
84      }
85      
86      /**
87       * Execute target table SQL.
88       * 
89       * @param targetConnection target connection
90       * @param sql SQL
91       * @throws SQLException SQL exception
92       */
93      protected void executeTargetTableSQL(final Connection targetConnection, final String sql) throws SQLException {
94          log.info("execute target table sql: {}", sql);
95          try (Statement statement = targetConnection.createStatement()) {
96              statement.execute(sql);
97          }
98      }
99      
100     /**
101      * Add if not exists for create table SQL.
102      * 
103      * @param createTableSQL create table SQL
104      * @return create table if not existed SQL
105      */
106     protected final String addIfNotExistsForCreateTableSQL(final String createTableSQL) {
107         if (PATTERN_CREATE_TABLE_IF_NOT_EXISTS.matcher(createTableSQL).find()) {
108             return createTableSQL;
109         }
110         return PATTERN_CREATE_TABLE.matcher(createTableSQL).replaceFirst("CREATE TABLE IF NOT EXISTS ");
111     }
112     
113     protected final String getCreateTargetTableSQL(final CreateTableEntry createTableEntry, final PipelineDataSourceManager dataSourceManager,
114                                                    final SQLParserEngine sqlParserEngine) throws SQLException {
115         DatabaseType databaseType = createTableEntry.getSourceDataSourceConfig().getDatabaseType();
116         DataSource sourceDataSource = dataSourceManager.getDataSource(createTableEntry.getSourceDataSourceConfig());
117         String schemaName = createTableEntry.getSourceName().getSchemaName().getOriginal();
118         String sourceTableName = createTableEntry.getSourceName().getTableName().getOriginal();
119         String targetTableName = createTableEntry.getTargetName().getTableName().getOriginal();
120         PipelineDDLGenerator generator = new PipelineDDLGenerator();
121         return generator.generateLogicDDL(databaseType, sourceDataSource, schemaName, sourceTableName, targetTableName, sqlParserEngine);
122     }
123 }
