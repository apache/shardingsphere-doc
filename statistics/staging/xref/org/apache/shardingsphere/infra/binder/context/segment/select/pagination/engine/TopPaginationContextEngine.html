34  import java.util.Collection;
35  import java.util.List;
36  import java.util.Optional;
37  import java.util.stream.Collectors;
38  
39  /**
40   * Pagination context engine for top.
41   */
42  public final class TopPaginationContextEngine {
43      
44      /**
45       * Create pagination context.
46       *
47       * @param topProjectionSegment top projection segment
48       * @param expressions expressions
49       * @param params SQL parameters
50       * @return pagination context
51       */
52      public PaginationContext createPaginationContext(final TopProjectionSegment topProjectionSegment, final Collection<ExpressionSegment> expressions, final List<Object> params) {
53          Collection<AndPredicate> andPredicates = expressions.stream().flatMap(each -> ExpressionExtractUtils.getAndPredicates(each).stream()).collect(Collectors.toList());
54          Optional<ExpressionSegment> rowNumberPredicate = expressions.isEmpty() ? Optional.empty() : getRowNumberPredicate(andPredicates, topProjectionSegment.getAlias());
55          Optional<PaginationValueSegment> offset = rowNumberPredicate.isPresent() ? createOffsetWithRowNumber(rowNumberPredicate.get()) : Optional.empty();
56          PaginationValueSegment rowCount = topProjectionSegment.getTop();
57          return new PaginationContext(offset.orElse(null), rowCount, params);
58      }
59      
60      private Optional<ExpressionSegment> getRowNumberPredicate(final Collection<AndPredicate> andPredicates, final String rowNumberAlias) {
61          for (AndPredicate each : andPredicates) {
62              for (ExpressionSegment expression : each.getPredicates()) {
63                  if (isRowNumberColumn(expression, rowNumberAlias) && isCompareCondition(expression)) {
64                      return Optional.of(expression);
65                  }
66              }
67          }
68          return Optional.empty();
69      }
70      
71      private boolean isRowNumberColumn(final ExpressionSegment predicate, final String rowNumberAlias) {
72          if (predicate instanceof BinaryOperationExpression) {
73              ExpressionSegment left = ((BinaryOperationExpression) predicate).getLeft();
74              return left instanceof ColumnSegment && ((ColumnSegment) left).getIdentifier().getValue().equalsIgnoreCase(rowNumberAlias);
75          }
76          return false;
77      }
78      
79      private boolean isCompareCondition(final ExpressionSegment predicate) {
80          if (predicate instanceof BinaryOperationExpression) {
81              String operator = ((BinaryOperationExpression) predicate).getOperator();
82              return ">".equals(operator) || ">=".equals(operator);
83          }
84          return false;
85      }
86      
87      private Optional<PaginationValueSegment> createOffsetWithRowNumber(final ExpressionSegment predicateSegment) {
88          if (!(predicateSegment instanceof BinaryOperationExpression)) {
89              return Optional.empty();
90          }
91          String operator = ((BinaryOperationExpression) predicateSegment).getOperator();
92          switch (operator) {
93              case ">":
94                  return Optional.of(createRowNumberValueSegment(((BinaryOperationExpression) predicateSegment).getRight(), false));
95              case ">=":
96                  return Optional.of(createRowNumberValueSegment(((BinaryOperationExpression) predicateSegment).getRight(), true));
97              default:
98                  return Optional.empty();
99          }
100     }
101     
102     private RowNumberValueSegment createRowNumberValueSegment(final ExpressionSegment expression, final boolean boundOpened) {
103         int startIndex = expression.getStartIndex();
104         int stopIndex = expression.getStopIndex();
105         return expression instanceof LiteralExpressionSegment
106                 ? new NumberLiteralRowNumberValueSegment(startIndex, stopIndex, Long.parseLong(((LiteralExpressionSegment) expression).getLiterals().toString()), boundOpened)
107                 : new ParameterMarkerRowNumberValueSegment(startIndex, stopIndex, ((ParameterMarkerExpressionSegment) expression).getParameterMarkerIndex(), boundOpened);
108     }
109 }
