203     @Override
204     public ASTNode visitSubquery(final SubqueryContext ctx) {
205         return visit(ctx.queryExpressionParens());
206     }
207     
208     @Override
209     public ASTNode visitQueryExpressionParens(final QueryExpressionParensContext ctx) {
210         if (null != ctx.queryExpressionParens()) {
211             return visit(ctx.queryExpressionParens());
212         }
213         HiveSelectStatement result = (HiveSelectStatement) visit(ctx.queryExpression());
214         if (null != ctx.lockClauseList()) {
215             result.setLock((LockSegment) visit(ctx.lockClauseList()));
216         }
217         result.addParameterMarkerSegments(getParameterMarkerSegments());
218         return result;
219     }
220     
221     @Override
222     public ASTNode visitLockClauseList(final LockClauseListContext ctx) {
223         LockSegment result = new LockSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
224         for (LockClauseContext each : ctx.lockClause()) {
225             if (null != each.tableLockingList()) {
226                 result.getTables().addAll(generateTablesFromTableAliasRefList(each.tableLockingList().tableAliasRefList()));
227             }
228         }
229         return result;
230     }
231     
232     @Override
233     public ASTNode visitQueryExpression(final QueryExpressionContext ctx) {
234         HiveSelectStatement result;
235         if (null != ctx.queryExpressionBody()) {
236             result = (HiveSelectStatement) visit(ctx.queryExpressionBody());
237         } else {
238             result = (HiveSelectStatement) visit(ctx.queryExpressionParens());
239         }
240         if (null != ctx.orderByClause()) {
241             result.setOrderBy((OrderBySegment) visit(ctx.orderByClause()));
242         }
243         if (null != ctx.limitClause()) {
244             result.setLimit((LimitSegment) visit(ctx.limitClause()));
245         }
246         return result;
247     }
248     
249     @Override
250     public ASTNode visitSelectWithInto(final SelectWithIntoContext ctx) {
251         if (null != ctx.selectWithInto()) {
252             return visit(ctx.selectWithInto());
253         }
254         HiveSelectStatement result = (HiveSelectStatement) visit(ctx.queryExpression());
255         if (null != ctx.lockClauseList()) {
256             result.setLock((LockSegment) visit(ctx.lockClauseList()));
257         }
258         return result;
259     }
260     
261     @Override
262     public ASTNode visitQueryExpressionBody(final QueryExpressionBodyContext ctx) {
263         if (1 == ctx.getChildCount() && ctx.getChild(0) instanceof QueryPrimaryContext) {
264             return visit(ctx.queryPrimary());
265         }
266         if (null != ctx.queryExpressionBody()) {
267             HiveSelectStatement result = new HiveSelectStatement();
268             SubquerySegment left = new SubquerySegment(ctx.queryExpressionBody().start.getStartIndex(), ctx.queryExpressionBody().stop.getStopIndex(),
269                     (HiveSelectStatement) visit(ctx.queryExpressionBody()), getOriginalText(ctx.queryExpressionBody()));
270             result.setProjections(left.getSelect().getProjections());
271             left.getSelect().getFrom().ifPresent(result::setFrom);
272             ((HiveSelectStatement) left.getSelect()).getTable().ifPresent(result::setTable);
273             result.setCombine(createCombineSegment(ctx.combineClause(), left));
274             return result;
275         }
276         if (null != ctx.queryExpressionParens()) {
277             HiveSelectStatement result = new HiveSelectStatement();
278             SubquerySegment left = new SubquerySegment(ctx.queryExpressionParens().start.getStartIndex(), ctx.queryExpressionParens().stop.getStopIndex(),
279                     (HiveSelectStatement) visit(ctx.queryExpressionParens()), getOriginalText(ctx.queryExpressionParens()));
280             result.setProjections(left.getSelect().getProjections());
281             left.getSelect().getFrom().ifPresent(result::setFrom);
282             ((HiveSelectStatement) left.getSelect()).getTable().ifPresent(result::setTable);
283             result.setCombine(createCombineSegment(ctx.combineClause(), left));
284             return result;
285         }
286         return visit(ctx.queryExpressionParens());
287     }
288     
289     private CombineSegment createCombineSegment(final CombineClauseContext ctx, final SubquerySegment left) {
290         CombineType combineType;
291         if (null != ctx.EXCEPT()) {
292             combineType = CombineType.EXCEPT;
293         } else {
294             combineType = null == ctx.combineOption() || null == ctx.combineOption().ALL() ? CombineType.UNION : CombineType.UNION_ALL;
295         }
296         ParserRuleContext ruleContext = null == ctx.queryPrimary() ? ctx.queryExpressionParens() : ctx.queryPrimary();
297         SubquerySegment right = new SubquerySegment(ruleContext.start.getStartIndex(), ruleContext.stop.getStopIndex(), (HiveSelectStatement) visit(ruleContext), getOriginalText(ruleContext));
298         return new CombineSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), left, combineType, right);
299     }
300     
301     @Override
302     public ASTNode visitQuerySpecification(final QuerySpecificationContext ctx) {
303         HiveSelectStatement result = new HiveSelectStatement();
304         result.setProjections((ProjectionsSegment) visit(ctx.projections()));
305         if (null != ctx.selectSpecification()) {
306             result.getProjections().setDistinctRow(isDistinct(ctx));
307         }
308         if (null != ctx.fromClause()) {
309             if (null != ctx.fromClause().tableReferences()) {
310                 TableSegment tableSource = (TableSegment) visit(ctx.fromClause().tableReferences());
311                 result.setFrom(tableSource);
312             }
313             if (null != ctx.fromClause().DUAL()) {
314                 TableSegment tableSource = new SimpleTableSegment(new TableNameSegment(ctx.fromClause().DUAL().getSymbol().getStartIndex(),
315                         ctx.fromClause().DUAL().getSymbol().getStopIndex(), new IdentifierValue(ctx.fromClause().DUAL().getText())));
316                 result.setFrom(tableSource);
317             }
318         }
319         if (null != ctx.whereClause()) {
320             result.setWhere((WhereSegment) visit(ctx.whereClause()));
321         }
322         if (null != ctx.groupByClause()) {
323             result.setGroupBy((GroupBySegment) visit(ctx.groupByClause()));
324         }
325         if (null != ctx.havingClause()) {
326             result.setHaving((HavingSegment) visit(ctx.havingClause()));
327         }
328         if (null != ctx.windowClause()) {
329             result.setWindow((WindowSegment) visit(ctx.windowClause()));
330         }
331         return result;
332     }

334     @Override
335     public ASTNode visitTableValueConstructor(final TableValueConstructorContext ctx) {
336         HiveSelectStatement result = new HiveSelectStatement();
337         int startIndex = ctx.getStart().getStartIndex();
338         int stopIndex = ctx.getStop().getStopIndex();
339         ValuesExpression valuesExpression = new ValuesExpression(startIndex, stopIndex);
340         valuesExpression.getRowConstructorList().addAll(createRowConstructorList(ctx.rowConstructorList()));
341         result.setProjections(new ProjectionsSegment(startIndex, stopIndex));
342         result.getProjections().getProjections().add(new ExpressionProjectionSegment(startIndex, stopIndex, getOriginalText(ctx), valuesExpression));
343         return result;
344     }
345     
346     private Collection<InsertValuesSegment> createRowConstructorList(final RowConstructorListContext ctx) {
347         Collection<InsertValuesSegment> result = new LinkedList<>();
348         for (AssignmentValuesContext each : ctx.assignmentValues()) {
349             result.add((InsertValuesSegment) visit(each));
350         }
351         return result;
352     }

354     @Override
355     public ASTNode visitTableStatement(final TableStatementContext ctx) {
356         HiveSelectStatement result = new HiveSelectStatement();
357         if (null != ctx.TABLE()) {
358             result.setFrom(new SimpleTableSegment(new TableNameSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(),
359                     new IdentifierValue(ctx.tableName().getText()))));
360         } else {
361             result.setTable((SimpleTableSegment) visit(ctx.tableName()));
362         }
363         return result;
364     }

1103     @Override
1104     public ASTNode visitProjection(final ProjectionContext ctx) {
1105         // FIXME :The stop index of project is the stop index of projection, instead of alias.
1106         if (null != ctx.qualifiedShorthand()) {
1107             return createShorthandProjection(ctx.qualifiedShorthand());
1108         }
1109         AliasSegment alias = null == ctx.alias() ? null : (AliasSegment) visit(ctx.alias());
1110         ASTNode exprProjection = visit(ctx.expr());
1111         if (exprProjection instanceof ColumnSegment) {
1112             ColumnProjectionSegment result = new ColumnProjectionSegment((ColumnSegment) exprProjection);
1113             result.setAlias(alias);
1114             return result;
1115         }
1116         if (exprProjection instanceof SubquerySegment) {
1117             SubquerySegment subquerySegment = (SubquerySegment) exprProjection;
1118             String text = ctx.start.getInputStream().getText(new Interval(subquerySegment.getStartIndex(), subquerySegment.getStopIndex()));
1119             SubqueryProjectionSegment result = new SubqueryProjectionSegment((SubquerySegment) exprProjection, text);
1120             result.setAlias(alias);
1121             return result;
1122         }
1123         if (exprProjection instanceof ExistsSubqueryExpression) {
1124             ExistsSubqueryExpression existsSubqueryExpression = (ExistsSubqueryExpression) exprProjection;
1125             String text = ctx.start.getInputStream().getText(new Interval(existsSubqueryExpression.getStartIndex(), existsSubqueryExpression.getStopIndex()));
1126             SubqueryProjectionSegment result = new SubqueryProjectionSegment(((ExistsSubqueryExpression) exprProjection).getSubquery(), text);
1127             result.setAlias(alias);
1128             return result;
1129         }
1130         return createProjection(ctx, alias, exprProjection);
1131     }

1150     private ASTNode createProjection(final ProjectionContext ctx, final AliasSegment alias, final ASTNode projection) {
1151         if (projection instanceof AggregationProjectionSegment) {
1152             ((AggregationProjectionSegment) projection).setAlias(alias);
1153             return projection;
1154         }
1155         if (projection instanceof ExpressionProjectionSegment) {
1156             ((ExpressionProjectionSegment) projection).setAlias(alias);
1157             return projection;
1158         }
1159         if (projection instanceof FunctionSegment) {
1160             FunctionSegment functionSegment = (FunctionSegment) projection;
1161             ExpressionProjectionSegment result = new ExpressionProjectionSegment(functionSegment.getStartIndex(), functionSegment.getStopIndex(), functionSegment.getText(), functionSegment);
1162             result.setAlias(alias);
1163             return result;
1164         }
1165         if (projection instanceof CommonExpressionSegment) {
1166             CommonExpressionSegment segment = (CommonExpressionSegment) projection;
1167             ExpressionProjectionSegment result = new ExpressionProjectionSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getText(), segment);
1168             result.setAlias(alias);
1169             return result;
1170         }
1171         // FIXME :For DISTINCT()
1172         if (projection instanceof ColumnSegment) {
1173             ExpressionProjectionSegment result = new ExpressionProjectionSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), getOriginalText(ctx), (ColumnSegment) projection);
1174             result.setAlias(alias);
1175             return result;
1176         }
1177         if (projection instanceof SubqueryExpressionSegment) {
1178             SubqueryExpressionSegment subqueryExpressionSegment = (SubqueryExpressionSegment) projection;
1179             String text = ctx.start.getInputStream().getText(new Interval(subqueryExpressionSegment.getStartIndex(), subqueryExpressionSegment.getStopIndex()));
1180             SubqueryProjectionSegment result = new SubqueryProjectionSegment(subqueryExpressionSegment.getSubquery(), text);
1181             result.setAlias(alias);
1182             return result;
1183         }
1184         if (projection instanceof BinaryOperationExpression) {
1185             int startIndex = ((BinaryOperationExpression) projection).getStartIndex();
1186             int stopIndex = null == alias ? ((BinaryOperationExpression) projection).getStopIndex() : alias.getStopIndex();
1187             ExpressionProjectionSegment result = new ExpressionProjectionSegment(startIndex, stopIndex, ((BinaryOperationExpression) projection).getText(), (BinaryOperationExpression) projection);
1188             result.setAlias(alias);
1189             return result;
1190         }
1191         if (projection instanceof ParameterMarkerExpressionSegment) {
1192             ParameterMarkerExpressionSegment result = (ParameterMarkerExpressionSegment) projection;
1193             result.setAlias(alias);
1194             return projection;
1195         }
1196         if (projection instanceof CaseWhenExpression) {
1197             ExpressionProjectionSegment result = new ExpressionProjectionSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), getOriginalText(ctx.expr()), (CaseWhenExpression) projection);
1198             result.setAlias(alias);
1199             return result;
1200         }
1201         if (projection instanceof VariableSegment) {
1202             ExpressionProjectionSegment result = new ExpressionProjectionSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), getOriginalText(ctx.expr()), (VariableSegment) projection);
1203             result.setAlias(alias);
1204             return result;
1205         }
1206         if (projection instanceof BetweenExpression) {
1207             ExpressionProjectionSegment result = new ExpressionProjectionSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), getOriginalText(ctx.expr()), (BetweenExpression) projection);
1208             result.setAlias(alias);
1209             return result;
1210         }
1211         if (projection instanceof InExpression) {
1212             ExpressionProjectionSegment result = new ExpressionProjectionSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), getOriginalText(ctx.expr()), (InExpression) projection);
1213             result.setAlias(alias);
1214             return result;
1215         }
1216         LiteralExpressionSegment column = (LiteralExpressionSegment) projection;
1217         ExpressionProjectionSegment result = null == alias
1218                 ? new ExpressionProjectionSegment(column.getStartIndex(), column.getStopIndex(), String.valueOf(column.getLiterals()), column)
1219                 : new ExpressionProjectionSegment(column.getStartIndex(), ctx.alias().stop.getStopIndex(), String.valueOf(column.getLiterals()), column);
1220         result.setAlias(alias);
1221         return result;
1222     }
