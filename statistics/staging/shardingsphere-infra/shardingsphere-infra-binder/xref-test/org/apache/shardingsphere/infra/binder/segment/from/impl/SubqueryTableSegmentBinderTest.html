48  class SubqueryTableSegmentBinderTest {
49      
50      @Test
51      void assertBindWithAlias() {
52          MySQLSelectStatement selectStatement = mock(MySQLSelectStatement.class);
53          when(selectStatement.getDatabaseType()).thenReturn(new MySQLDatabaseType());
54          when(selectStatement.getFrom()).thenReturn(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order"))));
55          ProjectionsSegment projectionsSegment = new ProjectionsSegment(0, 0);
56          projectionsSegment.getProjections().add(new ShorthandProjectionSegment(0, 0));
57          when(selectStatement.getProjections()).thenReturn(projectionsSegment);
58          SubqueryTableSegment subqueryTableSegment = new SubqueryTableSegment(new SubquerySegment(0, 0, selectStatement));
59          subqueryTableSegment.setAlias(new AliasSegment(0, 0, new IdentifierValue("temp")));
60          ShardingSphereMetaData metaData = createMetaData();
61          Map<String, TableSegmentBinderContext> tableBinderContexts = new CaseInsensitiveMap<>();
62          SubqueryTableSegment actual = SubqueryTableSegmentBinder.bind(subqueryTableSegment, metaData, DefaultDatabase.LOGIC_NAME, tableBinderContexts);
63          assertTrue(actual.getAlias().isPresent());
64          assertTrue(tableBinderContexts.containsKey("temp"));
65      }
66      
67      @Test
68      void assertBindWithoutAlias() {
69          MySQLSelectStatement selectStatement = mock(MySQLSelectStatement.class);
70          when(selectStatement.getDatabaseType()).thenReturn(new MySQLDatabaseType());
71          when(selectStatement.getFrom()).thenReturn(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order"))));
72          ProjectionsSegment projectionsSegment = new ProjectionsSegment(0, 0);
73          projectionsSegment.getProjections().add(new ShorthandProjectionSegment(0, 0));
74          when(selectStatement.getProjections()).thenReturn(projectionsSegment);
75          SubqueryTableSegment subqueryTableSegment = new SubqueryTableSegment(new SubquerySegment(0, 0, selectStatement));
76          ShardingSphereMetaData metaData = createMetaData();
77          Map<String, TableSegmentBinderContext> tableBinderContexts = new CaseInsensitiveMap<>();
78          SubqueryTableSegment actual = SubqueryTableSegmentBinder.bind(subqueryTableSegment, metaData, DefaultDatabase.LOGIC_NAME, tableBinderContexts);
79          assertFalse(actual.getAlias().isPresent());
80          assertTrue(tableBinderContexts.containsKey(""));
81      }
82      
83      private ShardingSphereMetaData createMetaData() {
84          ShardingSphereSchema schema = mock(ShardingSphereSchema.class, RETURNS_DEEP_STUBS);
85          when(schema.getTable("t_order").getColumnValues()).thenReturn(Arrays.asList(
86                  new ShardingSphereColumn("order_id", Types.INTEGER, true, false, false, true, false),
87                  new ShardingSphereColumn("user_id", Types.INTEGER, false, false, false, true, false),
88                  new ShardingSphereColumn("status", Types.INTEGER, false, false, false, true, false)));
89          ShardingSphereMetaData result = mock(ShardingSphereMetaData.class, RETURNS_DEEP_STUBS);
90          when(result.getDatabase(DefaultDatabase.LOGIC_NAME).getSchema(DefaultDatabase.LOGIC_NAME)).thenReturn(schema);
91          return result;
92      }
