49  /**
50   * Route SQL rewrite engine.
51   */
52  @RequiredArgsConstructor
53  public final class RouteSQLRewriteEngine {
54      
55      private final SQLTranslatorRule translatorRule;
56      
57      private final ShardingSphereDatabase database;
58      
59      private final RuleMetaData globalRuleMetaData;
60      
61      /**
62       * Rewrite SQL and parameters.
63       *
64       * @param sqlRewriteContext SQL rewrite context
65       * @param routeContext route context
66       * @param queryContext query context
67       * @return SQL rewrite result
68       */
69      public RouteSQLRewriteResult rewrite(final SQLRewriteContext sqlRewriteContext, final RouteContext routeContext, final QueryContext queryContext) {
70          Map<RouteUnit, SQLRewriteUnit> sqlRewriteUnits = new LinkedHashMap<>(routeContext.getRouteUnits().size(), 1F);
71          for (Entry<String, Collection<RouteUnit>> entry : aggregateRouteUnitGroups(routeContext.getRouteUnits()).entrySet()) {
72              Collection<RouteUnit> routeUnits = entry.getValue();
73              if (isNeedAggregateRewrite(sqlRewriteContext.getSqlStatementContext(), routeUnits)) {
74                  sqlRewriteUnits.put(routeUnits.iterator().next(), createSQLRewriteUnit(sqlRewriteContext, routeContext, routeUnits));
75              } else {
76                  addSQLRewriteUnits(sqlRewriteUnits, sqlRewriteContext, routeContext, routeUnits);
77              }
78          }
79          return new RouteSQLRewriteResult(translate(queryContext, sqlRewriteUnits));
80      }
81      
82      private SQLRewriteUnit createSQLRewriteUnit(final SQLRewriteContext sqlRewriteContext, final RouteContext routeContext, final Collection<RouteUnit> routeUnits) {
83          Collection<String> sql = new LinkedList<>();
84          List<Object> params = new LinkedList<>();
85          boolean containsDollarMarker = sqlRewriteContext.getSqlStatementContext() instanceof SelectStatementContext
86                  && ((SelectStatementContext) (sqlRewriteContext.getSqlStatementContext())).isContainsDollarParameterMarker();
87          for (RouteUnit each : routeUnits) {
88              sql.add(SQLUtils.trimSemicolon(new RouteSQLBuilder(sqlRewriteContext.getSql(), sqlRewriteContext.getSqlTokens(), each).toSQL()));
89              if (containsDollarMarker && !params.isEmpty()) {
90                  continue;
91              }
92              params.addAll(getParameters(sqlRewriteContext.getParameterBuilder(), routeContext, each));
93          }
94          return new SQLRewriteUnit(String.join(" UNION ALL ", sql), params);
95      }
96      
97      private void addSQLRewriteUnits(final Map<RouteUnit, SQLRewriteUnit> sqlRewriteUnits, final SQLRewriteContext sqlRewriteContext,
98                                      final RouteContext routeContext, final Collection<RouteUnit> routeUnits) {
99          for (RouteUnit each : routeUnits) {
100             sqlRewriteUnits.put(each, new SQLRewriteUnit(new RouteSQLBuilder(sqlRewriteContext.getSql(), sqlRewriteContext.getSqlTokens(), each).toSQL(),
101                     getParameters(sqlRewriteContext.getParameterBuilder(), routeContext, each)));
102         }
103     }
104     
105     private boolean isNeedAggregateRewrite(final SQLStatementContext sqlStatementContext, final Collection<RouteUnit> routeUnits) {
106         if (!(sqlStatementContext instanceof SelectStatementContext) || routeUnits.size() == 1) {
107             return false;
108         }
109         SelectStatementContext statementContext = (SelectStatementContext) sqlStatementContext;
110         boolean containsSubqueryJoinQuery = statementContext.isContainsSubquery() || statementContext.isContainsJoinQuery();
111         boolean containsOrderByLimitClause = !statementContext.getOrderByContext().getItems().isEmpty() || statementContext.getPaginationContext().isHasPagination();
112         boolean containsLockClause = statementContext.getSqlStatement().getLock().isPresent();
113         boolean needAggregateRewrite = !containsSubqueryJoinQuery && !containsOrderByLimitClause && !containsLockClause;
114         statementContext.setNeedAggregateRewrite(needAggregateRewrite);
115         return needAggregateRewrite;
116     }
117     
118     private Map<String, Collection<RouteUnit>> aggregateRouteUnitGroups(final Collection<RouteUnit> routeUnits) {
119         Map<String, Collection<RouteUnit>> result = new LinkedHashMap<>(routeUnits.size(), 1F);
120         for (RouteUnit each : routeUnits) {
121             String dataSourceName = each.getDataSourceMapper().getActualName();
122             result.computeIfAbsent(dataSourceName, unused -> new LinkedList<>()).add(each);
123         }
124         return result;
125     }
126     
127     private List<Object> getParameters(final ParameterBuilder paramBuilder, final RouteContext routeContext, final RouteUnit routeUnit) {
128         if (paramBuilder instanceof StandardParameterBuilder) {
129             return paramBuilder.getParameters();
130         }
131         return routeContext.getOriginalDataNodes().isEmpty()
132                 ? ((GroupedParameterBuilder) paramBuilder).getParameters()
133                 : buildRouteParameters((GroupedParameterBuilder) paramBuilder, routeContext, routeUnit);
134     }
135     
136     private List<Object> buildRouteParameters(final GroupedParameterBuilder paramBuilder, final RouteContext routeContext, final RouteUnit routeUnit) {
137         List<Object> result = new LinkedList<>();
138         int count = 0;
139         for (Collection<DataNode> each : routeContext.getOriginalDataNodes()) {
140             if (isInSameDataNode(each, routeUnit)) {
141                 result.addAll(paramBuilder.getParameters(count));
142             }
143             count++;
144         }
145         result.addAll(paramBuilder.getGenericParameterBuilder().getParameters());
146         return result;
147     }
148     
149     private boolean isInSameDataNode(final Collection<DataNode> dataNodes, final RouteUnit routeUnit) {
150         if (dataNodes.isEmpty()) {
151             return true;
152         }
153         for (DataNode each : dataNodes) {
154             if (routeUnit.findTableMapper(each.getDataSourceName(), each.getTableName()).isPresent()) {
155                 return true;
156             }
157         }
158         return false;
159     }
160     
161     private Map<RouteUnit, SQLRewriteUnit> translate(final QueryContext queryContext, final Map<RouteUnit, SQLRewriteUnit> sqlRewriteUnits) {
162         Map<RouteUnit, SQLRewriteUnit> result = new LinkedHashMap<>(sqlRewriteUnits.size(), 1F);
163         Map<String, StorageUnit> storageUnits = database.getResourceMetaData().getStorageUnits();
164         for (Entry<RouteUnit, SQLRewriteUnit> entry : sqlRewriteUnits.entrySet()) {
165             DatabaseType storageType = storageUnits.get(entry.getKey().getDataSourceMapper().getActualName()).getStorageType();
166             SQLTranslatorContext sqlTranslatorContext = translatorRule.translate(entry.getValue().getSql(), entry.getValue().getParameters(), queryContext, storageType, database, globalRuleMetaData);
167             SQLRewriteUnit sqlRewriteUnit = new SQLRewriteUnit(sqlTranslatorContext.getSql(), sqlTranslatorContext.getParameters());
168             result.put(entry.getKey(), sqlRewriteUnit);
169         }
170         return result;
171     }
