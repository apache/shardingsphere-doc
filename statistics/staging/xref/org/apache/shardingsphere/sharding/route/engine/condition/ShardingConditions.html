47  /**
48   * Sharding conditions.
49   */
50  @Getter
51  @ToString
52  public final class ShardingConditions {
53      
54      private final List<ShardingCondition> conditions;
55      
56      private final SQLStatementContext sqlStatementContext;
57      
58      private final ShardingRule rule;
59      
60      private final boolean subqueryContainsShardingCondition;
61      
62      public ShardingConditions(final List<ShardingCondition> conditions, final SQLStatementContext sqlStatementContext, final ShardingRule rule) {
63          this.conditions = conditions;
64          this.sqlStatementContext = sqlStatementContext;
65          this.rule = rule;
66          subqueryContainsShardingCondition = isSubqueryContainsShardingCondition(conditions, sqlStatementContext);
67      }
68      
69      /**
70       * Judge sharding conditions is always false or not.
71       *
72       * @return sharding conditions is always false or not
73       */
74      public boolean isAlwaysFalse() {
75          if (conditions.isEmpty()) {
76              return false;
77          }
78          for (ShardingCondition each : conditions) {
79              if (!(each instanceof AlwaysFalseShardingCondition)) {
80                  return false;
81              }
82          }
83          return true;
84      }
85      
86      /**
87       * Merge sharding conditions.
88       */
89      public void merge() {
90          if (conditions.size() > 1) {
91              Collection<ShardingCondition> result = new LinkedList<>();
92              result.add(conditions.remove(conditions.size() - 1));
93              while (!conditions.isEmpty()) {
94                  findUniqueShardingCondition(result, conditions.remove(conditions.size() - 1)).ifPresent(result::add);
95              }
96              conditions.addAll(result);
97          }
98      }
99      
100     private Optional<ShardingCondition> findUniqueShardingCondition(final Collection<ShardingCondition> conditions, final ShardingCondition condition) {
101         for (ShardingCondition each : conditions) {
102             if (isSameShardingCondition(rule, condition, each)) {
103                 return Optional.empty();
104             }
105         }
106         return Optional.of(condition);
107     }
108     
109     /**
110      * Judge whether sharding condition need merge or not.
111      *
112      * @return whether sharding condition need merge or not
113      */
114     public boolean isNeedMerge() {
115         boolean selectContainsSubquery = sqlStatementContext instanceof SelectStatementContext && ((SelectStatementContext) sqlStatementContext).isContainsSubquery();
116         boolean insertSelectContainsSubquery = sqlStatementContext instanceof InsertStatementContext && null != ((InsertStatementContext) sqlStatementContext).getInsertSelectContext()
117                 && ((InsertStatementContext) sqlStatementContext).getInsertSelectContext().getSelectStatementContext().isContainsSubquery();
118         return (selectContainsSubquery || insertSelectContainsSubquery) && !rule.getShardingLogicTableNames(((TableAvailable) sqlStatementContext).getTablesContext().getTableNames()).isEmpty();
119     }
120     
121     private boolean isSubqueryContainsShardingCondition(final List<ShardingCondition> conditions, final SQLStatementContext sqlStatementContext) {
122         Collection<SelectStatement> selectStatements = getSelectStatements(sqlStatementContext);
123         if (selectStatements.size() > 1) {
124             Map<Integer, List<ShardingCondition>> startIndexShardingConditions = new HashMap<>(conditions.size(), 1F);
125             for (ShardingCondition each : conditions) {
126                 startIndexShardingConditions.computeIfAbsent(each.getStartIndex(), unused -> new LinkedList<>()).add(each);
127             }
128             for (SelectStatement each : selectStatements) {
129                 if (each.getFrom().isPresent() && each.getFrom().get() instanceof SubqueryTableSegment) {
130                     continue;
131                 }
132                 if (!each.getWhere().isPresent() || !startIndexShardingConditions.containsKey(each.getWhere().get().getExpr().getStartIndex())) {
133                     return false;
134                 }
135             }
136         }
137         return true;
138     }
139     
140     private Collection<SelectStatement> getSelectStatements(final SQLStatementContext sqlStatementContext) {
141         Collection<SelectStatement> result = new LinkedList<>();
142         if (sqlStatementContext instanceof SelectStatementContext) {
143             result.add(((SelectStatementContext) sqlStatementContext).getSqlStatement());
144             for (SelectStatementContext each : ((SelectStatementContext) sqlStatementContext).getSubqueryContexts().values()) {
145                 result.add(each.getSqlStatement());
146             }
147         }
148         if (sqlStatementContext instanceof InsertStatementContext && null != ((InsertStatementContext) sqlStatementContext).getInsertSelectContext()) {
149             SelectStatementContext selectStatementContext = ((InsertStatementContext) sqlStatementContext).getInsertSelectContext().getSelectStatementContext();
150             result.add(selectStatementContext.getSqlStatement());
151             for (SelectStatementContext each : selectStatementContext.getSubqueryContexts().values()) {
152                 result.add(each.getSqlStatement());
153             }
154         }
155         return result;
156     }
157     
158     /**
159      * Judge whether all sharding conditions are same or not.
160      *
161      * @return whether all sharding conditions are same or not
162      */
163     public boolean isSameShardingCondition() {
164         Collection<String> hintStrategyTables = findHintStrategyTables(sqlStatementContext);
165         return 1 == hintStrategyTables.size() || subqueryContainsShardingCondition && 1 == conditions.size();
166     }
167     
168     private boolean isSameShardingCondition(final ShardingRule shardingRule, final ShardingCondition shardingCondition1, final ShardingCondition shardingCondition2) {
169         if (shardingCondition1.getValues().size() != shardingCondition2.getValues().size()) {
170             return false;
171         }
172         for (int i = 0; i < shardingCondition1.getValues().size(); i++) {
173             ShardingConditionValue shardingValue1 = shardingCondition1.getValues().get(i);
174             ShardingConditionValue shardingValue2 = shardingCondition2.getValues().get(i);
175             if (!isSameShardingConditionValue(shardingRule, shardingValue1, shardingValue2)) {
176                 return false;
177             }
178         }
179         return true;
180     }
181     
182     private boolean isSameShardingCondition(final ShardingConditionValue shardingValue1, final ShardingConditionValue shardingValue2) {
183         return shardingValue1.getTableName().equals(shardingValue2.getTableName()) && shardingValue1.getColumnName().equals(shardingValue2.getColumnName());
184     }
185     
186     private Collection<String> findHintStrategyTables(final SQLStatementContext sqlStatementContext) {
187         Collection<String> result = new HashSet<>(((TableAvailable) sqlStatementContext).getTablesContext().getTableNames().size(), 1F);
188         for (String each : ((TableAvailable) sqlStatementContext).getTablesContext().getTableNames()) {
189             Optional<ShardingTable> shardingTable = rule.findShardingTable(each);
190             if (!shardingTable.isPresent()) {
191                 continue;
192             }
193             ShardingStrategyConfiguration databaseHintStrategy = rule.getDatabaseShardingStrategyConfiguration(shardingTable.get());
194             ShardingStrategyConfiguration tableHintStrategy = rule.getTableShardingStrategyConfiguration(shardingTable.get());
195             boolean isDatabaseTableHintStrategy = databaseHintStrategy instanceof HintShardingStrategyConfiguration && tableHintStrategy instanceof HintShardingStrategyConfiguration;
196             boolean isDatabaseHintStrategy = databaseHintStrategy instanceof HintShardingStrategyConfiguration && tableHintStrategy instanceof NoneShardingStrategyConfiguration;
197             boolean isTableHintStrategy = databaseHintStrategy instanceof NoneShardingStrategyConfiguration && tableHintStrategy instanceof HintShardingStrategyConfiguration;
198             if (isDatabaseTableHintStrategy || isDatabaseHintStrategy || isTableHintStrategy) {
199                 result.add(each);
200             }
201         }
202         return result;
203     }
204     
205     private boolean isBindingTable(final ShardingRule shardingRule, final ShardingConditionValue shardingValue1, final ShardingConditionValue shardingValue2) {
206         Optional<BindingTableRule> bindingRule = shardingRule.findBindingTableRule(shardingValue1.getTableName());
207         return bindingRule.isPresent() && bindingRule.get().hasLogicTable(shardingValue2.getTableName());
208     }
209     
210     private boolean isSameShardingConditionValue(final ShardingRule shardingRule, final ShardingConditionValue shardingValue1, final ShardingConditionValue shardingValue2) {
211         if (isBindingTable(shardingRule, shardingValue1, shardingValue2)) {
212             return true;
213         }
214         return isSameShardingCondition(shardingValue1, shardingValue2) && isSameShardingValue(shardingValue1, shardingValue2);
215     }
216     
217     @SuppressWarnings({"rawtypes", "unchecked"})
218     private boolean isSameShardingValue(final ShardingConditionValue shardingConditionValue1, final ShardingConditionValue shardingConditionValue2) {
219         if (shardingConditionValue1 instanceof ListShardingConditionValue && shardingConditionValue2 instanceof ListShardingConditionValue) {
220             return SafeNumberOperationUtils.safeCollectionEquals(
221                     ((ListShardingConditionValue) shardingConditionValue1).getValues(), ((ListShardingConditionValue) shardingConditionValue2).getValues());
222         }
223         if (shardingConditionValue1 instanceof RangeShardingConditionValue && shardingConditionValue2 instanceof RangeShardingConditionValue) {
224             return SafeNumberOperationUtils.safeRangeEquals(
225                     ((RangeShardingConditionValue) shardingConditionValue1).getValueRange(), ((RangeShardingConditionValue) shardingConditionValue2).getValueRange());
226         }
227         return false;
228     }
