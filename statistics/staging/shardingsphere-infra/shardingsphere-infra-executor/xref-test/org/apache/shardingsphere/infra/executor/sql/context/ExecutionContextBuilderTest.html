51      @Test
52      void assertBuildGenericSQLRewriteResult() {
53          String sql = "sql";
54          GenericSQLRewriteResult genericSQLRewriteResult = new GenericSQLRewriteResult(new SQLRewriteUnit(sql, Collections.singletonList("foo_param")));
55          ShardingSphereResourceMetaData resourceMetaData = mock(ShardingSphereResourceMetaData.class);
56          String firstDataSourceName = "firstDataSourceName";
57          when(resourceMetaData.getAllInstanceDataSourceNames()).thenReturn(Arrays.asList(firstDataSourceName, "lastDataSourceName"));
58          ShardingSphereRuleMetaData ruleMetaData = new ShardingSphereRuleMetaData(Collections.emptyList());
59          ShardingSphereDatabase database = new ShardingSphereDatabase(DefaultDatabase.LOGIC_NAME, mock(DatabaseType.class), resourceMetaData, ruleMetaData, buildDatabase());
60          Collection<ExecutionUnit> actual = ExecutionContextBuilder.build(database, genericSQLRewriteResult, mock(SQLStatementContext.class));
61          Collection<ExecutionUnit> expected = Collections.singletonList(new ExecutionUnit(firstDataSourceName, new SQLUnit(sql, Collections.singletonList("foo_param"))));
62          assertThat(actual, is(expected));
63      }

65      @Test
66      void assertBuildRouteSQLRewriteResult() {
67          RouteUnit routeUnit1 = new RouteUnit(new RouteMapper("logicName1", "actualName1"), Collections.singletonList(new RouteMapper("logicName1", "actualName1")));
68          SQLRewriteUnit sqlRewriteUnit1 = new SQLRewriteUnit("sql1", Collections.singletonList("parameter1"));
69          RouteUnit routeUnit2 = new RouteUnit(new RouteMapper("logicName2", "actualName2"), Collections.singletonList(new RouteMapper("logicName1", "actualName1")));
70          SQLRewriteUnit sqlRewriteUnit2 = new SQLRewriteUnit("sql2", Collections.singletonList("parameter2"));
71          Map<RouteUnit, SQLRewriteUnit> sqlRewriteUnits = new HashMap<>(2, 1F);
72          sqlRewriteUnits.put(routeUnit1, sqlRewriteUnit1);
73          sqlRewriteUnits.put(routeUnit2, sqlRewriteUnit2);
74          ShardingSphereResourceMetaData resourceMetaData = new ShardingSphereResourceMetaData("sharding_db", Collections.emptyMap());
75          ShardingSphereRuleMetaData ruleMetaData = new ShardingSphereRuleMetaData(Collections.emptyList());
76          ShardingSphereDatabase database = new ShardingSphereDatabase(DefaultDatabase.LOGIC_NAME, mock(DatabaseType.class), resourceMetaData, ruleMetaData, buildDatabase());
77          Collection<ExecutionUnit> actual = ExecutionContextBuilder.build(database, new RouteSQLRewriteResult(sqlRewriteUnits), mock(SQLStatementContext.class));
78          ExecutionUnit expectedUnit1 = new ExecutionUnit("actualName1", new SQLUnit("sql1", Collections.singletonList("parameter1")));
79          ExecutionUnit expectedUnit2 = new ExecutionUnit("actualName2", new SQLUnit("sql2", Collections.singletonList("parameter2")));
80          Collection<ExecutionUnit> expected = new LinkedHashSet<>(2, 1F);
81          expected.add(expectedUnit1);
82          expected.add(expectedUnit2);
83          assertThat(actual, is(expected));
84      }

86      @Test
87      void assertBuildRouteSQLRewriteResultWithEmptyPrimaryKeyMeta() {
88          RouteUnit routeUnit2 = new RouteUnit(new RouteMapper("logicName2", "actualName2"), Collections.singletonList(new RouteMapper("logicName2", "actualName2")));
89          SQLRewriteUnit sqlRewriteUnit2 = new SQLRewriteUnit("sql2", Collections.singletonList("parameter2"));
90          Map<RouteUnit, SQLRewriteUnit> sqlRewriteUnits = new HashMap<>(2, 1F);
91          sqlRewriteUnits.put(routeUnit2, sqlRewriteUnit2);
92          ShardingSphereResourceMetaData resourceMetaData = new ShardingSphereResourceMetaData("sharding_db", Collections.emptyMap());
93          ShardingSphereRuleMetaData ruleMetaData = new ShardingSphereRuleMetaData(Collections.emptyList());
94          ShardingSphereDatabase database = new ShardingSphereDatabase(DefaultDatabase.LOGIC_NAME, mock(DatabaseType.class), resourceMetaData, ruleMetaData, buildDatabaseWithoutPrimaryKey());
95          Collection<ExecutionUnit> actual = ExecutionContextBuilder.build(database, new RouteSQLRewriteResult(sqlRewriteUnits), mock(SQLStatementContext.class));
96          ExecutionUnit expectedUnit2 = new ExecutionUnit("actualName2", new SQLUnit("sql2", Collections.singletonList("parameter2")));
97          assertThat(actual, is(Collections.singleton(expectedUnit2)));
98      }

100     private Map<String, ShardingSphereSchema> buildDatabaseWithoutPrimaryKey() {
101         Map<String, ShardingSphereTable> tables = new HashMap<>(3, 1F);
102         tables.put("logicName1", new ShardingSphereTable("logicName1", Arrays.asList(new ShardingSphereColumn("order_id", Types.INTEGER, true, false, false, true, false),
103                 new ShardingSphereColumn("user_id", Types.INTEGER, false, false, false, true, false),
104                 new ShardingSphereColumn("status", Types.INTEGER, false, false, false, true, false)), Collections.emptySet(), Collections.emptyList()));
105         tables.put("t_other", new ShardingSphereTable("t_other", Collections.singletonList(
106                 new ShardingSphereColumn("order_id", Types.INTEGER, true, false, false, true, false)), Collections.emptySet(), Collections.emptyList()));
107         return Collections.singletonMap("name", new ShardingSphereSchema(tables, Collections.emptyMap()));
108     }

110     private Map<String, ShardingSphereSchema> buildDatabase() {
111         Map<String, ShardingSphereTable> tables = new HashMap<>(3, 1F);
112         tables.put("logicName1", new ShardingSphereTable("logicName1", Arrays.asList(new ShardingSphereColumn("order_id", Types.INTEGER, true, false, false, true, false),
113                 new ShardingSphereColumn("user_id", Types.INTEGER, false, false, false, true, false),
114                 new ShardingSphereColumn("status", Types.INTEGER, false, false, false, true, false)), Collections.emptySet(), Collections.emptyList()));
115         tables.put("logicName2", new ShardingSphereTable("logicName2", Arrays.asList(new ShardingSphereColumn("item_id", Types.INTEGER, true, false, false, true, false),
116                 new ShardingSphereColumn("order_id", Types.INTEGER, false, false, false, true, false),
117                 new ShardingSphereColumn("user_id", Types.INTEGER, false, false, false, true, false),
118                 new ShardingSphereColumn("status", Types.VARCHAR, false, false, false, true, false),
119                 new ShardingSphereColumn("c_date", Types.TIMESTAMP, false, false, false, true, false)), Collections.emptySet(), Collections.emptyList()));
120         tables.put("t_other", new ShardingSphereTable("t_other", Collections.singletonList(
121                 new ShardingSphereColumn("order_id", Types.INTEGER, true, false, false, true, false)), Collections.emptySet(), Collections.emptyList()));
122         return Collections.singletonMap("name", new ShardingSphereSchema(tables, Collections.emptyMap()));
123     }
