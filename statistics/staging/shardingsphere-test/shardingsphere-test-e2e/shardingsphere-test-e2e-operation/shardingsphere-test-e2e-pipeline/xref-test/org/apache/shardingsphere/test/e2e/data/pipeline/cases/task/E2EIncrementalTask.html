26  import org.apache.shardingsphere.sharding.spi.KeyGenerateAlgorithm;

28  import org.apache.shardingsphere.test.e2e.data.pipeline.framework.helper.PipelineCaseHelper;
29  import org.apache.shardingsphere.test.e2e.data.pipeline.util.DataSourceExecuteUtils;

32  import javax.sql.DataSource;
33  import java.math.BigDecimal;

36  import java.time.LocalDateTime;
37  import java.time.LocalTime;
38  import java.time.OffsetDateTime;
39  import java.time.Year;
40  import java.util.ArrayList;
41  import java.util.Arrays;
42  import java.util.LinkedList;
43  import java.util.List;
44  import java.util.concurrent.ThreadLocalRandom;
45  
46  @RequiredArgsConstructor
47  @Slf4j
48  public final class E2EIncrementalTask extends BaseIncrementTask {
49      
50      private static final List<String> MYSQL_COLUMN_NAMES = Arrays.asList("order_id", "user_id", "status", "t_mediumint", "t_smallint", "t_tinyint", "t_unsigned_int", "t_unsigned_mediumint",
51              "t_unsigned_smallint", "t_unsigned_tinyint", "t_float", "t_double", "t_decimal", "t_timestamp", "t_datetime", "t_date", "t_time", "t_year", "t_bit", "t_binary", "t_varbinary", "t_blob",
52              "t_mediumblob", "t_char", "t_text", "t_mediumtext", "t_enum", "t_set", "t_json");
53      
54      private static final List<String> POSTGRESQL_COLUMN_NAMES = Arrays.asList("order_id", "user_id", "status", "t_int2", "t_numeric", "t_bool", "t_bytea", "t_char", "t_varchar", "t_float",
55              "t_double", "t_json", "t_jsonb", "t_text", "t_date", "t_time", "t_timestamp", "t_timestamptz");
56      
57      private final DataSource dataSource;
58      
59      private final String orderTableName;
60      
61      private final KeyGenerateAlgorithm primaryKeyGenerateAlgorithm;
62      
63      private final DatabaseType databaseType;
64      
65      private final int loopCount;
66      
67      @Override
68      public void run() {
69          List<Object[]> orderInsertData = PipelineCaseHelper.generateOrderInsertData(databaseType, primaryKeyGenerateAlgorithm, loopCount);
70          List<Object> primaryKeys = new LinkedList<>();
71          for (Object[] each : orderInsertData) {
72              primaryKeys.add(each[0]);
73              insertOrder(each);
74          }
75          ThreadLocalRandom random = ThreadLocalRandom.current();
76          for (int i = 0; i < Math.max(1, loopCount / 3); i++) {
77              // TODO 0000-00-00 00:00:00 now will cause consistency check failed of MySQL.
78              // DataSourceUtil.execute(dataSource, String.format("UPDATE %s SET t_datetime='0000-00-00 00:00:00' WHERE order_id = ?", orderTableName)
79              updateOrderById(primaryKeys.get(random.nextInt(0, primaryKeys.size())));
80          }
81          for (int i = 0; i < Math.max(1, loopCount / 3); i++) {
82              setNullToAllFields(primaryKeys.get(random.nextInt(0, primaryKeys.size())));
83              deleteOrderById(primaryKeys.get(random.nextInt(0, primaryKeys.size())));
84          }
85          log.info("increment task runnable execute successfully.");
86      }
87      
88      private void insertOrder(final Object[] orderInsertData) {
89          String sql;
90          if (databaseType instanceof MySQLDatabaseType) {
91              sql = SQLBuilderUtils.buildInsertSQL(MYSQL_COLUMN_NAMES, orderTableName);
92          } else if (databaseType instanceof PostgreSQLDatabaseType || databaseType instanceof OpenGaussDatabaseType) {
93              sql = SQLBuilderUtils.buildInsertSQL(POSTGRESQL_COLUMN_NAMES, orderTableName);
94          } else {
95              throw new UnsupportedOperationException();
96          }
97          DataSourceExecuteUtils.execute(dataSource, sql, orderInsertData);
98      }
99      
100     private void updateOrderById(final Object orderId) {
101         ThreadLocalRandom random = ThreadLocalRandom.current();
102         int randomInt = random.nextInt(-100, 100);
103         if (databaseType instanceof MySQLDatabaseType) {
104             String sql = SQLBuilderUtils.buildUpdateSQL(ignoreShardingColumns(MYSQL_COLUMN_NAMES), orderTableName, "?");
105             log.info("update sql: {}", sql);
106             int randomUnsignedInt = random.nextInt(10, 100);
107             LocalDateTime now = LocalDateTime.now();
108             DataSourceExecuteUtils.execute(dataSource, sql, new Object[]{"中文测试", randomInt, randomInt, randomInt, randomUnsignedInt, randomUnsignedInt, randomUnsignedInt,
109                     randomUnsignedInt, 1.0F, 1.0, new BigDecimal("999"), now, now, now.toLocalDate(), now.toLocalTime(), Year.now().getValue() + 1, new byte[]{}, new byte[]{1, 2, -1, -3},
110                     "D".getBytes(), "A".getBytes(), "T".getBytes(), "E", "text", "mediumText", "3", "3", PipelineCaseHelper.generateJsonString(32, true), orderId});
111             return;
112         }
113         if (databaseType instanceof PostgreSQLDatabaseType || databaseType instanceof OpenGaussDatabaseType) {
114             String sql = SQLBuilderUtils.buildUpdateSQL(ignoreShardingColumns(POSTGRESQL_COLUMN_NAMES), orderTableName, "?");
115             log.info("update sql: {}", sql);
116             DataSourceExecuteUtils.execute(dataSource, sql, new Object[]{"中文测试", randomInt, BigDecimal.valueOf(10000), true, new byte[]{}, "char", "varchar", PipelineCaseHelper.generateFloat(),
117                     PipelineCaseHelper.generateDouble(), PipelineCaseHelper.generateJsonString(10, true), PipelineCaseHelper.generateJsonString(20, true), "text-update", LocalDate.now(),
118                     LocalTime.now(), Timestamp.valueOf(LocalDateTime.now()), OffsetDateTime.now(), orderId});
119         }
120     }
121     
122     private List<String> ignoreShardingColumns(final List<String> columnNames) {
123         return new ArrayList<>(columnNames.subList(2, columnNames.size()));
124     }
125     
126     private void deleteOrderById(final Object orderId) {
127         String sql = SQLBuilderUtils.buildDeleteSQL(orderTableName, "order_id");
128         DataSourceExecuteUtils.execute(dataSource, sql, new Object[]{orderId});
129     }
130     
131     private void setNullToAllFields(final Object orderId) {
132         if (databaseType instanceof MySQLDatabaseType) {
133             String sql = SQLBuilderUtils.buildUpdateSQL(ignoreShardingColumns(MYSQL_COLUMN_NAMES), orderTableName, "null");
134             DataSourceExecuteUtils.execute(dataSource, sql, new Object[]{orderId});
135         }
136     }
137 }
