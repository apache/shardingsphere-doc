37  import java.util.List;
38  import java.util.stream.Collectors;

40  /**
41   * SQL RUL backend handler.
42   *
43   * @param <T> type of RUL statement
44   */
45  public final class SQLRULBackendHandler<T extends RULStatement> extends RULBackendHandler<T> {
46      
47      private List<QueryHeader> queryHeaders;
48      
49      private MergedResult mergedResult;
50      
51      @Override
52      public ResponseHeader execute() throws SQLException {
53          RULExecutor<T> executor = TypedSPILoader.getService(RULExecutor.class, getSqlStatement().getClass().getName());
54          queryHeaders = createQueryHeader(executor);
55          mergedResult = createMergedResult(executor);
56          return new QueryResponseHeader(queryHeaders);
57      }
58      
59      private List<QueryHeader> createQueryHeader(final RULExecutor<T> executor) {
60          return executor.getColumnNames().stream().map(each -> new QueryHeader("", "", each, each, Types.CHAR, "CHAR", 255, 0, false, false, false, false)).collect(Collectors.toList());
61      }
62      
63      private MergedResult createMergedResult(final RULExecutor<T> executor) throws SQLException {
64          if (executor instanceof ConnectionSessionRequiredRULExecutor) {
65              ShardingSphereMetaData metaData = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData();
66              return new LocalDataMergedResult(((ConnectionSessionRequiredRULExecutor<T>) executor).getRows(metaData, getConnectionSession(), getSqlStatement()));
67          }
68          return new LocalDataMergedResult(executor.getRows(getSqlStatement()));
69      }
70      
71      @Override
72      public boolean next() throws SQLException {
73          return null != mergedResult && mergedResult.next();
74      }
75      
76      @Override
77      public QueryResponseRow getRowData() throws SQLException {
78          List<QueryResponseCell> cells = new ArrayList<>(queryHeaders.size());
79          for (int i = 0; i < queryHeaders.size(); i++) {
80              cells.add(new QueryResponseCell(queryHeaders.get(i).getColumnType(), mergedResult.getValue(i + 1, Object.class)));
81          }
82          return new QueryResponseRow(cells);
83      }
84  }
