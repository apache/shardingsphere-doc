33  import java.util.Collections;
34  import java.util.List;
35  import java.util.stream.Collectors;
36  
37  /**
38   * YAML CDC job configuration swapper.
39   */
40  public final class YamlCDCJobConfigurationSwapper implements YamlPipelineJobConfigurationSwapper<YamlCDCJobConfiguration, CDCJobConfiguration> {
41      
42      private final YamlPipelineDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlPipelineDataSourceConfigurationSwapper();
43      
44      @Override
45      public YamlCDCJobConfiguration swapToYamlConfiguration(final CDCJobConfiguration data) {
46          YamlCDCJobConfiguration result = new YamlCDCJobConfiguration();
47          result.setJobId(data.getJobId());
48          result.setDatabaseName(data.getDatabaseName());
49          result.setSchemaTableNames(data.getSchemaTableNames());
50          result.setFull(data.isFull());
51          result.setSourceDatabaseType(data.getSourceDatabaseType().getType());
52          result.setDataSourceConfiguration(dataSourceConfigSwapper.swapToYamlConfiguration(data.getDataSourceConfig()));
53          result.setTablesFirstDataNodes(null == data.getTablesFirstDataNodes() ? null : data.getTablesFirstDataNodes().marshal());
54          List<String> jobShardingDataNodes = null == data.getJobShardingDataNodes() ? null : data.getJobShardingDataNodes().stream().map(JobDataNodeLine::marshal).collect(Collectors.toList());
55          result.setJobShardingDataNodes(jobShardingDataNodes);
56          result.setDecodeWithTX(data.isDecodeWithTX());
57          result.setSinkConfig(swapToYamlSinkConfiguration(data.getSinkConfig()));
58          result.setConcurrency(data.getConcurrency());
59          result.setRetryTimes(data.getRetryTimes());
60          return result;
61      }
62      
63      private YamlSinkConfiguration swapToYamlSinkConfiguration(final SinkConfiguration sinkConfig) {
64          YamlSinkConfiguration result = new YamlSinkConfiguration();
65          result.setSinkType(sinkConfig.getSinkType().name());
66          result.setProps(sinkConfig.getProps());
67          return result;
68      }
69      
70      @Override
71      public CDCJobConfiguration swapToObject(final YamlCDCJobConfiguration yamlConfig) {
72          List<JobDataNodeLine> jobShardingDataNodes = null == yamlConfig.getJobShardingDataNodes()
73                  ? Collections.emptyList()
74                  : yamlConfig.getJobShardingDataNodes().stream().map(JobDataNodeLine::unmarshal).collect(Collectors.toList());
75          YamlSinkConfiguration yamlSinkConfig = yamlConfig.getSinkConfig();
76          SinkConfiguration sinkConfig = new SinkConfiguration(CDCSinkType.valueOf(yamlSinkConfig.getSinkType()), yamlSinkConfig.getProps());
77          JobDataNodeLine tablesFirstDataNodes = null == yamlConfig.getTablesFirstDataNodes() ? null : JobDataNodeLine.unmarshal(yamlConfig.getTablesFirstDataNodes());
78          return new CDCJobConfiguration(yamlConfig.getJobId(), yamlConfig.getDatabaseName(), yamlConfig.getSchemaTableNames(), yamlConfig.isFull(),
79                  TypedSPILoader.getService(DatabaseType.class, yamlConfig.getSourceDatabaseType()),
80                  (ShardingSpherePipelineDataSourceConfiguration) dataSourceConfigSwapper.swapToObject(yamlConfig.getDataSourceConfiguration()), tablesFirstDataNodes,
81                  jobShardingDataNodes, yamlConfig.isDecodeWithTX(), sinkConfig, yamlConfig.getConcurrency(), yamlConfig.getRetryTimes());
82      }
83      
84      @Override
85      public CDCJobConfiguration swapToObject(final String jobParam) {
86          return null == jobParam ? null : swapToObject(YamlEngine.unmarshal(jobParam, YamlCDCJobConfiguration.class, true));
87      }
88  }
