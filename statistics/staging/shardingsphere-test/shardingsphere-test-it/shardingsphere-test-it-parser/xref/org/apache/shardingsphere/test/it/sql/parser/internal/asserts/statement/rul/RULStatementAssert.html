35  /**
36   * RUL statement assert.
37   */
38  @NoArgsConstructor(access = AccessLevel.PRIVATE)
39  public final class RULStatementAssert {
40      
41      /**
42       * Assert RUL statement is correct with expected parser result.
43       *
44       * @param assertContext assert context
45       * @param actual actual RUL statement
46       * @param expected expected RUL statement test case
47       */
48      public static void assertIs(final SQLCaseAssertContext assertContext, final RULStatement actual, final SQLParserTestCase expected) {
49          if (actual instanceof PreviewStatement) {
50              PreviewStatementAssert.assertIs(assertContext, (PreviewStatement) actual, (PreviewStatementTestCase) expected);
51          } else if (actual instanceof ParseStatement) {
52              ParseStatementAssert.assertIs(assertContext, (ParseStatement) actual, (ParseStatementTestCase) expected);
53          } else if (actual instanceof FormatStatement) {
54              FormatSQLStatementAssert.assertIs(assertContext, (FormatStatement) actual, (FormatSQLStatementTestCase) expected);
55          }
56      }
57  }
