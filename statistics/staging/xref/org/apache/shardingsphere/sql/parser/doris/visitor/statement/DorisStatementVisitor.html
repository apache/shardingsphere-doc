249 /**
250  * Statement visitor for Doris.
251  */
252 @Getter(AccessLevel.PROTECTED)
253 public abstract class DorisStatementVisitor extends DorisStatementBaseVisitor<ASTNode> {
254     
255     private final Collection<ParameterMarkerSegment> parameterMarkerSegments = new LinkedList<>();
256     
257     @Override
258     public final ASTNode visitParameterMarker(final ParameterMarkerContext ctx) {
259         return new ParameterMarkerValue(parameterMarkerSegments.size(), ParameterMarkerType.QUESTION);
260     }
261     
262     @Override
263     public final ASTNode visitLiterals(final LiteralsContext ctx) {
264         if (null != ctx.stringLiterals()) {
265             return visit(ctx.stringLiterals());
266         }
267         if (null != ctx.numberLiterals()) {
268             return visit(ctx.numberLiterals());
269         }
270         if (null != ctx.temporalLiterals()) {
271             return visit(ctx.temporalLiterals());
272         }
273         if (null != ctx.hexadecimalLiterals()) {
274             return visit(ctx.hexadecimalLiterals());
275         }
276         if (null != ctx.bitValueLiterals()) {
277             return visit(ctx.bitValueLiterals());
278         }
279         if (null != ctx.booleanLiterals()) {
280             return visit(ctx.booleanLiterals());
281         }
282         if (null != ctx.nullValueLiterals()) {
283             return visit(ctx.nullValueLiterals());
284         }
285         throw new IllegalStateException("Literals must have string, number, dateTime, hex, bit, boolean or null.");
286     }
287     
288     @Override
289     public final ASTNode visitStringLiterals(final StringLiteralsContext ctx) {
290         return new StringLiteralValue(ctx.getText());
291     }
292     
293     @Override
294     public ASTNode visitString_(final String_Context ctx) {
295         return new StringLiteralValue(ctx.getText());
296     }
297     
298     @Override
299     public final ASTNode visitNumberLiterals(final NumberLiteralsContext ctx) {
300         return new NumberLiteralValue(ctx.getText());
301     }
302     
303     @Override
304     public ASTNode visitTemporalLiterals(final TemporalLiteralsContext ctx) {
305         // TODO deal with TemporalLiterals
306         return new OtherLiteralValue(ctx.getText());
307     }
308     
309     @Override
310     public final ASTNode visitHexadecimalLiterals(final HexadecimalLiteralsContext ctx) {
311         // TODO deal with hexadecimalLiterals
312         return new OtherLiteralValue(ctx.getText());
313     }
314     
315     @Override
316     public final ASTNode visitBitValueLiterals(final BitValueLiteralsContext ctx) {
317         // TODO deal with bitValueLiterals
318         return new OtherLiteralValue(ctx.getText());
319     }
320     
321     @Override
322     public final ASTNode visitBooleanLiterals(final BooleanLiteralsContext ctx) {
323         return new BooleanLiteralValue(ctx.getText());
324     }
325     
326     @Override
327     public final ASTNode visitNullValueLiterals(final NullValueLiteralsContext ctx) {
328         return new NullLiteralValue(ctx.getText());
329     }
330     
331     @Override

597     @Override
598     public final ASTNode visitSimpleExpr(final SimpleExprContext ctx) {
599         int startIndex = ctx.start.getStartIndex();
600         int stopIndex = ctx.stop.getStopIndex();
601         if (null != ctx.subquery()) {
602             SubquerySegment subquerySegment = new SubquerySegment(
603                     ctx.subquery().getStart().getStartIndex(), ctx.subquery().getStop().getStopIndex(), (DorisSelectStatement) visit(ctx.subquery()), getOriginalText(ctx.subquery()));
604             if (null == ctx.EXISTS()) {
605                 return new SubqueryExpressionSegment(subquerySegment);
606             }
607             subquerySegment.setSubqueryType(SubqueryType.EXISTS);
608             return new ExistsSubqueryExpression(startIndex, stopIndex, subquerySegment);
609         }
610         if (null != ctx.parameterMarker()) {
611             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) visit(ctx.parameterMarker());
612             ParameterMarkerExpressionSegment result = new ParameterMarkerExpressionSegment(startIndex, stopIndex, parameterMarker.getValue(), parameterMarker.getType());
613             parameterMarkerSegments.add(result);
614             return result;
615         }
616         if (null != ctx.literals()) {
617             return SQLUtils.createLiteralExpression(visit(ctx.literals()), startIndex, stopIndex, ctx.literals().start.getInputStream().getText(new Interval(startIndex, stopIndex)));
618         }
619         if (null != ctx.intervalExpression()) {
620             return visit(ctx.intervalExpression());
621         }
622         if (null != ctx.functionCall()) {
623             return visit(ctx.functionCall());
624         }
625         if (null != ctx.collateClause()) {
626             ExpressionSegment expr = null == ctx.simpleExpr() ? null : (ExpressionSegment) visit(ctx.simpleExpr(0));
627             return new CollateExpression(startIndex, stopIndex, (SimpleExpressionSegment) visit(ctx.collateClause()), expr);
628         }
629         if (null != ctx.columnRef()) {
630             return visit(ctx.columnRef());
631         }
632         if (null != ctx.matchExpression()) {
633             return visit(ctx.matchExpression());
634         }
635         if (null != ctx.notOperator()) {
636             ASTNode expression = visit(ctx.simpleExpr(0));
637             if (expression instanceof ExistsSubqueryExpression) {
638                 ((ExistsSubqueryExpression) expression).setNot(true);
639                 return expression;
640             }
641             return new NotExpression(startIndex, stopIndex, (ExpressionSegment) expression, "!".equalsIgnoreCase(ctx.notOperator().getText()));
642         }
643         if (null != ctx.LP_() && 1 == ctx.expr().size()) {
644             ASTNode result = visit(ctx.expr(0));
645             if (result instanceof ColumnSegment) {
646                 ((ColumnSegment) result).getParentheses().add(new ParenthesesSegment(ctx.LP_().getSymbol().getStartIndex(), ctx.LP_().getSymbol().getStopIndex(), true));
647                 ((ColumnSegment) result).getParentheses().add(new ParenthesesSegment(ctx.RP_().getSymbol().getStartIndex(), ctx.RP_().getSymbol().getStopIndex(), false));
648             }
649             return result;
650         }
651         if (null != ctx.VERTICAL_BAR_() && 2 == ctx.VERTICAL_BAR_().size()) {
652             ExpressionSegment left = (ExpressionSegment) visit(ctx.simpleExpr(0));
653             ExpressionSegment right = (ExpressionSegment) visit(ctx.simpleExpr(1));
654             String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
655             return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, ctx.VERTICAL_BAR_(0).getText() + ctx.VERTICAL_BAR_(1).getText(), text);
656         }
657         return visitRemainSimpleExpr(ctx);
658     }

1730     private ASTNode createProjection(final ProjectionContext ctx, final AliasSegment alias, final ExpressionSegment projection) {
1731         if (projection instanceof AggregationProjectionSegment) {
1732             ((AggregationProjectionSegment) projection).setAlias(alias);
1733             return projection;
1734         }
1735         if (projection instanceof ExpressionProjectionSegment) {
1736             ((ExpressionProjectionSegment) projection).setAlias(alias);
1737             return projection;
1738         }
1739         if (projection instanceof FunctionSegment) {
1740             FunctionSegment functionSegment = (FunctionSegment) projection;
1741             ExpressionProjectionSegment result = new ExpressionProjectionSegment(functionSegment.getStartIndex(), functionSegment.getStopIndex(), functionSegment.getText(), functionSegment);
1742             result.setAlias(alias);
1743             return result;
1744         }
1745         if (projection instanceof CommonExpressionSegment) {
1746             CommonExpressionSegment segment = (CommonExpressionSegment) projection;
1747             ExpressionProjectionSegment result = new ExpressionProjectionSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getText(), segment);
1748             result.setAlias(alias);
1749             return result;
1750         }
1751         // FIXME :For DISTINCT()
1752         if (projection instanceof ColumnSegment) {
1753             ExpressionProjectionSegment result = new ExpressionProjectionSegment(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), getOriginalText(ctx), projection);
1754             result.setAlias(alias);
1755             return result;
1756         }
1757         if (projection instanceof SubqueryExpressionSegment) {
1758             SubqueryExpressionSegment subqueryExpressionSegment = (SubqueryExpressionSegment) projection;
1759             String text = ctx.start.getInputStream().getText(new Interval(subqueryExpressionSegment.getStartIndex(), subqueryExpressionSegment.getStopIndex()));
1760             SubqueryProjectionSegment result = new SubqueryProjectionSegment(subqueryExpressionSegment.getSubquery(), text);
1761             result.setAlias(alias);
1762             return result;
1763         }
1764         if (projection instanceof BinaryOperationExpression) {
1765             int startIndex = projection.getStartIndex();
1766             int stopIndex = null == alias ? projection.getStopIndex() : alias.getStopIndex();
1767             ExpressionProjectionSegment result = new ExpressionProjectionSegment(startIndex, stopIndex, projection.getText(), projection);
1768             result.setAlias(alias);
1769             return result;
1770         }
1771         if (projection instanceof ParameterMarkerExpressionSegment) {
1772             ParameterMarkerExpressionSegment result = (ParameterMarkerExpressionSegment) projection;
1773             result.setAlias(alias);
1774             return projection;
1775         }
1776         if (projection instanceof CaseWhenExpression || projection instanceof VariableSegment || projection instanceof BetweenExpression || projection instanceof InExpression
1777                 || projection instanceof CollateExpression || projection instanceof NotExpression) {
1778             return createExpressionProjectionSegment(ctx, alias, projection);
1779         }
1780         LiteralExpressionSegment column = (LiteralExpressionSegment) projection;
1781         ExpressionProjectionSegment result = null == alias
1782                 ? new ExpressionProjectionSegment(column.getStartIndex(), column.getStopIndex(), String.valueOf(column.getLiterals()), column)
1783                 : new ExpressionProjectionSegment(column.getStartIndex(), ctx.alias().stop.getStopIndex(), String.valueOf(column.getLiterals()), column);
1784         result.setAlias(alias);
1785         return result;
1786     }

1946         if (null == ctx.limitOffset()) {
1947             return new LimitSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), null, (PaginationValueSegment) visit(ctx.limitRowCount()));
1948         }
1949         PaginationValueSegment rowCount;
1950         PaginationValueSegment offset;
1951         if (null != ctx.OFFSET()) {
1952             rowCount = (PaginationValueSegment) visit(ctx.limitRowCount());
1953             offset = (PaginationValueSegment) visit(ctx.limitOffset());
1954         } else {
1955             offset = (PaginationValueSegment) visit(ctx.limitOffset());
1956             rowCount = (PaginationValueSegment) visit(ctx.limitRowCount());
1957         }
1958         return new LimitSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), offset, rowCount);
1959     }
