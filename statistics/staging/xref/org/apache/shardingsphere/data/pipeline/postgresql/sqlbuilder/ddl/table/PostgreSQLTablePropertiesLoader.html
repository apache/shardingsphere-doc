18  package org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.ddl.table;
19  
20  import org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.ddl.PostgreSQLDDLTemplateExecutor;
21  
22  import java.sql.Connection;
23  import java.sql.SQLException;
24  import java.util.Arrays;
25  import java.util.Collection;
26  import java.util.LinkedHashMap;
27  import java.util.Map;
28  
29  /**
30   * Table properties loader for PostgreSQL.
31   */
32  public final class PostgreSQLTablePropertiesLoader {
33      
34      private final String tableName;
35      
36      private final String schemaName;
37      
38      private final PostgreSQLDDLTemplateExecutor templateExecutor;
39      
40      public PostgreSQLTablePropertiesLoader(final Connection connection, final String tableName, final String schemaName, final int majorVersion, final int minorVersion) {
41          this.tableName = tableName;
42          this.schemaName = schemaName;
43          templateExecutor = new PostgreSQLDDLTemplateExecutor(connection, majorVersion, minorVersion);
44      }
45      
46      /**
47       * Load table properties.
48       *
49       * @return loaded table properties
50       * @throws SQLException SQL exception
51       */
52      public Map<String, Object> load() throws SQLException {
53          Map<String, Object> result = new LinkedHashMap<>();
54          fetchDataBaseId(result);
55          fetchSchemaId(result);
56          fetchTableId(result);
57          fetchTableProperties(result);
58          return result;
59      }
60      
61      private void fetchDataBaseId(final Map<String, Object> context) throws SQLException {
62          Map<String, Object> params = new LinkedHashMap<>();
63          params.put("databaseName", templateExecutor.getConnection().getCatalog());
64          context.putAll(templateExecutor.executeByTemplateForSingleRow(params, "component/table/%s/get_database_id.ftl"));
65      }
66      
67      private void fetchTableId(final Map<String, Object> context) {
68          Map<String, Object> params = new LinkedHashMap<>();
69          params.put("schemaName", schemaName);
70          params.put("tableName", tableName);
71          context.putAll(templateExecutor.executeByTemplateForSingleRow(params, "component/table/%s/get_table_id.ftl"));
72      }
73      
74      private void fetchSchemaId(final Map<String, Object> context) {
75          Map<String, Object> params = new LinkedHashMap<>();
76          params.put("schemaName", schemaName);
77          context.putAll(templateExecutor.executeByTemplateForSingleRow(params, "component/table/%s/get_schema_id.ftl"));
78      }
79      
80      private void fetchTableProperties(final Map<String, Object> context) throws SQLException {
81          context.putAll(templateExecutor.executeByTemplateForSingleRow(context, "component/table/%s/properties.ftl"));
82          updateAutovacuumProperties(context);
83          checkRlspolicySupport(context);
84          templateExecutor.formatSecurityLabels(context);
85      }
86      
87      private void updateAutovacuumProperties(final Map<String, Object> context) {
88          if (null == context.get("autovacuum_enabled")) {
89              context.put("autovacuum_enabled", "x");
90          } else if (Boolean.TRUE.toString().equalsIgnoreCase(context.get("autovacuum_enabled").toString())) {
91              context.put("autovacuum_enabled", "t");
92          } else {
93              context.put("autovacuum_enabled", "f");
94          }
95          if (null == context.get("toast_autovacuum_enabled")) {
96              context.put("toast_autovacuum_enabled", "x");
97          } else if (Boolean.TRUE.toString().equalsIgnoreCase(context.get("toast_autovacuum_enabled").toString())) {
98              context.put("toast_autovacuum_enabled", "t");
99          } else {
100             context.put("toast_autovacuum_enabled", "f");
101         }
102         context.put("autovacuum_custom", anyIsTrue(Arrays.asList(
103                 context.get("autovacuum_vacuum_threshold"),
104                 context.get("autovacuum_vacuum_scale_factor"),
105                 context.get("autovacuum_analyze_threshold"),
106                 context.get("autovacuum_analyze_scale_factor"),
107                 context.get("autovacuum_vacuum_cost_delay"),
108                 context.get("autovacuum_vacuum_cost_limit"),
109                 context.get("autovacuum_freeze_min_age"),
110                 context.get("autovacuum_freeze_max_age"),
111                 context.get("autovacuum_freeze_table_age"))) || "t".equals(context.get("autovacuum_enabled")) || "f".equals(context.get("autovacuum_enabled")));
112         context.put("toast_autovacuum", anyIsTrue(Arrays.asList(
113                 context.get("toast_autovacuum_vacuum_threshold"),
114                 context.get("toast_autovacuum_vacuum_scale_factor"),
115                 context.get("toast_autovacuum_analyze_threshold"),
116                 context.get("toast_autovacuum_analyze_scale_factor"),
117                 context.get("toast_autovacuum_vacuum_cost_delay"),
118                 context.get("toast_autovacuum_vacuum_cost_limit"),
119                 context.get("toast_autovacuum_freeze_min_age"),
120                 context.get("toast_autovacuum_freeze_max_age"),
121                 context.get("toast_autovacuum_freeze_table_age"))) || "t".equals(context.get("toast_autovacuum_enabled")) || "f".equals(context.get("toast_autovacuum_enabled")));
122     }
123     
124     private void checkRlspolicySupport(final Map<String, Object> context) {
125         if (context.containsKey("rlspolicy")) {
126             if (context.get("rlspolicy") instanceof String && Boolean.TRUE.toString().equals(context.get("rlspolicy"))) {
127                 context.put("rlspolicy", true);
128             }
129             if (context.get("forcerlspolicy") instanceof String && Boolean.TRUE.toString().equals(context.get("forcerlspolicy"))) {
130                 context.put("forcerlspolicy", true);
131             }
132         }
133     }
134     
135     private boolean anyIsTrue(final Collection<Object> collection) {
136         for (Object each : collection) {
137             if (each instanceof Boolean && (Boolean) each) {
138                 return true;
139             }
140         }
141         return false;
142     }
143 }
