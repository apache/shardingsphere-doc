18  package org.apache.shardingsphere.proxy.frontend.executor;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  
23  import java.util.Map;
24  import java.util.concurrent.ConcurrentHashMap;
25  import java.util.concurrent.ExecutorService;
26  import java.util.concurrent.LinkedBlockingQueue;
27  import java.util.concurrent.ThreadPoolExecutor;
28  import java.util.concurrent.TimeUnit;
29  
30  /**
31   * Connection thread executor group.
32   *
33   * <p>
34   * Manage the thread for each connection session invoking.
35   * This ensure XA transaction framework processed by current thread id.
36   * </p>
37   */
38  @NoArgsConstructor(access = AccessLevel.PRIVATE)
39  public final class ConnectionThreadExecutorGroup {
40      
41      private static final ConnectionThreadExecutorGroup INSTANCE = new ConnectionThreadExecutorGroup();
42      
43      private final Map<Integer, ExecutorService> executorServices = new ConcurrentHashMap<>();
44      
45      /**
46       * Get connection thread executor group.
47       *
48       * @return connection thread executor group
49       */
50      public static ConnectionThreadExecutorGroup getInstance() {
51          return INSTANCE;
52      }
53      
54      /**
55       * Register connection.
56       *
57       * @param connectionId connection id
58       */
59      public void register(final int connectionId) {
60          executorServices.put(connectionId, newSingleThreadExecutorService(connectionId));
61      }
62      
63      private ExecutorService newSingleThreadExecutorService(final int connectionId) {
64          return new ThreadPoolExecutor(0, 1, 1L, TimeUnit.HOURS, new LinkedBlockingQueue<>(), runnable -> new Thread(runnable, String.format("Connection-%d-ThreadExecutor", connectionId)));
65      }
66      
67      /**
68       * Get executor service of connection.
69       *
70       * @param connectionId connection id
71       * @return executor service of current connection
72       */
73      public ExecutorService get(final int connectionId) {
74          return executorServices.get(connectionId);
75      }
76      
77      /**
78       * Unregister connection and await termination.
79       *
80       * @param connectionId connection id
81       */
82      public void unregisterAndAwaitTermination(final int connectionId) {
83          ExecutorService executorService = executorServices.remove(connectionId);
84          if (null == executorService) {
85              return;
86          }
87          executorService.shutdown();
88          try {
89              executorService.awaitTermination(Long.MAX_VALUE, TimeUnit.MILLISECONDS);
90          } catch (final InterruptedException ignored) {
91              Thread.currentThread().interrupt();
92          }
93      }
94  }
