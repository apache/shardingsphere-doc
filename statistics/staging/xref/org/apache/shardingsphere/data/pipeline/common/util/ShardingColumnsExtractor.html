18  package org.apache.shardingsphere.data.pipeline.common.util;
19  
20  import org.apache.shardingsphere.data.pipeline.api.metadata.LogicTableName;
21  import org.apache.shardingsphere.infra.yaml.config.pojo.rule.YamlRuleConfiguration;
22  import org.apache.shardingsphere.sharding.api.config.ShardingRuleConfiguration;
23  import org.apache.shardingsphere.sharding.api.config.rule.ShardingAutoTableRuleConfiguration;
24  import org.apache.shardingsphere.sharding.api.config.rule.ShardingTableRuleConfiguration;
25  import org.apache.shardingsphere.sharding.api.config.strategy.sharding.ComplexShardingStrategyConfiguration;
26  import org.apache.shardingsphere.sharding.api.config.strategy.sharding.ShardingStrategyConfiguration;
27  import org.apache.shardingsphere.sharding.api.config.strategy.sharding.StandardShardingStrategyConfiguration;
28  import org.apache.shardingsphere.sharding.yaml.swapper.ShardingRuleConfigurationConverter;
29  
30  import java.util.Arrays;
31  import java.util.Collection;
32  import java.util.Collections;
33  import java.util.HashSet;
34  import java.util.Map;
35  import java.util.Optional;
36  import java.util.Set;
37  import java.util.concurrent.ConcurrentHashMap;
38  
39  /**
40   * Sharding columns extractor.
41   */
42  public final class ShardingColumnsExtractor {
43      
44      /**
45       * Get sharding columns map.
46       *
47       * @param yamlRuleConfigs YAML rule configurations
48       * @param logicTableNames logic table names
49       * @return sharding columns map
50       */
51      public Map<LogicTableName, Set<String>> getShardingColumnsMap(final Collection<YamlRuleConfiguration> yamlRuleConfigs, final Set<LogicTableName> logicTableNames) {
52          Optional<ShardingRuleConfiguration> shardingRuleConfigOptional = ShardingRuleConfigurationConverter.findAndConvertShardingRuleConfiguration(yamlRuleConfigs);
53          if (!shardingRuleConfigOptional.isPresent()) {
54              return Collections.emptyMap();
55          }
56          ShardingRuleConfiguration shardingRuleConfig = shardingRuleConfigOptional.get();
57          Set<String> defaultDatabaseShardingColumns = extractShardingColumns(shardingRuleConfig.getDefaultDatabaseShardingStrategy());
58          Set<String> defaultTableShardingColumns = extractShardingColumns(shardingRuleConfig.getDefaultTableShardingStrategy());
59          Map<LogicTableName, Set<String>> result = new ConcurrentHashMap<>();
60          for (ShardingTableRuleConfiguration each : shardingRuleConfig.getTables()) {
61              LogicTableName logicTableName = new LogicTableName(each.getLogicTable());
62              if (!logicTableNames.contains(logicTableName)) {
63                  continue;
64              }
65              Set<String> shardingColumns = new HashSet<>();
66              shardingColumns.addAll(null == each.getDatabaseShardingStrategy() ? defaultDatabaseShardingColumns : extractShardingColumns(each.getDatabaseShardingStrategy()));
67              shardingColumns.addAll(null == each.getTableShardingStrategy() ? defaultTableShardingColumns : extractShardingColumns(each.getTableShardingStrategy()));
68              result.put(logicTableName, shardingColumns);
69          }
70          for (ShardingAutoTableRuleConfiguration each : shardingRuleConfig.getAutoTables()) {
71              LogicTableName logicTableName = new LogicTableName(each.getLogicTable());
72              if (!logicTableNames.contains(logicTableName)) {
73                  continue;
74              }
75              ShardingStrategyConfiguration shardingStrategy = each.getShardingStrategy();
76              Set<String> shardingColumns = new HashSet<>(extractShardingColumns(shardingStrategy));
77              result.put(logicTableName, shardingColumns);
78          }
79          return result;
80      }
81      
82      private Set<String> extractShardingColumns(final ShardingStrategyConfiguration shardingStrategy) {
83          if (shardingStrategy instanceof StandardShardingStrategyConfiguration) {
84              return new HashSet<>(Collections.singleton(((StandardShardingStrategyConfiguration) shardingStrategy).getShardingColumn()));
85          }
86          if (shardingStrategy instanceof ComplexShardingStrategyConfiguration) {
87              return new HashSet<>(Arrays.asList(((ComplexShardingStrategyConfiguration) shardingStrategy).getShardingColumns().split(",")));
88          }
89          return Collections.emptySet();
90      }
91  }
