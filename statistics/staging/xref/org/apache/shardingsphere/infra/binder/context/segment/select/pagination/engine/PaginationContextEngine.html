41  /**
42   * Pagination context engine.
43   */
44  @RequiredArgsConstructor
45  public final class PaginationContextEngine {
46      
47      private final DatabaseType databaseType;
48      
49      /**
50       * Create pagination context.
51       *
52       * @param selectStatement SQL statement
53       * @param projectionsContext projections context
54       * @param params SQL parameters
55       * @param whereSegments where segments
56       * @return pagination context
57       */
58      public PaginationContext createPaginationContext(final SelectStatement selectStatement, final ProjectionsContext projectionsContext,
59                                                       final List<Object> params, final Collection<WhereSegment> whereSegments) {
60          Optional<LimitSegment> limitSegment = selectStatement.getLimit();
61          if (limitSegment.isPresent()) {
62              return new LimitPaginationContextEngine().createPaginationContext(limitSegment.get(), params);
63          }
64          Optional<TopProjectionSegment> topProjectionSegment = findTopProjection(selectStatement);
65          Collection<ExpressionSegment> expressions = new LinkedList<>();
66          for (WhereSegment each : whereSegments) {
67              expressions.add(each.getExpr());
68          }
69          if (topProjectionSegment.isPresent()) {
70              return new TopPaginationContextEngine().createPaginationContext(topProjectionSegment.get(), expressions, params);
71          }
72          if (!expressions.isEmpty() && containsRowNumberPagination(selectStatement)) {
73              return new RowNumberPaginationContextEngine(databaseType).createPaginationContext(expressions, projectionsContext, params);
74          }
75          return new PaginationContext(null, null, params);
76      }
77      
78      private boolean containsRowNumberPagination(final SelectStatement selectStatement) {
79          return selectStatement instanceof OracleStatement || selectStatement instanceof SQLServerStatement;
80      }
81      
82      private Optional<TopProjectionSegment> findTopProjection(final SelectStatement selectStatement) {
83          List<SubqueryTableSegment> subqueryTableSegments = selectStatement.getFrom().map(SQLUtils::getSubqueryTableSegmentFromTableSegment).orElse(Collections.emptyList());
84          for (SubqueryTableSegment subquery : subqueryTableSegments) {
85              SelectStatement subquerySelect = subquery.getSubquery().getSelect();
86              for (ProjectionSegment each : subquerySelect.getProjections().getProjections()) {
87                  if (each instanceof TopProjectionSegment) {
88                      return Optional.of((TopProjectionSegment) each);
89                  }
90              }
91          }
92          return Optional.empty();
93      }
94  }
