50  /**
51   * SQL federation table.
52   */
53  @RequiredArgsConstructor
54  public final class SQLFederationTable extends AbstractTable implements QueryableTable, TranslatableTable {
55      
56      private final ShardingSphereTable table;
57      
58      private final SQLFederationStatistic statistic;
59      
60      private final DatabaseType protocolType;
61      
62      @Setter
63      private EnumerableScanExecutor scanExecutor;
64      
65      @Override
66      public RelDataType getRowType(final RelDataTypeFactory typeFactory) {
67          return SQLFederationDataTypeUtils.createRelDataType(table, protocolType, typeFactory);
68      }
69      
70      @Override
71      public Expression getExpression(final SchemaPlus schema, final String tableName, final Class clazz) {
72          return Schemas.tableExpression(schema, getElementType(), tableName, clazz);
73      }
74      
75      @Override
76      public Type getElementType() {
77          return Object[].class;
78      }
79      
80      @Override
81      public <T> Queryable<T> asQueryable(final QueryProvider queryProvider, final SchemaPlus schema, final String tableName) {
82          throw new UnsupportedSQLOperationException("asQueryable");
83      }
84      
85      @Override
86      public RelNode toRel(final ToRelContext context, final RelOptTable relOptTable) {
87          return LogicalTableScan.create(context.getCluster(), relOptTable, Collections.emptyList());
88      }
89      
90      /**
91       * Execute.
92       *
93       * @param root data context
94       * @param sql sql
95       * @param paramIndexes param indexes
96       * @return enumerable result
97       */
98      public Enumerable<Object> execute(final DataContext root, final String sql, final int[] paramIndexes) {
99          return scanExecutor.execute(table, new EnumerableScanExecutorContext(root, sql, paramIndexes));
100     }
101     
102     @Override
103     public String toString() {
104         return "SQLFederationTable";
105     }
106     
107     @Override
108     public Statistic getStatistic() {
109         return statistic;
110     }
111 }
