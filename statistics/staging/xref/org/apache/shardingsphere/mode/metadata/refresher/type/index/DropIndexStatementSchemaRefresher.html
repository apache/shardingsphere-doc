38  /**
39   * Schema refresher for drop index statement.
40   */
41  public final class DropIndexStatementSchemaRefresher implements MetaDataRefresher<DropIndexStatement> {
42      
43      @Override
44      public void refresh(final MetaDataManagerPersistService metaDataManagerPersistService, final ShardingSphereDatabase database, final Collection<String> logicDataSourceNames,
45                          final String schemaName, final DatabaseType databaseType, final DropIndexStatement sqlStatement, final ConfigurationProperties props) {
46          for (IndexSegment each : sqlStatement.getIndexes()) {
47              String actualSchemaName = each.getOwner().map(optional -> optional.getIdentifier().getValue().toLowerCase()).orElse(schemaName);
48              Optional<String> logicTableName = findLogicTableName(database, sqlStatement, Collections.singletonList(each));
49              if (!logicTableName.isPresent()) {
50                  continue;
51              }
52              AlterSchemaMetaDataPOJO alterSchemaMetaDataPOJO = new AlterSchemaMetaDataPOJO(database.getName(), actualSchemaName);
53              ShardingSphereTable table = newShardingSphereTable(database.getSchema(actualSchemaName).getTable(logicTableName.get()));
54              table.removeIndex(each.getIndexName().getIdentifier().getValue());
55              alterSchemaMetaDataPOJO.getAlteredTables().add(table);
56              metaDataManagerPersistService.alterSchemaMetaData(alterSchemaMetaDataPOJO);
57          }
58      }
59      
60      private Optional<String> findLogicTableName(final ShardingSphereDatabase database, final DropIndexStatement sqlStatement, final Collection<IndexSegment> indexSegments) {
61          Optional<SimpleTableSegment> simpleTableSegment = DropIndexStatementHandler.getSimpleTableSegment(sqlStatement);
62          if (simpleTableSegment.isPresent()) {
63              return Optional.of(simpleTableSegment.get().getTableName().getIdentifier().getValue());
64          }
65          Collection<QualifiedTable> tableNames = IndexMetaDataUtils.getTableNames(database, database.getProtocolType(), indexSegments);
66          return tableNames.isEmpty() ? Optional.empty() : Optional.of(tableNames.iterator().next().getTableName());
67      }
68      
69      private ShardingSphereTable newShardingSphereTable(final ShardingSphereTable table) {
70          ShardingSphereTable result = new ShardingSphereTable(table.getName(), table.getColumnValues(), table.getIndexValues(), table.getConstraintValues(), table.getType());
71          result.getColumnNames().addAll(table.getColumnNames());
72          result.getVisibleColumns().addAll(table.getVisibleColumns());
73          result.getPrimaryKeyColumns().addAll(table.getPrimaryKeyColumns());
74          return result;
75      }
76      
77      @Override
78      public Class<DropIndexStatement> getType() {
79          return DropIndexStatement.class;
80      }
81  }
