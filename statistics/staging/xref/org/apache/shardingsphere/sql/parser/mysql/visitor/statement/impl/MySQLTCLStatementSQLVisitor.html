59  /**
60   * TCL Statement SQL visitor for MySQL.
61   */
62  @NoArgsConstructor
63  public final class MySQLTCLStatementSQLVisitor extends MySQLStatementSQLVisitor implements TCLSQLVisitor, SQLStatementVisitor {
64      
65      public MySQLTCLStatementSQLVisitor(final Properties props) {
66          super(props);
67      }
68      
69      @Override
70      public ASTNode visitSetTransaction(final SetTransactionContext ctx) {
71          MySQLSetTransactionStatement result = new MySQLSetTransactionStatement();
72          if (null != ctx.optionType()) {
73              OperationScope scope = null;
74              if (null != ctx.optionType().SESSION()) {
75                  scope = OperationScope.SESSION;
76              } else if (null != ctx.optionType().GLOBAL()) {
77                  scope = OperationScope.GLOBAL;
78              }
79              result.setScope(scope);
80          }
81          if (null != ctx.transactionCharacteristics().isolationLevel()) {
82              TransactionIsolationLevel isolationLevel = null;
83              if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().SERIALIZABLE()) {
84                  isolationLevel = TransactionIsolationLevel.SERIALIZABLE;
85              } else if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().COMMITTED()) {
86                  isolationLevel = TransactionIsolationLevel.READ_COMMITTED;
87              } else if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().UNCOMMITTED()) {
88                  isolationLevel = TransactionIsolationLevel.READ_UNCOMMITTED;
89              } else if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().REPEATABLE()) {
90                  isolationLevel = TransactionIsolationLevel.REPEATABLE_READ;
91              }
92              result.setIsolationLevel(isolationLevel);
93          }
94          if (null != ctx.transactionCharacteristics().transactionAccessMode()) {
95              TransactionAccessType accessType = null;
96              if (null != ctx.transactionCharacteristics().transactionAccessMode().ONLY()) {
97                  accessType = TransactionAccessType.READ_ONLY;
98              } else if (null != ctx.transactionCharacteristics().transactionAccessMode().WRITE()) {
99                  accessType = TransactionAccessType.READ_WRITE;
100             }
101             result.setAccessMode(accessType);
102         }
103         return result;
104     }
105     
106     @Override
107     public ASTNode visitSetAutoCommit(final SetAutoCommitContext ctx) {
108         MySQLSetAutoCommitStatement result = new MySQLSetAutoCommitStatement();
109         result.setAutoCommit(generateAutoCommitSegment(ctx.autoCommitValue).isAutoCommit());
110         return result;
111     }
112     
113     private AutoCommitSegment generateAutoCommitSegment(final Token ctx) {
114         boolean autoCommit = "1".equals(ctx.getText()) || "ON".equals(ctx.getText());
115         return new AutoCommitSegment(ctx.getStartIndex(), ctx.getStopIndex(), autoCommit);
116     }
117     
118     @Override
119     public ASTNode visitBeginTransaction(final BeginTransactionContext ctx) {
120         return new MySQLBeginTransactionStatement();
121     }
122     
123     @Override
124     public ASTNode visitCommit(final CommitContext ctx) {
125         return new MySQLCommitStatement();
126     }
127     
128     @Override
129     public ASTNode visitRollback(final RollbackContext ctx) {
130         MySQLRollbackStatement result = new MySQLRollbackStatement();
131         if (null != ctx.identifier()) {
132             result.setSavepointName(((IdentifierValue) visit(ctx.identifier())).getValue());
133         }
134         return result;
135     }
136     
137     @Override
138     public ASTNode visitSavepoint(final SavepointContext ctx) {
139         MySQLSavepointStatement result = new MySQLSavepointStatement();
140         result.setSavepointName(((IdentifierValue) visit(ctx.identifier())).getValue());
141         return result;
142     }
143     
144     @Override
145     public ASTNode visitReleaseSavepoint(final ReleaseSavepointContext ctx) {
146         MySQLReleaseSavepointStatement result = new MySQLReleaseSavepointStatement();
147         result.setSavepointName(((IdentifierValue) visit(ctx.identifier())).getValue());
148         return result;
149     }
150     
151     @Override
152     public ASTNode visitXa(final XaContext ctx) {
153         MySQLXAStatement result = new MySQLXAStatement();
154         result.setOp(ctx.getChild(1).getText().toUpperCase());
155         if (null != ctx.xid()) {
156             result.setXid(ctx.xid().getText());
157         }
158         return result;
159     }
160     
161     @Override
162     public ASTNode visitLock(final LockContext ctx) {
163         MySQLLockStatement result = new MySQLLockStatement();
164         if (null != ctx.tableLock()) {
165             result.getTables().addAll(getLockTables(ctx.tableLock()));
166         }
167         return result;
168     }
169     
170     private Collection<SimpleTableSegment> getLockTables(final List<TableLockContext> tableLockContexts) {
171         Collection<SimpleTableSegment> result = new LinkedList<>();
172         for (TableLockContext each : tableLockContexts) {
173             SimpleTableSegment simpleTableSegment = (SimpleTableSegment) visit(each.tableName());
174             if (null != each.alias()) {
175                 simpleTableSegment.setAlias((AliasSegment) visit(each.alias()));
176             }
177             result.add(simpleTableSegment);
178         }
179         return result;
180     }
181     
182     @Override
183     public ASTNode visitUnlock(final UnlockContext ctx) {
184         return new MySQLUnlockStatement();
185     }
