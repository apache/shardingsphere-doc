41  class FilterableTableScanExecutorTest {
42      
43      @Test
44      void assertExecuteWithShardingSphereData() {
45          OptimizerContext optimizerContext = mock(OptimizerContext.class, RETURNS_DEEP_STUBS);
46          when(optimizerContext.getParserContext(any()).getDatabaseType().getType()).thenReturn("PostgreSQL");
47          TableScanExecutorContext executorContext = mock(TableScanExecutorContext.class);
48          when(executorContext.getDatabaseName()).thenReturn("db");
49          when(executorContext.getSchemaName()).thenReturn("pg_catalog");
50          ShardingSphereData shardingSphereData = mock(ShardingSphereData.class, RETURNS_DEEP_STUBS);
51          ShardingSphereDatabaseData databaseData = mock(ShardingSphereDatabaseData.class, RETURNS_DEEP_STUBS);
52          when(shardingSphereData.getDatabaseData().get("db")).thenReturn(databaseData);
53          ShardingSphereSchemaData schemaData = mock(ShardingSphereSchemaData.class, RETURNS_DEEP_STUBS);
54          when(databaseData.getSchemaData().get("pg_catalog")).thenReturn(schemaData);
55          ShardingSphereTableData tableData = mock(ShardingSphereTableData.class);
56          when(tableData.getRows()).thenReturn(Collections.singletonList(new ShardingSphereRowData(Collections.singletonList(1))));
57          when(schemaData.getTableData().get("test")).thenReturn(tableData);
58          ShardingSphereTable shardingSphereTable = mock(ShardingSphereTable.class);
59          when(shardingSphereTable.getName()).thenReturn("test");
60          Enumerable<Object[]> enumerable = new FilterableTableScanExecutor(null, null, null, optimizerContext, null, executorContext, shardingSphereData)
61                  .execute(shardingSphereTable, mock(ScanNodeExecutorContext.class));
62          Enumerator<Object[]> actual = enumerable.enumerator();
63          actual.moveNext();
64          Object[] row = actual.current();
65          assertThat(row[0], is(1));
66      }
67  }
