18  package org.apache.shardingsphere.parser.distsql.handler.query;
19  
20  import lombok.Setter;
21  import org.apache.shardingsphere.distsql.handler.aware.DistSQLExecutorRuleAware;
22  import org.apache.shardingsphere.distsql.handler.engine.query.DistSQLQueryExecutor;
23  import org.apache.shardingsphere.infra.merge.result.impl.local.LocalDataQueryResultRow;
24  import org.apache.shardingsphere.mode.manager.ContextManager;
25  import org.apache.shardingsphere.parser.config.SQLParserRuleConfiguration;
26  import org.apache.shardingsphere.parser.distsql.statement.queryable.ShowSQLParserRuleStatement;
27  import org.apache.shardingsphere.parser.rule.SQLParserRule;
28  
29  import java.util.Arrays;
30  import java.util.Collection;
31  import java.util.Collections;
32  
33  /**
34   * Show SQL parser rule executor.
35   */
36  @Setter
37  public final class ShowSQLParserRuleExecutor implements DistSQLQueryExecutor<ShowSQLParserRuleStatement>, DistSQLExecutorRuleAware<SQLParserRule> {
38      
39      private SQLParserRule rule;
40      
41      @Override
42      public Collection<String> getColumnNames(final ShowSQLParserRuleStatement sqlStatement) {
43          return Arrays.asList("parse_tree_cache", "sql_statement_cache");
44      }
45      
46      @Override
47      public Collection<LocalDataQueryResultRow> getRows(final ShowSQLParserRuleStatement sqlStatement, final ContextManager contextManager) {
48          SQLParserRuleConfiguration ruleConfig = rule.getConfiguration();
49          return Collections.singleton(new LocalDataQueryResultRow(
50                  null == ruleConfig.getParseTreeCache() ? "" : ruleConfig.getParseTreeCache().toString(),
51                  null == ruleConfig.getSqlStatementCache() ? "" : ruleConfig.getSqlStatementCache().toString()));
52      }
53      
54      @Override
55      public Class<SQLParserRule> getRuleClass() {
56          return SQLParserRule.class;
57      }
58      
59      @Override
60      public Class<ShowSQLParserRuleStatement> getType() {
61          return ShowSQLParserRuleStatement.class;
62      }
63  }
