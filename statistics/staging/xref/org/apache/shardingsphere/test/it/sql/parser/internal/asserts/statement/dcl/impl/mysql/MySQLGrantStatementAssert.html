20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.sql.parser.sql.dialect.statement.mysql.dcl.MySQLGrantStatement;
23  import org.apache.shardingsphere.test.it.sql.parser.internal.asserts.SQLCaseAssertContext;
24  import org.apache.shardingsphere.test.it.sql.parser.internal.asserts.segment.generic.GrantLevelSegmentAssert;
25  import org.apache.shardingsphere.test.it.sql.parser.internal.cases.parser.jaxb.statement.dcl.GrantStatementTestCase;
26  
27  import java.util.Collections;
28  
29  import static org.hamcrest.CoreMatchers.is;
30  import static org.hamcrest.MatcherAssert.assertThat;
31  
32  /**
33   * MySQL grant statement assert.
34   */
35  @NoArgsConstructor(access = AccessLevel.PRIVATE)
36  public final class MySQLGrantStatementAssert {
37      
38      /**
39       * Assert MySQL grant statement is correct with expected parser result.
40       *
41       * @param assertContext assert context
42       * @param actual actual MySQL grant statement
43       * @param expected expected grant statement test case
44       */
45      public static void assertIs(final SQLCaseAssertContext assertContext, final MySQLGrantStatement actual, final GrantStatementTestCase expected) {
46          if (null == expected.getTables() || expected.getTables().isEmpty()) {
47              assertThat(assertContext.getText("Actual table should not exist."), actual.getTables(), is(Collections.emptyList()));
48          } else {
49              assertThat(expected.getTables().size(), is(1));
50              GrantLevelSegmentAssert.assertIs(assertContext, actual.getLevel(), expected.getTables());
51          }
52      }
53  }
