166     @Test
167     void assertPersistEphemeralExistKey() throws NacosException {
168         final String key = "/test/children/keys/ephemeral/1";
169         final Instance instance = new Instance();
170         instance.setEphemeral(true);
171         Map<String, String> metaDataMap = new HashMap<>(4, 1);
172         metaDataMap.put(PreservedMetadataKeys.HEART_BEAT_INTERVAL, String.valueOf(2000));
173         metaDataMap.put(PreservedMetadataKeys.HEART_BEAT_TIMEOUT, String.valueOf(4000));
174         metaDataMap.put(PreservedMetadataKeys.IP_DELETE_TIMEOUT, String.valueOf(6000));
175         metaDataMap.put(key, "value0");
176         instance.setMetadata(metaDataMap);
177         List<Instance> instances = new LinkedList<>();
178         buildParentPath(key, instances);
179         ServiceMetaData persistentService = serviceController.getPersistentService();
180         when(client.getAllInstances(persistentService.getServiceName(), false)).thenReturn(instances);
181         instances = new LinkedList<>();
182         instances.add(instance);
183         ServiceMetaData ephemeralService = serviceController.getEphemeralService();
184         when(client.getAllInstances(ephemeralService.getServiceName(), false)).thenReturn(instances);
185         doAnswer(AdditionalAnswers.answerVoid(getDeregisterInstanceAnswer())).when(client).deregisterInstance(anyString(), any(Instance.class));
186         doAnswer(AdditionalAnswers.answerVoid(getRegisterInstanceAnswer())).when(client).registerInstance(anyString(), any(Instance.class));
187         REPOSITORY.persistEphemeral(key, "value4");
188         ArgumentCaptor<Instance> instanceArgumentCaptor = ArgumentCaptor.forClass(Instance.class);
189         ArgumentCaptor<String> stringArgumentCaptor = ArgumentCaptor.forClass(String.class);
190         verify(client).deregisterInstance(anyString(), any(Instance.class));
191         verify(client).registerInstance(stringArgumentCaptor.capture(), instanceArgumentCaptor.capture());
192         Instance registerInstance = instanceArgumentCaptor.getValue();
193         String registerType = stringArgumentCaptor.getValue();
194         assertThat(registerType, is(ephemeralService.getServiceName()));
195         assertThat(registerInstance.isEphemeral(), is(true));
196         assertThat(NacosMetaDataUtils.getValue(registerInstance), is("value4"));
197         Map<String, String> metaData = registerInstance.getMetadata();
198         long timeToLiveSeconds = Long.parseLong(NacosPropertyKey.TIME_TO_LIVE_SECONDS.getDefaultValue());
199         assertThat(metaData.get(PreservedMetadataKeys.HEART_BEAT_INTERVAL), is(String.valueOf(timeToLiveSeconds * 1000 / 3)));
200         assertThat(metaData.get(PreservedMetadataKeys.HEART_BEAT_TIMEOUT), is(String.valueOf(timeToLiveSeconds * 1000 * 2 / 3)));
201         assertThat(metaData.get(PreservedMetadataKeys.IP_DELETE_TIMEOUT), is(String.valueOf(timeToLiveSeconds * 1000)));
202     }

204     private void buildParentPath(final String key, final List<Instance> instances) {
205         StringBuilder parentPath = new StringBuilder();
206         final String[] partPath = key.split(PersistRepository.PATH_SEPARATOR);
207         for (int index = 1; index < partPath.length - 1; index++) {
208             parentPath.append(PersistRepository.PATH_SEPARATOR);
209             parentPath.append(partPath[index]);
210             String path = parentPath.toString();
211             Instance instance = new Instance();
212             instance.setEphemeral(false);
213             instance.setMetadata(Collections.singletonMap(path, ""));
214             instances.add(instance);
215         }
216     }

273         ServiceMetaData ephemeralService = serviceController.getEphemeralService();
274         ephemeralService.setListener(null);

350     private VoidAnswer2<String, EventListener> getListenerAnswer(final Instance preInstance, final Event event) {
351         return (serviceName, listener) -> {
352             MemberAccessor accessor = Plugins.getMemberAccessor();
353             if (Objects.nonNull(preInstance)) {
354                 Map<String, Instance> preInstances = new HashMap<>();
355                 preInstances.put(NacosMetaDataUtils.getKey(preInstance), preInstance);
356                 accessor.set(listener.getClass().getDeclaredField("preInstances"), listener, preInstances);
357             }
358             listener.onEvent(event);
359         };
360     }
361     
362     private VoidAnswer2<String, Instance> getRegisterInstanceAnswer() {
363         return (serviceName, instance) -> {
364             List<Instance> instances = client.getAllInstances(serviceName, false);
365             instances.removeIf(each -> StringUtils.equals(each.getIp(), instance.getIp()) && each.getPort() == instance.getPort());
366             instances.add(instance);
367             when(client.getAllInstances(serviceName, false)).thenReturn(instances);
368         };
369     }
370     
371     private VoidAnswer2<String, Instance> getDeregisterInstanceAnswer() {
372         return (serviceName, instance) -> {
373             List<Instance> instances = client.getAllInstances(serviceName, false);
374             instances.remove(instance);
375             when(client.getAllInstances(serviceName, false)).thenReturn(instances);
376         };
