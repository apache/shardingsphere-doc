20  import org.apache.shardingsphere.data.pipeline.postgresql.sqlbuilder.ddl.PostgreSQLDDLTemplateExecutor;

22  import java.sql.Connection;
23  import java.util.Collection;
24  import java.util.HashMap;
25  import java.util.HashSet;
26  import java.util.LinkedHashMap;
27  import java.util.LinkedList;
28  import java.util.Map;
29  import java.util.Optional;
30  import java.util.Set;
31  import java.util.stream.Collectors;

33  /**
34   * Constraints properties appender for PostgreSQL.
35   */
36  public final class PostgreSQLConstraintsPropertiesAppender {
37      
38      private static final Integer PG_CONSTRAINTS_INCLUDE_VERSION = 11;
39      
40      private final PostgreSQLDDLTemplateExecutor templateExecutor;
41      
42      public PostgreSQLConstraintsPropertiesAppender(final Connection connection, final int majorVersion, final int minorVersion) {
43          templateExecutor = new PostgreSQLDDLTemplateExecutor(connection, majorVersion, minorVersion);
44      }
45      
46      /**
47       * Append constraints properties.
48       *
49       * @param context create table sql context
50       */
51      public void append(final Map<String, Object> context) {
52          loadPrimaryOrUniqueConstraint(context, "primary_key", "p");
53          loadPrimaryOrUniqueConstraint(context, "unique_constraint", "u");
54          context.put("foreign_key", fetchForeignKeys(context));
55          context.put("check_constraint", fetchCheckConstraints(context));
56          context.put("exclude_constraint", getExclusionConstraints(context));
57      }
58      
59      private Collection<Map<String, Object>> fetchCheckConstraints(final Map<String, Object> context) {
60          Collection<Map<String, Object>> result = new LinkedList<>();
61          for (Map<String, Object> each : getCheckConstraints((Long) context.get("tid"))) {
62              if (!isPartitionAndConstraintInherited(each, context)) {
63                  result.add(each);
64              }
65          }
66          return result;
67      }
68      
69      private Collection<Map<String, Object>> fetchForeignKeys(final Map<String, Object> context) {
70          Collection<Map<String, Object>> result = new LinkedList<>();
71          for (Map<String, Object> each : getForeignKeys((Long) context.get("tid"))) {
72              if (!isPartitionAndConstraintInherited(each, context)) {
73                  result.add(each);
74              }
75          }
76          return result;
77      }
78      
79      private void loadPrimaryOrUniqueConstraint(final Map<String, Object> context, final String name, final String type) {
80          Collection<Map<String, Object>> constraintsProps = fetchConstraintsProperties(context, type);
81          fetchConstraintsColumns(constraintsProps);
82          context.put(name, constraintsProps.stream().filter(each -> !isPartitionAndConstraintInherited(each, context)).collect(Collectors.toList()));
83      }
84      
85      private void fetchConstraintsColumns(final Collection<Map<String, Object>> constraintsProps) {
86          for (Map<String, Object> each : constraintsProps) {
87              Collection<Map<String, Object>> columns = new LinkedList<>();
88              for (Map<String, Object> col : fetchConstraintsCols(each)) {
89                  Map<String, Object> column = new HashMap<>();
90                  column.put("column", stripQuote((String) col.get("column")));
91                  columns.add(column);
92              }
93              each.put("columns", columns);
94              appendConstraintsInclude(each);
95          }
96      }
97      
98      private void appendConstraintsInclude(final Map<String, Object> constraintsProp) {
99          Map<String, Object> params = new LinkedHashMap<>();
100         params.put("cid", constraintsProp.get("oid"));
101         Collection<Object> includes = new LinkedList<>();
102         if (templateExecutor.getMajorVersion() >= PG_CONSTRAINTS_INCLUDE_VERSION) {
103             for (Map<String, Object> each : templateExecutor.executeByTemplate(params, "component/index_constraint/%s/get_constraint_include.ftl")) {
104                 includes.add(each.get("colname"));
105             }
106         }
107         constraintsProp.put("include", includes);
108     }
109     
110     private String stripQuote(final String column) {
111         String result = column;
112         if (column.startsWith("\"")) {
113             result = result.substring(1);
114         }
115         if (column.endsWith("\"")) {
116             result = result.substring(0, result.length() - 1);
117         }
118         return result;
119     }
120     
121     private Collection<Map<String, Object>> fetchConstraintsCols(final Map<String, Object> constraintColProps) {
122         Map<String, Object> params = new HashMap<>();
123         params.put("cid", constraintColProps.get("oid"));
124         params.put("colcnt", constraintColProps.get("col_count"));
125         return templateExecutor.executeByTemplate(params, "component/index_constraint/%s/get_costraint_cols.ftl");
126     }
127     
128     private Collection<Map<String, Object>> fetchConstraintsProperties(final Map<String, Object> context, final String constraintType) {
129         Map<String, Object> params = new HashMap<>();
130         params.put("did", context.get("did"));
131         params.put("tid", context.get("tid"));
132         params.put("cid", context.get("cid"));
133         params.put("constraint_type", constraintType);
134         return templateExecutor.executeByTemplate(params, "component/index_constraint/%s/properties.ftl");
135     }
136     
137     private Collection<Map<String, Object>> getExclusionConstraints(final Map<String, Object> context) {
138         Map<String, Object> params = new HashMap<>();
139         params.put("tid", context.get("tid"));
140         params.put("did", context.get("did"));
141         Collection<Map<String, Object>> result = templateExecutor.executeByTemplate(params, "component/exclusion_constraint/%s/properties.ftl");
142         for (Map<String, Object> each : result) {
143             getExclusionConstraintsColumns(each);
144         }
145         return result;
146     }
147     
148     private void getExclusionConstraintsColumns(final Map<String, Object> exclusionConstraintsProps) {
149         Map<String, Object> params = new HashMap<>();
150         params.put("cid", exclusionConstraintsProps.get("oid"));
151         params.put("col_count", exclusionConstraintsProps.get("col_count"));
152         Collection<Map<String, Object>> columns = new LinkedList<>();
153         for (Map<String, Object> each : templateExecutor.executeByTemplate(params, "component/exclusion_constraint/%s/get_constraint_cols.ftl")) {
154             boolean order = 0 == (((int) each.get("options")) & 1);
155             boolean nullsOrder = 0 != (((int) each.get("options")) & 2);
156             Map<String, Object> col = new HashMap<>();
157             col.put("column", strip((String) each.get("coldef")));
158             col.put("oper_class", each.get("opcname"));
159             col.put("order", order);
160             col.put("nulls_order", nullsOrder);
161             col.put("operator", each.get("oprname"));
162             col.put("col_type", each.get("datatype"));
163             col.put("is_exp", each.get("is_exp"));
164             columns.add(col);
165         }
166         exclusionConstraintsProps.put("columns", columns);
167         Map<String, Object> map = new HashMap<>();
168         map.put("cid", exclusionConstraintsProps.get("oid"));
169         Collection<String> include = new LinkedList<>();
170         if (templateExecutor.getMajorVersion() >= PG_CONSTRAINTS_INCLUDE_VERSION) {
171             for (Map<String, Object> each : templateExecutor.executeByTemplate(map, "exclusion_constraint/%s/get_constraint_include.ftl")) {
172                 include.add(each.get("colname").toString());
173             }
174         }
175         exclusionConstraintsProps.put("include", include);
176     }
177     
178     private Collection<Map<String, Object>> getForeignKeys(final Long tid) {
179         Map<String, Object> params = new HashMap<>();
180         params.put("tid", tid);
181         Collection<Map<String, Object>> result = templateExecutor.executeByTemplate(params, "component/foreign_key/%s/properties.ftl");
182         for (Map<String, Object> each : result) {
183             Collection<Map<String, Object>> columns = new LinkedList<>();
184             Set<String> cols = new HashSet<>();
185             for (Map<String, Object> col : getForeignKeysCols(tid, each)) {
186                 Map<String, Object> foreignKeysRef = new HashMap<>();
187                 foreignKeysRef.put("local_column", col.get("conattname"));
188                 foreignKeysRef.put("references", each.get("confrelid"));
189                 foreignKeysRef.put("referenced", col.get("confattname"));
190                 foreignKeysRef.put("references_table_name", each.get("refnsp") + "." + each.get("reftab"));
191                 columns.add(foreignKeysRef);
192                 cols.add((String) col.get("conattname"));
193             }
194             setRemoteName(each, columns);
195             Optional<String> coveringindex = searchCoveringIndex(tid, cols);
196             each.put("coveringindex", coveringindex.orElse(null));
197             each.put("autoindex", !coveringindex.isPresent());
198             each.put("hasindex", coveringindex.isPresent());
199             each.put("columns", columns);
200         }
201         return result;
202     }
203     
204     private void setRemoteName(final Map<String, Object> foreignKey, final Collection<Map<String, Object>> columns) {
205         Map<String, Object> params = new HashMap<>();
206         params.put("tid", columns.iterator().next().get("references"));
207         Map<String, Object> parents = templateExecutor.executeByTemplateForSingleRow(params, "component/foreign_key/%s/get_parent.ftl");
208         foreignKey.put("remote_schema", parents.get("schema"));
209         foreignKey.put("remote_table", parents.get("table"));
210     }
211     
212     private Collection<Map<String, Object>> getForeignKeysCols(final Long tid, final Map<String, Object> foreignKeyProps) {
213         Map<String, Object> params = new HashMap<>();
214         params.put("tid", tid);
215         Collection<Map<String, Object>> keys = new LinkedList<>();
216         Map<String, Object> key = new HashMap<>();
217         key.put("confkey", foreignKeyProps.get("confkey"));
218         key.put("conkey", foreignKeyProps.get("conkey"));
219         keys.add(key);
220         params.put("keys", keys);
221         return templateExecutor.executeByTemplate(params, "component/foreign_key/%s/get_constraint_cols.ftl");
222     }
223     
224     private boolean isPartitionAndConstraintInherited(final Map<String, Object> constraint, final Map<String, Object> context) {
225         return context.containsKey("relispartition") && (boolean) context.get("relispartition") && constraint.containsKey("conislocal") && (boolean) constraint.get("conislocal");
226     }
227     
228     private Optional<String> searchCoveringIndex(final Long tid, final Set<String> cols) {
229         Map<String, Object> params = new HashMap<>();
230         params.put("tid", tid);
231         for (Map<String, Object> each : templateExecutor.executeByTemplate(params, "component/foreign_key/%s/get_constraints.ftl")) {
232             Map<String, Object> map = new HashMap<>();
233             map.put("cid", each.get("oid"));
234             map.put("colcnt", each.get("col_count"));
235             Collection<Map<String, Object>> rows = templateExecutor.executeByTemplate(map, "component/foreign_key/%s/get_cols.ftl");
236             Set<String> indexCols = new HashSet<>();
237             for (Map<String, Object> row : rows) {
238                 indexCols.add(strip(row.get("column").toString()));
239             }
240             if (isSame(indexCols, cols)) {
241                 return Optional.of((String) each.get("idxname"));
242             }
243         }
244         return Optional.empty();
245     }
246     
247     private boolean isSame(final Set<String> indexCols, final Set<String> cols) {
248         Set<String> copyIndexCols = new HashSet<>(indexCols);
249         Set<String> copyCols = new HashSet<>(cols);
250         copyIndexCols.removeAll(copyCols);
251         if (copyIndexCols.isEmpty()) {
252             cols.removeAll(indexCols);
253             return cols.isEmpty();
254         }
255         return false;
256     }
257     
258     private String strip(final String column) {
259         if (column.startsWith("\"")) {
260             return column.substring(1);
261         }
262         if (column.endsWith("\"")) {
263             return column.substring(0, column.length() - 1);
264         }
265         return column;
266     }
267     
268     private Collection<Map<String, Object>> getCheckConstraints(final Long tid) {
269         Map<String, Object> params = new HashMap<>();
270         params.put("tid", tid);
271         return templateExecutor.executeByTemplate(params, "component/check_constraint/%s/properties.ftl");
272     }
