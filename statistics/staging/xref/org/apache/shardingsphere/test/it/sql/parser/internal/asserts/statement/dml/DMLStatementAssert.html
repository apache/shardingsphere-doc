56  /**
57   * DML statement assert.
58   */
59  @NoArgsConstructor(access = AccessLevel.PRIVATE)
60  public final class DMLStatementAssert {
61      
62      /**
63       * Assert DML statement is correct with expected parser result.
64       *
65       * @param assertContext assert context
66       * @param actual actual DML statement
67       * @param expected expected parser result
68       */
69      public static void assertIs(final SQLCaseAssertContext assertContext, final DMLStatement actual, final SQLParserTestCase expected) {
70          if (actual instanceof SelectStatement) {
71              SelectStatementAssert.assertIs(assertContext, (SelectStatement) actual, (SelectStatementTestCase) expected);
72          } else if (actual instanceof UpdateStatement) {
73              UpdateStatementAssert.assertIs(assertContext, (UpdateStatement) actual, (UpdateStatementTestCase) expected);
74          } else if (actual instanceof MergeStatement) {
75              MergeStatementAssert.assertIs(assertContext, (MergeStatement) actual, (MergeStatementTestCase) expected);
76          } else if (actual instanceof DeleteStatement) {
77              DeleteStatementAssert.assertIs(assertContext, (DeleteStatement) actual, (DeleteStatementTestCase) expected);
78          } else if (actual instanceof InsertStatement) {
79              InsertStatementAssert.assertIs(assertContext, (InsertStatement) actual, (InsertStatementTestCase) expected);
80          } else if (actual instanceof CallStatement) {
81              CallStatementAssert.assertIs(assertContext, (CallStatement) actual, (CallStatementTestCase) expected);
82          } else if (actual instanceof CopyStatement) {
83              CopyStatementAssert.assertIs(assertContext, (CopyStatement) actual, (CopyStatementTestCase) expected);
84          } else if (actual instanceof DoStatement) {
85              DoStatementAssert.assertIs(assertContext, (DoStatement) actual, (DoStatementTestCase) expected);
86          } else if (actual instanceof LoadDataStatement) {
87              LoadDataStatementAssert.assertIs(assertContext, (LoadDataStatement) actual, (LoadDataStatementTestCase) expected);
88          } else if (actual instanceof LoadXMLStatement) {
89              LoadXMLStatementAssert.assertIs(assertContext, (LoadXMLStatement) actual, (LoadXMLStatementTestCase) expected);
90          }
91      }
92  }
