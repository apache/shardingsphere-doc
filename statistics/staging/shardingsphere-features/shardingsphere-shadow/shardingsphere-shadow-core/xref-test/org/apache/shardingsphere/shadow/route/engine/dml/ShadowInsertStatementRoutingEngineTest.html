55  class ShadowInsertStatementRoutingEngineTest {
56      
57      private ShadowInsertStatementRoutingEngine shadowRouteEngine;
58      
59      @BeforeEach
60      void init() {
61          shadowRouteEngine = new ShadowInsertStatementRoutingEngine(createInsertStatementContext());
62      }
63      
64      private InsertStatementContext createInsertStatementContext() {
65          InsertStatementContext result = mock(InsertStatementContext.class);
66          when(result.getAllTables()).thenReturn(Collections.singleton(new SimpleTableSegment(new TableNameSegment(20, 25, new IdentifierValue("t_order")))));
67          when(result.getInsertColumnNames()).thenReturn(Arrays.asList("user_id", "order_code", "order_name"));
68          List<ExpressionSegment> valueExpressions = Arrays.asList(new LiteralExpressionSegment(0, 10, "1"),
69                  new LiteralExpressionSegment(11, 20, "orderCode"), new LiteralExpressionSegment(21, 30, "orderName"));
70          when(result.getInsertValueContexts()).thenReturn(Collections.singletonList(new InsertValueContext(valueExpressions, Collections.emptyList(), 0)));
71          MySQLInsertStatement insertStatement = new MySQLInsertStatement();
72          insertStatement.getCommentSegments().add(new CommentSegment("/*shadow:true,foo:bar*/", 0, 20));
73          insertStatement.getCommentSegments().add(new CommentSegment("/*aaa:bbb*/", 21, 30));
74          when(result.getSqlStatement()).thenReturn(insertStatement);
75          return result;
76      }
77      
78      @Test
79      void assertRouteAndParseShadowColumnConditions() {
80          RouteContext routeContext = mock(RouteContext.class);
81          when(routeContext.getRouteUnits()).thenReturn(Collections.singleton(new RouteUnit(new RouteMapper("ds", "ds_shadow"), Collections.emptyList())));
82          shadowRouteEngine.route(routeContext, new ShadowRule(createShadowRuleConfiguration()));
83          Optional<Collection<String>> sqlNotes = shadowRouteEngine.parseSQLComments();
84          assertTrue(sqlNotes.isPresent());
85          assertThat(sqlNotes.get().size(), is(2));
86          Iterator<String> sqlNotesIt = sqlNotes.get().iterator();
87          assertThat(sqlNotesIt.next(), is("/*shadow:true,foo:bar*/"));
88          assertThat(sqlNotesIt.next(), is("/*aaa:bbb*/"));
89      }
90      
91      private ShadowRuleConfiguration createShadowRuleConfiguration() {
92          ShadowRuleConfiguration result = new ShadowRuleConfiguration();
93          result.setDataSources(Collections.singletonList(new ShadowDataSourceConfiguration("shadow-data-source-0", "ds", "ds_shadow")));
94          result.setTables(Collections.singletonMap("t_order", new ShadowTableConfiguration(Collections.singleton("shadow-data-source-0"), Collections.singleton("user-id-insert-regex-algorithm"))));
95          result.setShadowAlgorithms(Collections.singletonMap("user-id-insert-regex-algorithm", createShadowAlgorithm()));
96          return result;
97      }
98      
99      private AlgorithmConfiguration createShadowAlgorithm() {
100         return new AlgorithmConfiguration("REGEX_MATCH", PropertiesBuilder.build(new Property("column", "user_id"), new Property("operation", "insert"), new Property("regex", "[1]")));
101     }
102     
103     @Test
104     void assertGetAllTables() {
105         Collection<SimpleTableSegment> actual = shadowRouteEngine.getAllTables();
106         assertThat(actual.size(), is(1));
107         assertThat(actual.iterator().next().getTableName().getIdentifier().getValue(), is("t_order"));
108     }
109 }
