53  /**
54   * Driver execute batch executor.
55   */
56  @RequiredArgsConstructor
57  public final class DriverExecuteBatchExecutor {
58      
59      private final ShardingSphereConnection connection;
60      
61      private final ShardingSphereMetaData metaData;
62      
63      private final BatchPreparedStatementExecutor batchPreparedStatementExecutor;
64      
65      /**
66       * Add batch.
67       * 
68       * @param queryContext query context
69       * @param database database
70       * @return execution context
71       */
72      public ExecutionContext addBatch(final QueryContext queryContext, final ShardingSphereDatabase database) {
73          Optional<String> trafficInstanceId = connection.getTrafficInstanceId(metaData.getGlobalRuleMetaData().getSingleRule(TrafficRule.class), queryContext);
74          ExecutionContext result = trafficInstanceId.map(optional -> createExecutionContext(queryContext, optional)).orElseGet(() -> createExecutionContext(queryContext, database));
75          batchPreparedStatementExecutor.addBatchForExecutionUnits(result.getExecutionUnits());
76          return result;
77      }
78      
79      private ExecutionContext createExecutionContext(final QueryContext queryContext, final String trafficInstanceId) {
80          return new ExecutionContext(queryContext, Collections.singleton(new ExecutionUnit(trafficInstanceId, new SQLUnit(queryContext.getSql(), queryContext.getParameters()))), new RouteContext());
81      }
82      
83      private ExecutionContext createExecutionContext(final QueryContext queryContext, final ShardingSphereDatabase database) {
84          RuleMetaData globalRuleMetaData = metaData.getGlobalRuleMetaData();
85          SQLAuditEngine.audit(queryContext, globalRuleMetaData, database);
86          return new KernelProcessor().generateExecutionContext(queryContext, database, globalRuleMetaData, metaData.getProps(), connection.getDatabaseConnectionManager().getConnectionContext());
87      }
88      
89      /**
90       * Execute batch.
91       * 
92       * @param database database
93       * @param sqlStatementContext SQL statement context
94       * @param generatedValues generated values
95       * @param statementOption statement option
96       * @param prepareEngine prepare engine
97       * @param executionContext execution context
98       * @param addCallback statement add callback
99       * @param replayCallback statement replay callback
100      * @param generatedKeyCallback generated key callback
101      * @return generated keys
102      * @throws SQLException SQL exception
103      */
104     @SuppressWarnings("rawtypes")
105     public int[] executeBatch(final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext, final Collection<Comparable<?>> generatedValues,
106                               final StatementOption statementOption, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
107                               final ExecutionContext executionContext, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback,
108                               final GeneratedKeyCallback generatedKeyCallback) throws SQLException {
109         // TODO add raw SQL executor
110         return doExecuteBatch(database, batchPreparedStatementExecutor,
111                 sqlStatementContext, generatedValues, statementOption, prepareEngine, executionContext, addCallback, replayCallback, generatedKeyCallback);
112     }
113     
114     @SuppressWarnings({"rawtypes", "unchecked"})
115     private int[] doExecuteBatch(final ShardingSphereDatabase database, final BatchPreparedStatementExecutor batchExecutor,
116                                  final SQLStatementContext sqlStatementContext, final Collection<Comparable<?>> generatedValues, final StatementOption statementOption,
117                                  final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine, final ExecutionContext executionContext,
118                                  final StatementAddCallback addCallback, final StatementReplayCallback replayCallback, final GeneratedKeyCallback generatedKeyCallback) throws SQLException {
119         initBatchPreparedStatementExecutor(database, batchExecutor, prepareEngine, executionContext, replayCallback);
120         int[] result = batchExecutor.executeBatch(sqlStatementContext);
121         if (statementOption.isReturnGeneratedKeys() && generatedValues.isEmpty()) {
122             addCallback.add(batchExecutor.getStatements(), Collections.emptyList());
123             generatedKeyCallback.generateKeys();
124         }
125         return result;
126     }
127     
128     private void initBatchPreparedStatementExecutor(final ShardingSphereDatabase database, final BatchPreparedStatementExecutor batchExecutor,
129                                                     final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
130                                                     final ExecutionContext executionContext, final StatementReplayCallback replayCallback) throws SQLException {
131         List<ExecutionUnit> executionUnits = new ArrayList<>(batchExecutor.getBatchExecutionUnits().size());
132         for (BatchExecutionUnit each : batchExecutor.getBatchExecutionUnits()) {
133             ExecutionUnit executionUnit = each.getExecutionUnit();
134             executionUnits.add(executionUnit);
135         }
136         batchExecutor.init(prepareEngine
137                 .prepare(executionContext.getRouteContext(), executionUnits, new ExecutionGroupReportContext(connection.getProcessId(), database.getName(), new Grantee("", ""))));
138         replayCallback.replay();
139     }
