18  package org.apache.shardingsphere.data.pipeline.core.importer;
19  
20  import lombok.RequiredArgsConstructor;
21  import org.apache.shardingsphere.data.pipeline.api.executor.AbstractLifecycleExecutor;
22  import org.apache.shardingsphere.data.pipeline.api.ingest.channel.PipelineChannel;
23  import org.apache.shardingsphere.data.pipeline.api.ingest.record.FinishedRecord;
24  import org.apache.shardingsphere.data.pipeline.api.ingest.record.PlaceholderRecord;
25  import org.apache.shardingsphere.data.pipeline.api.ingest.record.Record;
26  import org.apache.shardingsphere.data.pipeline.common.job.progress.listener.PipelineJobProgressListener;
27  import org.apache.shardingsphere.data.pipeline.common.job.progress.listener.PipelineJobProgressUpdatedParameter;
28  import org.apache.shardingsphere.infra.util.close.QuietlyCloser;
29  import org.apache.shardingsphere.data.pipeline.core.importer.sink.PipelineSink;
30  
31  import java.util.List;
32  import java.util.concurrent.TimeUnit;
33  import java.util.stream.Collectors;
34  
35  /**
36   * Single channel consumer importer.
37   */
38  @RequiredArgsConstructor
39  public final class SingleChannelConsumerImporter extends AbstractLifecycleExecutor implements Importer {
40      
41      private final PipelineChannel channel;
42      
43      private final int batchSize;
44      
45      private final int timeout;
46      
47      private final TimeUnit timeUnit;
48      
49      private final PipelineSink sink;
50      
51      private final PipelineJobProgressListener jobProgressListener;
52      
53      @Override
54      protected void runBlocking() {
55          while (isRunning()) {
56              List<Record> records = channel.fetchRecords(batchSize, timeout, timeUnit).stream().filter(each -> !(each instanceof PlaceholderRecord)).collect(Collectors.toList());
57              if (records.isEmpty()) {
58                  continue;
59              }
60              PipelineJobProgressUpdatedParameter updatedParam = sink.write("", records);
61              channel.ack(records);
62              jobProgressListener.onProgressUpdated(updatedParam);
63              if (FinishedRecord.class.equals(records.get(records.size() - 1).getClass())) {
64                  break;
65              }
66          }
67      }
68      
69      @Override
70      protected void doStop() {
71          QuietlyCloser.close(sink);
72      }
73  }
