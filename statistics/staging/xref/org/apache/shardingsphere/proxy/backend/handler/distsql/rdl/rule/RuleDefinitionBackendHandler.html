47  /**
48   * Rule definition backend handler.
49   *
50   * @param <T> type of rule definition statement
51   */
52  public final class RuleDefinitionBackendHandler<T extends RuleDefinitionStatement> extends RDLBackendHandler<T> {
53      
54      public RuleDefinitionBackendHandler(final T sqlStatement, final ConnectionSession connectionSession) {
55          super(sqlStatement, connectionSession);
56      }
57      
58      @SuppressWarnings({"rawtypes", "unchecked"})
59      @Override
60      protected ResponseHeader execute(final String databaseName, final T sqlStatement) {
61          RuleDefinitionUpdater ruleDefinitionUpdater = TypedSPILoader.getService(RuleDefinitionUpdater.class, sqlStatement.getClass());
62          Class<? extends RuleConfiguration> ruleConfigClass = ruleDefinitionUpdater.getRuleConfigurationClass();
63          ShardingSphereDatabase database = ProxyContext.getInstance().getDatabase(databaseName);
64          RuleConfiguration currentRuleConfig = findCurrentRuleConfiguration(database, ruleConfigClass).orElse(null);
65          ruleDefinitionUpdater.checkSQLStatement(database, sqlStatement, currentRuleConfig);
66          if (getRefreshStatus(sqlStatement, currentRuleConfig, ruleDefinitionUpdater)) {
67              ProxyContext.getInstance().getContextManager().getInstanceContext().getModeContextManager().alterRuleConfiguration(databaseName,
68                      processSQLStatement(database, sqlStatement, ruleDefinitionUpdater, currentRuleConfig));
69          }
70          return new UpdateResponseHeader(sqlStatement);
71      }
72      
73      private Optional<RuleConfiguration> findCurrentRuleConfiguration(final ShardingSphereDatabase database, final Class<? extends RuleConfiguration> ruleConfigClass) {
74          for (RuleConfiguration each : database.getRuleMetaData().getConfigurations()) {
75              if (ruleConfigClass.isAssignableFrom(each.getClass())) {
76                  return Optional.of(each);
77              }
78          }
79          return Optional.empty();
80      }
81      
82      @SuppressWarnings("rawtypes")
83      private Collection<RuleConfiguration> processSQLStatement(final ShardingSphereDatabase database,
84                                                                final T sqlStatement, final RuleDefinitionUpdater updater, final RuleConfiguration currentRuleConfig) {
85          Collection<RuleConfiguration> result = new LinkedList<>(database.getRuleMetaData().getConfigurations());
86          if (updater instanceof RuleDefinitionCreateUpdater) {
87              if (null != currentRuleConfig) {
88                  result.remove(currentRuleConfig);
89              }
90              RuleConfiguration createdRuleConfig = processCreate(sqlStatement, (RuleDefinitionCreateUpdater) updater, currentRuleConfig);
91              result.add(decorateRuleConfiguration(database, createdRuleConfig));
92          } else if (updater instanceof RuleDefinitionAlterUpdater) {
93              result.remove(currentRuleConfig);
94              RuleConfiguration alteredRuleConfig = processAlter(sqlStatement, (RuleDefinitionAlterUpdater) updater, currentRuleConfig);
95              result.add(decorateRuleConfiguration(database, alteredRuleConfig));
96          } else if (updater instanceof RuleDefinitionDropUpdater) {
97              processDrop(database, result, sqlStatement, (RuleDefinitionDropUpdater) updater, currentRuleConfig);
98          } else {
99              throw new UnsupportedSQLOperationException(String.format("Cannot support RDL updater type `%s`", updater.getClass().getName()));
100         }
101         return result;
102     }
103     
104     @SuppressWarnings({"unchecked", "rawtypes"})
105     private RuleConfiguration decorateRuleConfiguration(final ShardingSphereDatabase database, final RuleConfiguration ruleConfig) {
106         Optional<RuleConfigurationDecorator> decorator = TypedSPILoader.findService(RuleConfigurationDecorator.class, ruleConfig.getClass());
107         return decorator.map(optional -> optional.decorate(database.getName(),
108                 database.getResourceMetaData().getStorageUnits().entrySet().stream()
109                         .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
110                 database.getRuleMetaData().getRules(), ruleConfig)).orElse(ruleConfig);
111     }
112     
113     @SuppressWarnings({"rawtypes", "unchecked"})
114     private RuleConfiguration processCreate(final T sqlStatement, final RuleDefinitionCreateUpdater updater, final RuleConfiguration currentRuleConfig) {
115         RuleConfiguration toBeCreatedRuleConfig = updater.buildToBeCreatedRuleConfiguration(currentRuleConfig, sqlStatement);
116         if (null == currentRuleConfig) {
117             return toBeCreatedRuleConfig;
118         }
119         updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeCreatedRuleConfig);
120         return currentRuleConfig;
121     }
122     
123     @SuppressWarnings({"rawtypes", "unchecked"})
124     private RuleConfiguration processAlter(final T sqlStatement, final RuleDefinitionAlterUpdater updater, final RuleConfiguration currentRuleConfig) {
125         RuleConfiguration toBeAlteredRuleConfig = updater.buildToBeAlteredRuleConfiguration(sqlStatement);
126         updater.updateCurrentRuleConfiguration(currentRuleConfig, toBeAlteredRuleConfig);
127         return currentRuleConfig;
128     }
129     
130     @SuppressWarnings({"rawtypes", "unchecked"})
131     private void processDrop(final ShardingSphereDatabase database, final Collection<RuleConfiguration> configs, final T sqlStatement,
132                              final RuleDefinitionDropUpdater updater, final RuleConfiguration currentRuleConfig) {
133         if (!updater.hasAnyOneToBeDropped(sqlStatement, currentRuleConfig)) {
134             return;
135         }
136         if (updater.updateCurrentRuleConfiguration(sqlStatement, currentRuleConfig)) {
137             configs.remove(currentRuleConfig);
138         }
139         if (updater instanceof DropReadwriteSplittingRuleStatementUpdater) {
140             database.getRuleMetaData().findSingleRule(StaticDataSourceContainedRule.class)
141                     .ifPresent(optional -> ((DropReadwriteSplittingRuleStatement) sqlStatement).getNames().forEach(optional::cleanStorageNodeDataSource));
142         }
143     }
144     
145     private boolean getRefreshStatus(final SQLStatement sqlStatement, final RuleConfiguration currentRuleConfig, final RuleDefinitionUpdater<?, ?> updater) {
146         return !(updater instanceof RuleDefinitionDropUpdater) || ((RuleDefinitionDropUpdater) updater).hasAnyOneToBeDropped(sqlStatement, currentRuleConfig);
147     }
