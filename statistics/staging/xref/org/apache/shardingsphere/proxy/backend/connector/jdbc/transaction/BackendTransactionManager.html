22  import org.apache.shardingsphere.infra.session.connection.transaction.TransactionConnectionContext;
23  import org.apache.shardingsphere.infra.spi.type.ordered.OrderedSPILoader;
24  import org.apache.shardingsphere.proxy.backend.connector.ProxyDatabaseConnectionManager;
25  import org.apache.shardingsphere.proxy.backend.connector.TransactionManager;
26  import org.apache.shardingsphere.proxy.backend.context.ProxyContext;
27  import org.apache.shardingsphere.proxy.backend.util.TransactionUtils;
28  import org.apache.shardingsphere.transaction.savepoint.ConnectionSavepointManager;
29  import org.apache.shardingsphere.transaction.ShardingSphereTransactionManagerEngine;
30  import org.apache.shardingsphere.transaction.api.TransactionType;
31  import org.apache.shardingsphere.transaction.rule.TransactionRule;
32  import org.apache.shardingsphere.transaction.spi.ShardingSphereDistributedTransactionManager;
33  import org.apache.shardingsphere.transaction.spi.TransactionHook;
34  
35  import java.sql.Connection;
36  import java.sql.SQLException;
37  import java.util.Collection;
38  import java.util.Iterator;
39  import java.util.LinkedList;
40  import java.util.Map;
41  import java.util.Map.Entry;
42  
43  /**
44   * Backend transaction manager.
45   */
46  @SuppressWarnings({"rawtypes", "unchecked"})
47  public final class BackendTransactionManager implements TransactionManager {
48      
49      private final ProxyDatabaseConnectionManager connection;
50      
51      private final TransactionType transactionType;
52      
53      private final LocalTransactionManager localTransactionManager;
54      
55      private final ShardingSphereDistributedTransactionManager distributedTransactionManager;
56      
57      private final Map<ShardingSphereRule, TransactionHook> transactionHooks;
58      
59      public BackendTransactionManager(final ProxyDatabaseConnectionManager databaseConnectionManager) {
60          connection = databaseConnectionManager;
61          localTransactionManager = new LocalTransactionManager(databaseConnectionManager);
62          TransactionRule transactionRule = ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getGlobalRuleMetaData().getSingleRule(TransactionRule.class);
63          transactionType = transactionRule.getDefaultType();
64          ShardingSphereTransactionManagerEngine engine = transactionRule.getResource();
65          distributedTransactionManager = null == engine ? null : engine.getTransactionManager(transactionType);
66          transactionHooks = OrderedSPILoader.getServices(TransactionHook.class, ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData().getGlobalRuleMetaData().getRules());
67      }
68      
69      @Override
70      public void begin() {
71          if (!connection.getConnectionSession().getTransactionStatus().isInTransaction()) {
72              connection.getConnectionSession().getTransactionStatus().setInTransaction(true);
73              getTransactionContext().beginTransaction(String.valueOf(transactionType));
74              connection.closeHandlers(true);
75              connection.closeConnections(false);
76          }
77          DatabaseType databaseType = ProxyContext.getInstance().getDatabaseType();
78          for (Entry<ShardingSphereRule, TransactionHook> entry : transactionHooks.entrySet()) {
79              entry.getValue().beforeBegin(entry.getKey(), databaseType, getTransactionContext());
80          }
81          if (TransactionType.LOCAL == transactionType || null == distributedTransactionManager) {
82              localTransactionManager.begin();
83          } else {
84              distributedTransactionManager.begin();
85          }
86          for (Entry<ShardingSphereRule, TransactionHook> entry : transactionHooks.entrySet()) {
87              entry.getValue().afterBegin(entry.getKey(), databaseType, getTransactionContext());
88          }
89      }
90      
91      @Override
92      public void commit() throws SQLException {
93          DatabaseType databaseType = ProxyContext.getInstance().getDatabaseType();
94          for (Entry<ShardingSphereRule, TransactionHook> entry : transactionHooks.entrySet()) {
95              entry.getValue().beforeCommit(entry.getKey(), databaseType,
96                      connection.getCachedConnections().values(), getTransactionContext(), ProxyContext.getInstance().getContextManager().getComputeNodeInstanceContext().getLockContext());
97          }
98          if (connection.getConnectionSession().getTransactionStatus().isInTransaction()) {
99              try {
100                 if (TransactionType.LOCAL == TransactionUtils.getTransactionType(getTransactionContext()) || null == distributedTransactionManager) {
101                     localTransactionManager.commit();
102                 } else {
103                     distributedTransactionManager.commit(getTransactionContext().isExceptionOccur());
104                 }
105             } finally {
106                 for (Entry<ShardingSphereRule, TransactionHook> entry : transactionHooks.entrySet()) {
107                     entry.getValue().afterCommit(entry.getKey(), databaseType,
108                             connection.getCachedConnections().values(), getTransactionContext(), ProxyContext.getInstance().getContextManager().getComputeNodeInstanceContext().getLockContext());
109                 }
110                 for (Connection each : connection.getCachedConnections().values()) {
111                     ConnectionSavepointManager.getInstance().transactionFinished(each);
112                 }
113                 connection.getConnectionSession().getTransactionStatus().setInTransaction(false);
114                 connection.getConnectionSession().getConnectionContext().close();
115             }
116         }
117     }
118     
119     @Override
120     public void rollback() throws SQLException {
121         DatabaseType databaseType = ProxyContext.getInstance().getDatabaseType();
122         for (Entry<ShardingSphereRule, TransactionHook> entry : transactionHooks.entrySet()) {
123             entry.getValue().beforeRollback(entry.getKey(), databaseType, connection.getCachedConnections().values(), getTransactionContext());
124         }
125         if (connection.getConnectionSession().getTransactionStatus().isInTransaction()) {
126             try {
127                 if (TransactionType.LOCAL == TransactionUtils.getTransactionType(getTransactionContext()) || null == distributedTransactionManager) {
128                     localTransactionManager.rollback();
129                 } else {
130                     distributedTransactionManager.rollback();
131                 }
132             } finally {
133                 for (Entry<ShardingSphereRule, TransactionHook> entry : transactionHooks.entrySet()) {
134                     entry.getValue().afterRollback(entry.getKey(), databaseType, connection.getCachedConnections().values(), getTransactionContext());
135                 }
136                 for (Connection each : connection.getCachedConnections().values()) {
137                     ConnectionSavepointManager.getInstance().transactionFinished(each);
138                 }
139                 connection.getConnectionSession().getTransactionStatus().setInTransaction(false);
140                 connection.getConnectionSession().getConnectionContext().close();
141             }
142         }
143     }
144     
145     private TransactionConnectionContext getTransactionContext() {
146         return connection.getConnectionSession().getConnectionContext().getTransactionContext();
147     }
148     
149     @Override
150     public void setSavepoint(final String savepointName) throws SQLException {
151         for (Connection each : connection.getCachedConnections().values()) {
152             ConnectionSavepointManager.getInstance().setSavepoint(each, savepointName);
153         }
154         connection.getConnectionPostProcessors().add(target -> ConnectionSavepointManager.getInstance().setSavepoint(target, savepointName));
155     }
156     
157     @Override
158     public void rollbackTo(final String savepointName) throws SQLException {
159         Collection<SQLException> result = new LinkedList<>();
160         for (Connection each : connection.getCachedConnections().values()) {
161             try {
162                 ConnectionSavepointManager.getInstance().rollbackToSavepoint(each, savepointName);
163             } catch (final SQLException ex) {
164                 result.add(ex);
165             }
166         }
167         if (result.isEmpty() && getTransactionContext().isExceptionOccur()) {
168             getTransactionContext().setExceptionOccur(false);
169         }
170         throwSQLExceptionIfNecessary(result);
171     }
172     
173     @Override
174     public void releaseSavepoint(final String savepointName) throws SQLException {
175         Collection<SQLException> result = new LinkedList<>();
176         for (Connection each : connection.getCachedConnections().values()) {
177             try {
178                 ConnectionSavepointManager.getInstance().releaseSavepoint(each, savepointName);
179             } catch (final SQLException ex) {
180                 result.add(ex);
181             }
182         }
183         throwSQLExceptionIfNecessary(result);
184     }
185     
186     private void throwSQLExceptionIfNecessary(final Collection<SQLException> exceptions) throws SQLException {
187         if (exceptions.isEmpty()) {
188             return;
189         }
190         Iterator<SQLException> iterator = exceptions.iterator();
191         SQLException firstException = iterator.next();
192         while (iterator.hasNext()) {
193             firstException.setNextException(iterator.next());
194         }
195         throw firstException;
196     }
197 }
