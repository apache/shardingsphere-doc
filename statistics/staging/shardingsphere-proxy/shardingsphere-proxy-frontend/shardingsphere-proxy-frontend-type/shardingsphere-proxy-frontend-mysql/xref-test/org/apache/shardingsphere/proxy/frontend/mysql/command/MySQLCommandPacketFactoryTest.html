91      @SuppressWarnings("unchecked")
92      @Test
93      void assertNewInstanceWithComStmtExecutePacket() {
94          when(payload.readInt1()).thenReturn(MySQLNewParametersBoundFlag.PARAMETER_TYPE_EXIST.getValue());
95          when(payload.readInt4()).thenReturn(1);
96          when(payload.getByteBuf().getIntLE(anyInt())).thenReturn(1);
97          ServerPreparedStatementRegistry serverPreparedStatementRegistry = new ServerPreparedStatementRegistry();
98          when(connectionSession.getServerPreparedStatementRegistry()).thenReturn(serverPreparedStatementRegistry);
99          SQLStatementContext<SelectStatement> sqlStatementContext = mock(SQLStatementContext.class);
100         when(sqlStatementContext.getSqlStatement()).thenReturn(new MySQLSelectStatement());
101         serverPreparedStatementRegistry.addPreparedStatement(1, new MySQLServerPreparedStatement("select 1", sqlStatementContext, Collections.emptyList()));
102         assertThat(MySQLCommandPacketFactory.newInstance(MySQLCommandPacketType.COM_STMT_EXECUTE, payload, connectionSession, false), instanceOf(MySQLComStmtExecutePacket.class));
103     }
