35  import java.util.Collection;
36  import java.util.LinkedList;
37  import java.util.stream.Collectors;
38  
39  /**
40   * Create mask rule executor.
41   */
42  @Setter
43  public final class CreateMaskRuleExecutor implements DatabaseRuleCreateExecutor<CreateMaskRuleStatement, MaskRule, MaskRuleConfiguration> {
44      
45      private ShardingSphereDatabase database;
46      
47      private MaskRule rule;
48      
49      private boolean ifNotExists;
50      
51      @Override
52      public void checkBeforeUpdate(final CreateMaskRuleStatement sqlStatement) {
53          ifNotExists = sqlStatement.isIfNotExists();
54          if (!ifNotExists) {
55              checkDuplicatedRuleNames(sqlStatement);
56          }
57          checkAlgorithms(sqlStatement);
58      }
59      
60      private void checkDuplicatedRuleNames(final CreateMaskRuleStatement sqlStatement) {
61          if (null != rule) {
62              Collection<String> currentRuleNames = rule.getConfiguration().getTables().stream().map(MaskTableRuleConfiguration::getName).collect(Collectors.toList());
63              Collection<String> duplicatedRuleNames = sqlStatement.getRules().stream().map(MaskRuleSegment::getTableName).filter(currentRuleNames::contains).collect(Collectors.toList());
64              ShardingSpherePreconditions.checkMustEmpty(duplicatedRuleNames, () -> new DuplicateRuleException("mask", database.getName(), duplicatedRuleNames));
65          }
66      }
67      
68      private void checkAlgorithms(final CreateMaskRuleStatement sqlStatement) {
69          Collection<MaskColumnSegment> columns = new LinkedList<>();
70          sqlStatement.getRules().forEach(each -> columns.addAll(each.getColumns()));
71          columns.forEach(each -> TypedSPILoader.checkService(MaskAlgorithm.class, each.getAlgorithm().getName(), each.getAlgorithm().getProps()));
72      }
73      
74      @Override
75      public MaskRuleConfiguration buildToBeCreatedRuleConfiguration(final CreateMaskRuleStatement sqlStatement) {
76          return MaskRuleStatementConverter.convert(sqlStatement.getRules());
77      }
78      
79      @Override
80      public Class<MaskRule> getRuleClass() {
81          return MaskRule.class;
82      }
83      
84      @Override
85      public Class<CreateMaskRuleStatement> getType() {
86          return CreateMaskRuleStatement.class;
87      }
88  }
