45  /**
46   * Show sharding table rules executor.
47   */
48  @Setter
49  public final class ShowShardingTableRuleExecutor implements RQLExecutor<ShowShardingTableRulesStatement>, DatabaseRuleAwareRQLExecutor<ShardingRule> {
50      
51      private ShardingRule rule;
52      
53      @Override
54      public Collection<String> getColumnNames() {
55          return Arrays.asList("table", "actual_data_nodes", "actual_data_sources", "database_strategy_type", "database_sharding_column", "database_sharding_algorithm_type",
56                  "database_sharding_algorithm_props", "table_strategy_type", "table_sharding_column", "table_sharding_algorithm_type", "table_sharding_algorithm_props",
57                  "key_generate_column", "key_generator_type", "key_generator_props", "auditor_types", "allow_hint_disable");
58      }
59      
60      @Override
61      public Collection<LocalDataQueryResultRow> getRows(final ShowShardingTableRulesStatement sqlStatement, final ContextManager contextManager) {
62          String tableName = sqlStatement.getTableName();
63          Collection<ShardingTableRuleConfiguration> tables;
64          Collection<ShardingAutoTableRuleConfiguration> autoTables;
65          if (null == tableName) {
66              tables = rule.getConfiguration().getTables();
67              autoTables = rule.getConfiguration().getAutoTables();
68          } else {
69              tables = rule.getConfiguration().getTables().stream().filter(each -> tableName.equalsIgnoreCase(each.getLogicTable())).collect(Collectors.toList());
70              autoTables = rule.getConfiguration().getAutoTables().stream().filter(each -> tableName.equalsIgnoreCase(each.getLogicTable())).collect(Collectors.toList());
71          }
72          Collection<LocalDataQueryResultRow> result = new LinkedList<>();
73          for (ShardingTableRuleConfiguration each : tables) {
74              result.add(buildTableRowData(rule.getConfiguration(), each));
75          }
76          for (ShardingAutoTableRuleConfiguration each : autoTables) {
77              result.add(buildAutoTableRowData(rule.getConfiguration(), each));
78          }
79          return result;
80      }
81      
82      private LocalDataQueryResultRow buildTableRowData(final ShardingRuleConfiguration ruleConfig, final ShardingTableRuleConfiguration shardingTableRuleConfig) {
83          Optional<ShardingStrategyConfiguration> databaseShardingStrategyConfig = getDatabaseShardingStrategy(ruleConfig, shardingTableRuleConfig);
84          Optional<ShardingStrategyConfiguration> tableShardingStrategyConfig = getTableShardingStrategy(ruleConfig, shardingTableRuleConfig.getTableShardingStrategy());
85          return new LocalDataQueryResultRow(shardingTableRuleConfig.getLogicTable(), shardingTableRuleConfig.getActualDataNodes(), "",
86                  databaseShardingStrategyConfig.map(this::getStrategyType).orElse(""), databaseShardingStrategyConfig.map(this::getShardingColumn).orElse(""),
87                  databaseShardingStrategyConfig.map(optional -> getAlgorithmType(ruleConfig, optional)).orElse(""),
88                  databaseShardingStrategyConfig.map(optional -> getAlgorithmProperties(ruleConfig, optional)).orElse(""),
89                  tableShardingStrategyConfig.map(this::getStrategyType).orElse(""), tableShardingStrategyConfig.map(this::getShardingColumn).orElse(""),
90                  tableShardingStrategyConfig.map(optional -> getAlgorithmType(ruleConfig, optional)).orElse(""),
91                  tableShardingStrategyConfig.map(optional -> getAlgorithmProperties(ruleConfig, optional)).orElse(""),
92                  getKeyGenerateColumn(ruleConfig, shardingTableRuleConfig.getKeyGenerateStrategy()), getKeyGeneratorType(ruleConfig, shardingTableRuleConfig.getKeyGenerateStrategy()),
93                  getKeyGeneratorProps(ruleConfig, shardingTableRuleConfig.getKeyGenerateStrategy()), getAuditorTypes(ruleConfig, shardingTableRuleConfig.getAuditStrategy()),
94                  getAllowHintDisable(ruleConfig, shardingTableRuleConfig.getAuditStrategy()));
95      }
96      
97      private LocalDataQueryResultRow buildAutoTableRowData(final ShardingRuleConfiguration ruleConfig, final ShardingAutoTableRuleConfiguration shardingAutoTableRuleConfig) {
98          Optional<ShardingStrategyConfiguration> tableShardingStrategyConfig = getTableShardingStrategy(ruleConfig, shardingAutoTableRuleConfig.getShardingStrategy());
99          return new LocalDataQueryResultRow(shardingAutoTableRuleConfig.getLogicTable(), "", shardingAutoTableRuleConfig.getActualDataSources(), "", "", "", "",
100                 tableShardingStrategyConfig.map(this::getStrategyType).orElse(""), tableShardingStrategyConfig.map(this::getShardingColumn).orElse(""),
101                 tableShardingStrategyConfig.map(optional -> getAlgorithmType(ruleConfig, optional)).orElse(""),
102                 tableShardingStrategyConfig.map(optional -> getAlgorithmProperties(ruleConfig, optional)).orElse(""),
103                 getKeyGenerateColumn(ruleConfig, shardingAutoTableRuleConfig.getKeyGenerateStrategy()), getKeyGeneratorType(ruleConfig, shardingAutoTableRuleConfig.getKeyGenerateStrategy()),
104                 getKeyGeneratorProps(ruleConfig, shardingAutoTableRuleConfig.getKeyGenerateStrategy()), getAuditorTypes(ruleConfig, shardingAutoTableRuleConfig.getAuditStrategy()),
105                 getAllowHintDisable(ruleConfig, shardingAutoTableRuleConfig.getAuditStrategy()));
106     }
107     
108     private String getShardingColumn(final ShardingStrategyConfiguration shardingStrategyConfig) {
109         if (shardingStrategyConfig instanceof StandardShardingStrategyConfiguration) {
110             return ((StandardShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumn();
111         }
112         if (shardingStrategyConfig instanceof ComplexShardingStrategyConfiguration) {
113             return ((ComplexShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumns();
114         }
115         return "";
116     }
117     
118     private String getAlgorithmType(final ShardingRuleConfiguration ruleConfig, final ShardingStrategyConfiguration shardingStrategyConfig) {
119         return shardingStrategyConfig instanceof NoneShardingStrategyConfiguration ? "" : getAlgorithmConfiguration(ruleConfig, shardingStrategyConfig.getShardingAlgorithmName()).getType();
120     }
121     
122     private String getAlgorithmProperties(final ShardingRuleConfiguration ruleConfig, final ShardingStrategyConfiguration shardingStrategyConfig) {
123         return shardingStrategyConfig instanceof NoneShardingStrategyConfiguration
124                 ? ""
125                 : PropertiesConverter.convert(getAlgorithmConfiguration(ruleConfig, shardingStrategyConfig.getShardingAlgorithmName()).getProps());
126     }
127     
128     private Optional<ShardingStrategyConfiguration> getDatabaseShardingStrategy(final ShardingRuleConfiguration ruleConfig, final ShardingTableRuleConfiguration shardingTableRuleConfig) {
129         return null == shardingTableRuleConfig.getDatabaseShardingStrategy()
130                 ? Optional.ofNullable(ruleConfig.getDefaultDatabaseShardingStrategy())
131                 : Optional.of(shardingTableRuleConfig.getDatabaseShardingStrategy());
132     }
133     
134     private AlgorithmConfiguration getAlgorithmConfiguration(final ShardingRuleConfiguration ruleConfig, final String algorithmName) {
135         return ruleConfig.getShardingAlgorithms().get(algorithmName);
136     }
137     
138     private String getStrategyType(final ShardingStrategyConfiguration shardingStrategyConfig) {
139         return shardingStrategyConfig.getType();
140     }
141     
142     private Optional<ShardingStrategyConfiguration> getTableShardingStrategy(final ShardingRuleConfiguration ruleConfig, final ShardingStrategyConfiguration shardingStrategyConfig) {
143         return null == shardingStrategyConfig ? Optional.ofNullable(ruleConfig.getDefaultTableShardingStrategy()) : Optional.of(shardingStrategyConfig);
144     }
145     
146     private String getKeyGenerateColumn(final ShardingRuleConfiguration ruleConfig, final KeyGenerateStrategyConfiguration keyGenerateStrategyConfig) {
147         return getKeyGenerateStrategyConfiguration(ruleConfig, keyGenerateStrategyConfig).isPresent()
148                 ? getKeyGenerateStrategyConfiguration(ruleConfig, keyGenerateStrategyConfig).get().getColumn()
149                 : "";
150     }
151     
152     private String getKeyGeneratorType(final ShardingRuleConfiguration ruleConfig, final KeyGenerateStrategyConfiguration originalKeyGenerateStrategyConfig) {
153         Optional<KeyGenerateStrategyConfiguration> keyGenerateStrategyConfig = getKeyGenerateStrategyConfiguration(ruleConfig, originalKeyGenerateStrategyConfig);
154         return keyGenerateStrategyConfig.isPresent() ? ruleConfig.getKeyGenerators().get(keyGenerateStrategyConfig.get().getKeyGeneratorName()).getType() : "";
155     }
156     
157     private String getKeyGeneratorProps(final ShardingRuleConfiguration ruleConfig, final KeyGenerateStrategyConfiguration keyGenerateStrategyConfig) {
158         return getKeyGenerateStrategyConfiguration(ruleConfig, keyGenerateStrategyConfig)
159                 .map(optional -> PropertiesConverter.convert(ruleConfig.getKeyGenerators().get(optional.getKeyGeneratorName()).getProps())).orElse("");
160     }
161     
162     private Optional<KeyGenerateStrategyConfiguration> getKeyGenerateStrategyConfiguration(final ShardingRuleConfiguration ruleConfig,
163                                                                                            final KeyGenerateStrategyConfiguration keyGenerateStrategyConfig) {
164         return null == keyGenerateStrategyConfig ? Optional.ofNullable(ruleConfig.getDefaultKeyGenerateStrategy()) : Optional.of(keyGenerateStrategyConfig);
165     }
166     
167     private String getAuditorTypes(final ShardingRuleConfiguration ruleConfig, final ShardingAuditStrategyConfiguration shardingAuditStrategyConfig) {
168         Optional<ShardingAuditStrategyConfiguration> auditStrategyConfig = getShardingAuditStrategyConfiguration(ruleConfig, shardingAuditStrategyConfig);
169         Collection<String> auditorTypes = new LinkedList<>();
170         if (auditStrategyConfig.isPresent()) {
171             for (String each : auditStrategyConfig.get().getAuditorNames()) {
172                 auditorTypes.add(ruleConfig.getAuditors().get(each).getType());
173             }
174         }
175         return auditorTypes.isEmpty() ? "" : String.join(",", auditorTypes);
176     }
177     
178     private String getAllowHintDisable(final ShardingRuleConfiguration ruleConfig, final ShardingAuditStrategyConfiguration shardingAuditStrategyConfig) {
179         return getShardingAuditStrategyConfiguration(ruleConfig, shardingAuditStrategyConfig).isPresent()
180                 ? Boolean.toString(getShardingAuditStrategyConfiguration(ruleConfig, shardingAuditStrategyConfig).get().isAllowHintDisable())
181                 : "";
182     }
183     
184     private Optional<ShardingAuditStrategyConfiguration> getShardingAuditStrategyConfiguration(final ShardingRuleConfiguration ruleConfig,
185                                                                                                final ShardingAuditStrategyConfiguration shardingAuditStrategyConfig) {
186         return null == shardingAuditStrategyConfig ? Optional.ofNullable(ruleConfig.getDefaultAuditStrategy()) : Optional.of(shardingAuditStrategyConfig);
187     }
188     
189     @Override
190     public Class<ShardingRule> getRuleClass() {
191         return ShardingRule.class;
192     }
193     
194     @Override
195     public Class<ShowShardingTableRulesStatement> getType() {
196         return ShowShardingTableRulesStatement.class;
197     }
