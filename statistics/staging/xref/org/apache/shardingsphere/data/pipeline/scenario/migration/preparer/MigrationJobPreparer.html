81  /**
82   * Migration job preparer.
83   */
84  @Slf4j
85  public final class MigrationJobPreparer implements PipelineJobPreparer<MigrationJobItemContext> {
86      
87      private final PipelineJobItemManager<TransmissionJobItemProgress> jobItemManager = new PipelineJobItemManager<>(new MigrationJobType().getYamlJobItemProgressSwapper());
88      
89      @Override
90      public void prepare(final MigrationJobItemContext jobItemContext) throws SQLException {
91          ShardingSpherePreconditions.checkState(StandardPipelineDataSourceConfiguration.class.equals(
92                  jobItemContext.getTaskConfig().getDumperContext().getCommonContext().getDataSourceConfig().getClass()),
93                  () -> new UnsupportedSQLOperationException("Migration inventory dumper only support StandardPipelineDataSourceConfiguration."));
94          DatabaseType sourceDatabaseType = jobItemContext.getJobConfig().getSourceDatabaseType();
95          new PipelineDataSourceCheckEngine(sourceDatabaseType).checkSourceDataSources(Collections.singleton(jobItemContext.getSourceDataSource()));
96          ShardingSpherePreconditions.checkState(!jobItemContext.isStopping(), PipelineJobCancelingException::new);
97          prepareAndCheckTargetWithLock(jobItemContext);
98          ShardingSpherePreconditions.checkState(!jobItemContext.isStopping(), PipelineJobCancelingException::new);
99          boolean isIncrementalSupported = DatabaseTypedSPILoader.findService(DialectIncrementalDumperCreator.class, sourceDatabaseType).isPresent();
100         if (isIncrementalSupported) {
101             prepareIncremental(jobItemContext);
102         }
103         initInventoryTasks(jobItemContext);
104         if (isIncrementalSupported) {
105             initIncrementalTasks(jobItemContext);
106             ShardingSpherePreconditions.checkState(!jobItemContext.isStopping(), PipelineJobCancelingException::new);
107         }
108         log.info("Prepare job, jobId={}, shardingItem={}, inventoryTasks={}, incrementalTasks={}",
109                 jobItemContext.getJobId(), jobItemContext.getShardingItem(), jobItemContext.getInventoryTasks(), jobItemContext.getIncrementalTasks());
110     }
111     
112     @SuppressWarnings({"unchecked", "rawtypes"})
113     private void prepareAndCheckTargetWithLock(final MigrationJobItemContext jobItemContext) throws SQLException {
114         MigrationJobConfiguration jobConfig = jobItemContext.getJobConfig();
115         String jobId = jobConfig.getJobId();
116         PipelineContextKey contextKey = PipelineJobIdUtils.parseContextKey(jobId);
117         ContextManager contextManager = PipelineContextManager.getContext(contextKey).getContextManager();
118         LockContext lockContext = contextManager.getComputeNodeInstanceContext().getLockContext();
119         if (!jobItemManager.getProgress(jobId, jobItemContext.getShardingItem()).isPresent()) {
120             jobItemManager.persistProgress(jobItemContext);
121         }
122         LockDefinition lockDefinition = new GlobalLockDefinition(String.format(GlobalLockNames.PREPARE.getLockName(), jobConfig.getJobId()));
123         long startTimeMillis = System.currentTimeMillis();
124         if (lockContext.tryLock(lockDefinition, 600 * 1000L)) {
125             log.info("Lock success, jobId={}, shardingItem={}, cost {} ms.", jobId, jobItemContext.getShardingItem(), System.currentTimeMillis() - startTimeMillis);
126             try {
127                 PipelineJobOffsetGovernanceRepository offsetRepository = PipelineAPIFactory.getPipelineGovernanceFacade(contextKey).getJobFacade().getOffset();
128                 JobOffsetInfo offsetInfo = offsetRepository.load(jobId);
129                 if (!offsetInfo.isTargetSchemaTableCreated()) {
130                     jobItemContext.setStatus(JobStatus.PREPARING);
131                     jobItemManager.updateStatus(jobId, jobItemContext.getShardingItem(), JobStatus.PREPARING);
132                     prepareAndCheckTarget(jobItemContext, contextManager);
133                     offsetRepository.persist(jobId, new JobOffsetInfo(true));
134                 }
135             } finally {
136                 log.info("Unlock, jobId={}, shardingItem={}, cost {} ms.", jobId, jobItemContext.getShardingItem(), System.currentTimeMillis() - startTimeMillis);
137                 lockContext.unlock(lockDefinition);
138             }
139         } else {
140             log.warn("Lock failed, jobId={}, shardingItem={}.", jobId, jobItemContext.getShardingItem());
141         }
142     }
143     
144     private void prepareAndCheckTarget(final MigrationJobItemContext jobItemContext, final ContextManager contextManager) throws SQLException {
145         DatabaseType targetDatabaseType = jobItemContext.getJobConfig().getTargetDatabaseType();
146         if (jobItemContext.isSourceTargetDatabaseTheSame()) {
147             prepareTarget(jobItemContext, targetDatabaseType, contextManager);
148         }
149         if (null == jobItemContext.getInitProgress()) {
150             PipelineDataSource targetDataSource = jobItemContext.getDataSourceManager().getDataSource(jobItemContext.getTaskConfig().getImporterConfig().getDataSourceConfig());
151             new PipelineDataSourceCheckEngine(targetDatabaseType).checkTargetDataSources(Collections.singleton(targetDataSource), jobItemContext.getTaskConfig().getImporterConfig());
152         }
153     }
154     
155     private void prepareTarget(final MigrationJobItemContext jobItemContext, final DatabaseType targetDatabaseType, final ContextManager contextManager) throws SQLException {
156         MigrationJobConfiguration jobConfig = jobItemContext.getJobConfig();
157         Collection<CreateTableConfiguration> createTableConfigs = jobItemContext.getTaskConfig().getCreateTableConfigurations();
158         PipelineDataSourceManager dataSourceManager = jobItemContext.getDataSourceManager();
159         PipelineJobDataSourcePreparer preparer = new PipelineJobDataSourcePreparer(targetDatabaseType);
160         preparer.prepareTargetSchemas(new PrepareTargetSchemasParameter(targetDatabaseType, createTableConfigs, dataSourceManager));
161         ShardingSphereMetaData metaData = contextManager.getMetaDataContexts().getMetaData();
162         SQLParserEngine sqlParserEngine = metaData.getGlobalRuleMetaData().getSingleRule(SQLParserRule.class)
163                 .getSQLParserEngine(metaData.getDatabase(jobConfig.getTargetDatabaseName()).getProtocolType());
164         preparer.prepareTargetTables(new PrepareTargetTablesParameter(createTableConfigs, dataSourceManager, sqlParserEngine));
165     }
166     
167     private void prepareIncremental(final MigrationJobItemContext jobItemContext) {
168         MigrationTaskConfiguration taskConfig = jobItemContext.getTaskConfig();
169         JobItemIncrementalTasksProgress initIncremental = null == jobItemContext.getInitProgress() ? null : jobItemContext.getInitProgress().getIncremental();
170         try {
171             DatabaseType databaseType = taskConfig.getDumperContext().getCommonContext().getDataSourceConfig().getDatabaseType();
172             IngestPosition position = new IncrementalTaskPositionManager(databaseType).getPosition(initIncremental, taskConfig.getDumperContext(), jobItemContext.getDataSourceManager());
173             taskConfig.getDumperContext().getCommonContext().setPosition(position);
174         } catch (final SQLException ex) {
175             throw new PrepareJobWithGetBinlogPositionException(jobItemContext.getJobId(), ex);
176         }
177     }
178     
179     private void initInventoryTasks(final MigrationJobItemContext jobItemContext) {
180         InventoryDumperContext inventoryDumperContext = new InventoryDumperContext(jobItemContext.getTaskConfig().getDumperContext().getCommonContext());
181         InventoryTaskSplitter inventoryTaskSplitter = new InventoryTaskSplitter(jobItemContext.getSourceDataSource(), inventoryDumperContext, jobItemContext.getTaskConfig().getImporterConfig());
182         jobItemContext.getInventoryTasks().addAll(inventoryTaskSplitter.split(jobItemContext));
183     }
184     
185     private void initIncrementalTasks(final MigrationJobItemContext jobItemContext) {
186         MigrationTaskConfiguration taskConfig = jobItemContext.getTaskConfig();
187         IncrementalDumperContext dumperContext = taskConfig.getDumperContext();
188         PipelineExecuteEngine incrementalExecuteEngine = jobItemContext.getJobProcessContext().getIncrementalExecuteEngine();
189         IncrementalTaskProgress taskProgress = PipelineTaskUtils.createIncrementalTaskProgress(dumperContext.getCommonContext().getPosition(), jobItemContext.getInitProgress());
190         PipelineChannel channel = IncrementalChannelCreator.create(jobItemContext.getJobProcessContext().getProcessConfiguration().getStreamChannel(), taskProgress);
191         CreateIncrementalDumperParameter param = new CreateIncrementalDumperParameter(
192                 dumperContext, dumperContext.getCommonContext().getPosition(), channel, jobItemContext.getSourceMetaDataLoader(), jobItemContext.getDataSourceManager());
193         Dumper dumper = IncrementalDumperCreator.create(param);
194         Collection<Importer> importers = Collections.singletonList(new SingleChannelConsumerImporter(channel, 1, 5L, jobItemContext.getSink(), jobItemContext));
195         PipelineTask incrementalTask = new IncrementalTask(dumperContext.getCommonContext().getDataSourceName(), incrementalExecuteEngine, dumper, importers, taskProgress);
196         jobItemContext.getIncrementalTasks().add(incrementalTask);
197     }
