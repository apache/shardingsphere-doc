43  /**
44   * CDC job.
45   */
46  @RequiredArgsConstructor
47  @Slf4j
48  public final class CDCJob extends AbstractSimplePipelineJob {
49      
50      private final ImporterConnector importerConnector;
51      
52      private final CDCJobAPI jobAPI = new CDCJobAPI();
53      
54      private final CDCJobPreparer jobPreparer = new CDCJobPreparer();
55      
56      private final PipelineDataSourceManager dataSourceManager = new DefaultPipelineDataSourceManager();
57      
58      @Override
59      protected void doPrepare(final PipelineJobItemContext jobItemContext) {
60          jobPreparer.initTasks((CDCJobItemContext) jobItemContext);
61      }
62      
63      @Override
64      protected PipelineJobItemContext buildPipelineJobItemContext(final ShardingContext shardingContext) {
65          int shardingItem = shardingContext.getShardingItem();
66          CDCJobConfiguration jobConfig = new YamlCDCJobConfigurationSwapper().swapToObject(shardingContext.getJobParameter());
67          Optional<InventoryIncrementalJobItemProgress> initProgress = jobAPI.getJobItemProgress(shardingContext.getJobName(), shardingItem);
68          CDCProcessContext jobProcessContext = jobAPI.buildPipelineProcessContext(jobConfig);
69          CDCTaskConfiguration taskConfig = jobAPI.buildTaskConfiguration(jobConfig, shardingItem, jobProcessContext.getPipelineProcessConfig());
70          return new CDCJobItemContext(jobConfig, shardingItem, initProgress.orElse(null), jobProcessContext, taskConfig, dataSourceManager, importerConnector);
71      }
72      
73      protected PipelineTasksRunner buildPipelineTasksRunner(final PipelineJobItemContext pipelineJobItemContext) {
74          InventoryIncrementalJobItemContext jobItemContext = (InventoryIncrementalJobItemContext) pipelineJobItemContext;
75          return new CDCTasksRunner(jobItemContext, jobItemContext.getInventoryTasks(), jobItemContext.getIncrementalTasks());
76      }
77      
78      @Override
79      protected void doClean() {
80          dataSourceManager.close();
81          if (importerConnector instanceof AutoCloseable) {
82              CloseUtils.closeQuietly((AutoCloseable) importerConnector);
83          }
84      }
85  }
