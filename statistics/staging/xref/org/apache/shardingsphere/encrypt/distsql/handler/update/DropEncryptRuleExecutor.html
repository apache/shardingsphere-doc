32  import java.util.Collection;
33  import java.util.Collections;
34  import java.util.HashMap;
35  import java.util.LinkedList;
36  import java.util.Map;
37  import java.util.Optional;
38  import java.util.stream.Collectors;

40  /**
41   * Drop encrypt rule executor.
42   */
43  @DistSQLExecutorCurrentRuleRequired(EncryptRule.class)
44  @Setter
45  public final class DropEncryptRuleExecutor implements DatabaseRuleDropExecutor<DropEncryptRuleStatement, EncryptRule, EncryptRuleConfiguration> {
46      
47      private ShardingSphereDatabase database;
48      
49      private EncryptRule rule;
50      
51      @Override
52      public void checkBeforeUpdate(final DropEncryptRuleStatement sqlStatement) {
53          if (!sqlStatement.isIfExists()) {
54              checkToBeDroppedEncryptTableNames(sqlStatement);
55          }
56      }
57      
58      private void checkToBeDroppedEncryptTableNames(final DropEncryptRuleStatement sqlStatement) {
59          Collection<String> notExistedTableNames = sqlStatement.getTables().stream().filter(each -> !rule.getAllTableNames().contains(each)).collect(Collectors.toList());
60          ShardingSpherePreconditions.checkMustEmpty(notExistedTableNames, () -> new MissingRequiredRuleException("Encrypt", database.getName(), notExistedTableNames));
61      }
62      
63      @Override
64      public boolean hasAnyOneToBeDropped(final DropEncryptRuleStatement sqlStatement) {
65          return !Collections.disjoint(rule.getAllTableNames(), sqlStatement.getTables());
66      }
67      
68      @Override
69      public EncryptRuleConfiguration buildToBeDroppedRuleConfiguration(final DropEncryptRuleStatement sqlStatement) {
70          Collection<EncryptTableRuleConfiguration> toBeDroppedTables = new LinkedList<>();
71          Map<String, AlgorithmConfiguration> toBeDroppedEncryptors = new HashMap<>();
72          for (String each : sqlStatement.getTables()) {
73              toBeDroppedTables.add(new EncryptTableRuleConfiguration(each, Collections.emptyList()));
74              dropRule(each);
75          }
76          UnusedAlgorithmFinder.findUnusedEncryptor(rule.getConfiguration()).forEach(each -> toBeDroppedEncryptors.put(each, rule.getConfiguration().getEncryptors().get(each)));
77          return new EncryptRuleConfiguration(toBeDroppedTables, toBeDroppedEncryptors);
78      }
79      
80      private void dropRule(final String ruleName) {
81          Optional<EncryptTableRuleConfiguration> encryptTableRuleConfig = rule.getConfiguration().getTables().stream()
82                  .filter(each -> each.getName().equals(ruleName)).findAny();
83          encryptTableRuleConfig.ifPresent(optional -> rule.getConfiguration().getTables().remove(encryptTableRuleConfig.get()));
84      }
85      
86      @Override
87      public Class<EncryptRule> getRuleClass() {
88          return EncryptRule.class;
89      }
90      
91      @Override
92      public Class<DropEncryptRuleStatement> getType() {
93          return DropEncryptRuleStatement.class;
94      }
95  }
