18  package org.apache.shardingsphere.infra.binder.context.segment.insert.keygen.engine;
19  
20  import com.cedarsoftware.util.CaseInsensitiveSet;
21  import lombok.RequiredArgsConstructor;
22  import org.apache.shardingsphere.infra.binder.context.segment.insert.keygen.GeneratedKeyContext;
23  import org.apache.shardingsphere.infra.binder.context.segment.insert.values.InsertValueContext;
24  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereColumn;
25  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereSchema;
26  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.assignment.SetAssignmentSegment;
27  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.ExpressionSegment;
28  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.simple.LiteralExpressionSegment;
29  import org.apache.shardingsphere.sql.parser.statement.core.segment.dml.expr.simple.ParameterMarkerExpressionSegment;
30  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.InsertStatement;
31  
32  import java.util.List;
33  import java.util.Optional;
34  
35  /**
36   * Generated key context.
37   */
38  @RequiredArgsConstructor
39  public final class GeneratedKeyContextEngine {
40      
41      private final InsertStatement insertStatement;
42      
43      private final ShardingSphereSchema schema;
44      
45      /**
46       * Create generate key context.
47       *
48       * @param insertColumnNames insert column names
49       * @param insertValueContexts insert value contexts
50       * @param params SQL parameters
51       * @return generate key context
52       */
53      public Optional<GeneratedKeyContext> createGenerateKeyContext(final List<String> insertColumnNames, final List<InsertValueContext> insertValueContexts, final List<Object> params) {
54          String tableName = Optional.ofNullable(insertStatement.getTable()).map(optional -> optional.getTableName().getIdentifier().getValue()).orElse("");
55          return findGenerateKeyColumn(tableName).map(optional -> containsGenerateKey(insertColumnNames, optional)
56                  ? findGeneratedKey(insertColumnNames, insertValueContexts, params, optional)
57                  : new GeneratedKeyContext(optional, true));
58      }
59      
60      private Optional<String> findGenerateKeyColumn(final String tableName) {
61          if (!schema.containsTable(tableName)) {
62              return Optional.empty();
63          }
64          for (ShardingSphereColumn each : schema.getTable(tableName).getColumnValues()) {
65              if (each.isGenerated()) {
66                  return Optional.of(each.getName());
67              }
68          }
69          return Optional.empty();
70      }
71      
72      private boolean containsGenerateKey(final List<String> insertColumnNames, final String generateKeyColumnName) {
73          return insertColumnNames.isEmpty() ? schema.getVisibleColumnNames(insertStatement.getTable().getTableName().getIdentifier().getValue()).size() == getValueCountForPerGroup()
74                  : new CaseInsensitiveSet<>(insertColumnNames).contains(generateKeyColumnName);
75      }
76      
77      private int getValueCountForPerGroup() {
78          if (!insertStatement.getValues().isEmpty()) {
79              return insertStatement.getValues().iterator().next().getValues().size();
80          }
81          Optional<SetAssignmentSegment> setAssignment = insertStatement.getSetAssignment();
82          if (setAssignment.isPresent()) {
83              return setAssignment.get().getAssignments().size();
84          }
85          if (insertStatement.getInsertSelect().isPresent()) {
86              return insertStatement.getInsertSelect().get().getSelect().getProjections().getProjections().size();
87          }
88          return 0;
89      }
90      
91      private GeneratedKeyContext findGeneratedKey(final List<String> insertColumnNames, final List<InsertValueContext> insertValueContexts,
92                                                   final List<Object> params, final String generateKeyColumnName) {
93          GeneratedKeyContext result = new GeneratedKeyContext(generateKeyColumnName, false);
94          for (InsertValueContext each : insertValueContexts) {
95              ExpressionSegment expression = each.getValueExpressions().get(findGenerateKeyIndex(insertColumnNames, generateKeyColumnName.toLowerCase()));
96              if (expression instanceof ParameterMarkerExpressionSegment) {
97                  if (params.isEmpty()) {
98                      continue;
99                  }
100                 if (null != params.get(((ParameterMarkerExpressionSegment) expression).getParameterMarkerIndex())) {
101                     result.getGeneratedValues().add((Comparable<?>) params.get(((ParameterMarkerExpressionSegment) expression).getParameterMarkerIndex()));
102                 }
103             } else if (expression instanceof LiteralExpressionSegment) {
104                 result.getGeneratedValues().add((Comparable<?>) ((LiteralExpressionSegment) expression).getLiterals());
105             }
106         }
107         return result;
108     }
109     
110     private int findGenerateKeyIndex(final List<String> insertColumnNames, final String generateKeyColumnName) {
111         return insertColumnNames.isEmpty() ? schema.getVisibleColumnNames(insertStatement.getTable().getTableName().getIdentifier().getValue()).indexOf(generateKeyColumnName)
112                 : insertColumnNames.indexOf(generateKeyColumnName);
113     }
114 }
