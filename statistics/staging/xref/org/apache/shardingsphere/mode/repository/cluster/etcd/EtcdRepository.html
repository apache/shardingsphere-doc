56  /**
57   * Registry repository of ETCD.
58   */
59  @Slf4j
60  public final class EtcdRepository implements ClusterPersistRepository {
61      
62      private static final ExecutorService EVENT_LISTENER_EXECUTOR = Executors.newSingleThreadExecutor(new ThreadFactoryBuilder().setDaemon(true).setNameFormat("Etcd-EventListener-%d").build());
63      
64      private Client client;
65      
66      private EtcdProperties etcdProps;
67      
68      @Getter
69      private DistributedLockHolder distributedLockHolder;
70      
71      @Override
72      public void init(final ClusterPersistRepositoryConfiguration config, final ComputeNodeInstanceContext computeNodeInstanceContext) {
73          etcdProps = new EtcdProperties(config.getProps());
74          client = Client.builder().endpoints(Util.toURIs(Splitter.on(",").trimResults().splitToList(config.getServerLists())))
75                  .namespace(ByteSequence.from(config.getNamespace(), StandardCharsets.UTF_8))
76                  .maxInboundMessageSize((int) 32e9)
77                  .build();
78          distributedLockHolder = new DistributedLockHolder(getType(), client, etcdProps);
79      }
80      
81      @SneakyThrows({InterruptedException.class, ExecutionException.class})
82      @Override
83      public String query(final String key) {
84          List<KeyValue> keyValues = client.getKVClient().get(ByteSequence.from(key, StandardCharsets.UTF_8)).get().getKvs();
85          return keyValues.isEmpty() ? null : keyValues.iterator().next().getValue().toString(StandardCharsets.UTF_8);
86      }
87      
88      @SneakyThrows({InterruptedException.class, ExecutionException.class})
89      @Override
90      public List<String> getChildrenKeys(final String key) {
91          String prefix = key + PATH_SEPARATOR;
92          ByteSequence prefixByteSequence = ByteSequence.from(prefix, StandardCharsets.UTF_8);
93          GetOption getOption = GetOption.newBuilder().isPrefix(true).withSortField(GetOption.SortTarget.KEY).withSortOrder(GetOption.SortOrder.ASCEND).build();
94          List<KeyValue> keyValues = client.getKVClient().get(prefixByteSequence, getOption).get().getKvs();
95          return keyValues.stream().map(each -> getSubNodeKeyName(prefix, each.getKey().toString(StandardCharsets.UTF_8))).distinct().collect(Collectors.toList());
96      }
97      
98      @Override
99      public boolean isExisted(final String key) {
100         return false;
101     }
102     
103     private String getSubNodeKeyName(final String prefix, final String fullPath) {
104         String pathWithoutPrefix = fullPath.substring(prefix.length());
105         return pathWithoutPrefix.contains(PATH_SEPARATOR) ? pathWithoutPrefix.substring(0, pathWithoutPrefix.indexOf(PATH_SEPARATOR)) : pathWithoutPrefix;
106     }
107     
108     @SneakyThrows({InterruptedException.class, ExecutionException.class})
109     @Override
110     public void persist(final String key, final String value) {
111         buildParentPath(key);
112         client.getKVClient().put(ByteSequence.from(key, StandardCharsets.UTF_8), ByteSequence.from(value, StandardCharsets.UTF_8)).get();
113     }
114     
115     @Override
116     public void update(final String key, final String value) {
117         // TODO
118     }
119     
120     @SneakyThrows({InterruptedException.class, ExecutionException.class})
121     @Override
122     public void persistEphemeral(final String key, final String value) {
123         buildParentPath(key);
124         long leaseId = client.getLeaseClient().grant(etcdProps.getValue(EtcdPropertyKey.TIME_TO_LIVE_SECONDS)).get().getID();
125         client.getLeaseClient().keepAlive(leaseId, Observers.observer(response -> {
126         }));
127         client.getKVClient().put(ByteSequence.from(key, StandardCharsets.UTF_8), ByteSequence.from(value, StandardCharsets.UTF_8), PutOption.newBuilder().withLeaseId(leaseId).build()).get();
128     }
129     
130     @Override
131     public boolean persistExclusiveEphemeral(final String key, final String value) {
132         persistEphemeral(key, value);
133         return true;
134     }
135     
136     private void buildParentPath(final String key) throws ExecutionException, InterruptedException {
137         StringBuilder parentPath = new StringBuilder();
138         String[] partPath = key.split(PATH_SEPARATOR);
139         for (int index = 1; index < partPath.length - 1; index++) {
140             parentPath.append(PATH_SEPARATOR);
141             parentPath.append(partPath[index]);
142             String path = parentPath.toString();
143             List<KeyValue> keyValues = client.getKVClient().get(ByteSequence.from(path, StandardCharsets.UTF_8)).get().getKvs();
144             if (keyValues.isEmpty()) {
145                 client.getKVClient().put(ByteSequence.from(path, StandardCharsets.UTF_8), ByteSequence.from("", StandardCharsets.UTF_8)).get();
146             }
147         }
148     }
149     
150     @Override
151     public void delete(final String key) {
152         client.getKVClient().delete(ByteSequence.from(key, StandardCharsets.UTF_8), DeleteOption.newBuilder().isPrefix(true).build());
153     }
154     
155     @Override
156     public void watch(final String key, final DataChangedEventListener dataChangedEventListener) {
157         Watch.Listener listener = Watch.listener(response -> {
158             for (WatchEvent each : response.getEvents()) {
159                 Type type = getEventChangedType(each);
160                 if (Type.IGNORED != type) {
161                     dispatchEvent(dataChangedEventListener, each, type);
162                 }
163             }
164         });
165         ByteSequence prefix = ByteSequence.from(key, StandardCharsets.UTF_8);
166         Preconditions.checkNotNull(prefix, "prefix should not be null");
167         client.getWatchClient().watch(prefix,
168                 WatchOption.newBuilder().withRange(OptionsUtil.prefixEndOf(prefix)).build(), listener);
169     }
170     
171     @Override
172     public void removeDataListener(final String key) {
173         // TODO
174     }
175     
176     private Type getEventChangedType(final WatchEvent event) {
177         if (1 == event.getKeyValue().getVersion()) {
178             return Type.ADDED;
179         }
180         switch (event.getEventType()) {
181             case PUT:
182                 return Type.UPDATED;
183             case DELETE:
184                 return Type.DELETED;
185             default:
186                 return Type.IGNORED;
187         }
188     }
189     
190     private void dispatchEvent(final DataChangedEventListener dataChangedEventListener, final WatchEvent event, final Type type) {
191         CompletableFuture.runAsync(() -> dataChangedEventListener.onChange(new DataChangedEvent(event.getKeyValue().getKey().toString(StandardCharsets.UTF_8),
192                 event.getKeyValue().getValue().toString(StandardCharsets.UTF_8), type)), EVENT_LISTENER_EXECUTOR).whenComplete((unused, throwable) -> {
193                     if (null != throwable) {
194                         log.error("Dispatch event failed", throwable);
195                     }
196                 });
197     }
198     
199     @Override
200     public void close() {
201         client.close();
202         EVENT_LISTENER_EXECUTOR.shutdown();
203     }
204     
205     @Override
206     public String getType() {
207         return "etcd";
208     }
