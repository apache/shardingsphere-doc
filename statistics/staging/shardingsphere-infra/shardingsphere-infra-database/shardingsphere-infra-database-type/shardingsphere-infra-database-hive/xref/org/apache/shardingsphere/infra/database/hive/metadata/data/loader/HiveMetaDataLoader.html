44  /**
45   * Hive meta data loader.
46   */
47  public final class HiveMetaDataLoader implements DialectMetaDataLoader {
48      
49      private static final String HIVE_METASTORE_URIS = "hive.metastore.uris";
50      
51      @SuppressWarnings("SqlNoDataSourceInspection")
52      @Override
53      public Collection<SchemaMetaData> load(final MetaDataLoaderMaterial material) throws SQLException {
54          String hiveMetastoreUris;
55          try (Statement statement = material.getDataSource().getConnection().createStatement()) {
56              ResultSet resultSet = statement.executeQuery("SET hive.metastore.uris");
57              resultSet.next();
58              hiveMetastoreUris = resultSet.getString("set");
59          }
60          if ("hive.metastore.uris is undefined".equals(hiveMetastoreUris)) {
61              Collection<TableMetaData> tableMetaData = new LinkedList<>();
62              for (String each : material.getActualTableNames()) {
63                  TableMetaDataLoader.load(material.getDataSource(), each, material.getStorageType()).ifPresent(tableMetaData::add);
64              }
65              return Collections.singletonList(new SchemaMetaData(material.getDefaultSchemaName(), tableMetaData));
66          }
67          HiveMetaStoreClient storeClient = null;
68          try {
69              // TODO Support set hive.metastore uris when register storage unit.
70              HiveConf hiveConf = new HiveConf();
71              hiveConf.set(HIVE_METASTORE_URIS, hiveMetastoreUris);
72              storeClient = new HiveMetaStoreClient(hiveConf);
73              return Collections.singletonList(new SchemaMetaData(material.getDefaultSchemaName(),
74                      getTableMetaData(storeClient.getAllTables(material.getDefaultSchemaName()), storeClient, material)));
75          } catch (final TException ignored) {
76              throw new SQLException();
77          } finally {
78              if (null != storeClient) {
79                  storeClient.close();
80              }
81          }
82      }
83      
84      private Collection<TableMetaData> getTableMetaData(final Collection<String> tables, final HiveMetaStoreClient storeClient, final MetaDataLoaderMaterial material) throws TException, SQLException {
85          Map<String, Integer> dataTypes = getDataType(material.getDataSource());
86          Collection<TableMetaData> result = new LinkedList<>();
87          for (String each : tables) {
88              result.add(new TableMetaData(each, getColumnMetaData(storeClient.getTable(material.getDefaultSchemaName(), each), dataTypes), Collections.emptyList(), Collections.emptyList()));
89          }
90          return result;
91      }
92      
93      private Map<String, Integer> getDataType(final DataSource dataSource) throws SQLException {
94          try (
95                  Connection connection = dataSource.getConnection()) {
96              return new DataTypeLoader().load(connection.getMetaData(), getType());
97          }
98      }
99      
100     private Collection<ColumnMetaData> getColumnMetaData(final Table table, final Map<String, Integer> dataTypes) {
101         Collection<ColumnMetaData> result = new LinkedList<>();
102         for (FieldSchema each : table.getSd().getCols()) {
103             result.add(new ColumnMetaData(each.getName(), null == dataTypes.get(each.getType()) ? Types.VARCHAR : dataTypes.get(each.getType()),
104                     false, false, false, false, false, false));
105         }
106         return result;
107     }
108     
109     @Override
110     public String getDatabaseType() {
111         return "Hive";
112     }
113 }
