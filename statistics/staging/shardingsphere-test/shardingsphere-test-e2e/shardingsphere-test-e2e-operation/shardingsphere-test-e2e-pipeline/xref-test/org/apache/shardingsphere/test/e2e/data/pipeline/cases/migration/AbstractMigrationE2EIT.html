44  @Getter
45  @Slf4j
46  public abstract class AbstractMigrationE2EIT {
47      
48      private final MigrationDistSQLCommand migrationDistSQL;
49      
50      public AbstractMigrationE2EIT() {
51          migrationDistSQL = JAXB.unmarshal(Objects.requireNonNull(AbstractMigrationE2EIT.class.getClassLoader().getResource("env/common/migration-command.xml")), MigrationDistSQLCommand.class);
52      }
53      
54      protected void addMigrationSourceResource(final PipelineContainerComposer containerComposer) throws SQLException {
55          if (PipelineEnvTypeEnum.NATIVE == PipelineE2EEnvironment.getInstance().getItEnvType()) {
56              try {
57                  containerComposer.proxyExecuteWithLog("UNREGISTER MIGRATION SOURCE STORAGE UNIT ds_0", 2);
58              } catch (final SQLException ex) {
59                  log.warn("Drop sharding_db failed, maybe it's not exist. error msg={}", ex.getMessage());
60              }
61          }
62          String addSourceResource = migrationDistSQL.getRegisterMigrationSourceStorageUnitTemplate().replace("${user}", containerComposer.getUsername())
63                  .replace("${password}", containerComposer.getPassword())
64                  .replace("${ds0}", containerComposer.appendExtraParameter(containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_0, true)));
65          containerComposer.addResource(addSourceResource);
66      }
67      
68      protected void addMigrationTargetResource(final PipelineContainerComposer containerComposer) throws SQLException {
69          String addTargetResource = migrationDistSQL.getRegisterMigrationTargetStorageUnitTemplate().replace("${user}", containerComposer.getUsername())
70                  .replace("${password}", containerComposer.getPassword())
71                  .replace("${ds2}", containerComposer.appendExtraParameter(containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_2, true)))
72                  .replace("${ds3}", containerComposer.appendExtraParameter(containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_3, true)))
73                  .replace("${ds4}", containerComposer.appendExtraParameter(containerComposer.getActualJdbcUrlTemplate(PipelineContainerComposer.DS_4, true)));
74          containerComposer.addResource(addTargetResource);
75          List<Map<String, Object>> resources = containerComposer.queryForListWithLog("SHOW STORAGE UNITS from sharding_db");
76          assertThat(resources.size(), is(3));
77      }
78      
79      protected void createSourceSchema(final PipelineContainerComposer containerComposer, final String schemaName) throws SQLException {
80          if (DatabaseTypeUtils.isPostgreSQL(containerComposer.getDatabaseType())) {
81              containerComposer.sourceExecuteWithLog(String.format("CREATE SCHEMA IF NOT EXISTS %s", schemaName));
82              return;
83          }
84          if (DatabaseTypeUtils.isOpenGauss(containerComposer.getDatabaseType())) {
85              try {
86                  containerComposer.sourceExecuteWithLog(String.format("CREATE SCHEMA %s", schemaName));
87              } catch (final SQLException ex) {
88                  // only used for native mode.
89                  if (ex instanceof PSQLException && "42P06".equals(ex.getSQLState())) {
90                      log.info("Schema {} already exists.", schemaName);
91                  } else {
92                      throw ex;
93                  }
94              }
95          }
96      }
97

99          containerComposer.proxyExecuteWithLog(migrationDistSQL.getCreateTargetOrderTableRule(), 2);
100     }

107         containerComposer.proxyExecuteWithLog(migrationDistSQL.getCreateTargetOrderItemTableRule(), 2);
108     }
109     
110     protected void startMigration(final PipelineContainerComposer containerComposer, final String sourceTableName, final String targetTableName) throws SQLException {
111         containerComposer.proxyExecuteWithLog(migrationDistSQL.getMigrationSingleTable(sourceTableName, targetTableName), 5);
112     }
113     
114     protected void startMigrationWithSchema(final PipelineContainerComposer containerComposer, final String sourceTableName, final String targetTableName) throws SQLException {
115         containerComposer.proxyExecuteWithLog(migrationDistSQL.getMigrationSingleTableWithSchema(sourceTableName, targetTableName), 5);
116     }
117     
118     protected void addMigrationProcessConfig(final PipelineContainerComposer containerComposer) throws SQLException {
119         containerComposer.proxyExecuteWithLog(migrationDistSQL.getAlterMigrationRule(), 0);
120     }
121     
122     protected void stopMigrationByJobId(final PipelineContainerComposer containerComposer, final String jobId) throws SQLException {
123         containerComposer.proxyExecuteWithLog(String.format("STOP MIGRATION '%s'", jobId), 1);
124     }
125     
126     protected void startMigrationByJobId(final PipelineContainerComposer containerComposer, final String jobId) throws SQLException {
127         containerComposer.proxyExecuteWithLog(String.format("START MIGRATION '%s'", jobId), 1);
128     }
129     
130     protected void commitMigrationByJobId(final PipelineContainerComposer containerComposer, final String jobId) throws SQLException {
131         containerComposer.proxyExecuteWithLog(String.format("COMMIT MIGRATION '%s'", jobId), 1);
132     }
133     
134     protected List<String> listJobId(final PipelineContainerComposer containerComposer) {
135         List<Map<String, Object>> jobList = containerComposer.queryForListWithLog("SHOW MIGRATION LIST");
136         return jobList.stream().map(a -> a.get("id").toString()).collect(Collectors.toList());
137     }
138     
139     protected String getJobIdByTableName(final PipelineContainerComposer containerComposer, final String tableName) {
140         List<Map<String, Object>> jobList = containerComposer.queryForListWithLog("SHOW MIGRATION LIST");
141         return jobList.stream().filter(a -> a.get("tables").toString().equals(tableName)).findFirst().orElseThrow(() -> new RuntimeException("not find " + tableName + " table")).get("id").toString();
142     }
143     
144     @SneakyThrows(InterruptedException.class)
145     protected void assertCheckMigrationSuccess(final PipelineContainerComposer containerComposer, final String jobId, final String algorithmType) throws SQLException {
146         containerComposer.proxyExecuteWithLog(String.format("CHECK MIGRATION '%s' BY TYPE (NAME='%s')", jobId, algorithmType), 0);
147         // TODO Need to add after the stop then to start, can continue the consistency check from the previous progress
148         List<Map<String, Object>> resultList = Collections.emptyList();
149         for (int i = 0; i < 10; i++) {
150             resultList = containerComposer.queryForListWithLog(String.format("SHOW MIGRATION CHECK STATUS '%s'", jobId));
151             if (resultList.isEmpty()) {
152                 Thread.sleep(3000L);
153                 continue;
154             }
155             List<String> checkEndTimeList = resultList.stream().map(map -> map.get("check_end_time").toString()).filter(each -> !Strings.isNullOrEmpty(each)).collect(Collectors.toList());
156             if (checkEndTimeList.size() == resultList.size()) {
157                 break;
158             } else {
159                 Thread.sleep(3000L);
160             }
161         }
162         log.info("check job results: {}", resultList);
163         assertFalse(resultList.isEmpty());
164         for (Map<String, Object> each : resultList) {
165             assertTrue(Boolean.parseBoolean(each.get("result").toString()), String.format("%s check result is false", each.get("tables")));
166             assertThat("finished_percentage is not 100", each.get("finished_percentage").toString(), is("100"));
167         }
168     }
