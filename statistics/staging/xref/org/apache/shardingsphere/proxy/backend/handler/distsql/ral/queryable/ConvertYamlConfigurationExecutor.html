88      @Override
89      public Collection<LocalDataQueryResultRow> getRows(final ConvertYamlConfigurationStatement sqlStatement) {
90          File file = new File(sqlStatement.getFilePath());
91          YamlProxyDatabaseConfiguration yamlConfig;
92          try {
93              yamlConfig = YamlEngine.unmarshal(file, YamlProxyDatabaseConfiguration.class);
94          } catch (final IOException ex) {
95              throw new FileIOException(ex);
96          }
97          Preconditions.checkNotNull(yamlConfig, "Invalid yaml file `%s`", file.getName());
98          Preconditions.checkNotNull(yamlConfig.getDatabaseName(), "`databaseName` in file `%s` is required.", file.getName());
99          return Collections.singleton(new LocalDataQueryResultRow(generateDistSQL(yamlConfig)));
100     }
101     
102     private String generateDistSQL(final YamlProxyDatabaseConfiguration yamlConfig) {
103         StringBuilder result = new StringBuilder();
104         appendResourceDistSQL(yamlConfig, result);
105         for (RuleConfiguration each : swapToRuleConfigs(yamlConfig).values()) {
106             if (each instanceof EncryptRuleConfiguration) {
107                 appendEncryptDistSQL((EncryptRuleConfiguration) each, result);
108             } else if (each instanceof CompatibleEncryptRuleConfiguration) {
109                 appendEncryptDistSQL(((CompatibleEncryptRuleConfiguration) each).convertToEncryptRuleConfiguration(), result);
110             } else if (each instanceof ShadowRuleConfiguration) {
111                 appendShadowDistSQL((ShadowRuleConfiguration) each, result);
112             } else if (each instanceof MaskRuleConfiguration) {
113                 appendMaskDistSQL((MaskRuleConfiguration) each, result);
114             } else {
115                 ConvertRuleConfigurationProvider convertRuleConfigProvider = TypedSPILoader.getService(ConvertRuleConfigurationProvider.class, each.getClass().getName());
116                 result.append(convertRuleConfigProvider.convert(each));
117             }
118         }
119         return result.toString();
120     }
121     
122     private Map<Integer, RuleConfiguration> swapToRuleConfigs(final YamlProxyDatabaseConfiguration yamlConfig) {
123         Map<Integer, RuleConfiguration> result = new TreeMap<>(Comparator.reverseOrder());
124         for (YamlRuleConfiguration each : yamlConfig.getRules()) {
125             if (each instanceof YamlShardingRuleConfiguration) {
126                 YamlShardingRuleConfigurationSwapper swapper = new YamlShardingRuleConfigurationSwapper();
127                 result.put(swapper.getOrder(), swapper.swapToObject((YamlShardingRuleConfiguration) each));
128             } else if (each instanceof YamlReadwriteSplittingRuleConfiguration) {
129                 YamlReadwriteSplittingRuleConfigurationSwapper swapper = new YamlReadwriteSplittingRuleConfigurationSwapper();
130                 result.put(swapper.getOrder(), swapper.swapToObject((YamlReadwriteSplittingRuleConfiguration) each));
131             } else if (each instanceof YamlEncryptRuleConfiguration) {
132                 YamlEncryptRuleConfigurationSwapper swapper = new YamlEncryptRuleConfigurationSwapper();
133                 result.put(swapper.getOrder(), swapper.swapToObject((YamlEncryptRuleConfiguration) each));
134             } else if (each instanceof YamlCompatibleEncryptRuleConfiguration) {
135                 YamlCompatibleEncryptRuleConfigurationSwapper swapper = new YamlCompatibleEncryptRuleConfigurationSwapper();
136                 result.put(swapper.getOrder(), swapper.swapToObject((YamlCompatibleEncryptRuleConfiguration) each));
137             } else if (each instanceof YamlShadowRuleConfiguration) {
138                 YamlShadowRuleConfigurationSwapper swapper = new YamlShadowRuleConfigurationSwapper();
139                 result.put(swapper.getOrder(), swapper.swapToObject((YamlShadowRuleConfiguration) each));
140             } else if (each instanceof YamlMaskRuleConfiguration) {
141                 YamlMaskRuleConfigurationSwapper swapper = new YamlMaskRuleConfigurationSwapper();
142                 result.put(swapper.getOrder(), swapper.swapToObject((YamlMaskRuleConfiguration) each));
143             }
144         }
145         return result;
146     }
147     
148     private void appendResourceDistSQL(final YamlProxyDatabaseConfiguration yamlConfig, final StringBuilder stringBuilder) {
149         appendDatabase(yamlConfig.getDatabaseName(), stringBuilder);
150         appendResources(yamlConfig.getDataSources(), stringBuilder);
151     }
152     
153     private void appendDatabase(final String databaseName, final StringBuilder stringBuilder) {
154         stringBuilder.append(String.format(DistSQLScriptConstants.CREATE_DATABASE, databaseName)).append(System.lineSeparator())
155                 .append(String.format(DistSQLScriptConstants.USE_DATABASE, databaseName)).append(System.lineSeparator()).append(System.lineSeparator());
156     }
157     
158     private void appendResources(final Map<String, YamlProxyDataSourceConfiguration> dataSources, final StringBuilder stringBuilder) {
159         if (dataSources.isEmpty()) {
160             return;
161         }
162         stringBuilder.append(DistSQLScriptConstants.REGISTER_STORAGE_UNIT);
163         Iterator<Entry<String, YamlProxyDataSourceConfiguration>> iterator = dataSources.entrySet().iterator();
164         while (iterator.hasNext()) {
165             Entry<String, YamlProxyDataSourceConfiguration> entry = iterator.next();
166             DataSourceProperties dataSourceProps = DataSourcePropertiesCreator.create(HikariDataSource.class.getName(), dataSourceConfigSwapper.swap(entry.getValue()));
167             appendResource(entry.getKey(), dataSourceProps, stringBuilder);
168             if (iterator.hasNext()) {
169                 stringBuilder.append(DistSQLScriptConstants.COMMA);
170             }
171         }
172         stringBuilder.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
173     }
174     
175     private void appendResource(final String resourceName, final DataSourceProperties dataSourceProps, final StringBuilder stringBuilder) {
176         Map<String, Object> connectionProps = dataSourceProps.getConnectionPropertySynonyms().getStandardProperties();
177         String url = (String) connectionProps.get(DistSQLScriptConstants.KEY_URL);
178         String username = (String) connectionProps.get(DistSQLScriptConstants.KEY_USERNAME);
179         String password = (String) connectionProps.get(DistSQLScriptConstants.KEY_PASSWORD);
180         String props = getResourceProperties(dataSourceProps.getPoolPropertySynonyms(), dataSourceProps.getCustomDataSourceProperties());
181         if (Strings.isNullOrEmpty(password)) {
182             stringBuilder.append(String.format(DistSQLScriptConstants.RESOURCE_DEFINITION_WITHOUT_PASSWORD, resourceName, url, username, props));
183         } else {
184             stringBuilder.append(String.format(DistSQLScriptConstants.RESOURCE_DEFINITION, resourceName, url, username, password, props));
185         }
186     }
187     
188     private String getResourceProperties(final PoolPropertySynonyms poolPropertySynonyms, final CustomDataSourceProperties customDataSourceProps) {
189         StringBuilder result = new StringBuilder();
190         appendProperties(poolPropertySynonyms.getStandardProperties(), result);
191         if (!customDataSourceProps.getProperties().isEmpty()) {
192             result.append(DistSQLScriptConstants.COMMA);
193             appendProperties(customDataSourceProps.getProperties(), result);
194         }
195         return result.toString();
196     }
197     
198     private void appendProperties(final Map<String, Object> props, final StringBuilder stringBuilder) {
199         Iterator<Entry<String, Object>> iterator = props.entrySet().iterator();
200         while (iterator.hasNext()) {
201             Entry<String, Object> entry = iterator.next();
202             if (null == entry.getValue()) {
203                 continue;
204             }
205             stringBuilder.append(String.format(DistSQLScriptConstants.PROPERTY, entry.getKey(), entry.getValue()));
206             if (iterator.hasNext()) {
207                 stringBuilder.append(DistSQLScriptConstants.COMMA).append(' ');
208             }
209         }
210     }

212     private void appendEncryptDistSQL(final EncryptRuleConfiguration ruleConfig, final StringBuilder stringBuilder) {
213         if (ruleConfig.getTables().isEmpty()) {
214             return;
215         }
216         stringBuilder.append(DistSQLScriptConstants.CREATE_ENCRYPT);
217         Iterator<EncryptTableRuleConfiguration> iterator = ruleConfig.getTables().iterator();
218         while (iterator.hasNext()) {
219             EncryptTableRuleConfiguration tableRuleConfig = iterator.next();
220             stringBuilder.append(String.format(DistSQLScriptConstants.ENCRYPT, tableRuleConfig.getName(), getEncryptColumns(tableRuleConfig.getColumns(), ruleConfig.getEncryptors())));
221             if (iterator.hasNext()) {
222                 stringBuilder.append(DistSQLScriptConstants.COMMA).append(System.lineSeparator());
223             }
224         }
225         stringBuilder.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
226     }

228     private String getEncryptColumns(final Collection<EncryptColumnRuleConfiguration> ruleConfigs, final Map<String, AlgorithmConfiguration> encryptors) {
229         StringBuilder result = new StringBuilder();
230         Iterator<EncryptColumnRuleConfiguration> iterator = ruleConfigs.iterator();
231         while (iterator.hasNext()) {
232             EncryptColumnRuleConfiguration columnRuleConfig = iterator.next();
233             result.append(String.format(DistSQLScriptConstants.ENCRYPT_COLUMN, columnRuleConfig.getName(), getColumns(columnRuleConfig), getEncryptAlgorithms(columnRuleConfig, encryptors)));
234             if (iterator.hasNext()) {
235                 result.append(DistSQLScriptConstants.COMMA).append(System.lineSeparator());
236             }
237         }
238         return result.toString();
239     }
240     
241     private String getColumns(final EncryptColumnRuleConfiguration ruleConfig) {
242         StringBuilder result = new StringBuilder();
243         String cipherColumnName = ruleConfig.getCipher().getName();
244         if (!Strings.isNullOrEmpty(cipherColumnName)) {
245             result.append(String.format(DistSQLScriptConstants.CIPHER, cipherColumnName));
246         }
247         if (ruleConfig.getAssistedQuery().isPresent()) {
248             result.append(DistSQLScriptConstants.COMMA).append(' ').append(String.format(DistSQLScriptConstants.ASSISTED_QUERY_COLUMN, ruleConfig.getAssistedQuery().get().getName()));
249         }
250         if (ruleConfig.getLikeQuery().isPresent()) {
251             result.append(DistSQLScriptConstants.COMMA).append(' ').append(String.format(DistSQLScriptConstants.LIKE_QUERY_COLUMN, ruleConfig.getLikeQuery().get().getName()));
252         }
253         return result.toString();
254     }
255     
256     private String getEncryptAlgorithms(final EncryptColumnRuleConfiguration ruleConfig, final Map<String, AlgorithmConfiguration> encryptors) {
257         StringBuilder result = new StringBuilder();
258         String cipherEncryptorName = ruleConfig.getCipher().getEncryptorName();
259         String assistedQueryEncryptorName = ruleConfig.getAssistedQuery().map(EncryptColumnItemRuleConfiguration::getEncryptorName).orElse("");
260         String likeQueryEncryptorName = ruleConfig.getLikeQuery().map(EncryptColumnItemRuleConfiguration::getEncryptorName).orElse("");
261         if (!Strings.isNullOrEmpty(cipherEncryptorName)) {
262             result.append(String.format(DistSQLScriptConstants.ENCRYPT_ALGORITHM, getAlgorithmType(encryptors.get(cipherEncryptorName))));
263         }
264         if (!Strings.isNullOrEmpty(assistedQueryEncryptorName)) {
265             result.append(DistSQLScriptConstants.COMMA).append(' ')
266                     .append(String.format(DistSQLScriptConstants.ASSISTED_QUERY_ALGORITHM, getAlgorithmType(encryptors.get(assistedQueryEncryptorName))));
267         }
268         if (!Strings.isNullOrEmpty(likeQueryEncryptorName)) {
269             result.append(DistSQLScriptConstants.COMMA).append(' ').append(String.format(DistSQLScriptConstants.LIKE_QUERY_ALGORITHM, getAlgorithmType(encryptors.get(likeQueryEncryptorName))));
270         }
271         return result.toString();
272     }

274     private void appendShadowDistSQL(final ShadowRuleConfiguration ruleConfig, final StringBuilder stringBuilder) {
275         if (ruleConfig.getDataSources().isEmpty()) {
276             return;
277         }
278         stringBuilder.append(DistSQLScriptConstants.CREATE_SHADOW);
279         Iterator<ShadowDataSourceConfiguration> iterator = ruleConfig.getDataSources().iterator();
280         while (iterator.hasNext()) {
281             ShadowDataSourceConfiguration dataSourceConfig = iterator.next();
282             String shadowRuleName = dataSourceConfig.getName();
283             String shadowTables = getShadowTables(shadowRuleName, ruleConfig.getTables(), ruleConfig.getShadowAlgorithms());
284             stringBuilder.append(
285                     String.format(DistSQLScriptConstants.SHADOW, shadowRuleName, dataSourceConfig.getProductionDataSourceName(), dataSourceConfig.getShadowDataSourceName(), shadowTables));
286             if (iterator.hasNext()) {
287                 stringBuilder.append(DistSQLScriptConstants.COMMA);
288             }
289         }
290         stringBuilder.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
291     }
292     
293     private String getShadowTables(final String shadowRuleName, final Map<String, ShadowTableConfiguration> ruleConfig, final Map<String, AlgorithmConfiguration> algorithmConfigs) {
294         StringBuilder result = new StringBuilder();
295         Iterator<Entry<String, ShadowTableConfiguration>> iterator = ruleConfig.entrySet().iterator();
296         while (iterator.hasNext()) {
297             Entry<String, ShadowTableConfiguration> shadowTableConfig = iterator.next();
298             if (shadowTableConfig.getValue().getDataSourceNames().contains(shadowRuleName)) {
299                 String shadowTableTypes = getShadowTableTypes(shadowTableConfig.getValue().getShadowAlgorithmNames(), algorithmConfigs);
300                 result.append(String.format(DistSQLScriptConstants.SHADOW_TABLE, shadowTableConfig.getKey(), shadowTableTypes));
301             }
302             if (iterator.hasNext()) {
303                 result.append(DistSQLScriptConstants.COMMA).append(System.lineSeparator());
304             }
305         }
306         return result.toString();
307     }
308     
309     private String getShadowTableTypes(final Collection<String> shadowAlgorithmNames, final Map<String, AlgorithmConfiguration> algorithmConfigs) {
310         StringBuilder result = new StringBuilder();
311         Iterator<String> iterator = shadowAlgorithmNames.iterator();
312         while (iterator.hasNext()) {
313             result.append(getAlgorithmType(algorithmConfigs.get(iterator.next())));
314             if (iterator.hasNext()) {
315                 result.append(DistSQLScriptConstants.COMMA).append(' ');
316             }
317         }
318         return result.toString();
319     }

321     private void appendMaskDistSQL(final MaskRuleConfiguration ruleConfig, final StringBuilder stringBuilder) {
322         if (ruleConfig.getTables().isEmpty()) {
323             return;
324         }
325         stringBuilder.append(DistSQLScriptConstants.CREATE_MASK);
326         Iterator<MaskTableRuleConfiguration> iterator = ruleConfig.getTables().iterator();
327         while (iterator.hasNext()) {
328             MaskTableRuleConfiguration tableRuleConfig = iterator.next();
329             stringBuilder.append(String.format(DistSQLScriptConstants.MASK, tableRuleConfig.getName(), getMaskColumns(tableRuleConfig.getColumns(), ruleConfig.getMaskAlgorithms())));
330             if (iterator.hasNext()) {
331                 stringBuilder.append(DistSQLScriptConstants.COMMA).append(System.lineSeparator());
332             }
333         }
334         stringBuilder.append(DistSQLScriptConstants.SEMI).append(System.lineSeparator()).append(System.lineSeparator());
335     }
336     
337     private String getMaskColumns(final Collection<MaskColumnRuleConfiguration> columnRuleConfig, final Map<String, AlgorithmConfiguration> maskAlgorithms) {
338         StringBuilder result = new StringBuilder();
339         Iterator<MaskColumnRuleConfiguration> iterator = columnRuleConfig.iterator();
340         if (iterator.hasNext()) {
341             MaskColumnRuleConfiguration column = iterator.next();
342             result.append(String.format(DistSQLScriptConstants.MASK_COLUMN, column.getLogicColumn(), getMaskAlgorithms(column, maskAlgorithms)));
343         }
344         return result.toString();
345     }
346     
347     private String getMaskAlgorithms(final MaskColumnRuleConfiguration columnRuleConfig, final Map<String, AlgorithmConfiguration> maskAlgorithms) {
348         return getAlgorithmType(maskAlgorithms.get(columnRuleConfig.getMaskAlgorithm()));
349     }
350     
351     private String getAlgorithmType(final AlgorithmConfiguration algorithmConfig) {
352         StringBuilder result = new StringBuilder();
353         if (null == algorithmConfig) {
354             return result.toString();
355         }
356         String type = algorithmConfig.getType().toLowerCase();
357         if (algorithmConfig.getProps().isEmpty()) {
358             result.append(String.format(DistSQLScriptConstants.ALGORITHM_TYPE_WITHOUT_PROPS, type));
359         } else {
360             result.append(String.format(DistSQLScriptConstants.ALGORITHM_TYPE, type, getAlgorithmProperties(algorithmConfig.getProps())));
361         }
362         return result.toString();
363     }
364     
365     @SuppressWarnings({"unchecked", "rawtypes"})
366     private String getAlgorithmProperties(final Properties props) {
367         StringBuilder result = new StringBuilder();
368         Iterator<String> iterator = new TreeMap(props).keySet().iterator();
369         while (iterator.hasNext()) {
370             String key = iterator.next();
371             Object value = props.get(key);
372             if (null == value) {
373                 continue;
374             }
375             result.append(String.format(DistSQLScriptConstants.PROPERTY, key, value));
376             if (iterator.hasNext()) {
377                 result.append(DistSQLScriptConstants.COMMA).append(' ');
378             }
379         }
380         return result.toString();
381     }
