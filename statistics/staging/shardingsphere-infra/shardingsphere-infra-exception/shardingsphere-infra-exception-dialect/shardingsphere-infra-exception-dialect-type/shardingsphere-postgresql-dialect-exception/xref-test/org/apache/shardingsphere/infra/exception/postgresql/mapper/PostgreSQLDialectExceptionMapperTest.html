35  import static org.hamcrest.CoreMatchers.is;
36  import static org.hamcrest.MatcherAssert.assertThat;
37  import static org.mockito.Mockito.mock;
38  
39  class PostgreSQLDialectExceptionMapperTest {
40      
41      @ParameterizedTest(name = "{1} -> {0}")
42      @ArgumentsSource(TestCaseArgumentsProvider.class)
43      void assertConvert(final Class<SQLDialectException> sqlDialectExceptionClazz, final String sqlState) {
44          assertThat(new PostgreSQLDialectExceptionMapper().convert(mock(sqlDialectExceptionClazz)).getSQLState(), is(sqlState));
45      }
46      
47      private static class TestCaseArgumentsProvider implements ArgumentsProvider {
48          
49          @Override
50          public Stream<? extends Arguments> provideArguments(final ExtensionContext extensionContext) {
51              return Stream.of(Arguments.of(DatabaseCreateExistsException.class, "42P04"),
52                      Arguments.of(InTransactionException.class, PSQLState.TRANSACTION_STATE_INVALID.getState()),
53                      Arguments.of(InsertColumnsAndValuesMismatchedException.class, PSQLState.SYNTAX_ERROR.getState()),
54                      Arguments.of(InvalidParameterValueException.class, PSQLState.INVALID_PARAMETER_VALUE.getState()),
55                      Arguments.of(TooManyConnectionsException.class, PSQLState.CONNECTION_REJECTED.getState()));
56          }
57      }
58  }
