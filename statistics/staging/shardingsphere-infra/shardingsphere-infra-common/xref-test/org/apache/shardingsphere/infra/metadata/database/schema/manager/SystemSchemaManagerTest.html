18  package org.apache.shardingsphere.infra.metadata.database.schema.manager;
19  
20  import org.junit.jupiter.api.Test;
21  
22  import java.util.Collection;
23  
24  import static org.hamcrest.CoreMatchers.is;
25  import static org.hamcrest.MatcherAssert.assertThat;
26  import static org.junit.jupiter.api.Assertions.assertFalse;
27  import static org.junit.jupiter.api.Assertions.assertTrue;
28  
29  class SystemSchemaManagerTest {
30      
31      @Test
32      void assertValueOfSchemaPathSuccess() {
33          Collection<String> actualInformationSchema = SystemSchemaManager.getTables("MySQL", "information_schema");
34          assertThat(actualInformationSchema.size(), is(95));
35          Collection<String> actualMySQLSchema = SystemSchemaManager.getTables("MySQL", "mysql");
36          assertThat(actualMySQLSchema.size(), is(40));
37          Collection<String> actualPerformanceSchema = SystemSchemaManager.getTables("MySQL", "performance_schema");
38          assertThat(actualPerformanceSchema.size(), is(114));
39          Collection<String> actualSysSchema = SystemSchemaManager.getTables("MySQL", "sys");
40          assertThat(actualSysSchema.size(), is(53));
41          Collection<String> actualShardingSphereSchema = SystemSchemaManager.getTables("MySQL", "shardingsphere");
42          assertThat(actualShardingSphereSchema.size(), is(2));
43          Collection<String> actualPgInformationSchema = SystemSchemaManager.getTables("PostgreSQL", "information_schema");
44          assertThat(actualPgInformationSchema.size(), is(69));
45          Collection<String> actualPgCatalog = SystemSchemaManager.getTables("PostgreSQL", "pg_catalog");
46          assertThat(actualPgCatalog.size(), is(134));
47          Collection<String> actualOgInformationSchema = SystemSchemaManager.getTables("openGauss", "information_schema");
48          assertThat(actualOgInformationSchema.size(), is(66));
49          Collection<String> actualOgPgCatalog = SystemSchemaManager.getTables("openGauss", "pg_catalog");
50          assertThat(actualOgPgCatalog.size(), is(240));
51      }
52      
53      @Test
54      void assertIsisSystemTable() {
55          assertTrue(SystemSchemaManager.isSystemTable("information_schema", "columns"));
56          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_database"));
57          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_tables"));
58          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_aggregate"));
59          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_am"));
60          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_amop"));
61          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_amproc"));
62          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_attrdef"));
63          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_attribute"));
64          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_auth_members"));
65          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_authid"));
66          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_available_extension_versions"));
67          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_available_extensions"));
68          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_backend_memory_contexts"));
69          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_cast"));
70          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_range"));
71          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_replication_origin"));
72          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_rewrite"));
73          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_seclabel"));
74          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_sequence"));
75          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_roles"));
76          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_user_mapping"));
77          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_stat_database_conflicts"));
78          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_stat_gssapi"));
79          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_stat_progress_analyze"));
80          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_stat_progress_basebackup"));
81          assertTrue(SystemSchemaManager.isSystemTable("pg_catalog", "pg_stat_progress_cluster"));
82          assertFalse(SystemSchemaManager.isSystemTable("sharding_db", "t_order"));
83          assertTrue(SystemSchemaManager.isSystemTable("shardingsphere", "cluster_information"));
84          assertTrue(SystemSchemaManager.isSystemTable("shardingsphere", "sharding_table_statistics"));
85          assertFalse(SystemSchemaManager.isSystemTable("shardingsphere", "nonexistent"));
86      }
87  }
