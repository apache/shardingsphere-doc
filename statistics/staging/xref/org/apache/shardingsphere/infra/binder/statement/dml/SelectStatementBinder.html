32  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.SelectStatement;

34  import java.util.Collections;
35  import java.util.LinkedHashMap;
36  import java.util.Map;
37  import java.util.Optional;
38  
39  /**
40   * Select statement binder.
41   */
42  @RequiredArgsConstructor
43  public final class SelectStatementBinder implements SQLStatementBinder<SelectStatement> {
44      
45      private final Map<String, TableSegmentBinderContext> outerTableBinderContexts;
46      
47      public SelectStatementBinder() {
48          outerTableBinderContexts = Collections.emptyMap();
49      }
50      
51      @Override
52      public SelectStatement bind(final SelectStatement sqlStatement, final SQLStatementBinderContext binderContext) {
53          SelectStatement result = copy(sqlStatement);
54          Map<String, TableSegmentBinderContext> tableBinderContexts = new LinkedHashMap<>();
55          sqlStatement.getWithSegment()
56                  .ifPresent(optional -> result.setWithSegment(WithSegmentBinder.bind(optional, binderContext, tableBinderContexts, binderContext.getExternalTableBinderContexts())));
57          Optional<TableSegment> boundTableSegment = sqlStatement.getFrom().map(optional -> TableSegmentBinder.bind(optional, binderContext, tableBinderContexts, outerTableBinderContexts));
58          boundTableSegment.ifPresent(result::setFrom);
59          result.setProjections(ProjectionsSegmentBinder.bind(sqlStatement.getProjections(), binderContext, boundTableSegment.orElse(null), tableBinderContexts, outerTableBinderContexts));
60          sqlStatement.getWhere().ifPresent(optional -> result.setWhere(WhereSegmentBinder.bind(optional, binderContext, tableBinderContexts, outerTableBinderContexts)));
61          sqlStatement.getCombine().ifPresent(optional -> result.setCombine(CombineSegmentBinder.bind(optional, binderContext)));
62          sqlStatement.getLock().ifPresent(optional -> result.setLock(LockSegmentBinder.bind(optional, binderContext, tableBinderContexts, outerTableBinderContexts)));
63          // TODO support other segment bind in select statement
64          return result;
65      }
66      
67      @SneakyThrows(ReflectiveOperationException.class)
68      private SelectStatement copy(final SelectStatement sqlStatement) {
69          SelectStatement result = sqlStatement.getClass().getDeclaredConstructor().newInstance();
70          sqlStatement.getGroupBy().ifPresent(result::setGroupBy);
71          sqlStatement.getHaving().ifPresent(result::setHaving);
72          sqlStatement.getOrderBy().ifPresent(result::setOrderBy);
73          sqlStatement.getLimit().ifPresent(result::setLimit);
74          sqlStatement.getWindow().ifPresent(result::setWindow);
75          sqlStatement.getModelSegment().ifPresent(result::setModelSegment);
76          result.addParameterMarkerSegments(sqlStatement.getParameterMarkerSegments());
77          result.getCommentSegments().addAll(sqlStatement.getCommentSegments());
78          return result;
79      }
80  }
