18  package org.apache.shardingsphere.logging.yaml.swapper;
19  
20  import org.apache.shardingsphere.infra.yaml.config.swapper.rule.YamlRuleConfigurationSwapper;
21  import org.apache.shardingsphere.logging.config.LoggingRuleConfiguration;
22  import org.apache.shardingsphere.logging.constant.LoggingOrder;
23  import org.apache.shardingsphere.logging.rule.builder.DefaultLoggingRuleConfigurationBuilder;
24  import org.apache.shardingsphere.logging.yaml.config.YamlLoggingRuleConfiguration;
25  
26  import java.util.stream.Collectors;
27  
28  /**
29   * YAML logging rule configuration swapper.
30   */
31  public final class YamlLoggingRuleConfigurationSwapper implements YamlRuleConfigurationSwapper<YamlLoggingRuleConfiguration, LoggingRuleConfiguration> {
32      
33      private final YamlLoggerSwapper loggerSwapper = new YamlLoggerSwapper();
34      
35      private final YamlAppenderSwapper appenderSwapper = new YamlAppenderSwapper();
36      
37      @Override
38      public YamlLoggingRuleConfiguration swapToYamlConfiguration(final LoggingRuleConfiguration data) {
39          YamlLoggingRuleConfiguration result = new YamlLoggingRuleConfiguration();
40          result.setLoggers(data.getLoggers().stream().map(loggerSwapper::swapToYamlConfiguration).collect(Collectors.toList()));
41          result.setAppenders(data.getAppenders().stream().map(appenderSwapper::swapToYamlConfiguration).collect(Collectors.toList()));
42          return result;
43      }
44      
45      @Override
46      public LoggingRuleConfiguration swapToObject(final YamlLoggingRuleConfiguration yamlConfig) {
47          LoggingRuleConfiguration result = new LoggingRuleConfiguration(yamlConfig.getLoggers().stream().map(loggerSwapper::swapToObject).collect(Collectors.toList()),
48                  yamlConfig.getAppenders().stream().map(appenderSwapper::swapToObject).collect(Collectors.toList()));
49          if (null == result.getLoggers()) {
50              result = getDefaultLoggingRuleConfiguration();
51          }
52          return result;
53      }
54      
55      private LoggingRuleConfiguration getDefaultLoggingRuleConfiguration() {
56          return new DefaultLoggingRuleConfigurationBuilder().build();
57      }
58      
59      @Override
60      public Class<LoggingRuleConfiguration> getTypeClass() {
61          return LoggingRuleConfiguration.class;
62      }
63      
64      @Override
65      public String getRuleTagName() {
66          return "LOGGING";
67      }
68      
69      @Override
70      public int getOrder() {
71          return LoggingOrder.ORDER;
72      }
73  }
