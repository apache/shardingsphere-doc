18  package org.apache.shardingsphere.encrypt.sm.algorithm;
19  
20  import lombok.EqualsAndHashCode;
21  import org.apache.commons.codec.binary.Hex;
22  import org.apache.shardingsphere.encrypt.api.encrypt.standard.StandardEncryptAlgorithm;
23  import org.apache.shardingsphere.encrypt.exception.algorithm.EncryptAlgorithmInitializationException;
24  import org.apache.shardingsphere.encrypt.api.context.EncryptContext;
25  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
26  import org.bouncycastle.crypto.digests.SM3Digest;
27  import org.bouncycastle.jce.provider.BouncyCastleProvider;
28  
29  import java.nio.charset.StandardCharsets;
30  import java.security.Security;
31  import java.util.Properties;
32  
33  /**
34   * SM3 encrypt algorithm.
35   */
36  @EqualsAndHashCode
37  public final class SM3EncryptAlgorithm implements StandardEncryptAlgorithm<Object, String> {
38      
39      static {
40          Security.addProvider(new BouncyCastleProvider());
41      }
42      
43      private static final String SM3_SALT = "sm3-salt";
44      
45      private static final int SALT_LENGTH = 8;
46      
47      private byte[] sm3Salt;
48      
49      @Override
50      public void init(final Properties props) {
51          sm3Salt = createSm3Salt(props);
52      }
53      
54      private byte[] createSm3Salt(final Properties props) {
55          String salt = null == props.getProperty(SM3_SALT) ? "" : String.valueOf(props.getProperty(SM3_SALT));
56          ShardingSpherePreconditions.checkState(salt.isEmpty() || SALT_LENGTH == salt.length(),
57                  () -> new EncryptAlgorithmInitializationException("SM3", "Salt should be either blank or better " + SALT_LENGTH + " bytes long."));
58          return salt.isEmpty() ? new byte[0] : salt.getBytes(StandardCharsets.UTF_8);
59      }
60      
61      @Override
62      public String encrypt(final Object plainValue, final EncryptContext encryptContext) {
63          return null == plainValue ? null : Hex.encodeHexString(digest(String.valueOf(plainValue).getBytes(StandardCharsets.UTF_8), sm3Salt));
64      }
65      
66      @Override
67      public Object decrypt(final String cipherValue, final EncryptContext encryptContext) {
68          return cipherValue;
69      }
70      
71      private byte[] digest(final byte[] input, final byte[] salt) {
72          SM3Digest sm3Digest = new SM3Digest();
73          byte[] updateByte = concat(input, salt);
74          sm3Digest.update(updateByte, 0, updateByte.length);
75          byte[] result = new byte[sm3Digest.getDigestSize()];
76          sm3Digest.doFinal(result, 0);
77          return result;
78      }
79      
80      private byte[] concat(final byte[] input, final byte[] salt) {
81          int inputLength = input.length;
82          int saltLength = salt.length;
83          byte[] result = new byte[inputLength + saltLength];
84          System.arraycopy(input, 0, result, 0, inputLength);
85          System.arraycopy(salt, 0, result, inputLength, saltLength);
86          return result;
87      }
88      
89      @Override
90      public String getType() {
91          return "SM3";
92      }
93  }
