45  /**
46   * DAL statement visitor for MySQL.
47   */
48  public final class PrestoDALStatementVisitor extends PrestoStatementVisitor implements DALStatementVisitor {
49      
50      @Override
51      public ASTNode visitExplain(final ExplainContext ctx) {
52          MySQLExplainStatement result = new MySQLExplainStatement();
53          if (null != ctx.tableName()) {
54              result.setTable((SimpleTableSegment) visit(ctx.tableName()));
55          }
56          return result;
57      }
58      
59      @Override
60      public ASTNode visitShow(final ShowContext ctx) {
61          return new EmptyStatement();
62      }
63      
64      @Override
65      public ASTNode visitChannelOption(final ChannelOptionContext ctx) {
66          return visit(ctx.string_());
67      }
68      
69      @Override
70      public ASTNode visitBinaryLogFileIndexNumber(final BinaryLogFileIndexNumberContext ctx) {
71          return new NumberLiteralValue(ctx.getText());
72      }
73      
74      @Override
75      public ASTNode visitShowWhereClause(final ShowWhereClauseContext ctx) {
76          return new WhereSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), (ExpressionSegment) visit(ctx.expr()));
77      }
78      
79      @Override
80      public ASTNode visitShowFilter(final ShowFilterContext ctx) {
81          ShowFilterSegment result = new ShowFilterSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
82          if (null != ctx.showLike()) {
83              result.setLike((ShowLikeSegment) visit(ctx.showLike()));
84          }
85          if (null != ctx.showWhereClause()) {
86              result.setWhere((WhereSegment) visit(ctx.showWhereClause()));
87          }
88          return result;
89      }
90      
91      @Override
92      public ASTNode visitFromTable(final FromTableContext ctx) {
93          FromTableSegment result = new FromTableSegment();
94          result.setTable((SimpleTableSegment) visit(ctx.tableName()));
95          return result;
96      }
97      
98      @Override
99      public ASTNode visitFromSchema(final FromSchemaContext ctx) {
100         return new FromSchemaSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), (DatabaseSegment) visit(ctx.schemaName()));
101     }
102     
103     @Override
104     public ASTNode visitShowLike(final ShowLikeContext ctx) {
105         StringLiteralValue literalValue = (StringLiteralValue) visit(ctx.stringLiterals());
106         return new ShowLikeSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), literalValue.getValue());
107     }
108     
109 }
