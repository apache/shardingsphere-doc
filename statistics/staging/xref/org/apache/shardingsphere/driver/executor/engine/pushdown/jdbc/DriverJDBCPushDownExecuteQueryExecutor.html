53  /**
54   * Driver JDBC push down execute query executor.
55   */
56  public final class DriverJDBCPushDownExecuteQueryExecutor {
57      
58      private final ConnectionContext connectionContext;
59      
60      private final String processId;
61      
62      private final RuleMetaData globalRuleMetaData;
63      
64      private final ConfigurationProperties props;
65      
66      private final JDBCExecutor jdbcExecutor;
67      
68      private final Collection<Statement> statements;
69      
70      public DriverJDBCPushDownExecuteQueryExecutor(final ShardingSphereConnection connection, final ShardingSphereMetaData metaData, final JDBCExecutor jdbcExecutor) {
71          connectionContext = connection.getDatabaseConnectionManager().getConnectionContext();
72          processId = connection.getProcessId();
73          globalRuleMetaData = metaData.getGlobalRuleMetaData();
74          props = metaData.getProps();
75          this.jdbcExecutor = jdbcExecutor;
76          statements = new LinkedList<>();
77      }
78      
79      /**
80       * Execute query.
81       * 
82       * @param database database
83       * @param queryContext query context
84       * @param prepareEngine prepare engine
85       * @param statement statement
86       * @param columnLabelAndIndexMap column label and index map
87       * @param addCallback statement add callback
88       * @param replayCallback statement replay callback
89       * @return result set
90       * @throws SQLException SQL exception
91       */
92      @SuppressWarnings("rawtypes")
93      public ResultSet executeQuery(final ShardingSphereDatabase database, final QueryContext queryContext,
94                                    final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine, final Statement statement,
95                                    final Map<String, Integer> columnLabelAndIndexMap,
96                                    final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
97          List<QueryResult> queryResults = getQueryResults(database, queryContext, prepareEngine, addCallback, replayCallback);
98          return new ShardingSphereResultSetFactory(connectionContext, globalRuleMetaData, props, statements).newInstance(database, queryContext, queryResults, statement, columnLabelAndIndexMap);
99      }
100     
101     @SuppressWarnings({"rawtypes", "unchecked"})
102     private List<QueryResult> getQueryResults(final ShardingSphereDatabase database, final QueryContext queryContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
103                                               final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
104         statements.clear();
105         ExecutionContext executionContext = new KernelProcessor().generateExecutionContext(queryContext, database, globalRuleMetaData, props, connectionContext);
106         ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext = prepareEngine.prepare(database.getName(), executionContext.getRouteContext(), executionContext.getExecutionUnits(),
107                 new ExecutionGroupReportContext(processId, database.getName(), connectionContext.getGrantee()));
108         for (ExecutionGroup<JDBCExecutionUnit> each : executionGroupContext.getInputGroups()) {
109             Collection<Statement> statements = getStatements(each);
110             this.statements.addAll(statements);
111             addCallback.add(statements, JDBCDriverType.PREPARED_STATEMENT.equals(prepareEngine.getType()) ? getParameterSets(each) : Collections.emptyList());
112         }
113         replayCallback.replay();
114         ProcessEngine processEngine = new ProcessEngine();
115         try {
116             processEngine.executeSQL(executionGroupContext, queryContext);
117             return jdbcExecutor.execute(executionGroupContext, new ExecuteQueryCallbackFactory(prepareEngine.getType()).newInstance(database, queryContext));
118         } finally {
119             processEngine.completeSQLExecution(executionGroupContext.getReportContext().getProcessId());
120         }
121     }
122     
123     private Collection<Statement> getStatements(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
124         Collection<Statement> result = new LinkedList<>();
125         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
126             result.add(each.getStorageResource());
127         }
128         return result;
129     }
130     
131     private Collection<List<Object>> getParameterSets(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
132         Collection<List<Object>> result = new LinkedList<>();
133         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
134             result.add(each.getExecutionUnit().getSqlUnit().getParameters());
135         }
136         return result;
137     }
138 }
