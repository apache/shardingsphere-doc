18  package org.apache.shardingsphere.infra.database.type.checker;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.infra.database.type.DatabaseType;
23  import org.apache.shardingsphere.infra.exception.UnsupportedStorageTypeException;
24  import org.apache.shardingsphere.infra.util.exception.ShardingSpherePreconditions;
25  import org.apache.shardingsphere.infra.util.spi.type.typed.TypedSPILoader;
26  
27  import javax.sql.DataSource;
28  import java.sql.Connection;
29  import java.sql.SQLException;
30  import java.util.Arrays;
31  import java.util.Collection;
32  import java.util.Collections;
33  import java.util.HashSet;
34  import java.util.Map;
35  import java.util.Map.Entry;
36  
37  /**
38   * Database type checker.
39   */
40  @NoArgsConstructor(access = AccessLevel.PRIVATE)
41  public final class DatabaseTypeChecker {
42      
43      private static final Collection<String> MOCKED_URL_PREFIXES = new HashSet<>(Arrays.asList("jdbc:fixture", "jdbc:mock", "mock:jdbc"));
44      
45      private static final Collection<String> UNSUPPORTED_URL_PREFIXES = Collections.singletonList("jdbc:mysql:aws");
46      
47      private static final Collection<DatabaseType> SUPPORTED_STORAGE_TYPES = new HashSet<>(8, 1F);
48      
49      private static volatile boolean isChecked;
50      
51      static {
52          Arrays.asList("MySQL", "PostgreSQL", "openGauss", "Oracle", "SQLServer", "H2", "MariaDB")
53                  .forEach(each -> TypedSPILoader.findService(DatabaseType.class, each).ifPresent(SUPPORTED_STORAGE_TYPES::add));
54      }
55      
56      /**
57       * Check supported storage types.
58       *
59       * @param dataSources  data sources
60       * @param databaseName database name
61       * @param storageTypes storage types
62       * @throws SQLException SQL exception
63       */
64      public static void checkSupportedStorageTypes(final Map<String, DataSource> dataSources, final String databaseName, final Map<String, DatabaseType> storageTypes) throws SQLException {
65          if (isChecked || dataSources.isEmpty()) {
66              return;
67          }
68          for (Entry<String, DataSource> entry : dataSources.entrySet()) {
69              try (Connection connection = entry.getValue().getConnection()) {
70                  String url = connection.getMetaData().getURL();
71                  if (MOCKED_URL_PREFIXES.stream().anyMatch(url::startsWith)) {
72                      return;
73                  }
74                  ShardingSpherePreconditions.checkState(UNSUPPORTED_URL_PREFIXES.stream()
75                          .noneMatch(url::startsWith), () -> new UnsupportedStorageTypeException(databaseName, entry.getKey()));
76              }
77          }
78          storageTypes.forEach((key, value) -> ShardingSpherePreconditions.checkState(SUPPORTED_STORAGE_TYPES.stream()
79                  .anyMatch(each -> each.getClass().equals(value.getClass())), () -> new UnsupportedStorageTypeException(databaseName, key)));
80          isChecked = true;
81      }
82      
83      /**
84       * Check supported storage type.
85       *
86       * @param url URL
87       * @param dataSourceName data source name
88       */
89      public static void checkSupportedStorageType(final String url, final String dataSourceName) {
90          if (MOCKED_URL_PREFIXES.stream().anyMatch(url::startsWith)) {
91              return;
92          }
93          ShardingSpherePreconditions.checkState(UNSUPPORTED_URL_PREFIXES.stream()
94                  .noneMatch(url::startsWith), () -> new UnsupportedStorageTypeException(dataSourceName));
95          ShardingSpherePreconditions.checkState(SUPPORTED_STORAGE_TYPES.stream()
96                  .flatMap(storageType -> storageType.getJdbcUrlPrefixes().stream())
97                  .anyMatch(url::startsWith), () -> new UnsupportedStorageTypeException(dataSourceName));
98      }
99  }
