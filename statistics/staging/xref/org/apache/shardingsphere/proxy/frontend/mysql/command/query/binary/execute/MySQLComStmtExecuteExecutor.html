56  /**
57   * COM_STMT_EXECUTE command executor for MySQL.
58   */
59  @RequiredArgsConstructor
60  public final class MySQLComStmtExecuteExecutor implements QueryCommandExecutor {
61      
62      private final MySQLComStmtExecutePacket packet;
63      
64      private final ConnectionSession connectionSession;
65      
66      private ProxyBackendHandler proxyBackendHandler;
67      
68      @Getter
69      private ResponseType responseType;
70      
71      @Override
72      public Collection<DatabasePacket> execute() throws SQLException {
73          MySQLServerPreparedStatement preparedStatement = updateAndGetPreparedStatement();
74          List<Object> params = packet.readParameters(preparedStatement.getParameterTypes(), preparedStatement.getLongData().keySet(), preparedStatement.getParameterColumnDefinitionFlags());
75          preparedStatement.getLongData().forEach(params::set);
76          SQLStatementContext sqlStatementContext = preparedStatement.getSqlStatementContext();
77          if (sqlStatementContext instanceof ParameterAware) {
78              ((ParameterAware) sqlStatementContext).setUpParameters(params);
79          }
80          QueryContext queryContext = new QueryContext(sqlStatementContext, preparedStatement.getSql(), params, preparedStatement.getHintValueContext(), connectionSession.getConnectionContext(),
81                  ProxyContext.getInstance().getContextManager().getMetaDataContexts().getMetaData(), true);
82          connectionSession.setQueryContext(queryContext);
83          proxyBackendHandler = ProxyBackendHandlerFactory.newInstance(TypedSPILoader.getService(DatabaseType.class, "MySQL"), queryContext, connectionSession, true);
84          ResponseHeader responseHeader = proxyBackendHandler.execute();
85          return responseHeader instanceof QueryResponseHeader ? processQuery((QueryResponseHeader) responseHeader) : processUpdate((UpdateResponseHeader) responseHeader);
86      }
87      
88      private MySQLServerPreparedStatement updateAndGetPreparedStatement() {
89          MySQLServerPreparedStatement result = connectionSession.getServerPreparedStatementRegistry().getPreparedStatement(packet.getStatementId());
90          if (MySQLNewParametersBoundFlag.PARAMETER_TYPE_EXIST == packet.getNewParametersBoundFlag()) {
91              result.getParameterTypes().clear();
92              result.getParameterTypes().addAll(packet.getNewParameterTypes());
93          }
94          return result;
95      }
96      
97      private Collection<DatabasePacket> processQuery(final QueryResponseHeader queryResponseHeader) {
98          responseType = ResponseType.QUERY;
99          int characterSet = connectionSession.getAttributeMap().attr(MySQLConstants.MYSQL_CHARACTER_SET_ATTRIBUTE_KEY).get().getId();
100         return ResponsePacketBuilder.buildQueryResponsePackets(queryResponseHeader, characterSet, ServerStatusFlagCalculator.calculateFor(connectionSession));
101     }
102     
103     private Collection<DatabasePacket> processUpdate(final UpdateResponseHeader updateResponseHeader) {
104         responseType = ResponseType.UPDATE;
105         return ResponsePacketBuilder.buildUpdateResponsePackets(updateResponseHeader, ServerStatusFlagCalculator.calculateFor(connectionSession));
106     }
107     
108     @Override
109     public boolean next() throws SQLException {
110         return proxyBackendHandler.next();
111     }
112     
113     @Override
114     public MySQLPacket getQueryRowPacket() throws SQLException {
115         QueryResponseRow queryResponseRow = proxyBackendHandler.getRowData();
116         return new MySQLBinaryResultSetRowPacket(createBinaryRow(queryResponseRow));
117     }
118     
119     private BinaryRow createBinaryRow(final QueryResponseRow queryResponseRow) {
120         List<BinaryCell> result = new ArrayList<>(queryResponseRow.getCells().size());
121         for (QueryResponseCell each : queryResponseRow.getCells()) {
122             result.add(new BinaryCell(MySQLBinaryColumnType.valueOfJDBCType(each.getJdbcType()), each.getData()));
123         }
124         return new BinaryRow(result);
125     }
126     
127     @Override
128     public void close() throws SQLException {
129         proxyBackendHandler.close();
130     }
131 }
