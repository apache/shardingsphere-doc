18  package org.apache.shardingsphere.mask.algorithm.replace;
19  
20  import com.google.common.base.Strings;
21  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
22  import org.apache.shardingsphere.mask.exception.algorithm.MaskAlgorithmInitializationException;
23  import org.apache.shardingsphere.mask.spi.MaskAlgorithm;
24  
25  import java.security.SecureRandom;
26  import java.util.Properties;
27  import java.util.Random;
28  
29  /**
30   * Personal identity number random replace algorithm.
31   */
32  public final class PersonalIdentityNumberRandomReplaceAlgorithm implements MaskAlgorithm<Object, String> {
33      
34      private static final String ALPHA_TWO_COUNTRY_AREA_CODE = "alpha-two-country-area-code";
35      
36      private final Random random = new SecureRandom();
37      
38      private String alphaTwoCountryAreaCode;
39      
40      @Override
41      public void init(final Properties props) {
42          alphaTwoCountryAreaCode = props.getProperty(ALPHA_TWO_COUNTRY_AREA_CODE, "CN");
43          ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(alphaTwoCountryAreaCode),
44                  () -> new MaskAlgorithmInitializationException(getType(), String.format("%s can not be empty", ALPHA_TWO_COUNTRY_AREA_CODE)));
45      }
46      
47      @Override
48      public String mask(final Object plainValue) {
49          String result = null == plainValue ? null : String.valueOf(plainValue);
50          if (Strings.isNullOrEmpty(result)) {
51              return result;
52          }
53          if ("CN".equals(alphaTwoCountryAreaCode)) {
54              return randomReplaceForChinesePersonalIdentityNumber(result);
55          }
56          return result;
57      }
58      
59      private String randomReplaceForChinesePersonalIdentityNumber(final String result) {
60          switch (result.length()) {
61              case 15:
62                  return randomReplaceNumber(result, 6, 12);
63              case 18:
64                  return randomReplaceNumber(result, 6, 14);
65              default:
66          }
67          return result;
68      }
69      
70      private String randomReplaceNumber(final String result, final int from, final int to) {
71          char[] chars = result.toCharArray();
72          for (int i = from; i < to; i++) {
73              chars[i] = Character.forDigit(random.nextInt(10), 10);
74          }
75          return new String(chars);
76      }
77      
78      @Override
79      public String getType() {
80          return "PERSONAL_IDENTITY_NUMBER_RANDOM_REPLACE";
81      }
82  }
