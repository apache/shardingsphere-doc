44  /**
45   * Sharding SQL router.
46   */
47  @HighFrequencyInvocation
48  public final class ShardingSQLRouter implements EntranceSQLRouter<ShardingRule> {
49      
50      @Override
51      public RouteContext createRouteContext(final QueryContext queryContext, final RuleMetaData globalRuleMetaData, final ShardingSphereDatabase database,
52                                             final ShardingRule rule, final ConfigurationProperties props) {
53          if (rule.isShardingCacheEnabled()) {
54              Optional<RouteContext> result = new CachedShardingSQLRouter()
55                      .loadRouteContext(this::createRouteContext0, queryContext, globalRuleMetaData, database, rule.getShardingCache(), props);
56              if (result.isPresent()) {
57                  return result.get();
58              }
59          }
60          return createRouteContext0(queryContext, globalRuleMetaData, database, rule, props);
61      }
62      
63      private RouteContext createRouteContext0(final QueryContext queryContext, final RuleMetaData globalRuleMetaData, final ShardingSphereDatabase database, final ShardingRule rule,
64                                               final ConfigurationProperties props) {
65          SQLStatement sqlStatement = queryContext.getSqlStatementContext().getSqlStatement();
66          ShardingConditions shardingConditions = createShardingConditions(queryContext, globalRuleMetaData, database, rule);
67          Optional<ShardingStatementValidator> validator = ShardingStatementValidatorFactory.newInstance(sqlStatement, shardingConditions);
68          validator.ifPresent(optional -> optional.preValidate(rule, queryContext.getSqlStatementContext(), queryContext.getHintValueContext(), queryContext.getParameters(), database, props));
69          if (sqlStatement instanceof DMLStatement && shardingConditions.isNeedMerge()) {
70              shardingConditions.merge();
71          }
72          RouteContext result = ShardingRouteEngineFactory.newInstance(rule, database, queryContext, shardingConditions, props).route(rule);
73          validator.ifPresent(optional -> optional.postValidate(rule, queryContext.getSqlStatementContext(), queryContext.getHintValueContext(), queryContext.getParameters(), database, props, result));
74          return result;
75      }
76      
77      private ShardingConditions createShardingConditions(final QueryContext queryContext,
78                                                          final RuleMetaData globalRuleMetaData, final ShardingSphereDatabase database, final ShardingRule rule) {
79          List<ShardingCondition> shardingConditions;
80          if (queryContext.getSqlStatementContext().getSqlStatement() instanceof DMLStatement || queryContext.getSqlStatementContext() instanceof CursorAvailable) {
81              shardingConditions = new ShardingConditionEngine(globalRuleMetaData, database, rule).createShardingConditions(queryContext.getSqlStatementContext(), queryContext.getParameters());
82          } else {
83              shardingConditions = Collections.emptyList();
84          }
85          return new ShardingConditions(shardingConditions, queryContext.getSqlStatementContext(), rule);
86      }
87      
88      @Override
89      public int getOrder() {
90          return ShardingOrder.ORDER;
91      }
92      
93      @Override
94      public Class<ShardingRule> getTypeClass() {
95          return ShardingRule.class;
96      }
97  }
