27  import org.apache.shardingsphere.infra.rewrite.sql.token.generator.CollectionSQLTokenGenerator;
28  import org.apache.shardingsphere.infra.rewrite.sql.token.pojo.generic.InsertColumnsToken;
29  import org.apache.shardingsphere.sql.parser.sql.common.segment.dml.column.ColumnSegment;

31  import java.util.Collection;
32  import java.util.LinkedList;
33  import java.util.List;
34  import java.util.Optional;
35  
36  /**
37   * Assist query and plain insert columns token generator.
38   */
39  @Setter
40  public final class AssistQueryAndPlainInsertColumnsTokenGenerator implements CollectionSQLTokenGenerator<InsertStatementContext>, EncryptRuleAware {
41      
42      private EncryptRule encryptRule;
43      
44      @Override
45      public boolean isGenerateSQLToken(final SQLStatementContext<?> sqlStatementContext) {
46          return sqlStatementContext instanceof InsertStatementContext && ((InsertStatementContext) sqlStatementContext).containsInsertColumns();
47      }
48      
49      @Override
50      public Collection<InsertColumnsToken> generateSQLTokens(final InsertStatementContext insertStatementContext) {
51          Collection<InsertColumnsToken> result = new LinkedList<>();
52          Optional<EncryptTable> encryptTable = encryptRule.findEncryptTable(insertStatementContext.getSqlStatement().getTable().getTableName().getIdentifier().getValue());
53          Preconditions.checkState(encryptTable.isPresent());
54          for (ColumnSegment each : insertStatementContext.getSqlStatement().getColumns()) {
55              List<String> columns = getColumns(encryptTable.get(), each);
56              if (!columns.isEmpty()) {
57                  result.add(new InsertColumnsToken(each.getStopIndex() + 1, columns));
58              }
59          }
60          return result;
61      }
62      
63      private List<String> getColumns(final EncryptTable encryptTable, final ColumnSegment columnSegment) {
64          List<String> result = new LinkedList<>();
65          encryptTable.findAssistedQueryColumn(columnSegment.getIdentifier().getValue()).ifPresent(result::add);
66          encryptTable.findLikeQueryColumn(columnSegment.getIdentifier().getValue()).ifPresent(result::add);
67          encryptTable.findPlainColumn(columnSegment.getIdentifier().getValue()).ifPresent(result::add);
68          return result;
69      }
70  }
