66  @ExtendWith(MockitoExtension.class)
67  class ShardingUpdateStatementValidatorTest {
68      
69      @Mock
70      private ShardingRule shardingRule;
71      
72      @Test
73      void assertPreValidateWhenUpdateSingleTable() {
74          UpdateStatement updateStatement = createUpdateStatement();
75          updateStatement.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("user"))));
76          SQLStatementContext<UpdateStatement> sqlStatementContext = new UpdateStatementContext(updateStatement);
77          Collection<String> tableNames = sqlStatementContext.getTablesContext().getTableNames();
78          when(shardingRule.isAllShardingTables(tableNames)).thenReturn(true);
79          when(shardingRule.tableRuleExists(tableNames)).thenReturn(true);
80          new ShardingUpdateStatementValidator().preValidate(shardingRule, sqlStatementContext, Collections.emptyList(), mock(ShardingSphereDatabase.class), mock(ConfigurationProperties.class));
81      }
82      
83      @Test
84      void assertPreValidateWhenUpdateMultipleTables() {
85          UpdateStatement updateStatement = createUpdateStatement();
86          JoinTableSegment joinTableSegment = new JoinTableSegment();
87          joinTableSegment.setLeft(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("user"))));
88          joinTableSegment.setRight(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("order"))));
89          updateStatement.setTable(joinTableSegment);
90          SQLStatementContext<UpdateStatement> sqlStatementContext = new UpdateStatementContext(updateStatement);
91          Collection<String> tableNames = sqlStatementContext.getTablesContext().getTableNames();
92          when(shardingRule.isAllShardingTables(tableNames)).thenReturn(false);
93          when(shardingRule.tableRuleExists(tableNames)).thenReturn(true);
94          assertThrows(DMLWithMultipleShardingTablesException.class, () -> new ShardingUpdateStatementValidator().preValidate(
95                  shardingRule, sqlStatementContext, Collections.emptyList(), mock(ShardingSphereDatabase.class), mock(ConfigurationProperties.class)));
96      }
97      
98      @Test
99      void assertPostValidateWhenNotUpdateShardingColumn() {
100         UpdateStatementContext sqlStatementContext = new UpdateStatementContext(createUpdateStatement());
101         new ShardingUpdateStatementValidator().postValidate(shardingRule, sqlStatementContext, new HintValueContext(), Collections.emptyList(),
102                 mock(ShardingSphereDatabase.class), mock(ConfigurationProperties.class), mock(RouteContext.class));
103     }
104     
105     @Test
106     void assertPostValidateWhenUpdateShardingColumnWithSameRouteContext() {
107         mockShardingRuleForUpdateShardingColumn();
108         new ShardingUpdateStatementValidator().postValidate(shardingRule, new UpdateStatementContext(createUpdateStatement()), new HintValueContext(),
109                 Collections.emptyList(), mock(ShardingSphereDatabase.class), mock(ConfigurationProperties.class), createSingleRouteContext());
110     }
111     
112     @Test
113     void assertPostValidateWhenTableNameIsBroadcastTable() {
114         mockShardingRuleForUpdateShardingColumn();
115         when(shardingRule.isBroadcastTable("user")).thenReturn(true);
116         new ShardingUpdateStatementValidator().postValidate(shardingRule, new UpdateStatementContext(createUpdateStatement()), new HintValueContext(),
117                 Collections.emptyList(), mock(ShardingSphereDatabase.class), mock(ConfigurationProperties.class), createSingleRouteContext());
118     }
119     
120     @Test
121     void assertPostValidateWhenUpdateShardingColumnWithDifferentRouteContext() {
122         mockShardingRuleForUpdateShardingColumn();
123         assertThrows(UnsupportedUpdatingShardingValueException.class,
124                 () -> new ShardingUpdateStatementValidator().postValidate(shardingRule, new UpdateStatementContext(createUpdateStatement()), new HintValueContext(),
125                         Collections.emptyList(), mock(ShardingSphereDatabase.class), mock(ConfigurationProperties.class), createFullRouteContext()));
126     }
127     
128     private void mockShardingRuleForUpdateShardingColumn() {
129         TableRule tableRule = mock(TableRule.class);
130         when(tableRule.getActualDataSourceNames()).thenReturn(Arrays.asList("ds_0", "ds_1"));
131         when(tableRule.getActualTableNames("ds_1")).thenReturn(Collections.singleton("user"));
132         when(shardingRule.findShardingColumn("id", "user")).thenReturn(Optional.of("id"));
133         when(shardingRule.getTableRule("user")).thenReturn(tableRule);
134         StandardShardingStrategyConfiguration databaseStrategyConfig = mock(StandardShardingStrategyConfiguration.class);
135         when(databaseStrategyConfig.getShardingColumn()).thenReturn("id");
136         when(databaseStrategyConfig.getShardingAlgorithmName()).thenReturn("database_inline");
137         when(shardingRule.getDatabaseShardingStrategyConfiguration(tableRule)).thenReturn(databaseStrategyConfig);
138         when(shardingRule.getShardingAlgorithms()).thenReturn(createShardingAlgorithmMap());
139     }
140     
141     private Map<String, ShardingAlgorithm> createShardingAlgorithmMap() {
142         return Collections.singletonMap("database_inline", TypedSPILoader.getService(ShardingAlgorithm.class, "INLINE", PropertiesBuilder.build(new Property("algorithm-expression", "ds_${id % 2}"))));
143     }
144     
145     private RouteContext createSingleRouteContext() {
146         RouteContext result = new RouteContext();
147         result.getRouteUnits().add(new RouteUnit(new RouteMapper("ds_1", "ds_1"), Collections.singleton(new RouteMapper("user", "user"))));
148         return result;
149     }
150     
151     private RouteContext createFullRouteContext() {
152         RouteContext result = new RouteContext();
153         result.getRouteUnits().add(new RouteUnit(new RouteMapper("ds_0", "ds_0"), Collections.singleton(new RouteMapper("user", "user"))));
154         result.getRouteUnits().add(new RouteUnit(new RouteMapper("ds_1", "ds_1"), Collections.singleton(new RouteMapper("user", "user"))));
155         return result;
156     }
157     
158     private UpdateStatement createUpdateStatement() {
159         UpdateStatement result = new MySQLUpdateStatement();
160         result.setTable(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("user"))));
161         List<ColumnSegment> columns = new LinkedList<>();
162         columns.add(new ColumnSegment(0, 0, new IdentifierValue("id")));
163         AssignmentSegment assignment = new ColumnAssignmentSegment(0, 0, columns, new LiteralExpressionSegment(0, 0, 1));
164         result.setSetAssignment(new SetAssignmentSegment(0, 0, Collections.singleton(assignment)));
165         return result;
166     }
