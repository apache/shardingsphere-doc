18  package org.apache.shardingsphere.db.protocol.mysql.packet.command.query.binary.execute.protocol;
19  
20  import com.google.common.base.Preconditions;
21  import lombok.AccessLevel;
22  import lombok.NoArgsConstructor;
23  import org.apache.shardingsphere.db.protocol.binary.BinaryColumnType;
24  import org.apache.shardingsphere.db.protocol.mysql.constant.MySQLBinaryColumnType;
25  
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  /**
30   * Binary protocol value factory for MySQL.
31   */
32  @NoArgsConstructor(access = AccessLevel.PRIVATE)
33  public final class MySQLBinaryProtocolValueFactory {
34      
35      private static final Map<BinaryColumnType, MySQLBinaryProtocolValue> BINARY_PROTOCOL_VALUES = new HashMap<>();
36      
37      static {
38          setStringLenencBinaryProtocolValue();
39          setInt8BinaryProtocolValue();
40          setInt4BinaryProtocolValue();
41          setInt2BinaryProtocolValue();
42          setInt1BinaryProtocolValue();
43          setDoubleBinaryProtocolValue();
44          setFloatBinaryProtocolValue();
45          setDateBinaryProtocolValue();
46          setTimeBinaryProtocolValue();
47          setNullBinaryProtocolValue();
48      }
49      
50      private static void setStringLenencBinaryProtocolValue() {
51          MySQLStringLenencBinaryProtocolValue binaryProtocolValue = new MySQLStringLenencBinaryProtocolValue();
52          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.STRING, binaryProtocolValue);
53          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.VARCHAR, binaryProtocolValue);
54          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.VAR_STRING, binaryProtocolValue);
55          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.ENUM, binaryProtocolValue);
56          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.SET, binaryProtocolValue);
57          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.LONG_BLOB, binaryProtocolValue);
58          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.MEDIUM_BLOB, binaryProtocolValue);
59          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.BLOB, binaryProtocolValue);
60          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TINY_BLOB, binaryProtocolValue);
61          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.GEOMETRY, binaryProtocolValue);
62          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.BIT, binaryProtocolValue);
63          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DECIMAL, binaryProtocolValue);
64          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.NEWDECIMAL, binaryProtocolValue);
65      }
66      
67      private static void setInt8BinaryProtocolValue() {
68          MySQLInt8BinaryProtocolValue binaryProtocolValue = new MySQLInt8BinaryProtocolValue();
69          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.LONGLONG, binaryProtocolValue);
70      }
71      
72      private static void setInt4BinaryProtocolValue() {
73          MySQLInt4BinaryProtocolValue binaryProtocolValue = new MySQLInt4BinaryProtocolValue();
74          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.LONG, binaryProtocolValue);
75          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.INT24, binaryProtocolValue);
76      }
77      
78      private static void setInt2BinaryProtocolValue() {
79          MySQLInt2BinaryProtocolValue binaryProtocolValue = new MySQLInt2BinaryProtocolValue();
80          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.SHORT, binaryProtocolValue);
81          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.YEAR, binaryProtocolValue);
82      }
83      
84      private static void setInt1BinaryProtocolValue() {
85          MySQLInt1BinaryProtocolValue binaryProtocolValue = new MySQLInt1BinaryProtocolValue();
86          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TINY, binaryProtocolValue);
87      }
88      
89      private static void setDoubleBinaryProtocolValue() {
90          MySQLDoubleBinaryProtocolValue binaryProtocolValue = new MySQLDoubleBinaryProtocolValue();
91          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DOUBLE, binaryProtocolValue);
92      }
93      
94      private static void setFloatBinaryProtocolValue() {
95          MySQLFloatBinaryProtocolValue binaryProtocolValue = new MySQLFloatBinaryProtocolValue();
96          BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.FLOAT, binaryProtocolValue);
97      }
98      
99      private static void setDateBinaryProtocolValue() {
100         MySQLDateBinaryProtocolValue binaryProtocolValue = new MySQLDateBinaryProtocolValue();
101         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DATE, binaryProtocolValue);
102         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.DATETIME, binaryProtocolValue);
103         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TIMESTAMP, binaryProtocolValue);
104     }
105     
106     private static void setTimeBinaryProtocolValue() {
107         MySQLTimeBinaryProtocolValue binaryProtocolValue = new MySQLTimeBinaryProtocolValue();
108         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.TIME, binaryProtocolValue);
109     }
110     
111     private static void setNullBinaryProtocolValue() {
112         BINARY_PROTOCOL_VALUES.put(MySQLBinaryColumnType.NULL, null);
113     }
114     
115     /**
116      * Get binary protocol value.
117      * 
118      * @param binaryColumnType binary column type
119      * @return binary protocol value
120      */
121     public static MySQLBinaryProtocolValue getBinaryProtocolValue(final BinaryColumnType binaryColumnType) {
122         Preconditions.checkArgument(BINARY_PROTOCOL_VALUES.containsKey(binaryColumnType), "Cannot find MySQL type '%s' in column type when process binary protocol value", binaryColumnType);
123         return BINARY_PROTOCOL_VALUES.get(binaryColumnType);
124     }
125 }
