33  /**
34   * Unlock cluster updater.
35   */
36  @RequiredArgsConstructor
37  public final class UnlockClusterUpdater implements RALUpdater<UnlockClusterStatement> {
38      
39      @Override
40      @SuppressWarnings({"unchecked", "rawtypes"})
41      public void executeUpdate(final String databaseName, final UnlockClusterStatement sqlStatement) {
42          checkMode();
43          checkState();
44          ContextManager contextManager = ProxyContext.getInstance().getContextManager();
45          LockContext lockContext = contextManager.getInstanceContext().getLockContext();
46          GlobalLockDefinition lockDefinition = new GlobalLockDefinition(GlobalLockNames.CLUSTER_LOCK.getLockName());
47          if (lockContext.tryLock(lockDefinition, 3000L)) {
48              try {
49                  checkState();
50                  contextManager.getInstanceContext().getEventBusContext().post(new ClusterStatusChangedEvent(ClusterState.OK));
51                  // TODO unlock snapshot info if locked
52              } finally {
53                  lockContext.unlock(lockDefinition);
54              }
55          }
56      }
57      
58      private void checkMode() {
59          ShardingSpherePreconditions.checkState(ProxyContext.getInstance().getContextManager().getInstanceContext().isCluster(),
60                  () -> new UnsupportedSQLOperationException("Only allowed in cluster mode"));
61      }
62      
63      private void checkState() {
64          ClusterState currentState = ProxyContext.getInstance().getContextManager().getClusterStateContext().getCurrentState();
65          ShardingSpherePreconditions.checkState(ClusterState.OK != currentState, () -> new IllegalStateException("Cluster is not locked"));
66      }
67      
68      @Override
69      public String getType() {
70          return UnlockClusterStatement.class.getName();
71      }
72  }
