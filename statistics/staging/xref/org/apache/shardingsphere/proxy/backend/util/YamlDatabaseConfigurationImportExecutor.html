62  /**
63   * Yaml database configuration import executor.
64   */
65  public final class YamlDatabaseConfigurationImportExecutor {
66      
67      private final YamlProxyDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlProxyDataSourceConfigurationSwapper();
68      
69      private final DistSQLDataSourcePoolPropertiesValidator validateHandler = new DistSQLDataSourcePoolPropertiesValidator();
70      
71      /**
72       * Import proxy database from yaml configuration.
73       *
74       * @param yamlConfig yaml proxy database configuration
75       * @throws SQLException SQL exception
76       */
77      public void importDatabaseConfiguration(final YamlProxyDatabaseConfiguration yamlConfig) throws SQLException {
78          String databaseName = yamlConfig.getDatabaseName();
79          checkDatabase(databaseName);
80          checkDataSources(databaseName, yamlConfig.getDataSources());
81          addDatabase(databaseName);
82          try {
83              importDataSources(databaseName, yamlConfig.getDataSources());
84              importRules(databaseName, yamlConfig.getRules());
85          } catch (final ShardingSphereSQLException ex) {
86              dropDatabase(databaseName);
87              throw ex;
88          }
89      }
90      
91      private void checkDataSources(final String databaseName, final Map<String, YamlProxyDataSourceConfiguration> dataSources) {
92          ShardingSpherePreconditions.checkNotEmpty(dataSources, () -> new EmptyStorageUnitException(databaseName));
93      }
94      
95      private void checkDatabase(final String databaseName) {
96          ShardingSpherePreconditions.checkNotEmpty(databaseName, MissingRequiredDatabaseException::new);
97          ShardingSpherePreconditions.checkState(!ProxyContext.getInstance().databaseExists(databaseName), () -> new DatabaseCreateExistsException(databaseName));
98      }
99      
100     private void addDatabase(final String databaseName) throws SQLException {
101         ContextManager contextManager = ProxyContext.getInstance().getContextManager();
102         contextManager.getPersistServiceFacade().getMetaDataManagerPersistService().createDatabase(databaseName);
103         DatabaseType protocolType = DatabaseTypeEngine.getProtocolType(Collections.emptyMap(), contextManager.getMetaDataContexts().getMetaData().getProps());
104         contextManager.getMetaDataContexts().getMetaData().addDatabase(databaseName, protocolType, contextManager.getMetaDataContexts().getMetaData().getProps());
105     }
106     
107     private void importDataSources(final String databaseName, final Map<String, YamlProxyDataSourceConfiguration> yamlDataSourceMap) {
108         Map<String, DataSourcePoolProperties> propsMap = new LinkedHashMap<>(yamlDataSourceMap.size(), 1F);
109         for (Entry<String, YamlProxyDataSourceConfiguration> entry : yamlDataSourceMap.entrySet()) {
110             DataSourceConfiguration dataSourceConfig = dataSourceConfigSwapper.swap(entry.getValue());
111             propsMap.put(entry.getKey(), DataSourcePoolPropertiesCreator.create(dataSourceConfig));
112         }
113         validateHandler.validate(propsMap);
114         try {
115             ProxyContext.getInstance().getContextManager().getPersistServiceFacade().getMetaDataManagerPersistService().registerStorageUnits(databaseName, propsMap);
116         } catch (final SQLException ex) {
117             throw new StorageUnitsOperateException("import", propsMap.keySet(), ex);
118         }
119         Map<String, StorageUnit> storageUnits = ProxyContext.getInstance().getContextManager()
120                 .getMetaDataContexts().getMetaData().getDatabase(databaseName).getResourceMetaData().getStorageUnits();
121         Map<String, StorageNode> toBeAddedStorageNode = StorageUnitNodeMapCreator.create(propsMap);
122         for (Entry<String, DataSourcePoolProperties> entry : propsMap.entrySet()) {
123             storageUnits.put(entry.getKey(), new StorageUnit(toBeAddedStorageNode.get(entry.getKey()), entry.getValue(), DataSourcePoolCreator.create(entry.getValue())));
124         }
125     }
126     
127     private void importRules(final String databaseName, final Collection<YamlRuleConfiguration> yamlRuleConfigs) {
128         if (null == yamlRuleConfigs || yamlRuleConfigs.isEmpty()) {
129             return;
130         }
131         Collection<RuleConfiguration> ruleConfigs = new LinkedList<>();
132         MetaDataContexts metaDataContexts = ProxyContext.getInstance().getContextManager().getMetaDataContexts();
133         ShardingSphereDatabase database = metaDataContexts.getMetaData().getDatabase(databaseName);
134         swapToRuleConfigs(yamlRuleConfigs).values().forEach(each -> addRule(ruleConfigs, each, database));
135         ProxyContext.getInstance().getContextManager().getPersistServiceFacade().getMetaDataPersistService().getDatabaseRulePersistService()
136                 .persist(metaDataContexts.getMetaData().getDatabase(databaseName).getName(), ruleConfigs);
137     }
138     
139     private void addRule(final Collection<RuleConfiguration> ruleConfigs, final RuleConfiguration ruleConfig, final ShardingSphereDatabase database) {
140         RuleConfigurationCheckEngine.check(ruleConfig, database);
141         ruleConfigs.add(ruleConfig);
142         database.getRuleMetaData().getRules().add(buildRule(ruleConfig, database));
143     }
144     
145     @SuppressWarnings({"unchecked", "rawtypes"})
146     private ShardingSphereRule buildRule(final RuleConfiguration ruleConfig, final ShardingSphereDatabase database) {
147         DatabaseRuleBuilder ruleBuilder = OrderedSPILoader.getServices(DatabaseRuleBuilder.class, Collections.singleton(ruleConfig)).get(ruleConfig);
148         ComputeNodeInstanceContext computeNodeInstanceContext = ProxyContext.getInstance().getContextManager().getComputeNodeInstanceContext();
149         return ruleBuilder.build(ruleConfig, database.getName(), database.getProtocolType(), database.getResourceMetaData(), database.getRuleMetaData().getRules(), computeNodeInstanceContext);
150     }
151     
152     @SuppressWarnings({"rawtypes", "unchecked"})
153     private Map<Integer, RuleConfiguration> swapToRuleConfigs(final Collection<YamlRuleConfiguration> yamlRuleConfigs) {
154         Map<Integer, RuleConfiguration> result = new TreeMap<>(Comparator.reverseOrder());
155         for (YamlRuleConfiguration each : yamlRuleConfigs) {
156             YamlRuleConfigurationSwapper swapper = OrderedSPILoader.getServicesByClass(YamlRuleConfigurationSwapper.class, Collections.singleton(each.getRuleConfigurationType()))
157                     .get(each.getRuleConfigurationType());
158             result.put(swapper.getOrder(), (RuleConfiguration) swapper.swapToObject(each));
159         }
160         return result;
161     }
162     
163     private void dropDatabase(final String databaseName) throws SQLException {
164         ProxyContext.getInstance().getContextManager().getPersistServiceFacade().getMetaDataManagerPersistService().dropDatabase(databaseName);
165     }
