98      @BeforeEach
99      void setup() throws SQLException {
100         ContextManager contextManager = mockContextManager();
101         when(ProxyContext.getInstance().getContextManager()).thenReturn(contextManager);
102         ShardingSphereDatabase database = mockDatabase();
103         when(ProxyContext.getInstance().getDatabase("foo_db")).thenReturn(database);
104         ConnectionSession connectionSession = mock(ConnectionSession.class);
105         when(connectionSession.getDefaultDatabaseName()).thenReturn("foo_db");
106         when(ProxyBackendHandlerFactory.newInstance(any(PostgreSQLDatabaseType.class), anyString(), any(SQLStatement.class), eq(connectionSession), any(HintValueContext.class)))
107                 .thenReturn(proxyBackendHandler);
108         when(ProxyBackendHandlerFactory.newInstance(any(PostgreSQLDatabaseType.class), any(QueryContext.class), eq(connectionSession), anyBoolean())).thenReturn(proxyBackendHandler);
109         when(backendConnection.getConnectionSession()).thenReturn(connectionSession);
110     }

112     private ContextManager mockContextManager() {
113         ContextManager result = mock(ContextManager.class, Answers.RETURNS_DEEP_STUBS);
114         when(result.getMetaDataContexts().getMetaData().containsDatabase("foo_db")).thenReturn(true);
115         when(result.getMetaDataContexts().getMetaData().getProps().getValue(ConfigurationPropertyKey.SQL_SHOW)).thenReturn(false);
116         return result;
117     }

121         when(result.getProtocolType()).thenReturn(new PostgreSQLDatabaseType());
122         return result;
123     }

132     @Test
133     void assertExecuteSelectStatementAndReturnAllRows() throws SQLException {
134         QueryResponseHeader responseHeader = mock(QueryResponseHeader.class);
135         QueryHeader queryHeader = new QueryHeader("schema", "table", "columnLabel", "columnName", Types.INTEGER, "columnTypeName", 0, 0, false, false, false, false);
136         when(responseHeader.getQueryHeaders()).thenReturn(Collections.singletonList(queryHeader));
137         when(proxyBackendHandler.execute()).thenReturn(responseHeader);
138         when(proxyBackendHandler.next()).thenReturn(true, true, false);
139         when(proxyBackendHandler.getRowData()).thenReturn(new QueryResponseRow(Collections.singletonList(new QueryResponseCell(Types.INTEGER, 0))),
140                 new QueryResponseRow(Collections.singletonList(new QueryResponseCell(Types.INTEGER, 1))));
141         SelectStatementContext sqlStatementContext = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
142         when(sqlStatementContext.getSqlStatement()).thenReturn(new PostgreSQLSelectStatement());
143         PostgreSQLServerPreparedStatement preparedStatement =
144                 new PostgreSQLServerPreparedStatement("", sqlStatementContext, Collections.emptyList());
145         List<PostgreSQLValueFormat> resultFormats = new ArrayList<>(Arrays.asList(PostgreSQLValueFormat.TEXT, PostgreSQLValueFormat.BINARY));
146         Portal portal = new Portal("", preparedStatement, Collections.emptyList(), resultFormats, backendConnection);
147         portal.bind();
148         assertThat(portal.describe(), instanceOf(PostgreSQLRowDescriptionPacket.class));
149         List<PostgreSQLPacket> actualPackets = portal.execute(0);
150         assertThat(actualPackets.size(), is(3));
151         Iterator<PostgreSQLPacket> actualPacketsIterator = actualPackets.iterator();
152         assertThat(actualPacketsIterator.next(), instanceOf(PostgreSQLDataRowPacket.class));
153         assertThat(actualPacketsIterator.next(), instanceOf(PostgreSQLDataRowPacket.class));
154         assertThat(actualPacketsIterator.next(), instanceOf(PostgreSQLCommandCompletePacket.class));
155     }

157     @Test
158     void assertExecuteSelectStatementAndPortalSuspended() throws SQLException {
159         QueryResponseHeader responseHeader = mock(QueryResponseHeader.class);
160         QueryHeader queryHeader = new QueryHeader("schema", "table", "columnLabel", "columnName", Types.INTEGER, "columnTypeName", 0, 0, false, false, false, false);
161         when(responseHeader.getQueryHeaders()).thenReturn(Collections.singletonList(queryHeader));
162         when(proxyBackendHandler.execute()).thenReturn(responseHeader);
163         when(proxyBackendHandler.next()).thenReturn(true, true);
164         when(proxyBackendHandler.getRowData()).thenReturn(
165                 new QueryResponseRow(Collections.singletonList(new QueryResponseCell(Types.INTEGER, 0))),
166                 new QueryResponseRow(Collections.singletonList(new QueryResponseCell(Types.INTEGER, 1))));
167         SelectStatementContext selectStatementContext = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
168         when(selectStatementContext.getSqlStatement()).thenReturn(new PostgreSQLSelectStatement());
169         PostgreSQLServerPreparedStatement preparedStatement = new PostgreSQLServerPreparedStatement("", selectStatementContext, Collections.emptyList());
170         List<PostgreSQLValueFormat> resultFormats = new ArrayList<>(Arrays.asList(PostgreSQLValueFormat.TEXT, PostgreSQLValueFormat.BINARY));
171         Portal portal = new Portal("", preparedStatement, Collections.emptyList(), resultFormats, backendConnection);
172         portal.bind();
173         assertThat(portal.describe(), instanceOf(PostgreSQLRowDescriptionPacket.class));
174         List<PostgreSQLPacket> actualPackets = portal.execute(2);
175         assertThat(actualPackets.size(), is(3));
176         Iterator<PostgreSQLPacket> actualPacketsIterator = actualPackets.iterator();
177         assertThat(actualPacketsIterator.next(), instanceOf(PostgreSQLDataRowPacket.class));
178         assertThat(actualPacketsIterator.next(), instanceOf(PostgreSQLDataRowPacket.class));
179         assertThat(actualPacketsIterator.next(), instanceOf(PostgreSQLPortalSuspendedPacket.class));
180     }

208     @Test
209     void assertExecuteSetStatement() throws SQLException {
210         when(proxyBackendHandler.execute()).thenReturn(mock(UpdateResponseHeader.class));
211         when(proxyBackendHandler.next()).thenReturn(false);
212         String sql = "set client_encoding = utf8";
213         PostgreSQLSetStatement setStatement = new PostgreSQLSetStatement();
214         VariableAssignSegment variableAssignSegment = new VariableAssignSegment();
215         variableAssignSegment.setVariable(new VariableSegment());
216         setStatement.getVariableAssigns().add(variableAssignSegment);
217         PostgreSQLServerPreparedStatement preparedStatement = new PostgreSQLServerPreparedStatement(sql, new CommonSQLStatementContext<>(setStatement), Collections.emptyList());
218         Portal portal = new Portal("", preparedStatement, Collections.emptyList(), Collections.emptyList(), backendConnection);
219         portal.bind();
220         List<PostgreSQLPacket> actualPackets = portal.execute(0);
221         assertThat(actualPackets.size(), is(2));
222         assertThat(actualPackets.get(0), instanceOf(PostgreSQLCommandCompletePacket.class));
223         assertThat(actualPackets.get(1), instanceOf(PostgreSQLParameterStatusPacket.class));
224     }
