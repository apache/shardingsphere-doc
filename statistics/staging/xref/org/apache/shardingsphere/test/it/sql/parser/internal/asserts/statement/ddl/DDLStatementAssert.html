137 @NoArgsConstructor(access = AccessLevel.PRIVATE)

139     
140     /**
141      * Assert DDL statement is correct with expected parser result.
142      * 
143      * @param assertContext assert context
144      * @param actual actual DDL statement
145      * @param expected expected parser result
146      */
147     public static void assertIs(final SQLCaseAssertContext assertContext, final DDLStatement actual, final SQLParserTestCase expected) {
148         if (actual instanceof CreateTableStatement) {
149             CreateTableStatementAssert.assertIs(assertContext, (CreateTableStatement) actual, (CreateTableStatementTestCase) expected);
