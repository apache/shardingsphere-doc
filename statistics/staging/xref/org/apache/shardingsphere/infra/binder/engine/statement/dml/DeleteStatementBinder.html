18  package org.apache.shardingsphere.infra.binder.engine.statement.dml;
19  
20  import com.cedarsoftware.util.CaseInsensitiveMap.CaseInsensitiveString;
21  import com.google.common.collect.LinkedHashMultimap;
22  import com.google.common.collect.Multimap;
23  import lombok.SneakyThrows;
24  import org.apache.shardingsphere.infra.binder.engine.segment.from.TableSegmentBinder;
25  import org.apache.shardingsphere.infra.binder.engine.segment.from.context.TableSegmentBinderContext;
26  import org.apache.shardingsphere.infra.binder.engine.segment.where.WhereSegmentBinder;
27  import org.apache.shardingsphere.infra.binder.engine.statement.SQLStatementBinder;
28  import org.apache.shardingsphere.infra.binder.engine.statement.SQLStatementBinderContext;
29  import org.apache.shardingsphere.sql.parser.statement.core.statement.dml.DeleteStatement;
30  
31  /**
32   * Delete statement binder.
33   */
34  public final class DeleteStatementBinder implements SQLStatementBinder<DeleteStatement> {
35      
36      @Override
37      public DeleteStatement bind(final DeleteStatement sqlStatement, final SQLStatementBinderContext binderContext) {
38          DeleteStatement result = copy(sqlStatement);
39          Multimap<CaseInsensitiveString, TableSegmentBinderContext> tableBinderContexts = LinkedHashMultimap.create();
40          result.setTable(TableSegmentBinder.bind(sqlStatement.getTable(), binderContext, tableBinderContexts, LinkedHashMultimap.create()));
41          sqlStatement.getWhere().ifPresent(optional -> result.setWhere(WhereSegmentBinder.bind(optional, binderContext, tableBinderContexts, LinkedHashMultimap.create())));
42          return result;
43      }
44      
45      @SneakyThrows(ReflectiveOperationException.class)
46      private DeleteStatement copy(final DeleteStatement sqlStatement) {
47          DeleteStatement result = sqlStatement.getClass().getDeclaredConstructor().newInstance();
48          sqlStatement.getOrderBy().ifPresent(result::setOrderBy);
49          sqlStatement.getLimit().ifPresent(result::setLimit);
50          sqlStatement.getWithSegment().ifPresent(result::setWithSegment);
51          sqlStatement.getOutputSegment().ifPresent(result::setOutputSegment);
52          result.addParameterMarkerSegments(sqlStatement.getParameterMarkerSegments());
53          result.getCommentSegments().addAll(sqlStatement.getCommentSegments());
54          return result;
55      }
56  }
