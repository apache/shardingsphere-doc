41  /**
42   * Database type engine.
43   */
44  @NoArgsConstructor(access = AccessLevel.PRIVATE)
45  public final class DatabaseTypeEngine {
46      
47      private static final String DEFAULT_DATABASE_TYPE = "MySQL";
48      
49      /**
50       * Get database type.
51       *
52       * @param url database URL
53       * @return database type
54       */
55      public static DatabaseType getDatabaseType(final String url) {
56          Collection<DatabaseType> databaseTypes = ShardingSphereServiceLoader.getServiceInstances(DatabaseType.class).stream().filter(each -> matchURLs(url, each)).collect(Collectors.toList());
57          if (databaseTypes.isEmpty()) {
58              return TypedSPILoader.getService(DatabaseType.class, "SQL92");
59          }
60          for (DatabaseType each : databaseTypes) {
61              if (each instanceof BranchDatabaseType) {
62                  return each;
63              }
64          }
65          return databaseTypes.iterator().next();
66      }
67      
68      private static boolean matchURLs(final String url, final DatabaseType databaseType) {
69          return databaseType.getJdbcUrlPrefixes().stream().anyMatch(url::startsWith);
70      }
71      
72      /**
73       * Get protocol type.
74       * 
75       * @param databaseName database name
76       * @param databaseConfig database configuration
77       * @param props configuration properties
78       * @return protocol type
79       */
80      public static DatabaseType getProtocolType(final String databaseName, final DatabaseConfiguration databaseConfig, final ConfigurationProperties props) {
81          return findConfiguredDatabaseType(props).orElseGet(() -> getStorageType(DataSourceStateManager.getInstance().getEnabledDataSources(databaseName, databaseConfig)));
82      }
83      
84      /**
85       * Get protocol type.
86       *
87       * @param databaseConfigs database configurations
88       * @param props configuration properties
89       * @return protocol type
90       */
91      public static DatabaseType getProtocolType(final Map<String, ? extends DatabaseConfiguration> databaseConfigs, final ConfigurationProperties props) {
92          Optional<DatabaseType> configuredDatabaseType = findConfiguredDatabaseType(props);
93          return configuredDatabaseType.orElseGet(() -> getStorageType(getEnabledDataSources(databaseConfigs).values()));
94      }
95      
96      private static Map<String, DataSource> getEnabledDataSources(final Map<String, ? extends DatabaseConfiguration> databaseConfigs) {
97          Map<String, DataSource> result = new LinkedHashMap<>();
98          for (Entry<String, ? extends DatabaseConfiguration> entry : databaseConfigs.entrySet()) {
99              result.putAll(DataSourceStateManager.getInstance().getEnabledDataSourceMap(entry.getKey(), entry.getValue().getDataSources()));
100         }
101         return result;
102     }
103     
104     /**
105      * Get storage types.
106      *
107      * @param databaseName database name
108      * @param databaseConfig database configuration
109      * @return storage types
110      */
111     public static Map<String, DatabaseType> getStorageTypes(final String databaseName, final DatabaseConfiguration databaseConfig) {
112         Map<String, DatabaseType> result = new LinkedHashMap<>(databaseConfig.getDataSources().size(), 1F);
113         Map<String, DataSource> enabledDataSources = DataSourceStateManager.getInstance().getEnabledDataSourceMap(databaseName, databaseConfig.getDataSources());
114         for (Entry<String, DataSource> entry : enabledDataSources.entrySet()) {
115             result.put(entry.getKey(), getStorageType(entry.getValue()));
116         }
117         return result;
118     }
119     
120     /**
121      * Get storage types.
122      *
123      * @param dataSources data sources
124      * @return storage types
125      */
126     public static Map<String, DatabaseType> getStorageTypes(final Map<String, DataSource> dataSources) {
127         Map<String, DatabaseType> result = new LinkedHashMap<>(dataSources.size(), 1F);
128         for (Entry<String, DataSource> entry : dataSources.entrySet()) {
129             result.put(entry.getKey(), getStorageType(entry.getValue()));
130         }
131         return result;
132     }
133     
134     /**
135      * Get storage type.
136      *
137      * @param dataSources data sources
138      * @return storage type
139      */
140     public static DatabaseType getStorageType(final Collection<DataSource> dataSources) {
141         return dataSources.isEmpty() ? TypedSPILoader.getService(DatabaseType.class, DEFAULT_DATABASE_TYPE) : getStorageType(dataSources.iterator().next());
142     }
143     
144     private static DatabaseType getStorageType(final DataSource dataSource) {
145         try (Connection connection = dataSource.getConnection()) {
146             return getDatabaseType(connection.getMetaData().getURL());
147         } catch (final SQLException ex) {
148             throw new SQLWrapperException(ex);
149         }
150     }
151     
152     private static Optional<DatabaseType> findConfiguredDatabaseType(final ConfigurationProperties props) {
153         String configuredDatabaseType = props.getValue(ConfigurationPropertyKey.PROXY_FRONTEND_DATABASE_PROTOCOL_TYPE);
154         return configuredDatabaseType.isEmpty() ? Optional.empty() : Optional.of(DatabaseTypeEngine.getTrunkDatabaseType(configuredDatabaseType));
155     }
156     
157     /**
158      * Get trunk database type.
159      *
160      * @param name database name 
161      * @return trunk database type
162      */
163     public static DatabaseType getTrunkDatabaseType(final String name) {
164         DatabaseType databaseType = TypedSPILoader.getService(DatabaseType.class, name);
165         return databaseType instanceof BranchDatabaseType ? ((BranchDatabaseType) databaseType).getTrunkDatabaseType() : databaseType;
166     }
167     
168     /**
169      * Get name of trunk database type.
170      *
171      * @param databaseType database type
172      * @return name of trunk database type
173      */
174     public static String getTrunkDatabaseTypeName(final DatabaseType databaseType) {
175         return databaseType instanceof BranchDatabaseType ? ((BranchDatabaseType) databaseType).getTrunkDatabaseType().getType() : databaseType.getType();
176     }
177     
178     /**
179      * Get default schema name.
180      * 
181      * @param protocolType protocol type
182      * @param databaseName database name
183      * @return default schema name
184      */
185     public static String getDefaultSchemaName(final DatabaseType protocolType, final String databaseName) {
186         return protocolType.getDefaultSchema().orElseGet(() -> null == databaseName ? null : databaseName.toLowerCase());
187     }
188     
189     /**
190      * Get trunk and branch database types.
191      *
192      * @param trunkDatabaseTypes trunk database types
193      * @return database types
194      */
195     public static Collection<DatabaseType> getTrunkAndBranchDatabaseTypes(final Collection<String> trunkDatabaseTypes) {
196         Collection<DatabaseType> result = new LinkedList<>();
197         for (DatabaseType each : ShardingSphereServiceLoader.getServiceInstances(DatabaseType.class)) {
198             if (trunkDatabaseTypes.contains(each.getType()) || each instanceof BranchDatabaseType && trunkDatabaseTypes.contains(((BranchDatabaseType) each).getTrunkDatabaseType().getType())) {
199                 result.add(each);
200             }
201         }
202         return result;
203     }
