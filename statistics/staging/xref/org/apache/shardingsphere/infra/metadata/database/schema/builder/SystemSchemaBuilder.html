18  package org.apache.shardingsphere.infra.metadata.database.schema.builder;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import org.apache.shardingsphere.infra.config.props.ConfigurationProperties;
23  import org.apache.shardingsphere.infra.config.props.temporary.TemporaryConfigurationPropertyKey;
24  import org.apache.shardingsphere.infra.database.type.DatabaseType;
25  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereSchema;
26  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereTable;
27  import org.apache.shardingsphere.infra.yaml.schema.pojo.YamlShardingSphereTable;
28  import org.apache.shardingsphere.infra.yaml.schema.swapper.YamlTableSwapper;
29  import org.yaml.snakeyaml.Yaml;
30  
31  import java.io.InputStream;
32  import java.util.Collection;
33  import java.util.Collections;
34  import java.util.LinkedHashMap;
35  import java.util.LinkedList;
36  import java.util.Map;
37  import java.util.Properties;
38  
39  /**
40   * System schema builder.
41   */
42  @NoArgsConstructor(access = AccessLevel.PRIVATE)
43  public final class SystemSchemaBuilder {
44      
45      /**
46       * Build system schema.
47       * 
48       * @param databaseName database name
49       * @param databaseType database type
50       * @param props configuration properties                         
51       * @return ShardingSphere system schema map
52       */
53      public static Map<String, ShardingSphereSchema> build(final String databaseName, final DatabaseType databaseType, final ConfigurationProperties props) {
54          Map<String, ShardingSphereSchema> result = new LinkedHashMap<>(databaseType.getSystemSchemas().size(), 1F);
55          boolean isSystemSchemaMetaDataEnabled = isSystemSchemaMetaDataEnabled(props.getProps());
56          YamlTableSwapper swapper = new YamlTableSwapper();
57          for (String each : getSystemSchemas(databaseName, databaseType)) {
58              result.put(each.toLowerCase(), createSchema(each, getSchemaStreams(each, databaseType), swapper, isSystemSchemaMetaDataEnabled));
59          }
60          return result;
61      }
62      
63      private static boolean isSystemSchemaMetaDataEnabled(final Properties props) {
64          TemporaryConfigurationPropertyKey configKey = TemporaryConfigurationPropertyKey.SYSTEM_SCHEMA_METADATA_ENABLED;
65          return Boolean.parseBoolean(props.getOrDefault(configKey.getKey(), configKey.getDefaultValue()).toString());
66      }
67      
68      private static Collection<String> getSystemSchemas(final String originalDatabaseName, final DatabaseType databaseType) {
69          String databaseName = databaseType.getDefaultSchema().isPresent() ? "postgres" : originalDatabaseName;
70          return databaseType.getSystemDatabaseSchemaMap().getOrDefault(databaseName, Collections.emptyList());
71      }
72      
73      private static Collection<InputStream> getSchemaStreams(final String schemaName, final DatabaseType databaseType) {
74          SystemSchemaBuilderRule builderRule = SystemSchemaBuilderRule.valueOf(databaseType.getType(), schemaName);
75          Collection<InputStream> result = new LinkedList<>();
76          for (String each : builderRule.getTables()) {
77              result.add(Thread.currentThread().getContextClassLoader().getResourceAsStream("schema/" + databaseType.getType().toLowerCase() + "/" + schemaName + "/" + each + ".yaml"));
78          }
79          return result;
80      }
81      
82      private static ShardingSphereSchema createSchema(final String schemaName, final Collection<InputStream> schemaStreams, final YamlTableSwapper swapper,
83                                                       final boolean isSystemSchemaMetadataEnabled) {
84          Map<String, ShardingSphereTable> tables = new LinkedHashMap<>(schemaStreams.size(), 1F);
85          for (InputStream each : schemaStreams) {
86              YamlShardingSphereTable metaData = new Yaml().loadAs(each, YamlShardingSphereTable.class);
87              if (isSystemSchemaMetadataEnabled || KernelSupportedSystemTables.isSupportedSystemTable(schemaName, metaData.getName())) {
88                  tables.put(metaData.getName(), swapper.swapToObject(metaData));
89              }
90          }
91          return new ShardingSphereSchema(tables, Collections.emptyMap());
92      }
93  }
