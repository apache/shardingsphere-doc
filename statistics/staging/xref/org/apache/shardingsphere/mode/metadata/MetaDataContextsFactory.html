75      /**
76       * Create meta data contexts.
77       *
78       * @param persistService persist service
79       * @param param context manager builder parameter
80       * @param computeNodeInstanceContext compute node instance context
81       * @return meta data contexts
82       * @throws SQLException SQL exception
83       */
84      public static MetaDataContexts create(final MetaDataPersistService persistService, final ContextManagerBuilderParameter param,
85                                            final ComputeNodeInstanceContext computeNodeInstanceContext) throws SQLException {
86          boolean isDatabaseMetaDataExisted = !persistService.getDatabaseMetaDataService().loadAllDatabaseNames().isEmpty();
87          Map<String, DatabaseConfiguration> effectiveDatabaseConfigs = isDatabaseMetaDataExisted
88                  ? createEffectiveDatabaseConfigurations(getDatabaseNames(computeNodeInstanceContext, param.getDatabaseConfigs(), persistService), param.getDatabaseConfigs(), persistService)
89                  : param.getDatabaseConfigs();
90          // TODO load global data sources from persist service
91          Map<String, DataSource> globalDataSources = param.getGlobalDataSources();
92          Collection<RuleConfiguration> globalRuleConfigs;
93          if (isDatabaseMetaDataExisted) {
94              globalRuleConfigs = persistService.getGlobalRuleService().load();
95          } else if (computeNodeInstanceContext.isCluster()) {
96              globalRuleConfigs = new RuleConfigurationPersistDecorateEngine(computeNodeInstanceContext).tryRestore(param.getGlobalRuleConfigs());
97              param.getGlobalRuleConfigs().clear();
98              param.getGlobalRuleConfigs().addAll(globalRuleConfigs);
99          } else {
100             globalRuleConfigs = param.getGlobalRuleConfigs();
101         }
102         ConfigurationProperties props = isDatabaseMetaDataExisted ? new ConfigurationProperties(persistService.getPropsService().load()) : new ConfigurationProperties(param.getProps());
103         Map<String, ShardingSphereDatabase> databases = isDatabaseMetaDataExisted
104                 ? InternalMetaDataFactory.create(persistService, effectiveDatabaseConfigs, props, computeNodeInstanceContext)
105                 : ExternalMetaDataFactory.create(effectiveDatabaseConfigs, props, computeNodeInstanceContext);
106         ResourceMetaData globalResourceMetaData = new ResourceMetaData(globalDataSources);
107         RuleMetaData globalRuleMetaData = new RuleMetaData(GlobalRulesBuilder.buildRules(globalRuleConfigs, databases, props));
108         ShardingSphereMetaData shardingSphereMetaData = new ShardingSphereMetaData(databases, globalResourceMetaData, globalRuleMetaData, props);
109         ShardingSphereStatistics shardingSphereStatistics = initStatistics(persistService, shardingSphereMetaData);
110         MetaDataContexts result = new MetaDataContexts(shardingSphereMetaData, shardingSphereStatistics);
111         if (isDatabaseMetaDataExisted) {
112             restoreRules(result, computeNodeInstanceContext);
113         } else {
114             persistDatabaseConfigurations(result, param, persistService, computeNodeInstanceContext);
115             persistMetaData(result, persistService);
116         }
117         return result;
118     }

131     private static Collection<String> getDatabaseNames(final ComputeNodeInstanceContext computeNodeInstanceContext,
132                                                        final Map<String, DatabaseConfiguration> databaseConfigs, final MetaDataPersistService persistService) {
133         return computeNodeInstanceContext.getInstance().getMetaData() instanceof JDBCInstanceMetaData ? databaseConfigs.keySet() : persistService.getDatabaseMetaDataService().loadAllDatabaseNames();
134     }

136     private static Map<String, DatabaseConfiguration> createEffectiveDatabaseConfigurations(final Collection<String> databaseNames,
137                                                                                             final Map<String, DatabaseConfiguration> databaseConfigs, final MetaDataPersistService persistService) {
138         return databaseNames.stream().collect(Collectors.toMap(each -> each, each -> createEffectiveDatabaseConfiguration(each, databaseConfigs, persistService)));
139     }
140     
141     private static DatabaseConfiguration createEffectiveDatabaseConfiguration(final String databaseName,
142                                                                               final Map<String, DatabaseConfiguration> databaseConfigs, final MetaDataPersistService persistService) {
143         closeGeneratedDataSources(databaseName, databaseConfigs);
144         Map<String, DataSourceConfiguration> dataSources = persistService.loadDataSourceConfigurations(databaseName);
145         Collection<RuleConfiguration> databaseRuleConfigs = persistService.getDatabaseRulePersistService().load(databaseName);
146         return new DataSourceGeneratedDatabaseConfiguration(dataSources, databaseRuleConfigs);
147     }
148     
149     private static void closeGeneratedDataSources(final String databaseName, final Map<String, ? extends DatabaseConfiguration> databaseConfigs) {
150         if (databaseConfigs.containsKey(databaseName) && !databaseConfigs.get(databaseName).getStorageUnits().isEmpty()) {
151             databaseConfigs.get(databaseName).getDataSources().values().forEach(each -> new DataSourcePoolDestroyer(each).asyncDestroy());
152         }
153     }

155     private static ShardingSphereStatistics initStatistics(final MetaDataPersistService persistService, final ShardingSphereMetaData metaData) {
156         if (metaData.getDatabases().isEmpty()) {
157             return new ShardingSphereStatistics();
158         }
159         DatabaseType protocolType = metaData.getDatabases().values().iterator().next().getProtocolType();
160         DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(protocolType).getDialectDatabaseMetaData();
161         // TODO can `protocolType instanceof SchemaSupportedDatabaseType ? "PostgreSQL" : protocolType.getType()` replace to trunk database type?
162         DatabaseType databaseType = dialectDatabaseMetaData.getDefaultSchema().isPresent() ? TypedSPILoader.getService(DatabaseType.class, "PostgreSQL") : protocolType;
163         Optional<ShardingSphereStatisticsBuilder> statisticsBuilder = DatabaseTypedSPILoader.findService(ShardingSphereStatisticsBuilder.class, databaseType);
164         if (!statisticsBuilder.isPresent()) {
165             return new ShardingSphereStatistics();
166         }
167         ShardingSphereStatistics result = statisticsBuilder.get().build(metaData);
168         Optional<ShardingSphereStatistics> loadedStatistics = persistService.getShardingSphereDataPersistService().load(metaData);
169         loadedStatistics.ifPresent(optional -> useLoadedToReplaceInit(result, optional));
170         return result;
171     }
172     
173     private static void useLoadedToReplaceInit(final ShardingSphereStatistics initStatistics, final ShardingSphereStatistics loadedStatistics) {
174         for (Entry<String, ShardingSphereDatabaseData> entry : initStatistics.getDatabaseData().entrySet()) {
175             if (loadedStatistics.getDatabaseData().containsKey(entry.getKey())) {
176                 useLoadedToReplaceInitByDatabaseData(entry.getValue(), loadedStatistics.getDatabaseData().get(entry.getKey()));
177             }
178         }
179     }
180     
181     private static void useLoadedToReplaceInitByDatabaseData(final ShardingSphereDatabaseData initDatabaseData, final ShardingSphereDatabaseData loadedDatabaseData) {
182         for (Entry<String, ShardingSphereSchemaData> entry : initDatabaseData.getSchemaData().entrySet()) {
183             if (loadedDatabaseData.getSchemaData().containsKey(entry.getKey())) {
184                 useLoadedToReplaceInitBySchemaData(entry.getValue(), loadedDatabaseData.getSchemaData().get(entry.getKey()));
185             }
186         }
187     }
188     
189     private static void useLoadedToReplaceInitBySchemaData(final ShardingSphereSchemaData initSchemaData, final ShardingSphereSchemaData loadedSchemaData) {
190         for (Entry<String, ShardingSphereTableData> entry : initSchemaData.getTableData().entrySet()) {
191             if (loadedSchemaData.getTableData().containsKey(entry.getKey())) {
192                 entry.setValue(loadedSchemaData.getTableData().get(entry.getKey()));
193             }
194         }
195     }

197     @SuppressWarnings({"unchecked", "rawtypes"})
198     private static void restoreRules(final MetaDataContexts metaDataContexts, final ComputeNodeInstanceContext computeNodeInstanceContext) {
199         if (!computeNodeInstanceContext.isCluster()) {
200             return;
201         }
202         for (RuleConfigurationPersistDecorator each : ShardingSphereServiceLoader.getServiceInstances(RuleConfigurationPersistDecorator.class)) {
203             ShardingSphereRule rule = metaDataContexts.getMetaData().getGlobalRuleMetaData().getSingleRule(each.getRuleType());
204             metaDataContexts.getMetaData().getGlobalRuleMetaData().getRules().removeIf(eachRule -> each.getRuleType() == eachRule.getClass());
205             RuleConfiguration restoredRuleConfig = each.restore(rule.getConfiguration());
206             ShardingSphereRule rebuiltRule = GlobalRulesBuilder.buildRules(
207                     Collections.singleton(restoredRuleConfig), metaDataContexts.getMetaData().getDatabases(), metaDataContexts.getMetaData().getProps()).iterator().next();
208             metaDataContexts.getMetaData().getGlobalRuleMetaData().getRules().add(rebuiltRule);
209         }
210     }

212     private static void persistDatabaseConfigurations(final MetaDataContexts metadataContexts, final ContextManagerBuilderParameter param, final MetaDataPersistService persistService,
213                                                       final ComputeNodeInstanceContext computeNodeInstanceContext) {
214         RuleConfigurationPersistDecorateEngine ruleConfigPersistDecorateEngine = new RuleConfigurationPersistDecorateEngine(computeNodeInstanceContext);
215         persistService.persistGlobalRuleConfiguration(ruleConfigPersistDecorateEngine.decorate(param.getGlobalRuleConfigs()), param.getProps());
216         for (Entry<String, ? extends DatabaseConfiguration> entry : param.getDatabaseConfigs().entrySet()) {
217             String databaseName = entry.getKey();
218             persistService.persistConfigurations(entry.getKey(), entry.getValue(),
219                     metadataContexts.getMetaData().getDatabase(databaseName).getResourceMetaData().getStorageUnits().entrySet().stream()
220                             .collect(Collectors.toMap(Entry::getKey, each -> each.getValue().getDataSource(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new)),
221                     metadataContexts.getMetaData().getDatabase(databaseName).getRuleMetaData().getRules());
222         }
223     }

225     private static void persistMetaData(final MetaDataContexts metaDataContexts, final MetaDataPersistService persistService) {
226         metaDataContexts.getMetaData().getDatabases().values().forEach(each -> each.getSchemas().forEach((schemaName, schema) -> {
227             if (schema.isEmpty()) {
228                 persistService.getDatabaseMetaDataService().addSchema(each.getName(), schemaName);
229             }
230             persistService.getDatabaseMetaDataService().getTableMetaDataPersistService().persist(each.getName(), schemaName, schema.getTables());
231         }));
232         metaDataContexts.getStatistics().getDatabaseData().forEach((databaseName, databaseData) -> databaseData.getSchemaData().forEach((schemaName, schemaData) -> persistService
233                 .getShardingSphereDataPersistService().persist(databaseName, schemaName, schemaData, metaDataContexts.getMetaData().getDatabases())));
234     }

236     /**
237      * Create meta data contexts by switch resource.
238      *
239      * @param databaseName database name
240      * @param internalLoadMetaData internal load meta data
241      * @param switchingResource switching resource
242      * @param originalMetaDataContexts original meta data contexts
243      * @param metaDataPersistService meta data persist service
244      * @param computeNodeInstanceContext compute node instance context
245      * @return meta data contexts
246      * @throws SQLException SQL exception
247      */
248     public static MetaDataContexts createBySwitchResource(final String databaseName, final boolean internalLoadMetaData, final SwitchingResource switchingResource,
249                                                           final MetaDataContexts originalMetaDataContexts, final MetaDataPersistService metaDataPersistService,
250                                                           final ComputeNodeInstanceContext computeNodeInstanceContext) throws SQLException {
251         Map<String, ShardingSphereDatabase> changedDatabases =
252                 createChangedDatabases(databaseName, internalLoadMetaData, switchingResource, null, originalMetaDataContexts, metaDataPersistService, computeNodeInstanceContext);
253         ConfigurationProperties props = originalMetaDataContexts.getMetaData().getProps();
254         RuleMetaData changedGlobalMetaData = new RuleMetaData(
255                 GlobalRulesBuilder.buildRules(originalMetaDataContexts.getMetaData().getGlobalRuleMetaData().getConfigurations(), changedDatabases, props));
256         return create(metaDataPersistService, new ShardingSphereMetaData(changedDatabases, originalMetaDataContexts.getMetaData().getGlobalResourceMetaData(), changedGlobalMetaData, props));
257     }

282     /**
283      * Create changed databases by switch resource.
284      *
285      * @param databaseName database name
286      * @param internalLoadMetaData internal load meta data
287      * @param switchingResource switching resource
288      * @param ruleConfigs rule configurations
289      * @param originalMetaDataContext original meta data contexts
290      * @param metaDataPersistService meta data persist service
291      * @param computeNodeInstanceContext compute node instance context
292      * @return changed databases
293      * @throws SQLException SQL exception
294      */
295     public static Map<String, ShardingSphereDatabase> createChangedDatabases(final String databaseName, final boolean internalLoadMetaData,
296                                                                              final SwitchingResource switchingResource, final Collection<RuleConfiguration> ruleConfigs,
297                                                                              final MetaDataContexts originalMetaDataContext,
298                                                                              final MetaDataPersistService metaDataPersistService,
299                                                                              final ComputeNodeInstanceContext computeNodeInstanceContext) throws SQLException {
300         ResourceMetaData effectiveResourceMetaData = getEffectiveResourceMetaData(originalMetaDataContext.getMetaData().getDatabase(databaseName), switchingResource);
301         Collection<RuleConfiguration> toBeCreatedRuleConfigs = null == ruleConfigs
302                 ? originalMetaDataContext.getMetaData().getDatabase(databaseName).getRuleMetaData().getConfigurations()
303                 : ruleConfigs;
304         DatabaseConfiguration toBeCreatedDatabaseConfig = getDatabaseConfiguration(effectiveResourceMetaData, switchingResource, toBeCreatedRuleConfigs);
305         ShardingSphereDatabase changedDatabase = createChangedDatabase(originalMetaDataContext.getMetaData().getDatabase(databaseName).getName(), internalLoadMetaData,
306                 metaDataPersistService, toBeCreatedDatabaseConfig, originalMetaDataContext.getMetaData().getProps(), computeNodeInstanceContext);
307         Map<String, ShardingSphereDatabase> result = new LinkedHashMap<>(originalMetaDataContext.getMetaData().getDatabases());
308         result.put(databaseName.toLowerCase(), changedDatabase);
309         return result;
310     }

312     private static ResourceMetaData getEffectiveResourceMetaData(final ShardingSphereDatabase database, final SwitchingResource resource) {
313         Map<StorageNode, DataSource> storageNodes = getStorageNodes(database.getResourceMetaData().getDataSources(), resource);
314         Map<String, StorageUnit> storageUnits = getStorageUnits(database.getResourceMetaData().getStorageUnits(), resource);
315         return new ResourceMetaData(storageNodes, storageUnits);
316     }
317     
318     private static Map<StorageNode, DataSource> getStorageNodes(final Map<StorageNode, DataSource> currentStorageNodes, final SwitchingResource resource) {
319         Map<StorageNode, DataSource> result = new LinkedHashMap<>(currentStorageNodes.size(), 1F);
320         for (Entry<StorageNode, DataSource> entry : currentStorageNodes.entrySet()) {
321             if (null == resource || !resource.getStaleDataSources().containsKey(entry.getKey())) {
322                 result.put(entry.getKey(), entry.getValue());
323             }
324         }
325         return result;
326     }
327     
328     private static Map<String, StorageUnit> getStorageUnits(final Map<String, StorageUnit> currentStorageUnits, final SwitchingResource resource) {
329         Map<String, StorageUnit> result = new LinkedHashMap<>(currentStorageUnits.size(), 1F);
330         for (Entry<String, StorageUnit> entry : currentStorageUnits.entrySet()) {
331             if (null == resource || !resource.getStaleStorageUnitNames().contains(entry.getKey())) {
332                 result.put(entry.getKey(), entry.getValue());
333             }
334         }
335         return result;
336     }

338     private static DatabaseConfiguration getDatabaseConfiguration(final ResourceMetaData resourceMetaData, final SwitchingResource switchingResource,
339                                                                   final Collection<RuleConfiguration> toBeCreatedRuleConfigs) {
340         Map<String, DataSourcePoolProperties> propsMap = null == switchingResource ? resourceMetaData.getStorageUnits().entrySet().stream()
341                 .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSourcePoolProperties(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new))
342                 : switchingResource.getMergedDataSourcePoolPropertiesMap();
343         return new DataSourceProvidedDatabaseConfiguration(getMergedStorageNodeDataSources(resourceMetaData, switchingResource), toBeCreatedRuleConfigs, propsMap);
344     }
345     
346     private static Map<StorageNode, DataSource> getMergedStorageNodeDataSources(final ResourceMetaData currentResourceMetaData, final SwitchingResource switchingResource) {
347         Map<StorageNode, DataSource> result = currentResourceMetaData.getDataSources();
348         if (null != switchingResource && !switchingResource.getNewDataSources().isEmpty()) {
349             result.putAll(switchingResource.getNewDataSources());
350         }
351         return result;
352     }
353     
354     private static ShardingSphereDatabase createChangedDatabase(final String databaseName, final boolean internalLoadMetaData, final MetaDataPersistService persistService,
355                                                                 final DatabaseConfiguration databaseConfig, final ConfigurationProperties props,
356                                                                 final ComputeNodeInstanceContext computeNodeInstanceContext) throws SQLException {
357         return internalLoadMetaData
358                 ? InternalMetaDataFactory.create(databaseName, persistService, databaseConfig, props, computeNodeInstanceContext)
359                 : ExternalMetaDataFactory.create(databaseName, databaseConfig, props, computeNodeInstanceContext);
360     }
