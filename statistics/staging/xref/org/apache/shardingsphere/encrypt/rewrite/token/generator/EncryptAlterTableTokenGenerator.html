54  /**
55   * Alter table token generator for encrypt.
56   */
57  @Setter
58  public final class EncryptAlterTableTokenGenerator implements CollectionSQLTokenGenerator<AlterTableStatementContext>, EncryptRuleAware {
59      
60      private EncryptRule encryptRule;
61      
62      @Override
63      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
64          return sqlStatementContext instanceof AlterTableStatementContext;
65      }
66      
67      @Override
68      public Collection<SQLToken> generateSQLTokens(final AlterTableStatementContext alterTableStatementContext) {
69          String tableName = alterTableStatementContext.getSqlStatement().getTable().getTableName().getIdentifier().getValue();
70          EncryptTable encryptTable = encryptRule.getEncryptTable(tableName);
71          Collection<SQLToken> result = new LinkedList<>(getAddColumnTokens(encryptTable, alterTableStatementContext.getSqlStatement().getAddColumnDefinitions()));
72          result.addAll(getModifyColumnTokens(encryptTable, alterTableStatementContext.getSqlStatement().getModifyColumnDefinitions()));
73          result.addAll(getChangeColumnTokens(encryptTable, alterTableStatementContext.getSqlStatement().getChangeColumnDefinitions()));
74          List<SQLToken> dropColumnTokens = getDropColumnTokens(encryptTable, alterTableStatementContext.getSqlStatement().getDropColumnDefinitions());
75          String databaseName = alterTableStatementContext.getDatabaseType().getType();
76          if ("SQLServer".equals(databaseName)) {
77              result.addAll(mergeDropColumnStatement(dropColumnTokens, "", ""));
78          } else if ("Oracle".equals(databaseName)) {
79              result.addAll(mergeDropColumnStatement(dropColumnTokens, "(", ")"));
80          } else {
81              result.addAll(dropColumnTokens);
82          }
83          return result;
84      }
85      
86      private Collection<SQLToken> getAddColumnTokens(final EncryptTable encryptTable, final Collection<AddColumnDefinitionSegment> segments) {
87          Collection<SQLToken> result = new LinkedList<>();
88          for (AddColumnDefinitionSegment each : segments) {
89              result.addAll(getAddColumnTokens(encryptTable, each));
90          }
91          return result;
92      }
93      
94      private Collection<SQLToken> getAddColumnTokens(final EncryptTable encryptTable, final AddColumnDefinitionSegment segment) {
95          Collection<SQLToken> result = new LinkedList<>();
96          for (ColumnDefinitionSegment each : segment.getColumnDefinitions()) {
97              String columnName = each.getColumnName().getIdentifier().getValue();
98              if (encryptTable.isEncryptColumn(columnName)) {
99                  result.addAll(getAddColumnTokens(encryptTable.getEncryptColumn(columnName), segment, each));
100             }
101         }
102         getAddColumnPositionToken(encryptTable, segment).ifPresent(result::add);
103         return result;
104     }
105     
106     private Collection<SQLToken> getAddColumnTokens(final EncryptColumn encryptColumn,
107                                                     final AddColumnDefinitionSegment addColumnDefinitionSegment, final ColumnDefinitionSegment columnDefinitionSegment) {
108         Collection<SQLToken> result = new LinkedList<>();
109         result.add(new RemoveToken(columnDefinitionSegment.getStartIndex(), columnDefinitionSegment.getStopIndex()));
110         result.add(new EncryptColumnToken(columnDefinitionSegment.getStopIndex() + 1, columnDefinitionSegment.getStopIndex(),
111                 encryptColumn.getCipher().getName(), EncryptColumnDataType.DEFAULT_DATA_TYPE));
112         encryptColumn.getAssistedQuery().map(optional -> new EncryptColumnToken(addColumnDefinitionSegment.getStopIndex() + 1,
113                 addColumnDefinitionSegment.getStopIndex(), ", ADD COLUMN " + optional.getName(), EncryptColumnDataType.DEFAULT_DATA_TYPE)).ifPresent(result::add);
114         encryptColumn.getLikeQuery().map(optional -> new EncryptColumnToken(addColumnDefinitionSegment.getStopIndex() + 1,
115                 addColumnDefinitionSegment.getStopIndex(), ", ADD COLUMN " + optional.getName(), EncryptColumnDataType.DEFAULT_DATA_TYPE)).ifPresent(result::add);
116         return result;
117     }
118     
119     private Optional<SQLToken> getAddColumnPositionToken(final EncryptTable encryptTable, final AddColumnDefinitionSegment segment) {
120         Optional<ColumnPositionSegment> columnPositionSegment = segment.getColumnPosition().filter(optional -> null != optional.getColumnName());
121         if (columnPositionSegment.isPresent()) {
122             String columnName = columnPositionSegment.get().getColumnName().getIdentifier().getValue();
123             if (encryptTable.isEncryptColumn(columnName)) {
124                 return Optional.of(getPositionColumnToken(encryptTable.getEncryptColumn(columnName), segment.getColumnPosition().get()));
125             }
126         }
127         return Optional.empty();
128     }
129     
130     private EncryptAlterTableToken getPositionColumnToken(final EncryptColumn encryptColumn, final ColumnPositionSegment segment) {
131         return new EncryptAlterTableToken(segment.getColumnName().getStartIndex(), segment.getStopIndex(), encryptColumn.getCipher().getName(), null);
132     }
133     
134     private Collection<SQLToken> getModifyColumnTokens(final EncryptTable encryptTable, final Collection<ModifyColumnDefinitionSegment> segments) {
135         Collection<SQLToken> result = new LinkedList<>();
136         for (ModifyColumnDefinitionSegment each : segments) {
137             String columnName = each.getColumnDefinition().getColumnName().getIdentifier().getValue();
138             ShardingSpherePreconditions.checkState(!encryptTable.isEncryptColumn(columnName), () -> new UnsupportedOperationException("Unsupported operation 'modify' for the cipher column"));
139             each.getColumnPosition().flatMap(optional -> getColumnPositionToken(encryptTable, optional)).ifPresent(result::add);
140         }
141         return result;
142     }
143     
144     private Optional<SQLToken> getColumnPositionToken(final EncryptTable encryptTable, final ColumnPositionSegment segment) {
145         if (null == segment.getColumnName()) {
146             return Optional.empty();
147         }
148         String columnName = segment.getColumnName().getIdentifier().getValue();
149         return encryptTable.isEncryptColumn(columnName) ? Optional.of(getPositionColumnToken(encryptTable.getEncryptColumn(columnName), segment)) : Optional.empty();
150     }
151     
152     private Collection<SQLToken> getChangeColumnTokens(final EncryptTable encryptTable, final Collection<ChangeColumnDefinitionSegment> segments) {
153         Collection<SQLToken> result = new LinkedList<>();
154         for (ChangeColumnDefinitionSegment each : segments) {
155             String columnName = each.getPreviousColumn().getIdentifier().getValue();
156             ShardingSpherePreconditions.checkState(!encryptTable.isEncryptColumn(columnName), () -> new UnsupportedOperationException("Unsupported operation 'change' for the cipher column"));
157             result.addAll(getChangeColumnTokens(encryptTable, each));
158             each.getColumnPosition().flatMap(optional -> getColumnPositionToken(encryptTable, optional)).ifPresent(result::add);
159         }
160         return result;
161     }
162     
163     private Collection<SQLToken> getChangeColumnTokens(final EncryptTable encryptTable, final ChangeColumnDefinitionSegment segment) {
164         String previousColumnName = segment.getPreviousColumn().getIdentifier().getValue();
165         String columnName = segment.getColumnDefinition().getColumnName().getIdentifier().getValue();
166         isSameEncryptColumn(encryptTable, previousColumnName, columnName);
167         if (!encryptTable.isEncryptColumn(columnName) || !encryptTable.isEncryptColumn(previousColumnName)) {
168             return Collections.emptyList();
169         }
170         Collection<SQLToken> result = new LinkedList<>();
171         EncryptColumn previousEncryptColumn = encryptTable.getEncryptColumn(previousColumnName);
172         EncryptColumn encryptColumn = encryptTable.getEncryptColumn(columnName);
173         result.addAll(getPreviousColumnTokens(previousEncryptColumn, segment));
174         result.addAll(getColumnTokens(previousEncryptColumn, encryptColumn, segment));
175         return result;
176     }
177     
178     private void isSameEncryptColumn(final EncryptTable encryptTable, final String previousColumnName, final String columnName) {
179         Optional<EncryptAlgorithm> previousEncryptor = encryptTable.findEncryptor(previousColumnName);
180         Optional<EncryptAlgorithm> currentEncryptor = encryptTable.findEncryptor(columnName);
181         if (!previousEncryptor.isPresent() && !currentEncryptor.isPresent()) {
182             return;
183         }
184         ShardingSpherePreconditions.checkState(previousEncryptor.equals(currentEncryptor) && checkPreviousAndAfterHasSameColumnNumber(encryptTable, previousColumnName, columnName),
185                 () -> new EncryptColumnAlterException(encryptTable.getTable(), columnName, previousColumnName));
186     }
187     
188     private boolean checkPreviousAndAfterHasSameColumnNumber(final EncryptTable encryptTable, final String previousColumnName, final String columnName) {
189         EncryptColumn previousEncryptColumn = encryptTable.getEncryptColumn(previousColumnName);
190         EncryptColumn encryptColumn = encryptTable.getEncryptColumn(columnName);
191         if (previousEncryptColumn.getAssistedQuery().isPresent() && !encryptColumn.getAssistedQuery().isPresent()) {
192             return false;
193         }
194         if (previousEncryptColumn.getLikeQuery().isPresent() && !encryptColumn.getLikeQuery().isPresent()) {
195             return false;
196         }
197         return previousEncryptColumn.getAssistedQuery().isPresent() || !encryptColumn.getAssistedQuery().isPresent();
198     }
199     
200     private Collection<SQLToken> getPreviousColumnTokens(final EncryptColumn previousEncryptColumn, final ChangeColumnDefinitionSegment segment) {
201         Collection<SQLToken> result = new LinkedList<>();
202         result.add(new RemoveToken(segment.getPreviousColumn().getStartIndex(), segment.getPreviousColumn().getStopIndex()));
203         result.add(new EncryptAlterTableToken(segment.getPreviousColumn().getStopIndex() + 1, segment.getPreviousColumn().getStopIndex(), previousEncryptColumn.getCipher().getName(), null));
204         return result;
205     }
206     
207     private Collection<SQLToken> getColumnTokens(final EncryptColumn previousEncryptColumn, final EncryptColumn encryptColumn, final ChangeColumnDefinitionSegment segment) {
208         Collection<SQLToken> result = new LinkedList<>();
209         result.add(new RemoveToken(segment.getColumnDefinition().getColumnName().getStartIndex(), segment.getColumnDefinition().getStopIndex()));
210         result.add(new EncryptColumnToken(segment.getColumnDefinition().getStopIndex() + 1, segment.getColumnDefinition().getStopIndex(),
211                 encryptColumn.getCipher().getName(), EncryptColumnDataType.DEFAULT_DATA_TYPE));
212         previousEncryptColumn.getAssistedQuery().map(optional -> new EncryptColumnToken(segment.getStopIndex() + 1, segment.getStopIndex(),
213                 ", CHANGE COLUMN " + optional.getName() + " " + encryptColumn.getAssistedQuery().map(AssistedQueryColumnItem::getName).orElse(""),
214                 EncryptColumnDataType.DEFAULT_DATA_TYPE)).ifPresent(result::add);
215         previousEncryptColumn.getLikeQuery().map(optional -> new EncryptColumnToken(segment.getStopIndex() + 1, segment.getStopIndex(),
216                 ", CHANGE COLUMN " + optional.getName() + " " + encryptColumn.getLikeQuery().map(LikeQueryColumnItem::getName).orElse(""),
217                 EncryptColumnDataType.DEFAULT_DATA_TYPE)).ifPresent(result::add);
218         return result;
219     }
220     
221     private List<SQLToken> getDropColumnTokens(final EncryptTable encryptTable, final Collection<DropColumnDefinitionSegment> segments) {
222         List<SQLToken> result = new ArrayList<>();
223         for (DropColumnDefinitionSegment each : segments) {
224             result.addAll(getDropColumnTokens(encryptTable, each));
225         }
226         return result;
227     }
228     
229     private Collection<SQLToken> getDropColumnTokens(final EncryptTable encryptTable, final DropColumnDefinitionSegment segment) {
230         Collection<SQLToken> result = new LinkedList<>();
231         for (ColumnSegment each : segment.getColumns()) {
232             ShardingSpherePreconditions.checkState(!encryptTable.isEncryptColumn(each.getQualifiedName()),
233                     () -> new UnsupportedOperationException("Unsupported operation 'drop' for the cipher column"));
234         }
235         return result;
236     }
237     
238     private Collection<SQLToken> mergeDropColumnStatement(final List<SQLToken> dropSQLTokens, final String leftJoiner, final String rightJoiner) {
239         Collection<SQLToken> result = new LinkedList<>();
240         Collection<String> dropColumns = new LinkedList<>();
241         int lastStartIndex = -1;
242         for (int i = 0; i < dropSQLTokens.size(); i++) {
243             SQLToken token = dropSQLTokens.get(i);
244             if (token instanceof RemoveToken) {
245                 result.add(0 == i ? token : new RemoveToken(lastStartIndex, ((RemoveToken) token).getStopIndex()));
246             } else {
247                 EncryptAlterTableToken encryptAlterTableToken = (EncryptAlterTableToken) token;
248                 dropColumns.add(encryptAlterTableToken.getColumnName());
249                 if (i == dropSQLTokens.size() - 1) {
250                     result.add(new EncryptAlterTableToken(token.getStartIndex(), encryptAlterTableToken.getStopIndex(), leftJoiner + String.join(",", dropColumns) + rightJoiner, "DROP COLUMN"));
251                 }
252             }
253             lastStartIndex = ((Substitutable) token).getStartIndex();
254         }
255         return result;
256     }
