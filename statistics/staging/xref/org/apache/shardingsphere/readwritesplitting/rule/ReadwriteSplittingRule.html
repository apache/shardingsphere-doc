45  /**
46   * Readwrite-splitting rule.
47   */
48  public final class ReadwriteSplittingRule implements DatabaseRule {
49      
50      @Getter
51      private final ReadwriteSplittingRuleConfiguration configuration;
52      
53      private final Map<String, LoadBalanceAlgorithm> loadBalancers;
54      
55      @Getter
56      private final Map<String, ReadwriteSplittingDataSourceGroupRule> dataSourceRuleGroups;
57      
58      @Getter
59      private final RuleAttributes attributes;
60      
61      public ReadwriteSplittingRule(final String databaseName, final ReadwriteSplittingRuleConfiguration ruleConfig, final ComputeNodeInstanceContext computeNodeInstanceContext) {
62          configuration = ruleConfig;
63          loadBalancers = createLoadBalancers(ruleConfig);
64          dataSourceRuleGroups = createDataSourceGroupRules(databaseName, ruleConfig);
65          attributes = new RuleAttributes(
66                  new ReadwriteSplittingDataSourceMapperRuleAttribute(dataSourceRuleGroups.values()),
67                  new ReadwriteSplittingStaticDataSourceRuleAttribute(databaseName, dataSourceRuleGroups, computeNodeInstanceContext),
68                  new ReadwriteSplittingExportableRuleAttribute(dataSourceRuleGroups),
69                  new ReadwriteSplittingStorageConnectorReusableRuleAttribute());
70      }
71      
72      private Map<String, LoadBalanceAlgorithm> createLoadBalancers(final ReadwriteSplittingRuleConfiguration ruleConfig) {
73          Map<String, LoadBalanceAlgorithm> result = new HashMap<>(ruleConfig.getDataSourceGroups().size(), 1F);
74          for (ReadwriteSplittingDataSourceGroupRuleConfiguration each : ruleConfig.getDataSourceGroups()) {
75              if (ruleConfig.getLoadBalancers().containsKey(each.getLoadBalancerName())) {
76                  AlgorithmConfiguration algorithmConfig = ruleConfig.getLoadBalancers().get(each.getLoadBalancerName());
77                  result.put(each.getName() + "." + each.getLoadBalancerName(), TypedSPILoader.getService(LoadBalanceAlgorithm.class, algorithmConfig.getType(), algorithmConfig.getProps()));
78              }
79          }
80          return result;
81      }
82      
83      private Map<String, ReadwriteSplittingDataSourceGroupRule> createDataSourceGroupRules(final String databaseName, final ReadwriteSplittingRuleConfiguration ruleConfig) {
84          Map<String, ReadwriteSplittingDataSourceGroupRule> result = new HashMap<>(ruleConfig.getDataSourceGroups().size(), 1F);
85          for (ReadwriteSplittingDataSourceGroupRuleConfiguration each : ruleConfig.getDataSourceGroups()) {
86              result.putAll(createDataSourceGroupRules(databaseName, each));
87          }
88          return result;
89      }
90      
91      private Map<String, ReadwriteSplittingDataSourceGroupRule> createDataSourceGroupRules(final String databaseName, final ReadwriteSplittingDataSourceGroupRuleConfiguration config) {
92          LoadBalanceAlgorithm loadBalanceAlgorithm = loadBalancers.getOrDefault(config.getName() + "." + config.getLoadBalancerName(), TypedSPILoader.getService(LoadBalanceAlgorithm.class, null));
93          return createStaticDataSourceGroupRules(databaseName, config, loadBalanceAlgorithm);
94      }
95      
96      private Map<String, ReadwriteSplittingDataSourceGroupRule> createStaticDataSourceGroupRules(final String databaseName, final ReadwriteSplittingDataSourceGroupRuleConfiguration config,
97                                                                                                  final LoadBalanceAlgorithm loadBalanceAlgorithm) {
98          List<String> inlineLogicDataSourceNames = InlineExpressionParserFactory.newInstance(config.getName()).splitAndEvaluate();
99          List<String> inlineWriteDataSourceNames = InlineExpressionParserFactory.newInstance(config.getWriteDataSourceName()).splitAndEvaluate();
100         List<List<String>> inlineReadDataSourceNames = config.getReadDataSourceNames().stream()
101                 .map(each -> InlineExpressionParserFactory.newInstance(each).splitAndEvaluate()).collect(Collectors.toList());
102         ShardingSpherePreconditions.checkState(inlineWriteDataSourceNames.size() == inlineLogicDataSourceNames.size(),
103                 () -> new InvalidReadwriteSplittingActualDataSourceInlineExpressionException(
104                         ReadwriteSplittingDataSourceType.WRITE, new ReadwriteSplittingRuleExceptionIdentifier(databaseName, config.getName())));
105         inlineReadDataSourceNames.forEach(each -> ShardingSpherePreconditions.checkState(each.size() == inlineLogicDataSourceNames.size(),
106                 () -> new InvalidReadwriteSplittingActualDataSourceInlineExpressionException(
107                         ReadwriteSplittingDataSourceType.READ, new ReadwriteSplittingRuleExceptionIdentifier(databaseName, config.getName()))));
108         Map<String, ReadwriteSplittingDataSourceGroupRule> result = new HashMap<>(inlineLogicDataSourceNames.size(), 1F);
109         for (int i = 0; i < inlineLogicDataSourceNames.size(); i++) {
110             ReadwriteSplittingDataSourceGroupRuleConfiguration staticConfig = createStaticDataSourceGroupRuleConfiguration(
111                     config, i, inlineLogicDataSourceNames, inlineWriteDataSourceNames, inlineReadDataSourceNames);
112             result.put(inlineLogicDataSourceNames.get(i), new ReadwriteSplittingDataSourceGroupRule(staticConfig, config.getTransactionalReadQueryStrategy(), loadBalanceAlgorithm));
113         }
114         return result;
115     }
116     
117     private ReadwriteSplittingDataSourceGroupRuleConfiguration createStaticDataSourceGroupRuleConfiguration(final ReadwriteSplittingDataSourceGroupRuleConfiguration config, final int index,
118                                                                                                             final List<String> logicDataSourceNames, final List<String> writeDatasourceNames,
119                                                                                                             final List<List<String>> readDatasourceNames) {
120         List<String> readDataSourceNames = readDatasourceNames.stream().map(each -> each.get(index)).collect(Collectors.toList());
121         return new ReadwriteSplittingDataSourceGroupRuleConfiguration(logicDataSourceNames.get(index), writeDatasourceNames.get(index), readDataSourceNames, config.getLoadBalancerName());
122     }
123     
124     /**
125      * Get single data source group rule.
126      *
127      * @return readwrite-splitting data source group rule
128      */
129     public ReadwriteSplittingDataSourceGroupRule getSingleDataSourceGroupRule() {
130         return dataSourceRuleGroups.values().iterator().next();
131     }
132     
133     /**
134      * Find data source group rule.
135      *
136      * @param dataSourceName data source name
137      * @return readwrite-splitting data source group rule
138      */
139     public Optional<ReadwriteSplittingDataSourceGroupRule> findDataSourceGroupRule(final String dataSourceName) {
140         return Optional.ofNullable(dataSourceRuleGroups.get(dataSourceName));
141     }
142 }
