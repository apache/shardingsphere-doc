56  /**
57   * TCL Statement SQL visitor for MySQL.
58   */
59  public final class MySQLTCLStatementSQLVisitor extends MySQLStatementSQLVisitor implements TCLSQLVisitor {
60      
61      @Override
62      public ASTNode visitSetTransaction(final SetTransactionContext ctx) {
63          MySQLSetTransactionStatement result = new MySQLSetTransactionStatement();
64          if (null != ctx.optionType()) {
65              OperationScope scope = null;
66              if (null != ctx.optionType().SESSION()) {
67                  scope = OperationScope.SESSION;
68              } else if (null != ctx.optionType().GLOBAL()) {
69                  scope = OperationScope.GLOBAL;
70              }
71              result.setScope(scope);
72          }
73          if (null != ctx.transactionCharacteristics().isolationLevel()) {
74              TransactionIsolationLevel isolationLevel = null;
75              if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().SERIALIZABLE()) {
76                  isolationLevel = TransactionIsolationLevel.SERIALIZABLE;
77              } else if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().COMMITTED()) {
78                  isolationLevel = TransactionIsolationLevel.READ_COMMITTED;
79              } else if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().UNCOMMITTED()) {
80                  isolationLevel = TransactionIsolationLevel.READ_UNCOMMITTED;
81              } else if (null != ctx.transactionCharacteristics().isolationLevel().isolationTypes().REPEATABLE()) {
82                  isolationLevel = TransactionIsolationLevel.REPEATABLE_READ;
83              }
84              result.setIsolationLevel(isolationLevel);
85          }
86          if (null != ctx.transactionCharacteristics().transactionAccessMode()) {
87              TransactionAccessType accessType = null;
88              if (null != ctx.transactionCharacteristics().transactionAccessMode().ONLY()) {
89                  accessType = TransactionAccessType.READ_ONLY;
90              } else if (null != ctx.transactionCharacteristics().transactionAccessMode().WRITE()) {
91                  accessType = TransactionAccessType.READ_WRITE;
92              }
93              result.setAccessMode(accessType);
94          }
95          return result;
96      }
97      
98      @Override
99      public ASTNode visitSetAutoCommit(final SetAutoCommitContext ctx) {
100         MySQLSetAutoCommitStatement result = new MySQLSetAutoCommitStatement();
101         result.setAutoCommit(generateAutoCommitSegment(ctx.autoCommitValue).isAutoCommit());
102         return result;
103     }
104     
105     private AutoCommitSegment generateAutoCommitSegment(final Token ctx) {
106         boolean autoCommit = "1".equals(ctx.getText()) || "ON".equals(ctx.getText());
107         return new AutoCommitSegment(ctx.getStartIndex(), ctx.getStopIndex(), autoCommit);
108     }
109     
110     @Override
111     public ASTNode visitBeginTransaction(final BeginTransactionContext ctx) {
112         return new MySQLBeginTransactionStatement();
113     }
114     
115     @Override
116     public ASTNode visitCommit(final CommitContext ctx) {
117         return new MySQLCommitStatement();
118     }
119     
120     @Override
121     public ASTNode visitRollback(final RollbackContext ctx) {
122         MySQLRollbackStatement result = new MySQLRollbackStatement();
123         if (null != ctx.identifier()) {
124             result.setSavepointName(((IdentifierValue) visit(ctx.identifier())).getValue());
125         }
126         return result;
127     }
128     
129     @Override
130     public ASTNode visitSavepoint(final SavepointContext ctx) {
131         MySQLSavepointStatement result = new MySQLSavepointStatement();
132         result.setSavepointName(((IdentifierValue) visit(ctx.identifier())).getValue());
133         return result;
134     }
135     
136     @Override
137     public ASTNode visitReleaseSavepoint(final ReleaseSavepointContext ctx) {
138         MySQLReleaseSavepointStatement result = new MySQLReleaseSavepointStatement();
139         result.setSavepointName(((IdentifierValue) visit(ctx.identifier())).getValue());
140         return result;
141     }
142     
143     @Override
144     public ASTNode visitXa(final XaContext ctx) {
145         MySQLXAStatement result = new MySQLXAStatement();
146         result.setOp(ctx.getChild(1).getText().toUpperCase());
147         if (null != ctx.xid()) {
148             result.setXid(ctx.xid().getText());
149         }
150         return result;
151     }
152     
153     @Override
154     public ASTNode visitLock(final LockContext ctx) {
155         MySQLLockStatement result = new MySQLLockStatement();
156         if (null != ctx.tableLock()) {
157             result.getTables().addAll(getLockTables(ctx.tableLock()));
158         }
159         return result;
160     }
161     
162     private Collection<SimpleTableSegment> getLockTables(final List<TableLockContext> tableLockContexts) {
163         Collection<SimpleTableSegment> result = new LinkedList<>();
164         for (TableLockContext each : tableLockContexts) {
165             SimpleTableSegment simpleTableSegment = (SimpleTableSegment) visit(each.tableName());
166             if (null != each.alias()) {
167                 simpleTableSegment.setAlias((AliasSegment) visit(each.alias()));
168             }
169             result.add(simpleTableSegment);
170         }
171         return result;
172     }
173     
174     @Override
175     public ASTNode visitUnlock(final UnlockContext ctx) {
176         return new MySQLUnlockStatement();
177     }
