18  package org.apache.shardingsphere.globalclock.type.tso.provider;
19  
20  import com.google.common.base.Strings;
21  import redis.clients.jedis.Jedis;
22  import redis.clients.jedis.JedisPool;
23  import redis.clients.jedis.JedisPoolConfig;
24  
25  import java.util.Properties;
26  import java.util.concurrent.atomic.AtomicBoolean;
27  
28  /**
29   * Redis timestamp oracle provider.
30   */
31  public final class RedisTSOProvider implements TSOProvider {
32      
33      private static final String CSN_KEY = "csn";
34      
35      private static final long ERROR_CSN = 0L;
36      
37      private static final long INIT_CSN = Integer.MAX_VALUE;
38      
39      private final AtomicBoolean initialized = new AtomicBoolean(false);
40      
41      private JedisPool jedisPool;
42      
43      private Properties props;
44      
45      @Override
46      public void init(final Properties props) {
47          this.props = props;
48          if (initialized.compareAndSet(false, true)) {
49              createJedisPool();
50              checkJedisPool();
51              initCSN();
52          }
53      }
54      
55      private void createJedisPool() {
56          JedisPoolConfig config = new JedisPoolConfig();
57          config.setMaxIdle(Integer.parseInt(getValue(props, RedisTSOPropertyKey.MAX_IDLE)));
58          config.setMaxTotal(Integer.parseInt(getValue(props, RedisTSOPropertyKey.MAX_TOTAL)));
59          jedisPool = new JedisPool(config, getValue(props, RedisTSOPropertyKey.HOST),
60                  Integer.parseInt(getValue(props, RedisTSOPropertyKey.PORT)), Integer.parseInt(getValue(props, RedisTSOPropertyKey.TIMEOUT_INTERVAL)),
61                  getValue(props, RedisTSOPropertyKey.PASSWORD));
62      }
63      
64      private void checkJedisPool() {
65          try (Jedis jedis = jedisPool.getResource()) {
66              jedis.ping();
67          }
68      }
69      
70      private void initCSN() {
71          try (Jedis jedis = jedisPool.getResource()) {
72              String originalCSN = jedis.get(CSN_KEY);
73              if (Strings.isNullOrEmpty(originalCSN) || String.valueOf(ERROR_CSN).equals(originalCSN)) {
74                  jedis.set(CSN_KEY, String.valueOf(INIT_CSN));
75              }
76          }
77      }
78      
79      private String getValue(final Properties props, final RedisTSOPropertyKey propertyKey) {
80          return props.containsKey(propertyKey.getKey()) ? props.getProperty(propertyKey.getKey()) : propertyKey.getDefaultValue();
81      }
82      
83      @Override
84      public long getCurrentTimestamp() {
85          try (Jedis jedis = jedisPool.getResource()) {
86              return Long.parseLong(jedis.get(CSN_KEY));
87          }
88      }
89      
90      @Override
91      public long getNextTimestamp() {
92          try (Jedis jedis = jedisPool.getResource()) {
93              return jedis.incr(CSN_KEY);
94          }
95      }
96      
97      @Override
98      public String getType() {
99          return "TSO.redis";
100     }
101 }
