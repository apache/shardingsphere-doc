50  /**
51   * SQL token generator builder for encrypt.
52   */
53  @RequiredArgsConstructor
54  public final class EncryptTokenGenerateBuilder implements SQLTokenGeneratorBuilder {
55      
56      private final EncryptRule encryptRule;
57      
58      private final SQLStatementContext sqlStatementContext;
59      
60      private final Collection<EncryptCondition> encryptConditions;
61      
62      private final String databaseName;
63      
64      @Override
65      public Collection<SQLTokenGenerator> getSQLTokenGenerators() {
66          Collection<SQLTokenGenerator> result = new LinkedList<>();
67          addSQLTokenGenerator(result, new EncryptSelectProjectionTokenGenerator(encryptRule));
68          addSQLTokenGenerator(result, new EncryptInsertSelectProjectionTokenGenerator(encryptRule));
69          addSQLTokenGenerator(result, new EncryptInsertAssignmentTokenGenerator(encryptRule));
70          addSQLTokenGenerator(result, new EncryptUpdateAssignmentTokenGenerator(encryptRule));
71          addSQLTokenGenerator(result, new EncryptPredicateColumnTokenGenerator(encryptRule));
72          addSQLTokenGenerator(result, new EncryptInsertPredicateColumnTokenGenerator(encryptRule));
73          addSQLTokenGenerator(result, new EncryptPredicateRightValueTokenGenerator(encryptRule));
74          addSQLTokenGenerator(result, new EncryptInsertPredicateRightValueTokenGenerator(encryptRule));
75          addSQLTokenGenerator(result, new EncryptInsertValuesTokenGenerator(encryptRule));
76          addSQLTokenGenerator(result, new EncryptInsertDefaultColumnsTokenGenerator(encryptRule));
77          addSQLTokenGenerator(result, new EncryptInsertCipherNameTokenGenerator(encryptRule));
78          addSQLTokenGenerator(result, new EncryptInsertDerivedColumnsTokenGenerator(encryptRule));
79          addSQLTokenGenerator(result, new EncryptInsertOnUpdateTokenGenerator(encryptRule));
80          addSQLTokenGenerator(result, new EncryptGroupByItemTokenGenerator(encryptRule));
81          addSQLTokenGenerator(result, new EncryptIndexColumnTokenGenerator(encryptRule));
82          addSQLTokenGenerator(result, new EncryptCreateTableTokenGenerator(encryptRule));
83          addSQLTokenGenerator(result, new EncryptAlterTableTokenGenerator(encryptRule));
84          return result;
85      }
86      
87      private void addSQLTokenGenerator(final Collection<SQLTokenGenerator> sqlTokenGenerators, final SQLTokenGenerator toBeAddedSQLTokenGenerator) {
88          setUpSQLTokenGenerator(toBeAddedSQLTokenGenerator);
89          if (toBeAddedSQLTokenGenerator.isGenerateSQLToken(sqlStatementContext)) {
90              sqlTokenGenerators.add(toBeAddedSQLTokenGenerator);
91          }
92      }
93      
94      private void setUpSQLTokenGenerator(final SQLTokenGenerator toBeAddedSQLTokenGenerator) {
95          if (toBeAddedSQLTokenGenerator instanceof EncryptConditionsAware) {
96              ((EncryptConditionsAware) toBeAddedSQLTokenGenerator).setEncryptConditions(encryptConditions);
97          }
98          if (toBeAddedSQLTokenGenerator instanceof DatabaseNameAware) {
99              ((DatabaseNameAware) toBeAddedSQLTokenGenerator).setDatabaseName(databaseName);
100         }
101         if (toBeAddedSQLTokenGenerator instanceof DatabaseTypeAware) {
102             ((DatabaseTypeAware) toBeAddedSQLTokenGenerator).setDatabaseType(sqlStatementContext.getDatabaseType());
103         }
104     }
105 }
