18  package org.apache.shardingsphere.test.e2e.framework.database.impl;
19  
20  import org.apache.shardingsphere.test.e2e.framework.database.DatabaseAssertionMetaData;
21  
22  import javax.sql.DataSource;
23  import java.sql.Connection;
24  import java.sql.ResultSet;
25  import java.sql.SQLException;
26  import java.sql.Statement;
27  
28  /**
29   * PostgreSQL assertion meta data.
30   */
31  public final class PostgreSQLDatabaseAssertionMetaData implements DatabaseAssertionMetaData {
32      
33      @Override
34      public String getPrimaryKeyColumnName(final DataSource dataSource, final String tableName) throws SQLException {
35          String sql = String.format("SELECT a.attname, format_type(a.atttypid, a.atttypmod) AS data_type "
36                  + "FROM pg_index i JOIN pg_attribute a ON a.attrelid = i.indrelid AND a.attnum = ANY(i.indkey) WHERE i.indrelid = '%s'::regclass AND i.indisprimary", tableName);
37          try (
38                  Connection connection = dataSource.getConnection();
39                  Statement statement = connection.createStatement();
40                  ResultSet resultSet = statement.executeQuery(sql)) {
41              if (resultSet.next()) {
42                  return resultSet.getString("attname");
43              }
44              throw new SQLException(String.format("Can not get primary key of `%s`", tableName));
45          }
46      }
47  }
