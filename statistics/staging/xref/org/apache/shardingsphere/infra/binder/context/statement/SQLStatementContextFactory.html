118 /**
119  * SQL statement context factory.
120  */
121 @NoArgsConstructor(access = AccessLevel.PRIVATE)
122 public final class SQLStatementContextFactory {
123     
124     /**
125      * Create SQL statement context.
126      *
127      * @param metaData metadata
128      * @param sqlStatement SQL statement
129      * @param params SQL parameters
130      * @param currentDatabaseName current database name
131      * @return SQL statement context
132      */
133     public static SQLStatementContext newInstance(final ShardingSphereMetaData metaData, final SQLStatement sqlStatement, final List<Object> params, final String currentDatabaseName) {
134         if (sqlStatement instanceof DMLStatement) {
135             return getDMLStatementContext(metaData, (DMLStatement) sqlStatement, params, currentDatabaseName);
136         }
137         if (sqlStatement instanceof DDLStatement) {
138             return getDDLStatementContext(metaData, (DDLStatement) sqlStatement, params, currentDatabaseName);
139         }
140         if (sqlStatement instanceof DCLStatement) {
141             return getDCLStatementContext((DCLStatement) sqlStatement, currentDatabaseName);
142         }
143         if (sqlStatement instanceof DALStatement) {
144             return getDALStatementContext((DALStatement) sqlStatement, currentDatabaseName);
145         }
146         return new UnknownSQLStatementContext(sqlStatement);
147     }
148     
149     private static SQLStatementContext getDMLStatementContext(final ShardingSphereMetaData metaData, final DMLStatement sqlStatement, final List<Object> params, final String currentDatabaseName) {
150         if (sqlStatement instanceof SelectStatement) {
151             return new SelectStatementContext(metaData, params, (SelectStatement) sqlStatement, currentDatabaseName, Collections.emptyList());
152         }
153         if (sqlStatement instanceof UpdateStatement) {
154             return new UpdateStatementContext((UpdateStatement) sqlStatement, currentDatabaseName);
155         }
156         if (sqlStatement instanceof DeleteStatement) {
157             return new DeleteStatementContext((DeleteStatement) sqlStatement, currentDatabaseName);
158         }
159         if (sqlStatement instanceof InsertStatement) {
160             return new InsertStatementContext(metaData, params, (InsertStatement) sqlStatement, currentDatabaseName);
161         }
162         if (sqlStatement instanceof CallStatement) {
163             return new CallStatementContext((CallStatement) sqlStatement);
164         }
165         if (sqlStatement instanceof CopyStatement) {
166             return new CopyStatementContext((CopyStatement) sqlStatement, currentDatabaseName);
167         }
168         if (sqlStatement instanceof DoStatement) {
169             return new DoStatementContext((DoStatement) sqlStatement);
170         }
171         if (sqlStatement instanceof LoadDataStatement) {
172             return new LoadDataStatementContext((LoadDataStatement) sqlStatement, currentDatabaseName);
173         }
174         if (sqlStatement instanceof LoadXMLStatement) {
175             return new LoadXMLStatementContext((LoadXMLStatement) sqlStatement, currentDatabaseName);
176         }
177         if (sqlStatement instanceof MergeStatement) {
178             return new UnknownSQLStatementContext(sqlStatement);
179         }
180         throw new UnsupportedSQLOperationException(String.format("Unsupported SQL statement `%s`", sqlStatement.getClass().getSimpleName()));
181     }
182     
183     private static SQLStatementContext getDDLStatementContext(final ShardingSphereMetaData metaData, final DDLStatement sqlStatement, final List<Object> params, final String currentDatabaseName) {
184         if (sqlStatement instanceof CreateSchemaStatement) {
185             return new CreateSchemaStatementContext((CreateSchemaStatement) sqlStatement);
186         }
187         if (sqlStatement instanceof CreateTableStatement) {
188             return new CreateTableStatementContext((CreateTableStatement) sqlStatement, currentDatabaseName);
189         }
190         if (sqlStatement instanceof AlterTableStatement) {
191             return new AlterTableStatementContext((AlterTableStatement) sqlStatement, currentDatabaseName);
192         }
193         if (sqlStatement instanceof RenameTableStatement) {
194             return new RenameTableStatementContext((RenameTableStatement) sqlStatement, currentDatabaseName);
195         }
196         if (sqlStatement instanceof DropTableStatement) {
197             return new DropTableStatementContext((DropTableStatement) sqlStatement, currentDatabaseName);
198         }
199         if (sqlStatement instanceof CreateIndexStatement) {
200             return new CreateIndexStatementContext((CreateIndexStatement) sqlStatement, currentDatabaseName);
201         }
202         if (sqlStatement instanceof AlterIndexStatement) {
203             return new AlterIndexStatementContext((AlterIndexStatement) sqlStatement, currentDatabaseName);
204         }
205         if (sqlStatement instanceof DropIndexStatement) {
206             return new DropIndexStatementContext((DropIndexStatement) sqlStatement, currentDatabaseName);
207         }
208         if (sqlStatement instanceof TruncateStatement) {
209             return new TruncateStatementContext((TruncateStatement) sqlStatement, currentDatabaseName);
210         }
211         if (sqlStatement instanceof CreateFunctionStatement) {
212             return new CreateFunctionStatementContext((CreateFunctionStatement) sqlStatement, currentDatabaseName);
213         }
214         if (sqlStatement instanceof CreateProcedureStatement) {
215             return new CreateProcedureStatementContext((CreateProcedureStatement) sqlStatement, currentDatabaseName);
216         }
217         if (sqlStatement instanceof CreateViewStatement) {
218             return new CreateViewStatementContext((CreateViewStatement) sqlStatement, currentDatabaseName);
219         }
220         if (sqlStatement instanceof AlterViewStatement) {
221             return new AlterViewStatementContext((AlterViewStatement) sqlStatement, currentDatabaseName);
222         }
223         if (sqlStatement instanceof DropViewStatement) {
224             return new DropViewStatementContext((DropViewStatement) sqlStatement, currentDatabaseName);
225         }
226         if (sqlStatement instanceof PrepareStatement) {
227             return new PrepareStatementContext((PrepareStatement) sqlStatement, currentDatabaseName);
228         }
229         if (sqlStatement instanceof CommentStatement) {
230             return new CommentStatementContext((CommentStatement) sqlStatement, currentDatabaseName);
231         }
232         if (sqlStatement instanceof CursorStatement) {
233             return new CursorStatementContext(metaData, params, (CursorStatement) sqlStatement, currentDatabaseName);
234         }
235         if (sqlStatement instanceof CloseStatement) {
236             return new CloseStatementContext((CloseStatement) sqlStatement, currentDatabaseName);
237         }
238         if (sqlStatement instanceof MoveStatement) {
239             return new MoveStatementContext((MoveStatement) sqlStatement, currentDatabaseName);
240         }
241         if (sqlStatement instanceof FetchStatement) {
242             return new FetchStatementContext((FetchStatement) sqlStatement, currentDatabaseName);
243         }
244         return new UnknownSQLStatementContext(sqlStatement);
245     }
246     
247     private static SQLStatementContext getDCLStatementContext(final DCLStatement sqlStatement, final String currentDatabaseName) {
248         if (sqlStatement instanceof GrantStatement) {
249             return new GrantStatementContext((GrantStatement) sqlStatement, currentDatabaseName);
250         }
251         if (sqlStatement instanceof RevokeStatement) {
252             return new RevokeStatementContext((RevokeStatement) sqlStatement, currentDatabaseName);
253         }
254         if (sqlStatement instanceof DenyUserStatement) {
255             return new DenyUserStatementContext((DenyUserStatement) sqlStatement, currentDatabaseName);
256         }
257         return new UnknownSQLStatementContext(sqlStatement);
258     }
259     
260     private static SQLStatementContext getDALStatementContext(final DALStatement sqlStatement, final String currentDatabaseName) {
261         if (sqlStatement instanceof ExplainStatement) {
262             return new ExplainStatementContext((ExplainStatement) sqlStatement, currentDatabaseName);
263         }
264         if (sqlStatement instanceof ShowCreateTableStatement) {
265             return new ShowCreateTableStatementContext((ShowCreateTableStatement) sqlStatement, currentDatabaseName);
266         }
267         if (sqlStatement instanceof ShowColumnsStatement) {
268             return new ShowColumnsStatementContext((ShowColumnsStatement) sqlStatement, currentDatabaseName);
269         }
270         if (sqlStatement instanceof ShowTablesStatement) {
271             return new ShowTablesStatementContext((ShowTablesStatement) sqlStatement);
272         }
273         if (sqlStatement instanceof ShowTableStatusStatement) {
274             return new ShowTableStatusStatementContext((ShowTableStatusStatement) sqlStatement);
275         }
276         if (sqlStatement instanceof ShowIndexStatement) {
277             return new ShowIndexStatementContext((ShowIndexStatement) sqlStatement, currentDatabaseName);
278         }
279         if (sqlStatement instanceof AnalyzeTableStatement) {
280             return new AnalyzeTableStatementContext((AnalyzeTableStatement) sqlStatement, currentDatabaseName);
281         }
282         if (sqlStatement instanceof FlushStatement) {
283             return new FlushStatementContext((FlushStatement) sqlStatement, currentDatabaseName);
284         }
285         if (sqlStatement instanceof OptimizeTableStatement) {
286             return new OptimizeTableStatementContext((OptimizeTableStatement) sqlStatement, currentDatabaseName);
287         }
288         if (sqlStatement instanceof KillStatement) {
289             return new KillStatementContext((KillStatement) sqlStatement);
290         }
291         return new UnknownSQLStatementContext(sqlStatement);
292     }
