80  /**
81   * SQL statement visitor for migration DistSQL.
82   */
83  public final class MigrationDistSQLStatementVisitor extends MigrationDistSQLStatementBaseVisitor<ASTNode> implements SQLVisitor<ASTNode> {
84      
85      @Override
86      public ASTNode visitMigrateTable(final MigrateTableContext ctx) {
87          SourceTargetEntry sourceTargetEntry = buildSourceTargetEntry(ctx.sourceTableName(), ctx.targetTableName());
88          return new MigrateTableStatement(Collections.singletonList(sourceTargetEntry), sourceTargetEntry.getTargetDatabaseName());
89      }
90      
91      private SourceTargetEntry buildSourceTargetEntry(final SourceTableNameContext sourceContext, final TargetTableNameContext targetContext) {
92          List<String> source = Splitter.on('.').splitToList(getRequiredIdentifierValue(sourceContext));
93          List<String> target = Splitter.on('.').splitToList(getRequiredIdentifierValue(targetContext));
94          String sourceResourceName = source.get(0);
95          String sourceSchemaName = 3 == source.size() ? source.get(1) : null;
96          String sourceTableName = source.get(source.size() - 1);
97          String targetDatabaseName = target.size() > 1 ? target.get(0) : null;
98          String targetTableName = target.get(target.size() - 1);
99          SourceTargetEntry result = new SourceTargetEntry(targetDatabaseName, new DataNode(sourceResourceName, sourceTableName), targetTableName);
100         result.getSource().setSchemaName(sourceSchemaName);
101         return result;
102     }
103     
104     private String getRequiredIdentifierValue(final ParseTree context) {
105         return new IdentifierValue(context.getText()).getValue();
106     }

138     @Override
139     public ASTNode visitCheckMigration(final CheckMigrationContext ctx) {

141     }
142     
143     @Override
144     public ASTNode visitShowMigrationCheckAlgorithms(final ShowMigrationCheckAlgorithmsContext ctx) {
145         return new ShowMigrationCheckAlgorithmsStatement();
146     }
147     
148     @Override
149     public ASTNode visitAlgorithmDefinition(final AlgorithmDefinitionContext ctx) {
150         return new AlgorithmSegment(getIdentifierValue(ctx.algorithmTypeName()), getProperties(ctx.propertiesDefinition()));
151     }
152     
153     private String getIdentifierValue(final ParseTree context) {
154         return new IdentifierValue(context.getText()).getValue();
155     }
156     
157     @Override
158     public ASTNode visitStorageUnitDefinition(final MigrationDistSQLStatementParser.StorageUnitDefinitionContext ctx) {
159         String user = getIdentifierValue(ctx.user());
160         String password = null == ctx.password() ? "" : getPassword(ctx.password());
161         Properties props = getProperties(ctx.propertiesDefinition());
162         return null == ctx.urlSource()
163                 ? new HostnameAndPortBasedDataSourceSegment(getIdentifierValue(ctx.storageUnitName()),
164                         getIdentifierValue(ctx.simpleSource().hostname()), ctx.simpleSource().port().getText(), getIdentifierValue(ctx.simpleSource().dbName()), user, password, props)
165                 : new URLBasedDataSourceSegment(getIdentifierValue(ctx.storageUnitName()), getIdentifierValue(ctx.urlSource().url()), user, password, props);
166     }
167     
168     private String getPassword(final PasswordContext ctx) {
169         return getIdentifierValue(ctx);
170     }
171     
172     private Properties getProperties(final PropertiesDefinitionContext ctx) {
173         Properties result = new Properties();
174         if (null == ctx || null == ctx.properties()) {
175             return result;
176         }
177         for (PropertyContext each : ctx.properties().property()) {
178             result.setProperty(IdentifierValue.getQuotedContent(each.key.getText()), IdentifierValue.getQuotedContent(each.value.getText()));
179         }
180         return result;
181     }
182     
183     @Override
184     public ASTNode visitRegisterMigrationSourceStorageUnit(final RegisterMigrationSourceStorageUnitContext ctx) {
185         Collection<DataSourceSegment> dataSources = new LinkedList<>();
186         for (StorageUnitDefinitionContext each : ctx.storageUnitDefinition()) {
187             dataSources.add((DataSourceSegment) visit(each));
188         }
189         return new RegisterMigrationSourceStorageUnitStatement(dataSources);
190     }
191     
192     @Override
193     public ASTNode visitUnregisterMigrationSourceStorageUnit(final UnregisterMigrationSourceStorageUnitContext ctx) {
194         return new UnregisterMigrationSourceStorageUnitStatement(ctx.storageUnitName().stream().map(ParseTree::getText).map(each -> new IdentifierValue(each).getValue()).collect(Collectors.toList()));
195     }
196

198     public ASTNode visitShowMigrationSourceStorageUnits(final ShowMigrationSourceStorageUnitsContext ctx) {
199         return new ShowMigrationSourceStorageUnitsStatement();
200     }

205     }

215     }

220     }
