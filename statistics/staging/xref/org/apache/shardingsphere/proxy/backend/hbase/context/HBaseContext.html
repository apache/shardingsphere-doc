48  /**
49   * HBase context.
50   */
51  @NoArgsConstructor(access = AccessLevel.PRIVATE)
52  @Getter
53  @Slf4j
54  public final class HBaseContext implements AutoCloseable {
55      
56      private static final HBaseContext INSTANCE = new HBaseContext();
57      
58      private final HBaseBackgroundExecutorManager executorManager = new HBaseBackgroundExecutorManager();
59      
60      private HBaseRegionWarmUpContext warmUpContext;
61      
62      @Setter
63      private HBaseProperties props;
64      
65      private final String columnFamily = "i";
66      
67      private Collection<HBaseCluster> connections;
68      
69      private boolean isSyncWarmUp;
70      
71      private final Map<String, HBaseCluster> tableConnectionMap = new ConcurrentHashMap<>();
72      
73      /**
74       * Get instance of HBase context.
75       *
76       * @return got instance
77       */
78      public static HBaseContext getInstance() {
79          return INSTANCE;
80      }
81      
82      /**
83       * Initialize HBase context.
84       * 
85       * @param connections A connection for per HBase cluster
86       * @throws SQLException SQL exception
87       */
88      public void init(final Map<String, Connection> connections) throws SQLException {
89          this.connections = new ArrayList<>(connections.size());
90          warmUpContext = HBaseRegionWarmUpContext.getInstance();
91          warmUpContext.init(getWarmUpThreadSize());
92          isSyncWarmUp = HBaseContext.getInstance().getProps().<Boolean>getValue(HBasePropertyKey.IS_SYNC_WARM_UP);
93          for (Entry<String, Connection> entry : connections.entrySet()) {
94              HBaseCluster cluster = new HBaseCluster(entry.getKey(), entry.getValue());
95              loadTables(cluster);
96              this.connections.add(cluster);
97          }
98          log.info("{} tables loaded from {} clusters.", tableConnectionMap.size(), connections.size());
99      }
100     
101     /**
102      * Get warm up thread size.
103      * 
104      * @return warm up thread size, value should be in (0, 30]
105      */
106     private int getWarmUpThreadSize() {
107         int warmUpThreadSize = HBaseContext.getInstance().getProps().<Integer>getValue(HBasePropertyKey.WARM_UP_THREAD_NUM);
108         return warmUpThreadSize < 0 ? 1 : Math.min(warmUpThreadSize, 30);
109     }
110     
111     /**
112      * Load tables.
113      * 
114      * @param hbaseCluster HBase cluster
115      * @throws SQLException SQL exception
116      */
117     public synchronized void loadTables(final HBaseCluster hbaseCluster) throws SQLException {
118         warmUpContext.initStatisticsInfo(System.currentTimeMillis());
119         HTableDescriptor[] hTableDescriptor = HBaseExecutor.executeAdmin(hbaseCluster.getConnection(), Admin::listTables);
120         for (String each : Arrays.stream(hTableDescriptor).map(HTableDescriptor::getNameAsString).collect(Collectors.toList())) {
121             if (tableConnectionMap.containsKey(each)) {
122                 continue;
123             }
124             warmUpContext.addNeedWarmCount();
125             log.info("Load table `{}` from cluster `{}`.", each, hbaseCluster.getClusterName());
126             tableConnectionMap.put(each, hbaseCluster);
127             warmUpContext.submitWarmUpTask(each, hbaseCluster);
128         }
129         if (isSyncWarmUp) {
130             warmUpContext.syncExecuteWarmUp(hbaseCluster.getClusterName());
131             warmUpContext.clear();
132         }
133     }
134     
135     /**
136      * Get connection via table name.
137      * 
138      * @param tableName table name
139      * @return HBase connection
140      */
141     public Connection getConnection(final String tableName) {
142         ShardingSpherePreconditions.checkState(tableConnectionMap.containsKey(tableName), () -> new TableNotFoundException(tableName));
143         return tableConnectionMap.get(tableName).getConnection();
144     }
145     
146     /**
147      * Is table exists.
148      * 
149      * @param tableName table name
150      * @return table exists or not
151      */
152     public boolean isTableExists(final String tableName) {
153         return tableConnectionMap.containsKey(tableName);
154     }
155     
156     /**
157      * Get connection via cluster name.
158      * 
159      * @param clusterName cluster name
160      * @return HBase connection
161      */
162     public Connection getConnectionByClusterName(final String clusterName) {
163         Optional<HBaseCluster> cluster = connections.stream().filter(each -> each.getClusterName().equalsIgnoreCase(clusterName)).findFirst();
164         ShardingSpherePreconditions.checkState(cluster.isPresent(), () -> new HBaseOperationException(String.format("Cluster `%s` is not exists", clusterName)));
165         return cluster.get().getConnection();
166     }
167     
168     @Override
169     public void close() throws SQLException {
170         connections.clear();
171         tableConnectionMap.clear();
172         executorManager.close();
173         for (Connection connection : connections.stream().map(HBaseCluster::getConnection).collect(Collectors.toList())) {
174             try {
175                 connection.close();
176             } catch (final IOException ex) {
177                 throw new SQLException(ex);
178             }
179         }
180     }
181 }
