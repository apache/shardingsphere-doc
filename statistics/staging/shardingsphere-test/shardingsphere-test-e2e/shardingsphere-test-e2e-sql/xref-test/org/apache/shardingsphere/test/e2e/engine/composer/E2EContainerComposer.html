21  import lombok.Getter;
22  import lombok.SneakyThrows;

24  import org.apache.shardingsphere.test.e2e.container.compose.ContainerComposer;

26  import org.apache.shardingsphere.test.e2e.engine.TotalSuitesCountCalculator;
27  import org.apache.shardingsphere.test.e2e.env.runtime.scenario.path.ScenarioDataPath;
28  import org.apache.shardingsphere.test.e2e.framework.param.model.E2ETestParameter;
29  import org.h2.tools.RunScript;

31  import javax.sql.DataSource;
32  import java.io.FileReader;
33  import java.io.IOException;
34  import java.sql.Connection;
35  import java.sql.SQLException;
36  import java.util.Collection;
37  import java.util.HashSet;
38  import java.util.Map;
39  import java.util.Optional;
40  import java.util.concurrent.atomic.AtomicInteger;

42  /**
43   * E2E container composer.
44   */
45  @Getter
46  public abstract class E2EContainerComposer implements AutoCloseable {
47      
48      public static final String NOT_VERIFY_FLAG = "NOT_VERIFY";
49      
50      private static final ContainerComposerRegistry CONTAINER_COMPOSER_REGISTRY = new ContainerComposerRegistry();
51      
52      private static final int TOTAL_SUITES_COUNT = TotalSuitesCountCalculator.calculate();
53      
54      private static final AtomicInteger COMPLETED_SUITES_COUNT = new AtomicInteger(0);
55      
56      private static final Collection<String> INITIALIZED_SUITES = new HashSet<>();
57      
58      @Getter(AccessLevel.NONE)
59      private final ContainerComposer containerComposer;
60      
61      private final Map<String, DataSource> actualDataSourceMap;
62      
63      private final DataSource targetDataSource;
64      
65      private final Map<String, DataSource> expectedDataSourceMap;
66      
67      public E2EContainerComposer(final E2ETestParameter testParam) {
68          containerComposer = CONTAINER_COMPOSER_REGISTRY.getContainerComposer(testParam);
69          containerComposer.start();
70          actualDataSourceMap = containerComposer.getActualDataSourceMap();
71          targetDataSource = containerComposer.getTargetDataSource();
72          expectedDataSourceMap = containerComposer.getExpectedDataSourceMap();
73          executeLogicDatabaseInitSQLFileOnlyOnce(testParam, targetDataSource);
74      }
75      
76      @SneakyThrows({SQLException.class, IOException.class})
77      private void executeLogicDatabaseInitSQLFileOnlyOnce(final E2ETestParameter testParam, final DataSource targetDataSource) {
78          Optional<String> logicDatabaseInitSQLFile = new ScenarioDataPath(testParam.getScenario()).findActualDatabaseInitSQLFile(DefaultDatabase.LOGIC_NAME, testParam.getDatabaseType());
79          if (!logicDatabaseInitSQLFile.isPresent()) {
80              return;
81          }
82          String cacheKey = testParam.getKey() + "-" + System.identityHashCode(targetDataSource);
83          if (!INITIALIZED_SUITES.contains(cacheKey)) {
84              synchronized (INITIALIZED_SUITES) {
85                  if (!INITIALIZED_SUITES.contains(cacheKey)) {
86                      executeInitSQL(targetDataSource, logicDatabaseInitSQLFile.get());
87                      INITIALIZED_SUITES.add(cacheKey);
88                  }
89              }
90          }
91      }
92      
93      private void executeInitSQL(final DataSource dataSource, final String logicDatabaseInitSQLFile) throws SQLException, IOException {
94          try (
95                  Connection connection = dataSource.getConnection();
96                  FileReader reader = new FileReader(logicDatabaseInitSQLFile)) {
97              RunScript.execute(connection, reader);
98          }
99      }
100     
101     @Override
102     public void close() {
103         if (COMPLETED_SUITES_COUNT.incrementAndGet() == TOTAL_SUITES_COUNT) {
104             CONTAINER_COMPOSER_REGISTRY.close();
105         }
106     }
107 }
