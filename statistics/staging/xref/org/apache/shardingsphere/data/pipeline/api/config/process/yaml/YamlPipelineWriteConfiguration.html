18  package org.apache.shardingsphere.data.pipeline.api.config.process.yaml;
19  
20  import lombok.Getter;
21  import lombok.Setter;
22  import org.apache.shardingsphere.infra.util.yaml.YamlConfiguration;
23  import org.apache.shardingsphere.infra.yaml.config.pojo.algorithm.YamlAlgorithmConfiguration;
24  
25  /**
26   * YAML pipeline write configuration.
27   */
28  @Getter
29  @Setter
30  public final class YamlPipelineWriteConfiguration implements YamlConfiguration {
31      
32      private static final Integer DEFAULT_WORKER_THREAD = 40;
33      
34      private static final Integer DEFAULT_BATCH_SIZE = 1000;
35      
36      private Integer workerThread;
37      
38      private Integer batchSize;
39      
40      private YamlAlgorithmConfiguration rateLimiter;
41      
42      /**
43       * Build with default value.
44       *
45       * @return write configuration
46       */
47      public static YamlPipelineWriteConfiguration buildWithDefaultValue() {
48          YamlPipelineWriteConfiguration result = new YamlPipelineWriteConfiguration();
49          result.workerThread = DEFAULT_WORKER_THREAD;
50          result.batchSize = DEFAULT_BATCH_SIZE;
51          return result;
52      }
53      
54      /**
55       * Fill in null fields with default value.
56       */
57      public void fillInNullFieldsWithDefaultValue() {
58          if (null == workerThread) {
59              workerThread = DEFAULT_WORKER_THREAD;
60          }
61          if (null == batchSize) {
62              batchSize = DEFAULT_BATCH_SIZE;
63          }
64      }
65      
66      /**
67       * Copy non-null fields from another.
68       *
69       * @param another another configuration
70       */
71      public void copyNonNullFields(final YamlPipelineWriteConfiguration another) {
72          if (null == another) {
73              return;
74          }
75          if (null != another.workerThread) {
76              workerThread = another.workerThread;
77          }
78          if (null != another.batchSize) {
79              batchSize = another.batchSize;
80          }
81          if (null != another.rateLimiter) {
82              rateLimiter = another.rateLimiter;
83          }
84      }
85      
86      /**
87       * Set field to null.
88       *
89       * @param nodeName node name
90       */
91      public void setFieldNull(final String nodeName) {
92          switch (nodeName.toUpperCase()) {
93              case "WORKER_THREAD":
94                  workerThread = null;
95                  break;
96              case "BATCH_SIZE":
97                  batchSize = null;
98                  break;
99              case "RATE_LIMITER":
100                 rateLimiter = null;
101                 break;
102             default:
103                 break;
104         }
105     }
106 }
