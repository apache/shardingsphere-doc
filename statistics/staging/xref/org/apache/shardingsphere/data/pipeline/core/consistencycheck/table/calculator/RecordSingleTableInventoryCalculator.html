47  /**
48   * Record single table inventory calculator.
49   */
50  @HighFrequencyInvocation
51  @RequiredArgsConstructor
52  public final class RecordSingleTableInventoryCalculator extends AbstractStreamingSingleTableInventoryCalculator {
53      
54      private final int chunkSize;
55      
56      @Override
57      public Optional<SingleTableInventoryCalculatedResult> calculateChunk(final SingleTableInventoryCalculateParameter param) {
58          CalculationContext calculationContext = getOrCreateCalculationContext(param);
59          try {
60              List<Map<String, Object>> records = new LinkedList<>();
61              Object maxUniqueKeyValue = null;
62              InventoryColumnValueReaderEngine columnValueReaderEngine = new InventoryColumnValueReaderEngine(param.getDatabaseType());
63              ResultSet resultSet = calculationContext.getResultSet();
64              ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
65              while (resultSet.next()) {
66                  ShardingSpherePreconditions.checkState(!isCanceling(), () -> new PipelineJobCancelingException(
67                          "Calculate chunk canceled, schema name: %s, table name: %s", param.getSchemaName(), param.getLogicTableName()));
68                  Map<String, Object> columnRecord = new LinkedHashMap<>();
69                  for (int columnIndex = 1, columnCount = resultSetMetaData.getColumnCount(); columnIndex <= columnCount; columnIndex++) {
70                      columnRecord.put(resultSetMetaData.getColumnLabel(columnIndex), columnValueReaderEngine.read(resultSet, resultSetMetaData, columnIndex));
71                  }
72                  records.add(columnRecord);
73                  maxUniqueKeyValue = columnValueReaderEngine.read(resultSet, resultSetMetaData, param.getFirstUniqueKey().getOrdinalPosition());
74                  if (records.size() == chunkSize) {
75                      break;
76                  }
77              }
78              if (records.isEmpty()) {
79                  calculationContext.close();
80              }
81              return records.isEmpty() ? Optional.empty() : Optional.of(new RecordSingleTableInventoryCalculatedResult(maxUniqueKeyValue, records));
82          } catch (final PipelineSQLException | PipelineJobCancelingException ex) {
83              calculationContext.close();
84              throw ex;
85              // CHECKSTYLE:OFF
86          } catch (final SQLException | RuntimeException ex) {
87              // CHECKSTYLE:ON
88              calculationContext.close();
89              throw new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), ex);
90          }
91      }
92      
93      private CalculationContext getOrCreateCalculationContext(final SingleTableInventoryCalculateParameter param) {
94          CalculationContext result = (CalculationContext) param.getCalculationContext();
95          if (null != result) {
96              return result;
97          }
98          try {
99              result = createCalculationContext(param);
100             fulfillCalculationContext(result, param);
101             // CHECKSTYLE:OFF
102         } catch (final SQLException | RuntimeException ex) {
103             // CHECKSTYLE:ON
104             QuietlyCloser.close(result);
105             throw new PipelineTableDataConsistencyCheckLoadingFailedException(param.getSchemaName(), param.getLogicTableName(), ex);
106         }
107         return result;
108     }
109     
110     private CalculationContext createCalculationContext(final SingleTableInventoryCalculateParameter param) throws SQLException {
111         Connection connection = param.getDataSource().getConnection();
112         CalculationContext result = new CalculationContext();
113         result.setConnection(connection);
114         param.setCalculationContext(result);
115         return result;
116     }
117     
118     private void fulfillCalculationContext(final CalculationContext calculationContext, final SingleTableInventoryCalculateParameter param) throws SQLException {
119         String sql = getQuerySQL(param);
120         PreparedStatement preparedStatement = JDBCStreamQueryBuilder.build(param.getDatabaseType(), calculationContext.getConnection(), sql);
121         setCurrentStatement(preparedStatement);
122         if (!(param.getDatabaseType() instanceof MySQLDatabaseType)) {
123             preparedStatement.setFetchSize(chunkSize);
124         }
125         calculationContext.setPreparedStatement(preparedStatement);
126         setParameters(preparedStatement, param);
127         ResultSet resultSet = preparedStatement.executeQuery();
128         calculationContext.setResultSet(resultSet);
129     }
130     
131     private String getQuerySQL(final SingleTableInventoryCalculateParameter param) {
132         if (null == param.getFirstUniqueKey()) {
133             throw new UnsupportedOperationException("Record inventory calculator does not support table without unique key and primary key now");
134         }
135         PipelineDataConsistencyCalculateSQLBuilder pipelineSQLBuilder = new PipelineDataConsistencyCalculateSQLBuilder(param.getDatabaseType());
136         Collection<String> columnNames = param.getColumnNames().isEmpty() ? Collections.singleton("*") : param.getColumnNames();
137         boolean firstQuery = null == param.getTableCheckPosition();
138         return pipelineSQLBuilder.buildQueryAllOrderingSQL(param.getSchemaName(), param.getLogicTableName(), columnNames, param.getFirstUniqueKey().getName(), firstQuery);
139     }
140     
141     private void setParameters(final PreparedStatement preparedStatement, final SingleTableInventoryCalculateParameter param) throws SQLException {
142         Object tableCheckPosition = param.getTableCheckPosition();
143         if (null != tableCheckPosition) {
144             preparedStatement.setObject(1, tableCheckPosition);
145         }
146     }
147 }
