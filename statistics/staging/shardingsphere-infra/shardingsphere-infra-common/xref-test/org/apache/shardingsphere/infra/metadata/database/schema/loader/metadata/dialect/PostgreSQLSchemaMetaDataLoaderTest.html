50      private static final String BASIC_TABLE_META_DATA_SQL = "SELECT table_name, column_name, ordinal_position, data_type, udt_name, column_default, table_schema"
51              + " FROM information_schema.columns WHERE table_schema IN ('public')";
52      
53      private static final String TABLE_META_DATA_SQL_WITHOUT_TABLES = BASIC_TABLE_META_DATA_SQL + " ORDER BY ordinal_position";
54      
55      private static final String TABLE_META_DATA_SQL_WITH_TABLES = BASIC_TABLE_META_DATA_SQL + " AND table_name IN ('tbl') ORDER BY ordinal_position";
56      
57      private static final String PRIMARY_KEY_META_DATA_SQL = "SELECT tc.table_name, kc.column_name, kc.table_schema FROM information_schema.table_constraints tc"
58              + " JOIN information_schema.key_column_usage kc ON kc.table_schema = tc.table_schema AND kc.table_name = tc.table_name AND kc.constraint_name = tc.constraint_name"
59              + " WHERE tc.constraint_type = 'PRIMARY KEY' AND kc.ordinal_position IS NOT NULL AND kc.table_schema IN ('public')";
60      
61      private static final String BASIC_INDEX_META_DATA_SQL = "SELECT tablename, indexname, schemaname FROM pg_indexes WHERE schemaname IN ('public')";
62      
63      private static final String BASIC_CONSTRAINT_META_DATA_SQL = "SELECT tc.table_schema,tc.table_name,tc.constraint_name,pgo.relname refer_table_name FROM information_schema.table_constraints tc "
64              + "JOIN pg_constraint pgc ON tc.constraint_name = pgc.conname AND contype='f' "
65              + "JOIN pg_class pgo ON pgc.confrelid = pgo.oid "
66              + "WHERE tc.constraint_type = 'FOREIGN KEY' AND tc.table_schema IN ('public')";
67      
68      private static final String LOAD_ALL_ROLE_TABLE_GRANTS_SQL = "SELECT table_name FROM information_schema.role_table_grants";
69      
70      @Test
71      void assertLoadWithoutTables() throws SQLException {
72          DataSource dataSource = mockDataSource();
73          ResultSet schemaResultSet = mockSchemaMetaDataResultSet();
74          when(dataSource.getConnection().getMetaData().getSchemas()).thenReturn(schemaResultSet);
75          ResultSet tableResultSet = mockTableMetaDataResultSet();
76          when(dataSource.getConnection().prepareStatement(TABLE_META_DATA_SQL_WITHOUT_TABLES).executeQuery()).thenReturn(tableResultSet);
77          ResultSet primaryKeyResultSet = mockPrimaryKeyMetaDataResultSet();
78          when(dataSource.getConnection().prepareStatement(PRIMARY_KEY_META_DATA_SQL).executeQuery()).thenReturn(primaryKeyResultSet);
79          ResultSet indexResultSet = mockIndexMetaDataResultSet();
80          when(dataSource.getConnection().prepareStatement(BASIC_INDEX_META_DATA_SQL).executeQuery()).thenReturn(indexResultSet);
81          ResultSet constraintResultSet = mockConstraintMetaDataResultSet();
82          when(dataSource.getConnection().prepareStatement(BASIC_CONSTRAINT_META_DATA_SQL).executeQuery()).thenReturn(constraintResultSet);
83          ResultSet roleTableGrantsResultSet = mockRoleTableGrantsResultSet();
84          when(dataSource.getConnection().prepareStatement(startsWith(LOAD_ALL_ROLE_TABLE_GRANTS_SQL)).executeQuery()).thenReturn(roleTableGrantsResultSet);
85          assertTableMetaDataMap(getDialectTableMetaDataLoader().load(dataSource, Collections.emptyList(), "sharding_db"));
86      }
87      
88      private ResultSet mockSchemaMetaDataResultSet() throws SQLException {
89          ResultSet result = mock(ResultSet.class);
90          when(result.next()).thenReturn(true, false);
91          when(result.getString("TABLE_SCHEM")).thenReturn("public");
92          return result;
93      }

96      void assertLoadWithTables() throws SQLException {
97          DataSource dataSource = mockDataSource();
98          ResultSet schemaResultSet = mockSchemaMetaDataResultSet();
99          when(dataSource.getConnection().getMetaData().getSchemas()).thenReturn(schemaResultSet);
100         ResultSet tableResultSet = mockTableMetaDataResultSet();

102         ResultSet primaryKeyResultSet = mockPrimaryKeyMetaDataResultSet();
103         when(dataSource.getConnection().prepareStatement(PRIMARY_KEY_META_DATA_SQL).executeQuery()).thenReturn(primaryKeyResultSet);
104         ResultSet indexResultSet = mockIndexMetaDataResultSet();
105         when(dataSource.getConnection().prepareStatement(BASIC_INDEX_META_DATA_SQL).executeQuery()).thenReturn(indexResultSet);
106         ResultSet constraintResultSet = mockConstraintMetaDataResultSet();
107         when(dataSource.getConnection().prepareStatement(BASIC_CONSTRAINT_META_DATA_SQL).executeQuery()).thenReturn(constraintResultSet);
108         ResultSet roleTableGrantsResultSet = mockRoleTableGrantsResultSet();
109         when(dataSource.getConnection().prepareStatement(startsWith(LOAD_ALL_ROLE_TABLE_GRANTS_SQL)).executeQuery()).thenReturn(roleTableGrantsResultSet);
110         assertTableMetaDataMap(getDialectTableMetaDataLoader().load(dataSource, Collections.singletonList("tbl"), "sharding_db"));
111     }

176     private DialectSchemaMetaDataLoader getDialectTableMetaDataLoader() {
177         Optional<DialectSchemaMetaDataLoader> result = DatabaseTypedSPILoader.findService(DialectSchemaMetaDataLoader.class, TypedSPILoader.getService(DatabaseType.class, "PostgreSQL"));
178         assertTrue(result.isPresent());
179         return result.get();
180     }

182     private void assertTableMetaDataMap(final Collection<SchemaMetaData> schemaMetaDataList) {
183         assertThat(schemaMetaDataList.size(), is(1));
184         TableMetaData actualTableMetaData = schemaMetaDataList.iterator().next().getTables().iterator().next();
185         assertThat(actualTableMetaData.getColumns().size(), is(2));
186         Iterator<ColumnMetaData> columnsIterator = actualTableMetaData.getColumns().iterator();
187         assertThat(columnsIterator.next(), is(new ColumnMetaData("id", Types.INTEGER, true, true, true, true, false)));
188         assertThat(columnsIterator.next(), is(new ColumnMetaData("name", Types.VARCHAR, false, false, true, true, false)));
189         assertThat(actualTableMetaData.getIndexes().size(), is(1));
190         Iterator<IndexMetaData> indexesIterator = actualTableMetaData.getIndexes().iterator();
191         assertThat(indexesIterator.next(), is(new IndexMetaData("id")));
192         assertThat(actualTableMetaData.getConstraints().size(), is(1));
193         Iterator<ConstraintMetaData> constrainsIterator = actualTableMetaData.getConstraints().iterator();
194         assertThat(constrainsIterator.next(), is(new ConstraintMetaData("tbl_con", "refer_tbl")));
195     }
