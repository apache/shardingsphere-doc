202     @Test
203     void assertWatchUpdatedChangedType() throws ExecutionException, InterruptedException {
204         mockCache("/test/children_updated/1");
205         ChildData oldData = new ChildData("/test/children_updated/1", null, "value1".getBytes());
206         ChildData data = new ChildData("/test/children_updated/1", null, "value2".getBytes());
207         doAnswer(AdditionalAnswers.answerVoid(getListenerAnswer(CuratorCacheListener.Type.NODE_CHANGED, oldData, data))).when(listenable).addListener(any(CuratorCacheListener.class));
208         SettableFuture<DataChangedEvent> settableFuture = SettableFuture.create();
209         REPOSITORY.watch("/test/children_updated/1", settableFuture::set);
210         DataChangedEvent dataChangedEvent = settableFuture.get();
211         assertThat(dataChangedEvent.getType(), is(Type.UPDATED));
212         assertThat(dataChangedEvent.getKey(), is("/test/children_updated/1"));
213         assertThat(dataChangedEvent.getValue(), is("value2"));
214     }

243     @SneakyThrows(ReflectiveOperationException.class)
244     private void mockCache(final String key) {
245         Map<String, CuratorCache> caches = new HashMap<>();
246         caches.put(key, curatorCache);
247         Plugins.getMemberAccessor().set(ZookeeperRepository.class.getDeclaredField("caches"), REPOSITORY, caches);
248         when(curatorCache.listenable()).thenReturn(listenable);
249     }

251     private VoidAnswer1<CuratorCacheListener> getListenerAnswer(final CuratorCacheListener.Type type, final ChildData oldData, final ChildData data) {
252         return listener -> listener.event(type, oldData, data);
253     }
