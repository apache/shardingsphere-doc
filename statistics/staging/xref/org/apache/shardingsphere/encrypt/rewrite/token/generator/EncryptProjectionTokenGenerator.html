60  /**
61   * Projection token generator for encrypt.
62   */
63  @Setter
64  public final class EncryptProjectionTokenGenerator implements CollectionSQLTokenGenerator<SQLStatementContext>, PreviousSQLTokensAware, SchemaMetaDataAware, EncryptRuleAware, DatabaseTypeAware {
65      
66      private List<SQLToken> previousSQLTokens;
67      
68      private EncryptRule encryptRule;
69      
70      private String databaseName;
71      
72      private Map<String, ShardingSphereSchema> schemas;
73      
74      private DatabaseType databaseType;
75      
76      @Override
77      public boolean isGenerateSQLToken(final SQLStatementContext sqlStatementContext) {
78          return sqlStatementContext instanceof SelectStatementContext && !((SelectStatementContext) sqlStatementContext).getAllTables().isEmpty();
79      }
80      
81      @Override
82      public Collection<SQLToken> generateSQLTokens(final SQLStatementContext sqlStatementContext) {
83          Preconditions.checkState(sqlStatementContext instanceof SelectStatementContext);
84          Collection<SQLToken> result = new LinkedHashSet<>();
85          SelectStatementContext selectStatementContext = (SelectStatementContext) sqlStatementContext;
86          addGenerateSQLTokens(result, selectStatementContext);
87          for (SelectStatementContext each : selectStatementContext.getSubqueryContexts().values()) {
88              addGenerateSQLTokens(result, each);
89          }
90          return result;
91      }
92      
93      private void addGenerateSQLTokens(final Collection<SQLToken> sqlTokens, final SelectStatementContext selectStatementContext) {
94          Map<String, String> columnTableNames = getColumnTableNames(selectStatementContext);
95          for (ProjectionSegment each : selectStatementContext.getSqlStatement().getProjections().getProjections()) {
96              SubqueryType subqueryType = selectStatementContext.getSubqueryType();
97              if (each instanceof ColumnProjectionSegment) {
98                  ColumnProjectionSegment columnSegment = (ColumnProjectionSegment) each;
99                  ColumnProjection columnProjection = buildColumnProjection(columnSegment);
100                 String tableName = columnTableNames.get(columnProjection.getExpression());
101                 if (null == tableName) {
102                     continue;
103                 }
104                 Optional<EncryptTable> encryptTable = encryptRule.findEncryptTable(tableName);
105                 if (encryptTable.isPresent() && encryptTable.get().isEncryptColumn(columnProjection.getName().getValue()) && !containsTableSubquery(selectStatementContext)) {
106                     sqlTokens.add(generateSQLToken(encryptTable.get().getEncryptColumn(columnProjection.getName().getValue()), columnSegment, columnProjection, subqueryType));
107                 }
108             }
109             if (each instanceof ShorthandProjectionSegment) {
110                 ShorthandProjectionSegment shorthandSegment = (ShorthandProjectionSegment) each;
111                 Collection<Projection> actualColumns = getShorthandProjection(shorthandSegment, selectStatementContext.getProjectionsContext()).getActualColumns();
112                 if (!actualColumns.isEmpty()) {
113                     sqlTokens.add(generateSQLToken(shorthandSegment, actualColumns, selectStatementContext, subqueryType, columnTableNames));
114                 }
115             }
116         }
117     }
118     
119     private SubstitutableColumnNameToken generateSQLToken(final EncryptColumn encryptColumn, final ColumnProjectionSegment columnSegment,
120                                                           final ColumnProjection columnProjection, final SubqueryType subqueryType) {
121         Collection<Projection> projections = generateProjections(encryptColumn, columnProjection, subqueryType, false, null);
122         int startIndex = columnSegment.getColumn().getOwner().isPresent() ? columnSegment.getColumn().getOwner().get().getStopIndex() + 2 : columnSegment.getColumn().getStartIndex();
123         int stopIndex = columnSegment.getStopIndex();
124         return new SubstitutableColumnNameToken(startIndex, stopIndex, projections);
125     }
126     
127     private SubstitutableColumnNameToken generateSQLToken(final ShorthandProjectionSegment segment, final Collection<Projection> actualColumns,
128                                                           final SelectStatementContext selectStatementContext, final SubqueryType subqueryType, final Map<String, String> columnTableNames) {
129         List<Projection> projections = new LinkedList<>();
130         for (Projection each : actualColumns) {
131             String tableName = columnTableNames.get(each.getExpression());
132             Optional<EncryptTable> encryptTable = null == tableName ? Optional.empty() : encryptRule.findEncryptTable(tableName);
133             if (!encryptTable.isPresent() || !encryptTable.get().isEncryptColumn(each.getColumnLabel()) || containsTableSubquery(selectStatementContext)) {
134                 projections.add(each.getAlias().map(optional -> (Projection) new ColumnProjection(null, optional, null, databaseType)).orElse(each));
135             } else if (each instanceof ColumnProjection) {
136                 projections.addAll(generateProjections(encryptTable.get().getEncryptColumn(((ColumnProjection) each).getName().getValue()), (ColumnProjection) each, subqueryType, true, segment));
137             }
138         }
139         int startIndex = segment.getOwner().isPresent() ? segment.getOwner().get().getStartIndex() : segment.getStartIndex();
140         previousSQLTokens.removeIf(each -> each.getStartIndex() == startIndex);
141         DialectDatabaseMetaData dialectDatabaseMetaData = DatabaseTypedSPILoader.getService(DialectDatabaseMetaData.class, selectStatementContext.getDatabaseType());
142         return new SubstitutableColumnNameToken(startIndex, segment.getStopIndex(), projections, dialectDatabaseMetaData.getQuoteCharacter());
143     }
144     
145     private boolean containsTableSubquery(final SelectStatementContext selectStatementContext) {
146         if (selectStatementContext.getSqlStatement().getFrom() instanceof SubqueryTableSegment) {
147             return true;
148         } else if (selectStatementContext.getSqlStatement().getFrom() instanceof JoinTableSegment) {
149             JoinTableSegment joinTableSegment = (JoinTableSegment) selectStatementContext.getSqlStatement().getFrom();
150             return joinTableSegment.getLeft() instanceof SubqueryTableSegment || joinTableSegment.getRight() instanceof SubqueryTableSegment;
151         }
152         return false;
153     }
154     
155     private ColumnProjection buildColumnProjection(final ColumnProjectionSegment segment) {
156         IdentifierValue owner = segment.getColumn().getOwner().map(OwnerSegment::getIdentifier).orElse(null);
157         return new ColumnProjection(owner, segment.getColumn().getIdentifier(), segment.getAliasName().isPresent() ? segment.getAlias().orElse(null) : null, databaseType);
158     }
159     
160     private Map<String, String> getColumnTableNames(final SelectStatementContext selectStatementContext) {
161         Collection<ColumnProjection> columns = new LinkedList<>();
162         for (Projection projection : selectStatementContext.getProjectionsContext().getProjections()) {
163             if (projection instanceof ColumnProjection) {
164                 columns.add((ColumnProjection) projection);
165             }
166             if (projection instanceof ShorthandProjection) {
167                 columns.addAll(((ShorthandProjection) projection).getColumnProjections());
168             }
169         }
170         String defaultSchema = new DatabaseTypeRegistry(selectStatementContext.getDatabaseType()).getDefaultSchemaName(databaseName);
171         ShardingSphereSchema schema = selectStatementContext.getTablesContext().getSchemaName().map(schemas::get).orElseGet(() -> schemas.get(defaultSchema));
172         return selectStatementContext.getTablesContext().findTableNamesByColumnProjection(columns, schema);
173     }
174     
175     private Collection<Projection> generateProjections(final EncryptColumn encryptColumn, final ColumnProjection column,
176                                                        final SubqueryType subqueryType, final boolean shorthand, final ShorthandProjectionSegment segment) {
177         Collection<Projection> result = new LinkedList<>();
178         if (SubqueryType.PREDICATE_SUBQUERY == subqueryType) {
179             result.add(distinctOwner(generatePredicateSubqueryProjection(encryptColumn, column), shorthand));
180         } else if (SubqueryType.TABLE_SUBQUERY == subqueryType) {
181             result.addAll(generateTableSubqueryProjections(encryptColumn, column, shorthand));
182         } else if (SubqueryType.EXISTS_SUBQUERY == subqueryType) {
183             result.addAll(generateExistsSubqueryProjections(encryptColumn, column, shorthand));
184         } else {
185             result.add(distinctOwner(generateCommonProjection(encryptColumn, column, segment), shorthand));
186         }
187         return result;
188     }
189     
190     private ColumnProjection distinctOwner(final ColumnProjection column, final boolean shorthand) {
191         if (shorthand || !column.getOwner().isPresent()) {
192             return column;
193         }
194         return new ColumnProjection(null, column.getName(), column.getAlias().isPresent() ? column.getAlias().get() : null, databaseType);
195     }
196     
197     private ColumnProjection generatePredicateSubqueryProjection(final EncryptColumn encryptColumn, final ColumnProjection column) {
198         Optional<AssistedQueryColumnItem> assistedQueryColumn = encryptColumn.getAssistedQuery();
199         if (assistedQueryColumn.isPresent()) {
200             return new ColumnProjection(column.getOwner().orElse(null), new IdentifierValue(assistedQueryColumn.get().getName(), column.getName().getQuoteCharacter()), null, databaseType);
201         }
202         String cipherColumn = encryptColumn.getCipher().getName();
203         return new ColumnProjection(column.getOwner().orElse(null), new IdentifierValue(cipherColumn, column.getName().getQuoteCharacter()), null, databaseType);
204     }
205     
206     private Collection<ColumnProjection> generateTableSubqueryProjections(final EncryptColumn encryptColumn, final ColumnProjection column, final boolean shorthand) {
207         Collection<ColumnProjection> result = new LinkedList<>();
208         result.add(distinctOwner(new ColumnProjection(column.getOwner().orElse(null), new IdentifierValue(encryptColumn.getCipher().getName(),
209                 column.getName().getQuoteCharacter()), column.getAlias().orElse(column.getName()), databaseType), shorthand));
210         encryptColumn.getAssistedQuery().ifPresent(optional -> result.add(
211                 new ColumnProjection(column.getOwner().orElse(null), new IdentifierValue(optional.getName(), column.getName().getQuoteCharacter()), null, databaseType)));
212         return result;
213     }
214     
215     private Collection<ColumnProjection> generateExistsSubqueryProjections(final EncryptColumn encryptColumn, final ColumnProjection column, final boolean shorthand) {
216         Collection<ColumnProjection> result = new LinkedList<>();
217         result.add(distinctOwner(new ColumnProjection(column.getOwner().orElse(null), new IdentifierValue(encryptColumn.getCipher().getName(),
218                 column.getName().getQuoteCharacter()), null, databaseType), shorthand));
219         return result;
220     }
221     
222     private ColumnProjection generateCommonProjection(final EncryptColumn encryptColumn, final ColumnProjection column, final ShorthandProjectionSegment segment) {
223         String queryColumnName = encryptColumn.getCipher().getName();
224         IdentifierValue owner = (null == segment || !segment.getOwner().isPresent()) ? column.getOwner().orElse(null) : segment.getOwner().get().getIdentifier();
225         return new ColumnProjection(owner, new IdentifierValue(queryColumnName, column.getName().getQuoteCharacter()), column.getAlias().isPresent()
226                 ? column.getAlias().get()
227                 : column.getName(), databaseType);
228     }
229     
230     private ShorthandProjection getShorthandProjection(final ShorthandProjectionSegment segment, final ProjectionsContext projectionsContext) {
231         Optional<String> owner = segment.getOwner().isPresent() ? Optional.of(segment.getOwner().get().getIdentifier().getValue()) : Optional.empty();
232         for (Projection each : projectionsContext.getProjections()) {
233             if (each instanceof ShorthandProjection) {
234                 if (!owner.isPresent() && !((ShorthandProjection) each).getOwner().isPresent()) {
235                     return (ShorthandProjection) each;
236                 }
237                 if (owner.isPresent() && owner.get().equals(((ShorthandProjection) each).getOwner().map(IdentifierValue::getValue).orElse(null))) {
238                     return (ShorthandProjection) each;
239                 }
240             }
241         }
242         throw new IllegalStateException(String.format("Can not find shorthand projection segment, owner is `%s`", owner.orElse(null)));
243     }
