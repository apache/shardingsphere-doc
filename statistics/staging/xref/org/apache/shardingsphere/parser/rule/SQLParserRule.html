18  package org.apache.shardingsphere.parser.rule;
19  
20  import lombok.Getter;
21  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
22  import org.apache.shardingsphere.infra.parser.SQLParserEngine;
23  import org.apache.shardingsphere.infra.parser.ShardingSphereSQLParserEngine;
24  import org.apache.shardingsphere.infra.rule.scope.GlobalRule;
25  import org.apache.shardingsphere.parser.config.SQLParserRuleConfiguration;
26  import org.apache.shardingsphere.sql.parser.api.CacheOption;
27  
28  /**
29   * SQL parser rule.
30   */
31  @Getter
32  public final class SQLParserRule implements GlobalRule {
33      
34      private final SQLParserRuleConfiguration configuration;
35      
36      private final CacheOption sqlStatementCache;
37      
38      private final CacheOption parseTreeCache;
39      
40      public SQLParserRule(final SQLParserRuleConfiguration ruleConfig) {
41          configuration = ruleConfig;
42          sqlStatementCache = ruleConfig.getSqlStatementCache();
43          parseTreeCache = ruleConfig.getParseTreeCache();
44      }
45      
46      /**
47       * Get SQL parser engine.
48       *
49       * @param databaseType database type
50       * @return SQL parser engine
51       */
52      public SQLParserEngine getSQLParserEngine(final DatabaseType databaseType) {
53          return new ShardingSphereSQLParserEngine(databaseType, sqlStatementCache, parseTreeCache);
54      }
55  }
