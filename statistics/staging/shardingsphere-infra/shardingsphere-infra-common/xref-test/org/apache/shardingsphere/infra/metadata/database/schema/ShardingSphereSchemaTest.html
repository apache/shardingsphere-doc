18  package org.apache.shardingsphere.infra.metadata.database.schema;
19  
20  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereColumn;
21  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereIndex;
22  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereSchema;
23  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereTable;
24  import org.apache.shardingsphere.infra.metadata.database.schema.model.ShardingSphereView;
25  import org.junit.jupiter.api.Test;
26  
27  import java.util.Collections;
28  import java.util.HashSet;
29  
30  import static org.hamcrest.CoreMatchers.is;
31  import static org.hamcrest.MatcherAssert.assertThat;
32  import static org.junit.jupiter.api.Assertions.assertNull;
33  import static org.junit.jupiter.api.Assertions.assertTrue;
34  import static org.mockito.Mockito.mock;
35  
36  class ShardingSphereSchemaTest {
37      
38      @Test
39      void assertGetAllTableNames() {
40          assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl", mock(ShardingSphereTable.class)), Collections.emptyMap()).getAllTableNames(),
41                  is(new HashSet<>(Collections.singleton("tbl"))));
42      }
43      
44      @Test
45      void assertGetAllViewNames() {
46          assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl", mock(ShardingSphereTable.class)),
47                  Collections.singletonMap("tbl_view", mock(ShardingSphereView.class))).getAllViewNames(), is(new HashSet<>(Collections.singleton("tbl_view"))));
48      }
49      
50      @Test
51      void assertGetTable() {
52          ShardingSphereTable table = mock(ShardingSphereTable.class);
53          assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl", table), Collections.emptyMap()).getTable("tbl"), is(table));
54      }
55      
56      @Test
57      void assertGetView() {
58          ShardingSphereView view = mock(ShardingSphereView.class);
59          assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl", mock(ShardingSphereTable.class)), Collections.singletonMap("tbl_view", view)).getView("tbl_view"), is(view));
60      }
61      
62      @Test
63      void assertPutTable() {
64          ShardingSphereSchema actual = new ShardingSphereSchema(Collections.emptyMap(), Collections.emptyMap());
65          ShardingSphereTable table = mock(ShardingSphereTable.class);
66          actual.putTable("tbl", table);
67          assertThat(actual.getTable("tbl"), is(table));
68      }
69      
70      @Test
71      void assertRemoveTable() {
72          ShardingSphereSchema actual = new ShardingSphereSchema(Collections.singletonMap("tbl", mock(ShardingSphereTable.class)), Collections.emptyMap());
73          actual.removeTable("tbl");
74          assertNull(actual.getTable("tbl"));
75      }
76      
77      @Test
78      void assertContainsTable() {
79          assertTrue(new ShardingSphereSchema(Collections.singletonMap("tbl", mock(ShardingSphereTable.class)), Collections.emptyMap()).containsTable("tbl"));
80      }
81      
82      @Test
83      void assertContainsColumn() {
84          ShardingSphereTable table = new ShardingSphereTable("tbl", Collections.singletonList(
85                  new ShardingSphereColumn("col", 0, false, false, false, true, false, false)), Collections.emptyList(), Collections.emptyList());
86          assertTrue(new ShardingSphereSchema(Collections.singletonMap("tbl", table), Collections.emptyMap()).containsColumn("tbl", "col"));
87      }
88      
89      @Test
90      void assertGetAllColumnNamesWhenContainsKey() {
91          ShardingSphereTable table = new ShardingSphereTable("tbl", Collections.singletonList(
92                  new ShardingSphereColumn("col", 0, false, false, false, true, false, false)), Collections.emptyList(), Collections.emptyList());
93          assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl", table), Collections.emptyMap()).getAllColumnNames("tbl"), is(Collections.singletonList("col")));
94      }
95      
96      @Test
97      void assertGetAllColumnNamesWhenNotContainsKey() {
98          ShardingSphereTable table = new ShardingSphereTable("tbl", Collections.singletonList(
99                  new ShardingSphereColumn("col", 0, false, false, false, true, false, false)), Collections.emptyList(), Collections.emptyList());
100         assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl1", table), Collections.emptyMap()).getAllColumnNames("tbl2"), is(Collections.<String>emptyList()));
101     }
102     
103     @Test
104     void assertContainsIndex() {
105         ShardingSphereTable table = new ShardingSphereTable("tbl", Collections.emptyList(), Collections.singletonList(new ShardingSphereIndex("col_idx")), Collections.emptyList());
106         assertTrue(new ShardingSphereSchema(Collections.singletonMap("tbl", table), Collections.emptyMap()).containsIndex("tbl", "col_idx"));
107     }
108     
109     @Test
110     void assertGetVisibleColumnNamesWhenContainsKey() {
111         ShardingSphereTable table = new ShardingSphereTable("tbl", Collections.singletonList(
112                 new ShardingSphereColumn("col", 0, false, false, false, true, false, false)), Collections.emptyList(), Collections.emptyList());
113         assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl", table), Collections.emptyMap()).getVisibleColumnNames("tbl"), is(Collections.singletonList("col")));
114     }
115     
116     @Test
117     void assertGetVisibleColumnNamesWhenNotContainsKey() {
118         ShardingSphereTable table = new ShardingSphereTable("tbl", Collections.singletonList(
119                 new ShardingSphereColumn("col", 0, false, false, false, false, true, false)), Collections.emptyList(), Collections.emptyList());
120         assertThat(new ShardingSphereSchema(Collections.singletonMap("tbl", table), Collections.emptyMap()).getVisibleColumnNames("tbl"), is(Collections.emptyList()));
121     }
122 }
