57  @ExtendWith(MockitoExtension.class)
58  @MockitoSettings(strictness = Strictness.LENIENT)
59  class MySQLPacketCodecEngineTest {
60      
61      @Mock(answer = Answers.RETURNS_DEEP_STUBS)
62      private ChannelHandlerContext context;
63      
64      @Mock
65      private ByteBuf byteBuf;
66      
67      @BeforeEach
68      void setup() {
69          when(context.channel().attr(AttributeKey.<Charset>valueOf(Charset.class.getName())).get()).thenReturn(StandardCharsets.UTF_8);
70          when(context.channel().attr(MySQLConstants.MYSQL_SEQUENCE_ID).get()).thenReturn(new AtomicInteger());
71      }
72      
73      @Test
74      void assertIsValidHeader() {
75          assertTrue(new MySQLPacketCodecEngine().isValidHeader(50));
76      }
77      
78      @Test
79      void assertIsInvalidHeader() {
80          assertFalse(new MySQLPacketCodecEngine().isValidHeader(3));
81      }
82      
83      @Test
84      void assertDecode() {
85          when(byteBuf.markReaderIndex()).thenReturn(byteBuf);
86          when(byteBuf.readUnsignedMediumLE()).thenReturn(50);
87          when(byteBuf.readableBytes()).thenReturn(51);
88          List<Object> out = new LinkedList<>();
89          new MySQLPacketCodecEngine().decode(context, byteBuf, out);
90          assertThat(out.size(), is(1));
91      }
92      
93      @Test
94      void assertDecodeWithEmptyPacket() {
95          when(byteBuf.markReaderIndex()).thenReturn(byteBuf);
96          when(byteBuf.readableBytes()).thenReturn(1);
97          when(byteBuf.readUnsignedMediumLE()).thenReturn(0);
98          List<Object> out = new LinkedList<>();
99          new MySQLPacketCodecEngine().decode(context, byteBuf, out);
100         assertThat(out.size(), is(1));
101     }
102     
103     @Test
104     void assertDecodeWithStickyPacket() {
105         when(byteBuf.markReaderIndex()).thenReturn(byteBuf);
106         when(byteBuf.readUnsignedMediumLE()).thenReturn(50);
107         List<Object> out = new LinkedList<>();
108         new MySQLPacketCodecEngine().decode(context, byteBuf, out);
109         assertTrue(out.isEmpty());
110     }
111     
112     @Test
113     void assertDecodePacketMoreThan16MB() {
114         MySQLPacketCodecEngine engine = new MySQLPacketCodecEngine();
115         when(context.alloc().compositeBuffer(2)).thenReturn(new CompositeByteBuf(UnpooledByteBufAllocator.DEFAULT, false, 2));
116         List<Object> actual = new ArrayList<>(1);
117         for (ByteBuf each : preparePacketMoreThan16MB()) {
118             engine.decode(context, each, actual);
119         }
120         assertThat(actual.size(), is(1));
121         assertThat(((ByteBuf) actual.get(0)).readableBytes(), is((1 << 24) - 1));
122     }
123     
124     private List<ByteBuf> preparePacketMoreThan16MB() {
125         byte[] firstPacketData = new byte[4 + (1 << 24) - 1];
126         firstPacketData[0] = firstPacketData[1] = firstPacketData[2] = (byte) 0xff;
127         firstPacketData[3] = (byte) 0;
128         byte[] secondPacketData = new byte[]{0x00, 0x00, 0x00, 0x01};
129         return Arrays.asList(Unpooled.wrappedBuffer(firstPacketData), Unpooled.wrappedBuffer(secondPacketData));
130     }
131     
132     @Test
133     void assertEncode() {
134         when(byteBuf.writeInt(anyInt())).thenReturn(byteBuf);
135         when(byteBuf.markWriterIndex()).thenReturn(byteBuf);
136         when(byteBuf.readableBytes()).thenReturn(8);
137         MySQLPacket actualMessage = mock(MySQLPacket.class);
138         context.channel().attr(MySQLConstants.MYSQL_SEQUENCE_ID).get().set(1);
139         new MySQLPacketCodecEngine().encode(context, actualMessage, byteBuf);
140         verify(byteBuf).writeInt(0);
141         verify(byteBuf).markWriterIndex();
142         verify(actualMessage).write(any(MySQLPacketPayload.class));
143         verify(byteBuf).setMediumLE(0, 4);
144         verify(byteBuf).setByte(3, 1);
145     }
146     
147     @Test
148     void assertEncodePacketMoreThan16MB() {
149         CompositeByteBuf expected = new CompositeByteBuf(UnpooledByteBufAllocator.DEFAULT, false, 6);
150         when(context.alloc().compositeBuffer(6)).thenReturn(expected);
151         when(context.alloc().ioBuffer(4, 4)).thenReturn(Unpooled.buffer(4, 4), Unpooled.buffer(4, 4), Unpooled.buffer(4, 4));
152         MySQLPacketCodecEngine engine = new MySQLPacketCodecEngine();
153         engine.encode(context, mock(MySQLPacket.class), prepareByteBufAlmost32MB());
154         verify(context).write(expected);
155         assertThat(expected.numComponents(), is(5));
156         assertThat(expected.component(0).getUnsignedByte(3), is((short) 0));
157         assertThat(expected.component(1).readableBytes(), is(0xFFFFFF));
158         assertThat(expected.component(2).getUnsignedByte(3), is((short) 1));
159         assertThat(expected.component(3).readableBytes(), is(0xFFFFFF));
160         assertThat(expected.component(4).getUnsignedByte(3), is((short) 2));
161     }
162     
163     private ByteBuf prepareByteBufAlmost32MB() {
164         int max = 0xFFFFFF << 1;
165         ByteBuf result = Unpooled.buffer(max + 4);
166         result.writerIndex(max);
167         return result.retain();
168     }
169     
170     @Test
171     void assertEncodeOccursException() {
172         when(byteBuf.writeInt(anyInt())).thenReturn(byteBuf);
173         when(byteBuf.markWriterIndex()).thenReturn(byteBuf);
174         when(byteBuf.readableBytes()).thenReturn(12);
175         RuntimeException ex = mock(RuntimeException.class);
176         MySQLPacket actualMessage = mock(MySQLPacket.class);
177         doThrow(ex).when(actualMessage).write(any(MySQLPacketPayload.class));
178         context.channel().attr(MySQLConstants.MYSQL_SEQUENCE_ID).get().set(2);
179         new MySQLPacketCodecEngine().encode(context, actualMessage, byteBuf);
180         verify(byteBuf).writeInt(0);
181         verify(byteBuf).markWriterIndex();
182         verify(byteBuf).resetWriterIndex();
183         verify(byteBuf).setMediumLE(0, 8);
184         verify(byteBuf).setByte(3, 2);
185     }
186     
187     @Test
188     void assertCreatePacketPayload() {
189         assertThat(new MySQLPacketCodecEngine().createPacketPayload(byteBuf, StandardCharsets.UTF_8).getByteBuf(), is(byteBuf));
190     }
