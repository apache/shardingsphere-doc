18  package org.apache.shardingsphere.test.e2e.transaction.util;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  import lombok.SneakyThrows;
23  import org.apache.shardingsphere.test.e2e.transaction.cases.base.BaseTransactionTestCase;
24  
25  import java.io.File;
26  import java.io.IOException;
27  import java.io.UnsupportedEncodingException;
28  import java.net.JarURLConnection;
29  import java.net.URL;
30  import java.net.URLDecoder;
31  import java.util.Collection;
32  import java.util.Enumeration;
33  import java.util.LinkedList;
34  import java.util.List;
35  import java.util.Objects;
36  import java.util.jar.JarEntry;
37  import java.util.jar.JarFile;
38  
39  /**
40   * Scan the transaction test classes to be tested.
41   */
42  @NoArgsConstructor(access = AccessLevel.PRIVATE)
43  public final class TestCaseClassScanner {
44      
45      private static final String TEST_CASE_PACKAGE_NAME = "org.apache.shardingsphere.test.e2e.transaction.cases";
46      
47      private static final Collection<File> CLASS_FILES = new LinkedList<>();
48      
49      private static final String CLASS_SUFFIX = ".class";
50      
51      /**
52       * Scan transaction test case classes in classpath.
53       *
54       * @return transaction test case classes
55       */
56      @SneakyThrows({IOException.class, ClassNotFoundException.class})
57      public static List<Class<? extends BaseTransactionTestCase>> scan() {
58          Enumeration<URL> urls = Thread.currentThread().getContextClassLoader().getResources(TEST_CASE_PACKAGE_NAME.replace(".", File.separator));
59          return scanURL(urls);
60      }
61      
62      private static List<Class<? extends BaseTransactionTestCase>> scanURL(final Enumeration<URL> urls) throws IOException, ClassNotFoundException {
63          List<Class<? extends BaseTransactionTestCase>> result = new LinkedList<>();
64          while (urls.hasMoreElements()) {
65              URL url = urls.nextElement();
66              String protocol = url.getProtocol();
67              switch (protocol) {
68                  case "file":
69                      addTestCaseClassesFromClassFiles(url, result);
70                      break;
71                  case "jar":
72                      addTestCaseClassesInJars(url, result);
73                      break;
74                  default:
75                      break;
76              }
77          }
78          return result;
79      }
80      
81      private static void addTestCaseClassesFromClassFiles(final URL url, final List<Class<? extends BaseTransactionTestCase>> caseClasses) throws UnsupportedEncodingException, ClassNotFoundException {
82          String filepath = URLDecoder.decode(url.getFile(), "UTF-8");
83          File file = new File(filepath);
84          scanClassFiles(file);
85          addTestCaseClasses(caseClasses);
86      }
87      
88      private static void addTestCaseClassesInJars(final URL url, final List<Class<? extends BaseTransactionTestCase>> caseClasses) throws IOException, ClassNotFoundException {
89          JarURLConnection connection = (JarURLConnection) url.openConnection();
90          if (null == connection) {
91              return;
92          }
93          JarFile jarFile = connection.getJarFile();
94          if (null == jarFile) {
95              return;
96          }
97          Enumeration<JarEntry> jarEntryEnumeration = jarFile.entries();
98          while (jarEntryEnumeration.hasMoreElements()) {
99              JarEntry entry = jarEntryEnumeration.nextElement();
100             String jarEntryName = entry.getName();
101             if (jarEntryName.contains(CLASS_SUFFIX) && jarEntryName.replace(File.separator, ".").startsWith(TEST_CASE_PACKAGE_NAME)) {
102                 String className = jarEntryName.substring(0, jarEntryName.lastIndexOf(".")).replace(File.separator, ".");
103                 Class<?> clazz = Class.forName(className);
104                 addClass(caseClasses, clazz);
105             }
106         }
107     }
108     
109     private static void scanClassFiles(final File file) {
110         if (file.isDirectory()) {
111             for (File f : Objects.requireNonNull(file.listFiles())) {
112                 scanClassFiles(f);
113             }
114         } else {
115             if (file.getName().endsWith(CLASS_SUFFIX)) {
116                 CLASS_FILES.add(file);
117             }
118         }
119     }
120     
121     private static void addTestCaseClasses(final List<Class<? extends BaseTransactionTestCase>> caseClasses) throws ClassNotFoundException {
122         for (File file : CLASS_FILES) {
123             String fileName = file.getAbsolutePath();
124             if (fileName.endsWith(CLASS_SUFFIX)) {
125                 String noSuffixFileName = fileName.substring(8 + fileName.lastIndexOf("classes"), fileName.indexOf(CLASS_SUFFIX));
126                 String filePackage = noSuffixFileName.replace(File.separator, ".");
127                 Class<?> clazz = Class.forName(filePackage);
128                 addClass(caseClasses, clazz);
129             }
130         }
131     }
132     
133     @SuppressWarnings("unchecked")
134     private static void addClass(final List<Class<? extends BaseTransactionTestCase>> caseClasses, final Class<?> clazz) {
135         if (BaseTransactionTestCase.class.isAssignableFrom(clazz)) {
136             caseClasses.add((Class<? extends BaseTransactionTestCase>) clazz);
137         }
138     }
139 }
