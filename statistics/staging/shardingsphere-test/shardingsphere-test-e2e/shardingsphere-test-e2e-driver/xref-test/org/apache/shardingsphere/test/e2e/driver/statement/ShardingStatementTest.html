38      @Test
39      void assertGetGeneratedKeys() throws SQLException {
40          String sql = "INSERT INTO t_order_item(order_id, user_id, status) VALUES (%d, %d, '%s')";
41          try (
42                  Connection connection = getShardingSphereDataSource().getConnection();
43                  Statement statement = connection.createStatement()) {
44              assertFalse(statement.execute(String.format(sql, 1, 1, "init")));
45              assertFalse(statement.execute(String.format(sql, 1, 1, "init"), Statement.NO_GENERATED_KEYS));
46              assertFalse(statement.execute(String.format(sql, 1, 1, "init"), Statement.RETURN_GENERATED_KEYS));
47              ResultSet generatedKeysResultSet = statement.getGeneratedKeys();
48              assertTrue(generatedKeysResultSet.next());
49              assertThat(generatedKeysResultSet.getLong(1), is(3L));
50              assertFalse(statement.execute(String.format(sql, 1, 1, "init"), new int[]{1}));
51              generatedKeysResultSet = statement.getGeneratedKeys();
52              assertTrue(generatedKeysResultSet.next());
53              assertThat(generatedKeysResultSet.getLong(1), is(4L));
54              assertFalse(statement.execute(String.format(sql, 1, 1, "init"), new String[]{"user_id"}));
55              generatedKeysResultSet = statement.getGeneratedKeys();
56              assertTrue(generatedKeysResultSet.next());
57              assertThat(generatedKeysResultSet.getLong(1), is(5L));
58              assertFalse(statement.execute(String.format(sql, 1, 1, "init"), new int[]{2}));
59              generatedKeysResultSet = statement.getGeneratedKeys();
60              assertTrue(generatedKeysResultSet.next());
61              assertThat(generatedKeysResultSet.getLong(1), is(6L));
62              assertFalse(statement.execute(String.format(sql, 1, 1, "init"), new String[]{"status"}));
63              generatedKeysResultSet = statement.getGeneratedKeys();
64              assertTrue(generatedKeysResultSet.next());
65              assertThat(generatedKeysResultSet.getLong(1), is(7L));
66          }
67      }

69      @Test
70      void assertAddGetGeneratedKeysForNoGeneratedValues() throws SQLException {
71          String sql = "INSERT INTO t_product (product_name) VALUES ('%s')";
72          try (Statement statement = getShardingSphereDataSource().getConnection().createStatement()) {
73              statement.execute(String.format(sql, "cup"), Statement.RETURN_GENERATED_KEYS);
74              ResultSet generatedKeysResultSet = statement.getGeneratedKeys();
75              assertTrue(generatedKeysResultSet.next());
76              assertThat(generatedKeysResultSet.getInt(1), is(1));
77          }
78      }

130     @Test
131     void assertExecuteBatch() throws SQLException {
132         try (Connection connection = getShardingSphereDataSource().getConnection(); Statement statement = connection.createStatement()) {
133             statement.addBatch("UPDATE t_order SET status = 'closed' WHERE order_id = 1001");
134             statement.addBatch("UPDATE t_order SET status = 'closed' WHERE order_id = 1100 OR order_id = 1101");
135             statement.addBatch("DELETE FROM t_order WHERE order_id = 1000");
136             assertThat(statement.executeBatch(), is(new int[]{1, 2, 1}));
137             statement.clearBatch();
138             assertThat(statement.executeBatch(), is(new int[0]));
139         }
