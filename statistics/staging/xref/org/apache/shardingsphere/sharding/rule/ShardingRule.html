122     public ShardingRule(final ShardingRuleConfiguration ruleConfig, final Collection<String> dataSourceNames, final InstanceContext instanceContext) {
123         configuration = ruleConfig;
124         this.dataSourceNames = getDataSourceNames(ruleConfig.getTables(), ruleConfig.getAutoTables(), dataSourceNames);
125         ruleConfig.getShardingAlgorithms().forEach((key, value) -> shardingAlgorithms.put(key, TypedSPILoader.getService(ShardingAlgorithm.class, value.getType(), value.getProps())));
126         ruleConfig.getKeyGenerators().forEach((key, value) -> keyGenerators.put(key, TypedSPILoader.getService(KeyGenerateAlgorithm.class, value.getType(), value.getProps())));
127         ruleConfig.getAuditors().forEach((key, value) -> auditors.put(key, TypedSPILoader.getService(ShardingAuditAlgorithm.class, value.getType(), value.getProps())));
128         tableRules.putAll(createTableRules(ruleConfig.getTables(), ruleConfig.getDefaultKeyGenerateStrategy()));
129         tableRules.putAll(createAutoTableRules(ruleConfig.getAutoTables(), ruleConfig.getDefaultKeyGenerateStrategy()));
130         bindingTableRules.putAll(createBindingTableRules(ruleConfig.getBindingTableGroups()));
131         defaultDatabaseShardingStrategyConfig = createDefaultDatabaseShardingStrategyConfig(ruleConfig);
132         defaultTableShardingStrategyConfig = createDefaultTableShardingStrategyConfig(ruleConfig);
133         defaultAuditStrategy = null == ruleConfig.getDefaultAuditStrategy() ? new ShardingAuditStrategyConfiguration(Collections.emptyList(), true) : ruleConfig.getDefaultAuditStrategy();
134         defaultKeyGenerateAlgorithm = null == ruleConfig.getDefaultKeyGenerateStrategy()
135                 ? TypedSPILoader.getService(KeyGenerateAlgorithm.class, null)
136                 : keyGenerators.get(ruleConfig.getDefaultKeyGenerateStrategy().getKeyGeneratorName());
137         defaultShardingColumn = ruleConfig.getDefaultShardingColumn();
138         shardingTableDataNodes = createShardingTableDataNodes(tableRules);
139         ShardingSpherePreconditions.checkState(isValidBindingTableConfiguration(tableRules, new BindingTableCheckedConfiguration(this.dataSourceNames, shardingAlgorithms,
140                 ruleConfig.getBindingTableGroups(), defaultDatabaseShardingStrategyConfig, defaultTableShardingStrategyConfig, defaultShardingColumn)),
141                 InvalidBindingTablesException::new);
142         keyGenerators.values().stream().filter(InstanceContextAware.class::isInstance).forEach(each -> ((InstanceContextAware) each).setInstanceContext(instanceContext));
143         if (defaultKeyGenerateAlgorithm instanceof InstanceContextAware && -1 == instanceContext.getWorkerId()) {
144             ((InstanceContextAware) defaultKeyGenerateAlgorithm).setInstanceContext(instanceContext);
145         }
146         shardingCache = null != ruleConfig.getShardingCache() ? new ShardingCache(ruleConfig.getShardingCache(), this) : null;
147         logicalTableMapper = createLogicalTableMapper();
148         actualTableMapper = createActualTableMapper();
149     }

151     private ShardingStrategyConfiguration createDefaultDatabaseShardingStrategyConfig(final ShardingRuleConfiguration ruleConfig) {
152         Optional.ofNullable(ruleConfig.getDefaultDatabaseShardingStrategy()).ifPresent(optional -> checkManualShardingAlgorithm(optional.getShardingAlgorithmName(), "default"));
153         return null == ruleConfig.getDefaultDatabaseShardingStrategy() ? new NoneShardingStrategyConfiguration() : ruleConfig.getDefaultDatabaseShardingStrategy();
154     }

156     private ShardingStrategyConfiguration createDefaultTableShardingStrategyConfig(final ShardingRuleConfiguration ruleConfig) {
157         Optional.ofNullable(ruleConfig.getDefaultTableShardingStrategy()).ifPresent(optional -> checkManualShardingAlgorithm(optional.getShardingAlgorithmName(), "default"));
158         return null == ruleConfig.getDefaultTableShardingStrategy() ? new NoneShardingStrategyConfiguration() : ruleConfig.getDefaultTableShardingStrategy();
159     }

185     private Collection<String> getDataSourceNames(final Collection<ShardingTableRuleConfiguration> tableRuleConfigs,
186                                                   final Collection<ShardingAutoTableRuleConfiguration> autoTableRuleConfigs, final Collection<String> dataSourceNames) {
187         if (tableRuleConfigs.isEmpty() && autoTableRuleConfigs.isEmpty()) {
188             return dataSourceNames;
189         }
190         if (tableRuleConfigs.stream().map(ShardingTableRuleConfiguration::getActualDataNodes).anyMatch(each -> null == each || each.isEmpty())) {
191             return dataSourceNames;
192         }
193         Collection<String> result = new LinkedHashSet<>();
194         tableRuleConfigs.forEach(each -> result.addAll(getDataSourceNames(each)));
195         autoTableRuleConfigs.forEach(each -> result.addAll(getDataSourceNames(each)));
196         return result;
197     }

199     private Collection<String> getDataSourceNames(final ShardingAutoTableRuleConfiguration shardingAutoTableRuleConfig) {
200         List<String> actualDataSources = InlineExpressionParserFactory.newInstance().splitAndEvaluate(shardingAutoTableRuleConfig.getActualDataSources());
201         return new HashSet<>(actualDataSources);
202     }

204     private Collection<String> getDataSourceNames(final ShardingTableRuleConfiguration shardingTableRuleConfig) {
205         List<String> actualDataNodes = InlineExpressionParserFactory.newInstance().splitAndEvaluate(shardingTableRuleConfig.getActualDataNodes());
206         return actualDataNodes.stream().map(each -> new DataNode(each).getDataSourceName()).collect(Collectors.toList());
207     }

209     private Map<String, TableRule> createTableRules(final Collection<ShardingTableRuleConfiguration> tableRuleConfigs, final KeyGenerateStrategyConfiguration defaultKeyGenerateStrategyConfig) {
210         return tableRuleConfigs.stream().map(each -> createTableRule(each, defaultKeyGenerateStrategyConfig))
211                 .collect(Collectors.toMap(each -> each.getLogicTable().toLowerCase(), Function.identity(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new));
212     }
213     
214     private TableRule createTableRule(final ShardingTableRuleConfiguration tableRuleConfig, final KeyGenerateStrategyConfiguration defaultKeyGenerateStrategyConfig) {
215         Optional.ofNullable(tableRuleConfig.getDatabaseShardingStrategy()).ifPresent(optional -> checkManualShardingAlgorithm(optional.getShardingAlgorithmName(), tableRuleConfig.getLogicTable()));
216         Optional.ofNullable(tableRuleConfig.getTableShardingStrategy()).ifPresent(optional -> checkManualShardingAlgorithm(optional.getShardingAlgorithmName(), tableRuleConfig.getLogicTable()));
217         return new TableRule(tableRuleConfig, dataSourceNames, getDefaultGenerateKeyColumn(defaultKeyGenerateStrategyConfig));
218     }
219     
220     private void checkManualShardingAlgorithm(final String shardingAlgorithmName, final String logicTable) {
221         ShardingAlgorithm shardingAlgorithm = shardingAlgorithms.get(shardingAlgorithmName);
222         ShardingSpherePreconditions.checkState(!(shardingAlgorithm instanceof ShardingAutoTableAlgorithm),
223                 () -> new ShardingAlgorithmInitializationException(shardingAlgorithm.getType(), String.format("`%s` tables sharding configuration can not use auto sharding algorithm.", logicTable)));
224     }
225     
226     private Map<String, TableRule> createAutoTableRules(final Collection<ShardingAutoTableRuleConfiguration> autoTableRuleConfigs,
227                                                         final KeyGenerateStrategyConfiguration defaultKeyGenerateStrategyConfig) {
228         return autoTableRuleConfigs.stream().map(each -> createAutoTableRule(defaultKeyGenerateStrategyConfig, each))
229                 .collect(Collectors.toMap(each -> each.getLogicTable().toLowerCase(), Function.identity(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new));
230     }
231     
232     private TableRule createAutoTableRule(final KeyGenerateStrategyConfiguration defaultKeyGenerateStrategyConfig, final ShardingAutoTableRuleConfiguration autoTableRuleConfig) {
233         checkAutoShardingAlgorithm(autoTableRuleConfig.getShardingStrategy().getShardingAlgorithmName(), autoTableRuleConfig.getLogicTable());
234         ShardingAlgorithm shardingAlgorithm = shardingAlgorithms.get(autoTableRuleConfig.getShardingStrategy().getShardingAlgorithmName());
235         return new TableRule(autoTableRuleConfig, dataSourceNames, (ShardingAutoTableAlgorithm) shardingAlgorithm, getDefaultGenerateKeyColumn(defaultKeyGenerateStrategyConfig));
236     }

238     private void checkAutoShardingAlgorithm(final String shardingAlgorithmName, final String logicTable) {
239         ShardingAlgorithm shardingAlgorithm = shardingAlgorithms.get(shardingAlgorithmName);
240         ShardingSpherePreconditions.checkState(shardingAlgorithm instanceof ShardingAutoTableAlgorithm,
241                 () -> new ShardingAlgorithmInitializationException(shardingAlgorithm.getType(), String.format("`%s` autoTables sharding configuration must use auto sharding algorithm.", logicTable)));
242     }

248     private Map<String, BindingTableRule> createBindingTableRules(final Collection<ShardingTableReferenceRuleConfiguration> bindingTableGroups) {
249         Map<String, BindingTableRule> result = new LinkedHashMap<>();
250         for (ShardingTableReferenceRuleConfiguration each : bindingTableGroups) {
251             BindingTableRule bindingTableRule = createBindingTableRule(each.getReference());
252             for (String logicTable : bindingTableRule.getAllLogicTables()) {
253                 result.put(logicTable.toLowerCase(), bindingTableRule);
254             }
255         }
256         return result;
257     }

259     private BindingTableRule createBindingTableRule(final String bindingTableGroup) {
260         Map<String, TableRule> tableRules = Splitter.on(",").trimResults().splitToList(bindingTableGroup).stream()
261                 .map(this::getTableRule).collect(Collectors.toMap(each -> each.getLogicTable().toLowerCase(), Function.identity(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new));
262         BindingTableRule result = new BindingTableRule();
263         result.getTableRules().putAll(tableRules);
264         return result;
265     }

267     private boolean isValidBindingTableConfiguration(final Map<String, TableRule> tableRules, final BindingTableCheckedConfiguration checkedConfig) {
268         for (ShardingTableReferenceRuleConfiguration each : checkedConfig.getBindingTableGroups()) {
269             Collection<String> bindingTables = Splitter.on(",").trimResults().splitToList(each.getReference().toLowerCase());
270             if (bindingTables.size() <= 1) {
271                 continue;
272             }
273             Iterator<String> iterator = bindingTables.iterator();
274             TableRule sampleTableRule = getTableRule(iterator.next(), tableRules);
275             while (iterator.hasNext()) {
276                 TableRule tableRule = getTableRule(iterator.next(), tableRules);
277                 if (!isValidActualDataSourceName(sampleTableRule, tableRule) || !isValidActualTableName(sampleTableRule, tableRule)) {
278                     return false;
279                 }
280                 if (!isBindingShardingAlgorithm(sampleTableRule, tableRule, true, checkedConfig) || !isBindingShardingAlgorithm(sampleTableRule, tableRule, false, checkedConfig)) {
281                     return false;
282                 }
283             }
284         }
285         return true;
286     }

288     private boolean isValidActualDataSourceName(final TableRule sampleTableRule, final TableRule tableRule) {
289         return sampleTableRule.getActualDataSourceNames().equals(tableRule.getActualDataSourceNames());
290     }

292     private boolean isValidActualTableName(final TableRule sampleTableRule, final TableRule tableRule) {
293         for (String each : sampleTableRule.getActualDataSourceNames()) {
294             Collection<String> sampleActualTableNames =
295                     sampleTableRule.getActualTableNames(each).stream().map(actualTableName -> actualTableName.replace(sampleTableRule.getTableDataNode().getPrefix(), "")).collect(Collectors.toSet());
296             Collection<String> actualTableNames =
297                     tableRule.getActualTableNames(each).stream().map(optional -> optional.replace(tableRule.getTableDataNode().getPrefix(), "")).collect(Collectors.toSet());
298             if (!sampleActualTableNames.equals(actualTableNames)) {
299                 return false;
300             }
301         }
302         return true;
303     }

305     private boolean isBindingShardingAlgorithm(final TableRule sampleTableRule, final TableRule tableRule, final boolean databaseAlgorithm, final BindingTableCheckedConfiguration checkedConfig) {
306         return getAlgorithmExpression(sampleTableRule, databaseAlgorithm, checkedConfig).equals(getAlgorithmExpression(tableRule, databaseAlgorithm, checkedConfig));
307     }
308     
309     private Optional<String> getAlgorithmExpression(final TableRule tableRule, final boolean databaseAlgorithm, final BindingTableCheckedConfiguration checkedConfig) {
310         ShardingStrategyConfiguration shardingStrategyConfig = databaseAlgorithm
311                 ? getDatabaseShardingStrategyConfiguration(tableRule, checkedConfig.getDefaultDatabaseShardingStrategyConfig())
312                 : getTableShardingStrategyConfiguration(tableRule, checkedConfig.getDefaultTableShardingStrategyConfig());
313         ShardingAlgorithm shardingAlgorithm = checkedConfig.getShardingAlgorithms().get(shardingStrategyConfig.getShardingAlgorithmName());
314         String dataNodePrefix = databaseAlgorithm ? tableRule.getDataSourceDataNode().getPrefix() : tableRule.getTableDataNode().getPrefix();
315         String shardingColumn = getShardingColumn(shardingStrategyConfig, checkedConfig.getDefaultShardingColumn());
316         return null == shardingAlgorithm ? Optional.empty() : shardingAlgorithm.getAlgorithmStructure(dataNodePrefix, shardingColumn);
317     }

319     private String getShardingColumn(final ShardingStrategyConfiguration shardingStrategyConfig, final String defaultShardingColumn) {
320         String shardingColumn = defaultShardingColumn;
321         if (shardingStrategyConfig instanceof ComplexShardingStrategyConfiguration) {
322             shardingColumn = ((ComplexShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumns();
323         }
324         if (shardingStrategyConfig instanceof StandardShardingStrategyConfiguration) {
325             shardingColumn = ((StandardShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumn();
326         }
327         return null == shardingColumn ? "" : shardingColumn;
328     }

439     /**
440      * Judge whether logic table is all binding tables or not.
441      *
442      * @param logicTableNames logic table names
443      * @return whether logic table is all binding tables or not
444      */
445     public boolean isAllBindingTables(final Collection<String> logicTableNames) {
446         if (logicTableNames.isEmpty()) {
447             return false;
448         }
449         Optional<BindingTableRule> bindingTableRule = findBindingTableRule(logicTableNames);
450         if (!bindingTableRule.isPresent()) {
451             return false;
452         }
453         Collection<String> result = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
454         result.addAll(bindingTableRule.get().getAllLogicTables());
455         return !result.isEmpty() && result.containsAll(logicTableNames);
456     }

458     /**
459      * Judge whether logic table is all binding tables.
460      *
461      * @param database database
462      * @param sqlStatementContext sqlStatementContext
463      * @param logicTableNames logic table names
464      * @return whether logic table is all binding tables
465      */
466     public boolean isAllBindingTables(final ShardingSphereDatabase database, final SQLStatementContext sqlStatementContext, final Collection<String> logicTableNames) {
467         if (!(sqlStatementContext instanceof SelectStatementContext && ((SelectStatementContext) sqlStatementContext).isContainsJoinQuery())) {
468             return isAllBindingTables(logicTableNames);
469         }
470         if (!isAllBindingTables(logicTableNames)) {
471             return false;
472         }
473         String defaultSchemaName = new DatabaseTypeRegistry(sqlStatementContext.getDatabaseType()).getDefaultSchemaName(database.getName());
474         ShardingSphereSchema schema = sqlStatementContext.getTablesContext().getSchemaName().map(database::getSchema).orElseGet(() -> database.getSchema(defaultSchemaName));
475         SelectStatementContext select = (SelectStatementContext) sqlStatementContext;
476         return isJoinConditionContainsShardingColumns(schema, select, logicTableNames, select.getWhereSegments());
477     }

479     private Optional<BindingTableRule> findBindingTableRule(final Collection<String> logicTableNames) {
480         for (String each : logicTableNames) {
481             Optional<BindingTableRule> result = findBindingTableRule(each);
482             if (result.isPresent()) {
483                 return result;
484             }
485         }
486         return Optional.empty();
487     }

565     private Optional<String> findShardingColumn(final TableRule tableRule, final String columnName) {
566         Optional<String> databaseShardingColumn = findShardingColumn(getDatabaseShardingStrategyConfiguration(tableRule), columnName);
567         if (databaseShardingColumn.isPresent()) {
568             return databaseShardingColumn;
569         }
570         return findShardingColumn(getTableShardingStrategyConfiguration(tableRule), columnName);
571     }
572     
573     private Optional<String> findShardingColumn(final ShardingStrategyConfiguration shardingStrategyConfig, final String columnName) {
574         if (shardingStrategyConfig instanceof StandardShardingStrategyConfiguration) {
575             String shardingColumn = null == ((StandardShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumn()
576                     ? defaultShardingColumn
577                     : ((StandardShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumn();
578             return shardingColumn.equalsIgnoreCase(columnName) ? Optional.of(shardingColumn) : Optional.empty();
579         }
580         if (shardingStrategyConfig instanceof ComplexShardingStrategyConfiguration) {
581             List<String> shardingColumns = Splitter.on(",").trimResults().splitToList(((ComplexShardingStrategyConfiguration) shardingStrategyConfig).getShardingColumns());
582             for (String each : shardingColumns) {
583                 if (each.equalsIgnoreCase(columnName)) {
584                     return Optional.of(each);
585                 }
586             }
587         }
588         return Optional.empty();
589     }

729     private boolean isJoinConditionContainsShardingColumns(final ShardingSphereSchema schema, final SelectStatementContext select,
730                                                            final Collection<String> tableNames, final Collection<WhereSegment> whereSegments) {
731         Collection<String> databaseJoinConditionTables = new HashSet<>(tableNames.size(), 1F);
732         Collection<String> tableJoinConditionTables = new HashSet<>(tableNames.size(), 1F);
733         for (WhereSegment each : whereSegments) {
734             Collection<AndPredicate> andPredicates = ExpressionExtractUtils.getAndPredicates(each.getExpr());
735             if (andPredicates.size() > 1) {
736                 return false;
737             }
738             for (AndPredicate andPredicate : andPredicates) {
739                 databaseJoinConditionTables.addAll(getJoinConditionTables(schema, select, andPredicate.getPredicates(), true));
740                 tableJoinConditionTables.addAll(getJoinConditionTables(schema, select, andPredicate.getPredicates(), false));
741             }
742         }
743         TableRule tableRule = getTableRule(tableNames.iterator().next());
744         boolean containsDatabaseShardingColumns = !(getDatabaseShardingStrategyConfiguration(tableRule) instanceof StandardShardingStrategyConfiguration)
745                 || databaseJoinConditionTables.containsAll(tableNames);
746         boolean containsTableShardingColumns = !(getTableShardingStrategyConfiguration(tableRule) instanceof StandardShardingStrategyConfiguration) || tableJoinConditionTables.containsAll(tableNames);
747         return containsDatabaseShardingColumns && containsTableShardingColumns;
748     }
749     
750     private Collection<String> getJoinConditionTables(final ShardingSphereSchema schema, final SelectStatementContext select,
751                                                       final Collection<ExpressionSegment> predicates, final boolean isDatabaseJoinCondition) {
752         Collection<String> result = new LinkedList<>();
753         for (ExpressionSegment each : predicates) {
754             if (!isJoinConditionExpression(each)) {
755                 continue;
756             }
757             ColumnSegment leftColumn = (ColumnSegment) ((BinaryOperationExpression) each).getLeft();
758             ColumnSegment rightColumn = (ColumnSegment) ((BinaryOperationExpression) each).getRight();
759             Map<String, String> columnExpressionTableNames = select.getTablesContext().findTableNamesByColumnSegment(Arrays.asList(leftColumn, rightColumn), schema);
760             Optional<TableRule> leftTableRule = findTableRule(columnExpressionTableNames.get(leftColumn.getExpression()));
761             Optional<TableRule> rightTableRule = findTableRule(columnExpressionTableNames.get(rightColumn.getExpression()));
762             if (!leftTableRule.isPresent() || !rightTableRule.isPresent()) {
763                 continue;
764             }
765             ShardingStrategyConfiguration leftConfig = isDatabaseJoinCondition
766                     ? getDatabaseShardingStrategyConfiguration(leftTableRule.get())
767                     : getTableShardingStrategyConfiguration(leftTableRule.get());
768             ShardingStrategyConfiguration rightConfig = isDatabaseJoinCondition
769                     ? getDatabaseShardingStrategyConfiguration(rightTableRule.get())
770                     : getTableShardingStrategyConfiguration(rightTableRule.get());
771             if (findShardingColumn(leftConfig, leftColumn.getIdentifier().getValue()).isPresent()
772                     && findShardingColumn(rightConfig, rightColumn.getIdentifier().getValue()).isPresent()) {
773                 result.add(columnExpressionTableNames.get(leftColumn.getExpression()));
774                 result.add(columnExpressionTableNames.get(rightColumn.getExpression()));
775             }
776         }
777         return result;
778     }
779     
780     private boolean isJoinConditionExpression(final ExpressionSegment expression) {
781         if (!(expression instanceof BinaryOperationExpression)) {
782             return false;
783         }
784         BinaryOperationExpression binaryExpression = (BinaryOperationExpression) expression;
785         return binaryExpression.getLeft() instanceof ColumnSegment && binaryExpression.getRight() instanceof ColumnSegment && "=".equals(binaryExpression.getOperator());
786     }
