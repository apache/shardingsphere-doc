47  @ExtendWith(MockitoExtension.class)
48  class DeleteStatementContextTest {
49      
50      @Mock
51      private WhereSegment whereSegment;
52      
53      @Test
54      void assertMySQLNewInstance() {
55          assertNewInstance(new MySQLDeleteStatement());
56      }
57      
58      @Test
59      void assertOracleNewInstance() {
60          assertNewInstance(new OracleDeleteStatement());
61      }
62      
63      @Test
64      void assertPostgreSQLNewInstance() {
65          assertNewInstance(new PostgreSQLDeleteStatement());
66      }
67      
68      @Test
69      void assertSQL92NewInstance() {
70          assertNewInstance(new SQL92DeleteStatement());
71      }
72      
73      @Test
74      void assertSQLServerNewInstance() {
75          assertNewInstance(new SQLServerDeleteStatement());
76      }
77      
78      private void assertNewInstance(final DeleteStatement deleteStatement) {
79          when(whereSegment.getExpr()).thenReturn(mock(ExpressionSegment.class));
80          SimpleTableSegment table1 = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("tbl_1")));
81          SimpleTableSegment table2 = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("tbl_2")));
82          JoinTableSegment tableSegment = new JoinTableSegment();
83          tableSegment.setLeft(table1);
84          tableSegment.setRight(table2);
85          deleteStatement.setWhere(whereSegment);
86          deleteStatement.setTable(tableSegment);
87          DeleteStatementContext actual = new DeleteStatementContext(deleteStatement);
88          assertThat(actual.getTablesContext().getTableNames(), is(new HashSet<>(Arrays.asList("tbl_1", "tbl_2"))));
89          assertThat(actual.getWhereSegments(), is(Collections.singletonList(whereSegment)));
90          assertThat(actual.getAllTables().stream().map(each -> each.getTableName().getIdentifier().getValue()).collect(Collectors.toList()), is(Arrays.asList("tbl_1", "tbl_2")));
91      }
92  }
