18  package org.apache.shardingsphere.infra.database.mysql.checker;
19  
20  import org.apache.shardingsphere.infra.database.core.checker.DialectDatabaseEnvironmentChecker;
21  import org.apache.shardingsphere.infra.database.core.checker.PrivilegeCheckType;
22  import org.apache.shardingsphere.infra.database.core.exception.CheckDatabaseEnvironmentFailedException;
23  import org.apache.shardingsphere.infra.database.core.exception.MissingRequiredPrivilegeException;
24  import org.apache.shardingsphere.infra.database.core.exception.UnexpectedVariableValueException;
25  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
26  
27  import javax.sql.DataSource;
28  import java.sql.Connection;
29  import java.sql.PreparedStatement;
30  import java.sql.ResultSet;
31  import java.sql.SQLException;
32  import java.util.Arrays;
33  import java.util.Collection;
34  import java.util.Collections;
35  import java.util.HashMap;
36  import java.util.Map;
37  import java.util.Map.Entry;
38  import java.util.stream.Collectors;
39  
40  /**
41   * Database environment checker for MySQL.
42   */
43  public final class MySQLDatabaseEnvironmentChecker implements DialectDatabaseEnvironmentChecker {
44      
45      private static final String SHOW_GRANTS_SQL = "SHOW GRANTS";
46      
47      private static final int MYSQL_MAJOR_VERSION_8 = 8;
48      
49      // BINLOG MONITOR is a synonym for REPLICATION CLIENT for MariaDB
50      private static final String[][] PIPELINE_REQUIRED_PRIVILEGES =
51              {{"ALL PRIVILEGES", "ON *.*"}, {"REPLICATION SLAVE", "REPLICATION CLIENT", "ON *.*"}, {"REPLICATION SLAVE", "BINLOG MONITOR", "ON *.*"}};
52      
53      private static final String[][] XA_REQUIRED_PRIVILEGES = {{"ALL PRIVILEGES", "ON *.*"}, {"XA_RECOVER_ADMIN", "ON *.*"}};
54      
55      private static final Map<PrivilegeCheckType, String[][]> REQUIRED_PRIVILEGES = new HashMap<>(2, 1F);
56      
57      private static final Map<PrivilegeCheckType, Collection<String>> REQUIRED_PRIVILEGES_FOR_MESSAGE = new HashMap<>(2, 1F);
58      
59      private static final Map<String, String> REQUIRED_VARIABLES = new HashMap<>(3, 1F);
60      
61      private static final String SHOW_VARIABLES_SQL;
62      
63      static {
64          REQUIRED_PRIVILEGES.put(PrivilegeCheckType.PIPELINE, PIPELINE_REQUIRED_PRIVILEGES);
65          REQUIRED_PRIVILEGES_FOR_MESSAGE.put(PrivilegeCheckType.PIPELINE, Arrays.asList("REPLICATION SLAVE", "REPLICATION CLIENT"));
66          REQUIRED_PRIVILEGES.put(PrivilegeCheckType.XA, XA_REQUIRED_PRIVILEGES);
67          REQUIRED_PRIVILEGES_FOR_MESSAGE.put(PrivilegeCheckType.XA, Collections.singleton("XA_RECOVER_ADMIN"));
68          REQUIRED_VARIABLES.put("LOG_BIN", "ON");
69          REQUIRED_VARIABLES.put("BINLOG_FORMAT", "ROW");
70          // It does not exist in all versions of MySQL
71          REQUIRED_VARIABLES.put("BINLOG_ROW_IMAGE", "FULL");
72          SHOW_VARIABLES_SQL = String.format("SHOW VARIABLES WHERE Variable_name IN (%s)", REQUIRED_VARIABLES.keySet().stream().map(each -> "?").collect(Collectors.joining(",")));
73      }
74      
75      @Override
76      public void checkPrivilege(final DataSource dataSource, final PrivilegeCheckType privilegeCheckType) {
77          try (Connection connection = dataSource.getConnection()) {
78              if (PrivilegeCheckType.XA == privilegeCheckType && MYSQL_MAJOR_VERSION_8 != connection.getMetaData().getDatabaseMajorVersion()) {
79                  return;
80              }
81              checkPrivilege(connection, privilegeCheckType);
82          } catch (final SQLException ex) {
83              throw new CheckDatabaseEnvironmentFailedException(ex);
84          }
85      }
86      
87      private void checkPrivilege(final Connection connection, final PrivilegeCheckType privilegeCheckType) {
88          try (
89                  PreparedStatement preparedStatement = connection.prepareStatement(SHOW_GRANTS_SQL);
90                  ResultSet resultSet = preparedStatement.executeQuery()) {
91              while (resultSet.next()) {
92                  String privilege = resultSet.getString(1).toUpperCase();
93                  if (matchPrivileges(privilege, REQUIRED_PRIVILEGES.get(privilegeCheckType))) {
94                      return;
95                  }
96              }
97          } catch (final SQLException ex) {
98              throw new CheckDatabaseEnvironmentFailedException(ex);
99          }
100         throw new MissingRequiredPrivilegeException(REQUIRED_PRIVILEGES_FOR_MESSAGE.get(privilegeCheckType));
101     }
102     
103     private boolean matchPrivileges(final String privilege, final String[][] requiredPrivileges) {
104         return Arrays.stream(requiredPrivileges).anyMatch(each -> Arrays.stream(each).allMatch(privilege::contains));
105     }
106     
107     @Override
108     public void checkVariable(final DataSource dataSource) {
109         try (
110                 Connection connection = dataSource.getConnection();
111                 PreparedStatement preparedStatement = connection.prepareStatement(SHOW_VARIABLES_SQL)) {
112             int parameterIndex = 1;
113             for (Entry<String, String> entry : REQUIRED_VARIABLES.entrySet()) {
114                 preparedStatement.setString(parameterIndex++, entry.getKey());
115             }
116             try (ResultSet resultSet = preparedStatement.executeQuery()) {
117                 while (resultSet.next()) {
118                     String variableName = resultSet.getString(1).toUpperCase();
119                     String expectedValue = REQUIRED_VARIABLES.get(variableName);
120                     String actualValue = resultSet.getString(2);
121                     ShardingSpherePreconditions.checkState(expectedValue.equalsIgnoreCase(actualValue),
122                             () -> new UnexpectedVariableValueException(variableName, expectedValue, actualValue));
123                 }
124             }
125         } catch (final SQLException ex) {
126             throw new CheckDatabaseEnvironmentFailedException(ex);
127         }
128     }
129     
130     @Override
131     public String getDatabaseType() {
132         return "MySQL";
133     }
134 }
