49  class PostgreSQLAdminExecutorCreatorTest {
50      
51      private static final String PSQL_SELECT_DATABASES = "SELECT d.datname as \"Name\",\n"
52              + "       pg_catalog.pg_get_userbyid(d.datdba) as \"Owner\",\n"
53              + "       pg_catalog.pg_encoding_to_char(d.encoding) as \"Encoding\",\n"
54              + "       d.datcollate as \"Collate\",\n"
55              + "       d.datctype as \"Ctype\",\n"
56              + "       pg_catalog.array_to_string(d.datacl, E'\\n') AS \"Access privileges\"\n"
57              + "FROM pg_catalog.pg_database d\n"
58              + "ORDER BY 1";
59      
60      private static final String PSQL_SELECT_TABLESPACES = "SELECT spcname AS \"Name\",\n"
61              + "  pg_catalog.pg_get_userbyid(spcowner) AS \"Owner\",\n"
62              + "  pg_catalog.pg_tablespace_location(oid) AS \"Location\"\n"
63              + "FROM pg_catalog.pg_tablespace\n"
64              + "ORDER BY 1";
65      
66      private static final String SELECT_PG_CATALOG_WITH_SUBQUERY = "select * from (select * from pg_catalog.pg_namespace) t;";
67      
68      private static final String SELECT_PG_CLASS_AND_PG_NAMESPACE = "SELECT n.nspname as \"Schema\",\n"
69              + "       c.relname as \"Name\",\n"
70              + "CASE c.relkind WHEN 'r' THEN 'table' WHEN 'v' THEN 'view' WHEN 'm' THEN 'materialized view' WHEN 'i' THEN 'index' WHEN 'S' THEN 'sequence' WHEN 's' THEN 'special' "
71              + "WHEN 'f' THEN 'foreign table' WHEN 'p' THEN 'partitioned table' WHEN 'I' THEN 'partitioned index' END as \"Type\",\n"
72              + "pg_catalog.pg_get_userbyid(c.relowner) as \"Owner\"\n"
73              + "FROM pg_catalog.pg_class c\n"
74              + "       LEFT JOIN pg_catalog.pg_namespace n ON n.oid = c.relnamespace\n"
75              + "WHERE c.relkind IN ('r','p','v','m','S','f','')\n"
76              + "       AND n.nspname <> 'pg_catalog'\n"
77              + "       AND n.nspname <> 'information_schema'\n"
78              + "       AND n.nspname !~ '^pg_toast'\n"
79              + "       AND pg_catalog.pg_table_is_visible(c.oid)\n"
80              + "ORDER BY 1,2;";
81      
82      @Test
83      void assertCreateWithOtherSQLStatementContextOnly() {
84          assertThat(new PostgreSQLAdminExecutorCreator().create(new CommonSQLStatementContext<>(new PostgreSQLInsertStatement())), is(Optional.empty()));
85      }
86      
87      @Test
88      void assertCreateWithShowSQLStatement() {
89          Optional<DatabaseAdminExecutor> actual = new PostgreSQLAdminExecutorCreator().create(new CommonSQLStatementContext<>(new PostgreSQLShowStatement("client_encoding")));
90          assertTrue(actual.isPresent());
91          assertThat(actual.get(), instanceOf(PostgreSQLShowVariableExecutor.class));
92      }
93      
94      @Test
95      void assertCreateWithSelectDatabase() {
96          SQLStatement sqlStatement = parseSQL(PSQL_SELECT_DATABASES);
97          SelectStatementContext selectStatementContext = mock(SelectStatementContext.class);
98          when(selectStatementContext.getSqlStatement()).thenReturn((SelectStatement) sqlStatement);
99          Optional<DatabaseAdminExecutor> actual = new PostgreSQLAdminExecutorCreator().create(selectStatementContext, PSQL_SELECT_DATABASES, "");
100         assertTrue(actual.isPresent());
101         assertThat(actual.get(), instanceOf(SelectDatabaseExecutor.class));
102     }
103     
104     @Test
105     void assertCreateWithSelectTablespace() {
106         SQLStatement sqlStatement = parseSQL(PSQL_SELECT_TABLESPACES);
107         SelectStatementContext selectStatementContext = mock(SelectStatementContext.class);
108         when(selectStatementContext.getSqlStatement()).thenReturn((SelectStatement) sqlStatement);
109         Optional<DatabaseAdminExecutor> actual = new PostgreSQLAdminExecutorCreator().create(selectStatementContext, PSQL_SELECT_TABLESPACES, "");
110         assertTrue(actual.isPresent());
111         assertThat(actual.get(), instanceOf(SelectTableExecutor.class));
112     }
113     
114     @Test
115     void assertCreateWithSelectPgCatalogWithSubquery() {
116         SQLStatement sqlStatement = parseSQL(SELECT_PG_CATALOG_WITH_SUBQUERY);
117         SelectStatementContext selectStatementContext = mock(SelectStatementContext.class);
118         when(selectStatementContext.getSqlStatement()).thenReturn((SelectStatement) sqlStatement);
119         Optional<DatabaseAdminExecutor> actual = new PostgreSQLAdminExecutorCreator().create(selectStatementContext, SELECT_PG_CATALOG_WITH_SUBQUERY, "");
120         assertFalse(actual.isPresent());
121     }
122     
123     @Test
124     void assertCreateWithSelectPgNamespaceAndPgClass() {
125         SQLStatement sqlStatement = parseSQL(SELECT_PG_CLASS_AND_PG_NAMESPACE);
126         SelectStatementContext selectStatementContext = mock(SelectStatementContext.class);
127         when(selectStatementContext.getSqlStatement()).thenReturn((SelectStatement) sqlStatement);
128         Optional<DatabaseAdminExecutor> actual = new PostgreSQLAdminExecutorCreator().create(selectStatementContext, SELECT_PG_CLASS_AND_PG_NAMESPACE, "");
129         assertFalse(actual.isPresent());
130     }
131     
132     private static SQLStatement parseSQL(final String sql) {
133         CacheOption cacheOption = new CacheOption(0, 0);
134         SQLParserRule sqlParserRule = new SQLParserRule(new SQLParserRuleConfiguration(false, cacheOption, cacheOption));
135         return sqlParserRule.getSQLParserEngine("PostgreSQL").parse(sql, false);
136     }
137     
138     @Test
139     void assertCreateWithSelectNonPgCatalog() {
140         SelectStatementContext selectStatementContext = mock(SelectStatementContext.class);
141         when(selectStatementContext.getSqlStatement()).thenReturn(new PostgreSQLSelectStatement());
142         assertThat(new PostgreSQLAdminExecutorCreator().create(selectStatementContext, "select 1", ""), is(Optional.empty()));
143     }
144     
145     @Test
146     void assertCreateWithSetStatement() {
147         PostgreSQLSetStatement setStatement = new PostgreSQLSetStatement();
148         CommonSQLStatementContext<PostgreSQLSetStatement> sqlStatementContext = new CommonSQLStatementContext<>(setStatement);
149         Optional<DatabaseAdminExecutor> actual = new PostgreSQLAdminExecutorCreator().create(sqlStatementContext, "SET client_encoding = utf8", "");
150         assertTrue(actual.isPresent());
151         assertThat(actual.get(), instanceOf(PostgreSQLSetVariableAdminExecutor.class));
152     }
153     
154     @Test
155     void assertCreateWithResetStatement() {
156         Optional<DatabaseAdminExecutor> actual = new PostgreSQLAdminExecutorCreator()
157                 .create(new CommonSQLStatementContext<>(new PostgreSQLResetParameterStatement("client_encoding")), "RESET client_encoding", "");
158         assertTrue(actual.isPresent());
159         assertThat(actual.get(), instanceOf(PostgreSQLResetVariableAdminExecutor.class));
160     }
161     
162     @Test
163     void assertCreateWithDMLStatement() {
164         DeleteStatementContext sqlStatementContext = new DeleteStatementContext(new PostgreSQLDeleteStatement());
165         assertThat(new PostgreSQLAdminExecutorCreator().create(sqlStatementContext, "delete from t where id = 1", ""), is(Optional.empty()));
166     }

170         assertThat(new PostgreSQLAdminExecutorCreator().getType(), is("PostgreSQL"));
171     }
