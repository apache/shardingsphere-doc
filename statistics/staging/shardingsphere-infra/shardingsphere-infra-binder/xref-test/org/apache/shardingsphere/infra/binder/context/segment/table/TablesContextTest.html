58      @Test
59      void assertInstanceCreatedWhenNoExceptionThrown() {
60          SimpleTableSegment tableSegment = new SimpleTableSegment(new TableNameSegment(0, 10, new IdentifierValue("tbl")));
61          tableSegment.setOwner(new OwnerSegment(0, 0, new IdentifierValue("schema")));
62          TablesContext tablesContext = new TablesContext(Collections.singleton(tableSegment), TypedSPILoader.getService(DatabaseType.class, "FIXTURE"));
63          assertThat(tablesContext.getDatabaseName(), is(Optional.of("schema")));
64          assertThat(tablesContext.getSchemaName(), is(Optional.of("schema")));
65          assertThat(tablesContext.getTableNames(), is(Collections.singleton("tbl")));
66          assertTrue(tablesContext.getSubqueryTables().isEmpty());
67      }

100     @Test
101     void assertFindTableNameWhenColumnSegmentOwnerAbsentAndSchemaMetaDataContainsColumn() {
102         SimpleTableSegment tableSegment1 = createTableSegment("table_1", "tbl_1");
103         SimpleTableSegment tableSegment2 = createTableSegment("table_2", "tbl_2");
104         ShardingSphereSchema schema = mock(ShardingSphereSchema.class);
105         when(schema.getAllColumnNames("table_1")).thenReturn(Collections.singletonList("col"));
106         ColumnSegment columnSegment = createColumnSegment(null, "col");
107         Map<String, String> actual = new TablesContext(Arrays.asList(tableSegment1, tableSegment2),
108                 TypedSPILoader.getService(DatabaseType.class, "FIXTURE")).findTableNamesByColumnSegment(Collections.singletonList(columnSegment), schema);
109         assertFalse(actual.isEmpty());
110         assertThat(actual.get("col"), is("table_1"));
111     }

113     @Test
114     void assertFindTableNameWhenColumnSegmentOwnerAbsentAndSchemaMetaDataContainsColumnInUpperCase() {
115         SimpleTableSegment tableSegment1 = createTableSegment("TABLE_1", "TBL_1");
116         SimpleTableSegment tableSegment2 = createTableSegment("TABLE_2", "TBL_2");
117         ShardingSphereTable table = new ShardingSphereTable("TABLE_1",
118                 Collections.singletonList(new ShardingSphereColumn("COL", 0, false, false, true, true, false, false)), Collections.emptyList(), Collections.emptyList());
119         ShardingSphereSchema schema = new ShardingSphereSchema(Stream.of(table).collect(Collectors.toMap(ShardingSphereTable::getName, value -> value)), Collections.emptyMap());
120         ColumnSegment columnSegment = createColumnSegment(null, "COL");
121         Map<String, String> actual = new TablesContext(Arrays.asList(tableSegment1, tableSegment2),
122                 TypedSPILoader.getService(DatabaseType.class, "FIXTURE")).findTableNamesByColumnSegment(Collections.singletonList(columnSegment), schema);
123         assertFalse(actual.isEmpty());
124         assertThat(actual.get("col"), is("TABLE_1"));
125     }

127     private SimpleTableSegment createTableSegment(final String tableName, final String alias) {
128         SimpleTableSegment result = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue(tableName)));
129         AliasSegment aliasSegment = new AliasSegment(0, 0, new IdentifierValue(alias));
130         result.setAlias(aliasSegment);
131         return result;
132     }
133     
134     private ColumnSegment createColumnSegment(final String owner, final String name) {
135         ColumnSegment result = new ColumnSegment(0, 0, new IdentifierValue(name));
136         if (null != owner) {
137             result.setOwner(new OwnerSegment(0, 0, new IdentifierValue(owner)));
138         }
139         return result;
140     }
