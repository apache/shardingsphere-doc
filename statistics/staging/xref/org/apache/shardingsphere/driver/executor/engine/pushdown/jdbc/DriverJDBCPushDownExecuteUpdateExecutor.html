54  /**
55   * Driver JDBC push down execute update executor.
56   */
57  public final class DriverJDBCPushDownExecuteUpdateExecutor {
58      
59      private final ShardingSphereConnection connection;
60      
61      private final String processId;
62      
63      private final ConfigurationProperties props;
64      
65      private final JDBCExecutor jdbcExecutor;
66      
67      public DriverJDBCPushDownExecuteUpdateExecutor(final ShardingSphereConnection connection, final ShardingSphereMetaData metaData, final JDBCExecutor jdbcExecutor) {
68          this.connection = connection;
69          processId = connection.getProcessId();
70          props = metaData.getProps();
71          this.jdbcExecutor = jdbcExecutor;
72      }
73      
74      /**
75       * Execute update.
76       *
77       * @param database database
78       * @param executionContext execution context
79       * @param prepareEngine prepare engine
80       * @param updateCallback statement execute update callback
81       * @param addCallback statement add callback
82       * @param replayCallback statement replay callback
83       * @return affected rows
84       * @throws SQLException SQL exception
85       */
86      @SuppressWarnings("rawtypes")
87      public int executeUpdate(final ShardingSphereDatabase database, final ExecutionContext executionContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
88                               final StatementExecuteUpdateCallback updateCallback, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
89          return new DriverTransactionalExecutor(connection).execute(
90                  database, executionContext, () -> doExecuteUpdate(database, executionContext, prepareEngine, updateCallback, addCallback, replayCallback));
91      }
92      
93      @SuppressWarnings({"rawtypes", "unchecked"})
94      private int doExecuteUpdate(final ShardingSphereDatabase database, final ExecutionContext executionContext, final DriverExecutionPrepareEngine<JDBCExecutionUnit, Connection> prepareEngine,
95                                  final StatementExecuteUpdateCallback updateCallback, final StatementAddCallback addCallback, final StatementReplayCallback replayCallback) throws SQLException {
96          ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext = prepareEngine.prepare(database.getName(), executionContext.getRouteContext(), executionContext.getExecutionUnits(),
97                  new ExecutionGroupReportContext(processId, database.getName(), connection.getDatabaseConnectionManager().getConnectionContext().getGrantee()));
98          for (ExecutionGroup<JDBCExecutionUnit> each : executionGroupContext.getInputGroups()) {
99              addCallback.add(getStatements(each), JDBCDriverType.PREPARED_STATEMENT.equals(prepareEngine.getType()) ? getParameterSets(each) : Collections.emptyList());
100         }
101         replayCallback.replay();
102         ProcessEngine processEngine = new ProcessEngine();
103         try {
104             processEngine.executeSQL(executionGroupContext, executionContext.getQueryContext());
105             JDBCExecutorCallback<Integer> callback = new ExecuteUpdateCallbackFactory(prepareEngine.getType())
106                     .newInstance(database, executionContext.getQueryContext().getSqlStatementContext().getSqlStatement(), updateCallback);
107             List<Integer> updateCounts = jdbcExecutor.execute(executionGroupContext, callback);
108             if (MetaDataRefreshEngine.isRefreshMetaDataRequired(executionContext.getQueryContext().getSqlStatementContext())) {
109                 new MetaDataRefreshEngine(connection.getContextManager().getPersistServiceFacade().getMetaDataManagerPersistService(), database, props)
110                         .refresh(executionContext.getQueryContext().getSqlStatementContext(), executionContext.getRouteContext().getRouteUnits());
111             }
112             return isNeedAccumulate(database.getRuleMetaData().getRules(), executionContext.getQueryContext().getSqlStatementContext()) ? accumulate(updateCounts) : updateCounts.get(0);
113         } finally {
114             processEngine.completeSQLExecution(executionGroupContext.getReportContext().getProcessId());
115         }
116     }
117     
118     private Collection<Statement> getStatements(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
119         Collection<Statement> result = new LinkedList<>();
120         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
121             result.add(each.getStorageResource());
122         }
123         return result;
124     }
125     
126     private Collection<List<Object>> getParameterSets(final ExecutionGroup<JDBCExecutionUnit> executionGroup) {
127         Collection<List<Object>> result = new LinkedList<>();
128         for (JDBCExecutionUnit each : executionGroup.getInputs()) {
129             result.add(each.getExecutionUnit().getSqlUnit().getParameters());
130         }
131         return result;
132     }
133     
134     private boolean isNeedAccumulate(final Collection<ShardingSphereRule> rules, final SQLStatementContext sqlStatementContext) {
135         if (!(sqlStatementContext instanceof TableAvailable)) {
136             return false;
137         }
138         for (ShardingSphereRule each : rules) {
139             Optional<DataNodeRuleAttribute> ruleAttribute = each.getAttributes().findAttribute(DataNodeRuleAttribute.class);
140             if (ruleAttribute.isPresent() && ruleAttribute.get().isNeedAccumulate(((TableAvailable) sqlStatementContext).getTablesContext().getTableNames())) {
141                 return true;
142             }
143         }
144         return false;
145     }
146     
147     private int accumulate(final List<Integer> updateResults) {
148         int result = 0;
149         for (Integer each : updateResults) {
150             result += null == each ? 0 : each;
151         }
152         return result;
153     }
154 }
