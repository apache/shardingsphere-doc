49  /**
50   * SQL statement assert.
51   */
52  @NoArgsConstructor(access = AccessLevel.PRIVATE)
53  public final class SQLStatementAssert {
54      
55      /**
56       * Assert SQL statement is correct with expected parser result.
57       *
58       * @param assertContext assert context
59       * @param actual actual SQL statement
60       * @param expected expected parser result
61       */
62      public static void assertIs(final SQLCaseAssertContext assertContext, final SQLStatement actual, final SQLParserTestCase expected) {
63          if (actual instanceof OracleCreateProcedureStatement || actual instanceof OracleCreateFunctionStatement) {
64              PLSQLStatementAssert.assertIs(assertContext, actual, expected);
65              return;
66          }
67          ParameterMarkerAssert.assertCount(assertContext, actual.getParameterCount(), expected.getParameters().size());
68          CommentAssert.assertComment(assertContext, actual, expected);
69          if (actual instanceof DMLStatement) {
70              DMLStatementAssert.assertIs(assertContext, (DMLStatement) actual, expected);
71          } else if (actual instanceof DDLStatement) {
72              DDLStatementAssert.assertIs(assertContext, (DDLStatement) actual, expected);
73          } else if (actual instanceof TCLStatement) {
74              TCLStatementAssert.assertIs(assertContext, (TCLStatement) actual, expected);
75          } else if (actual instanceof DCLStatement) {
76              DCLStatementAssert.assertIs(assertContext, (DCLStatement) actual, expected);
77          } else if (actual instanceof DALStatement) {
78              DALStatementAssert.assertIs(assertContext, (DALStatement) actual, expected);
79          } else if (actual instanceof RDLStatement) {
80              RDLStatementAssert.assertIs(assertContext, (RDLStatement) actual, expected);
81          } else if (actual instanceof RQLStatement) {
82              RQLStatementAssert.assertIs(assertContext, (RQLStatement) actual, expected);
83          } else if (actual instanceof RALStatement) {
84              RALStatementAssert.assertIs(assertContext, (RALStatement) actual, expected);
85          } else if (actual instanceof RULStatement) {
86              RULStatementAssert.assertIs(assertContext, (RULStatement) actual, expected);
87          }
88      }
89  }
