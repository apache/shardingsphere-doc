118     static {
119         SYSTEM_CATALOG_TABLES.add(PG_DATABASE);

124     @Override
125     public Enumerable<Object> execute(final ShardingSphereTable table, final ScanExecutorContext scanContext) {
126         String databaseName = executorContext.getDatabaseName().toLowerCase();
127         String schemaName = executorContext.getSchemaName().toLowerCase();
128         DatabaseType databaseType = optimizerContext.getParserContext(databaseName).getDatabaseType();
129         if (new SystemDatabase(databaseType).getSystemSchemas().contains(schemaName)) {
130             return executeByShardingSphereData(databaseName, schemaName, table, databaseType);
131         }
132         SQLFederationExecutorContext federationContext = executorContext.getFederationContext();
133         QueryContext queryContext = createQueryContext(federationContext.getMetaData(), scanContext, databaseType, federationContext.getQueryContext().isUseCache());
134         ShardingSphereDatabase database = federationContext.getMetaData().getDatabase(databaseName);
135         ExecutionContext context = new KernelProcessor().generateExecutionContext(queryContext, database, globalRuleMetaData, executorContext.getProps(), new ConnectionContext());
136         if (federationContext.isPreview()) {
137             federationContext.getExecutionUnits().addAll(context.getExecutionUnits());
138             return createEmptyEnumerable();
139         }
140         try {
141             return createEnumerable(queryContext, database, context);
142         } finally {
143             processEngine.completeSQLExecution();
144         }
145     }
146     
147     private AbstractEnumerable<Object> createEnumerable(final QueryContext queryContext, final ShardingSphereDatabase database, final ExecutionContext context) {
148         return new AbstractEnumerable<Object>() {
149             
150             @SneakyThrows
151             @Override
152             public Enumerator<Object> enumerator() {
153                 computeConnectionOffsets(context);
154                 ExecutionGroupContext<JDBCExecutionUnit> executionGroupContext =
155                         prepareEngine.prepare(context.getRouteContext(), executorContext.getConnectionOffsets(), context.getExecutionUnits(), new ExecutionGroupReportContext(database.getName()));
156                 setParameters(executionGroupContext.getInputGroups());
157                 processEngine.executeSQL(executionGroupContext, context.getQueryContext());
158                 List<QueryResult> queryResults = jdbcExecutor.execute(executionGroupContext, callback).stream().map(QueryResult.class::cast).collect(Collectors.toList());
159                 MergeEngine mergeEngine = new MergeEngine(database, executorContext.getProps(), new ConnectionContext());
160                 MergedResult mergedResult = mergeEngine.merge(queryResults, queryContext.getSqlStatementContext());
161                 Collection<Statement> statements = getStatements(executionGroupContext.getInputGroups());
162                 return new SQLFederationRowEnumerator(mergedResult, queryResults.get(0).getMetaData(), statements);
163             }
164         };
165     }
166     
167     private void computeConnectionOffsets(final ExecutionContext context) {
168         for (ExecutionUnit each : context.getExecutionUnits()) {
169             if (executorContext.getConnectionOffsets().containsKey(each.getDataSourceName())) {
170                 int connectionOffset = executorContext.getConnectionOffsets().get(each.getDataSourceName());
171                 executorContext.getConnectionOffsets().put(each.getDataSourceName(), ++connectionOffset);
172             } else {
173                 executorContext.getConnectionOffsets().put(each.getDataSourceName(), 0);
174             }
175         }
176     }
177     
178     private Enumerable<Object> executeByShardingSphereData(final String databaseName, final String schemaName, final ShardingSphereTable table, final DatabaseType databaseType) {
179         // TODO move this logic to ShardingSphere statistics
180         if (databaseType instanceof OpenGaussDatabaseType && SYSTEM_CATALOG_TABLES.contains(table.getName().toLowerCase())) {
181             return createMemoryEnumerator(createSystemCatalogTableData(table));
182         }
183         Optional<ShardingSphereTableData> tableData = Optional.ofNullable(statistics.getDatabaseData().get(databaseName)).map(optional -> optional.getSchemaData().get(schemaName))
184                 .map(ShardingSphereSchemaData::getTableData).map(shardingSphereData -> shardingSphereData.get(table.getName()));
185         return tableData.map(this::createMemoryEnumerator).orElseGet(this::createEmptyEnumerable);
186     }
187     
188     private ShardingSphereTableData createSystemCatalogTableData(final ShardingSphereTable table) {
189         ShardingSphereTableData result = new ShardingSphereTableData(table.getName());
190         ShardingSphereMetaData metaData = executorContext.getFederationContext().getMetaData();
191         if (PG_DATABASE.equalsIgnoreCase(table.getName())) {
192             appendOpenGaussDatabaseData(result, metaData.getDatabases().values());
193         } else if (PG_TABLES.equalsIgnoreCase(table.getName())) {
194             for (ShardingSphereDatabase each : metaData.getDatabases().values()) {
195                 appendOpenGaussTableData(result, each.getSchemas());
196             }
197         } else if (PG_ROLES.equalsIgnoreCase(table.getName())) {
198             appendOpenGaussRoleData(result, metaData);
199         }
200         return result;
201     }
202     
203     private void appendOpenGaussDatabaseData(final ShardingSphereTableData tableData, final Collection<ShardingSphereDatabase> databases) {
204         for (ShardingSphereDatabase each : databases) {
205             Object[] rows = new Object[15];
206             rows[0] = each.getName();
207             rows[11] = DAT_COMPATIBILITY;
208             tableData.getRows().add(new ShardingSphereRowData(Arrays.asList(rows)));
209         }
210     }
211     
212     private void appendOpenGaussTableData(final ShardingSphereTableData tableData, final Map<String, ShardingSphereSchema> schemas) {
213         for (Entry<String, ShardingSphereSchema> entry : schemas.entrySet()) {
214             for (String each : entry.getValue().getAllTableNames()) {
215                 Object[] rows = new Object[10];
216                 rows[0] = entry.getKey();
217                 rows[1] = each;
218                 tableData.getRows().add(new ShardingSphereRowData(Arrays.asList(rows)));
219             }
220         }
221     }
222     
223     private void appendOpenGaussRoleData(final ShardingSphereTableData tableData, final ShardingSphereMetaData metaData) {
224         for (ShardingSphereUser each : metaData.getGlobalRuleMetaData().getSingleRule(AuthorityRule.class).getConfiguration().getUsers()) {
225             Object[] rows = new Object[27];
226             rows[0] = each.getGrantee().getUsername();
227             tableData.getRows().add(new ShardingSphereRowData(Arrays.asList(rows)));
228         }
229     }

231     private Enumerable<Object> createMemoryEnumerator(final ShardingSphereTableData tableData) {
232         return new AbstractEnumerable<Object>() {
233             
234             @Override
235             public Enumerator<Object> enumerator() {
236                 return new MemoryEnumerator(tableData.getRows());
237             }
238         };
239     }
240     
241     private Collection<Statement> getStatements(final Collection<ExecutionGroup<JDBCExecutionUnit>> inputGroups) {
242         Collection<Statement> result = new LinkedList<>();
243         for (ExecutionGroup<JDBCExecutionUnit> each : inputGroups) {
244             for (JDBCExecutionUnit executionUnit : each.getInputs()) {
245                 result.add(executionUnit.getStorageResource());
246             }
247         }
248         return result;
249     }
250     
251     private void setParameters(final Collection<ExecutionGroup<JDBCExecutionUnit>> inputGroups) {
252         for (ExecutionGroup<JDBCExecutionUnit> each : inputGroups) {
253             for (JDBCExecutionUnit executionUnit : each.getInputs()) {
254                 if (!(executionUnit.getStorageResource() instanceof PreparedStatement)) {
255                     continue;
256                 }
257                 setParameters((PreparedStatement) executionUnit.getStorageResource(), executionUnit.getExecutionUnit().getSqlUnit().getParameters());
258             }
259         }
260     }
261     
262     @SneakyThrows(SQLException.class)
263     private void setParameters(final PreparedStatement preparedStatement, final List<Object> params) {
264         for (int i = 0; i < params.size(); i++) {
265             preparedStatement.setObject(i + 1, params.get(i));
266         }
267     }
268     
269     private QueryContext createQueryContext(final ShardingSphereMetaData metaData, final ScanExecutorContext sqlString, final DatabaseType databaseType, final boolean useCache) {
270         String sql = sqlString.getSql().replace(System.lineSeparator(), " ");
271         SQLStatement sqlStatement = new SQLStatementParserEngine(databaseType,
272                 optimizerContext.getSqlParserRule().getSqlStatementCache(), optimizerContext.getSqlParserRule().getParseTreeCache()).parse(sql, useCache);
273         List<Object> params = getParameters(sqlString.getParamIndexes());
274         HintValueContext hintValueContext = new HintValueContext();
275         SQLStatementContext sqlStatementContext = new SQLBindEngine(metaData, executorContext.getDatabaseName(), hintValueContext).bind(sqlStatement, params);
276         return new QueryContext(sqlStatementContext, sql, params, hintValueContext, useCache);
277     }
278     
279     private List<Object> getParameters(final int[] paramIndexes) {
280         if (null == paramIndexes) {
281             return Collections.emptyList();
282         }
283         List<Object> result = new ArrayList<>();
284         for (int each : paramIndexes) {
285             result.add(executorContext.getFederationContext().getQueryContext().getParameters().get(each));
286         }
287         return result;
288     }
289     
290     private AbstractEnumerable<Object> createEmptyEnumerable() {
291         return new AbstractEnumerable<Object>() {
292             
293             @Override
294             public Enumerator<Object> enumerator() {
295                 return new EmptyRowEnumerator();
296             }
297         };
298     }
