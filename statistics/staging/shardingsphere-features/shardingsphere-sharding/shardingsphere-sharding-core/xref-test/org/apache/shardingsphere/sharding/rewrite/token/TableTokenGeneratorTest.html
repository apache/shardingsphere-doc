50  class TableTokenGeneratorTest {
51      
52      @Test
53      void assertIsGenerateSQLTokenWhenConfigAllBindingTables() {
54          TableTokenGenerator generator = new TableTokenGenerator();
55          ShardingRule shardingRule = mock(ShardingRule.class);
56          Collection<String> logicTableNames = Arrays.asList("t_order", "t_order_item");
57          when(shardingRule.getShardingLogicTableNames(logicTableNames)).thenReturn(logicTableNames);
58          when(shardingRule.isAllBindingTables(logicTableNames)).thenReturn(true);
59          generator.setShardingRule(shardingRule);
60          SQLStatementContext<SelectStatement> sqlStatementContext = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
61          when(sqlStatementContext.getTablesContext().getTableNames()).thenReturn(logicTableNames);
62          assertTrue(generator.isGenerateSQLToken(sqlStatementContext));
63      }
64      
65      @Test
66      void assertIsGenerateSQLTokenWhenContainsTableSharding() {
67          TableTokenGenerator generator = new TableTokenGenerator();
68          RouteContext routeContext = mock(RouteContext.class);
69          when(routeContext.containsTableSharding()).thenReturn(true);
70          generator.setShardingRule(mock(ShardingRule.class));
71          generator.setRouteContext(routeContext);
72          SQLStatementContext<SelectStatement> sqlStatementContext = mock(SelectStatementContext.class, RETURNS_DEEP_STUBS);
73          assertTrue(generator.isGenerateSQLToken(sqlStatementContext));
74      }
75      
76      @Test
77      void assertGenerateSQLTokenWhenSQLStatementIsTableAvailable() {
78          ShardingRule shardingRule = mock(ShardingRule.class);
79          when(shardingRule.findTableRule(anyString())).thenReturn(Optional.of(mock(TableRule.class)));
80          TableTokenGenerator generator = new TableTokenGenerator();
81          generator.setShardingRule(shardingRule);
82          CreateTableStatementContext sqlStatementContext = mock(CreateTableStatementContext.class);
83          when(sqlStatementContext.getAllTables()).thenReturn(Collections.singletonList(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")))));
84          Collection<TableToken> actual = generator.generateSQLTokens(sqlStatementContext);
85          assertThat(actual.size(), is(1));
86          assertThat(actual.iterator().next(), instanceOf(TableToken.class));
87      }
88      
89      @Test
90      void assertGenerateSQLTokenWhenSQLStatementIsNotTableAvailable() {
91          TableTokenGenerator generator = new TableTokenGenerator();
92          SQLStatementContext<CreateDatabaseStatement> sqlStatementContext = mock(CreateDatabaseStatementContext.class);
93          assertThat(generator.generateSQLTokens(sqlStatementContext), is(Collections.emptyList()));
94      }
95  }
