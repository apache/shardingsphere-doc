44  /**
45   * Test parameters for SQL rewrite engine builder.
46   */
47  @NoArgsConstructor(access = AccessLevel.PRIVATE)
48  public final class SQLRewriteEngineTestParametersBuilder {
49      
50      /**
51       * Load test parameters.
52       * 
53       * @param type type
54       * @param path path
55       * @param targetClass target class
56       * @return test parameters list for SQL rewrite engine
57       */
58      public static Collection<SQLRewriteEngineTestParameters> loadTestParameters(final String type, final String path, final Class<?> targetClass) {
59          Collection<SQLRewriteEngineTestParameters> result = new LinkedList<>();
60          for (Entry<String, RewriteAssertionsRootEntity> entry : loadAllRewriteAssertionsRootEntities(type, path, targetClass).entrySet()) {
61              result.addAll(createTestParameters(type, entry.getKey(), entry.getValue()));
62          }
63          return result;
64      }
65      
66      private static Map<String, RewriteAssertionsRootEntity> loadAllRewriteAssertionsRootEntities(final String type, final String path, final Class<?> targetClass) {
67          Map<String, RewriteAssertionsRootEntity> result = new LinkedHashMap<>();
68          File file = new File(targetClass.getProtectionDomain().getCodeSource().getLocation().getPath() + "/" + path);
69          for (File each : Objects.requireNonNull(file.listFiles())) {
70              if (each.isFile()) {
71                  appendFromFile(type, each, path, result);
72              } else {
73                  appendFromDirectory(type, each, path + "/" + each.getName(), result);
74              }
75          }
76          return result;
77      }
78      
79      private static void appendFromDirectory(final String type, final File directory, final String path, final Map<String, RewriteAssertionsRootEntity> result) {
80          for (File each : Objects.requireNonNull(directory.listFiles())) {
81              if (each.isFile()) {
82                  appendFromFile(type, each, path, result);
83              } else {
84                  appendFromDirectory(type, each, path + "/" + each.getName(), result);
85              }
86          }
87      }
88      
89      private static void appendFromFile(final String type, final File file, final String path, final Map<String, RewriteAssertionsRootEntity> result) {
90          if (file.getName().endsWith(".xml")) {
91              String key = path.toLowerCase().replace(type.toLowerCase() + "/", "") + "/" + file.getName();
92              result.put(key, new RewriteAssertionsRootEntityLoader().load(path + "/" + file.getName()));
93          }
94      }
95      
96      private static Collection<SQLRewriteEngineTestParameters> createTestParameters(final String type, final String fileName, final RewriteAssertionsRootEntity rootAssertions) {
97          Collection<SQLRewriteEngineTestParameters> result = new LinkedList<>();
98          for (RewriteAssertionEntity each : rootAssertions.getAssertions()) {
99              for (String databaseType : getDatabaseTypes(each.getDatabaseTypes())) {
100                 result.add(new SQLRewriteEngineTestParameters(type, each.getId(), fileName, rootAssertions.getYamlRule(), each.getInput().getSql(),
101                         createParameters(each.getInput().getParameters()), createOutputSQLs(each.getOutputs()), createOutputGroupedParameters(each.getOutputs()), databaseType));
102             }
103         }
104         return result;
105     }
106     
107     private static Collection<String> getDatabaseTypes(final String databaseTypes) {
108         return Strings.isNullOrEmpty(databaseTypes) ? getAllDatabaseTypes() : Splitter.on(',').trimResults().splitToList(databaseTypes);
109     }
110     
111     private static Collection<String> getAllDatabaseTypes() {
112         return Arrays.asList("MySQL", "PostgreSQL", "Oracle", "SQLServer", "SQL92", "openGauss");
113     }
114     
115     private static List<Object> createParameters(final String inputParams) {
116         if (null == inputParams) {
117             return Collections.emptyList();
118         }
119         return Splitter.on(",").trimResults().splitToList(inputParams).stream().map(SQLRewriteEngineTestParametersBuilder::createInputParameter).collect(Collectors.toList());
120     }
121     
122     private static Object createInputParameter(final String inputParam) {
123         if (StringUtils.isNumeric(inputParam)) {
124             return NumberUtils.createNumber(inputParam);
125         }
126         return "NULL".equals(inputParam) ? null : inputParam;
127     }
128     
129     private static List<String> createOutputSQLs(final List<RewriteOutputEntity> outputs) {
130         List<String> result = new ArrayList<>(outputs.size());
131         for (RewriteOutputEntity each : outputs) {
132             result.add(each.getSql());
133         }
134         return result;
135     }
136     
137     private static List<List<Object>> createOutputGroupedParameters(final List<RewriteOutputEntity> outputs) {
138         List<List<Object>> result = new ArrayList<>(outputs.size());
139         for (RewriteOutputEntity each : outputs) {
140             result.add(createParameters(each.getParameters()));
141         }
142         return result;
143     }
144 }
