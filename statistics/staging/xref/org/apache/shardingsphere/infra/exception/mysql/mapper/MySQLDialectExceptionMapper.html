49  /**
50   * MySQL dialect exception mapper.
51   */
52  public final class MySQLDialectExceptionMapper implements SQLDialectExceptionMapper {
53      
54      @Override
55      public SQLException convert(final SQLDialectException sqlDialectException) {
56          if (sqlDialectException instanceof UnknownDatabaseException) {
57              return null != ((UnknownDatabaseException) sqlDialectException).getDatabaseName()
58                      ? toSQLException(MySQLVendorError.ER_BAD_DB_ERROR, ((UnknownDatabaseException) sqlDialectException).getDatabaseName())
59                      : toSQLException(MySQLVendorError.ER_NO_DB_ERROR);
60          }
61          if (sqlDialectException instanceof NoDatabaseSelectedException) {
62              return toSQLException(MySQLVendorError.ER_NO_DB_ERROR);
63          }
64          if (sqlDialectException instanceof DatabaseCreateExistsException) {
65              return toSQLException(MySQLVendorError.ER_DB_CREATE_EXISTS_ERROR, ((DatabaseCreateExistsException) sqlDialectException).getDatabaseName());
66          }
67          if (sqlDialectException instanceof DatabaseDropNotExistsException) {
68              return toSQLException(MySQLVendorError.ER_DB_DROP_NOT_EXISTS_ERROR, ((DatabaseDropNotExistsException) sqlDialectException).getDatabaseName());
69          }
70          if (sqlDialectException instanceof TableExistsException) {
71              return toSQLException(MySQLVendorError.ER_TABLE_EXISTS_ERROR, ((TableExistsException) sqlDialectException).getTableName());
72          }
73          if (sqlDialectException instanceof DialectSQLParsingException) {
74              return toSQLException(MySQLVendorError.ER_PARSE_ERROR, sqlDialectException.getMessage(), ((DialectSQLParsingException) sqlDialectException).getSymbol(),
75                      ((DialectSQLParsingException) sqlDialectException).getLine());
76          }
77          if (sqlDialectException instanceof NoSuchTableException) {
78              return toSQLException(MySQLVendorError.ER_NO_SUCH_TABLE, ((NoSuchTableException) sqlDialectException).getTableName());
79          }
80          if (sqlDialectException instanceof InsertColumnsAndValuesMismatchedException) {
81              return toSQLException(MySQLVendorError.ER_WRONG_VALUE_COUNT_ON_ROW, ((InsertColumnsAndValuesMismatchedException) sqlDialectException).getMismatchedRowNumber());
82          }
83          if (sqlDialectException instanceof TableModifyInTransactionException) {
84              return toSQLException(MySQLVendorError.ER_ERROR_ON_MODIFYING_GTID_EXECUTED_TABLE, ((TableModifyInTransactionException) sqlDialectException).getTableName());
85          }
86          if (sqlDialectException instanceof TooManyConnectionsException) {
87              return toSQLException(MySQLVendorError.ER_CON_COUNT_ERROR);
88          }
89          if (sqlDialectException instanceof UnsupportedPreparedStatementException) {
90              return toSQLException(MySQLVendorError.ER_UNSUPPORTED_PS);
91          }
92          if (sqlDialectException instanceof TooManyPlaceholdersException) {
93              return toSQLException(MySQLVendorError.ER_PS_MANY_PARAM);
94          }
95          if (sqlDialectException instanceof UnknownCharsetException) {
96              return toSQLException(MySQLVendorError.ER_UNKNOWN_CHARACTER_SET, ((UnknownCharsetException) sqlDialectException).getCharset());
97          }
98          if (sqlDialectException instanceof UnknownCollationException) {
99              return toSQLException(MySQLVendorError.ER_UNKNOWN_COLLATION, ((UnknownCollationException) sqlDialectException).getCollationId());
100         }
101         if (sqlDialectException instanceof HandshakeException) {
102             return toSQLException(MySQLVendorError.ER_HANDSHAKE_ERROR);
103         }
104         if (sqlDialectException instanceof AccessDeniedException) {
105             AccessDeniedException ex = (AccessDeniedException) sqlDialectException;
106             return toSQLException(MySQLVendorError.ER_ACCESS_DENIED_ERROR, ex.getUsername(), ex.getHostname(), ex.isUsingPassword() ? "YES" : "NO");
107         }
108         if (sqlDialectException instanceof DatabaseAccessDeniedException) {
109             DatabaseAccessDeniedException ex = (DatabaseAccessDeniedException) sqlDialectException;
110             return toSQLException(MySQLVendorError.ER_DBACCESS_DENIED_ERROR, ex.getUsername(), ex.getHostname(), ex.getDatabaseName());
111         }
112         if (sqlDialectException instanceof UnknownSystemVariableException) {
113             return toSQLException(MySQLVendorError.ER_UNKNOWN_SYSTEM_VARIABLE, ((UnknownSystemVariableException) sqlDialectException).getVariableName());
114         }
115         if (sqlDialectException instanceof ErrorLocalVariableException) {
116             return toSQLException(MySQLVendorError.ER_LOCAL_VARIABLE, ((ErrorLocalVariableException) sqlDialectException).getVariableName());
117         }
118         if (sqlDialectException instanceof ErrorGlobalVariableException) {
119             return toSQLException(MySQLVendorError.ER_GLOBAL_VARIABLE, ((ErrorGlobalVariableException) sqlDialectException).getVariableName());
120         }
121         if (sqlDialectException instanceof IncorrectGlobalLocalVariableException) {
122             IncorrectGlobalLocalVariableException ex = (IncorrectGlobalLocalVariableException) sqlDialectException;
123             return toSQLException(MySQLVendorError.ER_INCORRECT_GLOBAL_LOCAL_VAR, ex.getVariableName(), ex.getScope());
124         }
125         return new UnknownSQLException(sqlDialectException).toSQLException();
126     }
127     
128     private SQLException toSQLException(final VendorError vendorError, final Object... messageArgs) {
129         return new SQLException(String.format(vendorError.getReason(), messageArgs), vendorError.getSqlState().getValue(), vendorError.getVendorCode());
130     }
131     
132     @Override
133     public String getDatabaseType() {
134         return "MySQL";
135     }
136 }
