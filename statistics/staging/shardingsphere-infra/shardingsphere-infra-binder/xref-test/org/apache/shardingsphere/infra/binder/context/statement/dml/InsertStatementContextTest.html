98      private void assertInsertStatementContextWithColumnNames(final InsertStatement insertStatement) {
99          SimpleTableSegment tableSegment = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("tbl")));
100         tableSegment.setOwner(new OwnerSegment(0, 0, new IdentifierValue(DefaultDatabase.LOGIC_NAME.toUpperCase())));
101         insertStatement.setTable(tableSegment);
102         InsertColumnsSegment insertColumnsSegment = new InsertColumnsSegment(0, 0, Arrays.asList(
103                 new ColumnSegment(0, 0, new IdentifierValue("id")), new ColumnSegment(0, 0, new IdentifierValue("name")), new ColumnSegment(0, 0, new IdentifierValue("status"))));
104         insertStatement.setInsertColumns(insertColumnsSegment);
105         setUpInsertValues(insertStatement);
106         InsertStatementContext actual = createInsertStatementContext(Arrays.asList(1, "Tom", 2, "Jerry"), insertStatement);
107         actual.setUpParameters(Arrays.asList(1, "Tom", 2, "Jerry"));
108         assertInsertStatementContext(actual);
109     }
110     
111     private InsertStatementContext createInsertStatementContext(final List<Object> params, final InsertStatement insertStatement) {
112         ShardingSphereDatabase database = mock(ShardingSphereDatabase.class, RETURNS_DEEP_STUBS);
113         ShardingSphereSchema schema = mock(ShardingSphereSchema.class);
114         String defaultSchemaName = insertStatement instanceof PostgreSQLStatement || insertStatement instanceof OpenGaussStatement ? "public" : DefaultDatabase.LOGIC_NAME;
115         when(database.getSchema(defaultSchemaName)).thenReturn(schema);
116         when(schema.getVisibleColumnNames("tbl")).thenReturn(Arrays.asList("id", "name", "status"));
117         return new InsertStatementContext(createShardingSphereMetaData(database), params, insertStatement, DefaultDatabase.LOGIC_NAME);
118     }
119     
120     private ShardingSphereMetaData createShardingSphereMetaData(final ShardingSphereDatabase database) {
121         return new ShardingSphereMetaData(Collections.singletonMap(DefaultDatabase.LOGIC_NAME, database), mock(ShardingSphereResourceMetaData.class),
122                 mock(ShardingSphereRuleMetaData.class), mock(ConfigurationProperties.class));
123     }

175     private void setUpInsertValues(final InsertStatement insertStatement) {
176         insertStatement.getValues().add(new InsertValuesSegment(0, 0, Arrays.asList(
177                 new ParameterMarkerExpressionSegment(0, 0, 1), new ParameterMarkerExpressionSegment(0, 0, 2), new LiteralExpressionSegment(0, 0, "init"))));
178         insertStatement.getValues().add(new InsertValuesSegment(0, 0, Arrays.asList(
179                 new ParameterMarkerExpressionSegment(0, 0, 3), new ParameterMarkerExpressionSegment(0, 0, 4), new LiteralExpressionSegment(0, 0, "init"))));
180     }

182     private void setUpOnDuplicateValues(final MySQLInsertStatement insertStatement) {
183         List<ColumnSegment> parameterMarkerExpressionAssignmentColumns = new LinkedList<>();
184         parameterMarkerExpressionAssignmentColumns.add(new ColumnSegment(0, 0, new IdentifierValue("on_duplicate_key_update_column_1")));
185         AssignmentSegment parameterMarkerExpressionAssignment = new ColumnAssignmentSegment(0, 0, parameterMarkerExpressionAssignmentColumns,
186                 new ParameterMarkerExpressionSegment(0, 0, 4));
187         List<ColumnSegment> literalExpressionAssignmentColumns = new LinkedList<>();
188         literalExpressionAssignmentColumns.add(new ColumnSegment(0, 0, new IdentifierValue("on_duplicate_key_update_column_2")));
189         AssignmentSegment literalExpressionAssignment = new ColumnAssignmentSegment(0, 0, literalExpressionAssignmentColumns,
190                 new LiteralExpressionSegment(0, 0, 5));
191         OnDuplicateKeyColumnsSegment onDuplicateKeyColumnsSegment = new OnDuplicateKeyColumnsSegment(0, 0, Arrays.asList(parameterMarkerExpressionAssignment, literalExpressionAssignment));
192         insertStatement.setOnDuplicateKeyColumns(onDuplicateKeyColumnsSegment);
193     }

195     private void assertInsertStatementContext(final InsertStatementContext actual) {
196         assertThat(actual.getTablesContext().getTableNames(), is(new HashSet<>(Collections.singleton("tbl"))));
197         assertThat(actual.getAllTables().size(), is(1));
198         SimpleTableSegment simpleTableSegment = actual.getAllTables().iterator().next();
199         assertThat(simpleTableSegment.getTableName().getStartIndex(), is(0));
200         assertThat(simpleTableSegment.getTableName().getStopIndex(), is(0));
201         assertThat(simpleTableSegment.getTableName().getIdentifier().getValue(), is("tbl"));
202         List<String> columnNames = new ArrayList<>(3);
203         actual.getDescendingColumnNames().forEachRemaining(columnNames::add);
204         assertThat(columnNames, is(Arrays.asList("status", "name", "id")));
205         assertThat(actual.getGeneratedKeyContext(), is(Optional.empty()));
206         assertThat(actual.getColumnNames(), is(Arrays.asList("id", "name", "status")));
207         assertThat(actual.getInsertValueContexts().size(), is(2));
208         assertTrue(actual.getInsertValueContexts().get(0).getLiteralValue(0).isPresent());
209         assertTrue(actual.getInsertValueContexts().get(0).getLiteralValue(1).isPresent());
210         assertTrue(actual.getInsertValueContexts().get(0).getLiteralValue(2).isPresent());
211         assertTrue(actual.getInsertValueContexts().get(1).getLiteralValue(0).isPresent());
212         assertTrue(actual.getInsertValueContexts().get(1).getLiteralValue(1).isPresent());
213         assertTrue(actual.getInsertValueContexts().get(1).getLiteralValue(2).isPresent());
214         assertThat(actual.getInsertValueContexts().get(0).getLiteralValue(0).get(), is(1));
215         assertThat(actual.getInsertValueContexts().get(0).getLiteralValue(1).get(), is("Tom"));
216         assertThat(actual.getInsertValueContexts().get(0).getLiteralValue(2).get(), is("init"));
217         assertThat(actual.getInsertValueContexts().get(1).getLiteralValue(0).get(), is(2));
218         assertThat(actual.getInsertValueContexts().get(1).getLiteralValue(1).get(), is("Jerry"));
219         assertThat(actual.getInsertValueContexts().get(1).getLiteralValue(2).get(), is("init"));
220     }
