18  package org.apache.shardingsphere.mode.metadata.refresher.type.schema;
19  
20  import org.apache.shardingsphere.infra.config.props.ConfigurationProperties;
21  import org.apache.shardingsphere.infra.database.core.type.DatabaseType;
22  import org.apache.shardingsphere.infra.metadata.database.ShardingSphereDatabase;
23  import org.apache.shardingsphere.infra.metadata.database.schema.pojo.AlterSchemaPOJO;
24  import org.apache.shardingsphere.mode.metadata.refresher.MetaDataRefresher;
25  import org.apache.shardingsphere.mode.service.MetaDataManagerPersistService;
26  import org.apache.shardingsphere.sql.parser.sql.common.statement.ddl.AlterSchemaStatement;
27  import org.apache.shardingsphere.sql.parser.sql.dialect.handler.ddl.AlterSchemaStatementHandler;
28  
29  import java.util.Collection;
30  import java.util.Optional;
31  
32  /**
33   * Schema refresher for alter schema statement.
34   */
35  public final class AlterSchemaStatementSchemaRefresher implements MetaDataRefresher<AlterSchemaStatement> {
36      
37      @Override
38      public void refresh(final MetaDataManagerPersistService metaDataManagerPersistService, final ShardingSphereDatabase database, final Collection<String> logicDataSourceNames,
39                          final String schemaName, final DatabaseType databaseType, final AlterSchemaStatement sqlStatement, final ConfigurationProperties props) {
40          Optional<String> renameSchemaName = AlterSchemaStatementHandler.getRenameSchema(sqlStatement).map(optional -> optional.getValue().toLowerCase());
41          if (!renameSchemaName.isPresent()) {
42              return;
43          }
44          metaDataManagerPersistService.alterSchema(new AlterSchemaPOJO(database.getName(), sqlStatement.getSchemaName().getValue().toLowerCase(),
45                  renameSchemaName.get(), logicDataSourceNames));
46      }
47      
48      @Override
49      public Class<AlterSchemaStatement> getType() {
50          return AlterSchemaStatement.class;
51      }
52  }
