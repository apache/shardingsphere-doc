55  /**
56   * Database rule configuration manager.
57   */
58  @Slf4j
59  public final class DatabaseRuleConfigurationManager {
60      
61      private final AtomicReference<MetaDataContexts> metaDataContexts;
62      
63      private final ComputeNodeInstanceContext computeNodeInstanceContext;
64      
65      private final MetaDataPersistService metaDataPersistService;
66      
67      public DatabaseRuleConfigurationManager(final AtomicReference<MetaDataContexts> metaDataContexts, final ComputeNodeInstanceContext computeNodeInstanceContext,
68                                              final PersistRepository repository) {
69          this.metaDataContexts = metaDataContexts;
70          this.computeNodeInstanceContext = computeNodeInstanceContext;
71          metaDataPersistService = new MetaDataPersistService(repository);
72      }
73      
74      /**
75       * Alter rule configuration.
76       *
77       * @param databaseName database name
78       * @param ruleConfig rule configurations
79       * @throws SQLException SQL Exception
80       */
81      @SuppressWarnings({"unchecked", "rawtypes"})
82      public synchronized void alterRuleConfiguration(final String databaseName, final RuleConfiguration ruleConfig) throws SQLException {
83          ShardingSphereDatabase database = metaDataContexts.get().getMetaData().getDatabase(databaseName);
84          Collection<ShardingSphereRule> rules = new LinkedList<>(database.getRuleMetaData().getRules());
85          Optional<ShardingSphereRule> toBeChangedRule = rules.stream().filter(each -> each.getConfiguration().getClass().equals(ruleConfig.getClass())).findFirst();
86          if (toBeChangedRule.isPresent() && toBeChangedRule.get() instanceof PartialRuleUpdateSupported && ((PartialRuleUpdateSupported) toBeChangedRule.get()).partialUpdate(ruleConfig)) {
87              ((PartialRuleUpdateSupported) toBeChangedRule.get()).updateConfiguration(ruleConfig);
88              return;
89          }
90          rules.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
91          rules.addAll(DatabaseRulesBuilder.build(databaseName, database.getProtocolType(), database.getRuleMetaData().getRules(),
92                  ruleConfig, computeNodeInstanceContext, database.getResourceMetaData()));
93          refreshMetadata(databaseName, database, rules);
94      }
95      
96      /**
97       * Drop rule configuration.
98       *
99       * @param databaseName database name
100      * @param ruleConfig rule configurations
101      * @throws SQLException SQL Exception
102      */
103     @SuppressWarnings({"unchecked", "rawtypes"})
104     public synchronized void dropRuleConfiguration(final String databaseName, final RuleConfiguration ruleConfig) throws SQLException {
105         ShardingSphereDatabase database = metaDataContexts.get().getMetaData().getDatabase(databaseName);
106         Collection<ShardingSphereRule> rules = new LinkedList<>(database.getRuleMetaData().getRules());
107         Optional<ShardingSphereRule> toBeChangedRule = rules.stream().filter(each -> each.getConfiguration().getClass().equals(ruleConfig.getClass())).findFirst();
108         if (toBeChangedRule.isPresent() && toBeChangedRule.get() instanceof PartialRuleUpdateSupported && ((PartialRuleUpdateSupported) toBeChangedRule.get()).partialUpdate(ruleConfig)) {
109             ((PartialRuleUpdateSupported) toBeChangedRule.get()).updateConfiguration(ruleConfig);
110             return;
111         }
112         rules.removeIf(each -> each.getConfiguration().getClass().isAssignableFrom(ruleConfig.getClass()));
113         if (!((DatabaseRuleConfiguration) ruleConfig).isEmpty()) {
114             rules.addAll(DatabaseRulesBuilder.build(databaseName, database.getProtocolType(), database.getRuleMetaData().getRules(),
115                     ruleConfig, computeNodeInstanceContext, database.getResourceMetaData()));
116         }
117         refreshMetadata(databaseName, database, rules);
118     }
119     
120     private void refreshMetadata(final String databaseName, final ShardingSphereDatabase database, final Collection<ShardingSphereRule> rules) throws SQLException {
121         database.getRuleMetaData().getRules().clear();
122         database.getRuleMetaData().getRules().addAll(rules);
123         MetaDataContexts reloadMetaDataContexts = createMetaDataContextsByAlterRule(databaseName, database.getRuleMetaData().getConfigurations());
124         metaDataContexts.set(reloadMetaDataContexts);
125         metaDataContexts.get().getMetaData().getDatabase(databaseName).getSchemas().putAll(buildShardingSphereSchemas(metaDataContexts.get().getMetaData().getDatabase(databaseName)));
126     }
127     
128     private MetaDataContexts createMetaDataContextsByAlterRule(final String databaseName, final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
129         Map<String, ShardingSphereDatabase> changedDatabases = createChangedDatabases(databaseName, false, null, ruleConfigs);
130         return newMetaDataContexts(new ShardingSphereMetaData(changedDatabases, metaDataContexts.get().getMetaData().getGlobalResourceMetaData(),
131                 metaDataContexts.get().getMetaData().getGlobalRuleMetaData(), metaDataContexts.get().getMetaData().getProps()));
132     }
133     
134     private ResourceMetaData getEffectiveResourceMetaData(final ShardingSphereDatabase database, final SwitchingResource resource) {
135         Map<StorageNode, DataSource> storageNodes = getStorageNodes(database.getResourceMetaData().getDataSources(), resource);
136         Map<String, StorageUnit> storageUnits = getStorageUnits(database.getResourceMetaData().getStorageUnits(), resource);
137         return new ResourceMetaData(storageNodes, storageUnits);
138     }
139     
140     private Map<StorageNode, DataSource> getStorageNodes(final Map<StorageNode, DataSource> currentStorageNodes, final SwitchingResource resource) {
141         Map<StorageNode, DataSource> result = new LinkedHashMap<>(currentStorageNodes.size(), 1F);
142         for (Entry<StorageNode, DataSource> entry : currentStorageNodes.entrySet()) {
143             if (null == resource || !resource.getStaleDataSources().containsKey(entry.getKey())) {
144                 result.put(entry.getKey(), entry.getValue());
145             }
146         }
147         return result;
148     }
149     
150     private Map<String, StorageUnit> getStorageUnits(final Map<String, StorageUnit> currentStorageUnits, final SwitchingResource resource) {
151         Map<String, StorageUnit> result = new LinkedHashMap<>(currentStorageUnits.size(), 1F);
152         for (Entry<String, StorageUnit> entry : currentStorageUnits.entrySet()) {
153             if (null == resource || !resource.getStaleStorageUnitNames().contains(entry.getKey())) {
154                 result.put(entry.getKey(), entry.getValue());
155             }
156         }
157         return result;
158     }
159     
160     private synchronized Map<String, ShardingSphereDatabase> createChangedDatabases(final String databaseName, final boolean internalLoadMetaData,
161                                                                                     final SwitchingResource switchingResource, final Collection<RuleConfiguration> ruleConfigs) throws SQLException {
162         ResourceMetaData effectiveResourceMetaData = getEffectiveResourceMetaData(metaDataContexts.get().getMetaData().getDatabase(databaseName), switchingResource);
163         Collection<RuleConfiguration> toBeCreatedRuleConfigs = null == ruleConfigs
164                 ? metaDataContexts.get().getMetaData().getDatabase(databaseName).getRuleMetaData().getConfigurations()
165                 : ruleConfigs;
166         DatabaseConfiguration toBeCreatedDatabaseConfig = getDatabaseConfiguration(effectiveResourceMetaData, switchingResource, toBeCreatedRuleConfigs);
167         ShardingSphereDatabase changedDatabase = createChangedDatabase(metaDataContexts.get().getMetaData().getDatabase(databaseName).getName(), internalLoadMetaData,
168                 metaDataPersistService, toBeCreatedDatabaseConfig, metaDataContexts.get().getMetaData().getProps(), computeNodeInstanceContext);
169         Map<String, ShardingSphereDatabase> result = new LinkedHashMap<>(metaDataContexts.get().getMetaData().getDatabases());
170         result.put(databaseName.toLowerCase(), changedDatabase);
171         return result;
172     }
173     
174     private DatabaseConfiguration getDatabaseConfiguration(final ResourceMetaData resourceMetaData, final SwitchingResource switchingResource,
175                                                            final Collection<RuleConfiguration> toBeCreatedRuleConfigs) {
176         Map<String, DataSourcePoolProperties> propsMap = null == switchingResource
177                 ? resourceMetaData.getStorageUnits().entrySet().stream()
178                         .collect(Collectors.toMap(Entry::getKey, entry -> entry.getValue().getDataSourcePoolProperties(), (oldValue, currentValue) -> oldValue, LinkedHashMap::new))
179                 : switchingResource.getMergedDataSourcePoolPropertiesMap();
180         return new DataSourceProvidedDatabaseConfiguration(getMergedStorageNodeDataSources(resourceMetaData, switchingResource), toBeCreatedRuleConfigs, propsMap);
181     }
182     
183     private Map<StorageNode, DataSource> getMergedStorageNodeDataSources(final ResourceMetaData currentResourceMetaData, final SwitchingResource switchingResource) {
184         Map<StorageNode, DataSource> result = currentResourceMetaData.getDataSources();
185         if (null != switchingResource && !switchingResource.getNewDataSources().isEmpty()) {
186             result.putAll(switchingResource.getNewDataSources());
187         }
188         return result;
189     }
190     
191     private ShardingSphereDatabase createChangedDatabase(final String databaseName, final boolean internalLoadMetaData, final MetaDataPersistService persistService,
192                                                          final DatabaseConfiguration databaseConfig, final ConfigurationProperties props,
193                                                          final ComputeNodeInstanceContext computeNodeInstanceContext) throws SQLException {
194         return internalLoadMetaData
195                 ? InternalMetaDataFactory.create(databaseName, persistService, databaseConfig, props, computeNodeInstanceContext)
196                 : ExternalMetaDataFactory.create(databaseName, databaseConfig, props, computeNodeInstanceContext);
197     }
198     
199     private Map<String, ShardingSphereSchema> buildShardingSphereSchemas(final ShardingSphereDatabase database) {
200         Map<String, ShardingSphereSchema> result = new LinkedHashMap<>(database.getSchemas().size(), 1F);
201         database.getSchemas().forEach((key, value) -> result.put(key, new ShardingSphereSchema(value.getTables(), value.getViews())));
202         return result;
203     }
204     
205     private MetaDataContexts newMetaDataContexts(final ShardingSphereMetaData metaData) {
206         return MetaDataContextsFactory.create(metaDataPersistService, metaData);
207     }
