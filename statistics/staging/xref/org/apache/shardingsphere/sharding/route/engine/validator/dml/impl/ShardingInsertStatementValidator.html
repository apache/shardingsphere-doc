50  /**
51   * Sharding insert statement validator.
52   */
53  @RequiredArgsConstructor
54  public final class ShardingInsertStatementValidator extends ShardingDMLStatementValidator {
55      
56      private final ShardingConditions shardingConditions;
57      
58      @Override
59      public void preValidate(final ShardingRule shardingRule, final SQLStatementContext sqlStatementContext,
60                              final List<Object> params, final ShardingSphereDatabase database, final ConfigurationProperties props) {
61          if (null == ((InsertStatementContext) sqlStatementContext).getInsertSelectContext()) {
62              validateMultipleTable(shardingRule, sqlStatementContext);
63          }
64          InsertStatement insertStatement = (InsertStatement) sqlStatementContext.getSqlStatement();
65          String tableName = insertStatement.getTable().getTableName().getIdentifier().getValue();
66          Optional<SubquerySegment> insertSelectSegment = insertStatement.getInsertSelect();
67          if (insertSelectSegment.isPresent() && isContainsKeyGenerateStrategy(shardingRule, tableName)
68                  && !isContainsKeyGenerateColumn(shardingRule, insertStatement.getColumns(), tableName)) {
69              throw new MissingGenerateKeyColumnWithInsertSelectException();
70          }
71          TablesContext tablesContext = ((InsertStatementContext) sqlStatementContext).getTablesContext();
72          if (insertSelectSegment.isPresent() && shardingRule.containsShardingTable(tablesContext.getTableNames())
73                  && !isAllSameTables(tablesContext.getTableNames()) && !shardingRule.isAllBindingTables(tablesContext.getTableNames())) {
74              throw new InsertSelectTableViolationException();
75          }
76      }
77      
78      private boolean isContainsKeyGenerateStrategy(final ShardingRule shardingRule, final String tableName) {
79          return shardingRule.findGenerateKeyColumnName(tableName).isPresent();
80      }
81      
82      private boolean isContainsKeyGenerateColumn(final ShardingRule shardingRule, final Collection<ColumnSegment> columns, final String tableName) {
83          return columns.isEmpty() || columns.stream().anyMatch(each -> shardingRule.isGenerateKeyColumn(each.getIdentifier().getValue(), tableName));
84      }
85      
86      private boolean isAllSameTables(final Collection<String> tableNames) {
87          return 1L == tableNames.stream().distinct().count();
88      }
89      
90      @Override
91      public void postValidate(final ShardingRule shardingRule, final SQLStatementContext sqlStatementContext, final HintValueContext hintValueContext, final List<Object> params,
92                               final ShardingSphereDatabase database, final ConfigurationProperties props, final RouteContext routeContext) {
93          InsertStatement insertStatement = (InsertStatement) sqlStatementContext.getSqlStatement();
94          Optional<SubquerySegment> insertSelect = insertStatement.getInsertSelect();
95          String tableName = insertStatement.getTable().getTableName().getIdentifier().getValue();
96          if (insertSelect.isPresent() && shardingConditions.isNeedMerge()) {
97              boolean singleRoutingOrSameShardingCondition = routeContext.isSingleRouting() || shardingConditions.isSameShardingCondition();
98              ShardingSpherePreconditions.checkState(singleRoutingOrSameShardingCondition, () -> new UnsupportedShardingOperationException("INSERT ... SELECT ...", tableName));
99          }
100         Collection<ColumnAssignmentSegment> assignments = InsertStatementHandler.getOnDuplicateKeyColumnsSegment(insertStatement)
101                 .map(OnDuplicateKeyColumnsSegment::getColumns).orElse(Collections.emptyList());
102         Optional<ShardingConditions> onDuplicateKeyShardingConditions = createShardingConditions(sqlStatementContext, shardingRule, assignments, params);
103         Optional<RouteContext> onDuplicateKeyRouteContext = onDuplicateKeyShardingConditions.map(optional -> new ShardingStandardRoutingEngine(tableName, optional,
104                 sqlStatementContext, hintValueContext, props).route(shardingRule));
105         if (onDuplicateKeyRouteContext.isPresent() && !isSameRouteContext(routeContext, onDuplicateKeyRouteContext.get())) {
106             throw new UnsupportedUpdatingShardingValueException(tableName);
107         }
108         if (!routeContext.isSingleRouting()) {
109             boolean isSingleDataNode = routeContext.getOriginalDataNodes().stream().allMatch(dataNodes -> 1 == dataNodes.size());
110             ShardingSpherePreconditions.checkState(isSingleDataNode, () -> new DuplicateInsertDataRecordException(shardingConditions, tableName));
111         }
112     }
