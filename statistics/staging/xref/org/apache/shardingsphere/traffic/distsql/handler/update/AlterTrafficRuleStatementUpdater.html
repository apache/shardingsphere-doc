18  package org.apache.shardingsphere.traffic.distsql.handler.update;
19  
20  import org.apache.shardingsphere.distsql.handler.exception.rule.MissingRequiredRuleException;
21  import org.apache.shardingsphere.distsql.handler.ral.update.GlobalRuleRALUpdater;
22  import org.apache.shardingsphere.infra.config.algorithm.AlgorithmConfiguration;
23  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
24  import org.apache.shardingsphere.infra.util.spi.type.typed.TypedSPILoader;
25  import org.apache.shardingsphere.traffic.api.config.TrafficRuleConfiguration;
26  import org.apache.shardingsphere.traffic.api.config.TrafficStrategyConfiguration;
27  import org.apache.shardingsphere.traffic.distsql.handler.convert.TrafficRuleConverter;
28  import org.apache.shardingsphere.traffic.distsql.parser.segment.TrafficRuleSegment;
29  import org.apache.shardingsphere.traffic.distsql.parser.statement.updatable.AlterTrafficRuleStatement;
30  import org.apache.shardingsphere.traffic.spi.TrafficAlgorithm;
31  import org.apache.shardingsphere.traffic.spi.TrafficLoadBalanceAlgorithm;
32  
33  import java.util.Collection;
34  import java.util.LinkedHashMap;
35  import java.util.LinkedList;
36  import java.util.Map;
37  import java.util.stream.Collectors;
38  
39  /**
40   * Alter traffic rule statement updater.
41   */
42  public final class AlterTrafficRuleStatementUpdater implements GlobalRuleRALUpdater<AlterTrafficRuleStatement, TrafficRuleConfiguration> {
43      
44      @Override
45      public void checkSQLStatement(final TrafficRuleConfiguration currentRuleConfig, final AlterTrafficRuleStatement sqlStatement) {
46          checkRuleNames(currentRuleConfig, sqlStatement);
47          checkAlgorithmNames(sqlStatement);
48      }
49      
50      private void checkRuleNames(final TrafficRuleConfiguration currentRuleConfig, final AlterTrafficRuleStatement sqlStatement) {
51          Collection<String> notExistRuleNames = getNotExistRuleNames(currentRuleConfig, sqlStatement);
52          ShardingSpherePreconditions.checkState(notExistRuleNames.isEmpty(), () -> new MissingRequiredRuleException("Traffic", notExistRuleNames));
53      }
54      
55      private Collection<String> getNotExistRuleNames(final TrafficRuleConfiguration currentRuleConfig, final AlterTrafficRuleStatement sqlStatement) {
56          Collection<String> currentRuleNames = currentRuleConfig.getTrafficStrategies().stream().map(TrafficStrategyConfiguration::getName).collect(Collectors.toSet());
57          return sqlStatement.getSegments().stream().map(TrafficRuleSegment::getName).filter(each -> !currentRuleNames.contains(each)).collect(Collectors.toSet());
58      }
59      
60      private void checkAlgorithmNames(final AlterTrafficRuleStatement sqlStatement) {
61          sqlStatement.getSegments().forEach(each -> {
62              TypedSPILoader.checkService(TrafficAlgorithm.class, each.getAlgorithm().getName(), each.getAlgorithm().getProps());
63              if (null != each.getLoadBalancer()) {
64                  TypedSPILoader.checkService(TrafficLoadBalanceAlgorithm.class, each.getLoadBalancer().getName(), each.getLoadBalancer().getProps());
65              }
66          });
67      }
68      
69      @Override
70      public TrafficRuleConfiguration buildAlteredRuleConfiguration(final TrafficRuleConfiguration currentRuleConfig, final AlterTrafficRuleStatement sqlStatement) {
71          TrafficRuleConfiguration result = new TrafficRuleConfiguration();
72          TrafficRuleConfiguration configFromSQLStatement = TrafficRuleConverter.convert(sqlStatement.getSegments());
73          result.getTrafficStrategies().addAll(createToBeAlteredStrategyConfigurations(currentRuleConfig, configFromSQLStatement));
74          result.getTrafficAlgorithms().putAll(createToBeAlteredTrafficAlgorithms(currentRuleConfig, configFromSQLStatement, getInUsedTrafficAlgorithm(result)));
75          result.getLoadBalancers().putAll(createToBeAlteredLoadBalancers(currentRuleConfig, configFromSQLStatement, getInUsedLoadBalancer(result)));
76          return result;
77      }
78      
79      private Collection<TrafficStrategyConfiguration> createToBeAlteredStrategyConfigurations(final TrafficRuleConfiguration currentConfig, final TrafficRuleConfiguration configFromSQLStatement) {
80          Collection<TrafficStrategyConfiguration> result = new LinkedList<>(currentConfig.getTrafficStrategies());
81          Collection<String> toBeAlteredConfigNames = configFromSQLStatement.getTrafficStrategies().stream().map(TrafficStrategyConfiguration::getName).collect(Collectors.toSet());
82          result.removeIf(each -> toBeAlteredConfigNames.contains(each.getName()));
83          result.addAll(configFromSQLStatement.getTrafficStrategies());
84          return result;
85      }
86      
87      private Collection<String> getInUsedTrafficAlgorithm(final TrafficRuleConfiguration config) {
88          return config.getTrafficStrategies().stream().map(TrafficStrategyConfiguration::getAlgorithmName).collect(Collectors.toSet());
89      }
90      
91      private Map<String, AlgorithmConfiguration> createToBeAlteredTrafficAlgorithms(final TrafficRuleConfiguration currentConfig, final TrafficRuleConfiguration configFromSQLStatement,
92                                                                                     final Collection<String> inUsedTrafficAlgorithm) {
93          Map<String, AlgorithmConfiguration> result = new LinkedHashMap<>(currentConfig.getTrafficAlgorithms());
94          result.putAll(configFromSQLStatement.getTrafficAlgorithms());
95          for (String each : result.keySet().stream().filter(each -> !inUsedTrafficAlgorithm.contains(each)).collect(Collectors.toSet())) {
96              result.remove(each);
97          }
98          return result;
99      }
100     
101     private Collection<String> getInUsedLoadBalancer(final TrafficRuleConfiguration config) {
102         return config.getTrafficStrategies().stream().map(TrafficStrategyConfiguration::getLoadBalancerName).collect(Collectors.toSet());
103     }
104     
105     private Map<String, AlgorithmConfiguration> createToBeAlteredLoadBalancers(final TrafficRuleConfiguration currentConfig,
106                                                                                final TrafficRuleConfiguration configFromSQLStatement, final Collection<String> inUsedLoadBalancer) {
107         Map<String, AlgorithmConfiguration> result = new LinkedHashMap<>(currentConfig.getLoadBalancers());
108         result.putAll(configFromSQLStatement.getLoadBalancers());
109         for (String each : result.keySet().stream().filter(each -> !inUsedLoadBalancer.contains(each)).collect(Collectors.toSet())) {
110             result.remove(each);
111         }
112         return result;
113     }
114     
115     @Override
116     public Class<TrafficRuleConfiguration> getRuleConfigurationClass() {
117         return TrafficRuleConfiguration.class;
118     }
119     
120     @Override
121     public String getType() {
122         return AlterTrafficRuleStatement.class.getName();
123     }
124 }
