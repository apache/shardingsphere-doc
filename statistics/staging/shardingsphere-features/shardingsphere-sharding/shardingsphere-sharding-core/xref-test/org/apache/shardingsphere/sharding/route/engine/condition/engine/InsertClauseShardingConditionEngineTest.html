62  @ExtendWith(MockitoExtension.class)
63  @MockitoSettings(strictness = Strictness.LENIENT)
64  class InsertClauseShardingConditionEngineTest {
65      
66      private InsertClauseShardingConditionEngine shardingConditionEngine;
67      
68      @Mock
69      private ShardingRule shardingRule;
70      
71      @Mock
72      private InsertStatementContext insertStatementContext;
73      
74      @BeforeEach
75      void setUp() {
76          ShardingSphereDatabase database = mock(ShardingSphereDatabase.class);
77          InsertStatement insertStatement = mockInsertStatement();
78          shardingConditionEngine = new InsertClauseShardingConditionEngine(database, shardingRule, new TimeServiceRule(new TimeServiceRuleConfiguration("System", new Properties())));
79          when(insertStatementContext.getSqlStatement()).thenReturn(insertStatement);
80          when(insertStatementContext.getColumnNames()).thenReturn(Collections.singletonList("foo_col"));
81          when(insertStatementContext.getInsertValueContexts()).thenReturn(Collections.singletonList(createInsertValueContext()));
82      }
83      
84      private InsertStatement mockInsertStatement() {
85          InsertStatement result = mock(InsertStatement.class);
86          when(result.getTable()).thenReturn(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("foo_table"))));
87          return result;
88      }
89      
90      private InsertValueContext createInsertValueContext() {
91          return new InsertValueContext(Collections.singletonList(new LiteralExpressionSegment(0, 10, "1")), Collections.emptyList(), 0);
92      }
93      
94      private InsertValueContext createInsertValueContextAsCommonExpressionSegmentEmptyText() {
95          return new InsertValueContext(Collections.singletonList(new CommonExpressionSegment(0, 10, "null")), Collections.emptyList(), 0);
96      }
97      
98      private InsertValueContext createInsertValueContextAsCommonExpressionSegmentWithNow() {
99          return new InsertValueContext(Collections.singletonList(new CommonExpressionSegment(0, 10, "now()")), Collections.emptyList(), 0);
100     }
101     
102     @Test
103     void assertCreateShardingConditionsInsertStatement() {
104         when(insertStatementContext.getGeneratedKeyContext()).thenReturn(Optional.empty());
105         List<ShardingCondition> shardingConditions = shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList());
106         assertThat(shardingConditions.get(0).getStartIndex(), is(0));
107         assertTrue(shardingConditions.get(0).getValues().isEmpty());
108     }
109     
110     @Test
111     void assertCreateShardingConditionsInsertStatementWithGeneratedKeyContextUsingCommonExpressionSegmentEmpty() {
112         when(insertStatementContext.getInsertValueContexts()).thenReturn(Collections.singletonList(createInsertValueContextAsCommonExpressionSegmentEmptyText()));
113         when(insertStatementContext.getGeneratedKeyContext()).thenReturn(Optional.of(mock(GeneratedKeyContext.class)));
114         when(shardingRule.findShardingColumn(any(), any())).thenReturn(Optional.of("foo_sharding_col"));
115         List<ShardingCondition> shardingConditions = shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList());
116         assertThat(shardingConditions.get(0).getStartIndex(), is(0));
117         assertThat(shardingConditions.get(0).getValues().size(), is(1));
118     }
119     
120     @Test
121     void assertCreateShardingConditionsInsertStatementWithMismatchColumns() {
122         InsertValueContext insertValueContext = new InsertValueContext(Arrays.asList(new LiteralExpressionSegment(0, 10, "1"), new LiteralExpressionSegment(0, 10, "1")), Collections.emptyList(), 0);
123         when(insertStatementContext.getInsertValueContexts()).thenReturn(Collections.singletonList(insertValueContext));
124         when(shardingRule.findShardingColumn(any(), any())).thenReturn(Optional.of("foo_sharding_col"));
125         when(insertStatementContext.getColumnNames()).thenReturn(Collections.singletonList("foo_col1"));
126         assertThrows(InsertColumnsAndValuesMismatchedException.class, () -> shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList()));
127     }
128     
129     @Test
130     void assertCreateShardingConditionsInsertStatementWithGeneratedKeyContextUsingCommonExpressionSegmentNow() {
131         when(insertStatementContext.getInsertValueContexts()).thenReturn(Collections.singletonList(createInsertValueContextAsCommonExpressionSegmentWithNow()));
132         when(insertStatementContext.getGeneratedKeyContext()).thenReturn(Optional.of(mock(GeneratedKeyContext.class)));
133         when(shardingRule.findShardingColumn(any(), any())).thenReturn(Optional.of("foo_sharding_col"));
134         List<ShardingCondition> shardingConditions = shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList());
135         assertThat(shardingConditions.get(0).getStartIndex(), is(0));
136         assertFalse(shardingConditions.get(0).getValues().isEmpty());
137     }
138     
139     @Test
140     void assertCreateShardingConditionsInsertStatementWithGeneratedKeyContext() {
141         when(insertStatementContext.getGeneratedKeyContext()).thenReturn(Optional.of(mock(GeneratedKeyContext.class)));
142         List<ShardingCondition> shardingConditions = shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList());
143         assertThat(shardingConditions.get(0).getStartIndex(), is(0));
144         assertTrue(shardingConditions.get(0).getValues().isEmpty());
145     }
146     
147     @Test
148     void assertCreateShardingConditionsInsertStatementWithGeneratedKeyContextAndTableRule() {
149         GeneratedKeyContext generatedKeyContext = mock(GeneratedKeyContext.class);
150         when(insertStatementContext.getGeneratedKeyContext()).thenReturn(Optional.of(generatedKeyContext));
151         when(generatedKeyContext.isGenerated()).thenReturn(true);
152         when(generatedKeyContext.getGeneratedValues()).thenReturn(Collections.singletonList("foo_col1"));
153         when(shardingRule.findTableRule(eq("foo_table"))).thenReturn(Optional.of(new TableRule(Collections.singletonList("foo_col"), "test")));
154         when(shardingRule.findShardingColumn(any(), any())).thenReturn(Optional.of("foo_sharding_col"));
155         List<ShardingCondition> shardingConditions = shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList());
156         assertThat(shardingConditions.get(0).getStartIndex(), is(0));
157         assertFalse(shardingConditions.get(0).getValues().isEmpty());
158     }
159     
160     @Test
161     void assertCreateShardingConditionsWithParameterMarkers() {
162         InsertValueContext insertValueContext = new InsertValueContext(Collections.singletonList(new ParameterMarkerExpressionSegment(0, 0, 0)), Collections.singletonList(1), 0);
163         when(insertStatementContext.getInsertValueContexts()).thenReturn(Collections.singletonList(insertValueContext));
164         when(shardingRule.findShardingColumn(any(), any())).thenReturn(Optional.of("foo_sharding_col"));
165         List<ShardingCondition> shardingConditions = shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.singletonList(1));
166         assertThat(shardingConditions.size(), is(1));
167         assertThat(shardingConditions.get(0).getValues().size(), is(1));
168         assertThat(shardingConditions.get(0).getValues().get(0).getParameterMarkerIndexes(), is(Collections.singletonList(0)));
169     }
170     
171     @Test
172     void assertCreateShardingConditionsSelectStatement() {
173         when(insertStatementContext.getGeneratedKeyContext()).thenReturn(Optional.empty());
174         when(insertStatementContext.getInsertSelectContext()).thenReturn(mock(InsertSelectContext.class));
175         assertTrue(shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList()).isEmpty());
176     }
177     
178     @Test
179     void assertCreateShardingConditionsSelectStatementWithGeneratedKeyContext() {
180         when(insertStatementContext.getGeneratedKeyContext()).thenReturn(Optional.of(mock(GeneratedKeyContext.class)));
181         when(insertStatementContext.getInsertSelectContext()).thenReturn(mock(InsertSelectContext.class));
182         assertTrue(shardingConditionEngine.createShardingConditions(insertStatementContext, Collections.emptyList()).isEmpty());
183     }
