18  package org.apache.shardingsphere.sharding.cosid.algorithm.sharding.interval;
19  
20  import com.google.common.collect.BoundType;
21  import com.google.common.collect.Range;
22  import me.ahoo.cosid.sharding.IntervalStep;
23  import me.ahoo.cosid.sharding.IntervalTimeline;
24  import me.ahoo.cosid.sharding.LocalDateTimeConvertor;
25  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
26  import org.apache.shardingsphere.sharding.api.sharding.standard.PreciseShardingValue;
27  import org.apache.shardingsphere.sharding.api.sharding.standard.RangeShardingValue;
28  import org.apache.shardingsphere.sharding.api.sharding.standard.StandardShardingAlgorithm;
29  import org.apache.shardingsphere.sharding.cosid.algorithm.CosIdAlgorithmConstants;
30  import org.apache.shardingsphere.sharding.exception.ShardingPluginException;
31  
32  import java.time.LocalDateTime;
33  import java.time.format.DateTimeFormatter;
34  import java.time.temporal.ChronoUnit;
35  import java.util.Collection;
36  import java.util.Properties;
37  
38  /**
39   * Abstract interval range sharding algorithm with CosId.
40   * 
41   * @param <T> type of sharding value
42   */
43  public abstract class AbstractCosIdIntervalShardingAlgorithm<T extends Comparable<?>> implements StandardShardingAlgorithm<T> {
44      
45      public static final String DEFAULT_DATE_TIME_PATTERN = "yyyy-MM-dd HH:mm:ss";
46      
47      public static final DateTimeFormatter DEFAULT_DATE_TIME_FORMATTER = DateTimeFormatter.ofPattern(DEFAULT_DATE_TIME_PATTERN);
48      
49      public static final String DATE_TIME_LOWER_KEY = "datetime-lower";
50      
51      public static final String DATE_TIME_UPPER_KEY = "datetime-upper";
52      
53      public static final String SHARDING_SUFFIX_FORMAT_KEY = "sharding-suffix-pattern";
54      
55      public static final String INTERVAL_UNIT_KEY = "datetime-interval-unit";
56      
57      public static final String INTERVAL_AMOUNT_KEY = "datetime-interval-amount";
58      
59      private IntervalTimeline intervalTimeline;
60      
61      private LocalDateTimeConvertor localDateTimeConvertor;
62      
63      @Override
64      public void init(final Properties props) {
65          intervalTimeline = createIntervalTimeline(props);
66          localDateTimeConvertor = createLocalDateTimeConvertor(props);
67      }
68      
69      private IntervalTimeline createIntervalTimeline(final Properties props) {
70          String logicNamePrefix = getRequiredValue(props, CosIdAlgorithmConstants.LOGIC_NAME_PREFIX_KEY);
71          LocalDateTime effectiveLower = LocalDateTime.parse(getRequiredValue(props, DATE_TIME_LOWER_KEY), DEFAULT_DATE_TIME_FORMATTER);
72          LocalDateTime effectiveUpper = LocalDateTime.parse(getRequiredValue(props, DATE_TIME_UPPER_KEY), DEFAULT_DATE_TIME_FORMATTER);
73          ChronoUnit stepUnit = ChronoUnit.valueOf(getRequiredValue(props, INTERVAL_UNIT_KEY));
74          int stepAmount = Integer.parseInt(props.getOrDefault(INTERVAL_AMOUNT_KEY, 1).toString());
75          DateTimeFormatter suffixFormatter = DateTimeFormatter.ofPattern(getRequiredValue(props, SHARDING_SUFFIX_FORMAT_KEY));
76          return new IntervalTimeline(logicNamePrefix, Range.closed(effectiveLower, effectiveUpper), IntervalStep.of(stepUnit, stepAmount), suffixFormatter);
77      }
78      
79      private String getRequiredValue(final Properties props, final String key) {
80          ShardingSpherePreconditions.checkState(props.containsKey(key), () -> new ShardingPluginException("%s can not be null.", key));
81          return props.getProperty(key);
82      }
83      
84      protected abstract LocalDateTimeConvertor createLocalDateTimeConvertor(Properties props);
85      
86      @Override
87      public String doSharding(final Collection<String> availableTargetNames, final PreciseShardingValue<T> shardingValue) {
88          return intervalTimeline.sharding(localDateTimeConvertor.toLocalDateTime(shardingValue.getValue()));
89      }
90      
91      @Override
92      public Collection<String> doSharding(final Collection<String> availableTargetNames, final RangeShardingValue<T> shardingValue) {
93          return intervalTimeline.sharding(toLocalDateTimeRange(shardingValue.getValueRange()));
94      }
95      
96      @SuppressWarnings("unchecked")
97      private Range<LocalDateTime> toLocalDateTimeRange(final Range<T> shardingValue) {
98          if (Range.all().equals(shardingValue)) {
99              return Range.all();
100         }
101         Object endpointValue = shardingValue.hasLowerBound() ? shardingValue.lowerEndpoint() : shardingValue.upperEndpoint();
102         if (endpointValue instanceof LocalDateTime) {
103             return (Range<LocalDateTime>) shardingValue;
104         }
105         if (shardingValue.hasLowerBound() && shardingValue.hasUpperBound()) {
106             LocalDateTime lower = localDateTimeConvertor.toLocalDateTime(shardingValue.lowerEndpoint());
107             LocalDateTime upper = localDateTimeConvertor.toLocalDateTime(shardingValue.upperEndpoint());
108             return Range.range(lower, shardingValue.lowerBoundType(), upper, shardingValue.upperBoundType());
109         }
110         if (shardingValue.hasLowerBound()) {
111             LocalDateTime lower = localDateTimeConvertor.toLocalDateTime(shardingValue.lowerEndpoint());
112             return BoundType.OPEN == shardingValue.lowerBoundType() ? Range.greaterThan(lower) : Range.atLeast(lower);
113         }
114         LocalDateTime upper = localDateTimeConvertor.toLocalDateTime(shardingValue.upperEndpoint());
115         return BoundType.OPEN == shardingValue.upperBoundType() ? Range.lessThan(upper) : Range.atMost(upper);
116     }
117 }
