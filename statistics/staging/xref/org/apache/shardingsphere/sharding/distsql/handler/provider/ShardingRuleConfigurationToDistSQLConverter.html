41  /**
42   * Sharding rule configuration to DistSQL converter.
43   */
44  public final class ShardingRuleConfigurationToDistSQLConverter implements RuleConfigurationToDistSQLConverter<ShardingRuleConfiguration> {
45      
46      @Override
47      public String convert(final ShardingRuleConfiguration ruleConfig) {
48          if (ruleConfig.getTables().isEmpty() && ruleConfig.getAutoTables().isEmpty()) {
49              return "";
50          }
51          StringBuilder result = new StringBuilder();
52          appendShardingTableRules(ruleConfig, result);
53          if (!ruleConfig.getBindingTableGroups().isEmpty()) {
54              result.append(System.lineSeparator()).append(System.lineSeparator());
55              appendShardingBindingTableRules(ruleConfig, result);
56          }
57          if (null != ruleConfig.getDefaultDatabaseShardingStrategy() || null != ruleConfig.getDefaultTableShardingStrategy()) {
58              result.append(System.lineSeparator()).append(System.lineSeparator());
59              appendDefaultShardingStrategy(ruleConfig, result);
60          }
61          return result.toString();
62      }
63      
64      private void appendShardingTableRules(final ShardingRuleConfiguration ruleConfig, final StringBuilder stringBuilder) {
65          String tableRules = getTableRules(ruleConfig);
66          String autoTableRules = getAutoTableRules(ruleConfig);
67          stringBuilder.append(ShardingDistSQLConstants.CREATE_SHARDING_TABLE).append(tableRules);
68          if (!Strings.isNullOrEmpty(tableRules) && !Strings.isNullOrEmpty(autoTableRules)) {
69              stringBuilder.append(DistSQLConstants.COMMA);
70          }
71          stringBuilder.append(autoTableRules).append(DistSQLConstants.SEMI);
72      }
73      
74      private void appendShardingBindingTableRules(final ShardingRuleConfiguration ruleConfig, final StringBuilder stringBuilder) {
75          stringBuilder.append(ShardingDistSQLConstants.SHARDING_BINDING_TABLE_RULES);
76          Iterator<ShardingTableReferenceRuleConfiguration> iterator = ruleConfig.getBindingTableGroups().iterator();
77          while (iterator.hasNext()) {
78              ShardingTableReferenceRuleConfiguration referenceRuleConfig = iterator.next();
79              stringBuilder.append(String.format(ShardingDistSQLConstants.BINDING_TABLES, referenceRuleConfig.getName(), referenceRuleConfig.getReference()));
80              if (iterator.hasNext()) {
81                  stringBuilder.append(DistSQLConstants.COMMA);
82              }
83          }
84          stringBuilder.append(DistSQLConstants.SEMI);
85      }
86      
87      private void appendDefaultShardingStrategy(final ShardingRuleConfiguration ruleConfig, final StringBuilder result) {
88          if (null != ruleConfig.getDefaultDatabaseShardingStrategy()) {
89              appendStrategy(ruleConfig.getDefaultDatabaseShardingStrategy(), ShardingDistSQLConstants.DEFAULT_DATABASE_STRATEGY, result, ruleConfig.getShardingAlgorithms());
90              result.append(DistSQLConstants.SEMI);
91          }
92          if (null != ruleConfig.getDefaultTableShardingStrategy()) {
93              if (null != ruleConfig.getDefaultDatabaseShardingStrategy()) {
94                  result.append(System.lineSeparator()).append(System.lineSeparator());
95              }
96              appendStrategy(ruleConfig.getDefaultTableShardingStrategy(), ShardingDistSQLConstants.DEFAULT_TABLE_STRATEGY, result, ruleConfig.getShardingAlgorithms());
97              result.append(DistSQLConstants.SEMI);
98          }
99      }
100     
101     private String getTableRules(final ShardingRuleConfiguration ruleConfig) {
102         StringBuilder result = new StringBuilder();
103         if (!ruleConfig.getTables().isEmpty()) {
104             Iterator<ShardingTableRuleConfiguration> iterator = ruleConfig.getTables().iterator();
105             while (iterator.hasNext()) {
106                 ShardingTableRuleConfiguration tableRuleConfig = iterator.next();
107                 result.append(String.format(ShardingDistSQLConstants.SHARDING_TABLE, tableRuleConfig.getLogicTable(), tableRuleConfig.getActualDataNodes(),
108                         appendTableStrategy(tableRuleConfig, ruleConfig)));
109                 if (iterator.hasNext()) {
110                     result.append(DistSQLConstants.COMMA);
111                 }
112             }
113         }
114         return result.toString();
115     }
116     
117     private String getAutoTableRules(final ShardingRuleConfiguration ruleConfig) {
118         StringBuilder result = new StringBuilder();
119         if (!ruleConfig.getAutoTables().isEmpty()) {
120             Iterator<ShardingAutoTableRuleConfiguration> iterator = ruleConfig.getAutoTables().iterator();
121             while (iterator.hasNext()) {
122                 ShardingAutoTableRuleConfiguration autoTableRuleConfig = iterator.next();
123                 result.append(String.format(ShardingDistSQLConstants.SHARDING_AUTO_TABLE, autoTableRuleConfig.getLogicTable(), autoTableRuleConfig.getActualDataSources(),
124                         appendAutoTableStrategy(autoTableRuleConfig, ruleConfig)));
125                 if (iterator.hasNext()) {
126                     result.append(DistSQLConstants.COMMA);
127                 }
128             }
129         }
130         return result.toString();
131     }
132     
133     private String appendTableStrategy(final ShardingTableRuleConfiguration tableRuleConfig, final ShardingRuleConfiguration ruleConfig) {
134         StringBuilder result = new StringBuilder();
135         appendStrategy(tableRuleConfig.getDatabaseShardingStrategy(), ShardingDistSQLConstants.DATABASE_STRATEGY, result, ruleConfig.getShardingAlgorithms());
136         appendStrategy(tableRuleConfig.getTableShardingStrategy(), ShardingDistSQLConstants.TABLE_STRATEGY, result, ruleConfig.getShardingAlgorithms());
137         appendKeyGenerateStrategy(ruleConfig.getKeyGenerators(), tableRuleConfig.getKeyGenerateStrategy(), result);
138         appendAuditStrategy(ruleConfig.getAuditors(), null != tableRuleConfig.getAuditStrategy() ? tableRuleConfig.getAuditStrategy() : ruleConfig.getDefaultAuditStrategy(), result);
139         return result.toString();
140     }
141     
142     private String appendAutoTableStrategy(final ShardingAutoTableRuleConfiguration autoTableRuleConfig, final ShardingRuleConfiguration ruleConfig) {
143         StringBuilder result = new StringBuilder();
144         StandardShardingStrategyConfiguration strategyConfig = (StandardShardingStrategyConfiguration) autoTableRuleConfig.getShardingStrategy();
145         String shardingColumn = Strings.isNullOrEmpty(strategyConfig.getShardingColumn()) ? ruleConfig.getDefaultShardingColumn() : strategyConfig.getShardingColumn();
146         result.append(String.format(ShardingDistSQLConstants.AUTO_TABLE_STRATEGY,
147                 shardingColumn, AlgorithmDistSQLConverter.getAlgorithmType(ruleConfig.getShardingAlgorithms().get(strategyConfig.getShardingAlgorithmName()))));
148         appendKeyGenerateStrategy(ruleConfig.getKeyGenerators(), autoTableRuleConfig.getKeyGenerateStrategy(), result);
149         appendAuditStrategy(ruleConfig.getAuditors(), null != autoTableRuleConfig.getAuditStrategy() ? autoTableRuleConfig.getAuditStrategy() : ruleConfig.getDefaultAuditStrategy(), result);
150         return result.toString();
151     }
152     
153     private void appendStrategy(final ShardingStrategyConfiguration strategyConfig, final String strategyType,
154                                 final StringBuilder stringBuilder, final Map<String, AlgorithmConfiguration> shardingAlgorithms) {
155         if (null == strategyConfig) {
156             return;
157         }
158         if (Objects.equals(strategyType, ShardingDistSQLConstants.DATABASE_STRATEGY) || Objects.equals(strategyType, ShardingDistSQLConstants.TABLE_STRATEGY)) {
159             stringBuilder.append(DistSQLConstants.COMMA).append(System.lineSeparator());
160         }
161         String type = strategyConfig.getType().toLowerCase();
162         String algorithmDefinition = AlgorithmDistSQLConverter.getAlgorithmType(shardingAlgorithms.get(strategyConfig.getShardingAlgorithmName()));
163         switch (type) {
164             case ShardingDistSQLConstants.STANDARD:
165                 StandardShardingStrategyConfiguration standardShardingStrategyConfig = (StandardShardingStrategyConfiguration) strategyConfig;
166                 stringBuilder.append(String.format(ShardingDistSQLConstants.SHARDING_STRATEGY_STANDARD, strategyType, type, standardShardingStrategyConfig.getShardingColumn(), algorithmDefinition));
167                 break;
168             case ShardingDistSQLConstants.COMPLEX:
169                 ComplexShardingStrategyConfiguration complexShardingStrategyConfig = (ComplexShardingStrategyConfiguration) strategyConfig;
170                 stringBuilder.append(String.format(ShardingDistSQLConstants.SHARDING_STRATEGY_COMPLEX, strategyType, type, complexShardingStrategyConfig.getShardingColumns(), algorithmDefinition));
171                 break;
172             case ShardingDistSQLConstants.HINT:
173                 stringBuilder.append(String.format(ShardingDistSQLConstants.SHARDING_STRATEGY_HINT, strategyType, type, algorithmDefinition));
174                 break;
175             case ShardingDistSQLConstants.NONE:
176                 stringBuilder.append(String.format(ShardingDistSQLConstants.SHARDING_STRATEGY_NONE, strategyType, "none"));
177                 break;
178             default:
179                 break;
180         }
181     }
182     
183     private void appendKeyGenerateStrategy(final Map<String, AlgorithmConfiguration> keyGenerators,
184                                            final KeyGenerateStrategyConfiguration keyGenerateStrategyConfig, final StringBuilder stringBuilder) {
185         if (null == keyGenerateStrategyConfig) {
186             return;
187         }
188         stringBuilder.append(DistSQLConstants.COMMA).append(System.lineSeparator());
189         String algorithmDefinition = AlgorithmDistSQLConverter.getAlgorithmType(keyGenerators.get(keyGenerateStrategyConfig.getKeyGeneratorName()));
190         stringBuilder.append(String.format(ShardingDistSQLConstants.KEY_GENERATOR_STRATEGY, keyGenerateStrategyConfig.getColumn(), algorithmDefinition));
191     }
192     
193     private void appendAuditStrategy(final Map<String, AlgorithmConfiguration> auditors, final ShardingAuditStrategyConfiguration auditStrategy, final StringBuilder stringBuilder) {
194         if (null != auditStrategy) {
195             stringBuilder.append(DistSQLConstants.COMMA).append(System.lineSeparator());
196             stringBuilder.append(String.format(ShardingDistSQLConstants.AUDIT_STRATEGY, getAlgorithmTypes(auditors, auditStrategy.getAuditorNames()), auditStrategy.isAllowHintDisable()));
197         }
198     }
199     
200     private String getAlgorithmTypes(final Map<String, AlgorithmConfiguration> auditors, final Collection<String> auditorNames) {
201         StringBuilder result = new StringBuilder();
202         if (!auditorNames.isEmpty()) {
203             Iterator<String> iterator = auditorNames.iterator();
204             while (iterator.hasNext()) {
205                 result.append(AlgorithmDistSQLConverter.getAlgorithmType(auditors.get(iterator.next())));
206                 if (iterator.hasNext()) {
207                     result.append(DistSQLConstants.COMMA);
208                 }
209             }
210         }
211         return result.toString();
212     }
213     
214     @Override
215     public Class<ShardingRuleConfiguration> getType() {
216         return ShardingRuleConfiguration.class;
217     }
