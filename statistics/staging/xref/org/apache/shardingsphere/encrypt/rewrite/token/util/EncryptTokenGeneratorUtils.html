42  /**
43   * Encrypt token generator utils.
44   */
45  @NoArgsConstructor(access = AccessLevel.PRIVATE)
46  public final class EncryptTokenGeneratorUtils {
47      
48      /**
49       * Judge whether all join conditions use same encryptor or not.
50       *
51       * @param joinConditions join conditions
52       * @param encryptRule encrypt rule
53       * @return whether all join conditions use same encryptor or not
54       */
55      public static boolean isAllJoinConditionsUseSameEncryptor(final Collection<BinaryOperationExpression> joinConditions, final EncryptRule encryptRule) {
56          for (BinaryOperationExpression each : joinConditions) {
57              if (!(each.getLeft() instanceof ColumnSegment) || !(each.getRight() instanceof ColumnSegment)) {
58                  continue;
59              }
60              EncryptAlgorithm leftColumnEncryptor = getColumnEncryptor(((ColumnSegment) each.getLeft()).getColumnBoundInfo(), encryptRule);
61              EncryptAlgorithm rightColumnEncryptor = getColumnEncryptor(((ColumnSegment) each.getRight()).getColumnBoundInfo(), encryptRule);
62              if (isDifferentEncryptor(leftColumnEncryptor, rightColumnEncryptor)) {
63                  return false;
64              }
65          }
66          return true;
67      }
68      
69      private static boolean isDifferentEncryptor(final EncryptAlgorithm leftColumnEncryptor, final EncryptAlgorithm rightColumnEncryptor) {
70          if (null != leftColumnEncryptor && null != rightColumnEncryptor) {
71              if (!leftColumnEncryptor.getType().equals(rightColumnEncryptor.getType())) {
72                  return true;
73              }
74              return !leftColumnEncryptor.equals(rightColumnEncryptor);
75          }
76          return null != leftColumnEncryptor || null != rightColumnEncryptor;
77      }
78      
79      private static EncryptAlgorithm getColumnEncryptor(final ColumnSegmentBoundInfo columnBoundInfo, final EncryptRule encryptRule) {
80          String tableName = columnBoundInfo.getOriginalTable().getValue();
81          String columnName = columnBoundInfo.getOriginalColumn().getValue();
82          if (!encryptRule.findEncryptTable(tableName).isPresent() || !encryptRule.getEncryptTable(tableName).isEncryptColumn(columnName)) {
83              return null;
84          }
85          EncryptTable encryptTable = encryptRule.getEncryptTable(tableName);
86          EncryptColumn encryptColumn = encryptTable.getEncryptColumn(columnName);
87          if (encryptColumn.getAssistedQuery().isPresent()) {
88              return encryptColumn.getAssistedQuery().get().getEncryptor();
89          }
90          return encryptColumn.getCipher().getEncryptor();
91      }
92      
93      /**
94       * Judge whether all insert select columns use same encryptor or not.
95       *
96       * @param insertColumns insert columns
97       * @param projections projections
98       * @param encryptRule encrypt rule
99       * @return whether all insert select columns use same encryptor or not 
100      */
101     public static boolean isAllInsertSelectColumnsUseSameEncryptor(final Collection<ColumnSegment> insertColumns, final Collection<Projection> projections, final EncryptRule encryptRule) {
102         Iterator<ColumnSegment> insertColumnsIterator = insertColumns.iterator();
103         Iterator<Projection> projectionIterator = projections.iterator();
104         while (insertColumnsIterator.hasNext()) {
105             ColumnSegment columnSegment = insertColumnsIterator.next();
106             EncryptAlgorithm leftColumnEncryptor = getColumnEncryptor(columnSegment.getColumnBoundInfo(), encryptRule);
107             Projection projection = projectionIterator.next();
108             ColumnSegmentBoundInfo columnBoundInfo = projection instanceof ColumnProjection
109                     ? new ColumnSegmentBoundInfo(null, null, ((ColumnProjection) projection).getOriginalTable(), ((ColumnProjection) projection).getOriginalColumn())
110                     : new ColumnSegmentBoundInfo(new IdentifierValue(projection.getColumnLabel()));
111             EncryptAlgorithm rightColumnEncryptor = getColumnEncryptor(columnBoundInfo, encryptRule);
112             if (isDifferentEncryptor(leftColumnEncryptor, rightColumnEncryptor)) {
113                 return false;
114             }
115         }
116         return true;
117     }
118     
119     /**
120      * Judge whether contains encrypt projection in combine statement or not.
121      *
122      * @param selectStatementContext select statement context
123      * @param encryptRule encrypt rule
124      * @return whether contains encrypt projection in combine statement or not
125      */
126     public static boolean isContainsEncryptProjectionInCombineStatement(final SelectStatementContext selectStatementContext, final EncryptRule encryptRule) {
127         if (!selectStatementContext.getSqlStatement().getCombine().isPresent()) {
128             return false;
129         }
130         CombineSegment combineSegment = selectStatementContext.getSqlStatement().getCombine().get();
131         List<Projection> leftProjections = new ArrayList<>(selectStatementContext.getSubqueryContexts().get(combineSegment.getLeft().getStartIndex()).getProjectionsContext().getExpandProjections());
132         List<Projection> rightProjections = new ArrayList<>(selectStatementContext.getSubqueryContexts().get(combineSegment.getRight().getStartIndex()).getProjectionsContext().getExpandProjections());
133         ShardingSpherePreconditions.checkState(leftProjections.size() == rightProjections.size(), () -> new UnsupportedSQLOperationException("Column projections must be same for combine statement"));
134         for (int i = 0; i < leftProjections.size(); i++) {
135             Projection leftProjection = leftProjections.get(i);
136             Projection rightProjection = rightProjections.get(i);
137             ColumnSegmentBoundInfo leftColumnBoundInfo = leftProjection instanceof ColumnProjection
138                     ? new ColumnSegmentBoundInfo(null, null, ((ColumnProjection) leftProjection).getOriginalTable(), ((ColumnProjection) leftProjection).getOriginalColumn())
139                     : new ColumnSegmentBoundInfo(new IdentifierValue(leftProjection.getColumnLabel()));
140             ColumnSegmentBoundInfo rightColumnBoundInfo = rightProjection instanceof ColumnProjection
141                     ? new ColumnSegmentBoundInfo(null, null, ((ColumnProjection) rightProjection).getOriginalTable(), ((ColumnProjection) rightProjection).getOriginalColumn())
142                     : new ColumnSegmentBoundInfo(new IdentifierValue(rightProjection.getColumnLabel()));
143             EncryptAlgorithm leftColumnEncryptor = getColumnEncryptor(leftColumnBoundInfo, encryptRule);
144             EncryptAlgorithm rightColumnEncryptor = getColumnEncryptor(rightColumnBoundInfo, encryptRule);
145             if (null != leftColumnEncryptor || null != rightColumnEncryptor) {
146                 return true;
147             }
148         }
149         return false;
150     }
