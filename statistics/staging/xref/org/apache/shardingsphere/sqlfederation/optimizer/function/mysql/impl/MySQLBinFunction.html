18  package org.apache.shardingsphere.sqlfederation.optimizer.function.mysql.impl;
19  
20  import lombok.AccessLevel;
21  import lombok.NoArgsConstructor;
22  
23  import java.math.BigInteger;
24  
25  /**
26   * MySQL bin function.
27   */
28  @NoArgsConstructor(access = AccessLevel.PRIVATE)
29  public final class MySQLBinFunction {
30      
31      /**
32       * Bin.
33       *
34       * @param value value
35       * @return binary string
36       */
37      @SuppressWarnings("unused")
38      public static String bin(final Object value) {
39          if (null == value) {
40              return null;
41          }
42          if (value instanceof Number && !(value instanceof BigInteger)) {
43              return Long.toBinaryString(((Number) value).longValue());
44          }
45          BigInteger bigIntegerValue = value instanceof BigInteger ? (BigInteger) value : new BigInteger(getFirstNumbers(String.valueOf(value)));
46          return -1 == bigIntegerValue.signum() ? bigIntegerValue.add(BigInteger.ONE.shiftLeft(Long.SIZE).subtract(BigInteger.ONE)).add(BigInteger.ONE).toString(2) : bigIntegerValue.toString(2);
47      }
48      
49      private static String getFirstNumbers(final String value) {
50          if (value.isEmpty()) {
51              return "0";
52          }
53          boolean isNegative = '-' == value.charAt(0);
54          StringBuilder result = new StringBuilder();
55          if (isNegative) {
56              result.append('-');
57          }
58          for (int i = isNegative ? 1 : 0; i < value.length(); i++) {
59              if (Character.isDigit(value.charAt(i))) {
60                  result.append(value.charAt(i));
61              } else {
62                  break;
63              }
64          }
65          if (0 == result.length() || isNegative && 1 == result.length()) {
66              return "0";
67          }
68          return result.toString();
69      }
70  }
