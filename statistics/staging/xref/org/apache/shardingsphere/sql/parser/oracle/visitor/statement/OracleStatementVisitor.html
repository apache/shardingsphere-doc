162     @Override
163     public final ASTNode visitLiterals(final LiteralsContext ctx) {
164         if (null != ctx.stringLiterals()) {
165             return visit(ctx.stringLiterals());
166         }
167         if (null != ctx.numberLiterals()) {
168             return visit(ctx.numberLiterals());
169         }
170         if (null != ctx.hexadecimalLiterals()) {
171             return visit(ctx.hexadecimalLiterals());
172         }
173         if (null != ctx.bitValueLiterals()) {
174             return visit(ctx.bitValueLiterals());
175         }
176         if (null != ctx.booleanLiterals()) {
177             return visit(ctx.booleanLiterals());
178         }
179         if (null != ctx.nullValueLiterals()) {
180             return visit(ctx.nullValueLiterals());
181         }
182         if (null != ctx.dateTimeLiterals()) {
183             return visit(ctx.dateTimeLiterals());
184         }
185         throw new IllegalStateException("Literals must have string, number, dateTime, hex, bit, boolean or null.");
186     }

322     @Override
323     public final ASTNode visitExpr(final ExprContext ctx) {
324         if (null != ctx.booleanPrimary()) {
325             return visit(ctx.booleanPrimary());
326         }
327         if (null != ctx.LP_()) {
328             return visit(ctx.expr(0));
329         }
330         if (null != ctx.andOperator()) {
331             return createBinaryOperationExpression(ctx, ctx.andOperator().getText());
332         }
333         if (null != ctx.orOperator()) {
334             return createBinaryOperationExpression(ctx, ctx.orOperator().getText());
335         }
336         if (null != ctx.datetimeExpr()) {
337             return createDatetimeExpression(ctx, ctx.datetimeExpr());
338         }
339         return new NotExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), (ExpressionSegment) visit(ctx.expr(0)), false);
340     }
341     
342     private ASTNode createDatetimeExpression(final ExprContext ctx, final DatetimeExprContext datetimeExpr) {
343         ExpressionSegment left = (ExpressionSegment) visit(ctx.expr(0));
344         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
345         if (null != datetimeExpr.expr()) {
346             ExpressionSegment right = new ExpressionProjectionSegment(datetimeExpr.getStart().getStartIndex(),
347                     datetimeExpr.getStop().getStopIndex(), datetimeExpr.getText(), (ExpressionSegment) visit(datetimeExpr.expr()));
348             return new DatetimeExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, text);
349         }
350         return new DatetimeExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, text);
351     }
352     
353     private ASTNode createBinaryOperationExpression(final ExprContext ctx, final String operator) {
354         ExpressionSegment left = (ExpressionSegment) visit(ctx.expr(0));
355         ExpressionSegment right = (ExpressionSegment) visit(ctx.expr(1));
356         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
357         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
358     }
359     
360     @Override
361     public final ASTNode visitBooleanPrimary(final BooleanPrimaryContext ctx) {
362         if (null == ctx.IS()) {
363             return null == ctx.comparisonOperator() && null == ctx.SAFE_EQ_() ? visit(ctx.predicate()) : createCompareSegment(ctx);
364         }
365         String rightText = "";
366         if (null != ctx.NOT()) {
367             rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(ctx.NOT().getSymbol().getStartIndex(), ctx.NOT().getSymbol().getStopIndex()))).concat(" ");
368         }
369         Token operatorToken = null;
370         if (null != ctx.NULL()) {
371             operatorToken = ctx.NULL().getSymbol();
372         }
373         if (null != ctx.TRUE()) {
374             operatorToken = ctx.TRUE().getSymbol();
375         }
376         if (null != ctx.FALSE()) {
377             operatorToken = ctx.FALSE().getSymbol();
378         }
379         int startIndex = null == operatorToken ? ctx.IS().getSymbol().getStopIndex() + 2 : operatorToken.getStartIndex();
380         rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(startIndex, ctx.stop.getStopIndex())));
381         ExpressionSegment right = new LiteralExpressionSegment(ctx.IS().getSymbol().getStopIndex() + 2, ctx.stop.getStopIndex(), rightText);
382         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
383         ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
384         String operator = "IS";
385         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
386     }
387     
388     private ASTNode createCompareSegment(final BooleanPrimaryContext ctx) {
389         ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
390         String operator = null == ctx.SAFE_EQ_() ? ctx.comparisonOperator().getText() : ctx.SAFE_EQ_().getText();
391         ExpressionSegment right = null != ctx.predicate() ? (ExpressionSegment) visit(ctx.predicate()) : (ExpressionSegment) visit(ctx.subquery());
392         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
393         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
394     }
395     
396     @Override
397     public final ASTNode visitPredicate(final PredicateContext ctx) {
398         if (null != ctx.IN()) {
399             return createInSegment(ctx);
400         }
401         if (null != ctx.BETWEEN()) {
402             return createBetweenSegment(ctx);
403         }
404         if (null != ctx.LIKE()) {
405             return createBinaryOperationExpressionFromLike(ctx);
406         }
407         return visit(ctx.bitExpr(0));
408     }
409     
410     private InExpression createInSegment(final PredicateContext ctx) {
411         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
412         ExpressionSegment right;
413         if (null == ctx.subquery()) {
414             ListExpression listExpression = new ListExpression(ctx.LP_().getSymbol().getStartIndex(), ctx.RP_().getSymbol().getStopIndex());
415             for (ExprContext each : ctx.expr()) {
416                 listExpression.getItems().add((ExpressionSegment) visit(each));
417             }
418             right = listExpression;
419         } else {
420             right = new SubqueryExpressionSegment(new SubquerySegment(ctx.subquery().start.getStartIndex(), ctx.subquery().stop.getStopIndex(), (OracleSelectStatement) visit(ctx.subquery())));
421         }
422         boolean not = null != ctx.NOT();
423         return new InExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, not);
424     }
425     
426     private BinaryOperationExpression createBinaryOperationExpressionFromLike(final PredicateContext ctx) {
427         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
428         ListExpression right = new ListExpression(ctx.simpleExpr(0).start.getStartIndex(), ctx.simpleExpr().get(ctx.simpleExpr().size() - 1).stop.getStopIndex());
429         for (SimpleExprContext each : ctx.simpleExpr()) {
430             right.getItems().add((ExpressionSegment) visit(each));
431         }
432         String operator = null != ctx.NOT() ? "NOT LIKE" : "LIKE";
433         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
434         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
435     }

457     private ASTNode createExpressionSegment(final ASTNode astNode, final ParserRuleContext context) {
458         if (astNode instanceof StringLiteralValue) {
459             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((StringLiteralValue) astNode).getValue());
460         }
461         if (astNode instanceof NumberLiteralValue) {
462             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((NumberLiteralValue) astNode).getValue());
463         }
464         if (astNode instanceof BooleanLiteralValue) {
465             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((BooleanLiteralValue) astNode).getValue());
466         }
467         if (astNode instanceof ParameterMarkerValue) {
468             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) astNode;
469             ParameterMarkerExpressionSegment segment = new ParameterMarkerExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(),
470                     parameterMarker.getValue(), parameterMarker.getType());
471             parameterMarkerSegments.add(segment);
472             return segment;
473         }
474         if (astNode instanceof SubquerySegment) {
475             return new SubqueryExpressionSegment((SubquerySegment) astNode);
476         }
477         if (astNode instanceof OtherLiteralValue) {
478             return new CommonExpressionSegment(context.getStart().getStartIndex(), context.getStop().getStopIndex(), context.getText());
479         }
480         return astNode;
481     }

483     @Override
484     public final ASTNode visitSimpleExpr(final SimpleExprContext ctx) {
485         int startIndex = ctx.getStart().getStartIndex();
486         int stopIndex = ctx.getStop().getStopIndex();
487         if (null != ctx.subquery()) {
488             return new SubquerySegment(startIndex, stopIndex, (OracleSelectStatement) visit(ctx.subquery()));
489         }
490         if (null != ctx.parameterMarker()) {
491             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) visit(ctx.parameterMarker());
492             ParameterMarkerExpressionSegment segment = new ParameterMarkerExpressionSegment(startIndex, stopIndex, parameterMarker.getValue(), parameterMarker.getType());
493             parameterMarkerSegments.add(segment);
494             return segment;
495         }
496         if (null != ctx.literals()) {
497             return SQLUtils.createLiteralExpression(visit(ctx.literals()), startIndex, stopIndex, ctx.literals().start.getInputStream().getText(new Interval(startIndex, stopIndex)));
498         }
499         if (null != ctx.functionCall()) {
500             return visit(ctx.functionCall());
501         }
502         if (null != ctx.columnName()) {
503             return visit(ctx.columnName());
504         }
505         return new CommonExpressionSegment(startIndex, stopIndex, ctx.getText());
506     }

545     private ASTNode createAggregationSegment(final AggregationFunctionContext ctx, final String aggregationType) {
546         AggregationType type = AggregationType.valueOf(aggregationType.toUpperCase());
547         String innerExpression = ctx.start.getInputStream().getText(new Interval(ctx.LP_().get(0).getSymbol().getStartIndex(), ctx.stop.getStopIndex()));
548         if (null != ctx.DISTINCT()) {
549             AggregationDistinctProjectionSegment result = new AggregationDistinctProjectionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(),
550                     type, innerExpression, getDistinctExpression(ctx));
551             result.getParameters().addAll(getExpressions(ctx));
552             return result;
553         }
554         AggregationProjectionSegment result = new AggregationProjectionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), type, innerExpression);
555         result.getParameters().addAll(getExpressions(ctx));
556         return result;
557     }

788     public final ASTNode visitOrderByItem(final OrderByItemContext ctx) {
789         OrderDirection orderDirection = null != ctx.DESC() ? OrderDirection.DESC : OrderDirection.ASC;
790         NullsOrderType nullsOrderType = generateNullsOrderType(ctx);
791         if (null != ctx.columnName()) {
792             ColumnSegment column = (ColumnSegment) visit(ctx.columnName());
793             return new ColumnOrderByItemSegment(column, orderDirection, nullsOrderType);
794         }
795         if (null != ctx.numberLiterals()) {
796             return new IndexOrderByItemSegment(ctx.numberLiterals().getStart().getStartIndex(), ctx.numberLiterals().getStop().getStopIndex(),

798         }
799         return new ExpressionOrderByItemSegment(ctx.expr().getStart().getStartIndex(), ctx.expr().getStop().getStopIndex(),
800                 getOriginalText(ctx.expr()), orderDirection, nullsOrderType, (ExpressionSegment) visit(ctx.expr()));
801     }
