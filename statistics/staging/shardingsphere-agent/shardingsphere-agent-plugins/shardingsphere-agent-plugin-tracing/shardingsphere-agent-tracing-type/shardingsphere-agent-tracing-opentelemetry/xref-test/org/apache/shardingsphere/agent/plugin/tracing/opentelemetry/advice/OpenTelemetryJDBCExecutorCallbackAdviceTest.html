63  class OpenTelemetryJDBCExecutorCallbackAdviceTest {
64      
65      public static final String DATA_SOURCE_NAME = "mock.db";
66      
67      public static final String SQL = "SELECT 1";
68      
69      private static final String DB_TYPE = "MySQL";
70      
71      private final InMemorySpanExporter testExporter = InMemorySpanExporter.create();
72      
73      private Span parentSpan;
74      
75      private TargetAdviceObject targetObject;
76      
77      private JDBCExecutionUnit executionUnit;
78      
79      @BeforeEach
80      void setup() {
81          SdkTracerProvider tracerProvider = SdkTracerProvider.builder().addSpanProcessor(SimpleSpanProcessor.create(testExporter)).build();
82          OpenTelemetrySdk.builder().setTracerProvider(tracerProvider).buildAndRegisterGlobal().getTracer(OpenTelemetryConstants.TRACER_NAME);
83          parentSpan = GlobalOpenTelemetry.getTracer(OpenTelemetryConstants.TRACER_NAME).spanBuilder("parent").startSpan();
84          RootSpanContext.set(parentSpan);
85          prepare();
86      }
87      
88      @SuppressWarnings({"rawtypes", "unchecked"})
89      @SneakyThrows({ReflectiveOperationException.class, SQLException.class})
90      private void prepare() {
91          Statement statement = mock(Statement.class);
92          Connection connection = mock(Connection.class);
93          DatabaseMetaData databaseMetaData = mock(DatabaseMetaData.class);
94          when(databaseMetaData.getURL()).thenReturn("mock_url");
95          when(connection.getMetaData()).thenReturn(databaseMetaData);
96          when(statement.getConnection()).thenReturn(connection);
97          executionUnit = new JDBCExecutionUnit(new ExecutionUnit(DATA_SOURCE_NAME, new SQLUnit(SQL, Collections.emptyList())), null, statement);
98          JDBCExecutorCallback jdbcExecutorCallback = new JDBCExecutorCallbackFixture(new MySQLDatabaseType(), Collections.singletonMap("ds", new MySQLDatabaseType()),
99                  new MySQLSelectStatement(), true);
100         Map<String, DataSourceMetaData> cachedDatasourceMetaData = (Map<String, DataSourceMetaData>) Plugins.getMemberAccessor()
101                 .get(JDBCExecutorCallback.class.getDeclaredField("CACHED_DATASOURCE_METADATA"), jdbcExecutorCallback);
102         cachedDatasourceMetaData.put("mock_url", mock(DataSourceMetaData.class));
103         Map<String, DatabaseType> storageTypes = Collections.singletonMap(DATA_SOURCE_NAME, new MySQLDatabaseType());
104         Plugins.getMemberAccessor().set(JDBCExecutorCallback.class.getDeclaredField("storageTypes"), jdbcExecutorCallback, storageTypes);
105         targetObject = (TargetAdviceObject) jdbcExecutorCallback;
106     }
107     
108     @AfterEach
109     void clean() {
110         GlobalOpenTelemetry.resetForTest();
111         parentSpan.end();
112         testExporter.reset();
113     }
114     
115     @Test
116     void assertMethod() {
117         OpenTelemetryJDBCExecutorCallbackAdvice advice = new OpenTelemetryJDBCExecutorCallbackAdvice();
118         advice.beforeMethod(targetObject, null, new Object[]{executionUnit, false}, "OpenTelemetry");
119         advice.afterMethod(targetObject, null, new Object[]{executionUnit, false}, null, "OpenTelemetry");
120         List<SpanData> spanItems = testExporter.getFinishedSpanItems();
121         assertCommonData(spanItems);
122         assertThat(spanItems.iterator().next().getStatus().getStatusCode(), is(StatusCode.OK));
123     }
124     
125     @Test
126     void assertExceptionHandle() {
127         OpenTelemetryJDBCExecutorCallbackAdvice advice = new OpenTelemetryJDBCExecutorCallbackAdvice();
128         advice.beforeMethod(targetObject, null, new Object[]{executionUnit, false}, "OpenTelemetry");
129         advice.onThrowing(targetObject, null, new Object[]{executionUnit, false}, new IOException(), "OpenTelemetry");
130         List<SpanData> spanItems = testExporter.getFinishedSpanItems();
131         assertCommonData(spanItems);
132         assertThat(spanItems.iterator().next().getStatus().getStatusCode(), is(StatusCode.ERROR));
133     }
134     
135     private void assertCommonData(final List<SpanData> spanItems) {
136         assertThat(spanItems.size(), is(1));
137         SpanData spanData = spanItems.iterator().next();
138         assertThat(spanData.getName(), is("/ShardingSphere/executeSQL/"));
139         Attributes attributes = spanData.getAttributes();
140         assertThat(attributes.get(AttributeKey.stringKey(AttributeConstants.COMPONENT)), is(AttributeConstants.COMPONENT_NAME));
141         assertThat(attributes.get(AttributeKey.stringKey(AttributeConstants.DB_TYPE)), is(DB_TYPE));
142         assertThat(attributes.get(AttributeKey.stringKey(AttributeConstants.DB_INSTANCE)), is(DATA_SOURCE_NAME));
143         assertThat(attributes.get(AttributeKey.stringKey(AttributeConstants.DB_STATEMENT)), is(SQL));
144     }
