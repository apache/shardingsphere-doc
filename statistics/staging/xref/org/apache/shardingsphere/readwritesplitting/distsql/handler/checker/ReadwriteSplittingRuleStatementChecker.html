52  /**
53   * Readwrite-splitting rule statement checker.
54   */
55  @NoArgsConstructor(access = AccessLevel.PRIVATE)
56  public final class ReadwriteSplittingRuleStatementChecker {
57      
58      /**
59       * Check create readwrite-splitting rule statement.
60       *
61       * @param database database
62       * @param segments segments
63       * @param currentRuleConfig current rule config
64       * @param ifNotExists rule if not exists
65       */
66      public static void checkCreation(final ShardingSphereDatabase database, final Collection<ReadwriteSplittingRuleSegment> segments,
67                                       final ReadwriteSplittingRuleConfiguration currentRuleConfig, final boolean ifNotExists) {
68          checkDuplicateRuleNames(database, segments, currentRuleConfig, ifNotExists);
69          String databaseName = database.getName();
70          checkDataSourcesExist(databaseName, segments, database);
71          checkDuplicatedDataSourceNames(databaseName, segments, currentRuleConfig, true);
72          checkTransactionalReadQueryStrategy(segments);
73          checkLoadBalancers(segments);
74      }
75      
76      /**
77       * Check alter readwrite-splitting rule statement.
78       *
79       * @param database database
80       * @param segments segments
81       * @param currentRuleConfig current rule config
82       */
83      public static void checkAlteration(final ShardingSphereDatabase database, final Collection<ReadwriteSplittingRuleSegment> segments, final ReadwriteSplittingRuleConfiguration currentRuleConfig) {
84          String databaseName = database.getName();
85          checkDuplicateRuleNamesWithSelf(databaseName, segments);
86          checkRuleNamesExist(segments, currentRuleConfig, databaseName);
87          checkDataSourcesExist(databaseName, segments, database);
88          checkDuplicatedDataSourceNames(databaseName, segments, currentRuleConfig, false);
89          checkTransactionalReadQueryStrategy(segments);
90          checkLoadBalancers(segments);
91      }
92      
93      private static void checkRuleNamesExist(final Collection<ReadwriteSplittingRuleSegment> segments, final ReadwriteSplittingRuleConfiguration currentRuleConfig, final String databaseName) {
94          Collection<String> requiredRuleNames = segments.stream().map(ReadwriteSplittingRuleSegment::getName).collect(Collectors.toList());
95          Collection<String> currentRuleNames = currentRuleConfig.getDataSourceGroups().stream().map(ReadwriteSplittingDataSourceGroupRuleConfiguration::getName).collect(Collectors.toList());
96          Collection<String> notExistedRuleNames = requiredRuleNames.stream().filter(each -> !currentRuleNames.contains(each)).collect(Collectors.toSet());
97          ShardingSpherePreconditions.checkMustEmpty(notExistedRuleNames, () -> new MissingRequiredRuleException("Readwrite-splitting", databaseName, notExistedRuleNames));
98      }
99      
100     private static void checkDuplicateRuleNames(final ShardingSphereDatabase database,
101                                                 final Collection<ReadwriteSplittingRuleSegment> segments, final ReadwriteSplittingRuleConfiguration currentRuleConfig, final boolean ifNotExists) {
102         checkDuplicateRuleNamesWithSelf(database.getName(), segments);
103         checkDuplicateRuleNamesWithExistsDataSources(database, segments);
104         if (!ifNotExists) {
105             checkDuplicateRuleNamesWithRuleConfiguration(database.getName(), currentRuleConfig, segments);
106         }
107     }
108     
109     private static void checkDuplicateRuleNamesWithSelf(final String databaseName, final Collection<ReadwriteSplittingRuleSegment> segments) {
110         Collection<String> duplicatedRuleNames = getDuplicated(segments.stream().map(ReadwriteSplittingRuleSegment::getName).collect(Collectors.toList()));
111         ShardingSpherePreconditions.checkMustEmpty(duplicatedRuleNames, () -> new DuplicateRuleException("Readwrite-splitting", databaseName, duplicatedRuleNames));
112     }
113     
114     private static Collection<String> getDuplicated(final Collection<String> required) {
115         return required.stream().collect(Collectors.groupingBy(each -> each, Collectors.counting())).entrySet().stream()
116                 .filter(each -> each.getValue() > 1L).map(Entry::getKey).collect(Collectors.toSet());
117     }
118     
119     private static void checkDuplicateRuleNamesWithExistsDataSources(final ShardingSphereDatabase database, final Collection<ReadwriteSplittingRuleSegment> segments) {
120         Collection<String> currentRuleNames = new HashSet<>();
121         ResourceMetaData resourceMetaData = database.getResourceMetaData();
122         if (null != resourceMetaData && null != resourceMetaData.getStorageUnits()) {
123             currentRuleNames.addAll(resourceMetaData.getStorageUnits().keySet());
124         }
125         currentRuleNames.addAll(getLogicDataSources(database));
126         Collection<String> toBeCreatedRuleNames = segments.stream().map(ReadwriteSplittingRuleSegment::getName).filter(currentRuleNames::contains).collect(Collectors.toList());
127         ShardingSpherePreconditions.checkMustEmpty(toBeCreatedRuleNames, () -> new InvalidRuleConfigurationException("Readwrite-splitting", toBeCreatedRuleNames,
128                 Collections.singleton(String.format("%s already exists in storage unit", toBeCreatedRuleNames))));
129     }
130     
131     private static void checkDuplicateRuleNamesWithRuleConfiguration(final String databaseName, final ReadwriteSplittingRuleConfiguration currentRuleConfig,
132                                                                      final Collection<ReadwriteSplittingRuleSegment> segments) {
133         Collection<String> currentRuleNames = new LinkedList<>();
134         if (null != currentRuleConfig) {
135             currentRuleNames.addAll(currentRuleConfig.getDataSourceGroups().stream().map(ReadwriteSplittingDataSourceGroupRuleConfiguration::getName).collect(Collectors.toList()));
136         }
137         Collection<String> toBeCreatedRuleNames = segments.stream().map(ReadwriteSplittingRuleSegment::getName).filter(currentRuleNames::contains).collect(Collectors.toList());
138         ShardingSpherePreconditions.checkMustEmpty(toBeCreatedRuleNames, () -> new DuplicateRuleException("Readwrite-splitting", databaseName, toBeCreatedRuleNames));
139     }
140     
141     private static void checkDataSourcesExist(final String databaseName, final Collection<ReadwriteSplittingRuleSegment> segments, final ShardingSphereDatabase database) {
142         Collection<String> requiredDataSources = new LinkedHashSet<>();
143         segments.forEach(each -> {
144             requiredDataSources.add(each.getWriteDataSource());
145             requiredDataSources.addAll(each.getReadDataSources());
146         });
147         Collection<String> notExistedDataSources = database.getResourceMetaData().getNotExistedDataSources(requiredDataSources);
148         ShardingSpherePreconditions.checkMustEmpty(notExistedDataSources, () -> new MissingRequiredStorageUnitsException(databaseName, notExistedDataSources));
149     }
150     
151     private static Collection<String> getLogicDataSources(final ShardingSphereDatabase database) {
152         Collection<String> result = new LinkedHashSet<>();
153         for (DataSourceMapperRuleAttribute each : database.getRuleMetaData().getAttributes(DataSourceMapperRuleAttribute.class)) {
154             result.addAll(each.getDataSourceMapper().keySet());
155         }
156         return result;
157     }
158     
159     private static void checkDuplicatedDataSourceNames(final String databaseName, final Collection<ReadwriteSplittingRuleSegment> segments,
160                                                        final ReadwriteSplittingRuleConfiguration currentRuleConfig, final boolean isCreating) {
161         Collection<String> existedWriteDataSourceNames = new HashSet<>();
162         Collection<String> existedReadDataSourceNames = new HashSet<>();
163         if (null != currentRuleConfig) {
164             Collection<String> toBeAlteredRuleNames = isCreating ? Collections.emptySet() : getToBeAlteredRuleNames(segments);
165             for (ReadwriteSplittingDataSourceGroupRuleConfiguration each : currentRuleConfig.getDataSourceGroups()) {
166                 if (toBeAlteredRuleNames.contains(each.getName())) {
167                     continue;
168                 }
169                 existedWriteDataSourceNames.add(each.getWriteDataSourceName());
170                 existedReadDataSourceNames.addAll(each.getReadDataSourceNames());
171             }
172         }
173         checkDuplicateWriteDataSourceNames(segments, databaseName, existedWriteDataSourceNames);
174         checkDuplicateReadDataSourceNames(segments, databaseName, existedReadDataSourceNames);
175     }
176     
177     private static Collection<String> getToBeAlteredRuleNames(final Collection<ReadwriteSplittingRuleSegment> segments) {
178         return segments.stream().map(ReadwriteSplittingRuleSegment::getName).collect(Collectors.toSet());
179     }
180     
181     private static void checkDuplicateWriteDataSourceNames(final Collection<ReadwriteSplittingRuleSegment> segments, final String databaseName,
182                                                            final Collection<String> writeDataSourceNames) {
183         for (ReadwriteSplittingRuleSegment each : segments) {
184             if (Strings.isNullOrEmpty(each.getWriteDataSource())) {
185                 continue;
186             }
187             String writeDataSource = each.getWriteDataSource();
188             ShardingSpherePreconditions.checkState(writeDataSourceNames.add(writeDataSource), () -> new DuplicateReadwriteSplittingActualDataSourceException(
189                     ReadwriteSplittingDataSourceType.WRITE, writeDataSource, new ReadwriteSplittingRuleExceptionIdentifier(databaseName, "")));
190         }
191     }
192     
193     private static void checkDuplicateReadDataSourceNames(final Collection<ReadwriteSplittingRuleSegment> segments, final String databaseName,
194                                                           final Collection<String> readDataSourceNames) {
195         for (ReadwriteSplittingRuleSegment each : segments) {
196             if (null != each.getReadDataSources()) {
197                 checkDuplicateReadDataSourceNames(each, databaseName, readDataSourceNames);
198             }
199         }
200     }
201     
202     private static void checkDuplicateReadDataSourceNames(final ReadwriteSplittingRuleSegment segment, final String databaseName,
203                                                           final Collection<String> readDataSourceNames) {
204         for (String each : segment.getReadDataSources()) {
205             ShardingSpherePreconditions.checkState(readDataSourceNames.add(each), () -> new DuplicateReadwriteSplittingActualDataSourceException(
206                     ReadwriteSplittingDataSourceType.READ, each, new ReadwriteSplittingRuleExceptionIdentifier(databaseName, "")));
207         }
208     }
209     
210     private static void checkTransactionalReadQueryStrategy(final Collection<ReadwriteSplittingRuleSegment> segments) {
211         Collection<String> validStrategyNames = Arrays.stream(TransactionalReadQueryStrategy.values()).map(Enum::name).collect(Collectors.toSet());
212         for (ReadwriteSplittingRuleSegment each : segments) {
213             if (null != each.getTransactionalReadQueryStrategy()) {
214                 ShardingSpherePreconditions.checkContains(validStrategyNames, each.getTransactionalReadQueryStrategy().toUpperCase(),
215                         () -> new MissingRequiredStrategyException("Transactional read query", Collections.singleton(each.getTransactionalReadQueryStrategy())));
216             }
217         }
218     }
219     
220     private static void checkLoadBalancers(final Collection<ReadwriteSplittingRuleSegment> segments) {
221         segments.stream().map(ReadwriteSplittingRuleSegment::getLoadBalancer).filter(Objects::nonNull)
222                 .forEach(each -> TypedSPILoader.checkService(LoadBalanceAlgorithm.class, each.getName(), each.getProps()));
223     }
