199     @Override
200     public final ASTNode visitExpr(final ClickHouseStatementParser.ExprContext ctx) {
201         if (null != ctx.booleanPrimary()) {
202             return visit(ctx.booleanPrimary());
203         }
204         if (null != ctx.LP_()) {
205             return visit(ctx.expr(0));
206         }
207         if (null != ctx.andOperator()) {
208             return createBinaryOperationExpression(ctx, ctx.andOperator().getText());
209         }
210         if (null != ctx.orOperator()) {
211             return createBinaryOperationExpression(ctx, ctx.orOperator().getText());
212         }
213         return new NotExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), (ExpressionSegment) visit(ctx.expr(0)), false);
214     }

216     private ASTNode createBinaryOperationExpression(final ClickHouseStatementParser.ExprContext ctx, final String operator) {
217         ExpressionSegment left = (ExpressionSegment) visit(ctx.expr(0));
218         ExpressionSegment right = (ExpressionSegment) visit(ctx.expr(1));
219         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
220         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
221     }

223     @Override
224     public final ASTNode visitBooleanPrimary(final ClickHouseStatementParser.BooleanPrimaryContext ctx) {
225         if (null != ctx.IS()) {
226             String rightText = "";
227             if (null != ctx.NOT()) {
228                 rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(ctx.NOT().getSymbol().getStartIndex(),
229                         ctx.NOT().getSymbol().getStopIndex()))).concat(" ");
230             }
231             Token operatorToken = null;
232             if (null != ctx.NULL()) {
233                 operatorToken = ctx.NULL().getSymbol();
234             }
235             if (null != ctx.TRUE()) {
236                 operatorToken = ctx.TRUE().getSymbol();
237             }
238             if (null != ctx.FALSE()) {
239                 operatorToken = ctx.FALSE().getSymbol();
240             }
241             int startIndex = null == operatorToken ? ctx.IS().getSymbol().getStopIndex() + 2 : operatorToken.getStartIndex();
242             rightText = rightText.concat(ctx.start.getInputStream().getText(new Interval(startIndex, ctx.stop.getStopIndex())));
243             ExpressionSegment right = new LiteralExpressionSegment(ctx.IS().getSymbol().getStopIndex() + 2, ctx.stop.getStopIndex(), rightText);
244             String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
245             ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
246             String operator = "IS";
247             return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
248         }
249         if (null != ctx.comparisonOperator() || null != ctx.SAFE_EQ_()) {
250             return createCompareSegment(ctx);
251         }
252         return visit(ctx.predicate());
253     }
254     
255     private ASTNode createCompareSegment(final ClickHouseStatementParser.BooleanPrimaryContext ctx) {
256         ExpressionSegment left = (ExpressionSegment) visit(ctx.booleanPrimary());
257         ExpressionSegment right;
258         if (null != ctx.predicate()) {
259             right = (ExpressionSegment) visit(ctx.predicate());
260         } else {
261             right = (ExpressionSegment) visit(ctx.subquery());
262         }
263         String operator = null == ctx.SAFE_EQ_() ? ctx.comparisonOperator().getText() : ctx.SAFE_EQ_().getText();
264         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
265         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
266     }

268     @Override
269     public final ASTNode visitPredicate(final ClickHouseStatementParser.PredicateContext ctx) {
270         if (null != ctx.IN()) {
271             return createInSegment(ctx);
272         }
273         if (null != ctx.BETWEEN()) {
274             return createBetweenSegment(ctx);
275         }
276         if (null != ctx.LIKE()) {
277             return createBinaryOperationExpressionFromLike(ctx);
278         }
279         return visit(ctx.bitExpr(0));
280     }
281     
282     private BinaryOperationExpression createBinaryOperationExpressionFromLike(final ClickHouseStatementParser.PredicateContext ctx) {
283         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
284         ListExpression right = new ListExpression(ctx.simpleExpr(0).start.getStartIndex(), ctx.simpleExpr().get(ctx.simpleExpr().size() - 1).stop.getStopIndex());
285         for (ClickHouseStatementParser.SimpleExprContext each : ctx.simpleExpr()) {
286             right.getItems().add((ExpressionSegment) visit(each));
287         }
288         String operator = null == ctx.NOT() ? "LIKE" : "NOT LIKE";
289         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
290         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
291     }
292     
293     private InExpression createInSegment(final ClickHouseStatementParser.PredicateContext ctx) {
294         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
295         ExpressionSegment right;
296         if (null != ctx.subquery()) {
297             right = new SubqueryExpressionSegment(new SubquerySegment(ctx.subquery().start.getStartIndex(), ctx.subquery().stop.getStopIndex(), (SQL92SelectStatement) visit(ctx.subquery()),
298                     getOriginalText(ctx.subquery())));
299         } else {
300             ListExpression listExpression = new ListExpression(ctx.LP_().getSymbol().getStartIndex(), ctx.RP_().getSymbol().getStopIndex());
301             for (ClickHouseStatementParser.ExprContext each : ctx.expr()) {
302                 listExpression.getItems().add((ExpressionSegment) visit(each));
303             }
304             right = listExpression;
305         }
306         boolean not = null != ctx.NOT();
307         return new InExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, not);
308     }
309     
310     private BetweenExpression createBetweenSegment(final ClickHouseStatementParser.PredicateContext ctx) {
311         ExpressionSegment left = (ExpressionSegment) visit(ctx.bitExpr(0));
312         ExpressionSegment between = (ExpressionSegment) visit(ctx.bitExpr(1));
313         ExpressionSegment and = (ExpressionSegment) visit(ctx.predicate());
314         boolean not = null != ctx.NOT();
315         return new BetweenExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, between, and, not);
316     }

318     @Override
319     public final ASTNode visitBitExpr(final ClickHouseStatementParser.BitExprContext ctx) {
320         if (null != ctx.simpleExpr()) {
321             return createExpressionSegment(visit(ctx.simpleExpr()), ctx);
322         }
323         ExpressionSegment left = (ExpressionSegment) visit(ctx.getChild(0));
324         ExpressionSegment right = (ExpressionSegment) visit(ctx.getChild(2));
325         String operator = ctx.getChild(1).getText();
326         String text = ctx.start.getInputStream().getText(new Interval(ctx.start.getStartIndex(), ctx.stop.getStopIndex()));
327         return new BinaryOperationExpression(ctx.start.getStartIndex(), ctx.stop.getStopIndex(), left, right, operator, text);
328     }

330     private ASTNode createExpressionSegment(final ASTNode astNode, final ParserRuleContext context) {
331         if (astNode instanceof StringLiteralValue) {
332             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((StringLiteralValue) astNode).getValue());
333         }
334         if (astNode instanceof NumberLiteralValue) {
335             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((NumberLiteralValue) astNode).getValue());
336         }
337         if (astNode instanceof BooleanLiteralValue) {
338             return new LiteralExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(), ((BooleanLiteralValue) astNode).getValue());
339         }
340         if (astNode instanceof ParameterMarkerValue) {
341             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) astNode;
342             ParameterMarkerExpressionSegment segment = new ParameterMarkerExpressionSegment(context.start.getStartIndex(), context.stop.getStopIndex(),
343                     parameterMarker.getValue(), parameterMarker.getType());
344             parameterMarkerSegments.add(segment);
345             return segment;
346         }
347         if (astNode instanceof SubquerySegment) {
348             return new SubqueryExpressionSegment((SubquerySegment) astNode);
349         }
350         if (astNode instanceof OtherLiteralValue) {
351             return new CommonExpressionSegment(context.getStart().getStartIndex(), context.getStop().getStopIndex(), ((OtherLiteralValue) astNode).getValue());
352         }
353         return astNode;
354     }

356     @Override
357     public final ASTNode visitSimpleExpr(final ClickHouseStatementParser.SimpleExprContext ctx) {
358         int startIndex = ctx.getStart().getStartIndex();
359         int stopIndex = ctx.getStop().getStopIndex();
360         if (null != ctx.subquery()) {
361             return new SubquerySegment(startIndex, stopIndex, (SQL92SelectStatement) visit(ctx.subquery()), getOriginalText(ctx.subquery()));
362         }
363         if (null != ctx.parameterMarker()) {
364             ParameterMarkerValue parameterMarker = (ParameterMarkerValue) visit(ctx.parameterMarker());
365             ParameterMarkerExpressionSegment segment = new ParameterMarkerExpressionSegment(startIndex, stopIndex, parameterMarker.getValue(), parameterMarker.getType());
366             parameterMarkerSegments.add(segment);
367             return segment;
368         }
369         if (null != ctx.literals()) {
370             return SQLUtils.createLiteralExpression(visit(ctx.literals()), startIndex, stopIndex, ctx.literals().start.getInputStream().getText(new Interval(startIndex, stopIndex)));
371         }
372         if (null != ctx.functionCall()) {
373             return visit(ctx.functionCall());
374         }
375         if (null != ctx.columnName()) {
376             return visit(ctx.columnName());
377         }
378         return new CommonExpressionSegment(startIndex, stopIndex, ctx.getText());
379     }

401     @Override
402     public final ASTNode visitAggregationFunction(final ClickHouseStatementParser.AggregationFunctionContext ctx) {
403         String aggregationType = ctx.aggregationFunctionName().getText();
404         return AggregationType.isAggregationType(aggregationType)
405                 ? createAggregationSegment(ctx, aggregationType)
406                 : new ExpressionProjectionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), getOriginalText(ctx));
407     }

409     private ASTNode createAggregationSegment(final ClickHouseStatementParser.AggregationFunctionContext ctx, final String aggregationType) {
410         AggregationType type = AggregationType.valueOf(aggregationType.toUpperCase());
411         if (null != ctx.distinct()) {
412             AggregationDistinctProjectionSegment result =
413                     new AggregationDistinctProjectionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), type, getOriginalText(ctx), getDistinctExpression(ctx));
414             result.getParameters().addAll(getExpressions(ctx));
415             return result;
416         }
417         AggregationProjectionSegment result = new AggregationProjectionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), type, getOriginalText(ctx));
418         result.getParameters().addAll(getExpressions(ctx));
419         return result;
420     }
421     
422     private Collection<ExpressionSegment> getExpressions(final ClickHouseStatementParser.AggregationFunctionContext ctx) {
423         if (null == ctx.expr()) {
424             return Collections.emptyList();
425         }
426         Collection<ExpressionSegment> result = new LinkedList<>();
427         for (ClickHouseStatementParser.ExprContext each : ctx.expr()) {
428             result.add((ExpressionSegment) visit(each));
429         }
430         return result;
431     }
432     
433     private String getDistinctExpression(final ClickHouseStatementParser.AggregationFunctionContext ctx) {
434         StringBuilder result = new StringBuilder();
435         for (int i = 3; i < ctx.getChildCount() - 1; i++) {
436             result.append(ctx.getChild(i).getText());
437         }
438         return result.toString();
439     }

444             return visit(ctx.castFunction());

449     @Override
450     public final ASTNode visitCastFunction(final ClickHouseStatementParser.CastFunctionContext ctx) {
451         calculateParameterCount(Collections.singleton(ctx.expr()));
452         FunctionSegment result = new FunctionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), ctx.CAST().getText(), getOriginalText(ctx));
453         ASTNode exprSegment = visit(ctx.expr());
454         if (exprSegment instanceof ColumnSegment) {
455             result.getParameters().add((ColumnSegment) exprSegment);
456         } else if (exprSegment instanceof LiteralExpressionSegment) {
457             result.getParameters().add((LiteralExpressionSegment) exprSegment);
458         }
459         result.getParameters().add((DataTypeSegment) visit(ctx.dataType()));
460         return result;
461     }

463     @Override
464     public final ASTNode visitRegularFunction(final ClickHouseStatementParser.RegularFunctionContext ctx) {
465         FunctionSegment result = new FunctionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), ctx.regularFunctionName().getText(), getOriginalText(ctx));
466         Collection<ExpressionSegment> expressionSegments = ctx.expr().stream().map(each -> (ExpressionSegment) visit(each)).collect(Collectors.toList());
467         result.getParameters().addAll(expressionSegments);
468         return result;
469     }

496     @Override
497     public final ASTNode visitOrderByItem(final ClickHouseStatementParser.OrderByItemContext ctx) {
498         OrderDirection orderDirection = null == ctx.DESC() ? OrderDirection.ASC : OrderDirection.DESC;
499         if (null != ctx.columnName()) {
500             ColumnSegment column = (ColumnSegment) visit(ctx.columnName());
501             return new ColumnOrderByItemSegment(column, orderDirection, null);
502         }
503         return new IndexOrderByItemSegment(ctx.numberLiterals().getStart().getStartIndex(), ctx.numberLiterals().getStop().getStopIndex(),
504                 SQLUtils.getExactlyNumber(ctx.numberLiterals().getText(), 10).intValue(), orderDirection, null);
505     }

507     @Override
