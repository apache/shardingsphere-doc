18  package org.apache.shardingsphere.encrypt.sm.algorithm;
19  
20  import lombok.Getter;
21  import lombok.SneakyThrows;
22  import org.apache.shardingsphere.encrypt.api.encrypt.standard.StandardEncryptAlgorithm;
23  import org.apache.shardingsphere.encrypt.exception.algorithm.EncryptAlgorithmInitializationException;
24  import org.apache.shardingsphere.encrypt.api.context.EncryptContext;
25  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
26  import org.bouncycastle.jce.provider.BouncyCastleProvider;
27  import org.bouncycastle.pqc.math.linearalgebra.ByteUtils;
28  
29  import javax.crypto.Cipher;
30  import javax.crypto.spec.IvParameterSpec;
31  import javax.crypto.spec.SecretKeySpec;
32  import java.nio.charset.StandardCharsets;
33  import java.security.GeneralSecurityException;
34  import java.security.Security;
35  import java.util.Arrays;
36  import java.util.HashSet;
37  import java.util.Properties;
38  import java.util.Set;
39  import java.util.Map;
40  import java.util.HashMap;
41  
42  /**
43   * SM4 encrypt algorithm.
44   */
45  public final class SM4EncryptAlgorithm implements StandardEncryptAlgorithm<Object, String> {
46      
47      static {
48          Security.addProvider(new BouncyCastleProvider());
49      }
50      
51      private static final String SM4_KEY = "sm4-key";
52      
53      private static final String SM4_IV = "sm4-iv";
54      
55      private static final String SM4_MODE = "sm4-mode";
56      
57      private static final String SM4_PADDING = "sm4-padding";
58      
59      private static final int KEY_LENGTH = 16;
60      
61      private static final int IV_LENGTH = 16;
62      
63      private static final Set<String> MODES = new HashSet<>(Arrays.asList("ECB", "CBC"));
64      
65      private static final Set<String> PADDINGS = new HashSet<>(Arrays.asList("PKCS5Padding", "PKCS7Padding"));
66      
67      @Getter
68      private Map<String, Object> props;
69      
70      private byte[] sm4Key;
71      
72      private byte[] sm4Iv;
73      
74      private String sm4ModePadding;
75      
76      @Override
77      public void init(final Properties props) {
78          String sm4Mode = createSm4Mode(props);
79          String sm4Padding = createSm4Padding(props);
80          sm4ModePadding = "SM4/" + sm4Mode + "/" + sm4Padding;
81          sm4Key = createSm4Key(props);
82          sm4Iv = createSm4Iv(props, sm4Mode);
83          this.props = createProps();
84      }
85      
86      private String createSm4Mode(final Properties props) {
87          ShardingSpherePreconditions.checkState(props.containsKey(SM4_MODE), () -> new EncryptAlgorithmInitializationException("SM4", String.format("%s can not be null", SM4_MODE)));
88          String result = String.valueOf(props.getProperty(SM4_MODE)).toUpperCase();
89          ShardingSpherePreconditions.checkState(MODES.contains(result), () -> new EncryptAlgorithmInitializationException("SM4", "Mode must be either CBC or ECB"));
90          return result;
91      }
92      
93      private byte[] createSm4Key(final Properties props) {
94          ShardingSpherePreconditions.checkState(props.containsKey(SM4_KEY), () -> new EncryptAlgorithmInitializationException("SM4", String.format("%s can not be null", SM4_KEY)));
95          byte[] result = ByteUtils.fromHexString(String.valueOf(props.getProperty(SM4_KEY)));
96          ShardingSpherePreconditions.checkState(KEY_LENGTH == result.length,
97                  () -> new EncryptAlgorithmInitializationException("SM4", "Key length must be " + KEY_LENGTH + " bytes long"));
98          return result;
99      }
100     
101     private byte[] createSm4Iv(final Properties props, final String sm4Mode) {
102         if (!"CBC".equalsIgnoreCase(sm4Mode)) {
103             return new byte[0];
104         }
105         ShardingSpherePreconditions.checkState(props.containsKey(SM4_IV), () -> new EncryptAlgorithmInitializationException("SM4", String.format("%s can not be null", SM4_IV)));
106         String sm4IvValue = String.valueOf(props.getProperty(SM4_IV));
107         byte[] result = ByteUtils.fromHexString(sm4IvValue);
108         ShardingSpherePreconditions.checkState(IV_LENGTH == result.length, () -> new EncryptAlgorithmInitializationException("SM4", "Iv length must be " + IV_LENGTH + " bytes long"));
109         return result;
110     }
111     
112     private String createSm4Padding(final Properties props) {
113         ShardingSpherePreconditions.checkState(props.containsKey(SM4_PADDING), () -> new EncryptAlgorithmInitializationException("SM4", String.format("%s can not be null", SM4_PADDING)));
114         String result = String.valueOf(props.getProperty(SM4_PADDING)).toUpperCase().replace("PADDING", "Padding");
115         ShardingSpherePreconditions.checkState(PADDINGS.contains(result), () -> new EncryptAlgorithmInitializationException("SM4", "Padding must be either PKCS5Padding or PKCS7Padding"));
116         return result;
117     }
118     
119     private Map<String, Object> createProps() {
120         Map<String, Object> result = new HashMap<>();
121         result.put("sm4Key", ByteUtils.toHexString(sm4Key));
122         result.put("sm4Iv", ByteUtils.toHexString(sm4Iv));
123         result.put("sm4ModePadding", sm4ModePadding);
124         return result;
125     }
126     
127     @Override
128     public String encrypt(final Object plainValue, final EncryptContext encryptContext) {
129         return null == plainValue ? null : ByteUtils.toHexString(encrypt(String.valueOf(plainValue).getBytes(StandardCharsets.UTF_8)));
130     }
131     
132     private byte[] encrypt(final byte[] plainValue) {
133         return handle(plainValue, Cipher.ENCRYPT_MODE);
134     }
135     
136     @Override
137     public Object decrypt(final String cipherValue, final EncryptContext encryptContext) {
138         return null == cipherValue ? null : new String(decrypt(ByteUtils.fromHexString(cipherValue)), StandardCharsets.UTF_8);
139     }
140     
141     private byte[] decrypt(final byte[] cipherValue) {
142         return handle(cipherValue, Cipher.DECRYPT_MODE);
143     }
144     
145     @SneakyThrows(GeneralSecurityException.class)
146     private byte[] handle(final byte[] input, final int mode) {
147         Cipher cipher = Cipher.getInstance(sm4ModePadding, BouncyCastleProvider.PROVIDER_NAME);
148         SecretKeySpec secretKeySpec = new SecretKeySpec(sm4Key, "SM4");
149         if (0 == sm4Iv.length) {
150             cipher.init(mode, secretKeySpec);
151         } else {
152             cipher.init(mode, secretKeySpec, new IvParameterSpec(sm4Iv));
153         }
154         return cipher.doFinal(input);
155     }
156     
157     @Override
158     public String getType() {
159         return "SM4";
160     }
161 }
