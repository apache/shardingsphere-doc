42  /**
43   * System schema builder.
44   */
45  @NoArgsConstructor(access = AccessLevel.PRIVATE)
46  public final class SystemSchemaBuilder {
47      
48      /**
49       * Build system schema.
50       *
51       * @param databaseName database name
52       * @param databaseType database type
53       * @param props configuration properties
54       * @return ShardingSphere system schema map
55       */
56      public static Map<String, ShardingSphereSchema> build(final String databaseName, final DatabaseType databaseType, final ConfigurationProperties props) {
57          SystemDatabase systemDatabase = new SystemDatabase(databaseType);
58          Map<String, ShardingSphereSchema> result = new LinkedHashMap<>(systemDatabase.getSystemSchemas().size(), 1F);
59          boolean isSystemSchemaMetaDataEnabled = isSystemSchemaMetaDataEnabled(props.getProps());
60          YamlTableSwapper swapper = new YamlTableSwapper();
61          for (String each : getSystemSchemas(databaseName, databaseType, systemDatabase)) {
62              result.put(each.toLowerCase(), createSchema(each, SystemSchemaManager.getAllInputStreams(databaseType.getType(), each), swapper, isSystemSchemaMetaDataEnabled));
63          }
64          return result;
65      }
66      
67      private static boolean isSystemSchemaMetaDataEnabled(final Properties props) {
68          TemporaryConfigurationPropertyKey configKey = TemporaryConfigurationPropertyKey.SYSTEM_SCHEMA_METADATA_ASSEMBLY_ENABLED;
69          return Boolean.parseBoolean(props.getOrDefault(configKey.getKey(), configKey.getDefaultValue()).toString());
70      }
71      
72      private static Collection<String> getSystemSchemas(final String originalDatabaseName, final DatabaseType databaseType, final SystemDatabase systemDatabase) {
73          DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(databaseType).getDialectDatabaseMetaData();
74          String databaseName = dialectDatabaseMetaData.getDefaultSchema().isPresent() ? "postgres" : originalDatabaseName;
75          return systemDatabase.getSystemDatabaseSchemaMap().getOrDefault(databaseName, Collections.emptyList());
76      }
77      
78      private static ShardingSphereSchema createSchema(final String schemaName, final Collection<InputStream> schemaStreams, final YamlTableSwapper swapper,
79                                                       final boolean isSystemSchemaMetadataEnabled) {
80          Map<String, ShardingSphereTable> tables = new LinkedHashMap<>(schemaStreams.size(), 1F);
81          for (InputStream each : schemaStreams) {
82              YamlShardingSphereTable metaData = new Yaml().loadAs(each, YamlShardingSphereTable.class);
83              if (isSystemSchemaMetadataEnabled || KernelSupportedSystemTables.isSupportedSystemTable(schemaName, metaData.getName())) {
84                  tables.put(metaData.getName(), swapper.swapToObject(metaData));
85              }
86          }
87          return new ShardingSphereSchema(schemaName, tables, Collections.emptyMap());
88      }
89  }
