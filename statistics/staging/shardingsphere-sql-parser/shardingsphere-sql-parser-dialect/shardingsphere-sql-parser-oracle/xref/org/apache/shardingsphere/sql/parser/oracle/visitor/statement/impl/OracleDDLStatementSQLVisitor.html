268     @SuppressWarnings("unchecked")
269     @Override
270     public ASTNode visitCreateTable(final CreateTableContext ctx) {
271         OracleCreateTableStatement result = new OracleCreateTableStatement();
272         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
273         if (null != ctx.createDefinitionClause()) {
274             CollectionValue<CreateDefinitionSegment> createDefinitions = (CollectionValue<CreateDefinitionSegment>) visit(ctx.createDefinitionClause());
275             for (CreateDefinitionSegment each : createDefinitions.getValue()) {
276                 if (each instanceof ColumnDefinitionSegment) {
277                     result.getColumnDefinitions().add((ColumnDefinitionSegment) each);
278                 } else if (each instanceof ConstraintDefinitionSegment) {
279                     result.getConstraintDefinitions().add((ConstraintDefinitionSegment) each);
280                 }
281             }
282         }
283         return result;
284     }
285     
286     @Override
287     public ASTNode visitCreateDefinitionClause(final CreateDefinitionClauseContext ctx) {
288         CollectionValue<CreateDefinitionSegment> result = new CollectionValue<>();
289         if (null == ctx.createRelationalTableClause()) {
290             return result;
291         }
292         for (RelationalPropertyContext each : ctx.createRelationalTableClause().relationalProperties().relationalProperty()) {
293             if (null != each.columnDefinition()) {
294                 result.getValue().add((ColumnDefinitionSegment) visit(each.columnDefinition()));
295             }
296             if (null != each.outOfLineConstraint()) {
297                 result.getValue().add((ConstraintDefinitionSegment) visit(each.outOfLineConstraint()));
298             }
299             if (null != each.outOfLineRefConstraint()) {
300                 result.getValue().add((ConstraintDefinitionSegment) visit(each.outOfLineRefConstraint()));
301             }
302         }
303         return result;
304     }
305     
306     @Override
307     public ASTNode visitColumnDefinition(final ColumnDefinitionContext ctx) {
308         ColumnSegment column = (ColumnSegment) visit(ctx.columnName());
309         DataTypeSegment dataType = (DataTypeSegment) visit(ctx.dataType());
310         boolean isPrimaryKey = ctx.inlineConstraint().stream().anyMatch(each -> null != each.primaryKey());
311         boolean isNotNull = ctx.inlineConstraint().stream().anyMatch(each -> null != each.NOT() && null != each.NULL());
312         ColumnDefinitionSegment result = new ColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), column, dataType, isPrimaryKey, isNotNull);
313         for (InlineConstraintContext each : ctx.inlineConstraint()) {
314             if (null != each.referencesClause()) {
315                 result.getReferencedTables().add((SimpleTableSegment) visit(each.referencesClause().tableName()));
316             }
317         }
318         if (null != ctx.inlineRefConstraint()) {
319             result.getReferencedTables().add((SimpleTableSegment) visit(ctx.inlineRefConstraint().tableName()));
320         }
321         return result;
322     }
323     
324     @SuppressWarnings("unchecked")
325     @Override
326     public ASTNode visitOutOfLineConstraint(final OutOfLineConstraintContext ctx) {
327         ConstraintDefinitionSegment result = new ConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
328         if (null != ctx.constraintName()) {
329             result.setConstraintName((ConstraintSegment) visit(ctx.constraintName()));
330         }
331         if (null != ctx.primaryKey()) {
332             result.getPrimaryKeyColumns().addAll(((CollectionValue<ColumnSegment>) visit(ctx.columnNames())).getValue());
333         }
334         if (null != ctx.UNIQUE()) {
335             result.getIndexColumns().addAll(((CollectionValue<ColumnSegment>) visit(ctx.columnNames())).getValue());
336         }
337         if (null != ctx.referencesClause()) {
338             result.setReferencedTable((SimpleTableSegment) visit(ctx.referencesClause().tableName()));
339         }
340         return result;
341     }
342     
343     @Override
344     public ASTNode visitOutOfLineRefConstraint(final OutOfLineRefConstraintContext ctx) {
345         ConstraintDefinitionSegment result = new ConstraintDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex());
346         if (null != ctx.constraintName()) {
347             result.setConstraintName((ConstraintSegment) visit(ctx.constraintName()));
348         }
349         if (null != ctx.referencesClause()) {
350             result.setReferencedTable((SimpleTableSegment) visit(ctx.referencesClause().tableName()));
351         }
352         return result;
353     }

355     @SuppressWarnings("unchecked")
356     @Override
357     public ASTNode visitAlterTable(final AlterTableContext ctx) {
358         OracleAlterTableStatement result = new OracleAlterTableStatement();
359         result.setTable((SimpleTableSegment) visit(ctx.tableName()));
360         if (null != ctx.alterDefinitionClause()) {
361             for (AlterDefinitionSegment each : ((CollectionValue<AlterDefinitionSegment>) visit(ctx.alterDefinitionClause())).getValue()) {
362                 if (each instanceof AddColumnDefinitionSegment) {
363                     result.getAddColumnDefinitions().add((AddColumnDefinitionSegment) each);
364                 } else if (each instanceof ModifyColumnDefinitionSegment) {
365                     result.getModifyColumnDefinitions().add((ModifyColumnDefinitionSegment) each);
366                 } else if (each instanceof DropColumnDefinitionSegment) {
367                     result.getDropColumnDefinitions().add((DropColumnDefinitionSegment) each);
368                 } else if (each instanceof AddConstraintDefinitionSegment) {
369                     result.getAddConstraintDefinitions().add((AddConstraintDefinitionSegment) each);
370                 } else if (each instanceof ModifyConstraintDefinitionSegment) {
371                     result.getModifyConstraintDefinitions().add((ModifyConstraintDefinitionSegment) each);
372                 } else if (each instanceof DropConstraintDefinitionSegment) {
373                     result.getDropConstraintDefinitions().add((DropConstraintDefinitionSegment) each);
374                 }
375             }
376         }
377         return result;
378     }

380     @SuppressWarnings("unchecked")
381     @Override
382     public ASTNode visitAlterDefinitionClause(final AlterDefinitionClauseContext ctx) {
383         CollectionValue<AlterDefinitionSegment> result = new CollectionValue<>();
384         if (null != ctx.columnClauses()) {
385             for (OperateColumnClauseContext each : ctx.columnClauses().operateColumnClause()) {
386                 if (null != each.addColumnSpecification()) {
387                     result.getValue().addAll(((CollectionValue<AddColumnDefinitionSegment>) visit(each.addColumnSpecification())).getValue());
388                 }
389                 if (null != each.modifyColumnSpecification()) {
390                     result.getValue().add((ModifyColumnDefinitionSegment) visit(each.modifyColumnSpecification()));
391                 }
392                 if (null != each.dropColumnClause()) {
393                     result.getValue().add((DropColumnDefinitionSegment) visit(each.dropColumnClause()));
394                 }
395             }
396         }
397         if (null != ctx.constraintClauses()) {
398             // TODO Support rename constraint
399             ConstraintClausesContext constraintClausesContext = ctx.constraintClauses();
400             if (null != constraintClausesContext.addConstraintSpecification()) {
401                 result.combine((CollectionValue<AlterDefinitionSegment>) visit(constraintClausesContext.addConstraintSpecification()));
402             }
403             if (null != constraintClausesContext.modifyConstraintClause()) {
404                 result.getValue().add((AlterDefinitionSegment) visit(constraintClausesContext.modifyConstraintClause()));
405             }
406             for (DropConstraintClauseContext each : constraintClausesContext.dropConstraintClause()) {
407                 if (null != each.constraintName()) {
408                     result.getValue().add((AlterDefinitionSegment) visit(each));
409                 }
410             }
411         }
412         return result;
413     }

415     @Override
416     public ASTNode visitAddColumnSpecification(final AddColumnSpecificationContext ctx) {
417         CollectionValue<AddColumnDefinitionSegment> result = new CollectionValue<>();
418         for (ColumnOrVirtualDefinitionContext each : ctx.columnOrVirtualDefinitions().columnOrVirtualDefinition()) {
419             if (null != each.columnDefinition()) {
420                 AddColumnDefinitionSegment addColumnDefinition = new AddColumnDefinitionSegment(
421                         each.columnDefinition().getStart().getStartIndex(), each.columnDefinition().getStop().getStopIndex(),
422                         Collections.singletonList((ColumnDefinitionSegment) visit(each.columnDefinition())));
423                 result.getValue().add(addColumnDefinition);
424             }
425         }
426         return result;
427     }

429     @Override
430     public ASTNode visitModifyColumnSpecification(final ModifyColumnSpecificationContext ctx) {
431         // TODO handle no columnDefinition and multiple columnDefinitions
432         ColumnDefinitionSegment columnDefinition = null;
433         for (ModifyColPropertiesContext each : ctx.modifyColProperties()) {
434             columnDefinition = (ColumnDefinitionSegment) visit(each);
435         }
436         return new ModifyColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columnDefinition);
437     }

441         ColumnSegment column = (ColumnSegment) visit(ctx.columnName());
442         DataTypeSegment dataType = (DataTypeSegment) visit(ctx.dataType());
443         // TODO visit pk and reference table
444         return new ColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), column, dataType, false, false);
445     }

447     @Override
448     public ASTNode visitDropColumnSpecification(final DropColumnSpecificationContext ctx) {
449         Collection<ColumnSegment> columns = new LinkedList<>();
450         if (null != ctx.columnOrColumnList().columnName()) {
451             columns.add((ColumnSegment) visit(ctx.columnOrColumnList().columnName()));
452         } else {
453             for (ColumnNameContext each : ctx.columnOrColumnList().columnNames().columnName()) {
454                 columns.add((ColumnSegment) visit(each));
455             }
456         }
457         return new DropColumnDefinitionSegment(ctx.getStart().getStartIndex(), ctx.getStop().getStopIndex(), columns);
458     }

460     @Override
461     public ASTNode visitAddConstraintSpecification(final AddConstraintSpecificationContext ctx) {
462         CollectionValue<AddConstraintDefinitionSegment> result = new CollectionValue<>();
463         for (OutOfLineConstraintContext each : ctx.outOfLineConstraint()) {
464             result.getValue().add(new AddConstraintDefinitionSegment(each.getStart().getStartIndex(), each.getStop().getStopIndex(), (ConstraintDefinitionSegment) visit(each)));
465         }
466         if (null != ctx.outOfLineRefConstraint()) {
467             result.getValue().add(new AddConstraintDefinitionSegment(ctx.outOfLineRefConstraint().getStart().getStartIndex(), ctx.outOfLineRefConstraint().getStop().getStopIndex(),
468                     (ConstraintDefinitionSegment) visit(ctx.outOfLineRefConstraint())));
469         }
470         return result;
471     }

641     @Override
642     public ASTNode visitAssociateStatistics(final AssociateStatisticsContext ctx) {
643         OracleAssociateStatisticsStatement result = new OracleAssociateStatisticsStatement();
644         if (null != ctx.columnAssociation()) {
645             for (TableNameContext each : ctx.columnAssociation().tableName()) {
646                 result.getTables().add((SimpleTableSegment) visit(each));
647             }
648             for (ColumnNameContext each : ctx.columnAssociation().columnName()) {
649                 result.getColumns().add((ColumnSegment) visit(each));
650             }
651         }
652         if (null != ctx.functionAssociation()) {
653             for (IndexNameContext each : ctx.functionAssociation().indexName()) {
654                 result.getIndexes().add((IndexSegment) visit(each));
655             }
656             for (FunctionContext each : ctx.functionAssociation().function()) {
657                 result.getFunctions().add((FunctionSegment) visit(each));
658             }
659             for (PackageNameContext each : ctx.functionAssociation().packageName()) {
660                 result.getPackages().add((PackageSegment) visit(each));
661             }
662             for (TypeNameContext each : ctx.functionAssociation().typeName()) {
663                 result.getTypes().add((TypeSegment) visit(each));
664             }
665             for (IndexTypeNameContext each : ctx.functionAssociation().indexTypeName()) {
666                 result.getIndexTypes().add((IndexTypeSegment) visit(each));
667             }
668         }
669         return result;
670     }

687         }

691             }
692         }
693         if (null != ctx.packageName()) {
694             for (PackageNameContext each : ctx.packageName()) {
695                 result.getPackages().add((PackageSegment) visit(each));
696             }
697         }
698         if (null != ctx.typeName()) {
