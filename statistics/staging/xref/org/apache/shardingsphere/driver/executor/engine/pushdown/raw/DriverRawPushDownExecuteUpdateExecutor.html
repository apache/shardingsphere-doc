39  /**
40   * Driver raw push down execute update executor.
41   */
42  public final class DriverRawPushDownExecuteUpdateExecutor {
43      
44      private final String processId;
45      
46      private final ConfigurationProperties props;
47      
48      private final Grantee grantee;
49      
50      private final RawExecutor rawExecutor;
51      
52      public DriverRawPushDownExecuteUpdateExecutor(final ShardingSphereConnection connection, final ShardingSphereMetaData metaData, final Grantee grantee, final RawExecutor rawExecutor) {
53          processId = connection.getProcessId();
54          props = metaData.getProps();
55          this.grantee = grantee;
56          this.rawExecutor = rawExecutor;
57      }
58      
59      /**
60       * Execute update.
61       * 
62       * @param database database
63       * @param executionContext execution context
64       * @return updated row count
65       * @throws SQLException SQL exception
66       */
67      public int executeUpdate(final ShardingSphereDatabase database, final ExecutionContext executionContext) throws SQLException {
68          return accumulate(rawExecutor.execute(createRawExecutionGroupContext(database, executionContext), executionContext.getQueryContext(), new RawSQLExecutorCallback()));
69      }
70      
71      private int accumulate(final Collection<ExecuteResult> results) {
72          int result = 0;
73          for (ExecuteResult each : results) {
74              result += ((UpdateResult) each).getUpdateCount();
75          }
76          return result;
77      }
78      
79      private ExecutionGroupContext<RawSQLExecutionUnit> createRawExecutionGroupContext(final ShardingSphereDatabase database, final ExecutionContext executionContext) throws SQLException {
80          int maxConnectionsSizePerQuery = props.<Integer>getValue(ConfigurationPropertyKey.MAX_CONNECTIONS_SIZE_PER_QUERY);
81          return new RawExecutionPrepareEngine(maxConnectionsSizePerQuery, database.getRuleMetaData().getRules()).prepare(database.getName(),
82                  executionContext.getRouteContext(), executionContext.getExecutionUnits(), new ExecutionGroupReportContext(processId, database.getName(), grantee));
83      }
84  }
