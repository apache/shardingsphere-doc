50  /**
51   * Cluster meta data manager persist service.
52   */
53  public final class ClusterMetaDataManagerPersistService implements MetaDataManagerPersistService {
54      
55      private final MetaDataPersistService metaDataPersistService;
56      
57      private final ListenerAssistedPersistService listenerAssistedPersistService;
58      
59      private final MetaDataContextManager metaDataContextManager;
60      
61      public ClusterMetaDataManagerPersistService(final PersistRepository repository, final MetaDataContextManager metaDataContextManager) {
62          metaDataPersistService = new MetaDataPersistService(repository);
63          listenerAssistedPersistService = new ListenerAssistedPersistService(repository);
64          this.metaDataContextManager = metaDataContextManager;
65      }
66      
67      @Override
68      public void createDatabase(final String databaseName) {
69          metaDataPersistService.getDatabaseMetaDataService().addDatabase(databaseName);
70          listenerAssistedPersistService.persistDatabaseNameListenerAssisted(new ListenerAssisted(databaseName, ListenerAssistedType.CREATE_DATABASE));
71      }
72      
73      @Override
74      public void dropDatabase(final String databaseName) {
75          String droppedDatabaseName = metaDataContextManager.getMetaDataContexts().get().getMetaData().getDatabase(databaseName).getName();
76          listenerAssistedPersistService.persistDatabaseNameListenerAssisted(new ListenerAssisted(droppedDatabaseName, ListenerAssistedType.DROP_DATABASE));
77          metaDataPersistService.getDatabaseMetaDataService().dropDatabase(droppedDatabaseName);
78      }
79      
80      @Override
81      public void createSchema(final String databaseName, final String schemaName) {
82          metaDataPersistService.getDatabaseMetaDataService().addSchema(databaseName, schemaName);
83      }
84      
85      @Override
86      public void alterSchema(final AlterSchemaPOJO alterSchemaPOJO) {
87          String databaseName = alterSchemaPOJO.getDatabaseName();
88          String schemaName = alterSchemaPOJO.getSchemaName();
89          ShardingSphereSchema schema = metaDataContextManager.getMetaDataContexts().get().getMetaData().getDatabase(databaseName).getSchema(schemaName);
90          DatabaseMetaDataPersistService databaseMetaDataService = metaDataPersistService.getDatabaseMetaDataService();
91          if (schema.isEmpty()) {
92              databaseMetaDataService.addSchema(databaseName, alterSchemaPOJO.getRenameSchemaName());
93          }
94          databaseMetaDataService.getTableMetaDataPersistService().persist(databaseName, alterSchemaPOJO.getRenameSchemaName(), schema.getTables());
95          databaseMetaDataService.getViewMetaDataPersistService().persist(databaseName, alterSchemaPOJO.getRenameSchemaName(), schema.getViews());
96          databaseMetaDataService.dropSchema(databaseName, schemaName);
97      }
98      
99      @Override
100     public void dropSchema(final String databaseName, final Collection<String> schemaNames) {
101         DatabaseMetaDataPersistService databaseMetaDataService = metaDataPersistService.getDatabaseMetaDataService();
102         schemaNames.forEach(each -> databaseMetaDataService.dropSchema(databaseName, each));
103     }
104     
105     @Override
106     public void alterSchemaMetaData(final AlterSchemaMetaDataPOJO alterSchemaMetaDataPOJO) {
107         String databaseName = alterSchemaMetaDataPOJO.getDatabaseName();
108         String schemaName = alterSchemaMetaDataPOJO.getSchemaName();
109         Map<String, ShardingSphereTable> tables = alterSchemaMetaDataPOJO.getAlteredTables().stream().collect(Collectors.toMap(ShardingSphereTable::getName, table -> table));
110         Map<String, ShardingSphereView> views = alterSchemaMetaDataPOJO.getAlteredViews().stream().collect(Collectors.toMap(ShardingSphereView::getName, view -> view));
111         DatabaseMetaDataPersistService databaseMetaDataService = metaDataPersistService.getDatabaseMetaDataService();
112         databaseMetaDataService.getTableMetaDataPersistService().persist(databaseName, schemaName, tables);
113         databaseMetaDataService.getViewMetaDataPersistService().persist(databaseName, schemaName, views);
114         alterSchemaMetaDataPOJO.getDroppedTables().forEach(each -> databaseMetaDataService.getTableMetaDataPersistService().delete(databaseName, schemaName, each));
115         alterSchemaMetaDataPOJO.getDroppedViews().forEach(each -> databaseMetaDataService.getViewMetaDataPersistService().delete(databaseName, schemaName, each));
116     }
117     
118     @Override
119     public void registerStorageUnits(final String databaseName, final Map<String, DataSourcePoolProperties> toBeRegisteredProps) throws SQLException {
120         MetaDataContexts originalMetaDataContexts = metaDataContextManager.getMetaDataContexts().get();
121         SwitchingResource switchingResource = metaDataContextManager.getResourceSwitchManager()
122                 .switchByRegisterStorageUnit(originalMetaDataContexts.getMetaData().getDatabase(databaseName).getResourceMetaData(), toBeRegisteredProps);
123         MetaDataContexts reloadMetaDataContexts = MetaDataContextsFactory.createBySwitchResource(databaseName, false,
124                 switchingResource, originalMetaDataContexts, metaDataPersistService, metaDataContextManager.getComputeNodeInstanceContext());
125         metaDataPersistService.getDataSourceUnitService().persistConfigurations(databaseName, toBeRegisteredProps);
126         afterStorageUnitsAltered(databaseName, originalMetaDataContexts, reloadMetaDataContexts);
127         reloadMetaDataContexts.close();
128     }
129     
130     @Override
131     public void alterStorageUnits(final String databaseName, final Map<String, DataSourcePoolProperties> toBeUpdatedProps) throws SQLException {
132         MetaDataContexts originalMetaDataContexts = metaDataContextManager.getMetaDataContexts().get();
133         SwitchingResource switchingResource = metaDataContextManager.getResourceSwitchManager()
134                 .switchByAlterStorageUnit(originalMetaDataContexts.getMetaData().getDatabase(databaseName).getResourceMetaData(), toBeUpdatedProps);
135         MetaDataContexts reloadMetaDataContexts = MetaDataContextsFactory.createBySwitchResource(databaseName, false,
136                 switchingResource, originalMetaDataContexts, metaDataPersistService, metaDataContextManager.getComputeNodeInstanceContext());
137         DataSourceUnitPersistService dataSourceService = metaDataPersistService.getDataSourceUnitService();
138         metaDataPersistService.getMetaDataVersionPersistService()
139                 .switchActiveVersion(dataSourceService.persistConfigurations(databaseName, toBeUpdatedProps));
140         afterStorageUnitsAltered(databaseName, originalMetaDataContexts, reloadMetaDataContexts);
141         reloadMetaDataContexts.close();
142     }
143     
144     @Override
145     public void unregisterStorageUnits(final String databaseName, final Collection<String> toBeDroppedStorageUnitNames) throws SQLException {
146         for (String each : getToBeDroppedResourceNames(databaseName, toBeDroppedStorageUnitNames)) {
147             MetaDataContexts originalMetaDataContexts = metaDataContextManager.getMetaDataContexts().get();
148             SwitchingResource switchingResource = metaDataContextManager.getResourceSwitchManager()
149                     .switchByUnregisterStorageUnit(originalMetaDataContexts.getMetaData().getDatabase(databaseName).getResourceMetaData(), Collections.singletonList(each));
150             MetaDataContexts reloadMetaDataContexts = MetaDataContextsFactory.createBySwitchResource(databaseName, false,
151                     switchingResource, originalMetaDataContexts, metaDataPersistService, metaDataContextManager.getComputeNodeInstanceContext());
152             metaDataPersistService.getDataSourceUnitService().delete(databaseName, each);
153             afterStorageUnitsDropped(databaseName, originalMetaDataContexts, reloadMetaDataContexts);
154             reloadMetaDataContexts.close();
155         }
156     }
157     
158     private Collection<String> getToBeDroppedResourceNames(final String databaseName, final Collection<String> toBeDroppedResourceNames) {
159         Map<String, DataSourcePoolProperties> propsMap = metaDataPersistService.getDataSourceUnitService().load(databaseName);
160         return toBeDroppedResourceNames.stream().filter(propsMap::containsKey).collect(Collectors.toList());
161     }
162     
163     @Override
164     public void alterSingleRuleConfiguration(final String databaseName, final Collection<RuleConfiguration> ruleConfigs) {
165         ruleConfigs.removeIf(each -> !each.getClass().isAssignableFrom(SingleRuleConfiguration.class));
166         metaDataPersistService.getMetaDataVersionPersistService()
167                 .switchActiveVersion(metaDataPersistService.getDatabaseRulePersistService().persistConfigurations(databaseName, ruleConfigs));
168     }
169     
170     @Override
171     public void alterRuleConfiguration(final String databaseName, final RuleConfiguration toBeAlteredRuleConfig) {
172         MetaDataContexts originalMetaDataContexts = metaDataContextManager.getMetaDataContexts().get();
173         if (null != toBeAlteredRuleConfig) {
174             Collection<MetaDataVersion> metaDataVersions = metaDataPersistService.getDatabaseRulePersistService()
175                     .persistConfigurations(databaseName, Collections.singleton(toBeAlteredRuleConfig));
176             metaDataPersistService.getMetaDataVersionPersistService().switchActiveVersion(metaDataVersions);
177             afterRuleConfigurationAltered(databaseName, originalMetaDataContexts);
178         }
179     }
180     
181     @Override
182     public void removeRuleConfigurationItem(final String databaseName, final RuleConfiguration toBeRemovedRuleConfig) {
183         if (null != toBeRemovedRuleConfig) {
184             metaDataPersistService.getDatabaseRulePersistService().deleteConfigurations(databaseName, Collections.singleton(toBeRemovedRuleConfig));
185         }
186     }
187     
188     @Override
189     public void removeRuleConfiguration(final String databaseName, final String ruleName) {
190         MetaDataContexts originalMetaDataContexts = metaDataContextManager.getMetaDataContexts().get();
191         metaDataPersistService.getDatabaseRulePersistService().delete(databaseName, ruleName);
192         afterRuleConfigurationDropped(databaseName, originalMetaDataContexts);
193     }
194     
195     @Override
196     public void alterGlobalRuleConfiguration(final RuleConfiguration toBeAlteredRuleConfig) {
197         metaDataPersistService.getGlobalRuleService().persist(Collections.singleton(toBeAlteredRuleConfig));
198     }
199     
200     @Override
201     public void alterProperties(final Properties props) {
202         metaDataPersistService.getPropsService().persist(props);
203     }
204     
205     private void afterStorageUnitsAltered(final String databaseName, final MetaDataContexts originalMetaDataContexts, final MetaDataContexts reloadMetaDataContexts) {
206         reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getSchemas().forEach((schemaName, schema) -> metaDataPersistService.getDatabaseMetaDataService()
207                 .persistByAlterConfiguration(reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getName(), schemaName, schema));
208         // TODO confirm
209         Optional.ofNullable(reloadMetaDataContexts.getStatistics().getDatabaseData().get(databaseName))
210                 .ifPresent(optional -> optional.getSchemaData().forEach((schemaName, schemaData) -> metaDataPersistService.getShardingSphereDataPersistService()
211                         .persist(databaseName, schemaName, schemaData, originalMetaDataContexts.getMetaData().getDatabases())));
212         metaDataPersistService.persistReloadDatabaseByAlter(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName),
213                 originalMetaDataContexts.getMetaData().getDatabase(databaseName));
214     }
215     
216     private void afterStorageUnitsDropped(final String databaseName, final MetaDataContexts originalMetaDataContexts, final MetaDataContexts reloadMetaDataContexts) {
217         reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getSchemas().forEach((schemaName, schema) -> metaDataPersistService.getDatabaseMetaDataService()
218                 .persistByDropConfiguration(reloadMetaDataContexts.getMetaData().getDatabase(databaseName).getName(), schemaName, schema));
219         Optional.ofNullable(reloadMetaDataContexts.getStatistics().getDatabaseData().get(databaseName))
220                 .ifPresent(optional -> optional.getSchemaData().forEach((schemaName, schemaData) -> metaDataPersistService.getShardingSphereDataPersistService()
221                         .persist(databaseName, schemaName, schemaData, originalMetaDataContexts.getMetaData().getDatabases())));
222         metaDataPersistService.persistReloadDatabaseByDrop(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName),
223                 originalMetaDataContexts.getMetaData().getDatabase(databaseName));
224     }
225     
226     private void afterRuleConfigurationAltered(final String databaseName, final MetaDataContexts originalMetaDataContexts) {
227         MetaDataContexts reloadMetaDataContexts = metaDataContextManager.getMetaDataContexts().get();
228         metaDataPersistService.persistReloadDatabaseByAlter(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName),
229                 originalMetaDataContexts.getMetaData().getDatabase(databaseName));
230     }
231     
232     private void afterRuleConfigurationDropped(final String databaseName, final MetaDataContexts originalMetaDataContexts) {
233         MetaDataContexts reloadMetaDataContexts = metaDataContextManager.getMetaDataContexts().get();
234         metaDataPersistService.persistReloadDatabaseByDrop(databaseName, reloadMetaDataContexts.getMetaData().getDatabase(databaseName),
235                 originalMetaDataContexts.getMetaData().getDatabase(databaseName));
236     }
